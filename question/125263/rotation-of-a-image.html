This is my code.

var button1 = Titanium.UI.createButton({

button1.addEventListener('click', function()
{
    var t3 = Ti.UI.create2DMatrix();
    t3 = t3.rotate(90);
    var a2 = Titanium.UI.createAnimation();

    var button1 = Titanium.UI.createButton({
    title:'Rotate',

    var a2 = Titanium.UI.createAnimation();
