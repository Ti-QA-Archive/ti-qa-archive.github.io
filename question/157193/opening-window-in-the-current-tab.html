  • Hello Mohamed Badr,

    I have added example code for opening window in current tab. I hope that will help you. By the way, you posted your code with unnecessary code. That’s very disgusting for debug.

    
var self = Ti.UI.createTabGroup();

var win1 = Ti.UI.createWindow({
    title : 'Home',
    backgroundColor : 'white'
});

var button = Ti.UI.createButton({
    height : 44,
    width : 200,
    title : L('openWindow'),
    top : 20
});
win1.add(button);

button.addEventListener('click', function() {
    //containingTab attribute must be set by parent tab group on
    //the window for this work
    win1.containingTab.open(Ti.UI.createWindow({
        title : L('newWindow'),
        backgroundColor : 'white'
    }));
});

win2 = Ti.UI.createWindow({
    title : 'Setting',
    backgroundColor : 'white'
});

var tab1 = Ti.UI.createTab({
    title : L('home'),
    icon : '/images/KS_nav_ui.png',
    window : win1
});
win1.containingTab = tab1;

self.addTab(tab1);

self.open();

    Thanks
