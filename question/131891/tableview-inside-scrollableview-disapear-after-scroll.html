    for some reason its now working - its look like every few compiles its making this bug.

    the code is really basic but i cant really paste all my code here so here is a sample code which will do the same. (my code retrieve the data from my server and loop through it)

    
function createTable()
{
    var sections = new Array();
    for(var j = 0; j < 4; j++)
    {
        var section = Titanium.UI.createTableViewSection(
            {
                            headerTitle:  'foo bar'
            }
        );

        for(var i = 0; i < 10; i++)
        {

            var row = Titanium.UI.createTableViewRow(
            {
                height : '40',
                 backgroundSelectedColor : '#000'
             });
            section.add(row);
        }
        sections.push(section);
    }

    var tableView = Titanium.UI.createTableView(
            {
                data : sections,
                allowsSelection : false,
                height : '98%',
                width : '98%',
                borderColor : '#000',
                backgroundColor : '#000',
                borderRadius : '8',
                bottom : '5',
                separatorColor : '#313031'
            });
    return tableView;
}

var view1 = createTable();
var view2 = createTable();


var scrollView = Titanium.UI.createScrollableView(
            {
                views:[view1,view2],
                showPagingControl: true,
                width: '100%',
                height: '100%',
                top: '0',
                pagingControlTimeout: '1000'
            });

win1.add(scrollView);
