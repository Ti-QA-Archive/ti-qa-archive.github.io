Swipe event not working on TableViewRow

I am trying to add a swipe event on tableviewrow, but it doesn't work.

Code:

function getAddLitData(){
    var par = 'message=[{"userID":"' + Ti.App.Properties.getString('userLoginID') + '"}]';

    util.requestServer({
        url : Alloy.CFG.httpProtocol + Alloy.CFG.APIDomain +"getItem.php",
        success : function(ev) {
            var response = JSON.parse(this.responseText);
            _data = []
            for (var i = 0; i < response.length; i++) {

                var _return = Ti.UI.createTableViewRow({
                    height : 45,
                    hasChild : true,
                    backgroundColor : '#FFF',
                    name : response[i].name,
                    id : response[i].id,
                });

                var caption = Ti.UI.createLabel({
                    text :  response[i].name,
                    font : {
                        fontSize : 20,
                        fontWeight : 'bold'
                    },
                    left : 20,
                    width : 270,
                    touchEnabled : false,
                    color : '#5a5a5a',
                });
                _return.add(caption);

                _return.addEventListener('swipe', function(e) {
                    alert("Hello");
                })

                _data.push(_return)
            }
            $.addItemTableView.setData(_data);        
        },
        method : 'POST',
        param : par,
    });
}

However, when I add the swipe event on tableView, it worked.

Can someone help me out with this issue ?

Thanks in advance
