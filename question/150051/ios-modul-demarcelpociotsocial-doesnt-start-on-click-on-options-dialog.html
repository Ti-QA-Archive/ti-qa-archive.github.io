iOS-modul "de.marcelpociot.social" doesn't start on click on options dialog

Hi,
I have a right nav button, open a options dialog and want to start a social sheet. If I Ti.App.Social.showSheet() in constructor of window, then it works. In the example of marcel he uses a normal button. But options dialog is modal ….

exports.create = function(_link) {
    var self = require('modules/mywindow').create({
        title : 'Videoplaylist'
    }, function() {
        self.dialog.show();
    });

    self.dialog = Titanium.UI.createOptionDialog({
        options : ['Videobookmark', 'Channelabo', 'Facebook', 'Twitter', 'Abbruch'],
        cancel : 4
    });
    / ***   some stuff to fill window    */
    self.dialog.addEventListener('click', function(e) {
        switch (e) {
            case 2:
                Ti.App.Social.showSheet({
                    service : 'facebook',
                    message : 'Hey, this is some cool facebook message!',
                    urls : [_link],
                    images : [],
                    success : function() {
                        alert("Tweet successfully sent");
                    }
                });
                break;
        }
    });
    return self;
}

In app.js:

Ti.App.Social = require('de.marcelpociot.social');

But I dont see the dialog sheet. Any ideas?
