call evalJS on webview beforeload

var win = Titanium.UI.createWindow({
    title: 'Home'
});

var webview = Ti.UI.createWebView({
    url: 'p1.html'
});

function swipe_page(event) {
    if(event.direction == "left") {
        alert('left swipe'); // this works
        webview.addEventListener('beforeload', function(){
            console.log('title', webview.evalJS('document.title')); //print p1's document title
            webview.evalJS('window.a = 1'); // window is p1's window, I want p2's window
        })
        webview.url = 'p2.html';
    }
}
win.addEventListener('swipe',swipe_page);

win.add(webview);
win.open({modal:true});

I want to evaluate something into webview using WebView.evalJS when change webview url, like 

webview.evalJS('window.a = 1');

I called it in beforeload handler.But window object is old one, not the new one of p2.html.
