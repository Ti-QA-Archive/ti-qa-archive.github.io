      var scrollView = Ti.UI.createScrollView({
    width : "100%",
    height : "100%",
    layout : "vertical"
});
$.winIndex.add(scrollView);

for (var i = 0; i < 10; i++) {
    var btnView = Ti.UI.createView({
        top : 5,
        left : "10dp",
        right : "10dp",
        borderColor : "green",
        borderRadius : 5,
        height : Titanium.UI.SIZE,
        btnId : i
    });
    scrollView.add(btnView);

    btnView.addEventListener("click",function(e){
        alert(e.source.btnId + " Clicked");
    });

    var imgView = Titanium.UI.createImageView({
        backgroundColor : "red",
        height : "50dp",
        width : "50dp",
        top : "15dp",
        bottom : "15dp",
        left : "15dp",
        touchEnabled : false,
    });
    btnView.add(imgView);

    var lblTitle = Titanium.UI.createLabel({
        left : "80dp",
        right : "15dp",
        text : "Title Text " + i,
        font : {
            fontSize : "14sp"
        },
        touchEnabled : false,
    });
    btnView.add(lblTitle);
}
