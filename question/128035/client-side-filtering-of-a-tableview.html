Client-side filtering of a tableview

I've been struggling to implement client-side filtering of a tableview.

I've written the filtering logic and have created a tabbar to allow the user to choose the filter.

But I can't seem to actually hide/show a row. I can remove them but that removes the item from the data.

Table filtering seems to be possible as the table view search filteres the rows correctly.

Not sure why the hide/show methods of the row do nothing.

// create window
var win = Titanium.UI.createWindow({
  backgroundColor: '#fff'
});

var data = ['A', 'B', 'C', 'D', 'E'];

// create tabbar
var tabBar = Titanium.UI.createTabbedBar({
    labels:['A', 'B'],
    index:0,
    style:Titanium.UI.iPhone.SystemButtonStyle.BAR
});
win.setTitleControl(tabBar);    

// create table
var table = Titanium.UI.createTableView({
  data: data
});
win.add(table);

// tabbar event listener
tabBar.addEventListener('click', function(e) {
  var i, j;
  for(i = 0, j = data.length; i < j; i++) {

    // if filter "A" selected...
    if (e.index == 0) {
      if(data[i] == 'A') {
        table.data[0].rows[i].hide();
      }
    }
    // if filter "B" selected...
    else if (e.index == 1) {
      if(data[i] == 'B') {
        table.data[0].rows[i].hide();
      }
    }
    else {
      // all other rows we need to show (if previouslly hidden)
      table.data[0].rows[i].show();
    }    
  }  
});

Any help would be appreciated.
