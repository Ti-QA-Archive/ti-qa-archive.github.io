How to close current view and parent view ?

Here is the structure of my app :

Resources
|-- app.js
|-- views
|   |-- menu.js
|   |-- vitrine.js
|   |-- vitrine
|   |   |-- liste.js

app.js : 

Titanium.UI.setBackgroundColor('#000');

var menu = Titanium.UI.createWindow({  
    title:'Menu principal',
    backgroundColor:'#fff',
    url:'views/menu.js'
});

menu.open();

menu.js :

var win = Ti.UI.currentWindow;

var button3 = Ti.UI.createButton({
    title: 'Ventes',
    top: 100,
    left:20,
    width: 130,
    height: 80
});

button3.addEventListener('click', function(e){
    var win3 = Ti.UI.createWindow({
        backgroundColor:'#ffffff',
        url:'vitrine.js'
    });
    win3.open();
});

win.add(button3);

vitrine.js

var win = Ti.UI.currentWindow;

Titanium.UI.setBackgroundColor('#000');

var tabGroup = Titanium.UI.createTabGroup();

var win1 = Titanium.UI.createWindow({  
    title:'Magasin 1',
    backgroundColor:'#fff',
    url:'vitrine/liste.js'
});

var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',

});

tabGroup.addTab(tab1);

tabGroup.open();

win.add(tabGroup);

liste.js :

var win = Ti.UI.currentWindow;

var label = Ti.UI.createLabel({

var rootButton = Ti.UI.createButton({
    title:'back to root'

win.leftNavButton = rootButton;

rootButton.addEventListener('click', function(e){
    // PLACE HERE THE CODE I NEED

So, I need the code for my button rootButton. By clicking on it, I want to close the current view, then the parent view, and go back to the root menu.

Do you have an idea ?

    Hi,

    To close current view & parent view, you need to keep a reference to parent view - vitrine window.

    //vitrine.js
var win = Ti.UI.currentWindow;

var win1 = Titanium.UI.createWindow({  
    title:'Magasin 1',
    backgroundColor:'#fff',
    url:'vitrine/liste.js',
    ext:{vitrine:win}//Keep a reference to vitrine
});

//liste.js
rootButton.addEventListener('click', function(e){
    // Hide parent view
    win.ext.vitrine.hide();
    // Hide current view
    win.hide();
});
