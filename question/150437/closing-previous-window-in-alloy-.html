Closing previous window in Alloy ?

How to close previous window in alloy? From list page navigating to detail page using this. 

function listItemClick() {
     var detailWin=Alloy.createController('detail');
     detailWin.getView().open();
     $.index.close();
}

And from detail page navigating back to list page on android device back button click.

var backEvent=function() {   
    var listWin=Alloy.createController('index');
         listWin.getView().open();
         $.detail.close();
    removeDeviceBack();
}

$.salesAreaTab.addEventListener('android:back',backEvent);

function removeDeviceBack(){
    $.salesAreaTab.removeEventListener('android:back',backEvent);    
}

Here every time i visit a page , it is opening a new page and pushing into the stack if i don't close window.So how can i close the window? I tried above snippet but getting "unable to open, window not closed".
