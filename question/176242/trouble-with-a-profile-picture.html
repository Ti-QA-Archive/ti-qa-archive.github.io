  • Or you build a compact solution: 

    tableviewrow.add(require('profilephoto.widget')(options));

    and the profilephoto.widget.js:

    module.exports = function(_options) {
    var self = Ti.UI.createImageView();
    for (var key in _options) {
        self[key] = _options[key];
    }
     self.addEventListener('click',function() {
        Ti.Media.openGallery({success: function(_res) {
            self.setImage(_res.media);
            }});
        });
    return self;
}
