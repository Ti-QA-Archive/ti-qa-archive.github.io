  • hi,

    i have create a test demo so just try it.
    i have saved cropped image in cachedirectory so see open that folder and then run this code so that you can see cropped image.

    var win = Ti.UI.createWindow({
});

win.open();

var view = Ti.UI.createView({
    top : 0,
    left : 0,
    width : 320,
    height : 460,
    backgroundImage : 'test.jpg'
});

win.add(view);

var view1 = Ti.UI.createView({
    width : 200,
    height : 200,

    borderColor : 'gray'
});

view1.addEventListener('touchstart', function(e) {

    startX = e.x;
    startY = e.y;
    Ti.API.info(view1.animatedCenter.x);
});

view1.addEventListener('touchmove', function(e) {

    var newX = e.x + view1.animatedCenter.x - view1.width / 2;
    var newY = e.y + view1.animatedCenter.y - view1.height / 2;
    view1.animate({
        center : {
            x : newX,
            y : newY
        },
        duration : 1
    }, function(e) {

});

view1.addEventListener('touchend', function(e) {
    var blob = view.toImage();
    var croped = blob.imageAsCropped({
        x : view1.animatedCenter.x - (view1.width / 2),
        y : view1.animatedCenter.y - (view1.height / 2),
        height : 200,
        width : 200
    });
    var file = Ti.Filesystem.getFile(Ti.Filesystem.applicationCacheDirectory, 'test.png');
    file.write(croped);
});
win.add(view1);
