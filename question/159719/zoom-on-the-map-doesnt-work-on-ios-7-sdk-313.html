    • You should be setting region after the map has loaded so try this;

      var map = Ti.Map.createView();
map.addEventListener('complete', function (e) {
    Ti.Geolocation.getCurrentPosition(function (e) {
        if (e.error) {
            alert('error!');
            var region = {
                latitude : 0,
                longitude : 0,
                userLocation : true,
                latitudeDelta : 1.1,
                longitudeDelta : 1.1
            };
        } else {
            var region = {
                latitude : e.coords.latitude,
                longitude : e.coords.longitude,
                userLocation : true,
                latitudeDelta : 1.1,
                longitudeDelta : 1.1
            };
        }
        map.setLocation(region);
    });
});
mapview.add(map);
mapview.open();

      This will create the map add it to the mapview and only try setting the region once the map is ready to receive it.
