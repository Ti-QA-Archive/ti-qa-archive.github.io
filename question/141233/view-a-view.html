    Hi Juan

    Change your second view to 

    var vista2 = Ti.UI.createView({
    top : 10,
    backgroundColor : '#eee',
    backgroundGradient : {
        type : 'linear',
        startPoint : {
            x : '0%',
            y : '50%'
        },
        endPoint : {
            x : '100%',
            y : '50%'
        },
        colors : [{
            color : 'red',
            offset : 0.0
        }, {
            color : 'blue',
            offset : 0.25
        }, {
            color : 'red',
            offset : 1.0
        }],
    },
    visible: false // new line
});
win1.add(vista2);

    Change your button to;

    btn_parque_cordillera.addEventListener('click', function(e) {
    vista1.setVisible(false);
    vista2.setVisible(true);
});
