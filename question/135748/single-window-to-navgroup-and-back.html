Single window to NavGroup and back

I am making an app that has several single window steps before the user gets to a tab group(s). The app starts up on a login page that is not part of a tab group. If the login is successful, the app goes to another window where the user will press 1 of two buttons that will take them to a tab group that corresponds to the button that they pressed. So they could go to there Store tabgroup and manage their store or they could go to their Blog tab group and manage their blog. They also have to be able to press a back button that will back them out till they reach the login page.

How do you mix single window views and tab groups and the navigation between them. So that pressing the left navigation back button took them back to the login page? Can you go from a single window view to a tab group and back?

thanks

    • Every TabGroup ( Store or Blog ) has a Setting tab containing a Return to Home button that allow you to get back Home screen.

      — commented April 21st 2012 by Minh Nguyen

    • Back button on every tab of tabgroup:

      function openTabGroup(e) {
    var tg = Ti.UI.createTabGroup();
    var logoutBtn = Ti.UI.createButton({title: 'Return to Home', width: 200, height: 40});

    if ( e.source._id == 'store' ) {
        var win1 = Ti.UI.createWindow({backgroundColor: 'blue'});
        var win2 = Ti.UI.createWindow({backgroundColor: 'green'});
        win1.leftNavButton = logoutBtn;
        win2.leftNavButton = logoutBtn;

        var t1 = Ti.UI.createTab({title: 'Tab 1', window: win1});
        var t2 = Ti.UI.createTab({title: 'Tab 2', window: win2});
        tg.addTab(t1);
        tg.addTab(t2);    

    } else {
        var win1 = Ti.UI.createWindow({backgroundColor: 'black'});
        var win2 = Ti.UI.createWindow({backgroundColor: 'white'});
        var win3 = Ti.UI.createWindow({backgroundColor: 'brown'});

        win1.leftNavButton = logoutBtn;
        win2.leftNavButton = logoutBtn;
        win3.leftNavButton = logoutBtn;

        var t1 = Ti.UI.createTab({title: 'Tab 1', window: win1});
        var t2 = Ti.UI.createTab({title: 'Tab 2', window: win2});
        var t3 = Ti.UI.createTab({title: 'Tab 3', window: win3});
        tg.addTab(t1);
        tg.addTab(t2);
        tg.addTab(t3);
    }

    tg.open();

    logoutBtn.addEventListener('click', function(){
        tg.close();
        tg = null;
    });
}
