  • Give the code below a try

    If you can figure out how to hide/show the tab bar after opening the tab group and navigating through the nav groups, please post.

    var tabGroup = Titanium.UI.createTabGroup();
var base1 = Titanium.UI.createWindow({
    navBarHidden : true,
});
var base2 = Titanium.UI.createWindow({
    navBarHidden : true,
});
var base3 = Titanium.UI.createWindow({
    navBarHidden : true,
});
var base4 = Titanium.UI.createWindow({
    navBarHidden : true,
});
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////

var window1 = Titanium.UI.createWindow({
    title : '1',
    url : '1.js',
});
var window2 = Titanium.UI.createWindow({
    title : '2',
    url : '2.js',
});
var window3 = Titanium.UI.createWindow({
    title : "3",
    url : '3.js',
});
var window4 = Titanium.UI.createWindow({
    title : "4",
    url : '4.js',
});
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////

var nav1 = Titanium.UI.iPhone.createNavigationGroup({
    window : window1,
});
var nav2 = Titanium.UI.iPhone.createNavigationGroup({
    window : window2,
});
var nav3 = Titanium.UI.iPhone.createNavigationGroup({
    window : window3,
});
var nav4 = Titanium.UI.iPhone.createNavigationGroup({
    window : window4,
});
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////

base1.add(nav1);
base2.add(nav2);
base3.add(nav3);
base4.add(nav4);
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////

window1._navGroup = nav1;
window2._navGroup = nav2;
window3._navGroup = nav3;
window4._navGroup = nav4;
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////

var tab1 = Titanium.UI.createTab({
    icon : Titanium.UI.iPhone.SystemIcon.FAVORITES,
    title : 'Favorites',
    window : base1
});
var tab2 = Titanium.UI.createTab({
    icon : Titanium.UI.iPhone.SystemIcon.HISTORY,
    title : 'Recent',
    window : base2
});
var tab3 = Titanium.UI.createTab({
    icon : Titanium.UI.iPhone.SystemIcon.HISTORY,
    title : 'History',
    window : base3
});
var tab4 = Titanium.UI.createTab({
    icon : Titanium.UI.iPhone.SystemIcon.SEARCH,
    title : 'Search',
    window : base4,
});
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////

tabGroup.addTab(tab1);
tabGroup.addTab(tab2);
tabGroup.addTab(tab3);
tabGroup.addTab(tab4);
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////
//////////////////////////////////////////////////////////

tabGroup.open();
