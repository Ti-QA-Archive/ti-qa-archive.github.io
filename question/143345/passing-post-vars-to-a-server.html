Passing POST vars to a server

I'm trying to post a JSON string (as a POST var) to a server via HTTPClient, but all I get back is <null> on e.error.

Here is my code:

var now = new Date().getTime();
    var xhr = Ti.Network.createHTTPClient({
        onload: function(e) {
            alert('success');
            Ti.API.info(this.responseText);
        },
        onerror: function(e){
            Ti.API.debug(e.error);
        }
    });
    xhr.open("POST", 'My URL');
    var jsonObj = {
        "Id" : -1,
        "ActionTime" : now,
        "Detail": "",
        "GeoZoom": "",
        "ItemID": id,
        "QuadTree": "",
        "StatusType": 1,
        "StatusTypeString": "",
        "TimeZoom": "",
        "UserActionType": type,
        "UserId": Ti.App.userInfo.Id,
        "UserInfo": "",
        "UserHash": '"'+Ti.App.userInfo.Hash+'"',
        "_brokenRules": ""
    };
    console.log(jsonObj);
    var params = "p_UserAction=" + JSON.stringify(jsonObj);
    xhr.send(params);

I'm not sure if this is a titanium problem or a server side problem, is there a way I can debug this or see what the response from the server was (even on error)? Or is there a different way I should do POST?

  • Got it working (finally), this is what worked for me.

    xhr.open("POST", 'My URL');
    var jsonObj = {
        "Id" : -1,
        "ActionTime" : '\/Date('+now+')\/',
        "Detail": "",
        "GeoZoom": "",
        "ItemID": id,
        "QuadTree": "",
        "StatusType": 1,
        "StatusTypeString": "",
        "TimeZoom": "",
        "UserActionType": type,
        "UserId": Ti.App.userInfo.Id,
        "UserInfo": null,
        "UserHash": Ti.App.userInfo.Hash,
        "_brokenRules": null
    };
    xhr.setRequestHeader("Content-Type", "application/json;charset=UTF-8");
    xhr.send('{"p_UserAction":'+ JSON.stringify(jsonObj)+'}');
