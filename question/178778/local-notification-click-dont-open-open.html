alloy.js

bg.js

if (Ti.Platform.name == "iPhone OS" && parseInt(Ti.Platform.version.split(".")[0]) >= 8) {
    Ti.App.iOS.registerUserNotificationSettings({

            Ti.App.iOS.USER_NOTIFICATION_TYPE_ALERT,

    alertAction: "Snooze",
