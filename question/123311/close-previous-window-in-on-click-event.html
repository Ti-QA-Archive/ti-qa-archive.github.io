close previous window in on click event

Hi,
I am creating one window , adding few UI components to it .Then i want to open a new window(through url property) with on click of a button. In the onclick event of the button i want to close the previous window which holds the UI components and open the window with url property.
So i m doing this

var win=Titanium.UI.currentWindow;


    var search = Titanium.UI.createSearchBar({
        barColor:'red', 
        showCancel:true,
        height:40,
        top:0,
    });

        win.add(search);

        var label1= Titanium.UI.createLabel({
            top:40,
            width:150,
            height:50,
            color:'yellow',
            left:150,
            font:{fontWeight:'bold'},
            text:'Yahoo'
        });
    win.add(label1);

    var button=Titanium.UI.createButton({
            title:'ABCD',
            top:320,
            width:70,
            height:40,
        });

        win.add(button);


button.addEventListener('click',function(){
            win.close();
            var win2=Titanium.UI.createWindow({
                url:'secondwin.js',
                title:'second',
                backgroundColor:'white'

            });
            win2.open({modal:true});
        });

But the issue which i am facing is onclick of the button the second window means win2 gets open and closed automatically. why this is happening.

Thanks.
