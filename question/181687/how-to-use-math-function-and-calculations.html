function AddUpdateRecord() {
    var id = 0;
    if(hdneditId.Value != "")
        id = parseInt(hdneditId.Value);
        #region Calculation

    var k1 = 0.0;
    if (txtLFlatK1.Text != "")
        k1 = parseFloat(txtLFlatK1.Text);

    var axisOfK1_deg = 0.0;
    if (txtLFlatK1Axis.Text != "")
        axisOfK1_deg = parseFloat(txtLFlatK1Axis.Text);

    var k2 = 0.0;
    if (txtLSteepK2.Text != "")
        k2 = parseFloat(txtLSteepK2.Text);

    var axisOfK2_deg = 0.0;
    if (txtLSteepK2Axis.Text != "")
        axisOfK2_deg = parseFloat(txtLSteepK2Axis.Text);

    var KNotation = parseInt(Common.KNotation.D);
    if (k1 >= 6.75 && k1 <= 9.64) {
            k1 = 1000 * (1.336 - 1) / k1;
            KNotation = parseInt(Common.KNotation.mm);
    }

    if (k2 >= 6.75 && k2 <= 9.64) {
        k2 = 1000 * (1.336 - 1) / k2;
        KNotation = parseInt(Common.KNotation.mm);
    }

    var IOL = 10.0;
    if (ddlIOLSperical.SelectedValue != "")
        IOL = parseFloat(ddlIOLSperical.SelectedValue);

    var M_SIA = 10.0;
    if (txtLSIA.Text != "")
        M_SIA = parseFloat(txtLSIA.Text);

    var IL_deg = 0.0;
    if (txtLAxis.Text != "")
        IL_deg = parseFloat(txtLAxis.Text);

    var POA = k2 - k1;
    var axisofPOA_deg = parseFloat(axisOfK2_deg);

    var SIA = parseFloat(M_SIA);
    var axisofSIA_deg = parseFloat(IL_deg);
    if (IL_deg > 90)
        axisofSIA_deg = axisofSIA_deg - 90;
    else if (IL_deg < 90)
        axisofSIA_deg = axisofSIA_deg + 90;

    // C1 = A1 @ Q1
    var A1 = parseFloat(POA);
    var Q1_deg = parseFloat(axisofPOA_deg);

    // C2 = A2 @ Q2
    var A2 = parseFloat(SIA);
    var Q2_deg = parseFloat(axisofSIA_deg);

    var Crx = A1 * Math.Cos(2 * (Q1_deg * (3.14/180))) + A2 * Math.Cos(2 * DegreeToRadian(Q2_deg));
    var Cry = A1 * Math.Sin(2 * (Q1_deg * (3.14/180))) + A2 * Math.Sin(2 * DegreeToRadian(Q2_deg));

    var CC = Math.Sqrt((Crx * Crx) + (Cry * Cry));

    var axisofCC_deg = 0;
    #region axisofCC
    if (CC < 0) {
        axisofCC_deg = 0;
    }
    else {
        if (Crx < 0) {
            axisofCC_deg = 90 + 28.66 * Math.Atan(Cry / Crx);
        }
        else {
            if (Cry < 0) {
                axisofCC_deg = 180 + 28.66 * Math.Atan(Cry / Crx);
            }
            else {
                if (Crx < 0) {
                    axisofCC_deg = 28.66 * Math.Atan(Cry / Crx);
                }
                else {
                    if (Cry < 0) {
                        axisofCC_deg = 28.66 * Math.Atan(Cry / Crx);
                    }
                    else {
                        if (Crx > 0) {
                            axisofCC_deg = 28.66 * Math.Atan(Cry / Crx);
                        }
                        else {
                            if (Cry > 0) {
                                axisofCC_deg = 28.66 * Math.Atan(Cry / Crx);
                            }
                        }
                    }
                }
            }
        }
    }
