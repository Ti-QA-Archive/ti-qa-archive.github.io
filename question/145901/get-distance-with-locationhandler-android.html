    • This is a simplified construct for the handler and other buttons to get you started, this is not a complete solution - if it was I would have created your entire app for you :)

      var points = {
    gotFirstRead: false,
    start: {
        lat: 0,
        lon: 0
    },
    current: {
        lat: 0,
        lon: 0
    }
};
function locationHandler(e) {
    //
    if (e.success) {       
        if (points.gotFirstRead === false) {
            points.gotFirstRead = true;
            points.start.lat = e.coords.latitude;
            points.start.lon = e.coords.longitude;
        } else {
            points.current.lat = e.coords.latitude;
            points.current.lon = e.coords.longitude;
        }
    }
}
var btnStart = Ti.UI.createButton({
    title: 'Start'
});
win.add(btnStart);
var btnStop = Ti.UI.createButton({
    title: 'Stop'
});
win.add(btnStop);

btnStart.adEventListener('click', function (e) {
    Ti.Geolocation.addEventListener("location", locationHandler);
});
btnStop.adEventListener('click', function (e) {
    Ti.Geolocation.removeEventListener("location", locationHandler);
    alert(JSON.stringify(points));
    alert(velo.getDistance(points.start.lat, points.start.lon, points.current.lat, points.current.lon);
});

      This should help you get started.
