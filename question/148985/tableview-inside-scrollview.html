    Sudeep,you need to define scrollview and tableview separately.I had the similar situation and this was the way I had done,hope it helps you:

    var win=Ti.UI.createWindow({
    backgroundColor:'white',
})
var view1=Ti.UI.createView({
    top:0,
    height:200,
    width:'auto',
    backgroundColor:'green'
})
var view2=Ti.UI.createView({
    top:220,
    height:200,
    width:'auto',
})
var scr=Ti.UI.createScrollView({
    top:0,
    contentWidth:'auto',
    contentHeight:'auto',
    scrollType:'vertical',
    showVerticalScrollIndicator:true
})
// Create a Label.
var label1 = Ti.UI.createLabel({
    text : 'Scrollview begins....',
    color : 'black',
    font : {fontSize:14},
    height : 40,
    width :'auto',
    top : 10,
    left : 50,
    textAlign : 'center'
});
scr.add(label1);
var label2 = Ti.UI.createLabel({
    text : 'ScrollView ends......',
    color : 'black',
    font : {fontSize:14},
    height : 40,
    width :'auto',
    top : 600,
    left : 50,
    textAlign : 'center'
});
scr.add(label2);
view1.add(scr);

// Create a TableView.
var aTableView = Ti.UI.createTableView({
    top:0
});
// Populate the TableView data.
var data = [
    {title:'Row 1', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 2', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 3', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 4', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 5', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 6', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 7', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 8', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 9', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 10', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 11', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 12', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 13', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 14', hasCheck:true, color:'blue', header:'Second'},
    {title:'Row 15', hasCheck:true, color:'blue', header:'Second'},
];
aTableView.setData(data);
// Add to the parent view.
view2.add(aTableView);

//add views to the window
win.add(view1);
win.add(view2);
win.open();
