  • I do this:

    var r_ajax =  function(url,on_ok, on_err, method) {
    var method = !!method ? method : 'POST';
    var xhr = Titanium.Network.createHTTPClient();

    xhr.onload = on_ok;
    xhr.onerror = on_err;

    xhr.open(method,url);

    return {
        xhr:xhr,
        send:function(data) { xhr.send(data); }
    };
};

    I set onload first , then open

    ( this is tested on iPhone )
