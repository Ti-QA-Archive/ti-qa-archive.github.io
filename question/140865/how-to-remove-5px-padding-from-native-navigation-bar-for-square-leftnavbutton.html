  • Got it working, obviously not ideal as it's a bit of a hack but it accomplishes what I was trying to do:

        var infoBtn = Ti.UI.createButton({
        style:0,
        backgroundImage:L('infoBtn'),
        backgroundSelectedImage:L('infoBtnSelected'),
        width:51,height:44,left:0,top:0
    });
    infoBtn.addEventListener('touchstart', function(e) {
        //touch area larger than button size for some reason
        if (e.x <= infoBtn.width && e.y <= infoBtn.height) {
            masterContainerWindow.setBarImage("barImageLeftSelected.png");
        }
    });
    infoBtn.addEventListener('touchmove', function(e) {
        //user may press and drag away from the button
        //button becomes unselected 70 pixels away
        if (e.x >= infoBtn.width + 70 || e.y >= infoBtn.height + 70) {
            masterContainerWindow.setBarImage("barImage.png");
        } else {
            masterContainerWindow.setBarImage("barImageLeftSelected.png");
        }
    });
    infoBtn.addEventListener('touchend', function(e) {
        masterContainerWindow.setBarImage("barImage.png");
    });
    var masterContainerWindow = Ti.UI.createWindow({
        barImage:"barImage.png",
        barColor:"white",
        leftNavButton:infoBtn
    });
    masterContainerWindow.add(homeView);

    Seems like a lot of effort but I'm very happy with the visual and performance results.
