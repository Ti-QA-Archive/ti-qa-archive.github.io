    Hi Aparna,

    Thanks for your concern.

    Please try my test code and see if it helps you.

    var win = Ti.UI.createWindow({
    backgroundColor:'#000',
});
var picker = Ti.UI.createPicker({
         width:'50%',
         top:30,
         left:70,
     }); 
    var data = [];
    data.push(Titanium.UI.createPickerRow({title:'Blood Group'}));
    data.push(Titanium.UI.createPickerRow({title:'A+'})); 
    data.push(Titanium.UI.createPickerRow({title:'A-'}));
    data.push(Titanium.UI.createPickerRow({title:'AB+'}));
    data.push(Titanium.UI.createPickerRow({title:'AB-'}));
    data.push(Titanium.UI.createPickerRow({title:'B+'})); 
    data.push(Titanium.UI.createPickerRow({title:'B-'}));
    data.push(Titanium.UI.createPickerRow({title:'O+'}));
    data.push(Titanium.UI.createPickerRow({title:'O-'}));
    picker.add(data);

picker.addEventListener('change',function(e){
    if(e.row.title=='Blood Group')
        title="";
    else 
           title=e.row.title;
           alert(title);
});

win.add(picker);
win.open();

    If you have any concern about my answer then feel free to ask.

    Thanks.
