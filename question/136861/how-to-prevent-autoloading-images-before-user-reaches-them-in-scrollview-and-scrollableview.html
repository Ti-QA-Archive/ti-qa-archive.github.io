  • Hi,
    this problem I know. You can prevent autoloading by setting the image-property to defaultImage and setting the image url by event listener.

    Additionaly you can use an image cache.

    var rows = [];
rows[i] = Ti.UI.createTableViewRow({height:60});
rows[i][.add(Ti.UI.createImageView({width:100,height:60,image:'DUMMY',imageset:false,myImage:'IMAGEtoshow'}))
………

tableView.addEventListener('scroll',function(e){
    var y = e.contentOffset;
    // calculation of indexes via tableviewheight and y
    rows[i].setImage(rows[i].myImage;
    rows[i].imageset=true);

})
