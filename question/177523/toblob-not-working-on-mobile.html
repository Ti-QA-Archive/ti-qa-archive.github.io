var win = Ti.UI.createWindow();
var image = Ti.UI.createImageView({

  width: 20,
  top:20,
  height: 20,
  image:'KS_nav_ui.png'
});

var blob = image.toBlob();

var image1 = Ti.UI.createImageView({

  width: 20,
  top:50,
  height: 20,
  image:blob
});


win.add(image);
win.add(image1);
win.open();

  • Accepted Answer

    toBlob() is support on only iOS and Android, not MobileWeb. The Kitchen Sink demo app was written primarily to showcase APIs on the native mobile platforms. MobileWeb is limited in many ways, due in large part to the limitations of the browser environment compared to the native operating systems support for such features. You should not expect 1:1 parity between native and browser-based functionality.
