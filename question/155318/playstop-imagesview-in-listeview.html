Play/Stop ImagesView in ListeView?

I have a listview with animated images, but I can not start the animation ( imageIV.play() or imageIV.stop() ). How can I do?

var win = Ti.UI.createWindow();

var images = [];
for(var i=1; i<13; i++)
    images.push("/images/loading/"+i+".png");

var itemTemplate = {
    childTemplates: [
        {
            type: 'Ti.UI.Label',  
            bindId: 'label',  
            properties: {
                left : "10dp",
                width : "100dp"   
            }
        },
        {
            type: 'Ti.UI.ImageView',  
            bindId: 'imageIV',  
            properties: {   
                top : "10dp",
                left : "110dp",
                width : "80dp",
                height : "80dp",
                images : images    
            }
        }                
   ]
};

var imagesLV = Ti.UI.createListView({
    templates : {'itemTemplate': itemTemplate},
    defaultItemTemplate : 'itemTemplate',
    backgroundColor : "transparent"
});

var data = [];
for(var j=0; j<5; j++){
    data.push({                
        // Sets the regular list data properties
        template : "itemTemplate",
        properties : {
            itemId: 'row ' + (j + 1),
            accessoryType: Ti.UI.LIST_ACCESSORY_TYPE_NONE,
            backgroundColor : "#fff",
            left : "0dp",
            right : "0dp",
            height : "100dp"
        },
        label : {
            text : "Row n. " + j
        },
        imageIV : {
            images : images
        }
    });
};

var section = Ti.UI.createListSection({items: data});
imagesLV.sections = [section]

win.add(imagesLV);
win.open();

  • ok, I found a solution :))
    I write it down, maybe it could help someone ;)

    var win = Ti.UI.createWindow();

var images = [];
for(var i=1; i<13; i++)
    images.push("/images/loading/"+i+".png");

var itemTemplate = {
    childTemplates: [
        {
            type: 'Ti.UI.Label',  
            bindId: 'label',  
            properties: {
                left : "10dp",
                width : "100dp"   
            }
        },
        {
            type: 'Ti.UI.ImageView',  
            bindId: 'imageIV',  
            properties: {   
                top : "10dp",
                left : "110dp",
                width : "80dp",
                height : "80dp",
                images : images ,
                repeatCount : 0
            },
            events: {
                postlayout: function(e){
                    //Ti.API.info(JSON.stringify(e))
                    var img = e.source;                    
                    img.start()
                    setTimeout(function(){
                        img.stop()
                    },10000)
                }
            }            
        }                
   ]
};

var imagesLV = Ti.UI.createListView({
    templates : {'itemTemplate': itemTemplate},
    defaultItemTemplate : 'itemTemplate',
    backgroundColor : "transparent"
});

var data = [];
for(var j=0; j<5; j++){
    data.push({                
        // Sets the regular list data properties
        template : "itemTemplate",
        properties : {
            itemId: 'row ' + (j + 1),
            accessoryType: Ti.UI.LIST_ACCESSORY_TYPE_NONE,
            backgroundColor : "#fff",
            left : "0dp",
            right : "0dp",
            height : "100dp"
        },
        label : {
            text : "Row n. " + j
        }
    });
};

var section = Ti.UI.createListSection({items: data});
imagesLV.sections = [section]

win.add(imagesLV);
win.open();
