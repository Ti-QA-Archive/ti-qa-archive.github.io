Android SearchBar and TableView Height issue with Ti.UI.SIZE

I was going to post this as a JIRA but thought I'd put it up here first.

Titanium Studio, build: 3.1.3, Android Emulator, 4.0.3

With the below code, when search: is set to the searchBar view, the height property of Ti.UI.SIZE is ignored, and the table view takes up the full screen. I've tinkered with it but not found a work around - or am I missing something?

app.js

var win = Ti.UI.createWindow({
    backgroundColor:'#fff'
});

var searchBar = Ti.UI.createSearchBar({
    top      : 0,
    width    : Ti.UI.FILL,
    height   : 43,
    barColor : '#FFF',
    hintText : 'Search'
});
var searchTable = Ti.UI.createTableView({
    //search  : searchBar,
    top     : 0,
    width   : Ti.UI.FILL,
    height  : Ti.UI.SIZE,
    backgroundColor: '#ff0000',       
});

win.add(searchTable);
win.open();
