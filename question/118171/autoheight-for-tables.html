  • if you know about the # of rows, you can set the height dynamically:

    var numRows = ... // e.g. set at runtime
var rowHeight = 50;
var maxTableHeight = 400;
var tableHeight = numRows > 8 ? maxTableHeight : numRows * rowHeight;
// createTableView...

  • Hi,

    Try this

    var win = Ti.UI.createWindow({
       title:'test',
       backgroundImage:'img.png' //some png 
});

var d = [
        {title:'Row 1',hasChild:true},
        {title:'Row 2',hasChild:true},
        {title:'Row 3',hasChild:true},
        {title:'Row 4',hasChild:true},
        {title:'Row 5',hasChild:true},
        {title:'Row 6',hasChild:true},
        ];

var tableView = Ti.UI.createTableView({
    data:d,
    backgroundColor:'transparent',
    minRowHeight:60,
    rowBackgroundColor: '#FFF',
    //style:Ti.UI.iPhone.TableViewStyle.GROUPED, (or)
    footerTitle:''
});
win.add(tableView);

win.open();

    hope it helps
