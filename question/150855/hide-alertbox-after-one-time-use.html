    Hello Rechard,

    Try like this way : 

    var win = Ti.UI.createWindow({
    title : 'Click window to test',
    backgroundColor : 'white',
    exitOnClose : true,
    fullscreen : false
});
win.addEventListener('click', function(e) {
    var dialog = Ti.UI.createAlertDialog({
        cancel : 1,
        buttonNames : ['OK', 'Dont'],
        message : 'Would you like to delete the file?',
        title : 'Delete'
    });
    dialog.addEventListener('click', function(e) {
        if (e.index == 0) {
            dialog.hide();
        } else {
            //
        }
    });
    dialog.show();
});
win.open();

    • Try like this way : 

      var win = Ti.UI.createWindow({
    title : 'Click window to test',
    backgroundColor : 'white',
    exitOnClose : true,
    fullscreen : false
});

var button = Ti.UI.createButton({
    title : "SAmple",
    height : "40dp",
    width : "120dp"
});
win.add(button);
var dialog = Ti.UI.createAlertDialog({
    cancel : 1,
    buttonNames : ['OK', 'Dont'],
    message : 'Would you like to delete the file?',
    title : 'Delete'
});
dialog.addEventListener('click', function(e) {
    if (e.index == 0) {
        Ti.App.Properties.setBool("dialog", true);
        dialog.hide();
    } else {
        //
    }
});
button.addEventListener('click', function(e) {
    if(Ti.App.Properties.getBool("dialog")){        
        dialog.hide();
    }
});
win.open();
