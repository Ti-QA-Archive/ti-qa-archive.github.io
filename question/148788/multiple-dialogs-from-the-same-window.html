  • Hey Jeremy,

    Please find below code, hopefully it will answer your question.

    Ti.UI.backgroundColor = 'white';
var win = Ti.UI.createWindow({
    exitOnClose : true,
});

var picker = Ti.UI.createPicker({
    type : Ti.UI.PICKER_TYPE_DATE,
    minDate : new Date(2009, 0, 1),
    maxDate : new Date(2014, 11, 31),
    value : new Date(2014, 3, 12),
});

var timePicker = Ti.UI.createPicker({
});


var startDate = Ti.UI.createButton({
    title : 'start date',
    height : 50,
    width : 100,
    top : 50,
    left : 10
});

win.add(startDate);

var startTime = Ti.UI.createButton({
    title : 'start time',
    height : 50,
    width : 100,
    top : 50,
    left : 150
});

win.add(startTime);

startDate.addEventListener('click', function(e) {
    picker.showDatePickerDialog({
        callback : function(e) {
            if (!e.cancel) {
                var date = e.value;
                Ti.API.info(' '+date.getDate()+":"+date.getMonth()+":"+date.getFullYear());
                startDate.title = (date.getDate()+":"+date.getMonth()+":"+date.getFullYear());
            }
        }
    });
});



startTime.addEventListener('click', function(e) {
    timePicker.showTimePickerDialog({
        callback : function(e) {
            if (!e.cancel) {
                var time = e.value;
                Ti.API.info(' '+time.getHours()+":"+time.getMinutes());
                startTime.title = (time.getHours()+":"+time.getMinutes());
            }
        }
    });
});


win.open();
