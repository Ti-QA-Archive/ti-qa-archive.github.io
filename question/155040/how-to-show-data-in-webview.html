in test.html

<!DOCTYPE HTML>
<html>

    Try this,

    <!DOCTYPE html>
<html>
<head>
<script>
var messageValue=message;
function assignvalue(){
email.value=messageValue;
}
function displayResult()
{
var x=email.value;
alert(x);
}
</script>
</head>
<body onload="assignvalue()">

<form>
Email: <input type="text" id="email" value="h">
</form>

<button type="button" onclick="displayResult()">Display value</button>

</body>
</html>
