    Hi Fabian do like this

    var view = Ti.UI.createScrollView({
        backgroundColor: '#3e3e3e',
        opacity: 0.6,
        left: 5, right: 5, 
        top: 55, bottom: 5, 
        zIndex: 1,
        contentWidth:'auto', 
        contentHeight:'auto',
        height: 150,  
        showVerticalScrollIndicator:true, 
        showHorizontalScrollIndicator:true,
});
win.add(view);
var textFieldArray=new Array(); 
var top = 5;
    for (var i=1, item; i <= 15; i++) {
        //item = data[i];

        view.add( 

          textFieldArray[i]= Ti.UI.createTextField({
                    color:'#000',
                    height : 30, width : 100,
                    textAlign:'center',
                    borderRadius: 5, borderColor: '#000',
                    borderStyle : Titanium.UI.INPUT_BORDERSTYLE_NONE,
                    right: 5,
                    top: top, bottom: 5,
                    opacity: 0.7,
                    backgroundColor:'white',
                    keyboardType: Ti.UI.KEYBOARD_NUMBER_PAD, 
                    font:{ fontSize:'11dp', fontFamily: 'monospace'},

            })

        ); 

        var top = top + 35;
}
