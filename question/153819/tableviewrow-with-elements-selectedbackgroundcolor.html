exports.ProductList = function(foodSpecies, infoText){
var foodDetailView = Titanium.UI.createView({
backgroundColor : "white"
});

    var tableView = Ti.UI.createTableView({
            height : "100%"
    });

    globals.headerTitle.setText(foodSpecies);
    var DataMangement = require("module/dataManagement");
    var dataManagement = new DataMangement();
    var foodData = dataManagement.getItemForCategory(foodSpecies);
    var infoImagePath = dataManagement.getValidImage(foodSpecies);

    var infoImage = Titanium.UI.createImageView({
            image : infoImagePath,
            width : "auto",
            height : 80
    });

    var infoImageView = Titanium.UI.createView({
            top:0,
            backgroundColor: "white",
            height: 80,
            width: '100%'
    });

    infoImageView.add(infoImage);

    var infoTextArea = Ti.UI.createLabel({
            text : infoText,
            width : "auto",
            font : {
                    fontSize : (Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? "20%" : "25%"
            },
            color : "black",
            left : 5
    });

    var infoTextAreaView = Titanium.UI.createView({
            backgroundColor : "white",
            top:80
    });

    infoTextAreaView.add(infoTextArea);

    var tblData = [];
    var rowInfoText = Ti.UI.createTableViewRow();
    rowInfoText.add(infoTextAreaView);
    rowInfoText.add(infoImageView);
    tblData.push(rowInfoText);

    for (var i = 0; i < foodData.length; i++) {
            var row = Ti.UI.createTableViewRow({
                    index : i+1,
                    selectedBackgroundColor : "#1E558A"
            });

            var leftImageView = Ti.UI.createImageView({
                     image : foodData[i].icon,
                     left : 10,
                     height : (Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? 30 : 50,
                     width : (Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? 30 : 50,
                     top : 15,
                     currentRowIndex : i + 1
            });
            row.add(leftImageView);

            var label = Ti.UI.createLabel({
                    top: (Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? 15 : 15,
                    left : 70,
                    right : (Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? 35 : 53,
                    text : foodData[i].title,
                    color : "black",
                    font : {
                            fontSize:(Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? "20%" : "30%"
                    },
                    touchEnabled : false
            });

            var textArea = Titanium.UI.createTextArea({
                    width : Titanium.Platform.displayCaps.platformWidth,
                    top : 85,
                    height:0,
                    bottom: 0,
                    left : 5,
                    font : {
                            fontSize:(Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? "15%" : "20%"
                    },
                    textAlign : "left",
                    color : "black",
                    backgroundColor : "white",
                    value : foodData[i].infoText,
                    editable : false,
                    zIndex : 1
            });

            var invisibleView = Titanium.UI.createView({
                    top:0,
                    width : "100%",
                    height : 'auto'
            });

            invisibleView.add(label);

            row.addEventListener("click", function(e) {
                    var section = tableView.sections[0];
                    var rs = section.getRows();
                    for (var h=1;h<rs.length;h++)
                            if (h!=e.index && rs[h].children[3].getImage()=='/image/ic_minus.png') {
                                    rs[h].children[2].children[0].top = 15;
                                    rs[h].children[1].height = 0;
                                    rs[h].children[3].setImage('/image/ic_plus.png');
                            }

                    var nameLabel = e.row.children[2].children[0];
                    var textArea = e.row.children[1];

                    if (e.row.children[3].getImage()=='/image/ic_plus.png') {
                            if(nameLabel.text.length > 38) {
                                    textArea.top = (Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? 200 : 250;
                            }
                            nameLabel.top = 15;
                            textArea.height = 'auto';
                            e.row.children[3].setImage('/image/ic_minus.png');
                    } else if (e.row.children[3].getImage()=='/image/ic_minus.png') {
                            textArea.height = 0;
                            e.row.children[3].setImage('/image/ic_plus.png');
                    }
                    tableView.scrollToIndex(e.rowData.index);
            });

            row.add(textArea);
            row.add(invisibleView);
            var plusImageView = Ti.UI.createImageView({
                    image : '/image/ic_plus.png',
                    right : 10,
                    height : (Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? 30 : 50,
                    width : (Titanium.Platform.displayCaps.platformHeight >= 480 && Titanium.Platform.displayCaps.platformWidth == 320) ? 30 : 50,
                    top : 15,
                    currentRowIndex : i + 1
            });             
            row.add(plusImageView);
            tblData.push(row);
    }

    tableView.setData(tblData);

    // Add to the parent view.
    foodDetailView.add(tableView);

    return foodDetailView;

}
