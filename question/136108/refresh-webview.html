  • This is my code…and I just need it so that everytime the application comes up, whether it's running in the background or not, it refreshes.

    var webview = Titanium.UI.createWebView({url:'http://google.com'});
    var window = Titanium.UI.createWindow();
    window.add(webview);
    window.open({modal:true});
