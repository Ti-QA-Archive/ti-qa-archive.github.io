    Hi Onoma,

    to get rid of the white line in your TableView, set the separatorColor to transparent.

    var t = Ti.UI.createTableView({
    data : [r],
    backgroundColor : "transparent",
    allowsSelection : false,
    separatorColor : 'transparent'
});

    Greetings,

    Matthias

  • Hi

    Not answer you have one of those, this is a answer to your follow-up question to Rainer. BTW Rainer is right and I certainly agree with his suggestion that a webView and a table should not mix (in most circumstances) also it is the wrong tool for the job. But as you are not sure of a better solution here is one.

    var data = [
    { title: 'One' },
    { title: 'Two' },
    { title: 'Three' },
    { title: 'Four' },
    { title: 'Five' }
];

function legend(obj) {
    var view = Ti.UI.createView({
        backgroundColor: '#000',
        height: Ti.UI.SIZE,
        layout: 'vertical',
        width: Ti.UI.FILL
    });
    var lblCompany = Ti.UI.createLabel({
        color: '#fff',
        height: Ti.UI.SIZE,
        left: 10,
        text: obj.company,
        textAlign: 'left',
        top: 10,
        width: Ti.UI.SIZE
    });
    view.add(lblCompany);
    var lblPowered = Ti.UI.createLabel({
        color: '#fff',
        height: Ti.UI.SIZE,
        left: 10,
        text: obj.powered,
        textAlign: 'left',
        top: 5,
        width: Ti.UI.SIZE
    });
    view.add(lblPowered);
    var lblVersion = Ti.UI.createLabel({
        bottom: 10,
        color: '#fff',
        height: Ti.UI.SIZE,
        left: 10,
        text: obj.version,
        textAlign: 'left',
        top: 20,
        width: Ti.UI.SIZE
    });
    view.add(lblVersion);
    return view;
}
var headerView = legend({
    company: 'Twente Travel Guide',
    powered: 'Powered by kutNL.',
    version: 'Version: 1.1'
});
var tbl = Ti.UI.createTableView({
    data: data,
    headerView: headerView,
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
win.add(tbl);

    This uses the headerView property and three labels to give you hat you need with significantly less resources and better performance.
