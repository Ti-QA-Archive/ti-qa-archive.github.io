Update text for Slider Label within table row ... not working for Android. Any ideas?

function currentLocationRow(){
            var sliderValTxt = " miles";
            var sliderView = Ti.UI.createView({width:hdc.platformWidth,height:46});
            var slider = Titanium.UI.createSlider({min:1,max:25,value:10,width:220,height:'auto',left:10,bottom:6});
            var sliderLb = Titanium.UI.createLabel({text:'Radius',textAlign:'left',top:0,left:10,width:'auto',font:{fontSize:12,fontWeight:'bold'},color:'#fff',height:'auto'});
            var sliderVal = Titanium.UI.createLabel({text:slider.value ,textAlign:'right',right:10,width:'auto',font:{fontSize:14,fontWeight:'bold'},color:'#fff',height:'auto'});
            sliderView.add(sliderLb);
            sliderView.add(sliderVal);
            if(hdc.os.isAndroid()) { slider.bottom =  0;}
            sliderView.add(slider);
            var row = Ti.UI.createTableViewRow({height: 'auto',className:'newRow',height:rowHeight, width:hdc.ui.deviceWidth(),rowName:'sliderRow'});
            row.add(sliderView);
            tableView.insertRowBefore(0,row,{animated:true});

            slider.addEventListener('change',function(e){
                Ti.API.info("sliderLb.text = " + sliderLb.text);
                sliderLb.text = Math.round(slider.value);
            });


        }

What is weird, the Ti.API.info spits out the correct number, but the actual label on screen never gets updated. Any ideas?

  • This is a BUG … I removed the slider code from the tablerow code and placed it within kitchen sink's slider_basic.js and it works fine. 

    var win = Titanium.UI.currentWindow;

var sliderValTxt = " miles";
var sliderView = Ti.UI.createView({width:'100%',height:46});
var slider = Titanium.UI.createSlider({min:1,max:25,value:10,width:220,height:'auto',left:10,bottom:6});
var sliderLb = Titanium.UI.createLabel({text:'Radius',textAlign:'left',top:0,left:10,width:'auto',font:{fontSize:12,fontWeight:'bold'},color:'#ff0000',height:'auto'});
var sliderVal = Titanium.UI.createLabel({text:slider.value ,textAlign:'right',right:10,width:'auto',font:{fontSize:14,fontWeight:'bold'},color:'#ff0000',height:'auto'});
sliderView.add(sliderLb);
sliderView.add(sliderVal);
sliderView.add(slider);

slider.addEventListener('change',function(e){
    Ti.API.info("sliderLb.text = " + sliderLb);
    sliderLb.text = Math.round(slider.value);
});

win.add(sliderView)
