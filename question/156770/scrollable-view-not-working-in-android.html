    Hi suhas murthy,

    You can use this following code to add image in a Scrollable view. I think i will be helpful for you.

     var win = Ti.UI.createWindow({
    backgroundColor : '#fff'
});

function createRow(i) {
    var row = Ti.UI.createView({
        backgroundColor : 'white',
        borderColor : '#bbb',
        borderWidth : 1,
        width : '100%',
        height : 70,
        top : 0,
        left : 0
    });

var imgView = Titanium.UI.createImageView({
        top : 5,
        left : 5,
        width : 60,
        height : 60,
        image:'about.png'
    });

    row.add(imgView);
    return row;
}

var scrollView = Ti.UI.createScrollView({
    bottom : 10,
    contentHeight : 'auto',
    layout : 'vertical'
});

for (var i = 0; i <= 20; i++) {
    var row = createRow(i);
    scrollView.add(row);
}
win.add(scrollView);

win.open();
