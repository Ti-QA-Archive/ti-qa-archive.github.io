Losing focus when Prepending rows to a TableView

I'm prepending TableViewRows to my TableView, for example: 

var win = Ti.UI.createWindow();
var table = Ti.UI.createTableView();

var i = 1;

function createRow() {
    var height = (i % 2) ? 40 : 90;
    var bgColor = (i % 2) ? "#eee" : "#fff";

    return Ti.UI.createTableViewRow({
        height: height,
        backgroundColor: bgColor,
        title: 'Row' + i++
    });
}

function prependRow() {
    table.insertRowBefore(0, createRow());
}

win.add(table);

setInterval(prependRow, 1000);

win.open();

// insert first (iOS needs this)
table.appendRow(createRow());

The problem is when the user is looking at a specific row in the middle of the TableView it will lose it's focus when a new one is added because it's pushed down. What do you guys suggest?
