  • This code was working, just had to change your request to match the API docs of the service you were trying to reach:

    var win=Titanium.UI.createWindow();
var xhr = Titanium.Network.createHTTPClient();
function soap(){
    //var xhr = Titanium.Network.createHTTPClient();

    xhr.open('POST','http://www.geosolutions.co.il/WebServices/WSGeoMarketing.asmx',true);

    xhr.setRequestHeader('Content-Type','text/xml; charset=utf-8');

    //xhr.setRequestHeader('SOAPAction','http://tempuri.org/HelloWorld');

    //xhr.onreadystatechange=doUpdate();

    var soapRequest= '<?xml version="1.0" encoding="utf-8"?><HelloWorld >';
    xhr.send(soapRequest);
    xhr.onreadystatechange=doUpdate();
    return false;
}



function doUpdate(){
    alert('i m in function doUpdate');
    debugger;
    alert(xhr.readyState);
    if(xhr.readyState==4)
    {

         Ti.API.info('Response from the server: ' + xhr.responseText);
        //var result_as_XML = xhr.responseXML;
        // Now do what you want with the XML
    }
}
soap();
win.open();

    • OK, what worked for me is changing this:

      to

      xhr.onreadystatechange=function() {

      }
