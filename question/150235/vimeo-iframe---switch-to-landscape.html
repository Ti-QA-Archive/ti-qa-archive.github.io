  • Vimeo you can directly embed via webview. There is a special vimeo link.

    case 'youtube' :
                for (var y = 0; y < items[i].content.length; y++) {
                    var movieUrl = "http://www.youtube.com/embed/" + items[i].content[y] + "?fs=1&autoplay=0";
                    container.add(Ti.UI.createWebView({
                        url : movieUrl,
                        top : 10,
                        width : 320,
                        height : 240
                    }));
                }
                break;
            case 'vimeo' :
                for (var y = 0; y < items[i].content.length; y++) {
                    var movieUrl = "http://player.vimeo.com/video/" +  items[i].content[y] +'?title=0&amp;byline=0&amp;portrait=0&amp;autostart=1&amp;color=ffffff' 
                    try {
                        container.add(Ti.UI.createWebView({
                            url : movieUrl,
                            width : 320,
                            height : 400

                        }));
                    } catch (E) {
                    }
                }
                break;
