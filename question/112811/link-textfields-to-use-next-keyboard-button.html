Link TextFields to use "Next" Keyboard button

I'm building a signup for and I'm trying to link text fields so that the "next" button on the soft keyboard takes you to the next field automatically.

I can hack something with something like this:

l_firstname = Titanium.UI.createTextField();
l_lastname = Titanium.UI.createTextField();
l_email = Titanium.UI.createTextField();

l_firstname.addEventListener('blur', function() { 
    l_lastname.focus();
});

But that has obvious issues – if you select a later field, say, email, to trigger the blur event, it forces selection of lastname.

Is there any way to do this short of hacking together event listeners?

  • If you combined all of the above, you would have the answer.

    var first_field = Titanium.UI.createTextField({
    returnKeyType: Ti.UI.RETURNKEY_NEXT
});
first_field.addEventListener('return', function() {
    second_field.focus();
});

  • Here is a function to help everyone

    I put the code bellow on a file named utils.js

    exports.iosKeyboardNavigation = function(controllers, buttonText) {
    if (OS_IOS) { // this constant is only for Alloy
        for (var i = 0; i < controllers.length; i++) {
            var current = controllers[i], nextIndex = i + 1;

            var btn = Titanium.UI.createButton({
                title : buttonText || 'Ok',
                style : Titanium.UI.iPhone.SystemButtonStyle.DONE,
                controller : current,
                nextController : controllers.length > nextIndex ? controllers[nextIndex] : null
            });

            btn.addEventListener('click', function(e) {
                if (this.nextController) {
                    this.nextController.focus();
                } else {
                    this.controller.blur();
                }
            });

            current.keyboardToolbar = [btn];
        };
    };
};

    How to use

    var utils = require('utils');
utils.iosKeyboardNavigation([$.txtFieldOne, $.txtFieldTwo, $.txtFieldThree]);
