  • Hi Nadeem,

    You can select multiple friends from facebook but i am not sure about photo tageed with selected friends

    Please refer below code for select multiple friends from facebook.
    I hope this code may help you. 

    Titanium.Facebook.dialog("apprequests", {
            app_id : 'YOUR_APP_ID',
            message : "YOUR MESSAGE",
            redirect_uri : '****'
        }, function(rr) {
            if (rr.success) {
                var query = "facebook query";
                Titanium.Facebook.request('fql.query', {
                    query : query
                }, function(r) {
                    if (!r.success) {
                        if (r.error) {
                            //error
                        } else {
                            //error
                        }
                        return;
                    }
                    var result = JSON.parse(r.result);
                });

            }
        });
