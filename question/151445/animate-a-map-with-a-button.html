Animate a map with a button!

Hi!
Can someone explain me why this code i wrote do not work?
When i press the buttom… the map would go down from 200 to 300 in 1 sec, but the code give me an error in the part "tView.animate(anim);"

var win = Ti.UI.createWindow({title : "Animation window"});
var tView = Titanium.Map.createView({
    mapType: Titanium.Map.STANDARD_TYPE,
    region: {latitude:37.389569, longitude:-122.050212,
            latitudeDelta:0.1, longitudeDelta:0.1},
    animate:true,
    animated: true,
    regionFit:true,
    userLocation:false,
    top: 200
});

var button = Ti.UI.createButton({
    top : 20,
    height : 20,
    width : 100,
    title : 'Up'
});
button.addEventListener('click', function(e) {
    if(button.title == 'Up') {
        button.title = 'Down'
        var anim = Ti.UI.createAnimation({
            top : 200,
            duration : 1000
        });

        tView.animate(anim);
    } else {
        button.title = 'Up'
        var anim = Ti.UI.createAnimation({
            top : 300,
            duration : 1000
        });

        tView.animate(anim);
    }
});

win.add(button);
win.add(tView);
win.open();

    Hello Giuseppe,

    The problem is you are animate the mapview using animate method but this is the property for the map view.

    Just check below link :

    • animate Property of mapview

    it mean that you call this property as a method of animation. So, That'y you got the error like : 

    Uncaught    TypeError : Property 'animate' of object    #<view> is not a function

    So just try below sample. Hope it works : 

    var win = Ti.UI.createWindow({
    title : "Animation window"
});

var view = Ti.UI.createView({
    top : 200,
    bottom : "0dp",
    width:"100%"    
});
win.add(view);

var tView = Titanium.Map.createView({
    mapType : Titanium.Map.STANDARD_TYPE,
    region : {
        latitude : 37.389569,
        longitude : -122.050212,
        latitudeDelta : 0.1,
        longitudeDelta : 0.1
    },
    animate : true,
    animated : true,
    regionFit : true,
    userLocation : false,
});
view.add(tView);

var button = Ti.UI.createButton({
    top : 20,
    height : 20,
    width : 100,
    title : 'Up'
});
var upAnimation = Ti.UI.createAnimation({
    top : 200,
    duration : 1000
});
var downAnimation = Ti.UI.createAnimation({
    top : 300,
    duration : 1000
});
button.addEventListener('click', function(e) {
    if (button.title == 'Up') {
        button.title = 'Down'
        view.animate(upAnimation);
    } else {
        button.title = 'Up'
        view.animate(downAnimation);
    }
});

win.add(button);
win.open();

    Let me know if it helps

    Thanks
    Jigar
