  • You can use a tabGroup with just one tab and then hide the tabBar; works like a charm, without the navGroup …. 

            var tabGroup;
        var mainWindow;

        // create tab group
        tabGroup = Titanium.UI.createTabGroup();

        //
        // create base UI tab and root window
        mainWindow = Titanium.UI.createWindow({
            title:'Main Window',
            backgroundColor:'#fff',
            tabBarHidden:true
        });
        var tab1 = Titanium.UI.createTab({
            window:mainWindow
        });

        var label1 = Titanium.UI.createLabel({
            color:'#999',
            text:'I am Window 1',
            font: {
                fontSize:20,
                fontFamily:'Helvetica Neue'
            },
            textAlign:'center',
            width:'auto'
        });

        mainWindow.add(label1);

        //
        //  add tabs
        tabGroup.addTab(tab1);

        // open tab group
        tabGroup.open();

    in fact you can use the tabGroup just like a navGroup by doing this

    var someWindow = Titanium.UI.createWindow({
  title:'Some Window',
  backgroundColor:'#fff'
});
tab1.open(someWindow);
