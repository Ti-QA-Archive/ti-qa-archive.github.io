      exports.createWindow = function(){

    var winPub = Ti.UI.createWindow({  
        backgroundColor:'#fff'
    });

        image: 'Default.png',
        top: -20
    });

    var tabs = require('ui/winTabs').createTabs();

    require('BusinessLogic/loadPub').loadBigPub(        
        function update(pub){
            viewGrandePub.addEventListener('load', function(e) {
                setTimeout(function() {
                    winPub.hide();
                    tabs.open();
                },3000);
            });

            viewGrandePub.image = pub.image;
        },

        function error(msg){
            winPub.hide();
            tabs.open();
        }
    );
