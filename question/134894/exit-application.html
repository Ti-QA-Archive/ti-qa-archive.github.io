Exit application

IN my iphone app am using two time outs and it works well ,but my problem is that my app doesn't close full,how can I do this

My Code 


Ti.API.info('FILE NAME : text.js, ACTIVITY : Start');


var win = Titanium.UI.createWindow(
{
    navBarHidden : true,
    exitOnClose: true,
    backgroundColor : '#000'
});


var button = Titanium.UI.createButton({
    title:'To Roll',
    height:35,
    width:130,
    top:10,
    left:20
});

button.addEventListener('click', function() {
    clearTimeout(to2);

    setTimeout(function() {
        alert('2 MIN');
        win.close();
    }, 120000);
});

var to2 = setTimeout(function() {

    win.close();

win.add(button);

win.open();
