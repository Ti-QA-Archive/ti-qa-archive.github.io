Need help posting action with Facebook open Graph

Hello all,

I need some help posting actions to my app's page by using the Facebook Module, it's the latest module and I'm using Titanium 3.2.2.

So, when going to the Facebook developers page and look at my action and object it gives these codes:

For the Action:

FB.api(
  'me/airlineroster:do',
  'post',
  {
    flight: "http://samples.ogp.me/329450007163327"
  },
  function(response) {
    // handle the response
  }
);

And for the object: 

FB.api(
  'me/objects/airlineroster:flight',
  'post',
  {
    app_id: ***********,
    type: "airlineroster:flight",
    url: "http://samples.ogp.me/329450007163327",
    title: "Sample Flight",
    image: "https://fbstatic-a.akamaihd.net/images/devsite/attachment_blank.png",
    departure: "Sample departure: Longitude", //geopoint
    destination: "Sample destination: Longitude", //geopoint
    route: "Sample route: Longitude", //array of geopoints
    description: ""
  },
  function(response) {
    // handle the response
  }
);

Can someone please show me how to post an action with an object using the Facebook module? Been at it for two days now and all I get is an 'Facebook SDK error 5'.
I have [publish_actions] permission when I check it using:

fb.api(me/permissions, 'GET', , function(e) {
    if (e.success) {
        alert("Success! Returned from FB: " + e.result);
    } else {
        if (e.error) {
            alert(e.error);
        } else {
            alert("Unknown result");
        }
    }
});

So something must be off with my code, but I can't seem to figure out what it is.

Regards,

Ron

    Hello Ron,

    See this link Modules.Facebook.

    try something like this :

    var fb = require('facebook');
fb.appid = FACEBOOK_APP_ID;
fb.permissions = ['publish_stream'];


var data = {
    type: "airlineroster:flight",
    url: "http://samples.ogp.me/329450007163327",
    title: "Sample Flight",
    image: "https://fbstatic-a.akamaihd.net/images/devsite/attachment_blank.png",
    departure: "Sample departure: Longitude", //geopoint
    destination: "Sample destination: Longitude", //geopoint
    route: "Sample route: Longitude", //array of geopoints
    description: ""
};

fb.requestWithGraphPath('me/objects/airlineroster:flight', data, 'POST', function(e) {
    if (e.success) {
        alert("Success! Returned from FB: " + e.result);
    } else {
        if (e.error) {
            alert(e.error);
        } else {
            alert("Unknown result");
        }
    }
});

    i hope this helps. thanks.

    • Moehoehaha! Victory!
      You were close Sanat, that's why I marked your answer as best. But the following code did it for me. Just posting for everybody that runs into the same frustration.

      You have to post the action, so in my case the 'do' and use the object, in my case the 'flight' as an imbedded data:

      var data = {
    type: "airlineroster:flight",
    url: "http://samples.ogp.me/***********",
    title: "Sample Flight",
    image: "https://fbstatic-a.akamaihd.net/images/devsite/attachment_blank.png",
    description: ""
};
Ti.API.info(data);

fb.requestWithGraphPath('me/airlineroster:do', {
    flight: JSON.stringify(data),
    access_token: fb.getAccessToken() //It failed on this part, include your access_token
    }, 'POST', function(e) {
        if (e.success) {
            alert("Success! Returned from FB: " + e.result);
        } else {
            if (e.error) {
                alert(e.error);
            } else {
                alert("Unknown result");
            }
        }
});

      I tried this before but didn't include the access_token in the request, so be sure to do so.

      Code strong~
