    Hello Amir,

    Tyr this : 

    var xhr = Ti.Network.createHTTPClient();
    xhr.onload = function() {
        var json = JSON.parse(this.responseText);

        var j; 
        for (j = 0; j < json.length; j++) {
            var data=[]; 
            data[j] = {title:json[j].nom, leftImage:"images/park.png", hasCheck:false}; 
        }

        for (j = 0; j < json.length; j++) {
            var annotation = Titanium.Map.createAnnotation({ 
                animate : true, 
                pincolor : Titanium.Map.ANNOTATION_RED, 
                title : json[j].nom, 
                latitude : json[j].lat, 
                longitude : json[j].longi, 
                image:"images/pin"+json[j].park_pin+".png", 
                rightButton : "images/Next-Track.png"
            });
            mapView.addAnnotation(annotation);
            var k;

            annotation.addEventListener("click", function(n) {
                if(n.clicksource == "rightButton") {
                    for(var d = 0; d< json.length; d++){
                        if(json[j].nom = "' + n.mapView.title + '"){
                                alert(json[d].nom);
                        }
                    }
                }
            });
        }
    };       
    xhr.onerror = function(e) {
        alert('There was an error retrieving the remote data. Try again.');
    }

xhr.open("GET", "http://p.speakmark.com/testvar1.php"); 
xhr.send(); 
win1.add(mapView);
