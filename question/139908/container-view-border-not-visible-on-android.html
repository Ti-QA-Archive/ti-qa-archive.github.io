        bottom : 10

    • Ok See this….

      I made a demo for this & code of this demo is….

      
var win1 = Titanium.UI.createWindow();
win1.open();

var mainContainer = Ti.UI.createView({
    backgroundColor : '#e3e3e3',
    top : 0,
    left : 0,
    right : 0,
    bottom : 0
});
win1.add(mainContainer);

var container = Ti.UI.createView({
    layout : 'vertical',
    backgroundColor : '#d4d4d4',
    left : 10,
    top : 10,
    right : 10,
    bottom : 10,
    borderRadius : 5
});

var content = Ti.UI.createView({
    backgroundColor : '#f8f8f8',
    layout : 'vertical',
    top : 4,
    left : 4,
    right : 4,
    height : 200,
    borderRadius : 5
});
container.add(content);

var content2 = Ti.UI.createView({
    backgroundColor : '#ffffff',
    layout : 'vertical',
    top : 2,
    left : 4,
    right : 4,
    bottom : 5,
    borderRadius : 5
});
container.add(content2);

mainContainer.add(container);
