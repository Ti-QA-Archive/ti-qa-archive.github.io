  • Hi Ahmed

    I had also face same problem in android.
    then i tried to remove all heavy controller like tableview, webview and assign it into null on window close.
    and some where it is working for me.. and also it's good practice to remove controller from parent view and assign it to null

    see reference

    and also if you want clear the application cache it will also affect on memory

    Alloy.Globals.clearAllData = function(e){
    var directory = Ti.Filesystem.tempDirectory;
    // var directory = Ti.Filesystem.applicationCacheDirectory;

    var dataDirectory = Ti.Filesystem.getFile(directory);

    var data = [];
    Ti.API.info('directory = ' + directory);

    if(dataDirectory.exists()){

        //get all sub-directories and file paths in your directory
        data = dataDirectory.getDirectoryListing();

        //if the files still exist, destroy
        for( var i = 0; i < data.length; i++){
            Ti.API.info('FILE = ' + directory,data[i]);
            var fileToDelete = Ti.Filesystem.getFile(directory,data[i]);

            if(fileToDelete.exists()){
                 Ti.API.info('FILE REMOVED');
                fileToDelete.deleteFile();
            }   
        }

    }
}
