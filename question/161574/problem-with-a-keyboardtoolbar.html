Problem with a keyboardToolbar

When 'Back' button is clicked on the navBar, it seems to me that keyboardToolbar stays active so when textField is activated on the another window both toolbars will be shown, toolbar from the current textField and one from the previously closed window. Is this known bug and is there solution? My project demands multiple window with textFileds and keyboardToolbar's attached to them and this is big problem for me now.
The following is my code:

var self = Ti.UI.createWindow({
        title:title,
        backgroundColor:'white'
    });

    var button = Ti.UI.createButton({
        height:44,
        width:200,
        title:L('open window'),
        top:20
    });
    self.add(button);

    var newWindow = Ti.UI.createWindow({
            title: L('newWindow'),
            backgroundColor: 'white'
    });

    var next2 = Titanium.UI.createButton({
        title :'Location New Win',
        style :Titanium.UI.iPhone.SystemButtonStyle.DONE,
        });

    var myToolbar2 = Titanium.UI.iOS.createToolbar({
        items:[next2],
        borderTop:true,
        borderBottom:false
    });

    var myView2 = Ti.UI.createView({
        top:60,
        backgroundColor:'blue',
        height:80
    });

    var myTextField2 = Ti.UI.createTextField({
        top:20, keyboardToolbar: myToolbar2, autocapitalization: Ti.UI.TEXT_AUTOCAPITALIZATION_NONE, autocorrect: false, width:'50%', color:'black', backgroundColor:'white',value:''
    });

    myView2.add(myTextField2);

    newWindow.add(myView2);

    var next = Titanium.UI.createButton({
        title :'Location',
        style :Titanium.UI.iPhone.SystemButtonStyle.DONE,
        });

    var myToolbar = Titanium.UI.iOS.createToolbar({
        items:[next],
        borderTop:true,
        borderBottom:false
    });

    var myView = Ti.UI.createView({
        top:60,
        backgroundColor:'blue'
    });

    var myTextField = Ti.UI.createTextField({
        top:60, keyboardToolbar: myToolbar, autocapitalization: Ti.UI.TEXT_AUTOCAPITALIZATION_NONE, autocorrect: false, width:'50%', color:'black', backgroundColor:'white',value:''
    });

    myView.add(myTextField);

    self.add(myView);

    button.addEventListener('click', function() {
        //containingTab attribute must be set by parent tab group on
        //the window for this work
        self.containingTab.open(newWindow);
    });

  • A simple :

    myTextField2.addEventListener('blur', function(){
    myToolbar2.hide();
});

    seems to be a solution.
