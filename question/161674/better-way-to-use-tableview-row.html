Better way to use Tableview row

The code below is running OK on Android. I think that are a better way to do this, because it fires a eventListener the number of rows in table view for each click. I need it because have to transport the button status and and the labelSection text to other form.

If anybody have a better way to do this, let me know please !

//——————————————————————–
var win = Ti.UI.createWindow({

    navBarHidden:true,

    backgroundColor: 'white',

    top:0

});

var tableData = [];

var tableView = Ti.UI.createTableView({

    top:40

});

for (var i = 0; i < 4; i++){

    var row = Titanium.UI.createTableViewRow({

        height:40

    });

//—————————————————————————–
var labelSection = Titanium.UI.createLabel({

        color : 'black',

        font : {fontSize : 17,fontWeight: 'bold'},

        height : 60,

        width: 400,

        left: 2,

        text: '-- Row ' + i

    }); 
    row.add(labelSection);

//—————————————————————————– 

    var btnDel = Titanium.UI.createButton({

        backgroundImage:'btnTrash.png',

        height: 32,

        width: 32,

        right: 70

    });

    row.add(btnDel);

//—————————————————————————– 

    var btnNext = Titanium.UI.createButton({

        backgroundImage:'btnNext.png',

        height: 32,

        width: 32,

        right: 5

    });        

    row.add(btnNext);

//———————————– EVENTS ————————————–
tableView.addEventListener('click', function(e){
alert(e.row.children[0].text);
});
//——————————————————- 

btnDel.addEventListener('click', function(e){

    alert ( 'btnDel Clicked.');

});

//——————————————————-
btnNext.addEventListener('click', function(e){

        alert('btnNext Clicked.');

    });

//——————————————————- 

labelSection.addEventListener('click', function(e){

    alert('labelSection Clicked');

});

tableData.push(row);       

tableView.setData(tableData);

win.add(tableView);

win.open();

}

    navBarHidden : true,

    backgroundColor : 'white',
