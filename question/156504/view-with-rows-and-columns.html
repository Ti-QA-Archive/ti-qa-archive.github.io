  • You can make a 3 x 3 view like :

    for(i=0;i<3;i++);
{
    var row = Ti.UI.createTableViewRow();
    var col1 = Ti.UI.createView({
        left : 0,
        width : "25%",
        backgroundColor : "blue"
    });
    var col2 = Ti.UI.createView({
        left : "25%",
        width : "25%",
        backgroundColor : "red"
    });
    var col3 = Ti.UI.createView({
        left : "50%",
        width : "25%",
        backgroundColor : "green"
    });

    row.add(col1);
    row.add(col2);
    row.add(col3);

    rowData.push(row);
}

var tblview = Ti.UI.createTableView({
    data : rowData,
    top : 100
});
