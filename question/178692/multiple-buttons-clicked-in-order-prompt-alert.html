  • Something like this perhaps ?

    var BTN_COUNT = 9;

var clickedArray = [];

var btnWidth = parseInt(Ti.Platform.displayCaps.platformWidth / 4);

function allButtonsClicked() {

    for (var x = 0; x < BTN_COUNT; x++) {
        if (clickedArray[x] == false)
            return false;
    }

    return true;
}

function buttonClickedHandler() {

    if (!clickedArray[this.btnId] == true) {
        clickedArray[this.btnId] = true;
        this.enabled = false;
    }

    if (allButtonsClicked()) {
        alert('all clicked');
    }
}

var w = Ti.UI.createWindow();

var v = Ti.UI.createView({
    layout : 'horizontal'
});

for (var x = 0; x < BTN_COUNT; x++) {

    clickedArray[x] = false;

    var b = Ti.UI.createButton({
        width : btnWidth,
        height : btnWidth,
        title : x.toString(),
        left : 0,
        btnId : x
    });

    b.addEventListener('click', buttonClickedHandler);
    v.add(b);
}

w.add(v);

w.open();
