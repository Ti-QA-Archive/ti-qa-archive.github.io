Simple Button Code

Hi
Im trying to open a new page or .js file in current view by clicking a button on thepage; I would like the new view to animate in;

Heres is the current code I have with 3 buttons, could someone show me how to add anevent listener to open a new page if a button is clicked; Thanks

var window = Titanium.UI.currentWindow;

var uwicbutton = Ti.UI.createButton({ top:200, left:150, width:50, height:50 });

var accombutton = Ti.UI.createButton({ top:200, left:150, width:50, height:50 });

var interbutton = Ti.UI.createButton({ top:200, left:150, width:50, height:50 });

window.add(uwicbutton);
window.add(accombutton);
window.add(interbutton);

window.open();
uwicbutton.animate({top:80,left:20,duration:1000});
accombutton.animate({top:80,left:250,duration:1000});
interbutton.animate({top:80,left:130,duration:1000});

  • Accepted Answer

    You just have to create an event listener for each button like so:

    uwicbutton.addEventListener('click', function(e) {

    var win = Ti.UI.createWindow({
        url: 'new_window.js',
        title: 'New animated window',
    });

    win.open( { animated:true });
}
