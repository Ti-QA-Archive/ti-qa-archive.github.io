Catching Errors With Titanium

Hi All,

I have been working with UDP sockets for the last week and i am getting to the stage where i am testing the app.

Usual problems everything works fine but when i close reopen the app i get errors and the app crashes, also if i close lock the phone and reopen the app recreate a socket the app crashes.

How do i catch a error like this one.

[ERROR] :  Error Hit! Error Domain=NSPOSIXErrorDomain Code=48 "The operation couldn’t be completed. Address already in use"

I am using the udp sockets as follows.

var UDP = require('ti.udp');
var socketStarted = false;

/*
 Create the socket.
 */
var connectSocket = UDP.createSocket();
connectSocket.addEventListener('started', function(evt) {
    Ti.API.info('started');
    socketStarted = true;
});

connectSocket.start({
            port : 6454
});

if (socketStarted) {
    connectSocket.sendBytes({
        host : '192.168.100.46',
        data : [12,43,565,43,634,65,7,89]
    });
} else {
    Ti.API.info('SOCKET WAS NOT CONNECTED PROPERLY');
}

Ti.App.addEventListener('resume', function() {

    Ti.API.info('CALLING SOUND RESUME');
        connectSocket.start({
            port : 6454
    });

});
Ti.App.addEventListener('pause', function() {
    if (connectSocket) {
        connectSocket.stop();
    }
});

I understand it says Address already in use but i am closing the socket before reopening the socket.

Any suggestions?
