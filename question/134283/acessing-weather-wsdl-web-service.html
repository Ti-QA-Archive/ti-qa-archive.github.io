  • Why are you doing this

    client.send({xml : soapRequest});

    1) if you have to send a post parameter, you need to have xml in quotes
    2) if you can just send the soap, do client.sen(soapRequest)

  • Try this:

    ctrl.getWeather = function(la, lo) {
    Ti.Yahoo.yql('select * from yahoo.maps.findLocation where q="' + la + ',' + lo + '" and gflags="R"', function(e) {
        var url = 'http://weather.yahooapis.com/forecastjson?w=' + e.data.ResultSet.Results.woeid + '&u=c';
        var xhr = Ti.Network.createHTTPClient({});
        xhr.onload = function() {
            try {
                var weather = JSON.parse(this.responseText);
                Ti.App.fireEvent('weather_ready', {
                    "weather" : weather
                });
            } catch(e) {
                Ti.API.log(e);
            };
        };
        xhr.open('GET', url);
        xhr.send();
    });
};
