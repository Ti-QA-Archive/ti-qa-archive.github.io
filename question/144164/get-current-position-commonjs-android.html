    Well to get a fresh position and not a cache version you can use
    Ti.Geolocation.addEventListener('location', geolocListenerfunction);
    To use the gps:
    Ti.Geolocation.PROVIDER_NETWORK = Ti.Geolocation.PROVIDER_GPS;

    All that inside a commonjs structure

    function geoloc(){
    }
    module.exports = geoloc;

    Check out the API http://docs.appcelerator.com/titanium/latest/#!/api/Titanium.Geolocation

var newDate, settings, mapview;

//Constructor
function permaGeolocation(map,set){

    settings=set;

    settings.minutos=settings.minutos*60000;
 if (Ti.Geolocation.locationServicesEnabled) {
    Ti.Geolocation.purpose = 'Get Current Location';
    Ti.Geolocation.accuracy = Ti.Geolocation.ACCURACY_BEST;
    Ti.Geolocation.distanceFilter = settings.metros;
    Ti.Geolocation.PROVIDER_NETWORK = Ti.Geolocation.PROVIDER_GPS;
    Ti.Geolocation.trackSignificantLocationChange = true;

    Ti.Geolocation.addEventListener('location', geolocListenerfunction);
} else {
    alert('Please enable location services');
}

}

//          turn on/off location services
permaGeolocation.prototype.apagar = function(v) {

//obtener timestamp
function getDate(){
    var currentTime = new Date();
    var hours = currentTime.getHours();
    var minutes = currentTime.getMinutes();
    var month = currentTime.getMonth() + 1;
    var day = currentTime.getDate();
    var year = currentTime.getFullYear();

    return hours +":"+minutes + " - "+day+"/"+month+"/"+year;
}
