audioPlayer progress event not working

I have following code for audio player but progress event of audio player not firing when song is playing. Am i missing some thing in following code? 


var initial_duration_time = 0;
var win = Titanium.UI.createWindow({
    backgroundColor:'#fff'
});

var playBut = Titanium.UI.createButton({
    title: 'Play Song',
    top: 100
});

var slider = Titanium.UI.createSlider({
    min:0,
    value:0,
    width: 200,
    height: 30,
    max: 363,
    top: 170
});

var currentTime = Titanium.UI.createLabel({
    top: 220,
    left: 50,
    text: '0:0'
});

var totalTime = Ti.UI.createLabel({
    top: 220,
    right: 50,
    text: '6:03'
});

var musicPlayer = Titanium.Media.createAudioPlayer();
playBut.addEventListener('click', function(e){
    musicPlayer.setUrl('song1.mp3');
    musicPlayer.start();  
});

musicPlayer.addEventListener('change', function(e){
    Ti.API.info('state: ' + e.state);
});

musicPlayer.addEventListener('progress',function(e){
    Ti.API.info('progress: ' + e.progress);
    var mind1 = initial_duration_time % (60 * 60);
    var minutes1 = Math.floor(mind1 / 60);
    var secd1 = mind1 % 60;
    var seconds1 = Math.ceil(secd1);
    var current_duration = minutes1 + ":" + seconds1;
    currentTime.text = current_duration;
    slider.value = initial_duration_time;
    initial_duration_time ++;
});



win.add(totalTime);
win.add(currentTime);
win.add(slider);
win.add(playBut);
win.open();

  • Here is some code that shows a working progress bar for a remote media file. It should be easy to change this to play a local file.

    //assing total duration in milliseconds and divide by 1000 from your server

var totalDurationFromBackend = 898 // its in your case.Also divided by 1000 

var audioPlayer = Ti.Media.createAudioPlayer({ 

    url: 'http://naturallyimprove.com/mp3/Drone.mp3',
    allowBackground: true
});

var pb=Titanium.UI.createProgressBar
({
   top:10,
   width:250,
   height:'auto',
   min:0,
   max:totalDurationFromBackend,
   value:0,
   color:'#fff',
   style:Titanium.UI.iPhone.ProgressBarStyle.PLAIN,
});
win.add(pb);
pb.show();

audioPlayer.addEventListener('progress',function(e) 
{
   Ti.API.info('Time Played: ' + Math.round(e.progress) + ' milliseconds');

   // set the progress bar's progress value with current time played.. (milliseconds) 
   pb.value =  Math.round(e.progress/1000) ; 
});

startStopButton.addEventListener('click',function() {

// When paused, playing returns false.
// If both are false, playback is stopped.
if (audioPlayer.playing || audioPlayer.paused)
{
    audioPlayer.stop();
    pauseResumeButton.enabled = false;
    if (Ti.Platform.name === 'android')
    { 
        audioPlayer.release();
    }   
}
else
{
    audioPlayer.start();
    pauseResumeButton.enabled = true;
}
 });
