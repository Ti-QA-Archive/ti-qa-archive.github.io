I'm facing a real strange problem. I try to display some texts from a UTF-8 encoded json-file (downloaded with xhr) with some tablerowviews and labels. On iOS everything works perfect, but as soon as I try to start the app on the android emulator the encoding fails. All my "German" special chars like ä, ö or ü appear completely wrong.

I faced the same problem with the Kitchen-Sink decoding example!

Thanks in advance for any help!

    app.behavior.unescape = function(input){
    var text = input;
    text = text.replace(/&(quot|#34);/gi, '\"');
    text = text.replace(/&(amp|#38);/gi, '&');
    text = text.replace(/&(apos|#39);/gi, '\'');
    text = text.replace(/&(lt|#60);/gi, '<');
    text = text.replace(/&(gt|#62);/gi, '>');
    text = text.replace(/&(nbsp|#160);/gi, ' ');
    text = text.replace(/&ldquo;/gi, '\"');
    text = text.replace(/&ndash;/gi, '-');
    text = text.replace(/&middot;/gi, '·');
    text = text.replace(/&#8362;/gi, '?');
    text = text.replace(/&#1488;/g, '?');
    text = text.replace(/&#1489;/g, '?');
    text = text.replace(/&#1490;/g, '?');
    text = text.replace(/&#1491;/g, '?');
    text = text.replace(/&#1492;/g, '?');
    text = text.replace(/&#1493;/g, '?');
    text = text.replace(/&#1494;/g, '?');
    text = text.replace(/&#1495;/g, '?');
    text = text.replace(/&#1496;/g, '?');
    text = text.replace(/&#1497;/g, '?');

    text = text.replace(/&#1500;/g, '?');
    text = text.replace(/&#1501;/g, '?');
    text = text.replace(/&#1502;/g, '?');
    text = text.replace(/&#1503;/g, '?');
    text = text.replace(/&#1504;/g, '?');
