  • Hi

    This should be no more complicated than the following, example code;

    var btn = Ti.UI.createButton({
    text: 'Test Button'
});
btn.addEventListener('click', function (e) {
   // only fires on a button click
});
var data = [
    { title: 'One' },
    { title: 'Two' },
    { title: 'Three' }
];
var tbl = Ti.UI.createTableView({
    data: data
});
tbl.addEventListener('click', function (e) {
   // only fires on a table click - of any child view inside the table or the table itself
});
tbl.addEventListener('scroll', function (e) {
   // only fires on when a table is scrolling
});
