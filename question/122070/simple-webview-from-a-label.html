simple webview from a label

i have a simple js page with labels andtrying to create a simple listener function to open a webview

var foo = Ti.UI.createLabel({ snip

but i cannot work out why this throws an exception:

    foo.addEventListener('click',function(){

    var wv1 = Ti.UI.createWebView({url:'http://blah.com/whatever.htm'});
    win.tab.open(wv1,{animated:true});    

    });

  • Accepted Answer

    you need to make a window that contains a webview and open that window not the webview.

    Try this:

    
foo.addEventListener('click',function(){

    var wv1 = Ti.UI.createWebView({url:'http://blah.com/whatever.htm'});
    var web_win = Ti.UI.createWindow(); 
    web_win.add(wv1);

   win.tab.open(web_win,{animated:true});  

});

  • i think you need to put the webView in a window

    var wv1 = Ti.UI.createWebView({url:'http://blah.com/whatever.htm'});
var w = Ti.UI.createWindow({top:0,left:0,width:'100%',height:'100%'});
w.add(wv1);
win.tab.open(w);
