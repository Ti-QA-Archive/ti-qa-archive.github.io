Acess data inside picker

Hi guys !

So, I have a picker with 4 elements and I want to do, for exemple, an alert of one of those elementes.
How can I do it ? I can't access to any of those elements and I need.

var data = [];
data[0] = Ti.UI.createPickerRow({
    title : '5 km',
    custom_item : '5'
});
data[1] = Ti.UI.createPickerRow({
    title : '10 km',
    custom_item : '10'
});
data[2] = Ti.UI.createPickerRow({
    title : '20 km',
    custom_item : '20'
});
data[3] = Ti.UI.createPickerRow({
    title : '50 km',
    custom_item : '50'
});

var inputraio = Ti.UI.createPicker({
    top : '270 px',
    height : '60 px',
    zIndex : '2',
    visible : false,
    font : FontNormal,
});

inputraio.selectionIndicator = true;
inputraio.add(data);

    Try this:

    inputraio.addEventListener('change', function(e){
    alert(e.row.custom_item);
});

inputraio.setSelectedRow(0, 0, false);
