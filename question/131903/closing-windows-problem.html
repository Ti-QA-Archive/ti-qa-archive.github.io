Closing windows problem

Ok… I'm going to try to explain what I want to do and what my problem is… Basically, I want to create a main menu window (I'm working without tabs) with a start button and a second window with the program, so when you hit the button you go to the second window. This second window has a "back" button, that obviously takes you back to the main menu. All this is done and work fine. The second window's background is an animated photo. The first time I hit the START button, it works perfect: the second window appears with its animated background, and when I hit BACK I go back to the main menu. The problem comes when I hit START again… The second window appears BUT THE BACKGROUND IS GONE! Here's my code:

app.js

Titanium.UI.setBackgroundColor('#000')

Titanium.include('windows/idioma.js');

var main= Titanium.UI.createWindow({
    title:'Main',
    backgroundImage:'images/menu.png'
});

var b_inici= Titanium.UI.createView({
    backgroundImage:'images/start_button_active_CAT.png',
    height:75,
    width:240,
    top:130,
    left:40
});

b_inici.addEventListener('click', function(){
    b_inici.backgroundImage='images/start_button_innactive_CAT.png',
    idioma.open({transition: Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT});
    b_inici.backgroundImage='images/start_button_active_CAT.png'
});

main.add(b_inici);

main.open();

idioma.js


var idioma=Titanium.UI.createWindow({
    title:'idioma'    
});

var images = [];
for (var i=1;i<3;i++)
{
    images.push('images/background' + i+'.tiff');
}

var imageView = Titanium.UI.createImageView({
    images:images,
    duration:100, 
    repeatCount:0,
    top:0
});

imageView.start();


var b2_inici= Titanium.UI.createView({
    backgroundImage:'images/back_button.png',
    height:75,
    width:240,
    top:130,
    left:40
});

b2_inici.addEventListener('click',function(){
    idioma.close({transition: Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});
});

idioma.add(imageView);
idioma.add(b2_inici);

Anyone can see what am I doing wrong? :'(

  • What you want to do is have a focus event listener that will start the images playing, rather then just it playing in-line… also stopping and nulling out the imageView when you close… so like this:

    var idioma=Titanium.UI.createWindow({
    title:'idioma'  
});

var images = [];
for (var i=1;i<3;i++)
{
    images.push('images/background' + i+'.tiff');
}

var imageView = Titanium.UI.createImageView({
    images:images,
    duration:100, 
    repeatCount:0,
    top:0
});

idioma.addEventListener('focus', function() {
    imageView.start();
});

var b2_inici= Titanium.UI.createView({
    backgroundImage:'images/back_button.png',
    height:75,
    width:240,
    top:130,
    left:40
});

b2_inici.addEventListener('click',function(){
    imageView.stop();
    imageView = undefined;
    idioma.close({transition: Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});
});

idioma.add(imageView);
idioma.add(b2_inici);
