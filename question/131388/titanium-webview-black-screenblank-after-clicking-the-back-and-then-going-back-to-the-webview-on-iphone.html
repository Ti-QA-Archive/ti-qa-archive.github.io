Titanium WebView black screen/blank after clicking the Back and then going back to the webview on iphone

Hi guys I'm technically a newbie with Titanium with less than 1 month experience using it, but so far Im very impressed with it.

Right now I'm having problems with webviews. I have a window that contains a link (eventlistener)which opens a webview, when I click the link the WebView appears just fine, but after clicking Back and then clicking the link again to go back to the WebView the WebView is blank, a blank black page appears.

How do I resolve this?

//workflow
Home –> WebView (The WebView displays fine) –> Back –> WebView (Blank Black Page is displayed)

//Below is the code for the webview

var win_with_webview = Titanium.UI.createWindow({  
    title:'Window with webview',    
});

var myWebView = Titanium.UI.createWebView({
   width: 320,
   height: 364,
   top: 0,
   url: 'example.htm'

});
win_with_webview.add(myWebView);

//the eventlistener calling the window with the webview

var tab1 = Titanium.UI.createTab({  
    // icon:'KS_nav_views.png',
    title:'sample app',
    window:win1
});

table_row1.addEventListener('click', function (e) {
    tab1.open(win_with_webview);
});

    Move the constructor of the win_with_webview in the click event.

    table_row1.addEventListener('click', function (e) {

var win_with_webview = Titanium.UI.createWindow({  
    title:'Window with webview',    
});

    var myWebView = Titanium.UI.createWebView({
          width: 320,
          height: 364,
          top: 0,
         url: 'example.htm'
    });

    win_with_webview.add(myWebView);
       tab1.open(win_with_webview);
});
