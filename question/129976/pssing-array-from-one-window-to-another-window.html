  • Hi Rohith,

    You can pass as window property.

    //win 1

var arr = ['1','2','3','4'],

var win2 = Titanium.UI.createWindow({  
        url:'win2.js',
        backgroundColor:'#fff',
        fullscreen:false,
        navBarHidden: false,
        winArr:arr
    });
    win2.open();

//win 2

vat win2 = Ti.UI.currentWindow;

win2.winArr // accessing passed array from win 1
