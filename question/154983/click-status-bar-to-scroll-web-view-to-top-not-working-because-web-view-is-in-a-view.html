var isToggled = false;

function menuFunction(){
    if( !isToggled ){
        main.animate(animateLeft1);
        leftMenu.opacity = 1;
        menu.backgroundImage ='images/close-btn.png';
        menu.backgroundSelectedImage ='images/close-btn-click.png';
        main.add(menuClose);
        isToggled = true;
    } else {
        main.animate(animateRight1);
        menu.backgroundImage ='images/menu-btn.png';
        menu.backgroundSelectedImage ='images/menu-btn-click.png';
        setTimeout(function()
            {
              leftMenu.opacity = 0;
            }, 300 );
        main.remove(menuClose);
        isToggled = false;
    }
}

function socialFunction(){
    if( !isToggled ){
        main.animate(animateLeft2);
        rightMenu.opacity = 1;
        rightMenu.zIndex = 6;
        social.backgroundImage ='images/close-btn.png';
        social.backgroundSelectedImage ='images/close-btn-click.png';
        main.add(socialClose);
        isToggled = true;
    } else {
        main.animate(animateRight2);
        social.backgroundImage ='images/twit-btn.png';
        social.backgroundSelectedImage ='images/twit-btn-click.png';
        setTimeout(function()
            {
              rightMenu.opacity = 0;
              rightMenu.zIndex = 4;
            }, 300 );
        main.remove(socialClose);
        isToggled = false;
    }
}

menu.addEventListener('click',function(e){
