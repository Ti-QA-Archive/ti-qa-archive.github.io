    var win = Ti.UI.createWindow({

    height : '300dp',
    width : '400dp'
});

var overlay = Ti.UI.createView({
    width: Ti.UI.FILL,
    height: Ti.UI.FILL
})
var button = Ti.UI.createButton({
    bottom : '20dp',
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,

});

    Ti.Media.takePicture();

Ti.Media.showCamera({
    success : function(event) {
        imageView.image = event.media;

    cancel : function() {
    },
    error : function(error) {

    allowEditing : false,
    overlay : overlay,
    showControls : false,
