    Hi Abdulrahman Alzanki,

    Please check below code.
    Its a sample code.

    var moment = require('moment');
var currentDate = new Date();
Cloud.Files.query({
    page: 1,
    per_page: 20,
    where: {
         created_at:{"$lte":currentDate, "$gte":new Date(moment().subtract('days', 2))} 
    }
}, function (e) {
    if (e.success) {
    } else {
    }
});
