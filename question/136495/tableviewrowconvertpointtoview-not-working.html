TableViewRow.convertPointToView not working?

Hi, it seems convertPointToView does't work on TableViewRow. I've been battling with this for a a few hours and had no luck.

Little example that illustrates this:

Ti.UI.setBackgroundColor('#000');

var win = Ti.UI.createWindow();
var view = Ti.UI.createView(); 

// Populate the TableView data.
var data = [
    Ti.UI.createTableViewRow({title:'Row 1', hasChild:true, color:'red', header:'First'}),
    Ti.UI.createTableViewRow({title:'Row 2', hasDetail:true, color:'green'}),
    Ti.UI.createTableViewRow({title:'Row 3', hasCheck:true, color:'blue', header:'Second'}),
    Ti.UI.createTableViewRow({title:'Click me to test', color:'orange'})
];

// Create a TableView.
var aTableView = Ti.UI.createTableView({data: data});

data[3].addEventListener('click', function(e) {
    Ti.API.info('e.x: ' + e.x + ', e.y: ' + e.y);
    var point = data[3].convertPointToView(e, view);
    alert('p.x: ' + point.x + ', p.y: ' + point.y);
});

// Add to the parent view.
view.add(aTableView);
win.add(view);
win.open();

convertPointToView returns null.

Tested with SDK 1.8.2 and 2.0.1GA2.
