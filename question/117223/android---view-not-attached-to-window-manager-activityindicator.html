  • Mike did you try using a view instead of a window?
    Also are you "closing" the window using close()?

    Try using a View and instead of closing it hide it.

    You can also try centralizing the system using a compound of Listeners and FireEvent in the app.js

    // in the app.js I do the following:
// Create an object with all the common functions 

var global_functions = {
                        function1: function1,
                        function2 : function2 // etc 
                     };

/**
*  function to communicate with windows
**/
function1(params)
{
    alert(params.message);
    welcome.fireEvent('my_progress_listener')
}

// then create my windows and tabs
//
// welcome TAB
//
welcome = Titanium.UI.createWindow({    
        title : 'Welcome!',
        barColor : '#001122',
        backgroundColor : '#000',
        url : 'main_windows/welcome.js',
        message_txt : 'hello world',
        orientationModes : [
    Titanium.UI.LANDSCAPE_LEFT,
    Titanium.UI.LANDSCAPE_RIGHT,
    Titanium.UI.PORTRAIT
    ],
    // HERE I MAKE MY GLOBAL FUNCTIONS ACCESSIBLE to window
    functions : global_functions   
});

var tab_welcome = Titanium.UI.createTab({    
        icon:'images/yormo_icon.png',
        title:'Welcome to Yormo!',
        window:welcome
});

// ... add the tab


//_________welcome window___________________________

// NOW IN THE welcome WINDOW I MAKE THESE FUNCTIONS ACCESSIBLE:

var win = Titanium.UI.currentWindow;
var functions = win.functions;

// ON PROGRESS
YOU CAN COMMUNICATE WITH THE FUNCTIONS
functions.function1({message : 'hello world'});

// IF YOU NEED THE WINDOW TO REACT TO A CENTRAL FUNCTION 
win.addEventListener('my_progress_event', function(e){
    handle_my_progress()
})

function handle_myProgress()
{
  //bla bla bla
}

/// THE MAIN WINDOW WILL GET FUNCTION1 TO FIRE THE EVENT
// THIS WORKS NICELY WHEN YOU NEED MULTIPLE WINDOWS
// TO WORK TOGETHER
