    Hi andres,

    the toFixed method is pretty unreliable. i made an alternative prototype method derived from this stackoverflow answer

    Number.prototype.toFixedAlt = function toFixed(precision) {
    return (+(Math.round(+(this + 'e' + precision)) + 'e' + -precision)).toFixed(precision);
}

    test case:

    var a = 1.005;

console.log("Normal: a = " + a.toFixed(2));
console.log("Alt: a = " + a.toFixedAlt(2));

    outputs 

    [INFO] :   Normal: a = 1.00 // toFixed fails 
[INFO] :   Alt: a = 1.01
