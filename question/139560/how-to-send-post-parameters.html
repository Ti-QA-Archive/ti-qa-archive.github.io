How to send POST parameters

I am trying to send post request to web-server using below code

Method: POST

Parameters (POST): cId, json (example: {"Text":"test","IsPublic":true})

var addMsgReq = Titanium.Network.createHTTPClient();
var data_to_send = '{"Text": "test text", "IsPublic": "1"}';
addMsgReq.open("POST",serverBaseUrl+'ProMessage/AddMessage?cId=1'+data_to_send);

addMsgReq.onload = function(){
    var response = this.responseText;
    alert("addMsgReq response = "+response);
    Ti.API.info("addMsgReq response = "+response);
};    
addMsgReq.send();

above code doesn't work for me. But I can able to get response in the web browser if run the following query

https://xxx.com//proMessage/AddMessage?cId=1&{"Text":"test","IsPublic":true}

How can I use the above URL in Titanium HTTPClient.

Thanks in Advance

  • Hello Prabhu,

    Try like this,

    
var addMsgReq = Titanium.Network.createHTTPClient();

addMsgReq.onload = function() {
    var response = this.responseText;
    alert("addMsgReq response = " + response);
    Ti.API.info("addMsgReq response = " + response);
};

addMsgReq.open("POST", serverBaseUrl + 'ProMessage/AddMessage?cId=1');
addMsgReq.send({
    Text : 'test',
    IsPublic : 'true'
});

    Best Luck…
