  • Here my working solution:

    view.createClockView = function() {
    var setClock = function() {
        var zeit = new Date();
        var mm = zeit.getMinutes();
        var hh = zeit.getHours() % 12;
        h.setTransform(Ti.UI.create2DMatrix().rotate(hh * 360 / 12));
        m.setTransform(Ti.UI.create2DMatrix().rotate(mm * 360 / 60));
    };                            
    var clockView = Ti.UI.createImageView({
        image:'assets/clock/hafenuhr.png',
        width:320,
        height:240,
       bottom:0
    });
    var center = {"x":160,"y":106};
    var m = Ti.UI.createImageView({
        image:'assets/clock/m.png',
        width:7,
        opacity:0.9,
        height:85,
        center: center,
        anchorPoint : {"x":0.5, "y":0.5}
    });
    var h = Ti.UI.createImageView({
        image:'assets/clock/h.png',
        width:10,
        height:73,
        opacity:0.9,
        center: center,
        anchorPoint : {"x":0.5, "y":0.5}
    });
    clockView.add(m);
    clockView.add(h);
    setClock();
    setInterval(setClock, 60000)
    return clockView;
};
