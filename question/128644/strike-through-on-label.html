  • Szilard Mandici, here's a handy little function

    module.strikeThrough = function(text) {
  var result = '';
  _.each(text.split(''), function(letter) {
       result += letter + '\u0336';
  });
  return result;
}

    Unfortunately not all fonts work, fonts that do work, some look beter than others.
    The following will display an example of all IOS fonts

    function buildTableRow(_font,strike) {
  var strikeText = _font +' Through'
  if (strike == true) {
    strikeText = strikeThrough(strikeText)
  }

    var row = Ti.UI.createTableViewRow();
    var normal = Ti.UI.createLabel({  text: strikeText ,font: { fontSize:16, 
      fontFamily:_font}, left:6, top:2, height:30, width:200 });
    row.add(normal);

    var bold = Ti.UI.createLabel({ text: strikeText, font: { fontSize:16,
      fontWeight:'bold', fontFamily:_font }, left:46, top:21, height:30, 
      width:200 } );
      row.add(bold);
  return row;
}

function showAllFontsWithStrikeThrough() {

    'Hiragino Kaku Gothic ProN','Marker Felt','Thonburi','Times New Roman',
    'Trebuchet MS','Verdana','Zapfino']
  var self = h.createWindow('view fonts')
  var view = Ti.UI.createView();
   var table = Ti.UI.createTableView();
   self.add(table);
   var rows = [];
   _(fonts).each(function(_font) {
    row  = buildTableRow(_font,true)
