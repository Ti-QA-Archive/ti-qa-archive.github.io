putting integers into textfields

When I have 2 text fields and want to put the (integer) value of one to an other (e.g. on blur) it is not possible. When it is a string, it does work. Now I have worked around this (see example). But is there a better way to do this? Is this a bug?

var txtexcl =  Titanium.UI.createTextField({
    color:'#369',
    height:35,
    top:10,
    left:10,
    width:250,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});



var txtbtw =  Titanium.UI.createTextField({
    color:'#369',
    height:35,
    top:55,
    left:10,
    width:250,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});

var txtincl =  Titanium.UI.createTextField({
    color:'#963',
    height:35,
    top:100,
    left:10,
    width:250,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});

txtexcl.addEventListener("blur", function(e){
var btw = 0.19;
var xincl = txtexcl.value*(1+btw);
var xbtw = txtexcl.value*btw;
txtbtw.value=''+xbtw;
txtincl.value=''+xincl;
});
