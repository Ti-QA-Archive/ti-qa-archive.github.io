    Hi Ranjith,

    try this.

    Ti.UI.setBackgroundColor('#fff');
var win1 = Ti.UI.createWindow({
    backgroundColor : '#fff'
});
var itemObj = [];

// create array of 10 TextFields
var totalTextFields = 10;
for (var i = 0; i < totalTextFields; i++) {
    itemObj[i] = Titanium.UI.createTextField({
        name : 'TextFieldName' + i,
        borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
        color : '#336699',
        top : 5 + (i * 35),
        layout : 'vertical',
        hintText : 'TextFieldHintText' + i
    });

};

// add TextFields to the window
for (var i = 0; i < itemObj.length; i++) {
    win1.add(itemObj[i]);
};

// create Button
var myButton = Ti.UI.createButton({
    top : 200,
    title : 'myButton',
    width : 100
});
myButton.addEventListener("click", function(e) {
    // type some text into TextFieldName3 and klick the Button
    alert('Name: '+ itemObj[3].name + '\n' +'Value: '+ itemObj[3].value);
});
win1.add(myButton);

win1.open();
