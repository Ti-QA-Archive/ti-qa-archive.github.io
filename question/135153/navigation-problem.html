    var imagewrapper = Ti.UI.createView({
    height : 222,
    width : 300,
    top : 100
});
var lblEventInfoImage = Titanium.UI.createImageView({
    image : 'a.jpg',
    height : 222,
    width : 300,
    // top : 10
});

imagewrapper.add(lblEventInfoImage);

//alert('Hello'+dd+'\n'+ loc +'\n'+enm);

var lblInfoText = Titanium.UI.createLabel({
    text : "TestInfoText",
    width : 200,
    height : 30,
    left : 20,
    textAlign : 'left',
    color : 'white',
    font : {
        fontSize : 12,
        fontColor : 'white'
    },
    touchEnabled : true,
    top : 200,
    zIndex : 200
});

imagewrapper.add(lblInfoText);

var btnInfo = Titanium.UI.createButton({
    title : '',
    backgroundImage : 'KS_nav_ui.png',
    width : 45,
    height : 45,
    right : 10,
    enabled : true,
    font : {
        fontSize : 12,
        fontColor : 'white'
    },
    touchEnabled : true
});
imagewrapper.add(btnInfo);
