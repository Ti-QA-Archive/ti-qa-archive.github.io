      function ApplicationTabGroup(Window) {
    //create module instance
    var self = Ti.UI.createTabGroup();
    CoreUIWindow = require('ui/common/CoreUIWindow'),
    PatUIWindow = require('ui/common/PatientUIWindow'),
    ProvUIWindow = require ('ui/common/ProviderUIWindow')
    //create app tabs
    var CoreUIWin = new Window(L('Core Measures_win_title')),
        PatUIWin = new Window(L('Patient_win_title'));
        ProvUIWin = new Window(L('Provider_win_title'));

    var CoreUITab = Ti.UI.createTab({
        title: L('Core Measures_win_title'),
        window: CoreUIWin
    });
    CoreUIWin.containingTab = CoreUITab;
    self.addTab(CoreUITab)

    var PatUITab = Ti.UI.createTab({
        title: L('Patient_win_title'),
        window: PatUIWin
    });
    PatUIWin.containingTab = PatUITab;
    self.addTab(PatUITab);

    var ProvUITab = Ti.UI.createTab({
        title: L('Provider_win_title'),
        window: ProvUIWin
    });
    ProvUIWin.containingTab = ProvUITab;

    self.setActiveTab(1);

};

module.exports = ApplicationTabGroup;
