  • thanks Nitin Chavda.

    but not resolved.

    I added the code below to the sample application (Single Window application FirstView) and ran it.
    please see video : 

    //FirstView Component Constructor
function FirstView() {
    var self = Ti.UI.createView();
    var button = Ti.UI.createButton({
        color:'#000000',
        title:"up",
        height:100,
        width:200,
        backgroundColor:"#aaa",
        zIndex:10
    });
    self.add(button);

    var label1 = Ti.UI.createLabel({
        color:'#fff',
        top:10,
        height:50,
        width:100,
        backgroundColor:"#aaa"
    });
    self.add(label1);
    var i=0;
    button.addEventListener('click', function(e) {
        label1.text=i++;
    });
    return self;
}
module.exports = FirstView;
