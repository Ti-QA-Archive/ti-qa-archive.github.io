     num = num.toString();
     num = num.replace(/,/gi,"")
    if(num.length>3){
        var dec = 0 
        var sep = ","
        var decChar = ","
        var pre = ""
        var post = ""
            num = isNaN(num) || num === '' || num === null ? 0.00 : num;
        var n = num.toString().split(decChar);
        return (pre || '') + n[0].replace(/(\d)(?=(\d\d\d)+(?!\d))/g, "$1" + sep) + (n.length > 1 ? decChar + n[1].substr(0,dec) : '') + (post || '');

  • This function works perfectly with this value = 2500000.00

    It returns: 2.500.000,00

    function formatCurrency(amount){
    var delimiter = "."; // replace comma if desired
    amount = new String(amount);
    var a = amount.split('.',2)
    var d = a[1];
    var i = parseInt(a[0]);
    if(isNaN(i)) { return ''; }
    var minus = '';
    if(i < 0) { minus = '-'; }
    i = Math.abs(i);
    var n = new String(i);
    var a = [];
    while(n.length > 3)
    {
        var nn = n.substr(n.length-3);
        a.unshift(nn);
        n = n.substr(0,n.length-3);
    }
    if(n.length > 0) { a.unshift(n); }
    n = a.join(delimiter);
    if(d.length < 1) { amount = n; }
    else { amount = n + ',' + d; }
    amount = minus + amount;
    return amount;
}
