Can't reload button function

Follow is my sample code:

var win = Titanium.UI.currentWindow;

var webview = Titanium.UI.createWebView({
    url:'http://www.google.com/',
    top:0,left:0,//width:500,height:win.height-40
    });
win.add(webview);

var camera = Titanium.UI.createButton({
    backgroundImage:'/images/camera.png',
    height:33,
    width:33
});
camera.addEventListener('click', function()
{    
    camera=null;

    var vd=Ti.UI.createView({
    backgroundColor:'black',
    borderColor:'red',borderWidth:5,borderRadius:5,
    width:250,height:150,opacity:0.5});
    win.add(vd);


    var cancel = Titanium.UI.createButton({
    title:'cancel',    
    backgroundColor:'000',
    height:33,
    width:70});

    vd.addEventListener('touchmove', function(e){ 
    vd.backgroundColor='red';
    var newX = Math.round(e.x + vd.animatedCenter.x - vd.width/2);
    var newY = Math.round(e.y + vd.animatedCenter.y - vd.height/2);
    vd.animate({center:{x:newX,y:newY}, duration:1});

    win.add(cancel);
    cancel.top=newY+vd.height/2+10;
    cancel.left=newX +vd.width/2-cancel.width;

    });
    cancel.addEventListener('click', function(){
    win.remove(vd);
    win.remove(cancel);
    win.removeEventListener('touchmove',e);
    camera=true;
    });    
    });

var toolbar1 = Titanium.UI.createToolbar({
    items:[camera],
    bottom:0,
    borderTop:true,
    borderBottom:false,
    opacity    :0.4,
    //translucent:true,
    barColor:'#000'
});    
win.add(toolbar1);
