    You can not able to use this styles directly in classic. To use this, put it in an object structure and used it in classic like below,

    
var styleObj = {
    "bluetext" : {
            color: 'blue'
    },
    "orangetext" : {
            color: 'orange'
    },
    "shadow" : {
            shadowColor: '#88f',
            shadowOffset: {x:1,y:3}
    }    
}

var lbl = Ti.UI.createLabel(styleObj["bluetext"]);
lbl.applyProperties((styleObj["shadow"]):
