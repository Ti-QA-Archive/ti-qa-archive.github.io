Move a webview up when orientation has changed so...

Ok i have this code set so when the iphone is put into landscape mode two objects are hidden but i also want to change the "top" property of a webview when this event is fired how do i do this.

//////////////// START OF WEB CONTROL ORIENTATION HIDE WHEN IN LANDSCAPE MODE //////////////////

Ti.Gesture.addEventListener('orientationchange',function(e) {
    // get orienation from event object
    var orientation = e.orientation

    switch (orientation)
    {
        case Titanium.UI.PORTRAIT:
            webcontrolsbg.show();
            webcontrolholder.show();
            break;

        case Titanium.UI.LANDSCAPE_RIGHT:
            webcontrolsbg.hide();
            webcontrolholder.hide();
            break;

        case Titanium.UI.LANDSCAPE_LEFT:
            webcontrolsbg.hide();
            webcontrolholder.hide();
            break;
    }
});

///////////////// END OF WEB CONTROL ORIENTATION HIDE WHEN IN LANDSCAPE MODE ///////////////////
