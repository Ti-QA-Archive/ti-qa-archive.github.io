Give a variable to a view, and get it on click event (gives undefined!)

Hello

I have created multiple views in my application, and I would like to give it a custom variable so i can get it on the click event.

Here's my code:

while(booths.isValidRow()) {

    var label = Titanium.UI.createLabel({
        text: booths.fieldByName('name'),
        textAlign: 'center',
        font: {
            fontFamily: 'Helvetica',
            fontSize: Titanium.App.Properties.getDouble("FloorplanFont")
        }
    });

    var name = booths.fieldByName('Name');

    var standView = Ti.UI.createView({
        borderRadius: 0,
        borderColor: 'Black',
    my_name:name,
        borderRadius: 2,
        my_id: booths.fieldByName('BoothId'),
        left: booths.fieldByName('LocationX'),
        top: booths.fieldByName('LocationY'),
        backgroundColor: '#bfbfbf',
        width: booths.fieldByName('Width'),
        height: booths.fieldByName('Height')
    });

    standView.addEventListener('click', function(e){
       var the_name = e.source.my_name;

       Titanium.API.info('user clicked on item: ' + the_name);
    });
    standView.add(label);
    view.add(standView);
    booths.next();
};

This piece of code gives as result: undefined. Output on titanium:

[INFO] user clicked on item: undefined

However when i try to get e.source.width it gives me the correct width!

Can anyone explain what I'm doing wrong here?
I'm using titanium SDK 1.6.1 and iPhone SDK 4.2.

Thanks!
