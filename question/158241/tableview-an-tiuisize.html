TableView an Ti.UI.SIZE

I´m using the TableView UI. The TableView contains one or more TableViewRow(s). Each TableViewRow contain a View with one or more Label(s). The View"Container" for the Labels, have the Height-Propertie: Ti.UI.SIZE. The error: If I will update the text of one label, the TableViewRow "blank" for a short time.
But it works with a solid value(10, 50, 100…) for Height-Propertie.
The error appeared since I used Titanium SDK 3.x.

  • When I change the "height propertie" of the lableContainer it will work, also when I use the 2.14 Titanium SDK.
    But it is really incommodious with an solid value.

    Regards,
    Sebastian

    var win = Ti.UI.createWindow();

var tableData = [];

var labelContainer = Ti.UI.createView({
  hasChild: true,
  backgroundColor: 'black',
  height: 'SIZE',     //it will work with an solid value
  width: '95%',
});

var label= Ti.UI.createLabel({
  color: 'white',
  text:  'test',
  font:{fontSize: 18},
});

labelContainer.add(label);

var row = Ti.UI.createTableViewRow({
   top:10,
   className: 'trial',
   objName: 'trial',
   height: 'SIZE',
});

row.add(labelContainer);

tableData.push(row);

var tableView = Ti.UI.createTableView({
  backgroundColor: 'white',
  data: tableData,
  top: 20,
  separatorStyle: Ti.UI.iPhone.TableViewSeparatorStyle.NONE,
  allowsSelection: false
});

win.add(tableView);

var x = 0;
setInterval(function(){
  x ++;
  label.setText('test: ' + x);
}, 500);

win.open();
