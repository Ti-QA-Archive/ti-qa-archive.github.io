  • Hello Detp,

    Thanks for your inquiry. You can use "Titanium.UI.OptionDialog" with your defined options. Take a look at this code below about, how you can add your own options in option dialog.

    Ti.UI.setBackgroundColor('gray');

var win = Ti.UI.createWindow({
  title: 'Swipe left to See option',
});

var view = Titanium.UI.createView({
   borderRadius:10,
   backgroundColor:'red',
   width:Ti.UI.FILL,
   height:Ti.UI.FILL
});

win.add(view);

var options=[];

for(var i=0,j=5; i<j; i++){
 options.push("Option "+i) ;
};

options.push("cancel") ;

var opts = {
  cancel: 6,
  options: options,
  selectedIndex: 2,
  destructive: 0,
  title: 'Options'
};

view.addEventListener('swipe', function(e){
    if (e.direction == 'left') {
     var dialog = Ti.UI.createOptionDialog(opts).show();
   } else if (e.direction == 'right') {
     alert('Swipe to left');
   }

});

win.open();

    Swipe left to see the options. Here you can use your own option to perform actions. Thanks.
