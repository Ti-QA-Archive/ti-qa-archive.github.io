  • hi,

    try this

    
var txt = Ti.UI.createTextField({
    width : 300,
    height : 40    
});
txt.addEventListener('change',function(e){
    if (e.source.value) {
        if (e.source.value.length > 4) {
            e.source.value = e.source.oldValue;
        } else {
            e.source.oldValue = e.value;
        }
    }
});
