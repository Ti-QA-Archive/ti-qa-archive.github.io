    You're not removing event listener, you need to pass function reference to "removeListenerFunction", something like this:

    var m=0;

var funcRef = function(){
    if(m>=title.length){
    forward_button.removeEventListener(funcRef);
    }
    else{
    webView.html=title[m+1].toString();
    region_direction={latitude:start_latitude[m+1],longitude:start_longitude[m+1],latitudeDelta:0.01, longitudeDelta:0.01};
    mapview.setLocation(region_direction);


    m++;
    }       
}
forward_button.addEventListener('click', funcRef);
