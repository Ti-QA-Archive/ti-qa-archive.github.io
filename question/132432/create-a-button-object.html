/*
 * create a viw object to hold the buttons
 * and add the buttons into the view
 */
function createRatingButtons(numButtons) {
    var bn=0;
    var button = Titanium.UI.createButton({
        title:                 '*',
        width:                 50,
        height:                50,
        color:                "black",
        backgroundColor:    "blue",
        left:                0,
    });
    var ratingView = Titanium.UI.createView({
        height:        200,
        color:        'white',
    });
    for(bn==0;bn<=numButtons;bn++) {
            button.setTitle(bn);
            button.setLeft(button.getLeft+button.getWidth+15);
            ratingView.add(button);
    }
    return ratingView;
}

    I think you need to create button in for loop

    

    var left = 0;
    for(bn==0;bn<=numButtons;bn++) {
            left = left + 50 + 15;//50 is width
            var button = Titanium.UI.createButton({
                title:              bn,
                width:              50,
                height:             50,
                color:              "black",
                backgroundColor:    "blue",
                left:               left,
            });
            ratingView.add(button);
    }
