    Sajid

    It seems as though your aim is for an Object Oriented approach. If so, some of these articles will help you to understand how to do it.

    I have rewritten your code using these techniques. It should run successfully without modification:

    app.js

    Ti.UI.setBackgroundColor('#000');
Ti.include('sub/mainWin.js'); //ensure this is spelled correctly
var mainWin = new mainWindow();
mainWin.openWin.title = "Open";
mainWin.mainWindow.open({transition:Ti.UI.iPhone.AnimationStyle.CURL_DOWN});

    sub/mainWin.js

    function mainWindow() {
    var _this = this;
    this.mainWindow = Ti.UI.createWindow({
        backgroundColor:"red"
    });
    this.openWin = Ti.UI.createButton({
        title:'Open nav group',
        top:150,
        width:200,
        height:40
    });

    this.onBtnClick = function(e){
        Ti.API.info("do fancy stuff");
        var win1 = Ti.UI.createWindow({  
            backgroundColor:'green',
            fullscreen:false,
            title:'Subwin'
        });
        var label1 = Ti.UI.createLabel({
            text:"click this message to return to the previous screen"
        });
        win1.add(label1);

        label1.addEventListener("click", function(){
           win1.close(); 
        });

        win1.open();
    }

    this.openWin.addEventListener('click', this.onBtnClick);
    this.mainWindow.add(this.openWin);

}

    Hope this helps
