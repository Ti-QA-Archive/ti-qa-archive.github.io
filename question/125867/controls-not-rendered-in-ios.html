controls not rendered in iOS

Hello all,

I am trying to port a simple application I wrote on a windows machine for an Android device (which works fine) over to a MAC so I can see how well it works and what changes would need to be made.

Unfortunately the main window, which is a simple login window isn't rendering its controls correclty. After changing some of the proeprty values I've found it seems to be because i'm using percentages to set the height and width of the views and the controls they contain.

I really don't want to use absolute sizes for controls as the application must be able run on both android and apple devices and I want to keep code branching down to a minimum so I was wondering is there anyway to use percentages for layout values of controls on iOS devices?

Here is the code I currently have;

var win1 = Titanium.UI.createWindow({  
    title:'Login',
    backgroundColor:'#fff',
    layout : 'vertical',
    tabBarHidden:true
});

win1.orientationModes = [Ti.UI.LANDSCAPE_LEFT];

var mainView = Titanium.UI.createView({height:'auto', width:'100%',layout:'vertical'}); 
var view1 = Titanium.UI.createView({height:'auto', width:'100%',layout:'horizontal'});
var view2 = Titanium.UI.createView({height:'auto', width:'100%',layout:'horizontal'});
var view3 = Titanium.UI.createView({height:'auto', layout:'vertical'});

var label1 = Titanium.UI.createLabel({
    color:'#999',
    text:'Username:',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'right',
    width:'50%'
});

var label2 = Titanium.UI.createLabel({
    color:'#999',
    text:'Password:',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'right',
    width:'50%'
});

var text1 = Titanium.UI.createTextField({ color:'#336699',
    width:'25%',
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});
var text2 = Titanium.UI.createTextField({ color:'#336699',
    width:'25%',
    passwordMask : true,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED

var button = Titanium.UI.createButton({title : 'Login', visible:true, enabled:true});

view1.add(label1);
view2.add(label2);
view1.add(text1);
view2.add(text2);
view3.add(button);
mainView.add(view1);

mainView.add(view3);
win1.add(mainView);
win1.open();

  • Yes, you can use percentage dimensions with iOS. I made a few small modifications to your layout dimensions and positioning to get you a cross platform layout that should work for you.

    var win1 = Titanium.UI.createWindow({  
    title:'Login',
    backgroundColor:'#fff',
    layout : 'vertical',
    tabBarHidden:true
});

win1.orientationModes = [Ti.UI.LANDSCAPE_LEFT];

var labelWidth = 150;
var view1 = Ti.UI.createView({ 
    top:10,
    height:'auto',
    width: '55%'
});
view1.add(Ti.UI.createLabel({
    color:'#999',
    text:'Username:',
    height:'auto',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    top:5,
    left:0,
    width:'auto'
}));  
view1.add(Titanium.UI.createTextField({ 
    color:'#336699',
    right:0,
    width:'50%',
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
}));  

var view2 = Ti.UI.createView({ 
    top:10,
    height:'auto',
    width: '55%'
});
view2.add(Ti.UI.createLabel({
    color:'#999',
    text:'Password:',
    height:'auto',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    top:5,
    left:0,
    width:'auto'
}));  
view2.add(Titanium.UI.createTextField({ 
    color:'#336699',
    right:0,
    width:'50%',
    passwordMask : true,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
})); 

var button = Titanium.UI.createButton({
    title: 'Login', 
    height:40,
    width:100,
    top:10
});

win1.add(view1);
win1.add(view2);
win1.add(button);
win1.open();
