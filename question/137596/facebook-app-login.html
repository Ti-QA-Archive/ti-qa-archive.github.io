var win = Ti.UI.createWindow();

    height: 60,
    width: 200
});
if(Ti.Facebook.loggedIn) {
    button.title = "Logout";
}
else {
    button.title = "Login";
}
win.add(button); 

button.addEventListener('click', function(e){
