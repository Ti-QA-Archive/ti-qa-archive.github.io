How to get row index?

Hello everyone.
I am new in titanium, and I have a problem with getting index of listener.
This is my code:
var addTask = function(e) {
var row = Ti.UI.createTableViewRow({…});
var label = Ti.UI.createLabel({…});
var DoneCheckbox = Ti.UI.createButton({…});
var DeleteCheckbox = Ti.UI.createButton({…});
var imageShow = "/appicon.png";
if ($.dishes.image == '/Vdishes.png')
imageShow = "/dishes.png";
else if ($.laundery.image == '/Vlaundery.png')
imageShow = "/laundery.png";
var image = Ti.UI.createView({…});
row.add(DoneCheckbox);
row.add(DeleteCheckbox);
row.add(image);
row.add(label);
$.tableView.appendRow(row);
DoneCheckbox.addEventListener("click", function(event) {
row.backgroundColor = "#abce12";
});
DeleteCheckbox.addEventListener("click", function(event) {
$.tableView.deleteRow(row);
});
};

But $.tableView.deleteRow(row); doesn't work.
I understood that for delete the row I need to use the index, but when I tried:
$.tableView.deleteRow(event.index);
event.index = undifined.
I tried also $.tableView.deleteRow(event.source.parent); but nothing.
someone can help?
Thanks alot.

    Hello Yair Cohen,

    Normally you will get the row index using this code.

    $.tableView.addEventListener('click', function(e) {
    alert(e.index);
});
