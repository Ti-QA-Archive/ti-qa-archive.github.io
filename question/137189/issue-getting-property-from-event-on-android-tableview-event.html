    try this

    
tableView.addEventListener("click",function(e){
    if(e.source.hasDetail || e.rowData.hasDetail){
        if(e.source.js || e.rowData.js){
            var w = Titanium.UI.createWindow({
                title:e.source.title,
                backgroundColor:"#FFFFFF",
                dataToPass:e.source.dataToPass || e.rowData.dataToPass,
                url:(e.source.js || e.rowData.js)        });
        }else{  
            var w = Titanium.UI.createWindow({
                title:e.source.title || e.rowData.title,
                backgroundColor:"#FFFFFF",
            })


            var label = Titanium.UI.createLabel({
                text:"A newly opened window from the " + (e.source.title || e.rowData.title) + " row",
                width:"auto",
                height:"auto",
                textAlign:"center"
            });

            w.add(label)
        }

        tab.open(w,{animated:true});
    }else{
        alert("No window to open :(")
    }
});

    Click ->> the debugger is your friend with issues like this
