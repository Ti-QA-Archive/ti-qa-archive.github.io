Transparent background Android

Hello,

When i make a view with a border, In IOS it shows like i want to, It shows a view with a black border and the background is transparent. But in Android off course, It shows a view with a black backgroung and the border is also black, so it just show a complete black background. I want this background to be transparant, but the border and all the content on this view must be normal. How can i do this? and why is the view in android black and in IOS transparent.

this is my code for the view:

var sessionsView = Ti.UI.createScrollView({
    contentHeight: "auto",
    showVerticalScrollIndicator: true,
    layout: "vertical"
});

//This is the logo for EyeQuestion.
var logo = Ti.UI.createImageView({
    image: '../../images/logo_xl.png',
    top: 25,
    height: 80,
});

function setupViewForTitle(title, index){
    var view = Ti.UI.createView({
        width: getDeviceWidth()-20,
        height: Ti.UI.SIZE,
        bottom: 10,
        borderRadius: 5,
        borderWidth: 3,
        borderColor: "black",
        top: 10,
        backgroundColor: null
    });
    var titleLabel = Ti.UI.createLabel({
        text: title,
        top: 10,
        height: Ti.UI.SIZE,
        width: Ti.UI.SIZE,
        textAlign: "center",
        color: "white",
    });

    view.add(titleLabel);
    view.add(getUserNameView(index));
    view.add(getClosedSessionsView(index));

    sessionsView.add(view);
}
