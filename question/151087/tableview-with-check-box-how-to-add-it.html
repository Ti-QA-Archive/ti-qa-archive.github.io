tableview with check box, how to add it??

function abc() {
var win = Titanium.UI.createWindow();

// create table view data object
var data = [];

data[0] = Ti.UI.createTableViewRow({hasChild:false,height:'auto'});
data[1] = Ti.UI.createTableViewRow({hasDetail:false,height:'auto'});

function addRow(idx,text,)
{
data[idx].add(Ti.UI.createLabel({
text:text,
height:'auto',
width:'auto',
left:10,
right:50,
top:10,
bottom:10
}));
}

addRow(0,'appcelerator ');
addRow(1,'titanium');

// create table view
var tableview = Titanium.UI.createTableView({
    data:data,
    minRowHeight:20,
    top:'70dp',
    rowBackgroundColor:'white',
    });

win.add(tableview);

var toggleSwitch = function (e) {
Ti.App.Properties.setBool(e.source.id, !Ti.App.Properties.getBool(e.source.id, false));
e.source.setImage(Ti.App.Properties.getBool(e.source.id, false) ? '/pics/box.jpg' : '/pics/uncheck.jpg');
};
var check11 = Ti.UI.createImageView({
id: 'check11',
image : Ti.App.Properties.getBool('check11', false) ? '/pics/box.jpg' : '/pics/uncheck.jpg',
width : '20dp',
height : '20dp',
top : '79dp',
left : '280dp'
});
check11.addEventListener('click', toggleSwitch);
tableview.add(check11);

module.exports = abc;

  • Hi devyani p,

    You can see this code .I think it becomes helpful for you how to add check box in a tableview .

    var win = Ti.UI.createWindow({
    title : 'table View',
    navBarHidden :'false'
});
// create table view data object
var data = [];

data[0] = Ti.UI.createTableViewRow({
    hasChild : true,
    title : 'Row 1'
});
data[1] = Ti.UI.createTableViewRow({
    hasDetail : true,
    title : 'Row 2'
});
data[2] = Ti.UI.createTableViewRow({
    hasCheck : true,
    title : 'Row 3'
});
data[3] = Ti.UI.createTableViewRow({
    title : 'Row 4'
});

// create table view
var tableview = Titanium.UI.createTableView({
    data : data
});

win.add(tableview);
win.open();
