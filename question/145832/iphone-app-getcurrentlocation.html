Im building a app for both Android and iPhone, and need to get the current location of the device.

Here is my code:

 Titanium.Geolocation.purpose = 'Get Current Location';
        Ti.Geolocation.accuracy = Ti.Geolocation.ACCURACY_BEST;
        Titanium.Geolocation.distanceFilter = 10;
        Ti.Geolocation.preferredProvider = "gps";

function getcurrentlocation(){
    //lets show an activity indicator that we are looking up the location
    //alert("GETTING CURRENT LOCATION")

    if (Titanium.Geolocation.locationServicesEnabled === false)
    {

        actIndCurrentloc.hide();//hide the activity indicator
        Titanium.UI.createAlertDialog({title:'Header', message:'Your device has geo turned off - turn it on.'}).show();
    }
    else
    {    
        Ti.API.info("trying to get position");
//Get the current position and set it to the mapview
        for (i = 0; i < 3; i++)//call it 3 times to make sure we dont have a cached position
        {
            Titanium.Geolocation.getCurrentPosition(function(e){

                if(i == 2)
                {
                    if (e.error)
                    {
                        alert('Error: ' + e.error);
                    }
                    else 
                    {

                           Ti.App.fireEvent("currentlocationready",
                        {
                            latitude:e.coords.latitude,
                            longitude: e.coords.longitude
                        });
                        //alert("CURRENTPOS IS READY")
                        Titanium.Geolocation.removeEventListener('location',  getcurrentlocation);
                        //Titanium.Geolocation.removeEventListener('location',  getcurrentlocation);

                    }
                }
            });
        }
    }
};    




    module.exports = getcurrentlocation;
