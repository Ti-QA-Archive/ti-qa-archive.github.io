  • This code is very old, but maybe it helps you. I've used this webview-workaround to load remote images faster than a image view.

    /**
 * creates a webview from an image-url to replace remote imageview
 * @param {String} url
 * @return {Ti.UI.WebView}
 */
function createRemoteImageWebView(url){
    var photo = Ti.UI.createWebView({
        html:'<body topmargin=0 leftmargin=0><img src="' + url + '" width=50 height=50></body>',
        width:50,
        height:50,
        touchEnabled:false
    });

    var didLoad=false;
    var fnLoad=function() { 
        didLoad=true;
    };
    var fnTimer=function() {
        if (!didLoad) {
            photo.image=url;    //retry!
        }
    };
    // if loaded then everything is ok..
    photo.addEventListener('load', function(e) {
        fnLoad();
    });
    setTimeout(function() {
            fnTimer();
    },300);    // seconds load time

    return photo;
}

    In your tableview row you can now easily add an image like

    row.add( createRemoteImageWebView( thumbUrl ) );

    if (file.exists()) {
        // If it has been cached, assign the local asset path to the image view object.
        imageViewObject.image = file.nativePath;
    } else {
        // If it hasn't been cached, grab the directory it will be stored in.
        var g = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, imageDirectoryName);

            // If the directory doesn't exist, make it
            g.createDirectory();
