    • try this simple Demo:

      ###Alloy.js

      var idleTime = null;
var interVal = null;

Ti.App.addEventListener('idleTime', function(e) {
    idleTime = 0;
    clearInterval(interVal);
    interVal = setInterval(function() {
        idleTime++;
        Ti.API.info('>> App Idel Time>>>>' + idleTime);
    }, 1000);
});

      ###index.xml

      <Alloy>
    <Window id="win" class="container">
        <Label id="label" onClick="doClick">Hello, World</Label>
    </Window>
</Alloy>

      ###index.js

      function doClick(e) {
    alert($.label.text);
}
$.win.open();

$.win.addEventListener('click',function(e){
    Ti.App.fireEvent("idleTime"); 
});

    You should check for any interaction and reset a timer every time an action is performed. Not an usable solution at all.

    You can otherwise use the pause and resume event, those will do for sure.
