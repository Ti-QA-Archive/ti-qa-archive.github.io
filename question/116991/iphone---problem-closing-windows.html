    I would recommend using a navigation group.

    
Titanium.UI.setBackgroundColor('#000');

var baseWin = Titanium.UI.createWindow({  
    title:'BASE',
    backgroundColor:'#fff'
});

var win1 = Titanium.UI.createWindow({  
    title:'Main Window',
    backgroundColor:'#fff'
});

function showNextWindow(data){
    var detailWindow = Ti.UI.createWindow( {
        title : "detail window one",
        backgroundColor:'#ff1'
    });

    var modalBtn = Titanium.UI.createButton({  
        title:'SHOW MODAL',
        height: 25,
        width : 150
    });
    detailWindow.add(modalBtn);

    modalBtn.addEventListener('click', showModal);

    navGroup.open(detailWindow);
};

function showModal(data){
    var modalWindow = Ti.UI.createWindow( {
        title : "modal image window",
        backgroundColor:'#f2f',        
        modal:true
    });

    var closeBtn = Titanium.UI.createButton({  
        title:'CLOSE',
        height: 25,
        width : 60
    });
    modalWindow.add(closeBtn);
    modalWindow.open();

    closeBtn.addEventListener('click', function(){
       modalWindow.close();
    });
};


var nextBtn = Titanium.UI.createButton({  
    title:'NEXT',
    height: 25,
    width : 60
});
win1.add(nextBtn);
nextBtn.addEventListener('click', showNextWindow);

var navGroup = Ti.UI.iPhone.createNavigationGroup( {
    window : win1
});

win1.navGroup = navGroup;
baseWin.add(navGroup);
baseWin.open();
