var win = Titanium.UI.createWindow({ 
  title:'Marquee',

var label = Titanium.UI.createLabel({
  left: 320,

  height:30,
  color:'black',
  text:'They are very social animals and live with other llamas as a herd.'

var animation = Titanium.UI.createAnimation({
  left:-250,

    Hello Devyani,

    Try this :

    var label = Titanium.UI.createLabel({
    left : 320,
    width : Ti.UI.SIZE,
    height : 30,
    color : 'black',
    text : 'They are very social animals and live with other llamas as a herd.'
});

var width = label.toImage().width;

Ti.API.info('width>>>>' + width);

label.width = width;

    i hope this helps. thanks.
