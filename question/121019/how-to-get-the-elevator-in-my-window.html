    Try (as Steve suggested) a scrollView :

    

var win = Ti.UI.currentWindow;

var container = Ti.UI.createScrollView({
    top:0,
    contentHeight:'auto',
    layout:'vertical',
    width:320,
    bottom:0,
    showVerticalScrollIndicator:true,
    showHorizontalScrollIndicator:false
});

var some_text = Ti.UI.createLabel({
    text:"Here is lots of text. Here is lots of text. Here is lots of text. Here is lots of text. Here is lots of text. Here is lots of text. Here is lots of text. Here is lots of text. \n\nHere is more text. Here is more text. Here is more text. Here is more text. Here is more text. Here is more text. Here is more text. Here is more text. Here is more text. Here is more text. Here is more text. \n\nIsn't this fun? Isn't this fun? Isn't this fun? Isn't this fun? Isn't this fun? Isn't this fun? Isn't this fun? Isn't this fun? Steve is a magician. Or Yoda. ",
    left:5,
    height:'auto',
    right:5,
    font:{fontSize:12,fontWeight:'bold'},
    color:'black',
    textAlign:'left'
});

container.add(some_text);

win.add(container);
