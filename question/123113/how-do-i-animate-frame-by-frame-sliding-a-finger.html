How do I Animate frame by frame sliding a finger

i want to have a series of images and be able to show them one by one while I slide my finger from the top of the screen to the bottom. Is that possible ?
Regards.

    Sure this is possible. One possible way to handle this would be to take the height of the display, divide that into the number of images you have, then listen for touchmove events on the main window gathering the coordinates from the event and displaying the image/frame accordingly

  • Thanks Anthony, following your recommendation here's a solution if anybody is looking for something similar.

    var currentPage=0;
//======= ANIMATION ==============
var pages = [];
    pages[0] = '../images/wrapper/wrapper-1.jpg';
    pages[1] = '../images/wrapper/wrapper-2.jpg';
    pages[2] = '../images/wrapper/wrapper-3.jpg';
    pages[3] = '../images/wrapper/wrapper-4.jpg';
    pages[4] = '../images/wrapper/wrapper-5.jpg';
    pages[5] = '../images/wrapper/wrapper-6.jpg';
    pages[6] = '../images/wrapper/wrapper-7.jpg';
    pages[7] = '../images/wrapper/wrapper-8.jpg';
    pages[8] = '../images/wrapper/wrapper-9.jpg';
    pages[9] = '../images/wrapper/wrapper-10.jpg';

var view= Ti.UI.createView({
    width:236,
    height:300,
    top:0,
    backgroundImage:'../images/wrapper/wrapper-1.jpg'
});
win.add(view);


win.addEventListener('touchstart', function(e) {
  //Ti.API.info("touhcstart: "+e.y);
});

var previousY=0;
var currentPageAnterior=0;
win.addEventListener('touchmove', function(e) {

   if(e.y>previousY){
       var moved = (e.y*100)/win.height; //calculate de % on the finger is on the window    
       currentPage= Math.floor(moved/10); //i have 10 images so i divided by 10
       Ti.API.info("currentPage: "+currentPage)
           if(currentPage>currentPageAnterior){
               currentPageAnterior=currentPage;
               if(currentPage>9)
               currentPage=9;
               view.backgroundImage = pages[currentPage];
           } //this if prevents the image to star again if the user lift the finger and start from the top again
   }
   previousY=e.y;  
});

view.addEventListener('touchend', function(e) {
   //Ti.API.info("touchEnd: "+e.y);
});
