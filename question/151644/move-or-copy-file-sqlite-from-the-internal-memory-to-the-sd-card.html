    Hello jamal,

    Use this kind of way : 

    function moveFile(sourceFile) {
    try {
        var sourceFile = Titanium.Filesystem.getFile(sourceFile);

        var tempFile = Titanium.Filesystem.getFile(Ti.Filesystem.externalStorageDirectory, 'temp.js');
        if(!tempFile.exists()){
            tempFile.write(sourceFile.read())
        }
    } catch(e) {
        Ti.API.error("ERROR>>>>>> " + JSON.stringify(e));
    }
}

    Let me know if it not work

    • yup just pass the path of the file which you are going to move and put the below condition for checking that the file is moved or not.

      function moveFile(sourceFile) {
    try {
        var sourceFile = Titanium.Filesystem.getFile(sourceFile);

        var tempFile = Titanium.Filesystem.getFile(Ti.Filesystem.externalStorageDirectory, 'temp.js');
        if(!tempFile.exists()){
            tempFile.write(sourceFile.read())
        }
        //using native path, you can get the value of the file path
        Ti.API.info('File Path : ' + tempFile.nativePath);
    } catch(e) {
        Ti.API.error("ERROR>>>>>> " + JSON.stringify(e));
    }
}
