  • Hello Simon,

    See this code for splash screen…

    
var img = Ti.UI.createImageView({
    image : 'Default.png',
    top : 0,
    left : 0,
    width : 320,
    height : 480
});

var splash = Ti.UI.createWindow();
splash.add(img);

splash.open();

var tabGroup = Titanium.UI.createTabGroup();

var win1 = Titanium.UI.createWindow({
    tabBarHidden : true,
    url : 'home.js'
});

var tab1 = Titanium.UI.createTab({
    window : win1
});
tabGroup.addTab(tab1);

var fadeOut = Ti.UI.createAnimation({
    opacity : 0.2,
    duration : 300
});
var fadeIn = Ti.UI.createAnimation({
    opacity : 1,
    duration : 1800
});

setTimeout(function(e) {
    splash.close(fadeOut);
    tabGroup.open(fadeIn);
}, 3000);

    Goodluck….
