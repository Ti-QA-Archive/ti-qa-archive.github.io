  • You could use the focus event on the windows.

    Eg:

    
var someWindow = Ti.UI.createWindow();

someWindow.addEventListener("focus", function() {
    // This will be fired everytime you click the corresponding tab.
});

var tab = Titanium.UI.createTab({
    window:someWindow,
    title:'Some window',
    icon:'someicon.png'
});
tabGroup.addTab(tab);
