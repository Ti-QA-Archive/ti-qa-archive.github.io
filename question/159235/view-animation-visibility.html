  • Try this demo :

    index.xml

    <Alloy>
    <Window backgroundColor="white" >
        <Button id="crossFade" onClick="crossFadeClick">Cross Fade</Button>
        <View id="v1" height='50' width ="50" backgroundColor="green" bottom="50" />
        <View id="v2" height='50' width ="50" backgroundColor="blue"bottom="50"/>
        <View id="v3" height='50' width ="50" backgroundColor="red" bottom="50"/>
        <View id="v4" height='50' width ="50" backgroundColor="#630150" bottom="50" />
    </Window>
</Alloy>

    index.js

    var animation = require('alloy/animation');

var arr = [$.v4, $.v3, $.v2, $.v1];
var counter = 0;

function crossFadeClick(e) {
    if(counter+2 > arr.length)
    {
        counter = 0;
    }
    animation.crossFade(arr[counter], arr[counter+1], 2000);
    counter++;

}
$.index.open();
