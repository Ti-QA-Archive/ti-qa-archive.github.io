Placing an image, label and long text inside an scrollview isnt working

I would like the following layout:

[image]
[label]
[long text]

Since there is a long text everything should be scrollable as one. So I placed all items in a scrollview. Inside that scrollview I placed a normal view to contain all elements. However I don't succeed in placing the element at their correct position. Also scrolling isn't working. I also tried using an textarea for the long text instead of a label but then only the textarea is scrolling.

Inside a modal window I have

var wDetail = Titanium.UI.createWindow({
    title:oColumn.t,
    backgroundColor:config.backgroundColor
});
var vScrollView = Ti.UI.createScrollView({
    contentHeight:'auto',
    showVerticalScrollIndicator:'true'
});
wDetail.add(vScrollView);
var oContainer = Ti.UI.createView({
    backgroundColor:'red',
    top:0
});
vScrollView.add(oContainer);

 var oImageView = Ti.UI.createImageView({
    image:translateImage(json.i),
    width:iDeviceWidth,
    height:iDeviceWidth/2,
    top:0,
    preventDefaultImage:true, // No image placeholder
});
oContainer.add(oImageView);

// Label
var lTitle = Ti.UI.createLabel({
    text:'test',
    left:5,
    right:5,
    font:{fontFamily:"Helvetica Neue", fontWeight:'bold', fontSize:18},
    color:'#5DBDC5'
});
oContainer.add(lTitle);

// Content
oContainer.add(Ti.UI.createLabel({
    text:json.c,
    top:100
}));

tMainGroup.activeTab.open(wDetail);

  • Hi Dreosj,

    Try to set Scrollview height and view's layout property as vertical.

    var vScrollView = Ti.UI.createScrollView({
    contentHeight:'auto',
   height : 420,
    showVerticalScrollIndicator:'true'
});
wDetail.add(vScrollView);
var oContainer = Ti.UI.createView({
    backgroundColor:'red',
    layout : 'vertical',
    top:0
});
vScrollView.add(oContainer);
