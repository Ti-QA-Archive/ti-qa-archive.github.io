How to activate cameara flash light?!

Hi,

I am using SDK 3.1.1 and I want to activate the flash light with a button. I tried using 

Ti.Media.setCameraFlashMode(1);

and

Ti.Media.setCameraFlashMode(Ti.Media.CAMERA_FLASH_ON);

but this doesn´t work.

Can anybody help me?

Thanks in advance.

P.S. Here is a code snippet I used:

var win = Ti.UI.createWindow({
        top: 0,
        left: 0,
        right: 0,
        bottom: 0,
        backgroundColor: "#fff"
    });

    var button = Ti.UI.createButton({
        left: 50,
        top: 50,
        right: 50,
        title: "Flash"
    });
    var foo = false;
    button.addEventListener("click", function(){
        if(foo){
            foo = false;
            Ti.Media.setCameraFlashMode(0);
            //Ti.Media.setCameraFlashMode(Ti.Media.CAMERA_FLASH_OFF);
        }else{
            foo = true;
            Ti.Media.setCameraFlashMode(1);
            //Ti.Media.setCameraFlashMode(Ti.Media.CAMERA_FLASH_ON);
        }
    });

    win.add(button);

    win.open();
