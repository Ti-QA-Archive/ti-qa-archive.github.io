  • Accepted Answer

    For android you can do following things:

    You can simply close the application by adding following property to the first window of your application.

         exitOnClose: true

    and then type the following code where you close your window or in your alertdialog's listener

         win.close();
     var activity = Titanium.Android.currentActivity;
     activity.finish();

    This will exit you from application.

  • var exitBtn = Titanium.UI.createButton({
    backgroundImage:"images/exit.png",
    height:"100",
    width:"100"
    });
win.add(exitBtn );

    exitBtn.addEventListener("click", function(e){
    //alert('clicked');
    var dialog_success = Titanium.UI.createOptionDialog({
    options: ['exit', 'cancel'],
    title: 'click exit to quit'
});
var msgbox = Titanium.UI.createAlertDialog({ 
        title: 'Exit', 
        message: 'Are you sure?', 
        buttonNames: ['Yes', 'No'], cancel: 1 
    });    
    msgbox.addEventListener('click', function(e) { 

        //Clicked cancel, first check is for iphone, second for android
        if (e.cancel === e.index || e.cancel === true) {
            return;
        }

        switch (e.index) {
          case 0: 
            //Clicked YES
            var activity = Titanium.Android.currentActivity; activity.finish();

            break;

          case 1: 
            //Clicked YES
            break;

          default:
            break;

        }
    });//end event lisener

    msgbox.show();
});
