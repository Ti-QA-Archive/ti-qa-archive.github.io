Ti.Accelerometer.addEventListener('update',function(){
var cur_top = e.y;
var cur_left = e.x;

cur_top = cur_top.toFixed(2)*100;
cur_left = cur_left.toFixed(2)*100;    

cur_top = (Ti.Platform.displayCaps.platformHeight)*cur_top/100;
cur_left = (Ti.Platform.displayCaps.platformWidth)*cur_left/100;

if(cur_left<=win_width*5/100){

        cur_left = (win_width*90/100)-30;

    }
    if(cur_top >= (win_height*95/100)-15){

button_view.animate({center : {x:cur_left, y : cur_top, duration : 250}});
