    I finally figured out the magic secret sauce that is missing….

    you do not specify the data parameter as a Blob, just give it a valid path to the video to upload

    Titanium.Facebook.execute('facebook.video.upload',{}, function(r)
{
    if (r.success)
    {
        Ti.UI.createAlertDialog({
        title:'Facebook', 
        message:'Your video was uploaded'}).show();
    }
    else
    {
        Ti.UI.createAlertDialog({
        title:'Facebook', 
        message:'Error ' + r.error}).show();

    }
    b1.title = 'Upload Video';


}, Ti.Filesystem.resourcesDirectory + '/movie.mp4');
