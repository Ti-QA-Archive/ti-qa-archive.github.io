  • Hi Anton

    I think your root window is a light-weight window causing this happen. Instead of using light-weight window, use heavy window.~~~

    • A window is heavyweight if it has any of these properties: fullscreen, navBarHidden, modal, windowSoftInputMode, or if the tabOpen property is true
      Otherwise, the window is lightweight

    • listWindow().open();
function listWindow(){
    var win = Ti.UI.createWindow({
        backgroundColor: '#000',
        navBarHidden: true
    });
    var headerView = Ti.UI.createView({
        backgroundImage: 'images/header.jpg',
        width: '100%',
        height: '50dip',
        top: 0
    });

    var mapButton = Ti.UI.createButton({
        right: 0,
        height: '100%',
        backgroundSelectedImage: 'images/btnselected.jpg',
        backgroundImage: 'images/headerbg.jpg',
        width: '112dip',
        title: 'Visa på karta',
        font: {fontSize:'16dip'}
    });
    mapButton.addEventListener('click',function(e){
        mapWindow().open({modal:true});
    });
    var separator = Ti.UI.createImageView({
        url:'images/separator.jpg',
        right: '112dip',
        width: '3px'
    });

    headerView.add(mapButton);
    headerView.add(separator);
    win.add(headerView);
    return win;
}
