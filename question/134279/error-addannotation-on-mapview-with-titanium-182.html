Error addAnnotation on MapView with Titanium 1.8.2

I has a mapView and I want to add some annotations depend on their status, if status = 1 annotation has image '1.png', status = 2 => annotation.image = '2.png', but mapview show image wrong

mapview = Titanium.Map.createView({
    mapType: filter.googleViewType,
    region: { latitude: 55, longitude: 3.5, latitudeDelta: 0.05, longitudeDelta: 0.05 },
    animate: true,
    regionFit: true,
    userLocation: true
});

var MapCL = {
    addMarkers: function (aData) {
        var _count = aData.length;
        mapview.removeAllAnnotations(); 
        for (var i = 0; i < _count; i++) {           
            var _marker = Titanium.Map.createAnnotation({
                latitude: aData[i].Lat,
                longitude: aData[i].Lng,
                title: aData[i].Ident,
                image: (aData[i].StatusID ==1) ? 'images/1.png' : 'images/2.png',
                pincolor: Titanium.Map.ANNOTATION_PURPLE,
                index: i
            });

            mapview.addAnnotation(_marker);
        }       
    }

But this code work OK on Titanium 1.7.0. Please help me !!! Thanks so much
