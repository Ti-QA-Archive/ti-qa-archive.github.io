Passing arrays of data into webviews

I'm trying to send data to a webview in order to use it with RGraph. I can send a single variable but I've been struggling to send larger arrays.

app.js 

//This works..
Ti.App.fireEvent("TestFiring", {
    data:'Caspar'
});

//This doesn't
var mydata =[{"SmallWine":0,"DoseageStart":1.301337982E9,"SessionID":3,"HalfPints":2,"DoseageChanged":1.301337992E9,"SingleSpirits":0,"ExitCode":"","SessionStart":0}, {"SmallWine":0,"DoseageStart":1.301337982E9,"SessionID":3,"HalfPints":2,"DoseageChanged":1.301338001E9,"SingleSpirits":1,"ExitCode":"","SessionStart":0}]  
Ti.App.fireEvent("TestFiring", {
    data: escape(mydata)
});

chart.html 

<script>
    Ti.App.addEventListener('TestFiring',function(e) {
        Ti.API.info("e inside webview" + JSON.stringify(e));
    });
</script>

The former prints tells me e contains {&quot;data&quot;:&quot;Caspar&quot;,&quot;type&quot;:&quot;TestFiring&quot;}, the latter {&quot;data&quot;:&quot;%5Bobject%20Object%5D%2C%5Bobject%20Object%5D%2C%5Bobject%20Object%5D%2C%5Bobject%20Object%5D%2C%5Bobject%20Object%5D&quot;,&quot;type&quot;:&quot;TestFiring&quot;}

Moreover, if I send the event to a listener in app.js, it works correctly in both cases. Can anyone tell me what I am doing wrong?

    have you tried to evaluate the json string in the webView

    webView.evalJs("someFunction(" + myData + ")");

    this would send the string to the html file and then you can process it in the javascript over there, no firing of events needed
