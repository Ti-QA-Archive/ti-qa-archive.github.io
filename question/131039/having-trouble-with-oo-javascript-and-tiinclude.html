Having trouble with O.O javascript and Ti.include()

Hello everyone.

I got a bit of a problem, I wrote the following code:


var client = {};
client.ui = {}; 

client.ui.mainView = function(){

};

client.ui.mainView.prototype = {
    getMainView : function(){
        var mainWindow = Ti.UI.createWindow({
            backgroundColor: 'gray',
            layout: 'vertical',
            exitOnClose: false
        });

        mainWindow.add(Ti.UI.createTextField());
        return mainWindow;
    }
}

var mainViewFactory = new client.ui.mainView();

var mainView = mainViewFactory.getMainView();

mainView.open();

When I wrote the code that way everything worked fine.
I have then tried to seperate the mainView logic from the app.js file and the code looked like this:

(I've created views directory in the root of the project and placed my file there)

var client = {};
client.ui = {}; 

Ti.include("views/mainView.js");

var mainViewFactory = new client.ui.mainView();

var mainView = mainViewFactory.getMainView();

mainView.open();

When I tried to run the code after the change I received the following error:

Uncaught TypeError: undefined is not a function

I have no idea why this happens, it looks as the Ti.include() does not work.

Does anyone has a clue?

Thanks in advance,
Danny

    Hi,

    Your code looks correct. I did tested with below code, and it works well

    //app.js
var client = {};
client.ui = {}; 

Ti.include("views/mainView.js");

var mainViewFactory = new client.ui.mainView();

var mainView = mainViewFactory.getMainView();

mainView.open();


//mainView.js
client.ui.mainView = function(){

};

client.ui.mainView.prototype = {
    getMainView : function(){
        var mainWindow = Ti.UI.createWindow({
            backgroundColor: 'green',
            layout: 'vertical',
            exitOnClose: false
        });

        mainWindow.add(Ti.UI.createTextField());
        return mainWindow;
    }
};

    You should clean up project (Project > Clean..) before try again.
