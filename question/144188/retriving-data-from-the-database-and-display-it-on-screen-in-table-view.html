      var vwin = Ti.UI.currentWindow;

var newsID;
var i = 0;
var dbdata = [];

var tableView = Titanium.UI.createTableView({
    // data : rowdata
});
vwin.add(tableView);

viewdata();

function viewdata() {

    var db = Ti.Database.open('details');
    var rows = "Select * from kisan order by Regid DESC";
    var rs = db.execute(rows);

    dbdata = [];

    while (rs.isValidRow()) {

        var this_tfrid = rs.fieldByName('Regid');
        var this_tfname = rs.fieldByName('Name');
        var this_tfphno = rs.fieldByName('Phno');

        // alert(this_tfrid +'  '+ this_tfname + ' '+ this_tfphno);

        var row = Titanium.UI.createTableViewRow({
            height : 40
        });

        var lblId = Ti.UI.createLabel({
            top : 0,
            left : 5,
            height : 30,
            width : 50,
            text : this_tfrid
        });

        var lblName = Ti.UI.createLabel({
            top : 5,
            left : lblId.left + lblId.width,
            height : 30,
            width : 100,
            text : this_tfname
        });

        var lblPhone = Ti.UI.createLabel({
            top : 5,
            left : lblName.left + lblName.width,
            height : 30,
            width : 100,
            text : this_tfphno
        });

        row.add(lblId);
        row.add(lblName);
        row.add(lblPhone);

        dbdata.push(row);
        rs.next();

    }

    tableView.setData(dbdata);
}

      I have just made simple demo that how to fetch the record and display in tableview.

      you can set more labels in row as per you want…
