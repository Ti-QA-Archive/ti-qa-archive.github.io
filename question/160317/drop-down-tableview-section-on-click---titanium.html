  • Hello Gerardo Siano !!!. I hope, This code will be a little bit helpful for you

    Ti.UI.backgroundColor = 'white';
var win = Ti.UI.createWindow();

    var campania = [{
            title : 'Battipaglia'
        }, {
            title : 'Napoli'
        }, {
            title : 'Salerno'
    }];

    var emilia = [{
            title : 'Budrio'
    }];

    var lazio = [{
            title : 'Fondi'
        }, {
            title : 'Sabaudia'
        }, {
            title : 'Terracina'
    }];

    var lombardia = [{
            title : 'Brescia'
    }];

    var puglia = [{
            title : 'Lecce'
        }, {
            title : 'Nardo'
        }, {
            title : 'Taranto'
    }];

    var regioni = ['Campania', 'Emilia', 'Lazio', 'Lombardia', 'Puglia'];
    var citta = [];
    var tbl;
    var rows = [];
    var labels = [];
    var flag=[0,0,0,0,0];
    for(var index = 0; index < 6; index++) {
        lbl = Ti.UI.createLabel({
            text : regioni[index],
        });
        var section = Ti.UI.createTableViewSection({
            headerView: lbl,
            id:index
        });
        lbl.addEventListener('click', function(e) {

              if(e.source.text==regioni[0])
              {
                  Items(campania,rows[0],0);
              }
              if(e.source.text==regioni[1])
              {
                  Items(emilia,rows[1],1);
              }
              if(e.source.text==regioni[2])
              {
                  Items(lazio,rows[2],2);
              }
              if(e.source.text==regioni[3])
              {
                  Items(lombardia,rows[3],3);
              }
              if(e.source.text==regioni[4])
              {
                  Items(puglia,rows[4],4);
              }
        });
        rows.push(section);
        labels.push(lbl);
    }

    tbl = Ti.UI.createTableView({
        data: rows,
    });

    win.add(tbl);
    win.open();

    function Items(citta,section,ind)
    {
        if(flag[ind]==0)
        {
            for(var j = 0; j < citta.length; j++)
              {
                  section.add(Ti.UI.createTableViewRow({
                        title: citta[j].title,
                    }));        
              }
              tbl.setData([]);
              tbl.setData(rows);
              flag[ind]=1;
        }
        else
        {
            flag[ind]=0;
            for(var a = 0; a< section.rowCount; a++)
              {
                  tbl.deleteRow(tbl.data[section.id].rows[a]);    
              }
              tbl.deleteRow(tbl.data[section.id].rows[0]);
        }

    }
