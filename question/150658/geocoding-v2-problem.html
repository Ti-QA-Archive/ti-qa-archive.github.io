    var url= "http://maps.googleapis.com/maps/api/geocode/json?address=1600+Amphitheatre+Parkway,+Mountain+View,+CA&sensor=true";

geoCode(url);

function geoCode(url){
var xhr = Titanium.Network.createHTTPClient();
  xhr.open('POST',url);
  xhr.onerror = function(e) {
  Ti.API.info('IN ERROR ' + e.error);
  }; 
 xhr.onload = function() {
     var i;
    Ti.API.info('this.responseText--->'+this.responseText);
    var json= JSON.parse(this.responseText);
    if(json.status === "OK"){
    alert("lat and long==>"+json.results[0].geometry.location.lat+",+                                    json.results[0].geometry.location.lng);

        Ti.API.info("latitude===>"+ json.results[0].geometry.location.lat);
        Ti.API.info("longitude==>"+ json.results[0].geometry.location.lng);
        }
   }
   xhr.send();
}
