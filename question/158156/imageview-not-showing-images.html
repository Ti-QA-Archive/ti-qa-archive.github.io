ImageView not showing images

Hi guys, I'm new to Titanium.

I created a module that creates rows:

exports.createNewRow = function(rownum, primarylabel, secondarylabel, myImage){
    var helpers = require('helpers');

    var row = Ti.UI.createTableViewRow({
        rowId: rownum,
        primaryLabel: primarylabel,
        secondaryLabel: secondarylabel,
        height:50
    });

    var mainview = Ti.UI.createView({
        layout: 'horizontal',
        height: 'auto'
    });

    var image = Ti.UI.createImageView({
        url: helpers.getImageUrl(myImage),
        height: 50,
        width: 50,
        top: 10,
        left: 10,
        bottom: 10
    });
    mainview.add(image);

    row.add(mainview);

    return row;
};

var imageurls = [
    { url: '/images/imageA.png'},
    { url: '/images/imageB.png'},
    { url: '/images/imageC.png'}
];

exports.getImageUrl = function(urlindex){
    if (urlindex > imageurls.length) {
        return imageurls[imageurls.length-1].url;
    }
    else
    {
        return imageurls[urlindex].url;
    }

};

exports.ImageEnum = {
    IMG1 : 0,
    IMG2 : 1,
    IMG3 : 2
}
