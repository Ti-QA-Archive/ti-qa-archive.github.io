Android: Change icon tab, if tab is pressed(active)

var tab1 = Titanium.UI.createTab({
window : win1,
icon : 'icon1.png',
});
var tab2 = Titanium.UI.createTab({
window : win2,
icon : 'icon2.png',
});

In android:

if tab 1 is active, the icon(tab1) change , and icon( tab2) again the same

if tab 2 is active, the icon(tab2) change , and icon (tab1) again the same

    Hi Alex,

    Here is an example that you can take a look for icon changes. Icon interchange by focusing on tab.

    var tabGroup = Ti.UI.createTabGroup();

var win1 = Ti.UI.createWindow({
    title : "Window 1",
    backgroundColor : 'white' 
});

var tab1 = Ti.UI.createTab({
    title : "Tab1",
    icon : "KS_nav_ui.png",
    window : win1
});

var win2 = Ti.UI.createWindow({
    title : "Window 2",
    backgroundColor : 'gray' 
});

var tab2 = Ti.UI.createTab({
    title : "Tab2",
    icon : "KS_nav_views.png",
    window : win2
});

tabGroup.addTab(tab1);
tabGroup.addTab(tab2);

tabGroup.addEventListener('focus', function(e){
    if(e.tab == tab1){
        tab1.setIcon("KS_nav_views.png");
        tab2.setIcon("KS_nav_ui.png");
    } else {
        tab2.setIcon("KS_nav_views.png");
        tab1.setIcon("KS_nav_ui.png");
    }
});

tabGroup.open();

    best of luck
