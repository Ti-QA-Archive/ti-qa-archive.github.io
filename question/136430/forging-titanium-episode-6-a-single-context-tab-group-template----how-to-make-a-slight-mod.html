Forging Titanium Episode 6: A Single Context Tab Group Template -- How to make a slight mod?

In the example: Forging Titanium Episode 6: A Single Context Tab Group Template, the provided example adds the same button to each window.

How would you modify this to add an individual button to each window?

Here's the supplied code:

exports.AppWindow = function(args) {
    var instance = Ti.UI.createWindow(args);

    var button = Ti.UI.createButton({
        height:44,
        width:200,
        title:'Open new window on tab',
        top:20
    });
    instance.add(button);

    button.addEventListener('click', function() {
        globals.tabs.currentTab.open(Ti.UI.createWindow({
            title: 'New Window',
            backgroundColor: 'white'
        }));
    });

    return instance;
};

  • just create your button before hand and pass it to that function

    exports.AppWindow = function(args, customButton) {
    var instance = Ti.UI.createWindow(args);

    instance.add(customButton);

    customButton.addEventListener('click', function() {
        globals.tabs.currentTab.open(Ti.UI.createWindow({
            title: 'New Window',
            backgroundColor: 'white'
        }));
    });

    return instance;
};

    you can also pass the button with an event listener or overwrite the default event listener or just add the button after after creating the window
