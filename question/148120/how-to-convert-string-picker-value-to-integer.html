  • Using your data as an example, to exand on the answer above, here's how to use a regEx to parse them

    var data = ['0.12 gr','0.15 gr','0.20 gr','0.23 gr','0.24 gr',
'0.25 gr','0.28 gr','0.30 gr','0.36 gr','0.40 gr','0.43 gr'];

for (var x=0;x<data.length;x++){
    var value = parseFloat(data[x].match(/[0-9\.]+/)[0]);
    alert(value);
}
