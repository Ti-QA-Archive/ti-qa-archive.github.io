  • hi,

    pause and resume are supported for ios only so you can call this events by using window's activity.

    try this

    var OnAppResume = function() {
    Ti.API.info('win > OnAppResume ');
}

var OnAppPause = function() {
    Ti.API.info('win > OnAppPause ');    
}

var OnAppDestroy = function() {
    Ti.API.info('win > OnAppDestroy ');        
}

win.addEventListener('open', function(ed) {

    var activity = win.activity;
    activity.addEventListener('resume', OnAppResume);
    activity.addEventListener('pause', OnAppPause);
    activity.addEventListener('destroy', OnAppDestroy);
});

    // app loses focus
});

    // app regains focus
});
