    image : '/images/icons/light.png', // 14 width on 20 height white bulb icon.
    // set the button size big enough for a finger.
    width : 32,
    height : 32
});
TB_invertView.addEventListener('click', function(e) {
    //set the button to disable until the new content is loaded.
    TB_invertView.enabled = false;

    // run invertScreen() function to invert screen colors, etc.
});

// init toolbar and populate items
topToolbar = Ti.UI.createToolbar({
    items : toolbarButtons,
    top : -50,
    borderTop : false,
    borderBottom : false,
    translucent : true,
    visible : false
});

//toogleToolbar with animation (slide up/down)
function toggleToolbarEvent(e) {
    if(!topToolbar.visible) {
        topToolbar.show();
        topToolbar.animate({
            top : 0
        });
    } else {
        topToolbar.animate({
            top : -50
        }, function() {
            topToolbar.hide();
        });
    }
}

  • Ok the trick is to reset the image property of the buttons so I did something like this:

    function refreshToolBarButtonsImages(){
    var i,length = toolbarButtons.length;
    for (i=0; i<length; ++i){
        toolbarButtons[i].image = toolbarButtons[i].image;
    }
}
