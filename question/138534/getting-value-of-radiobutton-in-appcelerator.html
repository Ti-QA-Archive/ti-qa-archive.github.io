  • Hi Rebin,

    Try this demo and check flag which i have set if you click first time you get true and if you click again then it become false for all button and to know click on search

    
var flag1 = false;
var flag2 = false;
var flag3 = false;

var radioBtnAll = Ti.UI.createButton({
    value : 0,
    backgroundImage : '/images/unselected.PNG',
    height : 20,
    width : 20,
    top : 10,
    left : '10%',
    title : 'O'
});
radioBtnAll.addEventListener('click', function(e) {
    if (flag1 == true) {
        flag1 = false;
    } else {
        flag1 = true;
    }
});
var radioBtn1 = Ti.UI.createButton({
    value : 1,
    backgroundImage : '/images/unselected.PNG',
    height : 20,
    width : 20,
    top : 10,
    left : '30%',
    title : '1'
});
radioBtn1.addEventListener('click', function(e) {
    if (flag2 == true) {
        flag2 = false;
    } else {
        flag2 = true;
    }
});
var radioBtn2 = Ti.UI.createButton({
    value : 2,
    backgroundImage : '/images/unselected.PNG',
    height : 20,
    width : 20,
    top : 10,
    left : '70%',
    title : "2"
});
radioBtn2.addEventListener('click', function(e) {
    if (flag3 == true) {
        flag3 = false;
    } else {
        flag3 = true;
    }
});
var btnSearchMsg = Ti.UI.createButton({
    title : 'search',
    top : 100
});
btnSearchMsg.addEventListener('click', function(e) {
    alert('value f1=' + flag1 + " f2=" + flag2 + " f3=" + flag3);
});
win.add(radioBtn1);
win.add(radioBtn2);
win.add(radioBtnAll);
win.add(btnSearchMsg);
