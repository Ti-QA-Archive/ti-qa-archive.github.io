    hello nicolò monili I am just modify your code.

    var mydata = [];
var slider = new Array();   // Take array of slider control
var btn1 = Ti.UI.createButton({
    title : 'click',
    width : 100,
    height : 50,
    top : 10
});
function creaTableViewRow() {
    //var mydata=[];
    for (var i = 0; i < 24; i++) {

        var row;

         slider[i] = Ti.UI.createSlider({
            width : 240,
            left : 30,
            min : 0,
            max : 40,
            value : 0.0,
            index : i
        });

        var labelOre = Ti.UI.createLabel({
            left : 5,
            text : '' + i,
            index : i,
            width : 'auto',
            color : 'white',
            textAlign : 'center',
            width : 20,
            top : 12
        });

        var labelValori = Ti.UI.createLabel({
            left : 275,
            text : "10°c",
            index : i,
            width : 'auto',
            color : 'white',
            textAlign : 'center',
            width : 40,
            top : 12
        });

        row = Ti.UI.createTableViewRow({
            height : '50dp',
            backgroundColor : 'transparent',
            selectionStyle : Titanium.UI.iPhone.TableViewCellSelectionStyle.NONE

        });

        row.add(slider[i]);   // add slider object in row 
        row.add(labelOre);
        row.add(labelValori);

        row.className = 'myData';
        mydata[i] = row;

    }

    var miaTabella = Ti.UI.createTableView({
        top : 75,
        height : 344,
        data : mydata,
        backgroundColor : 'transparent',
        //scrollable:false,
        //touchEnabled:false
    });

    view2.add(miaTabella);
}

// ON BUTTON CLICK EVENT YOU CAN GET ALL UPDATED VALUES IN ALL 24 SLIDERS

btn1.addEventListener('click', function(e) {

    var valoriSlider = new Array(4.0, 8.0, 6.0, 4.0, 5.0, 10.0, 11.0, 26.0, 15.0, 31.0, 32.0, 34.0, 40.0, 2.0, 5.0, 25.0, 20.0, 15.0, 10.0, 11.0, 4.0, 18.0, 29.0, 35.0);

    updateSlider(valoriSlider);    // pass array of updated slider values.

});

function updateSlider(val)
{
    for( var a =0 ;a<24; a++)
    {
        slider[a].value =val[a];
    }
}
