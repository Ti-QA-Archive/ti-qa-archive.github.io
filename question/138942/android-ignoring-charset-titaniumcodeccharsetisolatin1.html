Android ignoring charset: Titanium.Codec.CHARSET_ISO_LATIN_1

Hello,

I'm requesting a webservice which are returning cp1252 xml.
here a part of the code i use to do the job:

        onload: function ( response, rootNodeName, okFunction ) {
            try {
                var doc;
                if (response.responseText) {
                    doc = response.responseText;
                    Ti.API.info("RESULT responseText: " + doc);
                } else {
                    var myBlob = response.responseData;
                    var blobStream = Ti.Stream.createStream({ source: myBlob, mode: Ti.Stream.MODE_READ });
                    var newBuffer = Ti.createBuffer();
                    var bytes = blobStream.read(newBuffer);
                    doc = Ti.Codec.decodeString({
                          source: newBuffer,
                          charset: Titanium.Codec.CHARSET_ISO_LATIN_1
                    });
                    Ti.API.info("RESULT responseData: " + doc);
                }

                // Titanium hard reading XML as UTF8
                doc = doc.replace("encoding=\"cp1252\"", "encoding=\"utf8\"");
                doc = Titanium.XML.parseString(doc).documentElement;

On iOs the Ti.Codec.decodeString method works perfectly wit the CHARSET_ISO_LATIN_1, but on Android it ignore it and fires the following error : 

[ERROR][TiHttpClient(  415)] (KrollRuntimeThread) [4703,57103] Unable to decode using charset: UTF-8

Is it any way to decode this please?
