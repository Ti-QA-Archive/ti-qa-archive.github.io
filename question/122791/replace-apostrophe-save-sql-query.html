replace apostrophe (save sql query)

I try to replace an apostrophe in an sql query to insert it save into the sqllite db. But it looks like replace All ' with \' doesn't work at all. Any Ideas?

function sqlSave(sql) {
    if (sql != null) {
        var myRegExp = new RegExp("'","g");
        sql = sql.replace(myRegExp, "\'");
    } else {
        sql='';
    }
    return(sql);
}

    Here, you can use this function to escape data in Sqlite:

    function escape (value) {
    if (value && typeof value.replace == "function") {
        return value.replace(/'/g, "''");
    }
    else {
        return value;

    }
};
