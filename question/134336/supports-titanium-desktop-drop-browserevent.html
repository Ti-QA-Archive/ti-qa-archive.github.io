  • Okay, Guys, ther ist my Drop&Upload solution for Win32 in a contable Div

    var target=document.getElementById('editor');
target.addEventListener('DOMNodeInserted', function(event) 
    {
    var text = $('#dropArea').html();
    var files = text.match(/((files?\:\/\/)|(www\.))(\S+)(\w{2,4})(:[0-9]+)?(\/|\/([\w#!:.?+=&%@!\-\/]))?/gi);
    if(files.length != 0)
        {            
        var droppedFile = text.split('</a>');
        var FilePath = droppedFile[droppedFile.length-2];
        var FilePath = FilePath.split('file:///');    
        var fileName = FilePath[2].split('/');

        var Path = FilePath[2].replace('Dokumente%20und%20Einstellungen','Dokumente und Einstellungen');//German OS
        var upload = Titanium.Network.createHTTPClient();
        var file = Titanium.Filesystem.getFile(Path);
        upload.open('POST','[uploadscript.pfp]', true);
        upload.setRequestHeader("contentType", "multipart/form-data ; boundary=\""+fileName[fileName.length-1]+"\""); //to get Filename, to read from Header by php-script
        upload.sendFile(
            {
            file: file.read()
            });            

        }
    return false;
    event.preventDefault();
    });

    Have fun ;) Thanks for reply
