app closes when removing and adding a view to the window

I have a single window, which has Views added and removed from it. For whatever reason, when I go to remove an existing view and add a new one to the window, the app hangs for a second, then closes, without any notification as to why (for android). Any ideas why this may be happening?

AppController.prototype.runController = function(controller)
{
    if(this.m_current != null)
    {
        var last = this.m_current.GetView();
        RootWindow.remove(last);
        last.Release();
    }
    var view = controller.GetView();
    view.PreLoad();
    RootWindow.add(view);
    RootWindow.open();
}
