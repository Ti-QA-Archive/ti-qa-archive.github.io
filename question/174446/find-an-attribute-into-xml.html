  • Tested on iOS

    var xml = '<item><media:thumbnail xmlns:media="http://search.yahoo.com/mrss/" url="http://3.bp.blogspot.com/-TWErtKUFSvk/U3UMFuVnmVI/AAAAAAAAAyY/0_SEQwfJRYE/s72-c/Screen+Shot+2014-05-15+at+19.48.40.png" height="72" width="72" /><thr:total xmlns:thr="http://purl.org/syndication/thread/1.0">0</thr:total><feedburner:origLink xmlns:feedburner="http://rssnamespace.org/feedburner/ext/1.0">http://googleenterprisefrance.blogspot.com/2014/05/de-controleur-federateur-le-nouveau.html</feedburner:origLink></item>';

var doc = Ti.XML.parseString(xml);

var items = doc.getElementsByTagName('item');

for (var i = 0; i < items.length; i++) {
    var item = items.item(i);
    var d = item.getElementsByTagNameNS('http://search.yahoo.com/mrss/', 'thumbnail');
    var a = d.getAttributes.item(0).getAttribute("url");
    Ti.API.info(a);
}
