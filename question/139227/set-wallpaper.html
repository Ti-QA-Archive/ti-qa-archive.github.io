  • var setWall = Ti.UI.createButton({ title: "Set Wallpaper", width: 200, height:'20', });
    setWall.addEventListener("click", function(e) {
    var f = Ti.Filesystem.getFile('images/WP.png');
    var img = f.read();
    Ti.Media.Android.setSystemWallpaper(img, true);
    });
    win.add(setWall);
