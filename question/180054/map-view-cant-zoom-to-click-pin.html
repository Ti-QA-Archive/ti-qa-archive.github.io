  • Here my suggestion:

    Caller:

    parentView.add(require('map.widget')());

    map.widget.js:

    Ti.Map = require('ti.map');
var self = Ti.Map.createView({
    mapType : Ti.Map.STANDARD_TYPE,
    region : {
        latitude : 25.7602295,
        longitude : 44.8881224,
        latitudeDelta : 0.1,
        longitudeDelta : 13
    },
    animate : true,
    regionFit : false,
    userLocation : false,
    enableZoomControls : false,
    annotations : annotations 
    //I hope for you annotations are annotiation objects
});

self.addEventListener('click', function(evt) {
    mapview.setRegion({
        latitude : evt.annotation.latitude,
        longitude : evt.annotation.longitude,
        latitudeDelta : 0.05,
        longitudeDelta : 0.05,
        regionFit : true,
        animate : true,
    });
});

module.exports = self;
