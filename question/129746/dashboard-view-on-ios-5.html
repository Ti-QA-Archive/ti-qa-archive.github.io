Dashboard View on iOS 5.

Hello, i am using Titanium 1.7.5 , and i'm trying to use a Dashboard View. When building on the iOS Sim, it works perfect, but if i compile it for my iPad, iPhone or iPod Touch, it does not seem to work. - the items don't appear. only the nav and tab bars. is this a problem with the Titanium SDK? or my fault? … 

var win = Titanium.UI.currentWindow;
win.backgroundColor = '#C0C0C0';
var data = [];
var labels = ['News','Twitter','Facebook','Profile','Events','Photos','Videos','Music','TV Show','Bella143','Contact', 'Settings'];
for (var x=0;x<1;x++)
{
    for (var c=0;c<labels.length;c++)
    {
        var item = Titanium.UI.createDashboardItem({
            image:labels[c]+'.png',
            label:labels[c],
            canDelete:false
        });
        data.push(item);
    }
}

var dashboard = Titanium.UI.createDashboardView({
    data:data,
    top:20,
});
win.add(dashboard);
var toolbar = Ti.UI.createToolbar({
    barColor:'#660066',
    bottom:0,
    height:20
});
win.add(toolbar)
function nointernet(){
    var nointernet = Titanium.UI.createWindow({
        height:80,
        width:200,
        touchEnabled:false
    });

    var indView = Titanium.UI.createView({
        height:80,
        width:200,
        backgroundColor:'#000',
        borderRadius:10,
        opacity:0.8,
        touchEnabled:false
    });
    nointernet.add(indView);

    var message = Titanium.UI.createLabel({
        text:'No Internet Connection...',
        color:'#fff',
        textAlign:'center',
        font:{fontSize:18, fontWeight:'bold'},
        height:'auto',
        width:'auto',
    });
    nointernet.add(message);
    nointernet.open();

    var animationProperties = {delay: 1500, duration:1800, opacity:0.1};
    if (Ti.Platform.osname == "iPhone OS"){
        animationProperties.transform = Ti.UI.create2DMatrix().translate(-200,200).scale(0);
    }
    nointernet.animate(animationProperties, function(){
        nointernet.close();
    });
}

var cancel = Titanium.UI.createButton({
    systemButton:Titanium.UI.iPhone.SystemButton.DONE
});
cancel.addEventListener('click', function()
{
    dashboard.stopEditing();
    win.rightNavButton = null;
});

dashboard.addEventListener('edit',function()
{
    win.rightNavButton = cancel;
});

dashboard.addEventListener('commit',function()
{
    Ti.API.info('data ' + dashboard.data);
    for (var i=0;i<dashboard.data.length;i++)
    {
        Ti.API.info('label ' + dashboard.data[i].label);
    }
});

dashboard.addEventListener('move', function(e) {
    Ti.API.log('Moved item '+e.item.label);
});

dashboard.addEventListener('dragStart', function(e) {
    Ti.API.log('Dragging item '+e.item.label);
    win.rightNavButton = null;

});

dashboard.addEventListener('dragEnd', function(e) {
    Ti.API.log('Drag ended: ' + e.item.label);
    win.rightNavButton = cancel;
});
