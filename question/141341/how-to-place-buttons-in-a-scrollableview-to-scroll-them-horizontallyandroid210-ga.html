    Hello,

    Try this code

    
var win=Ti.UI.createWindow({
    backgroundColor:'#FFFFFF'
});


// initialize to all modes
win.orientationModes = [
    Titanium.UI.PORTRAIT,
    Titanium.UI.LANDSCAPE_LEFT,
    Titanium.UI.LANDSCAPE_RIGHT
];



var scrollView = Titanium.UI.createScrollView({
    contentHeight: 'auto',
    contentWidth: 'auto',
    height: 50,
    layout: 'horizontal',
    showHorizontalScrollIndicator: true,
    showVerticalScrollIndicator: true,
    width: 100,
});
win.add(scrollView);

var btn1 = Titanium.UI.createButton({
    bottom: 10, // used as margin - change to suit
    height: 40,
    left: 10, // used as margin - change to suit
    right: 10, // used as margin - change to suit
    top: 10, // used as margin - change to suit
    title: 'Button 1',
    width: 50
});
scrollView.add(btn1);

var btn2 = Titanium.UI.createButton({
    bottom: 10,
    height: 40,
    left: 10,
    right: 10,
    top: 10,
    title: 'Button 2',
    width: 50
});
scrollView.add(btn2);

var btn3 = Titanium.UI.createButton({
    bottom: 10,
    height: 40,
    left: 10,
    right: 10,
    top: 10,
    title: 'Button 3',
    width: 50
});
scrollView.add(btn3);
win.open();

  • Hi Raffaele,

    • Create your scrollableView with your views inside.
    • Create your buttons and add to the views
    • Add your scrollableView to your window

    Example : 

    var view1 = Ti.UI.createView({backgroundColor:'white'}); 
var view2 = Ti.UI.createView({backgroundColor:'pink'});
var view3 = Ti.UI.createView({backgroundColor:'red'}); 

var button1 = Ti.UI.createButton({ title: 'button1' });
var button2 = Ti.UI.createButton({ title: 'button2' });
var button3 = Ti.UI.createButton({ title: 'button3' });

view1.add(button1);
view2.add(button2);
view3.add(button3);

var scrollView = Ti.UI.createScrollableView({
     views:[view1,view2,view3], 
    showPagingControl:false
}); 

win.add(scrollView);

    Let me know if you have problem with this.
