Opening modal window from modal window doesn't work like expected?

I'm running into a problem, I have an app and in a normal window I have a table with a number of items. If I click on a row I open a modal row with the details of that row. In that modal window with details I have a small thumb and if people click on that thumb I want to open a new modal window to show a large variant of that image. Passing the data is no problem but when I open the second modal window it just won't popup and no error is triggered. I can however close the first modal window and then I will see my second window but without a back button (in which I would like to open the first modal window again).

What is the correct solution?

// Add the image event
vImage.addEventListener('click', function(e) {
    wCarPhotos.close();
    showFullscreenPhoto(wCarPhotos, e.source.largeImage);
});

/**
 * Show a large image by its url
 */
function showFullscreenPhoto(wCarPhotos, imageUrl)
{
    var wPhoto = Titanium.UI.createWindow({
        title:'Photo large',
        barColor:'#067238',
        backgroundColor:'#222222'
    });
    var bClose = Ti.UI.createButton({
        title:'Back',
        style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN
    });
    wPhoto.setLeftNavButton(bClose);
    bClose.addEventListener('click',function() {
        wPhoto.close();
        wCarPhotos.open();
    });

    wPhoto.add(Ti.UI.createImageView({
        image:decodeURIComponent((imageUrl).replace(/\+/g, '%20')), 
        preventDefaultImage:true
    }));

    wPhoto.open({modal:true});
}
