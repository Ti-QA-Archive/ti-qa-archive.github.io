Vertical/horizontal scrolling a la Yahoo Weather?

Hi all. I'm talking to a development house about building an app I've designed as I'd love to go with Titanium to save on the overall development effort, but one of the features/functionality of the app is to have both horizontal and vertical scrolling on some screens. That is, is "diagonal" scrolling not allowed (vertical and horizontal scrolling at the same time), or is it that no screen can offer both, yet independent, horizontal and vertical scrolling?

  • Hi,

    You can also try this for both horizontal and vertical scrolling at the same time on Android.

    var win = Titanium.UI.createWindow(
    {
        backgroundColor:'white'
    }
);

var viewImage = Ti.UI.createImageView({
    image : '/Unknown.jpg',
    height:'150%',
     width:'150%'
});

var scrollViewH = Titanium.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    top : 0,
    scrollType : 'horizontal',
    showHorizontalScrollIndicator : true,
    zoomScale : 2,
    maxZoomScale : 30,
    minZoomScale : 1
});
scrollViewH.add(viewImage);

var scrollViewV = Titanium.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    top : 0,
    scrollType : 'vertical',
    showVerticalScrollIndicator : true,
    zoomScale : 2,
    maxZoomScale : 30,
    minZoomScale : 1
});
scrollViewV.add(scrollViewH);

win.add(scrollViewV);
win.open();

    Hi, use a ScrollableView and add Scrollviews as subviews. works both on iOS and Android

    
var win = Ti.UI.createWindow();

var scrollView1 = Ti.UI.createScrollView({backgroundColor:"red"});
var scrollView2 = Ti.UI.createScrollView({backgroundColor:"green"});
var scrollView3 = Ti.UI.createScrollView({backgroundColor:"blue"});

var scrollableView = Ti.UI.createScrollableView({
    views:[scrollView1, scrollView2, scrollView3],
    contentWidth:Ti.UI.FILL,
    contentHeight:Ti.UI.SIZE
});

win.add(scrollableView);
win.open();
