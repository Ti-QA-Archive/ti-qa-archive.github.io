navigating among windows

i am trying to switch from one window to another and back to the previous one … but while doing so when i switch back to the first window .. the elements (buttons, views) are disabled and logcat shows that i am clicking on the window… here are my files

app.js 

var win1 = Ti.UI.createWindow({
           fullscreen:false,
        exitOnClose:true
    });

var view1 = Ti.UI.createView({
    backgroundColor:"green"
    });

var button1 = Titanium.UI.createButton({
    title:"test",
    width:75,
    height:30
    });

    win1.open();
    win1.add(view1);
    view1.add(button1);
    button1.addEventListener("click", function()
    {    
        var win = Ti.UI.createWindow({
            url:"test.js",

            title: "test"
            });
        win.open();
    });

test.js

var w = Titanium.UI.createWindow({
    exitOnClose: false,
    fullscreen:false
});
var view = Ti.UI.createView({
    backgroundColor:"red"
    });

var label1 = Ti.UI.createLabel({
    text: "test me",
    width: "auto"
    });

    w.addEventListener('android:back', function(e) {
    Ti.API.info("Pressing Back Will Not Close The Activity/Window");
    w.close();
});
    w.add(view);
    view.add(label1);
    w.open();

output on logcat

  • I experienced this same issue when using exitOnClose: true, anytime I would navigate to a different Window. However, I believe it was only with a specific Titanium Mobile SDK (1.6.2?). I believe this was a known issue that is fixed in the continuous build.

    A surefire way to solve it is to change exitOnClose to false or remove it, but obviously that's not ideal.
