                for (var i = 0, l = e.results.length; i < l; i++) {
                    var result = e.results[i];
                    var row = Alloy.createController('appRow', result).getView();

                    data.push(row);
                }

            $.tbl.setData(data);

Then I have the following event listener

$.tbl.addEventListener('click', function (evt) {
    showDetail(evt);
});

//Invoked by task record click
function showDetail(event){
    Alloy.createController('appRowDetail', event.row).getView().open({modal:true, fullscreen: true});
