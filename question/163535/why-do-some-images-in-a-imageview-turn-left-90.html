    • var padLeft = 4;
var countImg = 1;
var visible = false;
function selectPhoto(){
    Titanium.Media.openPhotoGallery({
        success : function(event){
            if(event.mediaType == Ti.Media.MEDIA_TYPE_PHOTO){
                if(countImg < 4){
                    var image = Ti.UI.createImageView({
                        borderRadius: 5,
                        top: '10%',
                        left: padLeft + '%',
                        image: event.media,
                        height: '80%',
                        width: '13%'
                    });
                    if(visible === false){
                        $.preview.show();
                        $.countPhotoView.show();
                        visible = true;    
                    };
                    padLeft += 15;
                    $.countPhotoLabel.text = countImg++;
                    $.btnPhoto.opacity = 1;
                    $.preview.add(image);

                    msgImgs.push(event.media);    
                }
            }
        }
    });
};
