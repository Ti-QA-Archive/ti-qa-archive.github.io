    ok, i worked it out myself - i wasnt passing the row tel number to the option dialog bit - here is the code that now works

    tableView.addEventListener('click',function(e){

var dialog = Titanium.UI.createOptionDialog({
    options:['Call', 'Cancel'],
    cancel:1,
    title:'blah'
});

var rowData = e.rowData.tel;

    if(e.rowData.location_id=='MORE'){
        more_index = e.index;
        page++;
        getVillas(page);
    } else {

dialog.show();

dialog.addEventListener('click',function(e)
{
    if (e.index == 0)
        {

    Ti.Platform.openURL('tel:'+rowData);

                }
});

    }

});
