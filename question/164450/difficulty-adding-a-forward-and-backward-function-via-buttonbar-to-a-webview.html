Difficulty adding a forward and backward function via buttonBar to a webView

Below is a simplified version of what I'd like to do, but I know there is something missing in the leftButtonbar.addEventListener

It's probably a simple error but I'm new to this.

By the way, I'm wanting to use a two button buttonBar to the left side of the NavBar because I have a another use planed (a button to hide/reveal the tabBar) for the right side of the NavBar.

Thanks in advance…..

// ========== CODE STARTS HERE ============

// current window

var win = Titanium.UI.currentWindow;

// ========== CREATE LEFT BUTTONBAR and ADD to LEFT NAVBUTTON ============

var leftButtonbar = Titanium.UI.createButtonBar({
labels:['Back', 'Forward'],
backgroundColor:'red'
});

win.leftNavButton = leftButtonbar;

// ========== CREATE and ADD WEBVIEW to WINDOW============

var webView = Ti.UI.createWebView({url:"http://www.google.com"});

win.add(webView);

// ========== START: FIND and ADD WEBPAGE TITLE ===========

webView.addEventListener('load',function(e)
{
var title = webView.evalJS("document.title");
win.title = title;

});

// ========== ADD LEFT BUTTONBAR EVENTS ===========

leftButtonbar.addEventListener('click', function(e){

if(e.value == 0) {
webView.goBack();
}
if(e.value == 1) {
webView.goForward();
}

});
