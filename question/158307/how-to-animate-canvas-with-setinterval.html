How to animate Canvas with setInterval

Hi, I'm trying to draw arcs using the Canvas library. I get the first blue arc to be drawn, but then nothing is happening inside the setInterval (same behaviour with setTimeout). How can I use the Canvas to build stuff dynamically?

Here is my code:

var canvas = Canvas.createView();

canvas.begin();

canvas.arc(120, 120, 50, 0 * Math.PI, 2 * Math.PI, 0);
canvas.lineWidth(10);
canvas.strokeStyle('blue');
canvas.stroke();

var pointFrom = 1.5;

var interval = setInterval(function() {
  var pointTo = pointFrom - 0.5;

  console.log('pointFrom : ' + pointFrom);
  console.log('pointTo : ' + pointTo);
  canvas.arc(120, 120, 50, pointFrom * Math.PI, pointTo * Math.PI, 1);
  canvas.lineWidth(15);
  canvas.strokeStyle('red');
  canvas.stroke();

  pointFrom = pointTo;

  if (pointFrom < 0) clearInterval(interval);
}, 2000);

my_view.add(canvas);

Thanks !!

  • I could use some help drawing a full circle arc by arc on an event (like every second).

    I was able to get individual ticks to display but not draw a full circle (controller code):

    var Canvas = require('ti.canvas');
var canvas = Canvas.createView();
$.vwCanvas.add(canvas);

var currentEndAngle = 0;
var currentStartAngle = 0;
var lineRadius = 112;
var lineWidth = 32;
var pWidth = Ti.Platform.displayCaps.platformWidth;
var pHeight = Ti.Platform.displayCaps.platformHeight;
var x = pWidth / 2;
var y = pHeight / 3;

function draw() {
    var radius;
    var width;

    var startAngle = currentStartAngle * Math.PI;
    var endAngle = currentEndAngle * Math.PI;

    currentStartAngle = currentEndAngle - 0.01;
    currentEndAngle = currentEndAngle + 0.01;

    if (Math.floor(currentStartAngle / 2) % 2) {
        currentColor = Alloy.CFG.theme_2.color_primary_1;
        radius = lineRadius - 1;
        width = lineWidth + 3;
    } else {
        currentColor = Alloy.CFG.theme_2.color_primary_0;
        radius = lineRadius;
        width = lineWidth;
    }

    var counterClockwise = false;

    canvas.begin();
    canvas.arc(x, y, radius, startAngle, endAngle, counterClockwise);

    Ti.API.info('arcInfo: x: ' + x + ' y: ' + y + ' radius: ' + radius + ' startAngle: ' + startAngle + ' endAngle: ' + endAngle);

    // line color
    canvas.strokeStyle(currentColor);
    canvas.lineWidth(lineWidth);
    canvas.stroke();
    canvas.commit();
}

var drawInterval;

$.testCanvas.addEventListener('open', function(e) {
    drawInterval = setInterval(draw, 60);
});

$.testCanvas.addEventListener('close', function(e) {
    clearInterval(drawInterval);
});

    I'm using the Ti.Canvas module not HTML5 as well as Alloy, Ti SDK 3.4.0.GA
