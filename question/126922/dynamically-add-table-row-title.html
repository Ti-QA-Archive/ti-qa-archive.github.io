  • you should create a label for each entry and then add the label to the row. It can be one lable, or severals or even pictures.
    something like:

    var profilesListWindow = Titanium.UI.createTableView({
        height:320,
        width:300,
});

var d = JSON.parse(e.result);
var row = d.data;

var data=[];

for(var i=0;i<row.lenght;i++)
{
var img = Ti.UI.createImageView({
image:picture,
top:5,
bottom:5,
height:50,
width:50,
touchEnabled:false,
left:5
});

var rowtitle = Ti.UI.createLabel({
text:fullname,
left:60,
color:textColor,
top:5,
height:'auto',
touchEnabled:false,
right:5
});                    

var row1 = Ti.UI.createTableViewRow({});                                     
row1.add(img);
row1.add(rowtitle);
row1.selectedBackgroundColor="#3c0182";
row1.backgroundColor="#3c01ff";
row1.className = 'uniqueClassName';
row1.addEventListener('click', function(e){
 //do something when clicked
});

data.push(row1);    
}//ends for 

profilesListWindow.setData(data);
