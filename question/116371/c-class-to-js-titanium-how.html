    I use asp.net with a web service to get info for my app. I use code that looks like this in Titanium:

    var loader = Titanium.Network.createHTTPClient();

//This is run after get data was successful
loader.onload = function()
{ 
    Ti.API.info(">>> App: Got the data! ... ");

    var json = JSON.parse(this.responseText);       

    //Set properties
    _id = json.Id;
    _date = json.Date;            
};

//handle errors
loader.onerror = function() {
    Titanium.API.info('>>> App: error response, http status code ' + this.status);

};

// Send the HTTP request
loader.open("GET",url);
loader.send();
