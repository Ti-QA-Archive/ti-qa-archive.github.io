  • Hello Andres,

    Try this code. it's working proper in android. i tested iin android 4.4.2 and Ti SDK 3.2.2.

    var win = Ti.UI.createWindow({
    backgroundColor : "white",
});

var txt = Ti.UI.createTextField({
    top : 15,
    height : 75,
    width : 250,
    hintText : "Vertical Align Center",
    borderColor : "black",
    borderWidth : 1,
    borderRadius : 5,
    verticalAlign : Ti.UI.TEXT_VERTICAL_ALIGNMENT_CENTER
});

win.add(txt);

var txt1 = Ti.UI.createTextField({
    top : 115,
    height : 75,
    width : 250,
    hintText : "Vertical Align Top",
    borderColor : "black",
    borderWidth : 1,
    borderRadius : 5,
    verticalAlign : Ti.UI.TEXT_VERTICAL_ALIGNMENT_TOP
});

win.add(txt1);

var txt2 = Ti.UI.createTextField({
    top : 210,
    height : 75,
    width : 250,
    hintText : "Vertical Align Bottom",
    borderColor : "black",
    borderWidth : 1,
    borderRadius : 5,
    verticalAlign : Ti.UI.TEXT_VERTICAL_ALIGNMENT_BOTTOM
});

win.add(txt2);

win.open();
