    function creaAnotacion(indice, fruteria){

        Id:fruteria.Id,
        latitude:fruteria.lat,
        longitude:fruteria.lng,
        title:'Fruter\xeda',
        pincolor:Titanium.Map.ANNOTATION_RED,
        animate:true,
        datos: fruteria,
        rightButton:'../images/platano.png',

return anotacion;
