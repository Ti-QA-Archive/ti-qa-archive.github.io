  • In the iPhone app I do the following:

    rows = this.db.execute("SELECT * FROM p WHERE r = '0' LIMIT 1");
if(rows.rowCount === 0){
    this.db.execute("INSERT INTO p (r) VALUES (0)");
    rows = this.db.execute("SELECT * FROM p WHERE r = '0' LIMIT 1");
}
//load the fields for direct editing
for(var i=0, count=rows.fieldCount(); i < count; i++){
    this[ rows.fieldName(i) ] = rows.field(i);
}
rows.close();
