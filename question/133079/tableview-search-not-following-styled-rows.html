Tableview search not following styled rows

Hi

I have a problem, which is basically the only thing holding me back from uploading to iTunes App Store..

I have a data-array (data.js) which I load onto rows applied to a tableview - this works like a charm, but I've also added a search-bar, but as I try to search my row-styling isn't applied and my rows return to standard-styling. On search-cancel the rows return to my styling.. I've tried almost everything - styling the search variable etc., but nothing seems to work..

Here's the code:

var search = Titanium.UI.createSearchBar({

});

// create table view
var tableview = Titanium.UI.createTableView({
    search: search,
    header: 'header',
    filterAttribute:'filter',
    backgroundColor:'transparent',
    separatorStyle: Ti.UI.iPhone.TableViewSeparatorStyle.NONE
});

function setData(){

Ti.include('data.js');

    var i = 0;

    for (var c = 0; c < data.length; c++) {

        var row = Ti.UI.createTableViewRow({
            height: 'auto',
            width: 'auto',
            backgroundColor:'transparent',
            selectedBackgroundColor: '#7d7d7d'
        });

        var post_title = Ti.UI.createLabel({
            text: data[c].title,
            color: '#FFF',
            textAlign: 'left',
            left: 12,
            height: 'auto',
            width: 'auto',
            top: 10,
            font: {
                fontWeight: 'bold',
                fontSize: 18
            }
        });
        row.add(post_title);

        var post_desc = Ti.UI.createLabel({
            text: data[c].desc,
            color: '#FFF',
            textAlign: 'left',
            left: 12,
            height: 'auto',
            width: 230,
            top: 40,
            bottom: 10,
            font: {
                fontSize: 13
            }
        });
        row.add(post_desc);

        row.filter = data[c].title + data[c].desc;
        row.header = data[c].header;
        row.title = data[c].title;

  • Try this:

    var win = Ti.UI.createWindow({backgroundColor: 'white'});
var searchbar = Ti.UI.createSearchBar({
    top: 0,
    height: 50,
    showCancel: false
});
win.add(searchbar);

// Function to dump data
function dumpData() {
    var data = [];
    for(var i = 0; i < 20; i ++) {
        data.push({title: Math.random().toString(36).substring(7)});
    }
    return data;
}

var data = dumpData();
var tbl = Ti.UI.createTableView({
    data: data,
    backgroundColor:'transparent',
    separatorStyle: Ti.UI.iPhone.TableViewSeparatorStyle.NONE,
    top: 50
});

win.add(tbl);
win.open();

// You can customize to whatever search algorithm you want
function mysearch(value, data) {
    var searchResults = [];
    for(var i = 0, len = data.length; i < len; i ++) {
        if (data[i].title.indexOf(value) != -1) { // Or data[i].title.indexOf(value) == 0 for search Start With...
            searchResults.push(data[i]);    
        }
    }
    return searchResults;
}

// This will do the trick :)
searchbar.addEventListener('change', function(e){
    tbl.setData([]);

    var searchResults = mysearch(e.value, data);

    if (searchResults.length == 0) {// "No Result"
        tbl.setData([
            {title: ''},
            {title: ''},
            {title: 'No Result'},
        ]);    
    } else {
        tbl.setData( searchResults );
    }
});

searchbar.addEventListener('return', function(e){
    searchbar.blur();    
});
