Alloy collection add event

Hi,

i have a collection of a model and listen to the 'add' event (backbonejs add event) of the collection like this:

var tests = Alloy.createCollection('test');

tests.on('fetch', function() {
    console.log('fetched', tests.length);
    var test = Alloy.createModel("test", {
        name: 'asdf'
    });
    test.save();    
});

tests.on('add', function() {
    console.log('added', tests.length);
});

tests.fetch();

Now, when i create a new model 'test' i was hoping the 'add' event would be raised.
But my callback is never called.

Am i doing something wrong here or is the add event not present in the alloy implementation?

Thanks in advance!
