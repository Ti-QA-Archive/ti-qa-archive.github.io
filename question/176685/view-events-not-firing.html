View events not firing

Within one of my Alloy controllers, i have a <window> containing two views. First view takes up the entire space on the screen. Second view is placed outside. I then made a slide animation that i use on the window itself, enabling me to see the second view.
However, i am not able to get any events fired from the second view.
If i use the slide animation on both of the views instead of the window, the same problem occurs.

  • index.js

    var menuOpen = false;

var animateLeft = Ti.UI.createAnimation({
    left : "-195dp",
    curve : Ti.UI.ANIMATION_CURVE_EASE_OUT,
    duration : 350
});

var animateReset = Ti.UI.createAnimation({
    left : 0,
    curve : Ti.UI.ANIMATION_CURVE_EASE_OUT,
    duration : 350
});

$.slideMenuButton.addEventListener('click', function() {
    if (menuOpen === false) {
        $.index.animate(animateLeft);
        menuOpen = true;
    } else {
        $.index.animate(animateReset);
        menuOpen = false;
    }
});

    index.xml

    <Alloy>
    <Window class="container" navBarHidden="true" backgroundImage="/images/background.png" >
        <View id="slideMenuView" backgroundColor="#3d444d" height="Ti.UI.FILL" width="195dp" left="320dp"/>
<View id="pageContent" backgroundColor="#3d444d" height="Ti.UI.FILL" width="320dp"  left="0dp"/>
</Window>
</Alloy>
