There is a window, I am adding a view to it.

});

var FriendListLabel = Ti.UI.createLabel({
  text:'Friends',
  color:'#FFF',
  font:{fontWeight:'bold'}
});

var FriendRequestLabel = Ti.UI.createLabel({
  text:'Requests',
  color:'#FFF',
  font:{fontWeight:'bold'}

var friendsListView = Ti.UI.createView();
var friendsRequestView = Ti.UI.createView();

friendsListView.add(FriendRequestLabel);
friendsRequestView.add(FriendListLabel)

FriendRequestLabel.addEventListener('click',function() {
