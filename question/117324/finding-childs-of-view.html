Finding childs of view

How can I find the child controls in a view eg. scrollview. For example how can I find all textfields in the scrollview?
Edit: Something like getElementById in pure javascript.

    my suggestion is to set id's on the object and then you can loop through the children to find the objects based on the id

    var win1 = Ti.UI.createWindow({
  backgroundColor:"#fff",
  title:"My App"
});

var label1 = Ti.UI.createLabel({id:"label1",  text: "label one" });
win1.add(label1);

var label2 = Ti.UI.createLabel({id:"label2", text: "label two" });
win1.add(label2);

var label3 = Ti.UI.createLabel({id:"label3", text: "label three" });
win1.add(label3);

for (var x in win1.children ) {
    Ti.API.info(win1.children[x].id);
}
