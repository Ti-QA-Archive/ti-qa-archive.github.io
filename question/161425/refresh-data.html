  • Hi Preety Shah!!

    The following code can be a one kind of solution for your problem 

    var aButton = Ti.UI.createButton({
    title : 'Refresh',
    top : 5,
});

var aTableView = Ti.UI.createTableView();
var data = [
    {title:'Name1'},
    {title:'Name2'},
    {title:'Name3'},
    {title:'Name4'},
    {title:'Name5'}
];
aTableView.setData(data);

aButton.addEventListener('click', function() {
    loadData();
    aTableView.setData(data);

});
function loadData()
{
      data.push(
        {
            title:"Amimul"
      });
}
