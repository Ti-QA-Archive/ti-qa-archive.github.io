  • You would just make individual views the size that you want them and place them in another view or window.
    Just use: top: ##
    bottom: ##
    left: ##
    right: ##
    To position the smaller views where you want them.
    The ## would actually be a number for the amount of pixels from the position,
    So for example…

    var mainView = Titanium.UI.createView({
    top: 0,
        left:0,
    backgroundColor:'#687990',
    border: 1,
    borderColor: '#000',
    width:600,
    height:800
    borderRadius:10
});

win.add(mainView);


var viewOne = Titanium.UI.createView({
    top: 0,
        left:0,
    backgroundColor:'#687990',
    border: 1,
    borderColor: '#000',
    width:200,
    height:800
    borderRadius:10
});

mainView.add(viewOne);


var viewTwo = Titanium.UI.createView({
    top: 0,
        left:200,
    backgroundColor:'#687990',
    border: 1,
    borderColor: '#000',
    width:400,
    height:400
    borderRadius:10
});

mainView.add(viewTwo);


var viewThree = Titanium.UI.createView({
    top: 400,
        left:200,
    backgroundColor:'#687990',
    border: 1,
    borderColor: '#000',
    width:400,
    height:400
    borderRadius:10
});

mainView.add(viewThree);
