  • Hi Mohammed,

    Try to add id in your paragraph tag for example:

    <p id='yourID'>This is simple paragraph.</p>

    in your code you can find out the paragraph by getElementById property.

    see following example:

    var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});

var label1 = Titanium.UI.createLabel({
    color:'#999',
    text:'Click Label to get Text in paragraph.',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto',
    top :  10
});

label1.addEventListener('click',function(e){
    alert(Ti.XML.parseString(webview.html).getElementById('first').getText());
});


win1.add(label1);

var webview = Ti.UI.createWebView({
    top : 100,
    height : 300,
    html : '<html><head><title></title></head><body><p id="first">This is simple text in paragraph</p></body></html>'
});
win1.add(webview);

win1.open();
