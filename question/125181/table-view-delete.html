  • OK, here is some code.

    
var tableview = Titanium.UI.createTableView({
    data:data,
    editable:true


});

tableview.addEventListener('delete', function(e) {
    var db = Titanium.Database.open('q1.db');
    var user = e.rowData.title

    var alertYesNo = Titanium.UI.createAlertDialog({
        message: 'Are you sure you want to delete ALL entries by' + user,
        buttonNames: ['Yes','No']
    });

    alertYesNo.addEventListener('click', function(e) {
        if (e.index == 0) { //clicked yes
                var sql = db.execute("DELETE FROM q1 WHERE name='" + user + "'");
        } else if (e.index == 1) { //clicked no 
        }

    });
    alertYesNo.show();



});
