ImageView bug

I have an easy home made advertising code but it is not working properly. sometimes the images are compressed on a few pixels, why? How can I fix this?

var images = [];
for (var i=1;i<5;i++)
{
    images.push('images/pub/pub' + ((i<10)?'0'+i:i)+'.png');
}

//
// CREATE IMAGE VIEW
//

var imageView = Titanium.UI.createImageView({
    images:images,
    duration:3000, // in milliseconds, the time before next frame is shown
    repeatCount:0,  // 0 means animation repeats indefinitely, use > 1 to control repeat count
    bottom:'0%',
    height:'15%',

imageView.start();
first.add(imageView);
