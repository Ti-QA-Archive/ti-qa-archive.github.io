    This is one approach… would need info on the image

    var stop = Titanium.UI.createButton({
   //title:'Stop', 
   backgroundImage : "images/play.png"
   height:40, 
   width:145, 
   right:10, 
   top:10 
}); 
stop.addEventListener('click', function() {
  sound1.stop(); 
  pb.value = 0; 
}); 

win1.add(stop);
