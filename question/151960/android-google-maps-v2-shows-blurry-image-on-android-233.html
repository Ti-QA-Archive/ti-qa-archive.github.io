    • did you find solution ?
      i have the same issue.
      my map is blurred, im using MAP V2 for android.
      my code :

      function displayMapForAndroid(locations){

    var currentLatitude, currentLongitude;
    Ti.Geolocation.getCurrentPosition(
        function locationResults(result){
            if(result.success){
                currentLatitude = result.coords.latitude;
                currentLongitude = result.coords.longitude;

                var MapModule = require('ti.map');
                map = MapModule.createView({
                    userLocation: true,
                    mapType: MapModule.NORMAL_TYPE,
                    animate: true,
                    region: {latitude: currentLatitude, longitude: currentLongitude, latitudeDelta: 0.1, longitudeDelta: 0.1 },
                    top: 88
                });

                map.addEventListener('click', function(e){
                    if(e.clicksource == 'infoWindow'){
                        var tel = 'tel:' + e.annotation.phone.replace(/\./g, '');
                        var dialog = Ti.UI.createAlertDialog({
                            buttonNames: ['Oui', 'Non'],
                            message: L('TelephonConfirm') + e.annotation.phone + ' ?',
                          });

                          dialog.addEventListener('click', function(_e){
                            if(_e.index == 0){
                                //Appeler le numéro                    
                                var intent = Ti.Android.createIntent({
                                    action: Ti.Android.ACTION_CALL,
                                    data: tel
                                });

                                Ti.Android.currentActivity.startActivity(intent);
                            }
                          });

                          dialog.show();
                      }
                });

                for(var i=0, j=locations.length; i<j; i++){
                    var boutique = MapModule.createAnnotation({
                        latitude: locations[i].latitude,
                        longitude: locations[i].longitude,
                        pincolor: MapModule.ANNOTATION_AZURE,
                        rightButton: '/images/croix_quitter.png',    
                        title: locations[i].titre,
                        subtitle: locations[i].adresse,
                        phone: locations[i].telephone
                    });

                    map.addAnnotation(boutique);
                }

                $.geoloc.add(map);
            }
        }
    );
}
