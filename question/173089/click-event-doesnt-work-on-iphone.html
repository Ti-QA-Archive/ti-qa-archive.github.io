Click event doesn't work on iPhone

Hi!

I have a mobile application for Android/Iphone with a tableview. When I click on a row, a sub view is opened. This sub view contains a button.

When I click on the button with Android, the click event works, but doesn't work on iPhone. The click on the tableview works on Android and iPhone.

Screenshot of my program

http://postimg.org/image/didw2bmir/

http://postimg.org/image/tunxs20ub/

I use :
CLI version 3.2.1, Titanium SDK version 3.2.2.GA
IOS Simulator Version 7.0 (463.9.4.2)

My code. Don't copy past it because the result will be different from the screenshot. I removed a lot of proprieties (height, width, colour, …) when creating the objects.

// Fill the table with all the symptoms
for (var i = 0; i < listSymptoms.length; i++) {
    var symptom = listSymptoms[i];


    // Create a Label.
    var aLabel = Ti.UI.createLabel({
        text : symptom.getDescription()}
        );

    // Create a Button.
    var aButton = Ti.UI.createButton({
        title : L('green_button_text'),
    });


    // Listen for click events
    aButton.addEventListener('click', function(e) {
        Ti.API.info(e.source.title);
    });

    // Sub View when row is clicked
      var subwin = Titanium.UI.createView();
    subwin.add(aLabel);
    subwin.add(aButton);


    // Create row and his label
    var rowLabel = Titanium.UI.createLabel({
        text : symptom.getTitle().getText()
    })

    var row = Titanium.UI.createTableViewRow({
        sub:  [subwin,]
    });        
    row.add(rowLabel);
    rowData.push(row);

}

//Add data to tableView Here
$.table.setData(rowData);
$.table.addEventListener('click', function(e) {
        var currentIndex = e.index;
        //getting Expandable
        var subRow = Titanium.UI.createTableViewRow();
        subRow.add(e.row.sub[0]);
        $.table.insertRowAfter(currentIndex, subRow);

});

Thank you
