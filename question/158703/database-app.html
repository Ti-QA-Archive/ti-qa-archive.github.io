    Code to create, insert and list details:

        var db = Ti.Database.install("test.sqlite", "test");
    db.execute('INSERT INTO test (id, name) VALUES (?, ?)', 2, 'Gowtham S');
    var rows = db.execute('SELECT * FROM test;');
    while (rows.isValidRow())
    {
        alert(rows.fieldByName('id')+' '+rows.fieldByName('name'));        
        rows.next();
    };
    db.close();
    rows.close();
