  • Hello Apichart apinyasudja,

    Thanks for writing us.I have tested your issue using method and properties for visibility of view.Both are working well without any issue in latest Ti builds.

    Please check the below code . It may help to you.

    Test Code:

    
var win = Ti.UI.createWindow({
    backgroundColor: 'orange',
    layout:'vertical'

    });

    // Create a Button.
 var button1 = Ti.UI.createButton({
    title : 'Test with properties',width:'300px',height:'50px',top:'30',backgroundColor:'pink',

});

var view1 = Ti.UI.createView({
    backgroundColor: 'Green',width:'150px',height:'50px',top:'30',

    });

    // Create a Button.
    var button2 = Ti.UI.createButton({
        title : 'Test using method',width:'300px',height:'50px',top:'50',backgroundColor:'pink',        
     });


   var view2 = Ti.UI.createView({
       width:'150px',height:'50px',top:'70',backgroundColor:'red'    

    });

    // Listen for click events.View will be hide using visible property
    button1.addEventListener('click', function() {
        view1.visible = false;

    });

    // Listen for click events.View will be hide using method setVisible
    button2.addEventListener('click', function() {
        view2.setVisible(false);             
    });

    // Add to the parent view.
    win.add(button1);win.add(button2);win.add(view1);win.add(view2);

    win.open();
