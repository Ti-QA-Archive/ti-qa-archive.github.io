Can't change visibility on object after animated on android. Alloy

Alloy 1.0, Android 2.3.3, Android Device, TI Studio 3.1.0

I need to animate objects (views, etc) and then change their "visible" property to hide them.
The doHide() function works, if the label is NOT animated using the doSetup() function.
The animation's "complete" function will not change the visibility either.

Anyone know why this is?

index.xml

<Alloy>
    <Window class="container" backgroundImage="/images/MainMenu/background_menu.png" onOpen="doSetup">
        <Button id="button" onClick="doHide" title="Hello" top="10" left="500" width="100" height="100" />
        <Label id="label1" color="#fff" shadowColor="#aaa" text="A simple label" backgroundColor="#000" textAlign="Ti.UI.TEXT_ALIGNMENT_CENTER" top="30" left="100" width="120" height="60" />
    </Window>

</Alloy>

index.js


function doHide(e) {
    if ($.label1.visible === "true") {
        $.label1.visible = "false";
    } else {
        $.label1.visible = "true";
    }
}
function doSetup(e) {

    var t = Ti.UI.create2DMatrix();
    var t = t.translate(100, 5);

        var a = {
            transform : t,
            duration : 2000,
            anchorPoint : {
                x : 0.5,
                y : 0.5
            },
        };
        $.label1.animate(a, function() {
            $.label1.backgroundColor = "#aaa";
            $.label1.left = "0";
            //$.label1.visible = "false";
        });
}
