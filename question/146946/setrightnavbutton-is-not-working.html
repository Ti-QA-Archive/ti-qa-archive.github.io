    Hi

    Are you seeing the title/nav bar either, from the code supplied you would not be as you do not have either a navigationGroup of a tabGroup which when then add the title bar that would allow the left and right buttons to be set.

    BTW the;

    win.backButtonTitle = 'Back';

    Only works if the window win was opened as a child of a navigationGroup or a tabGroup as that simply changes the caption that appears on the automatically applied back button - but only if that exists.

    backgroundColor: '#fff',

      win1.js create a new file using this name and paste the code below;

      var win = Ti.UI.currentWindow;

var btn = Ti.UI.createButton({
    title: 'Win'
});
btn.addEventListener('click', function (e) {
    var newWin = Ti.UI.createWindow({
        backgroundColor: '#fff',
        //backButtonTitle: 'Go Back', // uncomment this line if you want to specify a button caption
        height: Ti.UI.FILL,
        title: 'New Window',
        url: 'newwin.js',
        width: Ti.UI.FILL
    });
    var tab = Ti.UI.currentTab;
    tab.open(newWin, { animated: true });
});
win.setRightNavButton(btn);

var lbl = Ti.UI.createLabel({
    height: Ti.UI.SIZE,
    text: 'New Window',
    width: Ti.UI.SIZE
});
win.add(lbl);

      newwin.js create a new file using this name and paste the code below;

      var win = Ti.UI.currentWindow;

var btn = Ti.UI.createButton({
    title: 'Hello'
});
btn.addEventListener('click', function (e) {

var lbl = Ti.UI.createLabel({
    height: Ti.UI.SIZE,
    text: 'New Window',
    width: Ti.UI.SIZE
