  • This simple example works fine. Without any code from you it is practicably impossible to help you

    var win = Ti.UI.createWindow();

var p = Ti.UI.createPicker();

p.addEventListener('change', function(e) {
    alert('changed');
});

var rows = [];

for (var x = 0; x < 20; x++) {
    var r = Ti.UI.createPickerRow({
        title : x.toString()
    });

    rows.push(r);
}
p.add(rows)

win.add(p);
win.open();
