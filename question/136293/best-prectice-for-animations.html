Best Prectice for animations?

Hi,
I try to copy the dynamic navigation like in SPON (SPIEGEL Online):
Over a tableview is on bottom a navi bar. The navi bar appears if the tableview is on "bounds" (top and bottom).

Here my first try:

var anims = {
    "gal_is_up" : true,  // start
    "gal_go_down" : function() {
        var down = Ti.UI.createAnimation({
            duration : 400,
            bottom : -70
        });
        down.addEventListener('complete', function() {
            anims.galisup = false;
        });
    },
    "gal_go_up" : function() {
        var up = Ti.UI.createAnimation({
            duration : 400,
            bottom : 0
            });
        up.addEventListener('complete', function() {
            anims.gal_is_up = true;
        });
    }
};

and here the scroll listener:

tableView.addEventListener('scroll', function(_e) {
    var height = _e.contentSize.height - _e.size.height;
        if(_e.contentOffset.y < 100 || _e.contentOffset.y + 100 > height) {
            if(anims.gal_is_up == false) {
                galerie.animate(anims.gal_go_up());
            }
        } else {
            if(anims.gal_is_up) {
                galerie.animate(anims.gal_go_down());
            }
        }
});

But it doesnt work ;-((

Any ideas, whats going wrong?
