function afterLog(e) {

    var animate1=animation.fadeOut(log, 700, function(){
        var ani1=animation.fadeIn(win, 700, function () {});

    Hello Jily,

    Try to do like this…

    function afterLog(e) {
    var log=$.index;
    var animation = require('alloy/animation');
    var win = Alloy.createController("first").getView();
    var animate1=animation.fadeOut(log, 700, function(){
        win.open({activityEnterAnimation:Ti.Android.R.anim.fade_in});
    });
};
