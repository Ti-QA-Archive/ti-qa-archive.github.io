    Hi

    I think you are looking for the slider as it is known in Titanium.

    var slider = Ti.UI.createSlider({
    left: 20,
    right: 20,
    min: 0,
    max: 100,
    top: 50,
    value: 50,
    width: Ti.UI.FILL
});
slider.addEventListener('change', function (e) {
    Ti.API.info('slider.value: ' + slider.getValue());
});
win.add(slider);
