User login via a webservice in Titanium Alloy

Hi,
I am building a mobile application for both iPhone and Android in Titanium Alloy.

Currently, I am building a login screen where the user enters his email address and password and clicks to the login button to login. The code that I do for traditional Titanium ( as found on the web) is :

//Parsing the JSON String.
var loginReq = Titanium.Network.createHTTPClient();
loginReq.onload = function()
{
    var json = this.responseText;
    var response = JSON.parse(json);

    if (response.logged == "true")
    {
        alert("Welcome " + response.name + "Hello" + response.name);
    }
    else
    {
        alert(response.message);
    }
};


//Event listener for the login button
loginBtn.addEventListener('click',function(e)
{
    if (name.value != '' && password.value != '')
    {
        loginReq.open("POST","http://localhost:9000/login");
        loginReq.setRequestHeader('Content-Type','application/json');
        var params = {
            "name": name.value,
            "password": password.value
        };
        Ti.API.info('Params'+JSON.stringify(params));
        loginReq.send(JSON.stringify(params));
    }
    else
    {
        alert("Username/Password are required");
    }
});

Can some one please help me on how I can use the same code for Alloy.

Alternatively, if some one can point me to a tutorial or sample code for Alloy involving user login using web service that would be great as well.

Thanks a lot,

Bably
