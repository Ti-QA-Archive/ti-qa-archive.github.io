  • Hi Muhammad Waqas,

    This code may help you.

    function setupNewWindow(myTitle, myUrl, myIcon) {
    var menuWin = Titanium.UI.createWindow({
        url : '' + myUrl,
        //title : myTitle,
        orientationModes : [Titanium.UI.PORTRAIT]
    });

    var menuTab = Titanium.UI.createTab({
        icon : myIcon,
        height : 39,
        width : 39,
        title : myTitle,
        window : menuWin
    });
    tabGroup.addTab(menuTab)
}

setupNewWindow("Search", "search.js", "pics/search@2x.png");
