Error during facebook.authorize()

TiSDK 3.1.2.v20130619101604 and iPhone 5 + iOS 6.1.4

The Facebook app is in sandbox mode. My partner and I have the save device/os and are both admins of the FB app. When I try to authorize, the normal FB authorize message flips onto the screen, I agree to access and posting and everything works. Other app admins can see my app's posts in Facebook.

When my partner tries to authorize, FB auth screen flips on, she agrees to access and posting and then she gets the red error header "Sorry, this feature isn't available right now: An error occurred while processing this request. Please try again later."

Here is the auth code I am using:

var fb = require('facebook');
fb.appid = 'xxxxx';

fb.permissions = [
                'publish_stream', 
                'friends_photos', 'user_photos', 
                'user_interests', 'friends_interests', 
                'user_hometown', 'friends_hometown'
            ];
fb.addEventListener('login', facebookLoginEvent);
fb.authorize();


function facebookLoginEvent(e) {
    fb.removeEventListener('login', facebookLoginEvent);
    if (e.success) {
        Ti.API.info('FB logged in = '+fb.loggedIn);

        $.facebookControl.setStatus(true);

        gameReady();
    } else if (e.error) {
        alert(e.error);
    } else if (e.cancelled) {
        alert("Canceled");
    }
}
