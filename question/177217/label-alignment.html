  • Hi

    As Ahmed mentioned, you should put them in a view and then you are able to use the horizontal property of layout. In addition you need to nest your childTemplate's to achieve this and here is an example:

    var plainTemplate = {
    childTemplates : [{
        type : 'Ti.UI.View',
        bindId : 'view',
        properties : {
            height : Ti.UI.SIZE,
            layout : 'horizontal',
            left : 31,
            width : 200
        },
        childTemplates : [{
                type : 'Ti.UI.Label',
                bindId : 'firstname',
                properties : {
                    font : {
                        fontFamily : 'Arial',
                        fontSize : '20dp',
                        fontWeight : 'bold'
                    },
                    height : Ti.UI.SIZE,
                    left : 0,
                    textAlign : 'left',
                    touchEnabled : 'false',
                    top : 10,
                    width : Ti.UI.SIZE
                }
            }, 
            {
                type : 'Ti.UI.Label',
                bindId : 'lastName',
                properties : {
                    font : {
                        fontFamily : 'Arial',
                        fontSize : '20dp',
                        fontWeight : 'bold'
                    },
                    height : Ti.UI.SIZE,
                    left : 10,
                    touchEnabled : 'false',
                    top : 10,
                    width : Ti.UI.SIZE
                }
            }
        ],
        {
            type : 'Ti.UI.ImageView',
            bindId : 'cellImage',
            properties : {
                height : 30,
                image : '/images/images.jpeg',
                left : 1,
                top : 10,
                width : 30
            }
        }
    ]
};

    Hope this helps
