Sliders in ScrollableView doesn't work on iOS

I have 3 views added to a ScrollableView, each view has sliders controls. On Android I can move the slider thumb to set up a value but on iOS the scrollableview captures the events and I can only move between the views, the slider doesn't catch the touch event on the thumb.

Any ideas?.

Thanks in advance

I don't know if this is a bug of Titanium.

  • I've tested apart with this code and it works, I'm checking the whole code again.

    function Test(){
    //Here's the first window...
    var first = Ti.UI.createWindow({
      backgroundColor:"#fff",
      title:"My App"
    });

    var view1 = Ti.UI.createView();
    var view2 = Ti.UI.createView();


    var label1 = Ti.UI.createLabel({text:'Slider 1 - View 1'});
    var slider1 = Titanium.UI.createSlider({
                                        top:20,
                                        min: 0,
                                        max: 5,
                                        minRange: 0,
                                        maxRange:5,
                                        left: '3%',
                                        width: '83%',
                                    });

    var label2 = Ti.UI.createLabel({text:'Slider 2 - View 2'});                            
    var slider2 = Titanium.UI.createSlider({
                                    top:20,
                                    min: 0,
                                    max: 5,
                                    minRange: 0,
                                    maxRange:5,
                                    left: '3%',
                                    width: '83%',
                                });
    view1.add(label1);
    view1.add(slider1);
    view2.add(label2);
    view2.add(slider2);

    var scrollableview = Ti.UI.createScrollableView({
        views: [view1, view2]
    });


    first.add(scrollableview);

    return first;    
}


module.exports = Test;
