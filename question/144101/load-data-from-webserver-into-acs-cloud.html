  • Hello. The following script might help shed some light on how you might do this. Essentially, pull data from source A, and then send that data to source B. The following example is how you might go about migrating ACS.Places objects. Please note that the object's unique IDs would be different since you are essentially creating new data in ACS. This example is from ACS to ACS, but the idea could be updated for other datasources too.

    function transfer(req, res){
    /*
        Step 0: Setup needed libs. ACS used for querying from ACS. request for creating new places.
    */
    var ACS = require('acs').ACS;
    var request = require('request');

    // Step 1: Login to source ACS data with admin
    ACS.init('<SOURCE API KEY>');
    ACS.Users.login({
        login:'<ACS ADMIN ID>',
        password:'<ACS ADMIN PASS>'
    },function(init_login){
        if(init_login.success){
            console.log('Initial login sucess');

            // Step 2: Query ACS for the places you want to to transfer
            ACS.Places.query({
                per_page:'1000'
            },function(places_query){
                if(places_query.success){
                    console.log('Query success');

                    // Step 3: Login to destination ACS data
                    var login_options = {
                        url:"https://api.cloud.appcelerator.com/v1/users/login.json?key=<DESTINATION ACS API KEY>",
                        form:{login:'<DESTINATION ADMIN ID>',password:'<DESTINATION ADMIN PASS>'}
                    }
                    request.post(login_options,function handleResponse(error, response, body){
                        var session_id = JSON.parse(body).meta.session_id;
                        var create_options = {
                            url:"https://api.cloud.appcelerator.com/v1/places/create.json?key=<DESTINATION ACS API KEY>&_session_id=" + session_id
                        }

                        // Step 4: Create new Places objects into destination ACS data.
                        // The create_options form object would be filled out with whatever fields you want to carry over.
                        for(var i = 0; i < places_query.places.length; i++){
                            create_options.form = {name:places_query.places[i].name, address:places_query.places[i].address};
                            request.post(create_options,function(error, response, body){
                                console.log(body);
                            });
                        }
                    });
                    res.send(200);
                }
                else{
                    console.log('error');
                }
            });
        }


    });


}
