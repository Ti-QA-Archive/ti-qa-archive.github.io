  • The Titanium.Media.showCamera method has an optional parameter called saveToPhotoGallery. Set it to false and just push up the blob/media object returned in the success callback to ACS. There's no need to save it to your device first as far as I'm aware.

  • Still have some problem though. Its not saving the photo…please see my code:

    camera.js (a common js module I created):

    function camera() {

    Titanium.Media.showCamera({

    success:function(event)
    {
        function savePhoto()
        {
                     var Cloud = require('ti.cloud');

                    var blobdata = profilePic.toImage();
                    //var blobdata = profilePic.toBlob();

                    Cloud.Photos.create({
                        photo: blobdata
                    }, function (e) 

                    {
                        if (e.success) 
                        {
                            var photo = e.photos[0];
                            alert("success");
                            updateUserInfo(photo.id);

                        } else 
                        {
                            alert('Error:\\n' + ((e.error && e.message) || JSON.stringify(e)));
                        }
                    });
           savePhoto();
        }

    },
    cancel:function()
    {
    },
    error:function(error)
    {
        var a = Titanium.UI.createAlertDialog({title:'Camera'});
        if (error.code == Titanium.Media.NO_CAMERA)
        {
            a.setMessage('Please run this test on device');
        }
        else
        {
            a.setMessage('Unexpected error: ' + error.code);
        }
        a.show();
    },
    showControls:true,    // don't show system controls
    mediaTypes:Ti.Media.MEDIA_TYPE_PHOTO,
    autohide:true,    // tell the system not to auto-hide and we'll do it ourself
    saveToPhotoGallery: false
});

    };

    module.exports = camera;

    And the codesnippet in the receiving window:

    // BUTTON CAMERA
    btn_Camera = Ti.UI.createButton({

    });

    v_Buttons.add(btn_Camera);

photo();
