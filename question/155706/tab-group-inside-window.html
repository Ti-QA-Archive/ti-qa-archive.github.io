  • Here is common js slider menu using tab Group in titanium

    app.js

    Ti.UI.backgroundColor = '#2d2d2d';

if (Ti.Platform.osname == "iphone")
    Ti.UI.iPhone.setStatusBarStyle(Ti.UI.iPhone.StatusBar.OPAQUE_BLACK);

var HomeScreen = require("HomeScreen");

var rootWindow = Ti.UI.createTabGroup();

if (Ti.Platform.osname == "iphone" || Ti.Platform.osname == 'ipad') {
    var navGroup = Ti.UI.createTab({
        window : new HomeScreen(rootWindow),

    });

    rootWindow.addTab(navGroup);
    rootWindow.open();
} else {
    new HomeScreen().open();

}

if (Ti.Platform.osname == "android") {
    Ti.Gesture.addEventListener("orientationchange", function(e) {
        Ti.Android.currentActivity.setRequestedOrientation(Ti.Android.SCREEN_ORIENTATION_PORTRAIT);
    });
}

    HomeScreen.js

    var MenuScreen = require("MenuScreen");
function HomeScreen(rootWindow) {
    new MenuScreen().open();
    var self = Ti.UI.createWindow({
        backgroundColor : 'red',
    });
    var menuButton = Ti.UI.createButton({
        title : 'Menu',
        toggle : false, // Custom property for menu toggle
    });
    self.setLeftNavButton(menuButton);
    menuButton.addEventListener('click', function(e) {
        //    new MenuScreen().open();
        if (e.source.toggle == true) {
            rootWindow.animate({
                left : 0,
                duration : 400,
                curve : Ti.UI.ANIMATION_CURVE_EASE_IN_OUT
            });
            e.source.toggle = false;
        }
        // If the menu isn't opened
        else {

            rootWindow.animate({
                left : 150,
                duration : 400,
                curve : Ti.UI.ANIMATION_CURVE_EASE_IN_OUT
            });
            //new MenuScreen().open();
            e.source.toggle = true;
        }
    });

    return self;
}

module.exports = HomeScreen;

    MenuScreen.js

    function MenuScreen() {
    var self = Ti.UI.createWindow({
        left : 0,
        width : '150dp'
    });
    //    self.open();
    var menuTitles = [{
        title : 'Menu 1'
    }, {
        title : 'Menu 2'
    }, {
        title : 'Menu 3'
    }, {
        title : 'Menu 4'
    }, {
        title : 'Menu 5'
    }, {
        title : 'Menu 6'
    }];
    // Tableview
    var tableView = Ti.UI.createTableView({
        top : '0%',
        data : menuTitles
    });
    tableView.addEventListener('click', function(e) {
        alert('I am clicked');
    })
    self.add(tableView);
    return self;
}

module.exports = MenuScreen;

    Run and enjoy the code
