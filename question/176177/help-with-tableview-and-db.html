Help with tableview and db

Hi guys, I'm using this code to show a table view that read the variable from a db:

    var db = Ti.Database.install('ui/common/accountdb.sqlite','account');


var rows = db.execute('SELECT DISTINCT nome FROM account');

var tableview = Ti.UI.createTableView({
    top:'12.5%',
    backgroundColor: '#03447f',
    height:(Ti.Platform.displayCaps.platformHeight)-90
});

var dataArray = [];

while (rows.isValidRow())
{
    dataArray.push({title:'' + rows.fieldByName('nome') + ''});
    rows.next();    


};

tableview.setData(dataArray);



self.add(tableview);

};

setData();

Now I'd like to custom the rows, because I need to add some row background image and change the text color (now is black).

How can I do that?

    Hello Simone,

    it is as easy as u think.

    See

    while (rows.isValidRow())
{
    var row = Ti.UI.createTableViewRow({
        backgroundColor : "#484848", //Change the row background color
        height : "50dp",
        title:'' + rows.fieldByName('nome') + '',
    color : "white" //change text color
    });

    dataArray.push(row);
    rows.next();
};

    Using this You can create tablview row and you can also add some Titanium compponents in this.

    Like,

    while (rows.isValidRow())
{
    var row = Ti.UI.createTableViewRow({
        backgroundColor : "#484848", //Change the row background color
        height : "50dp",
        title:'' + rows.fieldByName('nome') + '',
    color : "white" //change text color
    });

    //Add imageview
    var imageview = Ti.UI.createImageView({
        image : "path.png",
        height : "30",
        width : "30"
    });
    row.add(imageview); 

    dataArray.push(row);
    rows.next();
};
