Assigning eventlisteners to touchEnabled label

Hi,

I created a very simple text-based nav menu like this:

function makeButton(text, left) {
    return Ti.UI.createLabel({
        top: 10,
        left: left,
        height: 23,
        touchEnabled: true
    });
}
win.add(makeButton('Home', '30'));
win.add(makeButton('Media', '100'));
win.add(makeButton('Info', '200'));

How can I assign identifier to each element so I can add an eventlisteners to it?

many thanks.

    function makeButton(text, left) {
    var label = Ti.UI.createLabel({
        top: 10,
        left: left,
        height: 23,
        touchEnabled: true
    });
    label.addEventListener('click', function() {
        //do somethiner here
    });
    return label;
}
