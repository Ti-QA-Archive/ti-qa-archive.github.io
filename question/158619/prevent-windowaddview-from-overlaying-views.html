Prevent window.add(view) from overlaying views?

I am trying to use the .add() function to add multiple views to a window. My problem is that they all overlay each other. Is there a way to get them to stack instead of covering each other up?

var mainWin = Titanium.UI.createWindow();
            var topAd = Ti.UI.createImageView({id:'adImage',image:'images/title_bar.png', height:'auto', width:'100%', top:'0', zIndex:'0'});
            mainWin.add(topAd);
            var view = Alloy.createWidget(container.get('container_type') + '-container', {container:container, mainWin:mainWin, open:function(w){mainWin.add(w);}});
            mainWin.add(view.getView());
            tab.window = mainWin;
            $.tabGroup.addTab(tab);
