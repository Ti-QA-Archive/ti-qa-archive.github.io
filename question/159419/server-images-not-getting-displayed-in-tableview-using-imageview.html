Server images not getting displayed in tableview using imageview

Hi guys i am trying to display images using tableview . I followed code from github. But its not displaying images. I am getting proper data length and also it s displaying rows but not images . When i debug i see ll images but its not getting displayed in emulator or phone .

Please let me know where i am going wrong . Thanks in advance

var GalleryWindow = function(/*Sub Category ID*/subCategoryID, /*Tab object reference*/containingTab) {
    var win = Ti.UI.createWindow({
        backgroundColor : 'transparent',
        backgroundImage : '/images/BG3.png',
        barColor : '#6d0a0c'
    });

    var db = require('lib/network');
    db.galleryDetails(subCategoryID, function(galleryDetails) {
        //alert("Length "+galleryDetails.length);
        //alert("Image link "+galleryDetails[1].produtId);
        // loop thru and display images
        var table = Ti.UI.createTableView();
        var data = [];
        for (var i = 0; i < galleryDetails.length; i += 2) {
            //alert('Inside loop');
            // Empty Table Row
            var row = Ti.UI.createTableViewRow();
            //alert('Image '+galleryDetails[i].image);
            // The image on the left of the row
            var image1 = Ti.UI.createImageView({
                url : galleryDetails[i].image,
                //url:'/images/goldt.png',
                left : 10,
                top : 2,
                width : 100,
                height : 100,
                preventDefaultImage:true
            });

            // The image on the right of the row
            var image2 = Ti.UI.createImageView({
                url : galleryDetails[i + 1].image,
                //url:'/images/goldt.png',
                left : 190,
                top : 2,
                width : 100,
                height : 100,
                preventDefaultImage:true
            });

            // Add the two images to the newly created row
            row.add(image1);
            row.add(image2);

            // A mut for performance reasons
            row.className = 'two_images_per_row';

            // Add the row to the data variable
            data.push(row);

        }
        alert('Data length is '+data.length);
        table.setData(data);
        win.add(table);
    });
    return win;
};
module.exports = GalleryWindow;
