TabGroup show navbar double on last tab on iOS

I'm using tabgroup and i have 5 tabs,
but when I'm click tab number five for the first time, It's show navbar double,
when I'm click another tab then back to tab5, it's will be back to normal.

my code is like that

    var tabGroup = Titanium.UI.createTabGroup({
        tabsBackgroundColor:'#05bee0',
        tabsBackgroundSelectedColor:'#f1f1f1',
        barColor:'#f1f1f1',
    });

    var tab1 = Titanium.UI.createTab
    ({  
          window:Gees.ui.createCategoryWindow(),
          icon:'/images/android/home.png',
          width:64,
          title:'Home'
    });

    var tab2 = Titanium.UI.createTab
    ({  
        window:Gees.ui.createExclusiveWindow(),
        icon:'/images/android/exclusive.png',
        width:64,
        title:'Exclusive'
    });

    var tab3 = Titanium.UI.createTab
    ({  
        window:Gees.ui.createMapWindow(),
        icon:'/images/android/map.png',
        width:64,
        title:'Map'
    });

    var tab4 = Titanium.UI.createTab
    ({  
        window:Gees.ui.createSearchWindow(),
        icon:'/images/android/search.png',
        width:64,
        title:'Search'
    });

    var tab5 = Titanium.UI.createTab
    ({          
        window:Gees.ui.createHighlightWindow(),
        icon:'/images/android/highlight.png',
        width:64,
        title:'Highlight'
    });


    //  add tabs
    tabGroup.addTab(tab1);
    tabGroup.addTab(tab2);
    tabGroup.addTab(tab3);
    tabGroup.addTab(tab4);
    tabGroup.addTab(tab5);

    if(Gees.isAndroid()){
        tabGroup.open();
    } else {
        tabGroup.open
        ({          
            transition: Titanium.UI.iPhone && Titanium.UI.iPhone.AnimationStyle.NONE

   };
