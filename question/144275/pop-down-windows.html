pop down windows

basically,
I open windows when someone clicks on a button
and when the window opens, its contents pop down. like the contents start at the top then move down a bit then move back up.
How can I prevent it from doing so when I open new windows?
I am opening windows as follows

function openWindow(url, event)
{
    var cwin = Titanium.UI.currentWindow;
    var swin    = Titanium.UI.createWindow();

    swin.url            = url;

    if(event.zipcode)
        swin.zipcode        = event.zipcode;

    if(event.user_id)
        swin.user_id        = event.user_id;

    if(event.service_id)
        swin.service_id     = event.service_id;

    if(event.service_name)
        swin.service_name    = event.service_name;

    if(event.user_uniqid)
        swin.user_uniqid    = event.user_uniqid;

    if(event.user_name)
        swin.user_name      = event.user_name;

    if(event.user_email)
        swin.user_email     = event.user_email;  

    swin.backgroundColor = '#f7f7f7';

    swin._parent = cwin;
    swin.navGroup = cwin.navGroup;
    swin.rootWindow = cwin.rootWindow;
    cwin.navGroup.open(swin);
}

and I use scrollView to add all the items to a scroll view in each window

var scrollView = Titanium.UI.createScrollView({
    contentWidth:'auto',
    contentHeight:'auto',
    top:0,
    layout: 'vertical',
    showVerticalScrollIndicator:true,
    showHorizontalScrollIndicator:false
});
win.add(scrollView);

Titanium.UI.iPhone.setStatusBarStyle(Titanium.UI.iPhone.StatusBar.OPAQUE_BLACK);

var main = Titanium.UI.createWindow({
    title:'Example',
    fullscreen:false, 
    navBarHidden : true,
});

var first = Ti.UI.createWindow({
    title:'Example',
    url:'main_windows/main.js',
    backButtonTitleImage : '../images/backb.gif',
    fullscreen: false,
    navBarHidden : true,
});

var navGroup = Ti.UI.iPhone.createNavigationGroup({
    window:first
});

Ti.App.addEventListener('login', function(event)
{
    var lwin = Ti.UI.createWindow({
        height:Ti.Platform.displayCaps.platformHeight,  
        width:Ti.Platform.displayCaps.platformWidth,  
        url:'main_windows/login.js',
        barImage: 'images/lback.gif', 
        fullscreen:true, 
        backgroundColor: '#f7f7f7',
        navBarHidden : false,
        _parent : Titanium.UI.currentWindow,
        navGroup : navGroup,
        rootWindow : first 
    });

    navGroup.open(lwin);
});

Ti.App.addEventListener('signup', function(event)
{

    var rwin = Ti.UI.createWindow({
        height:Ti.Platform.displayCaps.platformHeight,  
        width:Ti.Platform.displayCaps.platformWidth,  
        url:'main_windows/register.js',
        barImage: 'images/sback.gif', 
        backgroundColor: '#f7f7f7',
        fullscreen:true, 
        _parent : Titanium.UI.currentWindow,
        navGroup : navGroup,
        rootWindow : first, 
        navBarHidden : false
    });
    navGroup.open(rwin);
});




Ti.App.addEventListener('zipcode', function(event)
{
    zwin.url         = 'zipcode.js';
    zwin.user_id     = event.user_id;
    zwin.user_name     = event.user_name;
    zwin.user_email = event.user_email;  
    navGroup.open(zwin);
});

    first.navGroup = navGroup;

    main.add(navGroup);

    main.open();
