Simulate TableView row click with fireEvent

Hi;

My codes as below. But fireEvent doesn't work. Where is the wrong?

$.TableViewNavigator.addEventListener('click', FncNavigatorTableViewRowsClick);

function FncNavigatorTableViewRowsClick(e) {

    var Rows = $.TableViewNavigator.data[0].rows;

    for (x in Rows) {
        Rows[x].backgroundColor = "transparent";
        Rows[x].selectionStyle = "none";
        Rows[x].selectedBackgroundColor = "transparent";
    }

    e.row.backgroundColor = "transparent";

    .....

}

$.TableViewNavigator.fireEvent("click", {
            data: 2
        });

Thank you in advance
