Reusing views in windows

I am building an app that uses the standard convention of tabGroups loading different windows with different .js files.

I have a standardised toolbar created using a view with nested views and eventListners.

When i add the view to one window it works fine if i add it to multiple windows it unloads it from previous windows.

I don't want to duplicate all that code. any ideas?

var winHome = Titanium.UI.createWindow({  
    navBarHidden:true,
    backgroundColor:'#5c5c5c',
    url:'home.js',
});


var winMATS = Titanium.UI.createWindow({  
    navBarHidden:true,
    backgroundColor:'#fff',
    url:'mats.js'
});

var winService = Titanium.UI.createWindow({  
    navBarHidden:true,
    backgroundColor:'#fff',
    url:'service.js'
});

var view = Titanium.UI.createView({
   backgroundImage:'img/menubg.png',
   width:'100%',
   height:54,
   top:0,
   left:0,
   zIndex:5
});

var topInfo = Titanium.UI.createView({  
   backgroundImage:'img/icons/info.png',
   width:30,
   height:31,
   top:10,
   left:10,
});

var topLogo = Titanium.UI.createView({  
   backgroundImage:'img/logo.png',
   width:115,
   height:42,
   top:4,
   center:true,
});

var topPhone = Titanium.UI.createView({  
   backgroundImage:'img/icons/tel.png',
   width:30,
   height:31,
   top:10,
   right:10,
});

var topBack = Titanium.UI.createView({  
   backgroundImage:'img/back_off.png',
   width:50,
   height:29,
   top:14,
   left:7,
   visible:false
});

view.add(topInfo);
view.add(topLogo);
view.add(topPhone);
view.add(topBack);

winHome.add(view);
winMATS.add(view);
winService.add(view);


topInfo.addEventListener('click', function(e){
    topInfo.visible=false;
    topBack.visible=true;
});

topBack.addEventListener('click', function(e){
    topInfo.visible=true;
    topBack.visible=false;
});

  • Write a function that creates components: 

    //instead of this:
var myView = Ti.UI.createView({ backgroundColor: '#00f' });

//write a function that creates views
function BlueView() {
    return Ti.UI.createView({ backgroundColor: '#00f' });
}

var myView = BlueView();
//now you can reuse if:
var myAnotherView = BlueView();
