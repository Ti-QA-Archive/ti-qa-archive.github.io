TableViewRow on Android, auto height not working

Screenshot Posted Here

var aRow = Ti.UI.createTableViewRow({
    height:'auto',
    width:'100%' ,
    className:'@row1',
    layout:'vertical'
});

var firstRow = Ti.UI.createLabel({
    text: "This is the first row of the text and when it wraps it becomes a problem and there becomes overlapping issues",
    height:'auto',
    width:220,
    left:80,
    top:2,
    textAlign: "left",
    fontWeight:'bold',
    font:{fontSize:14, fontWeight:'bold'}
});

var secondRow = Ti.UI.createLabel({
    text: "the second row of text",
    height:'auto',
    width:220,
    left:80,
    textAlign: "left",
});

aRow.add(firstRow);
aRow.add(secondRow);


var categoriesList = Titanium.UI.createTableView({
    data:[aRow],
    allowsSelection:true
});

I know there is a workaround, based on vertical_layout_table_view.js in kitchen sink, but I was looking for a solution where "auto" would work. I have to believe there is some performance issue with creating an extra control for every row

    I have tested the following code in Titanium 1.7.0 (2011/05/27 17:54 be59d3…) on Android 2.2, and have not observed an issue.

    var win1 = Titanium.UI.createWindow({
  backgroundColor:'white',
  exitOnClose:true,
  title:'win1: Main Window'
});

win1.open();

var aRow = Ti.UI.createTableViewRow({
  height:'auto',
  width:'100%' ,
  className:'@row1',
  layout:'vertical'
});

var firstRow = Ti.UI.createLabel({
  text: "This is the first row of the text and when it wraps it becomes a problem and there becomes overlapping issues",
  height:'auto',
  width:220,
  left:80,
  top:2,
  textAlign: "left",
  fontWeight:'bold',
  font:{fontSize:14, fontWeight:'bold'}
});

var secondRow = Ti.UI.createLabel({
  text: "the second row of text",
  height:'auto',
  width:220,
  left:80,
  textAlign: "left"
});

aRow.add(firstRow);
aRow.add(secondRow);


var categoriesList = Titanium.UI.createTableView({
  data:[aRow],
  allowsSelection:true
});

win1.add(categoriesList);
