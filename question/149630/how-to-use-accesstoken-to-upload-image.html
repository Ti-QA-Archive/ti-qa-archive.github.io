  • Nadeem Here is the example of user authentication in Api it resembles to your problem maybe it help you

    
    var doLogin = function(){
        //alert(userField.value + ' ' + passField.value);
        userField.blur();
        passField.blur();
        if(Titanium.Network.networkType != Titanium.Network.NETWORK_NONE){
            var activityIndicator = Ti.UI.createActivityIndicator({
              color:'#000',
              font: {fontFamily:'Helvetica Neue', fontSize:26, fontWeight:'bold'},
              message: 'Please wait...',
              style:Ti.UI.iPhone.ActivityIndicatorStyle.DARK,
              top:'70dp',
              height:'auto',
              width:'auto'
            });

            window.add(activityIndicator);
            activityIndicator.show();
            //get the data
            var url = "http://login.expressionguestregistry.com/api/login.json";
            //var url = "http://staging.epitech.mercury.stellarbluewebdesign.com/api/login.json";
            var xhr = Ti.Network.createHTTPClient({
                onload: function(e) {
                    // this function is called when data is returned from the server and available for use
                    // this.responseText holds the raw text return of the message (used for text/JSON)
                    // this.responseXML holds any returned XML (including SOAP)
                    // this.responseData holds any returned binary data
                    Ti.API.debug(this.responseText);
                    activityIndicator.hide();

                    var response = JSON.parse(this.responseText);

                    if(response.success){
                        //alert('logged in');
                        //open events selection
                        logoutPin = passField.value;
                        Ti.App.Properties.setString('password',logoutPin);
                        passField.value = '';
                        Window = require('ui/tablet/EventsWindow');
                        new Window().open();
                    } else {
                        alert('Invalid username / password supplied.');
                    }

                    //alert(response);
                },
                onerror: function(e) {
                    // this function is called when an error occurs, including a timeout
                    Ti.API.debug(e.error);
                    activityIndicator.hide();
                    alert('Could not connect to server.');
                },
                enableKeepAlive:false,
                timeout:5000  /* in milliseconds */
            });
            xhr.open("POST", url);
            var postdata = {};
            postdata.username = userField.value;
            postdata.password = passField.value;
            xhr.send(postdata);  // request is actually sent with this statement
        } else {
            //offline mode
            if(passField.value  == Ti.App.Properties.getString('password')){
                logoutPin = passField.value;
                passField.value = '';
                Window = require('ui/tablet/EventsWindow');
                new Window().open();
            } else {
                alert('Invalid username / password supplied.');
            }
        }
    };
