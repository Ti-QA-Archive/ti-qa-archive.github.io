    Hello Sabari,

    Check below code.

    
var win = Ti.UI.createWindow({
    backgroundColor : "white"
});

var txtFld = Ti.UI.createTextField({
    height : 40,
    top : 50,
    borderColor : "gray",
    borderWidth : 1,
    borderRadius : 5,
    width : 200
});

win.add(txtFld);

var btn = Ti.UI.createButton({
    title : 'set & get Value',
    height : 100,
    width : 200,
    top : 150,
});
// Listen for click events.
btn.addEventListener('click', function() {
    var val = txtFld.value;
    Ti.App.Properties.setString('Number', val);
    var getVal = Ti.App.Properties.getString('Number');
    alert('get>>>>' + getVal);
});

win.add(btn);

win.open();

    for more information see this *Titanium.App.Properties.
