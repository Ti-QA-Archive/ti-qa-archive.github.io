  • Hello User,

    You can track the progress by using "ondatastream" property that exist in Ti.Network.createHTTPClient object.

    http://docs.appcelerator.com/titanium/latest/#!/api/Titanium.Network.HTTPClient-property-ondatastream

    Code is as follows

    var client = Ti.Network.createHTTPClient({

    onload: function (e) {
},

onerror: function (e) {
},
ondatastream: function (e) {
  Ti.API.info('message - ondatastream');
  //Ti.API.info(e);
  //Ti.API.info(JSON.stringify(e));
  var progressNow = e.progress;
  Ti.API.info(progressNow);
  if (progressNow === 1.0) {
    Ti.API.info('data downloaded successfully');
  }
},
timeout: 100000

    });

    client.open(method, url);

    client.send(payload);
