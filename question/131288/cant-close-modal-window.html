Can't close modal window

I seem to be having an issue on iphone (works on android). when I open a modal window, I can't close it. Here's a simple example:

var appWin = Ti.UI.createWindow({
  backgroundColor: 'red',
  title: 'testApp'
});
var openBtn = Ti.UI.createButton({
  title:'Open'
});
appWin.add(openBtn);

openBtn.addEventListener('click', function(e) {
  var modalWindow = Ti.UI.createWindow({
    modal:true, 
    title:'Modal Window',
    backgroundColor:'#000'
  });
  var closeBtn = Ti.UI.createLabel({
    text:'Return to Main Window', 
    textAlign:'center',
    width:300,
    height:300,
    backgroundColor:'#ccc'
  })
  modalWindow.add(closeBtn);
  modalWindow.open();
  closeBtn.addEventListener(‘click’, function(e){
    modalWindow.close();
  });
});
appWin.open();

I may just be hitting the wall and my brain is mush, but I could sure use some help now. Thank you and have a good weekend (if you haven't started yet).

Regards,
