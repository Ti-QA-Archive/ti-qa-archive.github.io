Passing data to new open window

Hi guys,

When user tap on the 'departOnBtn' button, it will open a new window (calendarView.js) and send a variable called text to it.

Any idea how to do it?

Many thanks!

MyTripWindow.js

function MyTripWindow() {
    var win = Ti.UI.createWindow({
        title : "My Trip",
        navBarHidden : false,
        layout : 'vertical'
    });

    var departOnBtn = Ti.UI.createButton({
        title : 'TAP TO CHOOSE DEPARTURE DATE',
        color : '#ffffff', // colour of the text: white
        backgroundColor : '#2c3e50', 
        width : '100%',
        height : '20%',
        font : {
            fontFamily : 'Helvetica Neue',
            fontWeight : 'bold',
            fontSize : 18
        },
        top : '5%'
    });

    win.add(departOnBtn);

    departOnBtn.addEventListener('click', function(e) {

        //Call a export function
        var win = require('calendarView'); //calendarView.js

        var text = 'departure'; // send this variable 

        //Create new instance
        var newCalendarView = new win();

        //Open the register window
        newCalendarView.open({
            activityEnterAnimation : Ti.App.Android.R.anim.slide_in_right,
            activityExitAnimation : Ti.App.Android.R.anim.slide_out_left
        });

    return win;
});

module.exports = MyTripWindow;

calendarView.js

function calendarView() {

    Ti.API.info(text); // gives me departure as a string.

    // create window, label, etc, functions
});

module.exports = calendarView;
