    yourTextfield.addEventListener('blur', function(){
    yourScrollView.scrollTo(0,0);
    yourScrollView.contentHeight = ###; //Set this back to whatever it was before the keyboard added height to the scrollView
});

  • I had the same problem and here is my solution hope it will helps.
    I first set the contentHeight to 480 px but still the same problem the window was scrollable so i changed it to 0 px and it works !!
    here is the my code: 

    var winLogin = Ti.UI.createWindow({
        backgroundColor: '#212121',
           navBarHidden: true,
           tabBarHidden: true  
    });

    var scrollView = Ti.UI.createScrollView({
        layout: 'vertical',
        contentHeight: 0,
        disableBounce: true
    });

    var usernameTextField = Ti.UI.createTextField({
        borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
        color: 'black',
        top: 50, 
        width: '80%', height: '7%',
        autocorrect: false,
        returnKeyType: Titanium.UI.RETURNKEY_DONE,
        hintText: L('email')
    });

    usernameTextField.addEventListener('return', function(e) {
        scrollView.scrollTo(0, 0);
        scrollView.contentHeight = 0;
    });

    var passwordTextField = Ti.UI.createTextField({
        borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
        color: 'black',
        top: 15, 
        width: '80%', height: '7%',
        autocorrect: false,
        passwordMask: true,
        returnKeyType: Titanium.UI.RETURNKEY_DONE,
        hintText: L('password')
    });

    passwordTextField.addEventListener('return', function(e) {
        scrollView.scrollTo(0, 0);
        scrollView.contentHeight = 0;
    });

    scrollView.add(usernameTextField);
    scrollView.add(passwordTextField);    
    winLogin.add(scrollView);
