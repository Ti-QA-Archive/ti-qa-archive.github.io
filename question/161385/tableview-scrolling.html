        onload: function(_data) {
            renderNewRows(_data);
        }
    });

  • Hi Preety Shah!!!

    Please you can try the following code. It may helpful to you.

    var aTableView = Ti.UI.createTableView();
var data = [
    {title:'Row 1'},
    {title:'Row 2'},
    {title:'Row 3'},
    {title:'Row 4'},
    {title:'Row 1'},
    {title:'Row 2'},
    {title:'Row 3'},
    {title:'Row 4'},
    {title:'Row 1'},
    {title:'Row 2'},
    {title:'Row 3'},
    {title:'Row 4'}
];
aTableView.setData(data);
aTableView.addEventListener('scrollend', function(e) {
    loadData();
    aTableView.setData(data);
});
function loadData()
{
    for (var i=0; i <5; i++) 
    {
      data.push(
          {
              title:"Amimul--"+i,
      });
    }

}
