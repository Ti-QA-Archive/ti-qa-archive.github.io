  • @Adam: Here I am including the code.. Please suggest something..

    //app.js

    Titanium.UI.setBackgroundColor('#000');

// Create a window
var win = Titanium.UI.createWindow({
    title:'Web Test',
    backgroundColor:'#fff'
});

var webview = Titanium.UI.createWebView({
    url: 'test3.html',
    top: 42,
    left: 0,
    width: 310,
    height: 275
});
win.add(webview);


Ti.App.addEventListener('webviewToTi', function(e){
    alert('I am in app.js: ' + e.id);
});

win.open();

    //test3.html

    <html>
<head>
 Capturing Click Events From WebView
</head>
<body>
Hello

        //alert('I am in webview');
        Titanium.App.fireEvent('webviewToTi', {id:0});

bye
</body>
</html>
