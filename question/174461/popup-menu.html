    • already I tried with that but when add more of three options closes the app.
      This is my code:

      Ti.UI.setBackgroundColor('white');
var win = Ti.UI.createWindow({
    title:'Pruebas de menu',
    backgroundColor:'white',
    exitOnClose:true,
    fullscreen:false
});



var   button2=Ti.UI.createButton({
       title:'Menu 2',
    borderRadius:80,
    backgroundColor:'black',
    borderColor:'blue',
    borderWidth:5,
    font:{fontSize:10, fontFamily:'arial'},
    animatedCenter:{x:100,y:200},
    width:50,
    height:50,
    left:30,
    top:30,
    color:'white'
   });


var    ops2={
        cancel:1,
   options:['Ventana1','Ventana2','Ventana3,'Ventana4','Ventana5'],
        selectedIndex:0,
        destructive:2,
        title:'Menu'
    };

button2.addEventListener('longpress',function(e){
    var dialog2 =Ti.UI.createOptionDialog(ops2).show();
});
win.add(button2);
win.open();
