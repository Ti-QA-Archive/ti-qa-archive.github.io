  • Hi Erik Eisinga,

    I have seen your code . You should follow the coding pattern of ScrollView at first you should add this and then you can add one or more view this ScrollView then you can add any item to the view. After all you just add this ScrollView to the window. You just follow this code i think i will be helpful for you.

    
var basicWin = Ti.UI.createWindow({
    backgroundColor : '#f2f2f2'
});

var contentWindow = Ti.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    top : 0,
    showVerticalScrollIndicator : true,
    showHorizontalScrollIndicator : true
});

var loginScreen = Ti.UI.createView({
    backgroundColor : '#336699',
    borderRadius : 10,
    width : 300,
    height : 1000,
    top : 10,
    layout : 'vertical'
});
contentWindow.add(loginScreen);

var headView = Ti.UI.createView({
    backgroundColor : '#e94f36',
    top : 0,
    width : '100%',
    height : 50,
});

var headViewLabel = Ti.UI.createLabel({
    text : 'Test',
    font : {
        fontFamily : 'Open Sans',
        fontSize : 16
    },
    color : '#ffffff',
    top : '10',
});
headView.add(headViewLabel);
loginScreen.add(headView);

var headViewGradient = Ti.UI.createView({
    backgroundColor : 'blue',
    backgroundRepeat : true,
    top : 10,
    width : '100%',
    height : 70,
});
loginScreen.add(headViewGradient);

var btn = Ti.UI.createButton({
    top : 20,
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
    title : 'Aanmelden',
    font : {
        fontFamily : 'Open Sans',
        fontWeight : 'bold',
        color : '#ffffff',
        fontSize : 16
    },
});

btn.addEventListener('click', function(e) {
    alert('you click Aanmelden');
});
headViewGradient.add(btn);

basicWin.add(contentWindow);

basicWin.open();
