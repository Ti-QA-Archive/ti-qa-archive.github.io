// getting device token
function getDeviceToken(){
    Titanium.Network.registerForPushNotifications({
        types: [
            Titanium.Network.NOTIFICATION_TYPE_BADGE,
            Titanium.Network.NOTIFICATION_TYPE_ALERT,
            Titanium.Network.NOTIFICATION_TYPE_SOUND
        ],
    success:function(e)
    {
        deviceToken = e.deviceToken;
        glb_deviceToken = deviceToken;
        registerForPush();
    },
    error:function(e)
    {
        alert("Error getting Device Token: "+e.message);
    },
    callback:function(e)
    {
        ui.displayDynaMeetNotice(e);
    }
    });
}

// register for push notification on cloud server
function registerForPush(){
    Cloud.PushNotifications.subscribe({
        channel: 'dynameet_'+deviceToken,
        type:'ios',
        device_token: deviceToken
    }, function (e) {
        if (e.success) {
           // alert('Success2 :'+((e.error && e.message) || JSON.stringify(e)));
        } else {
            alert('Error registering device for Push Notifications:' + ((e.error && e.message) || JSON.stringify(e)));
        }
    });
}



function addBreadCrumb(strContent){
    var content = strContent;

    Ti.API.info("index.js : Adding "+content+" to bread crumb");
    breadCrumb.push(content);
    Ti.API.info("index.js : Bread Crumb :"+JSON.stringify(breadCrumb));

}

function getLastBreadCrumb(){
    var lastContent;

    Ti.API.info("index.js : Removing element from :"+JSON.stringify(breadCrumb));
    breadCrumb.pop();
    lastContent = breadCrumb[breadCrumb.length - 1];
    Ti.API.info("index.js : Getting last element which is :"+lastContent);

    return lastContent;

}
