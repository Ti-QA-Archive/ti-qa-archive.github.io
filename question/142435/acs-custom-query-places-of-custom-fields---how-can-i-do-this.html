ACS Custom Query Places of Custom Fields - how can I do this?

App Type: Mobile
Titanium SDK: 2.12

I'm trying to run a custom query on a custom field on a places database I'm building. I'm not sure how to structure my query using Titanium.

Here's my attempt

Cloud.Places.query({
    where : {
        custom_fields : {
            googleID : placesArray[e.source.rowIndex].pID
        }
    }
}, function(e) {
    if (e.success) {
        if (e.places.length == 0) {

            // No Google Places matches so this is a new place
            Ti.API.info("No match on place query");
        } else {

            // We have a Google Place match so ask user to check in
            Ti.API.info("We have a match on place: " + placesArray[e.source.rowIndex].pName);
        };
    } else if (e.error) {
        Ti.API.info("Error in query! Meow!");
    }
});

However the query constantly returns no matches - when I've checked manually and there should be

All help appreciated - I think its just a query syntax snafu

Thx,
Ken

    Cloud.Places.query({
    where : {
          'googleID' : placesArray[e.source.rowIndex].pID
    }
}, function(e) {
    if (e.success) {
        if (e.places.length == 0) {

            // No Google Places matches so this is a new place
            Ti.API.info("No match on place query");
        } else {

            // We have a Google Place match so ask user to check in
            Ti.API.info("We have a match on place: " + placesArray[e.source.rowIndex].pName);
        };
    } else if (e.error) {
        Ti.API.info("Error in query! Meow!");
    }
});
