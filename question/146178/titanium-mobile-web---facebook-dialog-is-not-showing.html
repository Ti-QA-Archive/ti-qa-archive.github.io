Titanium Mobile Web - Facebook dialog is not showing.

Titanium SDK : 2.1.4
Platform : Mobile web .

I want the functionality of login with facebook in my web app. But the following code is not working for me.

Titanium.Facebook.appid = "304324776345833";

var win = Titanium.UI.createWindow({
title : 'Tab 1',
backgroundColor : '#fff'
});

var btn = Ti.UI.createButton({
height : '58dp',
width : '318dp',
title:'Login'
});
btn.addEventListener('click', function() {

if (Ti.Network.online) {

    if (Ti.Facebook.loggedIn == false)
        Ti.Facebook.authorize();
    else
        alert("Already logged In.");

} else
    alert("No internet !!.");

});

win.add(btn);

win.open();

Please help !!

    If I recall correctly, the Facebook App ID is tied to the domain name. If you're testing your app on localhost or something, it won't work. You'll probably need to create multiple App IDs for various test and production domains.
