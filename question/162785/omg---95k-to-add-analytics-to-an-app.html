    • analytics.js (place in lib)

      var GA = require('analytics.google');
//GA.optOut = true;
GA.debug = true;
GA.trackUncaughtExceptions = true;
var tracker = GA.getTracker("{(your_ga_id_here}}");
module.exports = tracker;

      Then in the controller of the file you want to track stuff in:

      var tracker = require('analytics');

$.myWindow.addEventListener('open', function(e){
 tracker.trackScreen('My Window');
}
