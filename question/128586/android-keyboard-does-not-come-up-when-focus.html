    • Hi Nikunj
      Here is my code
      Textfield

      view.textfield = Ti.UI.createTextField({
            //hintText : Text('stock_input_hint'),
            hintText : viewParam.hintTxt,
            top : '2%',
            left : '2%',
            right : '2%',
            height : '10%',
            //backgroundImage : Ti.App.Dir + 'img/controls/inputfield.png',
            //paddingLeft : 20,
            //clearOnEdit : true, ***IOS only***
            keyboardType : Titanium.UI.KEYBOARD_DEFAULT,
            returnKeyType : Titanium.UI.RETURNKEY_DONE,
            autocorrect : false,
            borderStyle : Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
            //softKeyboardOnFocus : Titanium.UI.Android.SOFT_KEYBOARD_SHOW_ON_FOCUS
            // keyboardType : Titanium.UI.KEYBOARD_NUMBERS_PUNCTUATION,
            // returnKeyType : Titanium.UI.RETURNKEY_DONE,
            //clearButtonMode : Ti.UI.INPUT_BUTTONMODE_ONFOCUS
        });

      Event

      myBatchDetail.UI.table.addEventListener('click', function(e) {
        var rowObj = e.rowData;
            myBatchDetail.qtyUpdateRow = rowObj;
        if(e.x < 50)
        {
            myBatchDetail.status = 'update';
            mySTKViewFactory.chooseKeyBoard(myBatchDetail.UI.textfield,'numeric');
            myBatchDetail.UI.textfield.focus();
        }
});

      SDK:1.7.5
      Android:2.3.6
