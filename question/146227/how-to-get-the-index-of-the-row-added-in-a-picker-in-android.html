    • Hi Anuj,

      i was thinking again about your question and my first answer.
      In my first answer (code) the change event gets fired if the user selects another row.

      But i assume now, you want to get the index-value of the newly added row if you are adding a new row programmatically.

      So there seems to be an issue with Android and the change event.

      As a workaround you can use the picker.columns[0].rowCount as i implemented in this new code.

      This works on Android and iOS.

      Ti.UI.setBackgroundColor('#000');
var win = Ti.UI.createWindow({
    backgroundColor : '#ffffff',
    height : Ti.UI.FILL,
    title : 'Test',
    width : Ti.UI.FILL
});

var picker = Ti.UI.createPicker({
    top : 10,
    selectionIndicator : true
});

picker.add(Ti.UI.createPickerRow({
    title : 'manufacturer1'
}));

picker.add(Ti.UI.createPickerRow({
    title : 'manufacturer2'
}));

picker.addEventListener('change', function(e) {
    alert('change index: ' + e.rowIndex );
});

win.add(picker);

button = Ti.UI.createButton({
    top : 300,
    title : 'add row'
})
button.addEventListener('click', function(e) {
    // count rows before adding new one
    // index = rowCount - 1
    var rowCount = picker.columns[0].rowCount;

    // adding a new row
    picker.add(Ti.UI.createPickerRow({
        title : 'manufacturer' + (rowCount + 1)
    }));
    label.text = 'index for new row = ' + rowCount;
});

win.add(button);

label = Ti.UI.createLabel({
    top:350,
    width:'auto',
    textAlign: Ti.UI.TEXT_ALIGNMENT_LEFT,
    color: '#900'
})

win.add(label);

win.open();

      Hope that helps,

      Matthias

  • You have to set picker row while adding row to the picker to get row I index because picker selection is prereq for calling picker change . 

    var win =Ti.UI.createWindow({ backgroundColor:'white'});


var data1 = [];

            data1[0]=Ti.UI.createPickerRow({title:'ABC'});
            data1[1]=Ti.UI.createPickerRow({title:'DEF'});
            data1[2]=Ti.UI.createPickerRow({title:'GHI'});
            data1[3]=Ti.UI.createPickerRow({title:'JKL'});


var picker1 = Ti.UI.createPicker();

            picker1.selectionIndicator = true;

            //get counter of picker data[] 
            var counter= data1.length;
            picker1.add(data1);

win.add(picker1);



            // Submit button //
            var subButton = Ti.UI.createButton({
                title : 'Add Picker Row',
                height : Ti.UI.SIZE,
                width :  200,
                top : 20,
            });


              picker1.addEventListener('change', function(e){ 
                  alert(e.rowIndex);

           });


            subButton.addEventListener('click', function() {

              //counter++;

             picker1.add(Ti.UI.createPickerRow({
                title : 'row #'+counter
             }) );

             // selecting picker row having only one colum
             picker1.setSelectedRow(0,counter,true);

             counter++;
             //alert(counter);


           });

win.add(subButton);
win.open();
