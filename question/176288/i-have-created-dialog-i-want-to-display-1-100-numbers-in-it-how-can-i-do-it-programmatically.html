I have created dialog. I want to display 1-100 numbers in it. How can I do it programmatically?

 var genre = [
    '1',
    '2',
    '3',
    '4'..............100
];

var dialog = Ti.UI.createOptionDialog({
    options: genre,
    title: 'Pick a number'
});
dialog.addEventListener('click', function(evt)
{
    alert('You picked ' + genre[evt.index]);
});
dialog.show();

    oh,

    var genre = [];

for (var x=1;x<101;x++){
    genre.push(x):
} 

var dialog = Ti.UI.createOptionDialog({
    options: genre,
    title: 'Pick a number'
});
dialog.addEventListener('click', function(evt)
{
    alert('You picked ' + genre[evt.index]);
});
dialog.show();
