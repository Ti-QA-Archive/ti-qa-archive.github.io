  • There are a few errors. You used 'createWebview' instead of 'createWebView' and 'open(a)' instead of 'open(anim)'. Now works fine.

    
var webwindow = Titanium.UI.createWindow({
    height:0,
    bottom:0
});
var anim = Titanium.UI.createAnimation();
    anim.height = 420;
    anim.duration = 300;

var webview = Titanium.UI.createWebView({
    url:'http://cuv.acura.ca/inventory/mobile/search?province_id=2',
    height:'auto',
    fullscreen: true,
    navBarHidden: false        
});
webwindow.add(webview);
webwindow.open(anim)
