    You need to use brackets (square) around the feedReqList index rather than parenthesis (round). See the following improved code:

    var CustomData = [];
for (var j=0, jlen=feedReqList.length;j<jlen ;j++){
    CustomData.push(
        {
            tit:feedReqList.item[j].getElementsByTagName("title").item(0).text, 
            des:feedReqList.item[j].getElementsByTagName("description").item(0).text, 
            hasChild:true
        }
    );
// alert(CustomData[0].des);
    alert(CustomData.length);
}

    Cheers
