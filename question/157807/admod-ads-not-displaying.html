  • Hi Madusanka,
    you can test the functionality with this code:

    Admob.addEventListener('didReceiveAd', function() {
    alert('Did receive ad!');
});
Admob.addEventListener('didFailToReceiveAd', function() {
    alert('Failed to receive ad!');
});
Admob.addEventListener('willPresentScreen', function() {
    alert('Presenting screen!');
});
Admob.addEventListener('willDismissScreen', function() {
    alert('Dismissing screen!');
});
Admob.addEventListener('didDismissScreen', function() {
    alert('Dismissed screen!');
});
Admob.addEventListener('willLeaveApplication', function() {
    alert('Leaving the app!');
});
