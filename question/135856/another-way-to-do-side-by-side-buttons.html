Another Way to Do Side-By-Side Buttons?

I currently have a table and at the bottom I have a Calculate and Reset button side by side using left and right properties of the button and then I placed them in a tableRow. Problem is that they now have the border from the table around them and it looks bad because there is a space in between them (see image below). I had two questions:

  1. Is there any other way to do add the two buttons below the table such that they scrolled with the table. If I create a view and put the buttons inside of it and then place it using top or bottom margins it doesn't scroll with the table. And without top or bottom margins it just sits on top of the table, floating.

  2. Is there a way to add two tableViews or remove that specific border? All questions I have seen answered involve removing the separator of the entire table and that's not something I want to do since the upper table has fields that need to be separated.

The code below is my button definition and adding to the table.

calcButton = Ti.UI.createButton({
    font:{fontSize:17, fontWeight:'bold'},
    height: 45,
    width: '48%',
    left: 0,
    //bottom: 85,
    title: 'Calculate'
});
calcButton.addEventListener('click', formValidator);

resetButton = Ti.UI.createButton({
    font:{fontSize:17, fontWeight:'bold'},
    height: 45,
    width: '48%',
    right: 0,
    //bottom: 85,
    title: 'Reset'
});
resetButton.addEventListener('click', calcReset);

function addButtonRow()
{
    row = Ti.UI.createTableViewRow({
        //style: Ti.UI.iPhone.TableViewStyle.PLAIN,
        height:45,
        backgroundColor: 'transparent',
        borderWidth: '0',
        borderColor: 'transparent',
        borderStyle:Titanium.UI.INPUT_BORDERSTYLE_NONE
    });
    //textFieldsArray[hintName] = textField;

    row.add(calcButton);
    row.add(resetButton);

    row.selectionStyle = Ti.UI.iPhone.TableViewCellSelectionStyle.NONE;
    return row;
}

    • var win = Ti.UI.createWindow({
    backgroundColor : 'red',
});

// create table view data object
var data = [{
    title : 'Alan',
    // hasChild : true,
    // header : 'header'
}, {
    title : 'Alice',
    // hasDetail : true
}, {
    title : 'Alexander',
    // hasCheck : true
}, {
    title : 'Amos'
}, {
    title : 'Alonzo',
    // footer : 'footer'
}];

// create table view
var tableview = Titanium.UI.createTableView({
    data : data
});

var view = Ti.UI.createView({
    height : 40
});

var btn1 = Ti.UI.createButton({
    title : 'Test',
    left : 0,
    width : 100,
    height : 40
});

btn1.addEventListener('click', function(e) {
    alert(' << BUTTON 1 >> ');
});

view.add(btn1);

var btn2 = Ti.UI.createButton({
    title : 'Test',
    right : 0,
    width : 100,
    height : 40
});

btn2.addEventListener('click', function(e) {
    alert(' << BUTTON 2 >> ');
});

view.add(btn2);
tableview.footerView = view;
win.add(tableview);
