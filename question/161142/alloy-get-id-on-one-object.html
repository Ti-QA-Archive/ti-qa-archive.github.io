Alloy get id on one object

I am trying to return one object from a collection, here is what I have tried:

model

    config: {
        "columns": {
            "item":"String",
                "ident":"integer",
        },
        "adapter": {
            "type": "sql",
            "collection_name": "entries"
        }
    },

controller

var entries = Alloy.Collections.entries;
//doesn't work
var model2 = $model.get({ident:1});
//doesn't work
var model3 = entries.get({ident:1});

  • Use idAttribute to tell the collection which field to use as an id. Change your code to,

    model

    config: {
        "columns": {
            "item":"String",
                "ident":"integer",
        },
        "adapter": {
            "type": "sql",
            "collection_name": "entries"
        "idAttribute": "ident"
        }
    },

    controller

    var entries = Alloy.Collections.entries;
var model = entries.get(1);
