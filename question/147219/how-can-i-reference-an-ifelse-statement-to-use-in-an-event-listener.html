    so you can wrap that in a function and call that function from inside of event listener.

    function ifElseFunction(f){

if (f == "mmHg") {
        var p = 1;
    } else if (f == "Pascal") {
        var p = 0.00750063755419211;
    } else if (f == "atm") {
        var p = 760.002100178515;
    } else if (f == "bar") {
        var p = 750.063755419211;
    } else if (f == "cmH2O") {
        var p = 0.735561538478802;
    } else if (f == "cmHg") {
        var p = 10;
    } else if (f == "ftH2O") {
        var p = 22.4199156928339;
    } else if (f == "gm/sqcm") {
        var p = 0.735561538478802;
    } else if (f == "inH2O") {
        var p = 1.86832630773616;
    } else if (f == "inHg") {
        var p = 25.4000840071406;
    } else if (f == "kPa") {
        var p = 7.50063755419211;
    } else if (f == "mbar") {
        var p = 0.750063755419211;
    } else if (f == "psi") {
        var p = 51.7150957831416;
    } else {
        var p = 1;
    }

return p;
}

    now use it in eventListener.

    buttonSub.object.addEventListener("click", function(e) {
    var p = ifElseFunction("mmHg");
    Ti.API.info(p)
