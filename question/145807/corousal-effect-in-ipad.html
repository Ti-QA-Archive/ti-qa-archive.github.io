    • This code will be a little lengthy but it will help you customize the carousel effects easily.. Hope it is useful! :)

      var win = Titanium.UI.currentWindow;

//create a view
var view=Titanium.UI.createView();
win.add(view)

//position the 7 image views in U shape
var image1=Titanium.UI.createImageView
({
    top:10,
    left:10,
}); 
view.add(image1);

var image2=Titanium.UI.createImageView
({
    top:100,
    left:10,
}); 
view.add(image2);

var image3=Titanium.UI.createImageView
({
    top:200,
    left:10,
}); 
view.add(image3);

//image at the center of the U 
var image=Titanium.UI.createImageView
({
    top:300,
}); 
view.add(image4);

//similarly create imagesViews on the right to complete the U shape

ani_down1=Titanium.UI.createAnimation();
ani_down1.bottom=100;
ani_down1.duration = 400;

ani_down2=Titanium.UI.createAnimation();
ani_down2.bottom=200;
ani_down2.duration = 400;

//similarly create animations for shifting the views up
ani_up1=Titanium.UI.createAnimation();
ani_up1.duration = 400;
.
.
.
ani_center=Titanium.UI.createAnimation();
ani_center.bottom=300;
ani_center.duration = 400;

Ti.App.flag=1;  //to create 7 cases for 7 different positions of each image

view.addEventListener('swipe', function(e)
{

  if (e.direction == 'left')                    
   {
    if(Ti.App.flag==1)
    {
        image7.animate(ani_down1);
        image6.animate(ani_down2);
        image5.animate(ani_down3);
        image4.animate(ani_center);
        image3.animate(ani_up3);
        image2.animate(ani_up2);
        image1.animate(ani_up1);
        Ti.App.flag=2;                // dont forget to update the flag!
    }

    else if(Ti.App.flag==2)
    {
        //animate the images now according to their new positions
        Ti.App.flag=3;    
    }
    .
    .
    .
    .
    else        // Ti.App.flag=7
    {
        Ti.App.flag=1;
    }

    }

    else            // right swipe
    {
    exactly opposite animations for the right swipe
    }

});
