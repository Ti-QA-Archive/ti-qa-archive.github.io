  • This solution didnt work for me. I found my own:

    scrollableView.addEventListener('scroll',function(e){
    if (scrollableView.timeout)
    {
    clearTimeout(scrollableView.timeout);
    }
    scrollableView.timeout = setTimeout(function() {
    alert("Stopped!");
    },100);
    });
