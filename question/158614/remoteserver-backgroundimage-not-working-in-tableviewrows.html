remote/server backgroundImage not working in TableViewRows

When I set the backgroundImage property of a TableViewRow, if it's a remote url image not a local file it does not load the image, any workarounds for this issue?

Example bellow:

var win = Ti.UI.createWindow({

}) ;

// Create an array of explicitly defined custom TableViewRows
var tbl_data = [];
for (var i = 0; i < 10; i++) {
    var row = Ti.UI.createTableViewRow({
        backgroundImage:i%2==0?'KS_nav_ui.png':'http://www.origami-instructions.com/images/simple-yatch/thumbnails/11-origami-simple-yatch_11.JPG',
    });
    var label = Ti.UI.createLabel({
        left: 10,
        text: 'Row ' + (i+1)
    });

    row.add(label);
    tbl_data.push(row);
}

// now assign that array to the table's data property to add those objects as rows
var table = Titanium.UI.createTableView({
    data:tbl_data
});

win.add(table);
win.open();
