  • To remove an event you need to provide the event and the function attached to it.
    So you wil have to do something like ( pseudo-code):

    var on_location =  function(){ .... };
Ti.Geolocation.addEventListener('location', on_location);

    and to remove

    Ti.Geolocation.removeEventListener('location', on_location);

         title:'Tab Map',
         fullscreen:false

Ti.Geolocation.preferredProvider = "gps";
var mapview = Titanium.Map.createView({
         mapType: Titanium.Map.STANDARD_TYPE,
         region:{
                  latitude:45.73837,
                  longitude:4.65477,
                  latitudeDelta:0.5,
                  longitudeDelta:0.5},
         animate:true,
         regionFit:true,
         userLocation:true
});
Ti.Geolocation.getCurrentPosition(function(e){});
win.add(mapview);

win.open();
