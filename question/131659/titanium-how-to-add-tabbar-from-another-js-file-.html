  • createTabbar.js

    function createTabbar(params){
return Titanium.UI.iOS.createTabbedBar(params);
}

exports.create = createTabbar;

    then in your firstView.js

    
var factory = require('createTabbar');

var myTabBar = factory.create({});

  • Solved :

    There was a mistake in my code. I was creating new window in every .js file.
    Like :

    var currentWindow = Titanium.UI.createWindow;
 ({
     title:'A List',
     backgroundColor:'white',
     top:0,
     height:416
 });

    instead of 

    var currentWindow = Titanium.UI.currentWindow;
