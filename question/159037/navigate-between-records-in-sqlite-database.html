    For navigation you need to add event handlers. Events will controls your SQL table id and UI also.
    Here I have simple demo code you can use this for you help.

MyUI (recordID,title,location);
function MyUI (recordID,title,location){
    labelrecordID1.text = recordID;
    labelTitle1.text = title;
}
var button_previous = Titanium.UI.createButton({
    title : 'Previous',
    color : 'white',
    backgroundColor : '#464646',

        fontFamily : 'Helvetica Neue'
    },
    bottom : 20,
    right : 10,
    width : 100,
    height : 60
});

currentWin.add(button_next);
button_previous.addEventListener('click', function() {
    var db = Titanium.Database.open('photos');
    alert(id);
    var sql = db.execute('SELECT * FROM photos where id < ? order by id desc limit 1', id);

    var recordID = sql.fieldByName('id');
    var title = sql.fieldByName('title');
    var location = sql.fieldByName('location');

    db.close();

    if(parseInt(recordID)){
    id = recordID;
    alert(recordID + ' ' +title + ' ' +location);
    MyUI (recordID,title,location);
    button_next.show();
    }else{
        button_previous.hide();
    }
});

button_next.addEventListener('click', function() {

    var db = Titanium.Database.open('photos');

    var sql = db.execute('SELECT * FROM photos where id > ? order by id asc limit 1', id);

    var recordID =sql.fieldByName('id');
    var title = sql.fieldByName('title');
    var location = sql.fieldByName('location');

    db.close();

    if(parseInt(recordID)){
    id = recordID;
    alert(recordID + ' ' +title + ' ' +location);
    MyUI (recordID,title,location);
    button_previous.show();
    }else{
        button_next.hide();
    }

});
currentWin.open();
