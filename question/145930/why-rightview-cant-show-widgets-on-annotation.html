why rightView cant show widgets on annotation?

hi; I have a sample mapview that need a view in a annotation,
but seems the annotation cant show the rightView correctly on iphone simulator(5.0/5.1);
same code worked properly in android emulator. can anyone tell me the reason?
I'm using the 3.0ga mobilesdk; thanks in advance, and try the code below:


var win = Titanium.UI.createWindow();

var labelPriceIndex = Ti.UI.createLabel({
    text : "Store Price Index:\n0.1234565",
    color : "#f90",
    font : {
        fontWeight : "bold",
        fontSize : "8sp"
    }
});

var viewRight = Ti.UI.createView({
    width: Ti.UI.SIZE,
    height: Ti.UI.SIZE
});
viewRight.add(labelPriceIndex);

var mountainView = Titanium.Map.createAnnotation({
    latitude:37.390749,
    longitude:-122.081651,
    title:"Appcelerator Headquarters",
    subtitle:'Mountain View, CA',
    pincolor:Titanium.Map.ANNOTATION_RED,
    animate:true,
    rightView: viewRight,
    myid:1 // Custom property to uniquely identify this annotation.
});

var mapview = Titanium.Map.createView({
    mapType: Titanium.Map.STANDARD_TYPE,
    region: {latitude:37.390749, longitude:-122.081651, 
            latitudeDelta:0.01, longitudeDelta:0.01},
    animate:true,
    regionFit:true,
    userLocation:true,
    annotations:[mountainView]
});

win.add(mapview);
win.open();

    Hello de huang,

    Try to set height & width of Label…

    see like this…

    var labelPriceIndex = Ti.UI.createLabel({
    text : "Store Price Index:\n0.1234565",
    color : "#f90",
    font : {
        fontWeight : "bold",
        fontSize : "8sp"
    },
    height : 30,
    width : 30,    
    color : 'red',
    backgroundColor : 'pink'
});

    Clean build and run it again…
