Putting view on mapview

I want to put translucent view on a mapview. However I have a trouble. I want to control the mapview from the view (double tap the view to zoom, drag to the view region change).

var win = Ti.UI.createWindow();

var map = Ti.Map.createView({
    animate : true,
    bottom : 0,
    height : Ti.UI.FILL,
    mapType : Ti.Map.STANDARD_TYPE,
    regionFit : true,
    top : 0,
    userLocation : true,
    width : Ti.UI.FILL
});

var view = Ti.UI.createView({
    backgroundColor : "blue",
    height : "200dp",
    width : "200dp",
    opacity : 0.5
});

win.add(map);
win.add(view);

win.open();

  • exports.create= function(){
    var win = Ti.UI.createWindow();
     win.map = Ti.Map.createView({
            animate : true,
            mapType : Ti.Map.STANDARD_TYPE,
            regionFit : true,
            userLocation : true,
            region: {} // you have to fill out
    }); 
     win.controler = Ti.UI.createView({
           height : "200dp",
            width : "200dp",
        });
    win.controler.add(Ti.UI.createView({backgroundColor:'black',opacity:0.6}));    
    win.controler.add(Ti.UI.createLabel({color:'white',text:'test'}));
     win.add(win.map);
    win.add(win.control);
    return win;
}
