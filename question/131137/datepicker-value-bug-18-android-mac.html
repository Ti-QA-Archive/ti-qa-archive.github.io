DatePicker value bug 1.8 android mac

I have an application having a date picker which used to work fine with version 1.7.5 but when i compiled it with V8 on Ti 1.8 it stopped updating its value once changed so when I change the date value and click the button it passes the default date.

var picker_date = Titanium.UI.createPicker({
  type:  Titanium.UI.PICKER_TYPE_DATE,
  value: default_date,
  top:20,
  left:'auto'
});


var button = Titanium.UI.createButton({
    title: ' Calc ',
    top:20,
    height:40,
    right:0,
    left:0
 });

 button.addEventListener('click',function(e)
{
    var d = new Calc(picker_date.value);

});

for it to work again I had to add a change event and a variable




var picker_date = Titanium.UI.createPicker({
  type:  Titanium.UI.PICKER_TYPE_DATE,
  value: default_date,
  top:20,
  left:'auto'
});

var dateVal ;
picker_date.addEventListener('change',function(e){
    dateVal = e.value;
//    Ti.API.info('Date is '+ dateVal);
});

var button = Titanium.UI.createButton({
    title: ' Calc ',
    top:20,
    height:40,
    right:0,
    left:0
 });

 button.addEventListener('click',function(e)
{
    var d = new Calc(dateVal);

});
