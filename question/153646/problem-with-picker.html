Problem with Picker

hello to all guys.I have a problem in taking the present value of a custom picker created in the following way

for (var i = minn; i <= maxx; i++) {
    addRow("" + i + " " + m);
 };

picker.setSelectedRow(0, (vall - minn), true);

function addRow(text) {

            var row = Ti.UI.createPickerRow();
            var label = Ti.UI.createLabel({
                text : text,
                font : {
                    fontSize : 24,
                    fontWeight : 'bold'
                },
                width : 'auto',
                height : 'auto'
            });
            row.add(label);
            picker.add(row);
}

I'd like pressing a button, or after a certain action, could save in a variable the current value of the picker.
I tried in all ways but to no avail
thanks
