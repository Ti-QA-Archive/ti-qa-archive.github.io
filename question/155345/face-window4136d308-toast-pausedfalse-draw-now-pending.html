  • Hi Pradeep

    in http request set Timeout Property

    HTTP.request({
                timeout: 10000,
                type: "GET",
                format: "text",
                url: _params.url,
                passthrough: _params.callback,
                success: this.handleData,
                failure: function(_error) {
                    alert("Unable to connect. Please try again later.");
                }
