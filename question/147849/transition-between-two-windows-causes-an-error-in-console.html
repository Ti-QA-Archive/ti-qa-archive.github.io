  • Hi

    I suspect the problem may rest with the Alloy mark-up as your appears fine. To test this create a new default project and then replace the code in app.js with the code below;

    var win1 = Ti.UI.createWindow({
    backgroundColor: '#fff',
    height: Ti.UI.FILL,
    title: 'Window 1',
    width: Ti.UI.FILL
});
var tab1 = Ti.UI.createTab({
    title: 'Tab 1',
    window: win1
});

var btn = Ti.UI.createButton({
    title: 'Open'
});
win1.add(btn);

btn.addEventListener('click', function (e) {
    var win = Ti.UI.createWindow({
        backgroundColor: 'orange',
        height: Ti.UI.FILL,
        title: 'Window 1',
        width: Ti.UI.FILL
    });
    win.open({ transition: Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT });
});

var tabGroup = Ti.UI.createTabGroup({});
tabGroup.addTab(tab1);
tabGroup.open();

    If pressing the button works for you then provide you Alloy related mark-up and I will take a look.
