  • sure, i used this library for the titanium app https://github.com/iamyellow/socket.io-client

    and the code in the app is 

    var uri = 'ws://127.0.0.1:8080'; 
    var io = require('socket/socket.io'),

    socket = io.connect(uri);
    socket.on('connect', function () {
        Ti.API.log('connected!');
        socket.emit('info',{
            lotto:1, //this is my object

        });

        socket.on('nuova offerta',function(result){
            b.valore = result.valore;
            b.title = 'Offri '+result.valore;
            Ti.API.warn(result.tipo);
        });

    });

    and this is my server (in nodejs)

    
var app = require('express')()
  , server = require('http').createServer(app)
  , io = require('socket.io').listen(server);

server.listen(8080);


app.get('/', function (req, res) {
    console.log('PROVA');
});

app.get('/ciao', function (req, res) {
    io.sockets.on('info',function(data){
            res.send(data);
      });
    res.send(io.sockets.clients('stanza'));
});


io.sockets.on('connection', function (socket) {

  socket.on('newconnect',function (data) {
      console.log(data);
  });


  socket.on('info',function(data){
    console.log(data);
  });

  socket.on('offerta',function(data){
    if(data.lotto){
        nuovoValore = parseInt(data.offerta) + 50;
        socket.emit('nuova offerta',{
            valore:nuovoValore,
            tipo:'direct',
            lotto:data.lotto.id
        });
        socket.broadcast.emit('nuova offerta',{
            valore:nuovoValore,
            tipo:'broadcast',
            lotto:data.lotto.id
        });
    }   
  });

});

    this is only a test
