    Hi

    Thanks for the updated details, it helps you to get better answers.

    The key to opening a new window within the current tab is using the method;

    tab1.open(win, { animated: true });

    Where tab1 is the reference to the tab you are in and win is a reference to the new window you want open - shich you will have created before this line.

    If you are using a an older technique to link separate JavaScript files together namely the url property of the window - let me know and I will adjust the code accordingly.

    To see this code work, create a new simple project and replace the contents of the app.js file with the code you find below.

    Ti.UI.setBackgroundColor('#000');

var tab1, tab2;

(function () {
    var win = Ti.UI.createWindow({
        backgroundColor: '#fff',
        exitOnClose: true,
        title: 'Window 1'
    });
    tab1 = Ti.UI.createTab({  
        icon: 'KS_nav_views.png',
        title: 'Tab 1',
        window: win
    });    
    var btnTest1 = Ti.UI.createButton({
        title: 'Test 1a',
        top: 100
    });
    win.add(btnTest1);
    btnTest1.addEventListener('click', function (e) {
        var win = Ti.UI.createWindow({
            backgroundColor: '#fff',
            title: 'Test 1a'
        });
        var btnAnother = Ti.UI.createButton({
            title: 'Another'
        });
        win.add(btnAnother);
        btnAnother.addEventListener('click', function (e) {
            var win = Ti.UI.createWindow({
                backgroundColor: '#fff',
                title: 'Another'
            });
            tab1.open(win, { animated: true });            
        });
        tab1.open(win, { animated: true });
    });
    var btnTest2 = Ti.UI.createButton({
        title: 'Test 1b',
        top: 200
    });
    win.add(btnTest2);
    btnTest2.addEventListener('click', function (e) {
        var win = Ti.UI.createWindow({
            backgroundColor: '#fff',
            title: 'Test 1b'
        });
        tab1.open(win, { animated: true });
    });
})();

(function () {
    var win = Ti.UI.createWindow({
        backgroundColor: '#fff',
        exitOnClose: true,
        title: 'Window 2'
    });
    tab2 = Ti.UI.createTab({  
        icon: 'KS_nav_views.png',
        title: 'Tab 2',
        window: win
    });    
    var btnTest1 = Ti.UI.createButton({
        title: 'Test 2a',
        top: 100
    });
    win.add(btnTest1);
    btnTest1.addEventListener('click', function (e) {
        var win = Ti.UI.createWindow({
            backgroundColor: '#fff',
            title: 'Test 2a'
        });
        tab2.open(win, { animated: true });
    });
    var btnTest2 = Ti.UI.createButton({
        title: 'Test 2b',
        top: 200
    });
    win.add(btnTest2);
    btnTest2.addEventListener('click', function (e) {
        var win = Ti.UI.createWindow({
            backgroundColor: '#fff',
            title: 'Test 2b'
        });
        tab2.open(win, { animated: true });
    });
})();

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);
tabGroup.addTab(tab2);
tabGroup.open();

    • This should be what you need. Same as before - new project…

      When you use the property url you can access the current tab from within the dependant JavaScript file by this line;

      var tab = Ti.UI.currentTab;

      This property only works when inside a window loaded by the url property.

      Code below ================================================

      app.js

      Ti.UI.setBackgroundColor('#000');

var win1 = Ti.UI.createWindow({
    backgroundColor: '#fff',
    exitOnClose: true,
    title: 'Window 1',
    url: 'win1.js'
});
var tab1 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 1',
    window: win1
});    

var win2 = Ti.UI.createWindow({
    backgroundColor: '#fff',
    exitOnClose: true,
    title: 'Window 2',
    url: 'win2.js'
});
var tab2 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 2',
    window: win2
});    

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);
tabGroup.addTab(tab2);
tabGroup.open();

      win1.js

      var win = Ti.UI.currentWindow;
var tab = Ti.UI.currentTab;

var btnTest = Ti.UI.createButton({
    title: 'Test 1',
    top: 100
});
win.add(btnTest);
btnTest.addEventListener('click', function (e) {
    var win = Ti.UI.createWindow({
        backgroundColor: '#fff',
        title: 'Test 1',
        url: 'new.js'
    });
    tab.open(win, { animated: true });
});

      win2.js

      var win = Ti.UI.currentWindow;
var tab = Ti.UI.currentTab;

var lbl = Ti.UI.createLabel({
    text: 'Window 2'
});
win.add(lbl);

      new.js

      var win = Ti.UI.currentWindow;

var lbl = Ti.UI.createLabel({
    text: 'Hello'
});
win.add(lbl);
