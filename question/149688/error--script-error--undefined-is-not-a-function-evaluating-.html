Trying to call an objects function from another file causes Error. Not sure if I am missing something here.

[ERROR] : Script Error = 'undefined' is not a function (evaluating 'testObject.testObject()') at app.js (line 9).

Below the code, two files "TestObject.js" and "App.js".

If line 9 in App.js is commented, it working fine.

TestObject.js

//TestObject.js file
function TestObject() {

    var label = Titanium.UI.createLabel({
        color:'#999',
        text:'I am label',
        font:{fontSize:20,fontFamily:'Helvetica Neue'},
        textAlign:'center',
        width:'auto'
    });

    var UpdateLabel = function() {
        this.label.setText('I am updated via UpdateLabel function!');
    }

}

module.exports = TestObject;

App.js

// App.js file
var win = Ti.UI.createWindow({
    backgroundColor:'#fff'
});


var TestObject = require('TestObject');

var testObject = new TestObject();
testObject.UpdateLabel();              // line 9 --- Error 'undefined'

win.open();

  • hi,

    try this one

    
~~~
//TestObject.js file
function TestObject() {

    var label = Titanium.UI.createLabel({
        color:'#999',
        text:'I am label',
        font:{fontSize:20,fontFamily:'Helvetica Neue'},
        textAlign:'center',
        width:'auto'
    });

    label.UpdateLabel = function() {
            this.label.setText('I am updated via UpdateLabel function!');
    }

    return label; 

}

module.exports = TestObject;
~~~

```App.js

    // App.js file
var win = Ti.UI.createWindow({
    backgroundColor:'#fff'
});

var TestObject = require('TestObject');

var testObject = new TestObject();
win.add(testObject);

testObject.UpdateLabel();              // line 9 --- Error 'undefined'

win.open();
