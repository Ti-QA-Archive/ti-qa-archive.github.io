Scroll to top an html file of a webview after clicking a button

Today I needed a way to move to top a webview after clicking on the button of the navbar. Thought existed a function type for the scrollToTop webview, but instead does not exist. Then I found this solution that I share with you, if you ever need it ;)

app.js

var win = Ti.UI.createWindow();

var scrollTopBtn = Ti.UI.createButton({
    title : "Top"
});

win.rightNavButton = scrollTopBtn;

var webView = Ti.UI.createWebView({
    height : "100%",
    width : "100%",
    url: "file.html"
)};

scrollTopBtn.addEventListener("click",function(){
    Ti.App.fireEvent("scrollToTop");
});

win.add(webView);
win.open();

file.html

<script>                
    Ti.App.addEventListener("scrollToTop", function(){
        window.scrollTo(0,0)
    });
</script>
