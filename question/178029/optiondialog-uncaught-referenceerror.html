OptionDialog, Uncaught ReferenceError

I'm trying to populate a Tableview in a OptionDialog filling data from an array but i got an error:

Uncaught ReferenceError: regiornArea is not defined

here is my code:

var regionArea = ["A", "B", "C"];
var data = [];

var v1 = Ti.UI.createView({
    backgroundColor : '#212429'
});

var tableView = Ti.UI.createTableView({
    separatorColor : '#111214'
});
v1.add(tableView);

for(var i=0; i < regiornArea.length; i++){
    var Row = Ti.UI.createTableViewRow({
        height : '50dp',
        touchEnabled : false,
        width : '100%',
        backgroundColor : '#212429'
    });
    var label = Ti.UI.createLabel({
        text : regiornArea[i],
        color : 'white',
        touchEnabled : false,
        font : {fontSize : '18dp'}
    });
    Row.add(label);
    data.push(Row);    
}
tableView.setData(data);

var dialog  = Ti.UI.createOptionDialog({
    androidView:v1,
    buttonNames:['Cancel','Get it']
});

$.settingButton.addEventListener('click',function(e){
    dialog.show();
});

what's wrong?
