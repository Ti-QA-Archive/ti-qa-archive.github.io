  • Maybe something like this?

    //->app.js
var win = Ti.UI.createWindow(),
alertDialog = Ti.UI.createAlertDialog({title:'Foo',message:'Bar');
win.open();
setTimeout(function() {
  alertDialog.show();
}, 2000);
alertDialog.addEventListener('click', function() {
  var w = Ti.UI.createWindow({url:'main.js'});
  w.open();
});
