  • I'm not sure about this, but maybe you can write wrapper around "Ti.UI.Window" component (or you can override it).

    
var focusedWindow;
var Window = function(options) {
  var win = Ti.UI.createWindow(options);
  win.addEventListener('focus', function() {
    focusedWindow = win;
  }
};

Window.getFocusedWindow = function() {
  return focusedWindow;
}

    I haven't test it and I'm not sure will it work, but you can give it a try.
