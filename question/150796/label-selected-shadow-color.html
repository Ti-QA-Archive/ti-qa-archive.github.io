Label selected shadow color

I have a tableViewRow, which contains a label. The label has shadowColor and -Offset. If I tap the row, it will highlight, the label adjusts it's color to highlightedColor, but the shadow remains the same. Is there any way to change the color of the shadow when it is selected?

(Ok, I know I can add an event listener on touchDown and Up on every label, but it is very slow (you have to tap for a several ms to get the effect) and very-very-very ugly and unmaintainable in code.)

  • This solution works, but it has to be some better way:

            row.addEventListener('touchstart', function() {
            title.applyProperties({
                shadowColor: null,
                shadowOffset: null,
            });
            date.applyProperties({
                shadowColor: null,
                shadowOffset: null,
            });
            description.applyProperties({
                shadowColor: null,
                shadowOffset: null,
            });
        });

        var end = function() {
            title.applyProperties({
                shadowColor: '#eee',
                shadowOffset: { x: 0, y: 1 },
            });
            date.applyProperties({
                shadowColor: '#eee',
                shadowOffset: { x: 0, y: 1 },
            });
            description.applyProperties({
                shadowColor: '#eee',
                shadowOffset: { x: 0, y: 1 },
            });
        };

        row.addEventListener('touchend', end);
        row.addEventListener('touchcancel', end);
