Table View and adding labels

i am trying to add labels to a table view so the text will wrap around instead of getting cut off.
i have tried the code below and a few other things if some one has done this or has an idea on how to please help me. Thanks in advance!

var r1 = Titanium.UI.createTableViewRow({height:'auto'});
var tableview = Titanium.UI.createTableView({
    height:'auto'
});
var label1 = Ti.UI.createLabel({
  height: 'auto',
  text: 'content of the label'
});
r1.add(label1);
Titanium.UI.currentWindow.add(tableview);

  • try this change

    var label1 = Ti.UI.createLabel({
  height: 'auto',
  width:'100%', //<===
  text: 'content of the label'
});

  • or set the width to 'auto'.

    var row1 = Titanium.UI.createTableViewRow();
row1.height = '280';
row1.backgroundImage='/images/bg2.png'

var image1 = Ti.UI.createImageView({
        image: '/images/teacher3.jpg',
        width: '270'
});

row1.add(image1);

var text1 = Ti.UI.createLabel({
    color:'#000',
    text:'Read Me when... ',
    font:{fontSize:16},
    top:70,
    height:'auto',
    width:'auto',
    textAlign:'center',
    bottom:10

});
row1.add(text1);
