    Hi IVIedia Power,

    You can do it using your custom button on top of the window.

    var btnBack = Ti.UI.createButton({
        backgroundImage : 'Back.png',
        title : 'Back',
        width : 47,
        height : 44,
        left : 0,
        top : 0,
        zIndex : 111
    });
    btnBack.addEventListener('click', function(e) { 
        Ti.UI.currentWindow.close();
    });
Ti.UI.currentWindow.add(btnBack);
