btn_login = Ti.UI.createButton({
    title:"LOGIN",
    bottom: 170,
});

btn_login.addEventListener("click", function(e) {

    c = Titanium.Network.createHTTPClient();
    c.setTimeout(25000);

        svar = JSON.parse(this.responseText);

        Ti.API.info("on error");
        alert(e);

    c.open('GET', 'https://api.parse.com/1/login/');
