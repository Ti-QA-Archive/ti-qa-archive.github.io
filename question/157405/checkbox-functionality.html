    • Sample code :

      var win = Ti.UI.createWindow({
    objName: "mainwindow",
    backgroundColor:"white",
    layout:"vertical"
});

var checkboxArray = [];

for (var i = 0; i<6; i++){
    var checkbox = Ti.UI.createSwitch({
        objName:"checkbox-" + i,
        style:Ti.UI.Android.SWITCH_STYLE_CHECKBOX,
        title:"Sound Enabled",
        value:true
    });

    win.add(checkbox);
    checkboxArray.push(checkbox);

    checkbox.addEventListener("change", function(e){
        Ti.API.info(e.source.objName + " has been set to " + e.source.value);
    });

}

win.addEventListener("click", function(e){
    if(e.source.objName === "mainwindow"){
        for (var i=0, ilen=checkboxArray.length; i<ilen; i++){
            Ti.API.info(checkboxArray[i].objName + " has been set to " + checkboxArray[i].value);
        }
    }
});
win.open();
