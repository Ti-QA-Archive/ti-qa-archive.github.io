URGENT TITANIUM BUG [PARSEINT]

In the code below if you have:

var corner2X = parseInt(corner2.center.x);

the value of corner2X will be 0, even if corner2.center.x = 310.

But if you use var corner2X = parseInt(corner2.center.x)+1+parseInt(corner2.center.x);

the value of corner2X = 311

If instead of 1 you put 0, the value of corner2X is 0

I find this to be a very important BUG in titanium.

I tested this on 1.7.5, 1.6.2, 1.8.0, on all the result is the same.

Who wants to see it in action, copy the code below and watch the INFO sent in the debug, and try the different cases i added above, you will have a laugh :)

I will add this as a ticket.

If anyone has any questions, i will haply answer them.

var win = Titanium.UI.currentWindow;
win.backgroundColor = "#ffffff";


var corner1 = Titanium.UI.createView({
    height:50,
    width:50,
    borderRadius:50,
    backgroundColor:'#336699',
    center: {
        x:10,
        y:10
    }
});

var corner2 = Titanium.UI.createView({
    height:50,
    width:50,
    borderRadius:50,
    backgroundColor:'#336699',
    center: {
        x:310,
        y:10
    }
});





var corner1X = parseInt(corner1.center.x);
var corner1Y = parseInt(corner1.center.y);
var corner2X = parseInt(corner2.center.x)+1+parseInt(corner2.center.x);
var corner2Y = parseInt(corner2.center.y);



var topLine = Titanium.UI.createView({
    backgroundColor:'black',
    height:1,
    left:corner1X,
    top:corner1Y+200,
    width:(corner2X-corner1X),
});

Ti.API.info ("C1X:" + corner1.center.x);
Ti.API.info ("C1Y:" + corner1.center.y);
Ti.API.info ("C2X:" + corner2.center.x);
Ti.API.info ("C2Y:" + corner2.center.y);


Ti.API.info ("C1XI:" + corner1X);
Ti.API.info ("C1YI:" + corner1Y);
Ti.API.info ("C2XI:" + corner2X + " ORIGINAL: " + parseInt(corner2.center.x));
Ti.API.info ("C2YI:" + corner2Y);



Ti.API.info ("toplineH:" + topLine.height);
Ti.API.info ("toplineW:" + topLine.width);
Ti.API.info ("toplineT:" + topLine.top);
Ti.API.info ("toplineL:" + topLine.left);
