i have this code :

if (listLink != null){
    for(var i = 0; i < listLink.length; i++) {
        var rowLink = Ti.UI.createView();
        rowLink.backgroundColor = '#FFF';
        rowLink.backgroundSelectedColor = '#444';
        rowLink.height = 'auto';
        rowLink.top = 10;
        rowLink.width = lebarLayar - 30;
        rowLink.left = lebarLayar / 2 - rowDetail.width / 2;

        rowLink.link = listLink[i];

        var labelLink = Ti.UI.createLabel({
            color : '#000',

    };
}

    if (listLink != null){
    for(var i = 0; i < listLink.length; i++) {
        var rowLink = Ti.UI.createView();
        rowLink.backgroundColor = '#FFF';
        rowLink.backgroundSelectedColor = '#444';
        rowLink.height = 'auto';
        rowLink.top = 10;
        rowLink.width = lebarLayar - 30;
        rowLink.left = lebarLayar / 2 - rowDetail.width / 2;
        rowLink.borderRadius = 10;
        rowLink.borderColor = '#CC9966';
        rowLink.borderWidth = 1;
        rowLink.link = listLink[i];

        var labelLink = Ti.UI.createLabel({
            color : '#000',
            text : listLink[i],
            font : {
                fontSize : 12,
                fontFamily : 'Helvetica Neue'
            },
            top : 5,
            bottom : 5,
            left : 10,

        link:listLink[i],

        rowLink.add(labelLink);

            Ti.Platform.openURL(e.source.link);
        });
}

  • The above code does the right thing. Since its a for loop only the last eventListener will get fired as its declared last.

    You need to create separate views and each view having a distinct eventListener.
