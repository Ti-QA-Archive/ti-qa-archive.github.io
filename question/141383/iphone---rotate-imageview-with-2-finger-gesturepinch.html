//Define empty click counters for Rotate function
var clicksC = 0;

//Rotate image function
function rotateImage(choice, count){
    var t = Ti.UI.create2DMatrix();
    var spin = Titanium.UI.createAnimation();
    if (choice =='clock'){
        var degrees = (-(clicksA *10))+(clicksC *10);
