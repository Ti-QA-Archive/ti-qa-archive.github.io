  • Hi,

    put crewservice_screen.js code in app.js in a function then just call fire event from crewservice_screen.js window with selected tab index like below

    app.js

    
function openTabGroup(selectedIndex){
Titanium.UI.setBackgroundColor('#fff');  
var tabGroup1 = Titanium.UI.createTabGroup();  
var booking = Titanium.UI.createWindow({  

    title:'ASA FILMCREW APP INFORMATION : Bookings',  
    url:'/screens/bookingselection.js'  
});  
var bookingTab = Titanium.UI.createTab({  
    title:"BOOKINGS",  
    icon:"/radio_check/booking_tab.png",
    window:booking  
});  
var rates = Titanium.UI.createWindow({  
    title:'ASA FILMCREW APP INFORMATION : RATES',  
    url:'/screens/rates.js'  
});  
var ratesTab = Titanium.UI.createTab({  
    title:'RATES',  
    icon:"/radio_check/rates_tab.png",
    window:rates  
});  


var abooutUs = Titanium.UI.createWindow({  
    title:'ASA FILMCREW APP INFORMATION : ABOUT US',  
    url:'/screens/aboutus.js'  
});  
var abooutUsTab = Titanium.UI.createTab({  
    title:"ABOUT ASA",  
    icon:"/radio_check/aboutus_tab.png",
    window:abooutUs  
});  
var contactUs = Titanium.UI.createWindow({  
    title:'ASA FILMCREW APP INFORMATION : CONTACT US',  
    url:'/screens/ContactUs_Screen.js'  
});  
var contactUsTab = Titanium.UI.createTab({  
    title:'CONTACT US',  
    icon:"/radio_check/contactus_tab.png",
    window:contactUs  
});  

tabGroup1.addTab(bookingTab);
tabGroup1.addTab(ratesTab); 
tabGroup1.addTab(abooutUsTab);  
tabGroup1.addTab(contactUsTab); 
tabGroup1.setActiveTab(selectedIndex);
tabGroup1.open();
}

Ti.App.addEventListener('openTabView',function(e){
    openTabGroup(e.index);
});

var main = Ti.UI.createWindow({
    url:'/screens/Main_Screen.js',
    height:Ti.Platform.displayCaps.platformHeight,
    width:Ti.Platform.displayCaps.platformWidth,

    • ok, its so simple. just create one function in app.js and open login screen by calling that function and also create one custom add event listener to call from CrewService_Screen.js.

      like

      app.js

      Titanium.UI.setBackgroundColor('#fff');  
function openLoginScreen(){
var main = Ti.UI.createWindow({
    url:'screens/Login_Screen.js',
    height:Ti.Platform.displayCaps.platformHeight,
    width:Ti.Platform.displayCaps.platformWidth,
    fullscreen:false,
    titleControl:false
});
main.open();
}

openLoginScreen();

Ti.App.addEventListener('openLogin',openLoginScreen);

      Login_screen.js

      loginBtn.addEventListener('click', function(e) {
    var main = Ti.UI.createWindow({
        url : '/screens/CrewService_Screen.js',
        height : Ti.Platform.displayCaps.platformHeight,
        width : Ti.Platform.displayCaps.platformWidth,
        fullscreen : false,
        titleControl : false
    });
    main.open();
    win.close();
});

      Crew serivice.js

      Ti.App.addEventListener('closeTabView',function(e){
    Ti.App.fireEvent('openLogin');
    main.close();
    tabGroup1.close();

});
