    Hello Rakesh,

    Try this : 

    function setDates() {
    var date = new Date(), y = date.getFullYear(), m = date.getMonth();
    var lastDay = daysInMonth(m, y);
    var leftpos = 87;
    for ( i = 1; i <= lastDay; i++) {
        var lbl = Titanium.UI.createLabel({
            top : 147,
            left : leftpos,
            height : 22,
            width : 22,
            color : '#111',
            text : i,
            font : {
                fontSize : 13,
                fontFamily : 'Arial'
            },
            zIndex : 999,
            textAlign : 'center',
            id : 'lbldate_' + i,
            value : i,
        });
        Ti.API.info('Label=>' + 'lbldate_' + i);
        lbl.addEventListener('click', function(e) {
            setSelectedDate(e.source.value);
        });
        Dates.push(lbl);
        $.winSchedule.add(Dates);
        if (i > 9) {
            leftpos = leftpos + 22;
        } else {
            leftpos = leftpos + 18;
        }
    }
}

function setSelectedDate(i) {
    alert(i);
    alert(Dates);
}
