[ANDROID] App crashes when using file.write()

Hello,

my app is crashing when I want to write to the DataDirectory, and I think it's beacuse of the grow heap it causes. When run my app in debug mode, it's not crashing and showing me "grow heap size allocated to xx" and without it's just crashing.

I'm making an HttpRequest and write the responseData to the ApplicationDirectory, however, it's working on iOS fine.

my code (written in coffee):

// Generated by CoffeeScript 1.6.3
var APP, CONFIG;

APP = require("/core");

CONFIG = arguments[0];

$.init = function() {
  var _blob, _h, _w;
  $.WrapperPhoto.setWidth(APP.Device.width);
  if (CONFIG.image != null) {
    $.file = Ti.Filesystem.getFile(APP.remoteURLDataStorage, Ti.Utils.md5HexDigest(CONFIG.image + '') + '.' + CONFIG.image.split('.').pop());
    if (!$.file.exists()) {
      APP.log("info", "FILE NOT FOUND, DOWNLOADING!");
      $.Wrapper.addEventListener("postlayout", $.loadImageOnRequest);
    } else {
      _blob = $.file.read();
      _w = _blob.width;
      _h = _blob.height;
      $.finishImageHandling(_w, _h);
      _blob = null;
    }
  } else {
    $.Wrapper.remove($.WrapperPhoto);
  }
};

$.loadImageOnRequest = function() {
  var xhr;
  $.Wrapper.removeEventListener("postlayout", $.loadImageOnRequest);
  APP.log("info", "LOAD IMAGE ON REQUEST!");
  xhr = Ti.Network.createHTTPClient({
    timeout: 5000,
    async: true,
    onload: function(e) {
      var _blob, _h, _w;
      if (e.success) {
        if ($.file.write(this.responseData)) {
          _blob = $.file.read();
          _w = _blob.width;
          _h = _blob.height;
          $.finishImageHandling(_w, _h);
          return _blob = null;
        } else {
          return alert("Couldn't write to Directory.");
        }
      } else {
        return alert("Couldn't download the image, please try again later.");
      }
    }
  });
  xhr.open("GET", CONFIG.image);
  return xhr.send();
};

$.finishImageHandling = function(w, h) {
  $.Photo.setWidth(w / 2);
  $.Photo.setHeight(h / 2);
  $.Photo.image = $.file.nativePath;
  return $.file = null;
};
