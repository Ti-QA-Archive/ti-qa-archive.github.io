  • Ok, I found by myself, but I still don't know how to remove/convert HTML tags…

    var win = Ti.UI.currentWindow;

var tableView = Ti.UI.createTableView();

var url = 'http://localhost/rss/makefulltextfeed.php?url=rss.lemonde.fr%2Fc%2F205%2Ff%2F3050%2Findex.rss&max=5&links=remove&exc=1&format=json&submit=Create+Feed';

var xhr = Titanium.Network.createHTTPClient();
    xhr.open('GET',url);
    xhr.setRequestHeader('Content-Type', 'application/json; charset=utf-8');
    //ERROR
    xhr.onerror = function(e) {
        alert("ERROR " + e.error); 
         Titanium.API.info("some error");
    }; 
    //SUCCESS
    xhr.onload = function() {
        //callback(JSON.parse(this.responseText));
        Titanium.API.info(this.responseText);
        Titanium.API.info(this.status);
        var json = JSON.parse(this.responseText);
        var item = json.rss.channel.item;
        //
        var customData = [];
        for (var i=0; i<item.length; i++) {
            var title = item[i].title;
            var description = item[i].description;
            customData.push({titre:title, description:description});
        }
        //
        var data = [];

        for (var i = 0; i < customData.length; i++) {
            var row = Titanium.UI.createTableViewRow({
                height:100
            });

            var titre = Titanium.UI.createLabel({
                text:customData[i].titre,
                font:{fontSize:13,fontWeight:'bold'},
                width:'auto',
                textAlign:'left',
                top:0,
                left:10,
                height:30
            });

            var description = Titanium.UI.createLabel({
                text:customData[i].description,
                font:{fontSize:11},
                width:'auto',
                textAlign:'left',
                top:35,
                left:10,
                height:50
            });

            row.add(titre);
            row.add(description);     
            row.className = 'rss';

            data.push(row);
        };
        //
        tableView.setData(data);
        win.add(tableView);
    };
    xhr.send();

win.add(tableView);
