            </TableViewRow>
            <TableViewRow>

and the .js controller like this

//requires the database
var db = require('DBController');

/*################################################################
  FUNÇÃO DOS EVENTOS DE TELA
#################################################################*/
function SalvarLeitura(){
    var read = {
        consumption_id: consumption_id,
        consumption_name: consumption.name,
        unit_id: unit_id,
        unit_number:unit.number,
        condo_id: condo_id, 
        condo_name:unit.condo_name,
        value: $.txtValorLeitura.value, 
        date: mes_fator, 
        isdirty: 1
    };

    $.tblLeitura.hide();
    $.activity_loading.show();

    db.CondoUnitConsumptionReads.Save(read);

    var alertDialog = Titanium.UI.createAlertDialog({
                title: 'Sucesso',
                message: 'Dados salvos com sucesso. Ir para o próximo registro', 
                buttonNames: ['OK'] 
            });

    alertDialog.addEventListener('click',function(ev){

        var next_nextunit_id = db.CondoUnits.GetNextUnit(condo_id, nextunit_id);
        //navegaremos para a próxima tela
        var view = Alloy.createController('unidade_leitura',{ consumption_id: consumption_id, mes_fator: mes_fator, unit_id: nextunit_id, nextunit_id: next_nextunit_id}).getView();
        if (OS_IOS){
            Alloy.Globals.Navigation.openWindow(view);    
        }
        else{
            view.open();
        }
        $.unidade_leitura.close();
    });

    alertDialog.show();

}

/*################################################################
  EXECUTADO NO MOMENTO QUE A TELA É CARREGADA
#################################################################*/
var args = arguments[0] || {};
//busca dos parâmetros passados pela tela anterior
var condo_id = Ti.App.Properties.getString('condo_id');
var consumption_id = args.consumption_id;
var consumption = db.CondoConsumptions.GetById(condo_id,consumption_id);
var unit_id = args.unit_id;
var unit = db.CondoUnits.Get(unit_id);
var mes_fator = args.mes_fator;
var nextunit_id = args.nextunit_id;


//mostro o loading
$.activity_loading.show();

var date = new Date();
if (mes_fator==-1){
    if (date.getMonth()==1){
        date = new Date(date.getFullYear()-1,date.getMonth()-1,1);
    }
    else{
        date = new Date(date.getFullYear,date.getMonth-1,1);    
    }
}

//carrega da base os dados dessa leitura
var read = db.CondoUnitConsumptionReads.Get(consumption_id, unit_id, mes_fator);
var unit = db.CondoUnits.Get(unit_id);
var consumption = db.CondoConsumptions.GetById(condo_id,consumption_id);

//define os dados na tela
$.lblLeitura.text = consumption.name;
$.lblCondominio.text = unit.condo_name;
$.lblUnidade.text = unit.number;

if (mes_fator==0){
    $.lblData.text = "Mês atual";
}
else{
    $.lblData.text = "Mês anterior";
}

if (read != undefined && read.value != undefined){
    $.txtValorLeitura.value = read.value;
}
else{
    $.txtValorLeitura.value = '';    
}

$.activity_loading.hide();

$.unidade_leitura.addEventListener("open", function(event, type) {
    $.txtValorLeitura.focus();
});
