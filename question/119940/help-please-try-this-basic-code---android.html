Help please try this basic code ! ( Android)

This code works fine, until the app pause, please try this code, click on a row to open the window, and pause the app ( pressing Home button). when you go back to the app, nothing works like before : the window size change and the click doesn't work anymore.

app.js

Titanium.UI.setBackgroundColor('#000');

var tabGroup = Titanium.UI.createTabGroup();

var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});

var data = [];
for (i = 0 ; i < 30 ; i++)
    data.push({title:'Row n°' + i, color:'#000'});

var tableView1 = Titanium.UI.createTableView({
    data:data
});

tableView1.addEventListener('click',function(row){

    var newWindow = Titanium.UI.createWindow({
        url:'window.js',
        backgroundColor:'#FFF'
    });
    newWindow.open();
});

win1.add(tableView1);
tabGroup.addTab(tab1);  
tabGroup.open();

window.js

var win = Titanium.UI.currentWindow ;
var backbutton = Titanium.UI.createButton({
    top:5,
    title:'Close window',
    height:50
});

backbutton.addEventListener('click',function(){
    win.close();
});

win.add(backbutton);

Bug with Android (emulator and device)from 1.6 to 2.3.3

    Maxime

    …and which Titanium and Android SDK versions are you using? How can people be expected to test your results unless they have this? This should be the bare minimum information that you should include when you create any question in this forum.

    Regarding your issue, tested with the latest Continuous Build 1.7.X on Android 2.3.3….

    Personally, I would not expect my own code to run successfully unless it validates correctly first. Titanium Studio has this functionality built-in. After resolving the issues in your code that it brought to my attention, your code ran without issue. Hence, this is the working code:

    app.js:

    Titanium.UI.setBackgroundColor('#000');

var tabGroup = Titanium.UI.createTabGroup();

var win1 = Titanium.UI.createWindow({
    title:'Tab 1',
    backgroundColor:'white'
});
var tab1 = Titanium.UI.createTab({
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});

var data = [];
for (var i = 0; i < 30; i++){
    data.push({title:'Row n°' + i, color:'#000'});
}

var tableView1 = Titanium.UI.createTableView({
    data:data
});

tableView1.addEventListener('click',function(row){

    var newWindow = Titanium.UI.createWindow({
        url:'window.js',
        backgroundColor:'red'
    });
    newWindow.open();
});

win1.add(tableView1);
tabGroup.addTab(tab1);
tabGroup.open();

    window.js:

    var win = Titanium.UI.currentWindow ;
var backbutton = Titanium.UI.createButton({
    top:5,
    title:'Close window',
    height:50
});

backbutton.addEventListener('click',function(){
    win.close();
});

win.add(backbutton);

    Cheers
