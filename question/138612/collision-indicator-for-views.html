  • You need to make a collision function like so
    ~~~
    function isCollide(a, b) {
    return !(
    ((a.y + a.height) < (b.y)) ||
    (a.y > (b.y + b.height)) ||
    ((a.x + a.width) < b.x) ||
    (a.x > (b.x + b.width))
    );
    }
