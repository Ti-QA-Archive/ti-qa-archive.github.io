    Hi, it is simple as you think: Please look to XMLTools from David Bankier:

    var xhr = Ti.Network.HTTPClient({
    onload: function() {
        var XMLTools = require("XMLTools");
        var xml = new XMLTools(this.responseText);
         var my_object = xml.toObject();
    }    
});
xhr.open('GET',url),xhr.send();

    exports.getchannel() function(_callback) {
    var xhr = Ti.Network.HTTPClient({
            onload: function() {
        var XMLTools = require("XMLTools");

        _callback && _callback(xml.toObject());
    }       
    });
    xhr.open('GET','http://www.indiansinkuwait.com/RSS/ForumFeed.aspx'),xhr.send();
