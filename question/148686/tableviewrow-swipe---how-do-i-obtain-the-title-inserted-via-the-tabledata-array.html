    backgroundColor : 'white'

var tb = [{title: 'Allergy Diagnoses', header : 'Allergy / Immunology'},{title: 'Asthma'},{title: 'Hives'},{title : 'Swelling'},{title:'Seasonal Allergy'},{title : 'Immunizations' },{title : 'Frequent Skin Infections' },
{title : 'Recurrent Sinus Infections'},{title : 'Recurrent Throat Infections' },{ title : 'Recurrent Bronchitis'},{title : 'Recurrent Ear Infections' },{title : 'Recurrent Pneumonia' },{  title : 'Itching Eyes or Nose' }];


var Allergytable = Ti.UI.createTableView({
    objName : 'table'
});

for (var i = 0; i < tb.length; i++) {
    var row = Ti.UI.createTableViewRow({
        //       title:tableData[i].title,
        className : 'row',
        objName : 'row',
        touchEnabled : true,
        height : 100,
        category : 'allergy',
        hasCheck : false
    });
    Ti.API.info('Creating Allergy Table ' + i);

    var enabledWrapperView = Ti.UI.createView({
        backgroundColor : '#008FD5',
        objName : 'enabledWrapperView',
        rowID : i,
        width : Ti.UI.FILL,
        height : '100%',
        rowText: tb[i].title
    });

    var disabledWrapperView = Ti.UI.createView({
        backgroundColor : '#A2E0FF',
        objName : 'disabledWarpperView',
        touchEnabled : false,
        width : 300,
        height : '80%'
    });
    enabledWrapperView.add(disabledWrapperView);

    //   enabledWrapperView.add(disabledWrapperView);

    var label = Ti.UI.createLabel({
//        backgroundColor : '#313F48',
        color : 'white',
        objName : 'label',
        text : tb[i].title,
        touchEnabled : false,
        left : 0,
        width : 200
    });
    disabledWrapperView.add(label);

    row.add(enabledWrapperView);

    tableData.push(row);

};

        // log e

    }

});

  • Hi

    Whilst the click event returns the row and rowData objects, the swipe event does not.

    • click event documentation
    • swipe event documentation

    If you tweak your code as follows (I know you have more in the click event so add mine to yours);

    Allergytable.addEventListener('swipe', function (e) {
    Ti.API.info('Swipe object: ' + JSON.stringify(e));
});

Allergytable.addEventListener('click', function (e) {
    Ti.API.info('click object: ' + JSON.stringify(e));
    Ti.API.info('click title: ' + e.rowData.title);
});

    You will see the differences between the two events and what data they have access to.

      Swipe simply does not return the two properties that actually give you access to the objects you would need; row and rowData.
