  • function create_win( old_win, new_win ) {

if(old_win) old_win.animate({right:320,left:-320});

if(new_win) new_win.animate({left:0,right:0});

}

my_win_1 = my_win_2 = Ti.UI.createWindows({left:320, right:320});
create_win( null, my_win_1);

my_win_1.addEventListener('click',function(){ create_win(my_win_1, my_win_2);});
my_win_2.addEventListener('click',function(){ create_win(my_win_2, my_win_1);});
