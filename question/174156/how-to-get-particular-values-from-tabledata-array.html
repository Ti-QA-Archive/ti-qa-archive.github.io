    Hello Sabari,

    Simple Demo for You.

    var win = Ti.UI.createWindow();
win.open();

var Customdata = [{
    title : 'A',
    call : '01'
}, {
    title : 'B',
    call : '02'
}, {
    title : 'a',
    call : '03'
}, {
    title : 'b',
    call : '04'
}];
var data  = [];
var tableView = Titanium.UI.createTableView();

for (var i =0; i <  Customdata.length ; i++) {

    var row = Ti.UI.createTableViewRow({
        height : 45,
        backgroundColor : '#FFF',
        id : i,
    });

    var lbl = Ti.UI.createLabel({
        text : Customdata[i].title,
        font : {
            fontSize : 16,
            fontWeight : 'bold'
        },
        left : 20,
        width : 270,
        touchEnabled : false,
        color : '#5a5a5a',
    });
    row.add(lbl);

    data.push(row);
};
tableView.data = data;
tableView.addEventListener('click', selectRow);

function selectRow(e) {

    // event data
    var data = Customdata[e.index].call;
    Ti.API.info('>>> Selected Row Data :: ' + data);


}
win.add(tableView);

    i hope you get your solution. thanks.
