Activity problem

In my android app i need to start a activity by clicking a button and when i clicked that button an activity starts and shows a message" rolling" for 5 seconds and it should direct to a background image.how can i do this ,for this i had created the app upto showing the message but i dont know how to set the time and direct it to the image ,any one help me.Below is my code.

Ti.API.info('FILE NAME : text.js, ACTIVITY : Start');

var win = Titanium.UI.createWindow(
{
navBarHidden : true,
exitOnClose: true,
backgroundColor : '#000'
});
var actInd = Titanium.UI.createActivityIndicator({

bottom:10, 
height:50,
width:10

});

var button = Titanium.UI.createButton({
title:'To Roll',
height:35,
width:130,
top:10,
left:20
});

button.addEventListener('click', function()
{

actInd.font = {fontFamily:'Helvetica Neue', fontSize:15,fontWeight:'bold'};
actInd.color = 'white';
actInd.message = 'Rolling...';
actInd.width = 210;
actInd.show();

});

win.add(button);
win.add(actInd);
win.open();

Ti.API.info('FILE NAME : text.js, ACTIVITY : Stop');

    button.addEventListener('click', function() {

actInd.font = {fontFamily:'Helvetica Neue', fontSize:15,fontWeight:'bold'};
actInd.color = 'white';
actInd.message = 'Rolling...';
actInd.width = 210;
actInd.show();

setTimeout(function(e){
    actInd.hide();
},5000);

});

    5000 means 5 secs.
    1 sec = 1000 ;
