  • Tips:
    -please add a className in tableviewrow property

    var row = Titanium.UI.createTableViewRow({
    height: '100',
    className: 'myrows'
});

    -dont use view for the image… use the imageview object..

    var img = Titanium.UI.createImageView({
    url: 'type_your_image_url',
    height: '100',
    width: '100',
    left: 0,
    top: 0
});
row.add(img);

    exports.TableRow = function(data){
    var tableData = [];
    if(data[0].id != undefined){
        for(var x = 0; x < data.length; x++){
            var row = Titanium.UI.createTableViewRow({
                height: 140,
                className : 'datarow',
                selectedBackgroundColor : '#fff'
            });

            var bDistance = Titanium.UI.createLabel({
                text : (data[x].distance!=0) ? Math.round(data[x].distance).toFixed(2) + " KM" : "Map not available",
                top: 0,
                right: 5,
                height:20,
                width:300,
                textAlign: "right",
                clickName:'distance',
                font:{fontSize:14,fontWeight:'normal', fontFamily:'Arial'},
                color:'#999'
            });
            row.add(bDistance);

            var bPhoto = Titanium.UI.createImageView({
                url: (data[x].id) ? "http://www.gohunt.ph/business_logo/" + data[x].id + "/thumbnail.jpg" : "../images/no_photo.jpg",
                top:5,
                left:10,
                width:100,
                height:100,
                clickName:'photo'
            });
            row.add(bPhoto);

            var bName = Titanium.UI.createLabel({
                color:'#576996',
                font:{fontSize:16,fontWeight:'bold', fontFamily:'Arial'},
                left:120,
                top:5,
                height:50,
                textAlign: "left",
                width:200,
                clickName:'tradename',
                text: data[x].trade_name + ""
            });
            row.filter = bName.text;
            row.add(bName);

            var bAddress = Titanium.UI.createLabel({
                color:'#222',
                font:{fontSize:15,fontWeight:'normal', fontFamily:'Arial'},
                left:120,
                top:51,
                height:50,
                width:300,
                textAlign: "left",
                clickName:'address',
                text: data[x].address + ""
            });
            row.add(bAddress);

            var bRating = Titanium.UI.createImageView({
                url: "../images/stars.png",
                bottom:1,
                left:120,
                width:200,
                clickName:'rating',
                height:50
            });
            row.add(bRating);

            var bContact = Titanium.UI.createImageView({
                url: "../images/contacts.png",
                top:35,
                right:5,
                width:50,
                clickName:'contact',
                height:50
            });
            row.add(bContact);
            var bReview = Titanium.UI.createLabel({
                color:'#999',
                font:{fontSize:14,fontWeight:'normal', fontFamily:'Arial'},
                right:5,
                bottom:5,
                height:20,
                width:200,
                textAlign: "right",
                clickName:'review',
                text: "(#) of reviews"
            });
            row.add(bReview);    
            tableData.push(row);
        }
    }
    return tableData;
}
