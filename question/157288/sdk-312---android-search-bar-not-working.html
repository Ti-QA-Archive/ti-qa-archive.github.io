var tableView = Titanium.UI.createTableView({
    top : 5,
    left : 5,
    right : 5,
    bottom : 5,
    separatorColor : "black",
    backgroundColor : "transparent",
    separatorStyle : 1
});

win.backgroundColor = "white";

var dataUsers = [];


var Search = Titanium.UI.createSearchBar({
    showCancel : true,
    hintText : 'Search',
    autocorrect : false
});
Search.addEventListener('blur', function() {
    if (Ti.Platform.name === "android") {
        Ti.UI.Android.hideSoftKeyboard();
    }
});
tableView.search = Search;
var getFriends_list = function() {
    try {
        var json = JS_FEED;
        Ti.API.info("json.contacts.length: " + json.contacts.length);

        for (var i = 0; i < json.contacts.length; i++) {
            var rowFriends = Titanium.UI.createTableViewRow({
                height : (Titanium.Platform.osname != "android") ? 70 : 120,
                width : 307,
                backgroundColor : 'gray',
                selectedBackgroundColor : "transparent",
                selectionStyle : (Ti.Platform.osname != "android") ? Titanium.UI.iPhone.TableViewCellSelectionStyle.NONE : "",
                title : (Titanium.Platform.osname == "android") ? json.contacts[i].name : ""
            });

            var view_frame = Titanium.UI.createImageView({
                left : 10,
                top : 6,
                height : 55,
                width : 55,
            });
            var img_avatar = Ti.UI.createImageView({
                top : 2,
                left : 2,
                height : 46,
                width : 48
            });
            img_avatar.image = "/appicon.png";
            if (Titanium.Platform.osname != "android") {
                view_frame.add(img_avatar);
                rowFriends.add(view_frame);
            } else {
                rowFriends.add(img_avatar);
            }

            var name_lbl = Titanium.UI.createLabel({
                top : 20,
                height : 20,
                left : 85,
                font : {
                    fontSize : 15
                },
                color : 'black'
            });
            name_lbl.text = json.contacts[i].name;
            rowFriends.add(name_lbl);
            rowFriends.filter = name_lbl.text;
            dataUsers.push(rowFriends);
        }
        tableView.setData(dataUsers);
    } catch(exception) {
        alert(exception);
    }

}
getFriends_list();
tableView.filterAttribute = (Titanium.Platform.osname != "android") ? 'filter' : 'title';
win.add(tableView);

win.open();
