    I found the way.. you can also check Link that mitul Posted…

    but i am posting this code directly so you would get better idea

    function convertomiles(miles) {

    var Currentlatitude = 22.163242; //current latitude

    var scalingFactor = Math.abs((Math.cos(2 * 3.14 * Currentlatitude / 360.0) ));

    var latDelta = miles / 69.0;
    var longDelta = miles / (scalingFactor * 69.0);

    var regiion = {
        latitude : 22.163242, //current latitude
        longitude : 70.916748,//current Longitude
        latitudeDelta : latDelta,
        longitudeDelta : longDelta
    };

    yourMapView.setRegion(regiion);
}

convertomiles(5); // here 5 is in miles
