     HeritageApp.ui.createMapListWindow = function () {
        var mapListWindow = Titanium.UI.createWindow({

            backgroundColor: '#FFF',
            barImage: HeritageApp.ui.barImage,
            navBarHidden: false,

            contentWidth:'auto',

            layout:'vertical',

            showHorizontalScrollIndicator:true
        });

            backgroundColor:'#336699',
            top: 0,
            height: 800,
        });
        // create table view
        var item = new Item();
        var data = [];

        var timer = setInterval(function(){
            if(HeritageApp.geo.hasGeo == true) {
                mapview.region = {
                    latitude: HeritageApp.geo.latitude,
                    longitude: HeritageApp.geo.longitude,
                    latitudeDelta:0.04,
                    longitudeDelta:0.04
                }
                fetchTableData();
            }

        fetchTableData = function() {
            var latitude = Ti.App.properties.getString('latitude');
            var longitude = Ti.App.properties.getString('longitude');

            var rows = item.getByDistance(latitude, longitude);
            Ti.API.info(rows.length);
            for ( var i=0, len=rows.length; i<len; ++i ){
                Ti.API.info('ID: ' + rows[i].id);
                Ti.API.info('County: ' + rows[i].county);
                var tableViewDataItem = Ti.UI.createTableViewRow();
                var title = Ti.UI.createLabel({
                    color:'#7e8834',
                    font:{fontWeight:'bold',fontSize:18},
                    left:5,
                    top:0,
                    height:30,
                    text: rows[i].title
                });
                tableViewDataItem.add(title);

                data.push(tableViewDataItem);
                tableview.setData(data);
                if(HeritageApp.geo.hasGeo == true) {
                    clearInterval(timer);
                };
            }
        };

        var tableview = Titanium.UI.createTableView({
            top: 0,
            data:data,
            backgroundColor: '#efece9',
            scrollable: false,
            shadow:{
                shadowRadius:3,
                shadowOpacity:0.5,
                shadowOffset:{x:0, y:-3}
            }
        });

        view.add(tableview);


        scrollView.add(search);
        scrollView.add(mapview);
        scrollView.add(view);

        mapListWindow.add(scrollView);

        return mapListWindow;
    };
})();

  • I actually fixed the problem. For me it was a tableview. The solution, for me, was to not to add an event listener to the tableview before I added the tablewiew to the window. Adding the event listener after the tableview was added fixed the problem. I was creating rows. I added the rows and events as data after the tableview was added to the window. Hope that helps someone.
