    Here is example

    var win=Ti.Ui.createWindow();
var slider1 = Titanium.UI.createSlider({
    top: '50%',
    min: 0,
    max: 100,
    width: '80%',
    value: 50
    });
win.add(slider1)
var slider2 = Titanium.UI.createSlider({
    top:'40'%',
    min: 0,
    max: 100,
    width: '60%',
    value: 50
    });
win.add(slider2)
win.open();
