MapView delta values centering away from lat & lng

I have manually tested the coordinates that are being supplied to my mapView, and they are accurate. However when I set the location using the latitudeDelta and longitudeDelta, the map centers about 100 miles from the intended location. Does anyone have a suggestion as to why this might be happening? I have never had this problem outside of Alloy.

Ti.Geolocation.getCurrentPosition(function(e) {
    Alloy.Globals.currentCoords.longitude = e.coords.longitude;
    Alloy.Globals.currentCoords.latitude = e.coords.latitude;
});

<View id="mapContainer">
    <View ns="Ti.Map" id="currentMap" />
</View>

"#mapContainer": {
    width: Ti.Platform.displayCaps.platformWidth,
    height: '360dp',
    left: '0dp',
    bottom: '0dp',
    zIndex: 5
}

$.currentMap.setLocation({
    latitide: Alloy.Globals.currentCoords.latitude,

    latitudeDelta: 0.04,
    longitudeDelta: 0.04,
    regionFit: true,
    animate: true
});
