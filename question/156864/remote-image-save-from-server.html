  • Hello john,

    Please use my following code for saving images to Photo Gallery. I hope it will help you.

    var win = Ti.UI.createWindow({
        title : 'Save to Photo Gallery'
    });
    var xhr = Titanium.Network.createHTTPClient();

    xhr.onload = function() {
        Titanium.Media.saveToPhotoGallery(this.responseData);
        Titanium.UI.createAlertDialog({
            title : 'Photo Gallery',
            message : 'Check your photo gallery for a titanium logo'
        }).show();
    };
    // open the client
    xhr.open('GET', 'http://static.appcelerator.com/images/header/appc_logo.png');

    // send the data
    xhr.send();
    return win;

    And for more code suggestion you can follow Kitchen Sink project.
