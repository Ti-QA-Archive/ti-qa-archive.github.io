Click Events on animation views

I think I know the answer to this one but I thought I would throw it out there. Is it possible to fire the click event while an Image View is animating. The below code does nadda!

// -- Duck 
var bird = Ti.UI.createImageView({image:'bird.gif',width:120,height:120,right:400,bottom:0})
var anim = Ti.UI.createAnimation({left:400, top:10, duration:3000})

bird.addEventListener('click', function(e){
    // Stop The Animation
    bird.animate();
    // Fire Death Animation
    var anim = Ti.UI.createAnimation({left:bird.left, top:500, duration:1000});
})

bird.animate(anim);

// Base Windows
var win1 = Titanium.UI.createWindow({  
    fullScreen:true,
    backgroundColor:'#fff'
});

bird.addEventListener('click', function(e) {
    alert(this.left)
})

win1.add(bird)

// open tab group
win1.open();

Has anyone found a solution around this?
