    You cannot access controller UI elements from outside commonJS modules, I am sure that's not the correct way to structure your code, but if you insist you need another way to pass a reference to the UI element:

    for example you can pass the element to the module as a parameter. Like :

    inter.js

    exports.BoundLoginEvents = function(elt){
    elt.addEventListener('click', function(e) {
             face.Login();
    });
};

    index.js

    var inter=require('scripts/inter');
inter.BoundLoginEvents($.FaceButton);
