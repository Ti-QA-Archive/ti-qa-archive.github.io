LeftNavButton not showing up.

I can't seem to make a custom button show up on a NavBar. What am I doing wrong here?

var btnCancel = Titanium.UI.createButton({
    title:'Cancel',
    backgroundImage:'views/images/headerButton.png'
});


var win = Titanium.UI.createWindow({  
    title:'Sign In',
    barImage:'views/images/brownHeaderBar.png',
    backgroundImage:'views/images/background.png',
    leftNavButton: btnCancel,
    tabBarHidden:true
});


//HIDE TAB GROUP BUT SHOW NAVBAR
var tabGroup = Ti.UI.createTabGroup();

var tab = Ti.UI.createTab({
    title:"Sign In",
    window: win
});

tabGroup.addTab(tab);
tabGroup.open();

  • Was having this issue too, fix is:

    var btnCancel = Titanium.UI.createButton({
  height: 50, // Height of png in pixels
  width: 24, // Width of png in pixels
  title:'Cancel',
  backgroundImage:'views/images/headerButton.png'
});

    Note the width takes into account the LeftCap (see Titanium.UI.Button-object:Properties:backgroundLeftCap)

    • Dan, actually it seems to be a problem with the backgroundImage of the button. It works with barImage.
      This works (SDK 1.6, iOS):

      var btnCancel = Titanium.UI.createButton({
    title:'Cancel'
});


var win = Titanium.UI.createWindow({
    title:'Sign Inn',
    barImage:'views/images/brownHeaderBar.png',
    backgroundImage:'views/images/background.png',
    leftNavButton: btnCancel,
    tabBarHidden:true
});
