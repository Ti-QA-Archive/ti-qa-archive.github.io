  • Hello Douglas,

    Try below sample : 

    var win = Ti.UI.createWindow({
    width:"100%",
    height:"100%",
    backgroundColor : "yellow"
});

var view = Ti.UI.createView({
    height:"100%",
    width:"100%",
    backgroundColor:"yellow",
    layout : "horizontal" //This property set all the images in horizontal manner
});
win.add(view);

for(var i=0; i<5; i++){
    var image = Ti.UI.createImageView({
        image : "a.png"
    });
    view.add(image);
}
win.open();
