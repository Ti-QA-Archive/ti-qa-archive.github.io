    • Thanks, but I don't like that very much…
      I now solved it by doing it with a TableViewSection like this.

      var inputField = Ti.UI.createTextField({
        height:34,
        width:Ti.UI.FILL,
        left:30,
        right:30,
        value:$content.title || "",
        hintText:"Vul een titel in",
        keyboardType:Ti.UI.KEYBOARD_DEFAULT,
        returnKeyType:Ti.UI.RETURNKEY_DEFAULT,
        borderStyle:Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});


var inputFieldSection = Ti.UI.createTableViewSection({
        headerView:Ti.UI.createView({
            height:34,
            children:[inputField]
        })
});
