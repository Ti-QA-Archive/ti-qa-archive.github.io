  • I use Titanium SDK 1.7.2, Android SDK 3.0, and the phone is an HTC Desire S.
    The thing I don't understand is on the SDCard, the photo is "normal", but blurred in the app…
    the code:

        var win = Titanium.UI.currentWindow;
    var startVideoButton = Titanium.UI.createButton({
        bottom:10, width:150, height:150,
        title:'Click-it'
    });
    startVideoButton.addEventListener('click' , function(){
        Ti.Media.takePicture();
    });
    var overlay = Titanium.UI.createView();
    var imageView = Titanium.UI.createImageView({
        left:0,
        top:0,
    });
    win.add(imageView);
    overlay.add(startVideoButton);
    Titanium.Media.showCamera({
        overlay:overlay,
        showControls:true,
        success:function(event) {
            imageView.image = event.media;
        },
        cancel:function(){},
        error:function(error) {
            var a = Titanium.UI.createAlertDialog({title:'Camera'});
            if (error.code == Titanium.Media.NO_CAMERA) {
                a.setMessage('Please run this test on a device with a camera');
            }
            else {
                a.setMessage(error);
            }
            a.show();
        }
    });
