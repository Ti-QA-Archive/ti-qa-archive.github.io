    Hi Husain,

    after open method of HTTPClient you need to send a request using request method.

    e.g try this one

    var xhr = Ti.Network.createHTTPClient({
    onload : function(e) {
    alert(this.responseText);

    },
    onerror : function(e) {
        Ti.API.debug(e.error);
        alert('error');
    },
    timeout : 5000
});
xhr.open("GET", "http://file.bimser.com.tr/DOSYALAR/G2/veriler.js");
xhr.send();
