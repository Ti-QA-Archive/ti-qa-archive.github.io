    hello Nash,

    just try it

    var win = Ti.UI.currentWindow;

var a = Ti.UI.createAnimation();
a.top = 300;
a.duration = 1000;

var b = Ti.UI.createAnimation();
b.top = 100;
b.duration = 1000;

var circle = Titanium.UI.createView({
    height:100,
    width:100,
    borderRadius:50,
    backgroundColor:'#336699',
    top:0,
    left:0
});

win.add(circle);

setInterval(function(e)
{
    circle.animate(a);
},2000);

a.addEventListener('complete',function(e){
    circle.animate(b);    
});

    create new animation.js file and put this code into it.

    call this js, I m not sure but this is the animation that you want.
