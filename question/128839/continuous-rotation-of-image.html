Continuous rotation of image

Hi guys

I want to rotate an image view repeatedly so that it will look like any other GIF loader out there.

This is my code so far:

        var t = Ti.UI.create2DMatrix();
        function rotate() {
            var a = Titanium.UI.createAnimation();
            a.addEventListener('complete', rotate);
            t = t.rotate(90);
            a.transform = t;
            a.duration = 1000;
            myImage.animate(a);
        }
        rotate();

It works terribly. It stops in between animations and it not fluent at all.

How may I do this?

  • Try this code this is working :

    var t = Ti.UI.create2DMatrix();
        function rotate() {
            var a = Titanium.UI.createAnimation();
            a.addEventListener('complete', rotate);
            t = t.rotate(90);
            a.transform = t;
            a.duration = 1000;
            myImage.animate(a);
        }
        setInterval(rotate,100);
