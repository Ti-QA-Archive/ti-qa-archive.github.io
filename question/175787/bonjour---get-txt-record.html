function updateBonjourServices(e) {
        services = e['services'];

        for (var i=0; i < services.length; i++) {
                var service = services[i];

                Ti.API.info("BONJOUR!");

                if (service.socket == null || !service.socket.isValid) {

                        Ti.API.info("Bonjour Resolve!");

                        service.resolve();

                        service.socket.addEventListener('read', function(x) {
                            Ti.API.info("Bonjour: "+x['data'].text);
                        });

                        service.socket.connect();
