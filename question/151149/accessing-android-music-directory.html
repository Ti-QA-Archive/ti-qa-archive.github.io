  • here you go running audio from sdcard

    var f = Ti.Filesystem.getFile(Ti.Filesystem.externalStorageDirectory,filename);
alert(f.exists());
audioPlayer(f)
function audioPlayer (f) {

    var audioPlayer = Ti.Media.createSound({ 
        url: f
       });
    return audioPlayer;
}
module.exports = audioPlayer;
