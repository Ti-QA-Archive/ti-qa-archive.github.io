var win=Ti.UI.createWindow({
    layout:'vertical',
    backgroundColor:'white'
});

var db=Ti.Database.open('am');
db.execute('create table if not exists am1(name text, id text)');
db.execute('create table if not exists am2(name text, id text)');

db.execute('insert into am1 values(?,?)','amimul1','1');
db.execute('insert into am1 values(?,?)','amimul2','2');
db.execute('insert into am1 values(?,?)','amimul3','3');

db.execute('insert into am2 values(?,?)','amimul11','11');
db.execute('insert into am2 values(?,?)','amimul22','22');
db.execute('insert into am2 values(?,?)','amimul33','33');
db.execute('insert into am2 values(?,?)','amimul44','44');
db.execute('insert into am2 values(?,?)','amimul55','55');

var pickerValues = [];

        rows: pickerValues,fontSize: "40dp",//width:"300"

    //l++;

 var picker = Ti.UI.createPicker({
  top:'14%',
  borderColor:'#B6B6B4',

  useSpinner: true
