  • Hello Brandon,

    Yes this way you can set first TableViewRow differently…

    You have to create multiple TableViewRow…

    
var win = Ti.UI.currentWindow;

var data = ["Home", "Friends", "Messages", "Favorites", "Reports", "Settings", "Sign Out"];

var menuArr = [];

var tableView = Ti.UI.createTableView({
    backgroundColor : "pink",    
});

var menuRowProfileView = Ti.UI.createTableViewRow({
    height : 200,
    backgroundColor : 'blue',
    hasChild : true,
    title : 'Different Row'
});

menuArr.push(menuRowProfileView);

for (var i = 0; i < data.length; i++) {
    var menuRow = Ti.UI.createTableViewRow({
        height : 50,
        backgroundColor : 'yellow',
        hasDetail : true
    });

    var lblMenu = Ti.UI.createLabel({
        text : data[i],
        color : 'white',
        font : {
            fontSize : 16
        },
        color : 'red',
        left : 40,
        height : Ti.UI.SIZE,
        width : Ti.UI.SIZE,
        backgroundColor : 'transparent'
    });
    menuRow.add(lblMenu);
    menuArr.push(menuRow);
};
tableView.setData(menuArr);

win.add(tableView);

  • Hello Brandon, Why not use Listview instead of Tableview?. That way you can style the "ListSection" and using "Template" you can enter data same way as in "tableViewRow". Try the below code. To style the list section go to this Link.

    var win = Ti.UI.createWindow({backgroundColor: 'white'});
var myTemplate = {
    childTemplates: [
        {                            // Image justified left
            type: 'Ti.UI.ImageView', // Use an image view for the image
            bindId: 'pic',           // Maps to a custom pic property of the item data
            properties: {            // Sets the image view  properties
                width: '50dp', height: '50dp', left: 0
            }
        },
        {                            // Title 
            type: 'Ti.UI.Label',     // Use a label for the title 
            bindId: 'info',          // Maps to a custom info property of the item data
            properties: {            // Sets the label properties
                color: 'black',
                font: { fontFamily:'Arial', fontSize: '20dp', fontWeight:'bold' },
                left: '60dp', top: 0,
            }
        },
        {                            // Subtitle
            type: 'Ti.UI.Label',     // Use a label for the subtitle
            bindId: 'es_info',       // Maps to a custom es_info property of the item data
            properties: {            // Sets the label properties
                color: 'gray',
                font: { fontFamily:'Arial', fontSize: '14dp' },
                left: '60dp', top: '25dp',
            }
        }
    ]
};

var listView = Ti.UI.createListView({
    templates: { 'template': myTemplate },
    defaultItemTemplate: 'template'
});

var view = Titanium.UI.createView({
   backgroundColor:'gray',
   width:Ti.UI.FILL,
   height:50
});

var label1 = Ti.UI.createLabel({
  color: '#900',
  text: 'Fruits / Frutas',
  textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER,
  width: Ti.UI.SIZE, height: Ti.UI.SIZE
});

view.add(label1);

var sections = [];

var fruitSection = Ti.UI.createListSection({ 

    headerView : view

    });
var fruitDataSet = [

    { info: {text: 'Apple'}, es_info: {text: 'Manzana'}, pic: {image: 'apple.png'}},
    { info: {text: 'Banana'}, es_info: {text: 'Banana'}, pic: {image: 'banana.png'}},
    { info: {text: 'Apple'}, es_info: {text: 'Manzana'}, pic: {image: 'apple.png'}},
    { info: {text: 'Banana'}, es_info: {text: 'Banana'}, pic: {image: 'banana.png'}},
    { info: {text: 'Apple'}, es_info: {text: 'Manzana'}, pic: {image: 'apple.png'}},
    { info: {text: 'Banana'}, es_info: {text: 'Banana'}, pic: {image: 'banana.png'}},
];
fruitSection.setItems(fruitDataSet);
sections.push(fruitSection);

listView.setSections(sections);
win.add(listView);
win.open();

    Hope this helps. If so, please mark as best answer. Thanks.
