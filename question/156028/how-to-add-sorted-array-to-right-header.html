    You need to add items one by one and check for change in the name's first character like :

    var viewData = [];
var prevChar = '';

sortedArray.forEach(function(itemRow) {
    var s = itemRow.name;
    var row = {};

    if (s !== '') {
        row.title = itemRow.name;

        if (prevChar !== s.charAt(0)) {
            prevChar = s.charAt(0);
            row.header = prevChar;
        }

        viewData.push(row);
    }

});

table.data = viewData;
