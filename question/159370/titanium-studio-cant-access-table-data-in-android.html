  • Hi Aalem,

    Try this :

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});
var Data = [{
    title : 'row1',
    name : 'one'
}, {
    title : 'row2',
    name : 'two'
}, {
    title : 'row3',
    name : 'three'
}, {
    title : 'row4',
    name : 'four'
}];

var table = Ti.UI.createTableview({
    data : Data
});

win.add(table);
win.open();
table.addEventListener('click', function(e) {
    alert(e.rowData.name);
});
