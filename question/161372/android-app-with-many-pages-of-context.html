  • Hi Edin

    The below code is to be a simple solution for your problem if you like. So please check the code and i hope, it will helpful to you to complete your actual task

    Ti.UI.backgroundColor='#fff';
var win = Ti.UI.createWindow(
    {backgroundColor:'red'}
);

var view1 = Ti.UI.createView();
var view2 = Ti.UI.createView();
var view3 = Ti.UI.createView();


var aLabel = Ti.UI.createLabel({
    text : 'aLabel',
    color : 'green',
    font : {fontSize:15},
    textAlign : 'center'
});

 var aLabel2 = Ti.UI.createLabel({
    text : 'This is a simple test for testing the applecation and I hope, it may be an one approach to do your actual task. This is the built in process of titanium api',
    color : 'green',
    font : {fontSize:15},
    textAlign : 'center'
});

view1.add(aLabel);
view2.add(aLabel2);
view3.add(aLabel);


var scrollableView = Ti.UI.createScrollableView({
  views:[view1,view2,view3],
  showPagingControl:true
});

win.add(scrollableView);
win.open();
