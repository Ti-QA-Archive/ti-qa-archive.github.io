change title with button

    var b2 = Titanium.UI.createButton({
        title:'Change Label 2',
        height:40,
        width:200,
        top:280
    });
    var changed=false;
    b2.addEventListener('click', function()
    {
        if (!changed)
        {
            l2.color = '#ff9900';
            l2.shadowColor = '#336699';
            l2.font = {fontSize:20};
            changed=true;
            size.text = l2.size.height + ' ' + l2.size.width;  // where is size defined?

        {

            l2.font = {fontSize:48};
            size.text = l2.size.height + ' ' + l2.size.width;
            changed=false;
        }
    });
    win.add(b2);

but this code changes font color and size. i want to change text, not size and color.

    Hi

    Here is a simply reusable solution. It works by adding two simple custom properties directly into the button. Here you set the title for the two states (think on/off). You set the initial title value to match the stateFalse property.

    Then the event listener checks the current state against the value inside the actual button and swaps it to the other.

    var b2 = Titanium.UI.createButton({
    height: 40,
    stateFalse: 'Off', // custom property
    stateTrue: 'On', // custom property
    title: 'Off', // same as false to start
    top: 280,
    width: 200
});
b2.addEventListener('click', function (e) {
    // this checks to see if the current title is the same as 
    // "stateFalse" if it is it swaps it the "stateTrue" value
    // otherwise it sets it to "stateFalse" value
    // e.source = whatever control this event listener is attached to
    if (e.source.title === e.source.stateFalse) {
        e.source.setTitle(e.source.stateTrue);
        // you can also set other thing here if you wish
    } else {
        e.source.setTitle(e.source.stateFalse);
        // and even here
    }
});

    This should give you a very simply way of solving your problem.

    • One more benefit of this solution is you can handle the toggle for multiple buttons with a simgle function, for example;

      var toggleState = function (e) {
    // this checks to see if the current title is the same as 
    // "stateFalse" if it is it swaps it the "stateTrue" value
    // otherwise it sets it to "stateFalse" value
    // e.source = whatever control this event listener is attached to
    if (e.source.title === e.source.stateFalse) {
        e.source.setTitle(e.source.stateTrue);
        // you can also set other thing here if you wish
    } else {
        e.source.setTitle(e.source.stateFalse);
        // and even here
    }
};
var b1 = Titanium.UI.createButton({
    height: 40,
    stateFalse: 'Black', // custom property
    stateTrue: 'White', // custom property
    title: 'Black', // same as false to start
    top: 100,
    width: 200
});
var b2 = Titanium.UI.createButton({
    height: 40,
    stateFalse: 'Off', // custom property
    stateTrue: 'On', // custom property
    title: 'Off', // same as false to start
    top: 150,
    width: 200
});
var b3 = Titanium.UI.createButton({
    height: 40,
    stateFalse: 'Down', // custom property
    stateTrue: 'Up', // custom property
    title: 'Down', // same as false to start
    top: 200,
    width: 200
});
win.add(b1);
win.add(b2);
win.add(b3);
b1.addEventListener('click', toggleState);
b2.addEventListener('click', toggleState);
b3.addEventListener('click', toggleState);

      Using this variation, you can add multiple buttons that use the EXACT same handler, because it does not need to know about the original button or their variables or their choices. They are set into the button.

      Much shorter and more readable code.
