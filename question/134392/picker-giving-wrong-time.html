Picker giving wrong time?

According to the Picker change event, it should return an object with "selectedValue". However, it returns "value" and it is not the selected time or the current time.

var win = Titanium.UI.createWindow({ backgroundColor:'#fff' });

var view = Ti.UI.createView({});

var picker = Ti.UI.createPicker({
        top: 0,
        left: 0,
        bottom: 0,
        minuteInterval: 15,
        type: Ti.UI.PICKER_TYPE_TIME,
});
view.add(picker);

picker.addEventListener('change', function(e){
    Ti.API.log(e);
});


win.add(view);
win.open();
