    'touchstart',
    function (e)

            nextImg = images[get_img_nr(1)];

win1.addEventListener(
    'touchend',
    function (e)
    {
        if(active == true){
            touch_pos_end = e.x;
            delta_pos = touch_pos - touch_pos_end;

            if (delta_pos > 0) {
                dir = 1;

                    dir = 0;

                            nextImg.left = 0;

// get_img_nr():
// returns img_nr of
// dir = 1 > next image.
// dir = 0 > current image.
// dir = -1 > prev image

function get_img_nr(dir){
    var i = 0;
    var curImg = 0;
    var index = 0;
    var maxImg = images.length - 1;

    while(images[i] != null){
        if(images[i].left == 0){
            curImg = i;
        }
        i++;
    }

    if(dir == 0){
        return curImg;
    } else  {
        if(dir == 1){
            index = curImg + 1;            
        } else {
            index = curImg - 1;            
        }

        return index;
    }
}
