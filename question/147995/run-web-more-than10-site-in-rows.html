  • Hi

    I am really not sure what you are after, but here is some code that "might" be it.

    To test, create a new default project and replace the contents of the app.js file with the code below.

    Ti.UI.setBackgroundColor('#fff');

var win1 = Ti.UI.createWindow({
    backgroundColor: '#FFF',
    height: Ti.UI.FILL,
    title: 'Websites',
    width: Ti.UI.FILL
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title: 'Websites',
    window: win1
});

function addRow(obj) {
    var strColor = obj.color || '';
    var row = Ti.UI.createTableViewRow({
        backgroundColor: '#fff',
        data: obj,
        hasChild: true,
        height: Ti.UI.SIZE,
        width: Ti.UI.FILL
    });
    var lblTitle = Ti.UI.createLabel({
        color: '#000',
        font: {
            fontSize: 16
        },
        height: Ti.UI.SIZE,
        highlightedColor: '#fff',
        left: 10,
        text: obj.title || '',
        touchEnabled: false,
        width: Ti.UI.SIZE
    });
    row.add(lblTitle);
    return row;
}

var data = [
    { title: 'Apple', url: 'http://www.apple.com/' },
    { title: 'Microsoft', url: 'http://www.microsoft.com/' },
    { title: 'Appcelerator', url: 'http://www.appcelerator.com/' },
    { title: 'BBC News', url: 'http://www.bbc.co.uk/news/' },
    { title: 'Engadget', url: 'http://www.engadget.com/' },
    { title: '9to5Mac', url: 'http://www.9to5Mac.com/' }
];
var intRow = 0, intRows = data.length, row, rows = [];
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    row = addRow(data[intRow]);
    rows.push(row);
}
var tbl = Ti.UI.createTableView({
    data: rows,
    height: Ti.UI.FILL,
    minRowHeight: 44,
    width: Ti.UI.FILL
});
tbl.addEventListener('click', function (e) {
    var win = Ti.UI.createWindow({
        backgroundColor: '#FFF',
        height: Ti.UI.FILL,
        title: e.rowData.data.title,
        width: Ti.UI.FILL
    });
    var web = Ti.UI.createWebView({
        height: Ti.UI.FILL,
        url: e.rowData.data.url,
        width: Ti.UI.FILL
    });
    win.add(web);
    tab1.open(win, { animated: true });
});
win1.add(tbl);

var tabGroup = Titanium.UI.createTabGroup();
tabGroup.addTab(tab1);
tabGroup.open();
