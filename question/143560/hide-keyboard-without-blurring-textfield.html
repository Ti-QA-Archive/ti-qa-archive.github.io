Hide keyboard without blurring textfield

Hi all,

I'm using the google autocomplete places api, which is way too cool, but I need to hide the keyboard without blurring the textfield to get it to work better.

Is there a way to do this? Android has the hide keyboard method but I can't find an equivalent for iOS

Using Ti2.13

Here's the code I'm working with:

searchField.addEventListener("change", function(e) {
        clearTimeout(searchFieldID);
        searchFieldID = setTimeout(function() {
            Ti.API.info("Trigger autocomplete");
            places.autocomplete(searchField.value, latitude, longitude, "false", function(e) {

                //
                // Fill return data into table
                //
                Ti.API.info("Returned data: " + e.predictions.length);
                for ( i = 0; i < e.predictions.length; i++) {
                    Ti.API.info("Array Pos/Description/Ref/Type: " + i + "/" + e.predictions[i].description + "/" + e.predictions[i].reference + "/" + e.predictions[i].types[0]);
                    var row = Ti.UI.createTableViewRow({
                        title : e.predictions[i].description,
                    });
                    searchTableArray.push(row);
                };
                Ti.API.info("Table Array Size: " + searchTableArray.length);
                searchTable.setData(searchTableArray);
                Ti.UI.currentWindow.add(searchTable);
            }, function(e) {

                //
                // Error handling
                //
            });
        }, 500);
    });

  • Hi Ken,

    unfortunately on iOS this is not possible.
    If you are using textFields, the only way to hide the keyboard is to 'blur' it.
