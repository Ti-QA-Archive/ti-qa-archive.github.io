tableview - why the code shows no rows?


var win = Ti.UI.createWindow();
tableview = Ti.UI.createTableView();
var section = Titanium.UI.createTableViewSection();
section.headerTitle = "Hello";
var row1 = Titanium.UI.createTableViewRow({
    title : "Hello 1"
});
var row2 = Titanium.UI.createTableViewRow({
    title : "Hello 2"
});
section.add(row1);
section.add(row2);
tableview.add(section);
win.add(tableview);
win.open();

What s wrong with the code above? It shows an empty tableview, no row titles, no section header..

  • Hi

    Try changing this line;

    tableview.add(section);

    To this instead;

    tableview.setData([section]);

    • setData accepts an array, so the first example forces an array with the [ ] brackets.

      Whereas this example is passed an array directly. I have played with your code slightly to expand on the example;

      var win = Ti.UI.createWindow();
tableview = Ti.UI.createTableView();
var sections = [];
var section1 = Ti.UI.createTableViewSection({
    headerTitle: 'Section 1'
});
var row1 = Ti.UI.createTableViewRow({
    title: "Hello 1"
});
var row2 = Ti.UI.createTableViewRow({
    title: "Hello 2"
});
section1.add(row1);
section1.add(row2);
sections.push(section1);
var section2 = Ti.UI.createTableViewSection({
    headerTitle: 'Section 2'
});
var row3 = Ti.UI.createTableViewRow({
    title: "Hello 3"
});
var row4 = Ti.UI.createTableViewRow({
    title: "Hello 4"
});
section2.add(row3);
section2.add(row4);
sections.push(section2);

tableview.setData(sections);
win.add(tableview);
win.open();
