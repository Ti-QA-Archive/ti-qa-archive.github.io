  • Hi,gursewak singh
    Try this code 

    var window = Titanium.UI.createWindow({
    title : 'Send Message',
    backgroundColor : '#fff',
    navBarHidden : true,
    orientationModes : [Titanium.UI.LANDSCAPE_RIGHT],
    tabBarHidden : true
});
 window.open();

Ti.Gesture.addEventListener(
'orientationchange', function(e) {
   // alert("Called");
    win.orientationModes = [
    Titanium.UI.LANDSCAPE_LEFT,
    Titanium.UI.LANDSCAPE_RIGHT,
    Titanium.UI.PORTRAIT,
    Titanium.UI.UPSIDE_PORTRAIT
  ];
  if (e.orientation == Titanium.UI.LANDSCAPE_LEFT ||
        e.orientation == Titanium.UI.LANDSCAPE_RIGHT) {

        Ti.UI.orientation = Titanium.UI.LANDSCAPE_RIGHT;            
    } 
});
