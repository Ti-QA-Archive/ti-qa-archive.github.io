  • Hi

    You got things a little out of order.

    These two lines, MUST be called AFTER you have defined the code that actually processes the response.

    // create the http object
// set any parameters
// define load, error and any other event handlers
loader.open('GET', 'http://rebooks.me/api/get_recent_posts/');
loader.send();

    I have tweaked the order of your code and added a very simple error callback so you know if there is a problem with the returned data. Try the code below;

    var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    title: 'read Book'
});

function loadBooks() {

    var rowData = [];
    var loader = Titanium.Network.createHTTPClient();
    loader.onload = function (e) {
        alert('loader.onload')
        //var books = eval('('+this.responseText+')'); 
        var books = JSON.parse(this.responseText);
        for (var i = 0; i & lt; books.length; i++) {
            var title = books[i].posts.title;
            var posts = books[i].posts.content;

            var row = Ti.UI.createTableViewRow();
            var view = Ti.UI.createView({
                height: 100,
                layout: 'vertical',
                top: 5,
                right: 5,
                bottom: 5,
                left: 5
            });
            var title = Ti.UI.createlabel({
                text: title,
                left: 54,
                width: 120,
                bottom: 2,
                textAlign: 'right'
            });
            view.add(title);
            var posts = Ti.UI.createTable({
                text: posts,
                hight: 50,
                top: 5,
                right: 54,
                width: 235,
                font: {
                    fontSize: 14
                }
            });
            view.add(posts);
            row.add(view);
            rowData.push(row);
            //row.className = 'item'+i;
            var table = Ti.UI.createTableView({
                data: rowData
            });
            win.add(table);

            alert('moghazy')
        };
    }
    loader.onerror = function (e) {
        alert('There was a problem');
    };
    loader.open('GET', 'http://rebooks.me/api/get_recent_posts/');
    loader.send();
}

      var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    title: 'Read Book'
});

function loadBooks() {
    var rowData = [];
    var loader = Titanium.Network.createHTTPClient();

    loader.onload = function (e) {
        var books = JSON.parse(this.responseText);

        for (var i = 0; i < books.length; i++) {
            var title = books[i].posts.title;
            var posts = books[i].posts.content;

            var row = Ti.UI.createTableViewRow();
            var view = Ti.UI.createView({
                height: 100,
                layout: 'vertical',
                top: 5,
                right: 5,
                bottom: 5,
                left: 5
            });
            var titleLabel = Ti.UI.createlabel({
                text: title,
                left: 54,
                width: 120,
                bottom: 2,
                textAlign: 'right'
            });
            view.add(titleLabel);
            var contentLabel = Ti.UI.createLabel({
                text: posts,
                hight: 50,
                top: 5,
                right: 54,
                width: 235,
                font: {
                    fontSize: 14
                }
            });
            view.add(contentLabel);
            row.add(view);
            rowData.push(row);
        };
        var table = Ti.UI.createTableView({
            data: rowData
        });
        win.add(table);
    }
    loader.onerror = function (e) {
        alert('There was a problem');
    };

    loader.open('GET', 'http://rebooks.me/api/get_recent_posts/');
    loader.send();
}

loadBooks();
win.open();
