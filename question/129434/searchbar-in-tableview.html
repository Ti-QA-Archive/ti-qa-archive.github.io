  • Please look at kitchensink example for that.. or you can try this

    var win = Ti.UI.createWindow({
    fullscreen: false
});
var search = Ti.UI.createSearchBar({
    showCancel: true,
    height: 30,
    width: 100
});
var tableviewrow = [];
for(var x = 0; x < 10; x++){
    var row = Ti.UI.createTableViewRow({
        className: "datarow",
        height: "100"
    });
var lbl = Ti.UI.createLabel({
    text : "Row " + x
});

row.filter = lbl.text; //create what to filter
row.add(lbl);
tableviewrow.push(row); //add to array
}
var tableview = Ti.UI.createTableView({
    search: search,
    filterAttribute: 'filter',    //refrence to what to be filtered
    data : tableviewrow
})

win.add(tableview);
