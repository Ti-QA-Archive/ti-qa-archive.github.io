  • Hi

    You do not have to do anything special to convert any string with escape sequences into or out of a database. This is a repeatable example of how the string is inserted into the db.

    var label1 = Titanium.UI.createLabel({
    color: '#444',
    height: Ti.UI.SIZE,
    font: {
        fontSize: 14,
        fontFamily: 'Helvetica Neue'
    },
    text: "str",
    textAlign: 'left',
    top: 0,
    width: Ti.UI.FILL
});
win1.add(label1);
var btn = Ti.UI.createButton({
    bottom: 20,
    title: 'Test DB string'
});
btn.addEventListener('click', function (e) {
    label1.setText("");
    var str = "Yes. \n If you already have an account with any of the 3 driving centers (Bukit Batok, Comfort, Singapore Safety). you can use their online services to book a test date for your Basic Theory Test (BTT) or Final Theory Test (FTT).\n\nIf you dont have one yet, you need to go down personally with your IC to open an account first, then you can book your BTT or FTT test date online. \n\nNote: the account usually will last for a period of 6 months\n";
    var db = Ti.Database.open("quizDBTest");
    db.execute('Drop Table IF EXISTS Test;');
    db.execute('Create Table IF NOT EXISTS Test (id INTEGER PRIMARY KEY, name TEXT);');
    db.execute('Insert Into Test (name) values (?);', str);

    var rows = db.execute('SELECT name FROM Test;');
    while (rows.isValidRow()) {
        label1.setText(rows.fieldByName('name'));
        rows.next();
    }
    db = null;
});
win1.add(btn);

    This assumes you have a windows called win1, make sure you do NOT overwrite your own database.
