    I don't know if it's possible to do that.. But here's a javascript code to create an excel spreadsheet that I've found. I haven't tested it though (And I have no idea if it will work on Titanium)!:

    function executeExcel() 
{ 
// Declare the variables 
//var Book; 

// Create the Excel application object. 
Excel = new ActiveXObject("Excel.Application"); 

// Make Excel visible. 
//Excel.Visible = true; 

// Create a new work book. 
Book = Excel.Workbooks.Add() 

// Place some text in the first cell of the sheet. 
Book.ActiveSheet.Cells(1,1).Value = "This is column A, row 1"; 

// Save the sheet. 
Book.SaveAs("C:\\data\\TEST.XLS"); 

// Close Excel with the Quit method on the Application object. 
Excel.Application.Quit(); 
}
