      var win = Ti.UI.currentWindow;

var categoryArray = [];
var catRows = [];


if (categoryArray.length == 0) {
    var db = Titanium.Database.open('table');
    var dbrows = db.execute('select location_id, name, tel FROM table WHERE stuff="stuff" ORDER BY NAME ASC');
    while (dbrows.isValidRow()) {
        categoryArray.push({
            catid:dbrows.fieldByName('location_id'),
            tel:dbrows.fieldByName('tel'),    
            title:dbrows.fieldByName('name')

        }); 
        dbrows.next();
    }
    dbrows.close();
    db.close();
}


for (var c=0;c<categoryArray.length;c++) {
    var row = Ti.UI.createTableViewRow({height:40,backgroundColor:'#ffffff',selectedBackgroundColor:'#eeee33',hasChild:false}); 
    var item = categoryArray[c];

    row.catname = item.title;
    row.catid = item.catid;
    row.tel = item.tel;    

    var catName = Ti.UI.createLabel({
        text: item.title,
        color: '#334499',
        textAlign:'left',
        left:40,
        top:4,
        height:'auto',

    row.add(catName);
