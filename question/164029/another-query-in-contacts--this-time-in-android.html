                 });  

                     var people = con[i];

    if(people['phone'].mobile == people.fullName){

         var p = con[i].fullName;

         lbl1.text = p;

    }


   else{
          var q =  lbl1.text  = people.fullName +"\n"+people['phone'].mobile;
           lbl1.text = q;
   }


             row2.add(lbl1);
             rowData.push(row2);

            }


            var aTableView = Ti.UI.createTableView();

            aTableView.setData([]);    
            aTableView.setData(rowData);



win.add(aTableView);
