    alert(hold); will show empty string because HTTPClient requests are asynchronous. You can only get response in the callback onload.

    Try this:

    var hold = '';
function loadXML() {
    var xhr = Titanium.Network.createHTTPClient();
    xhr.open('GET', 'http://hids/abc/xyz.xml');
    xhr.onload = function() {
        var hold = this.responseXML;
        alert(hold);
    };
    xhr.send();
} 
//alert(hold);
