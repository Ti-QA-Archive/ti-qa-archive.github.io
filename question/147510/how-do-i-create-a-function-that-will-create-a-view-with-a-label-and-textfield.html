  • Make a common js module for the purpose and access it every time you want it
    Here is an example you can customize it according to your need.You can also send parameters like windows name and label height and width and other parameter

    function CBSScreen() {
   // alert('I am in CBS Screen')
    var displayCaps = Ti.Platform.getDisplayCaps();

    var self = Ti.UI.createWindow({
        title : "CBS Fantasy",
        backgroundColor : "white",
    //    navBarHidden : true
    });
    var tfUserName = Titanium.UI.createTextField({
        height : '10%',
        top : '20%',
        left : '10%',
        width : '80%',
        right : '10%',
        borderStyle : Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
    });
    self.add(tfUserName);
    var tfPassword = Titanium.UI.createTextField({
        height : '10%',
        top : '40%',
        left : '10%',
        width : '80%',
        right : '10%',
        borderStyle : Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
    });
    self.add(tfPassword);
    var labUserName = Titanium.UI.createLabel({
        text : 'User Name:',
        top : '13%',
        left : '10%',
        height : 'auto',
        width : 'auto',
        shadowColor : '#aaa',
        color : 'black',
        font : {
            fontSize : 28
        },
        textAlign : 'left'
    });
    self.add(labUserName);
    var labPassword = Titanium.UI.createLabel({
        text : 'Password:',
        left : '10%',
        top : '33%',
        height : 'auto',
        width : 'auto',
        shadowColor : '#aaa',
        color : 'black',
        font : {
            fontSize : 28
        },
        textAlign : 'left'
    });
    self.add(labPassword);
    self.addEventListener("android:back", function() {
    });
    return self;
};

module.exports = CBSScreen;

  • Hi Matt,
    In your code replace your param textField with the following code:

    var parameter = Ti.UI.createTextField({
    top: 5,
    value: param,            // you can use param as a value of textField not name of textField
    left: 200,
    width: 100,
    font: {fontSize: 14 },  
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});

    just add value as a property in your textField. You will get what you want.
    Hope it will help.
