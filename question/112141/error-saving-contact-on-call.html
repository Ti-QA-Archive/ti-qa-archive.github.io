Error saving contact on call

I'm trying to save a contact on call, but are having trouble making it work. Here is my code:

var phoneButton = Ti.UI.createButton({
    title:'Call',
    width:150,
    top: 100,
    height:50
});

phoneButton.addEventListener('click', function(e){
    var confirmCall = Titanium.UI.createAlertDialog({
        message:'Do you want to call?', 
        buttonNames: ['Yes','No']
    });
    confirmCall.show();
    confirmCall.addEventListener('click',function(e) {
        if (e.index == 0) 
        {
                var contact = Titanium.Contacts.createPerson();
                }
                contact.firstName = 'Firstname';
                contact.lastName = 'Lastname';
                contact.save();
        {
            call();

        }
    });
});

function call(){

    Titanium.Platform.openURL('tel:123456');    
}            


Titanium.UI.currentWindow.add(phoneButton);

UPDATE:

I changed

contact.save();

to 

Titanium.Contacts.save();

and now it works!
