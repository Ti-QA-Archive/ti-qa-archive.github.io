    Hello Juni….

    See this…

    
var win = Ti.UI.createWindow();

var scrollview = Ti.UI.createScrollView({
    top : 0,
    left : 0,
    width : 320,
    contentHeight : 'auto',
    backgroundColor : 'white',
    layout : 'vertical'
});
win.add(scrollview);

for(var i = 0; i < 9; i++) {
    var view = Ti.UI.createTextField({
        top : 30,
        height : 35,
        width : 250,
        borderColor : 'black'
    });
    scrollview.add(view);
}
win.open();
