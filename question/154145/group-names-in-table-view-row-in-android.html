                var tableview1=Ti.UI.createTableView({

    Hello renato,

    Provide headerTitle to tablView : 

    var tableview1=Ti.UI.createTableView({
     data:data,
    style:Titanium.UI.iPhone.TableViewStyle.GROUPED,
     headerTitle : "Sample"
});

    http://docs.appcelerator.com/titanium/latest/#!/api/Titanium.UI.TableView-property-headerTitle

    Let me know if it works

    • Good thanks Jigar, that link was exactly what I wanted. I created sections and it worked well in Android. :)

      // Create the first TableViewSection
var section1 = Ti.UI.createTableViewSection({
    headerTitle:'Header 1'
});
// use a loop to add some rows
for (var i=0; i < 4; i++) {
    section1.add(Ti.UI.createTableViewRow({
        title:'Row '+i
    }));
}
// do it all again...
var section2 = Ti.UI.createTableViewSection({
    headerTitle: 'Section 2'
});
for (var i=4; i < 10; i++) {
    section2.add(Ti.UI.createTableViewRow({
        title:'Row '+i
    }));
}
// Now, here's our table, and we're setting the data to hold the sections
var tv = Ti.UI.createTableView({
    data:[section1,section2]
});
