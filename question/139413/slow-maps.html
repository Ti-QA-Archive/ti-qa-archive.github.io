Slow Maps

var trackuser = function addCurrentPhonePositionListener() {
    central.phonePosition(the_map, zoom_lat, zoom_long, function(curr_lat, curr_long) {
        if (nearby_state.toString().trim() == "on") {
            Ti.App.TrackUser++;
            if (Ti.App.TrackUser == 1) {
                the_map.setLocation({
                    longitude : curr_long,
                    latitude : curr_lat,
                    animate : false,
                    latitudeDelta : zoom_lat,
                    longitudeDelta : zoom_long
                });
            }
        }
    });
};

Ti.App.addEventListener('trackUser', function(e) {
    Ti.API.info("Track User");
    if (win_controller.getWindowTitle() == map_view.win.title) {
        /*
         * Adding a Listener, which shows the current location of the user.
         * It also repositions the map based on the user.
         */
        Ti.API.info("nearby_state:"+nearby_state);
        if (nearby_state == "on") {
            Ti.API.info("A");
            if (Ti.App.TrackUser == 0) Titanium.Geolocation.addEventListener('location', trackuser);
        }
        else {
            Ti.API.info("B");
            Ti.API.info("counter:"+Ti.App.TrackUser);
            if (Ti.App.TrackUser > 0) {
                Titanium.Geolocation.removeEventListener('location',trackuser);
                Ti.App.TrackUser = 0;
            }
        }
    }
});
