    you can append items to the existing view like this…

    button.addEventListener("click", function (e) {
  var item2 = Ti.UI.createDashboardItem({
     badge:dboard.data.length, 
     image: "KS_nav_ui.png"
   });
  Titanium.API.info("Button clicked");
  dboard.setData([item2]);
  dboard.show();
  Titanium.API.info("dboard.data length "+dboard.data.length);
  Titanium.API.info("dboard.data " + dboard.data);
});

    BUT the bug I have noticed is that he dashboard view data property does not give back the expected results for the length of the data array

  • you do not add dashboard items with the "add" method, you add dashboard items by setting the data property on the dashboardview object

    var item = Ti.UI.createDashboardItem({
    label: 'View Messages', image: "KS_nav_ui.png"
});
var dashboard = Titanium.UI.createDashboardView({
    data:[item]
});
dashBoardWindow.add(dashboard);
