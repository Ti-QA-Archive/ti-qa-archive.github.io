  • Titanium.UI.currentTab.open this code open window with current tab. you can not use transition with it

    while below code will open window on top of tabgroup separately..

    var winnew = Titanium.UI.createWindow({

        navBarHidden : false,
    backgroundColor : 'red'
});
winnew.open({
    transition : Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT
});

    window : win2
});

tabGroup.addTab(tab2);
tabGroup.open();
