callback in controler

Hi,

if I call ctrl.nearme.getLocation(&#39;Purpose&#39;,myFunc); all works fine.

But if I call ctrl.nearme.getProperties(myFunc); the the error callback is not a function.

Whats going wrong?

ctrl.nearme = ( function() {
    var api = {};
    api.getLocation = function(purpose, callback) {
        try {
            Ti.Geolocation.purpose = purpose;
            Ti.Geolocation.preferredProvider = Ti.Geolocation.PROVIDER_GPS;
            Ti.Geolocation.getCurrentPosition(function(e) {
                if(e.error) {
                    callback(e.error);
                } else {
                    callback(e.coords);
                }
            });
        } catch(e) {
            callback(e);
        };
    };
    api.getProperties = function(callback) {
        if (typeof(datas.objects) == 'object') {
            callback(datas.objects);  // 'callback is not a function'
        }
    };
    return api;
}());

  • JavaScript will allow you to pass a single parameter to a two parameter function. However, it's most likely assuming your single parameter is the "purpose" param and your "callback" param is probably being set to null. Please use two params to call the existing function or update the function to detect and correct the supplied params. more info on javascript function overloading
