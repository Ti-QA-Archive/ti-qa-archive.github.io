  • If I understand you right, you're complaining that no data is show in the detail window?

    This is because you dont add any labels or views to the detail win.
    you should open the detail win after the data onload event has been fired, otherwise you will have no data when you open the detail win.

    I tried to fix your code with a (unformatted) label on the detail win:

    // this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');

// create tab group
var tabGroup = Titanium.UI.createTabGroup();

// // create base UI tab and root window //
var win1 = Titanium.UI.createWindow({
    title : 'Air Line',
    backgroundColor : '#fff'
});
var tab1 = Titanium.UI.createTab({
    icon : 'KS_nav_views.png',
    title : 'Air Line',
    window : win1
});

var search = Titanium.UI.createSearchBar({
    barColor : '#000',
    showCancel : true,
    height : 43,
    top : 0,
});

var table = Titanium.UI.createTableView();

var url = "http://api.milwaukeecounty.org/MobileAPI.svc/airlineNames";
var airlineNames = Ti.Network.createHTTPClient({
    onload : function(e) {
        Ti.API.info(this.responseText);
        //alert('success');
        var myData = JSON.parse(this.responseText);
        var airThingy = [];
        for(var i = 0; i < myData.length; i++) {
            Ti.API.info(myData[i]);
            var name = {
                title : myData[i]
            };
            airThingy.push(name);
        }
        table.setData(airThingy);
        Ti.API.info(airThingy);
    }
});
airlineNames.open("GET", url);
airlineNames.send();

table.addEventListener('click', function(e) {

    var tabWin = Titanium.UI.createWindow({
        backgroundColor : '#00ff00',
        title : 'Details Window'
    });

    var url = 'http://api.milwaukeecounty.org/MobileAPI.svc/getDepartures';
    var departure = Ti.Network.createHTTPClient({
        onload : function(e) {
            Ti.API.info(this.responseText);
            //alert('success');
            var myData = JSON.parse(this.responseText);


            //test label to show data in detail window (unformated)
            var detailLabel = Ti.UI.createLabel({
                text : this.responseText
            });
            //add test-Label to detail window
            tabWin.add(detailLabel);

            //start the opening process here - it will open the win AFTER the data has arrived
            if(Ti.Platform.name != 'android') {
                var closeButton = Ti.UI.createButton({
                    title : 'close',
                    top : 0,
                    left : 0,
                    height : 30,
                    width : 100
                });
                tabWin.rightNavButton = closeButton;
                closeButton.addEventListener('click', function(e) {
                    tabWin.close();
                });
            }

            //tabWin.add(closeButton);
            tabWin.open({
                modal : true
            });

        }
    });

    departure.open("GET", url);
    departure.send();

});
win1.add(table);

//tabWin.open();

tabGroup.addTab(tab1);
tabGroup.open();

    At first glance (although hard to read) your code looks fine - reference the KitchenSink for transversing JSON - you may be doing it incorrectly.

    Use fiddler on the API to determine the content returned.
