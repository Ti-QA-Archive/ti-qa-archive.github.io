  • Hi

    This is a much more complicated answer than your question allows for as you have not specified where the reference is being called from and what it needs to open.

    However the general answer for your question and assuming tabs;

    var win1 = Ti.UI.createWindow({
    title: 'One'
});
var tab1 = Titanium.UI.createTab({
    icon: 'myicon.png',
    title: 'One',
    window: win1
});
var btn = Ti.UI.createButton({
    title: 'Press me'
});
btn.addEventListener('click', function (e) {
    var win = Ti.UI.createWindow({
        title: 'New Window'
    });
    tab1.open(win, { animated : true });
});
win1.add(btn).
var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);
tabGroup.open();

    • A specific tableView method for comparison.

      var data = [
    { title: 'One' },
    { title: 'Two' },
    { title: 'Three' }
];
var tbl = Ti.UI.createTableView({
    data; data,
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
tbl.addEventListener('click', function (e) {
    Ti.API.info('row: ' + JSON.stringify(e.row));
    var win = Ti.UI.createWindow({
        title: e.row.title
    });
    tab1.open(win, { animated : true });
});
win.add(tbl);
