    Hi nhu thep nguyen,

    Please try this code. I hope that it will helpful for you. Here i have used only one line. you can modify as your need

    fb.requestWithGraphPath('me/photos', {}, 'GET', function(e){
            if (e.success) {
               var pic=JSON.parse(e.result);
               alert(pic.data.length);
               var data = Ti.UI.createImageView({ image:pic.data[0].picture }); // first image
               var showAlbum = Ti.UI.createWindow({
                    title: 'My Album',
                    backgroundColor: 'white'
                });

                showAlbum.add(data);
                showAlbum.open({modal:true});

            } else {
                if (e.error) {
                    alert(e.error);
                } else {
                    alert("Unkown result");
                }
            }
        });
