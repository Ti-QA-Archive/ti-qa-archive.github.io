TableViewRow

I need to change a backgroundImage each time it is clicked. I need to know how can I do this. On this code below a backgroundImage changes at the last row.

for (var i; i< 5; i++){
var row = createTableViewRow({
height: 40
}

var myBtn = Ti.UI.createButton({
         backgroundImage = 'image1.png'.
     height:32,
     width: 32,
     left: 2,
     btnno: i+1
});
row.add(myBtn);

}
myBtn.addEventListener('click', function(e){
try{
switch(myBtn.backgroundImage)
{
case 'image1.png':
mybtn.backgroundImage = 'image2.png';
break;
case 'image2.png':
mybtn.backgroundImage = 'image3.png';
break;
case 'image3.png':
mybtn.backgroundImage = 'image1.png';
break;
}
}
catch(e)
{
alert(e.message);
}
});

  • Hi

    Please try the below code. It works successfully in my pc. So i hope , it will work in your emulator

    var win=Ti.UI.createWindow({
    layout:'vertical',
    backgroundColor:'#fff'
});
var aTableView = Ti.UI.createTableView();

win.add(aTableView);

for (var i=0; i< 5; i++)
{ 
var row = Ti.UI.createTableViewRow({
    height:40
});

var myBtn = Ti.UI.createButton({
     backgroundImage : 'KS_nav_ui.png',
     height:32,
     width: 32,
     left: 2,
     btnno: i+1
});
row.add(myBtn);
aTableView.appendRow(row);
myBtn.addEventListener('click', function(e){
    changeImage(e);
});

}
function changeImage(es)
{
    try{
          switch(es.source.backgroundImage) 
          { 

              case 'KS_nav_ui.png': 
              es.source.backgroundImage = 'KS_nav_views.png'; 
              break; 
              case 'KS_nav_views.png': 
              es.source.backgroundImage = 'KS_nav_ui.png'; 
              break; 
              case 'KS_nav_ui.png': 
              es.source.backgroundImage = 'KS_nav_views.png'; 
              break; 
          } 
       } 
          catch(e) 
          { 
              alert("e.message"); 
          } 
}
win.open();
