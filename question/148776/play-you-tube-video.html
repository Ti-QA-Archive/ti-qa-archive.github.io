  • Hello Rahman,

    Just paste this code into your app.js file

    
var vidWin = Titanium.UI.createWindow({
    title : 'Video View Demo',
    backgroundColor : '#fff'
});

var videoPlayer = Titanium.Media.createVideoPlayer({
    top : 2,
    autoplay : true,
    backgroundColor : 'blue',
    height : 300,
    width : 300,
    mediaControlStyle : Titanium.Media.VIDEO_CONTROL_DEFAULT,
    scalingMode : Titanium.Media.VIDEO_SCALING_ASPECT_FIT
});

videoPlayer.url = 'http://www.youtube.com/watch?v=bSiDLCf5u3s'; // replace this url with your video url
vidWin.add(videoPlayer);
vidWin.open();

    for more info about video please take a look Video Player.
