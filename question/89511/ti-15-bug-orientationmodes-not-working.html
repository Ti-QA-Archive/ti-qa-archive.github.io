  • 1 way:

    var win = Ti.UI.createWindow({
    width: 1024,
    heigth: 768,
    backgroundColor: '#fff',

    orientationModes : [
Titanium.UI.PORTRAIT,
Titanium.UI.LANDSCAPE_RIGHT,
    Titanium.UI.LANDSCAPE_LEFT
    ]
});

    2nd way:

    Titanium.UI.orientation = Titanium.UI.LANDSCAPE_LEFT;

    and there is another way by using a gesture event and listen to orientation changes.
