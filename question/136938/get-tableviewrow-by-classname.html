Get TableViewRow by className?

I am using insertRowAfter on a TableView on a swipe event to make an accordian-like menu item slide out under a row to show more options. I want to make it so when you swipe another row, that previously added row slides back up and gets deleted. The row I'm adding that acts as the menu is given a className. Is there any way to target a TableViewRow object by className?

I've tried fiddling around with saving the index of the swiped row and adding 1 to it, but that didn't seem to work.

var rowMenu = Titanium.UI.createTableViewRow({
    top:28,
    height:20,
    backgroundColor:'#000000',
    touchEnabled:false,
    className:'bus_menu'
});

var openMenuRow;

row.addEventListener('swipe', function(e){
    if(e.direction = 'left'){
        if(openMenuRow){
            tableView.deleteRow(openMenuRow + 1,{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.DOWN});
        }

        openMenuRow = e.index;
        tableView.insertRowAfter(e.index,rowMenu,{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.DOWN});
    }
});

What I have in mind is basically getting ALL the rows with the className 'bus_menu' and calling deleteRow on them.

  • So, I couldn't figure out a way to loop through all the rows in my TableView (TableView.data, Tableview.getData(), and TableView.getChildren() were all empty or returned null) so I ended up detecting the rows with e.index. It's not pretty, but it seems to work. I'll share it here:

    var openMenu;

tableView.addEventListener('swipe', function(e){
    if(e.direction = 'left'){
        //do nothing if they swipe the same row twice
        if(e.index !== openMenu - 1){
            //only delete a row if a menu is already open
            if(openMenu){
                //depending if you swipe before or after the previously opened "menu" row, we have to change our target index for where to add the NEW "menu" row and what to set the openMenu index value to, because all the indexes have changed.
                var diff = e.index - openMenu;
                if(diff > 0){
                    tableView.deleteRow(openMenu,{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.TOP});
                    tableView.insertRowAfter(e.index - 1,{title:'New Row Menu'},{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.BOTTOM});
                    openMenu = e.index;
                }else if(diff < 0){
                    tableView.deleteRow(openMenu,{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.TOP});
                    tableView.insertRowAfter(e.index,{title:'New Row Menu'},{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.BOTTOM});
                    openMenu = e.index + 1;
                }
            }else{
                tableView.insertRowAfter(e.index,{title:'New Row Menu'},{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.BOTTOM});
                openMenu = e.index + 1;
            }
        }
    }
});

//adding a click listener to close any open "menu" row if you tap anywhere.
tableView.addEventListener('click', function(){
    if(openMenu){
        tableView.deleteRow(openMenu,{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.TOP});
        openMenu = null;
    }
});

    I can and will optimize this. It would be really awesome if I knew a way to somehow loop through all the rows a TableView currently has, and remove specific rows with a certain TableViewRow.className. But I'll leave this here for anyone who has the same issue until someone finds a solution.
