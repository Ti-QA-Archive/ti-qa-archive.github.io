    var xhr = Titanium.Network.createHTTPClient({

    onload : function(e) {
        // Ti.API.info(this.responseText);
        if (this.status == 200) {
            var response = JSON.parse(this.responseText);
            for (var i = 0; i < response.feed.entry.length; i++) {
                var obj = response.feed.entry[i];
                Ti.API.info('Player : ' + obj.media$group.media$player[0].url);
                Ti.API.info('Title : '+obj.media$group.media$title.$t);
                Ti.API.info('Thumbnail : ' + obj.media$group.media$thumbnail[0].url);
                Ti.API.info('Seconds : ' + obj.media$group.yt$duration.seconds);

            }
        }
    },
    onerror : function(e) {

        obj.callback('error');
    },
    timeout : 120000
});
