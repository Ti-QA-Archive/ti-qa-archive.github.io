  • Here is a sample code for list view grouping

    var fruitSection = Ti.UI.createListSection({ headerTitle: 'Fruits'});
    var fruitDataSet = [
    {properties: { title: 'Apple', canEdit: true, canMove: true,height : 20}},
    {properties: { title: 'Banana', canEdit: true, canMove: true,height : 20}}
    ];
    fruitSection.setItems(fruitDataSet);
    $. sections.push(fruitSection);

    var vegSection = Ti.UI.createListSection({ headerTitle: 'Vegetables'});
    var vegDataSet = [
    {properties: { title: 'Carrots'}},
    {properties: { title: 'Potatoes'}}
    ];
    vegSection.setItems(vegDataSet);
    $. sections.push(vegSection);

    var fishSection = Ti.UI.createListSection({ headerTitle: 'Fish'});
    var fishDataSet = [
    {properties: { title: 'Cod'}},
    {properties: { title: 'Haddock'}}
    ];
    fishSection.setItems(fishDataSet);
    $. sections.push(fishSection);

    for more details go throughout the below doc

    • If you want to customise your section area then you need to create a header view inside that section.
      here is one sample
      <Window fullscreen="true">
      <ListView>
      <ListSection id="LstSection">
      <HeaderView>
      <View backgroundColor="#DDD" layout="horizontal" height="Ti.UI.SIZE">
      <Label id="lstHeader"></Label>
      <ImageView image="/images/close.png" width="Ti.UI.SIZE" height="Ti.UI.SIZE"></ImageView>
      </View>>
      </HeaderView>
      </ListSection>
      </ListView>
      </Window>

      var fruitDataSet = [
      {properties: { title: 'Apple', canEdit: true, canMove: true,height : 20}},
      {properties: { title: 'Banana', canEdit: true, canMove: true,height : 20}}
      ];
      $.lstHeader.setText("Fruits"); //Set your section Title here
      $.LstSection.setItems(fruitDataSet);

      Hope this helps you..
