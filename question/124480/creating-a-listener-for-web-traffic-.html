  • Here is a solution, ugly, incomplete, but it works :)
    You will have to play a lot with and optimize it(save images once they are generated, delete the db once is installed and images saved, use a temp folder, etc), but may be a start.

    Makes sense, is what you need?

    page.html

    <!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Transitional//EN"
    "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd">

<html xmlns="http://www.w3.org/1999/xhtml" xml:lang="en" lang="en">
<head>
    <meta http-equiv="Content-Type" content="text/html; charset=utf-8"/>
</head>

<body>the next text is an image

<img id="img" width="300"/>

</body>
</html>

    app.js

    var window = Ti.UI.createWindow({  
    backgroundColor:'#fff',
    fullscreen: true
});



var webview = Ti.UI.createWebView({
    url:'page.html'
});


webview.addEventListener('load', function(){
    var img = Ti.UI.createLabel({
        text:'Boo',
        color:'#f00',
        backgroundColor:'#ff0',
        font:{
            fontSize:40,
            fontWeight:'bold'
        }
    }).toImage();

    var testfile = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, 'text.png');
    testfile.write(img);
    webview.evalJS("document.getElementById('img').src='"+testfile.nativePath+"'");
})


window.add(webview);



window.open();
