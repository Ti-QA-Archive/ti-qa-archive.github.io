    var json = JSON.parse(this.responseText);
var imgs = [];
for(i=0;i < json.length;i++)
{ 
    imgs=json[i].url.thumb;
} 

var view = Titanium.UI.iOS.createCoverFlowView({
    images:imgs,
    backgroundColor:'#000'
});

win.add(view);
