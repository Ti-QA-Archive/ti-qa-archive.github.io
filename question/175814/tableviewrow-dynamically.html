  • Hello Kusuma,

    You are declare one TableViewRow Object in the XML file…

    See this Sample..:

    index.xml

    <Alloy>
    <Window id="samplewindow">
        <TableView id="tableviewsample">
        </TableView>
    </Window>
</Alloy>

    index.js

    for (var c = 0; c < 10; c++) {
    $.tableviewsample.appendRow(Alloy.createController("row", {
        "title" : "Row " + c
    }).getView());
}

    row.xml

    <Alloy>
    <TableViewRow id="tableviewrowsample">
        <Label id="namelabel></Label>
    </TableViewRow>
<Alloy>

    row.js

    var ARG = arguments[0] || {};

//Set text to Label
$.namelabel.text = ARG.title;
