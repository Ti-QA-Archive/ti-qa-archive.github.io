Opening a detailview from annotationview - Android.

Im having problem opening a detailview, another window, from an annotationview in Android. This code works fine for IOS but in Android it will not open, no API connection.

SDK 3.1.0GA
Android ver 2.3 - 4.2 

Ti.App.addEventListener("missionspop", function(){


        api.missions.near.forEach(function(mi){
            var pinPay = Ti.UI.createLabel({
                text: "R$ " + Number(mi.pay).toFixed(2).replace(".",","),                            
                color: "#000",
                height: 40,
                width: 40,
                touchEnabled: false
            });


        var pin = map.createAnnotation({
               image: "img/transactions/logo@2x.png",
               latitude: mi.location[0],
               longitude: mi.location[1],
               title: mi.title,
               subtitle: mi.address,
               leftView: pinPay,
               animate: false,
               mission: mi                                                
             });

            m.addEventListener("click", function(e){
                if (e.clicksource == "leftPane") {
                    var missionView = require("mission")(this.mission);
                missionView.open();
                }
            });

            m.addAnnotation(pin);
        });
    });

Thanks in advance!
