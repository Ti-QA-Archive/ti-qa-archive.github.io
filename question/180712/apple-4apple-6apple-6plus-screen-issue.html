    Hi Aparna,

    It just because iOS not because of Phone. From iOS 7 or greater Apple make status bar transparent. that's y your label and button start from top..

    This problem i resolved by using add view with 20 height if is > iOS7.

    e.g

    
function isIOS7Plus() {

    if (Titanium.Platform.name == 'iPhone OS') {
        var version = Titanium.Platform.version.split(".");
        var major = parseInt(version[0], 10);
        if (major >= 7) {
            return true;
        }
    }
    return false;
}

var isIOS7Plus = isIOS7Plus();


var win = Ti.UI.createWindow({
        modal : true,
        backgroundColor : 'white',
    layout : "vertical"
    });
    //title
var statusBar = Ti.UI.createView({
    top : 0,
    height : ((isIOS7Plus) ? 20 : 0),
    width : "100%",
    backgroundColor : "white", //Whatever you want
});
win.add(statusBar);

var navView = Ti.UI.createView({
    top : 0,
    height : 45,
    width : "100%",
    backgroundColor : "white", //Whatever you want
});
win.add(navView);

    var titlelbl = Ti.UI.createLabel({
        top : 0,
        left : 0,
        right : 0,
        height : Ti.Platform.displayCaps.platformHeight*0.065,
        text : 'Window',
        textAlign : 'center',
        color : '#fff',
        font : {
            fontSize : 18,
            fontWeight : 'bold'
        },
        backgroundColor : 'black'
    });
    navView.add(titlelbl);
    //ios back button
    var backbtn = Ti.UI.createButton({
        title : "<Back",
        top : 0,
        style : Titanium.UI.iPhone.SystemButtonStyle.BORDERED,
        height:Ti.Platform.displayCaps.platformHeight*0.065,
        width:Ti.Platform.displayCaps.platformWidth*0.2,
        backgroundColor:'white',
        color:'black',
        textAlign:'left',
        left:0
    });
    navView.add(backbtn);
    var hbtn = Ti.UI.createImageView({
        image : '/images/home.png',
        width : Ti.Platform.displayCaps.platformWidth*0.15,
        height : Ti.Platform.displayCaps.platformHeight*0.065,
        right : 0,
        top : 0
    });
    navView.add(hbtn);
