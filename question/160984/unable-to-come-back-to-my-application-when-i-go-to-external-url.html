  • Hi Ramesh/Mitul,

    You can try the code as below which should fix your problem:

    function openUrl(){

     var openurl = "https://maps.google.com/maps";

//The web view allows you to open an HTML5 based view which can load either local or remote content.

var btnBack = Titanium.UI.createButton({
    title : ' < Go Back ',
    top : 0,
    left : 0,
    height : '10%',
});

    var webView = Titanium.UI.createWebView({
    url : openurl,
    canGoBack : true,
    top : '10%',
    height : '90%',
    });

    btnBack.addEventListener('click', function() {
    webView.goBack();
});

var win1 = Titanium.UI.createWindow({  
    backgroundColor:'#fff',
});

win1.add(btnBack);

win1.add(webView);
win1.open();

    }
