  • Hello Mick,

    use this: 

    var allRows = [
    ['Chest', 'size', true], 
    ['Neck','size', true], 
];

var view = Ti.UI.createView({
    top : "30dp",
    layout : "vertical",
    height : "auto"   
});
$.win.add(view);

for (var i = 0;i < allRows.length; i++){
        if (allRows[i][2]){
            var label = Ti.UI.createLabel({
                top : "25dp",
                font : {
                    fontSie : "13dp"
                },
                text: allRows[i][0],
                id: 'heading' + allRows[i][1],
            });
            view.add(label);
        }
}
$.win.open();

    • @Jigar nice.

      I few revisions to consider, not required but simplifies the loop and creates a handler to add the label as this will make any future event handling easier.

      var allRows = [
    ['Chest', 'size', true], 
    ['Neck','size', true]
];

var view = Ti.UI.createView({
    height: Ti.UI.FILL,
    layout: "vertical",
    top: "30dp",
    width: Ti.UI.FILL
});
$.win.add(view);
var addLabel = function (obj) {
    var label = Ti.UI.createLabel({
        font : {
            fontSie : "13dp"
        },
        id: 'heading' + obj.option,
        text: obj.title,
        top : "25dp"
    });
    return label;
};
_.each(allRows, function (row) {
    if (row[2]) {
        view.add(addLabel({
            option: row[1],
            title: row[0]
        }));
    }
});
$.win.open();
