            fontWeight : 'bold'
        },
        message : 'Loading...',

        top : 10,
        left : 10,

    });

    parent_obj.win.add(activityIndicator);
}

    var style;

  style = Ti.UI.iPhone.ActivityIndicatorStyle.DARK;

else {
  style = Ti.UI.ActivityIndicatorStyle.DARK;

var activityIndicator = Ti.UI.createActivityIndicator({

  height:Ti.UI.SIZE,
  width:Ti.UI.SIZE
