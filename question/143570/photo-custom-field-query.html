    JSONObject photoResponseJSON = photoResponse.getResponseData();

    CCMeta photoMeta = photoResponse.getMeta();

    if("ok".equals(photoMeta.getStatus()) && photoMeta.getCode() == 200 && "createPhoto".equals(photoMeta.getMethod())) {
    try {
    JSONArray photos = photoResponseJSON.getJSONArray("photos");
    JSONObject firstPhoto = photos.getJSONObject(0);
    return new PluginResult(PluginResult.Status.OK, firstPhoto.getString("id"));
