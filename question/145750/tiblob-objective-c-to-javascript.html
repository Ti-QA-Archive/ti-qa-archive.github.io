  • Hi,
    There is no need of module for importing image from Gallery , You can use this code to import images to your application:

    
Titanium.Media.openPhotoGallery({
    mediaTypes:[Ti.Media.MEDIA_TYPE_PHOTO],

    success:function(e) {
          var image = e.media;
          filename = 'someFileName.jpeg';
          var f = Titanium.Filesystem.getFile(Titanium.Filesystem.tempDirectory, filename+ ".jpeg");
          f.write(image);
          file = f.getNativePath();

        }
});

    Here the image is getting saved in application data directory.

      here what i did : 

      void (^assetForURLResultBlock)(ALAsset *) = ^(ALAsset *result)
    {
        if( result != nil )
        {
            ALAssetRepresentation *rep = [result defaultRepresentation];
            //NSString* type = [result valueForProperty:ALAssetPropertyType];


                //UIImage *largeimage;
                //largeimage = [UIImage imageWithCGImage:[rep fullScreenImage]];
            long size = [rep size];
            Byte *buffer = (Byte*)malloc(size);
            [rep getBytes:buffer fromOffset:0 length:size error:nil];
            NSData *data = [NSData dataWithBytesNoCopy:buffer length:size freeWhenDone:YES];


                NSDictionary *event = [NSDictionary
                                       dictionaryWithObjectsAndKeys:
                                       [[[TiBlob alloc] initWithData:data mimetype:nil] autorelease],
                                       @"media",
                                       //type,
                                       //@"type"
                                       nil];
                if (assetUrlCallback!=nil)
                {
                    [self _fireEventToListener:@"onAsset" withObject:event listener:assetUrlCallback thisObject:nil];
                }

        }
    };
