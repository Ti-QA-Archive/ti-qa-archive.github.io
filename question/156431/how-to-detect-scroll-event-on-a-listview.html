How to detect scroll event on a ListView?

Does anyone know of a way to detect a vertical scroll/swipe gesture on a ListView? I tried listening to swipe event but it only fires on left-to-right and right-to-left gestures, not up or down. I also tried listening on the parent View which resulted in the same behavior.

Thanks in advance for any tips or tricks! :)

  • Hi Brian Burns,

    Views added to the scroll view will be scrolled based on the size of the scrollable region of content. If a scrollable region fits within the size of its scroll view, the view will not scroll. You can follow this sample code; I hope it will be helpful for you. And you can also follow the link ScrollView

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

function createRow(i) {
    var row = Ti.UI.createView({
        backgroundColor : 'white',
        borderColor : '#bbb',
        borderWidth : 1,
        width : '100%',
        height : 70,
        top : 0,
        left : 0
    });

    // Create a Label.
    var aLabel = Ti.UI.createLabel({
        text : 'value' + i,
        color : '#000',
        font : {
            fontSize : 15
        },
        height : Titanium.UI.SIZE,
        width : Titanium.UI.FILL,
        top : 10,
        left : '10%',
        textAlign : 'center'
    });

    row.add(aLabel);
    var listView = Ti.UI.createListView();
    listView.add(row);
    return row;
}

var scrollView = Ti.UI.createScrollView({
    bottom : 120,
    contentHeight : 'auto',
    layout : 'vertical'
});

for (var i = 0; i <= 20; i++) {
    var row = createRow(i);
    scrollView.add(row);
}
win.add(scrollView);

win.open();
