Activity Indicator - Crashes on Andriod

Core.prototype.addActivity = function(parent_obj, txt) {
    var spinner_style = Ti.UI.iPhone.ActivityIndicatorStyle.PLAIN;
    parent_obj.activity_view = Ti.UI.createView({
        backgroundColor : '#5e5e5e',
        width : '70%',
        height : '10%',
        borderRadius : 20,
        top : 330
    });
    parent_obj.activity = Ti.UI.createActivityIndicator({
        color : 'white',
        font : {
            fontFamily : 'Helvetica Neue',
            fontSize : 18,
            fontWeight : 'normal'
        },
        message : txt,
        style : (Ti.Platform.name === 'iPhone OS' ? Ti.UI.iPhone.ActivityIndicatorStyle.PLAIN : Ti.UI.ActivityIndicatorStyle.PLAIN),
        top : 'auto',
        left : 'auto',
        height : 'auto',
        width : 'auto'
    });
    parent_obj.activity_view.add(parent_obj.activity);
    if (!isAndroid)
        parent_obj.win.add(parent_obj.activity_view);

    /*
     * Add Ability to remove activity.
     *
     */
    parent_obj.activity_view.addEventListener('click', function(e) {
        object_ref.removeActivity(parent_obj);
    });
    parent_obj.activity.show();
}

I am using this for the iPhone, and works well, but crashes the andriod app.
