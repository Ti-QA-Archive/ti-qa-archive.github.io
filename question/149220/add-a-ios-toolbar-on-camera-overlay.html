add a ios toolbar on camera overlay

Hi, i want to add a toolbar on camera overlay … I am building a custom camera overlay which includes an imageview(to hold the thumbnail image of the picture taken ) and a camera button to take picture

Below is my code 

var container = Titanium.UI.createView({
    });
    containerflexSpace = Titanium.UI.createButton({
        systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
    });
    var containerCameraButton = Titanium.UI.createButton({
        width : 100,
        height : 30,
        bottom : 3,
        borderRadius : 3,
        style : Titanium.UI.iPhone.SystemButtonStyle.DONE,
        title : 'Take Picture',
        zIndex : 1
    });
    var containerImageView = Titanium.UI.createImageView({
        bottom : 3,
        left : 5,
        height : 30,
        width : 30,
        zIndex : 4
    });
    containerparentView.add(containerImageView);
    var containerToolbar = Titanium.UI.iOS.createToolbar({
        items : [containerImageView, containerflexSpace, containerCameraButton, containerflexSpace],
        barColor : '#498ECF',
        bottom : 3
    });
    container.add(containerToolbar);
    containerImageView.addEventListener('click', function() {
        var GalleryWindow = require('components/windows/GalleryWindow').GalleryWindow;
        Ti.App.navController.open(new GalleryWindow());
    });
    containerCameraButton.addEventListener('click', function() {
        Titanium.Media.takePicture();
    })

The actual problem is my click events are transfered to the toolbar … but i want that events on the imageview and camera button respectively …
