Here Creates a navigation group with the first window colored red. Press the button to open the blue window. Use the back button to return to the red root window. The back button is displayed in the left part of the navigation.

    Check the attachment

    Test Case:

    index.xml

    <Alloy>
    <NavigationWindow id="win1" platform="ios">
        <Window id="win2" title="Red Window" backgroundColor="red">
            <Button id="button" onClick="openBlueWindow">Open Blue Window</Button>                      
        </Window>
    </NavigationWindow>
</Alloy>

    index.js

    function openBlueWindow(e) {
    win3 = Alloy.createController('bluewin', {
       // id: uid,
       // module : 'aT'
    }).getView();

    $.win1.openWindow(win3);
}

$.win1.open();

    bluewin.xml

    <Alloy>
    <Window id="win3" title="Blue Window" backgroundColor="blue">
        <Button onClick="closeWindow">Close Window</Button>
        <LeftNavButton>
                        <Button id="leftOne" title="back" onClick="backWindow" ></Button>
          </LeftNavButton>

    </Window>
</Alloy>

    bluewin.js

    function closeWindow(){
    $.win3.close();
}

 function backWindow(){
    $.win3.close();
}
