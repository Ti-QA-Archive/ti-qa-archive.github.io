function registerUser(name, email, mobile, address, pass1){
  var request = Titanium.Network.createHTTPClient();
  var url = "http://103.21.171.61/pakarga/topup.svc/registerjson?account=PK10001&email=" +email+ "&mobile=" +mobile+ "&pass=" +pass1+ "&name="+ name+ "&address="+address;
  request.open("GET",url);

  request.onload = function(){
            var arrData = [];

            arrData = eval('('+this.responseText+')');
            var result = arrData.status;
            if(result == "success"){
                 var newWin=Ti.UI.createWindow({
                        url:'regResults.js'
                    });            
                Ti.UI.currentTab.open(newWin,{animate:true});    
            } else if (result == "failed"){
                var alrt_Sorry = Titanium.UI.createAlertDialog({
                  title: 'Sorry!',
                  message: 'According to our database, that username is already in use.',
                  buttonNames: ['OK']
                  });
                  alrt_Sorry.show();
            } else {
                  var alrt_Fail = Titanium.UI.createAlertDialog({
                    title: 'Error!',
                    message: 'It seems we are experiencing problems... please try again later.',
                    buttonNames: ['OK']
                  });
                  alrt_Fail.show();
                  }

  };
  request.send();
};
