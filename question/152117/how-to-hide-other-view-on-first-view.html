    Hi Prerna,

    For hiding and showing you can use switch case.

    var index;

function viewShowFunction(index){
    switch(index) {
        case 1 :
            view1.hide();
            break;
        case 2 :
            view2.hide();
            break;
        case 3 :
            view3.hide();
            break;
    }
}


function view1Click(){
    view1.show();

    viewShowFunction(index)

    index=1;
}

function btn2Click(){
    view2.show();

    viewShowFunction(index)

    index=2;
}

function btn3Click(){
    view3.show();

    viewShowFunction(index)

    index=3;
}
