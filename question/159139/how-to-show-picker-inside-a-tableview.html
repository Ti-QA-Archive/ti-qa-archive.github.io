How to show Picker inside a TableView

I'm building an application where I have a TableViewRow that represents a date, and when I tap it I want to reveal a Picker below the row. I want the behavior to be identical to how the Starts and Ends fields on the Add Event screen in the iPhone Calendar app. According to this question (http://developer.appcelerator.com/question/119198/animating-row-height-changes) it appears that animating a TableViewRow isn't supported, and when I tried using the setTimeout workaround (setInterval in my example) I get nasty visual artifacts where the row below it doesn't change position smoothly. Is there some feature I'm missing that might let me do what I need?


    var PICKER_ROW_HEIGHT_HIDDEN = 1,
        PICKER_ROW_HEIGHT_VISIBLE = 162,
        PICKER_ROW_INCREMENT = 10,
        PICKER_ROW_INTERVAL = 10;

    function animatePickerRow(row, show) {
        var intervalId,
            rowHeight = show ? PICKER_ROW_HEIGHT_HIDDEN : PICKER_ROW_HEIGHT_VISIBLE;

        function animateRow() {
            if (show) {
                rowHeight = Math.min(PICKER_ROW_HEIGHT_VISIBLE, rowHeight + PICKER_ROW_INCREMENT);
            } else {
                rowHeight = Math.max(PICKER_ROW_HEIGHT_HIDDEN, rowHeight - PICKER_ROW_INCREMENT);
            }

            row.setHeight(rowHeight);

            if ((show && rowHeight >= PICKER_ROW_HEIGHT_VISIBLE) || (!show && rowHeight <= PICKER_ROW_HEIGHT_HIDDEN)) {
                clearInterval(intervalId);
            }
        }

        intervalId = setInterval(animateRow, PICKER_ROW_INTERVAL);
    }
