  • This is what i wanted to see,

    The url property of the window is somehow bugged then. This runs a fresh .js file each time it is called. So everytime you open the same win it will run a .js file again and again.

    To fix this you can create a class which you call when you need it. In the class you open the window the way you do now.

    Heres an example hope it fixes your problem:

    // photos.js

var class_photos = function photos(){
    var win = Titanium.UI.currentWindow;

    function loadPhotos(){
        ... loads views and text etc.. and add its to win.
    }

    win.addEventListener('open', loadPhotos);

    win.addEventListener('android:back', function (e) {
        win.close();
    });

    // open window when this class is called
    Titanium.UI.currentTab.open(win,{animated:true});
};

    And

    // dash.js

// include the class, will do nothing untill you call it
Ti.include('photos.js');

var photos = Titanium.UI.createWindow({
        title:'Accident Photos',
        navBarHidden:false
        });

var item4 = Titanium.UI.createImageView({
        url:'/images/dashboard/photos_off.png',
        width:75,
        height:90,
        left:newWidth,
        bottom:90
            });
    win.add(item4);
item4.addEventListener('click', function(e)
{
    // Now lets call the class
    class_photos();                
});
