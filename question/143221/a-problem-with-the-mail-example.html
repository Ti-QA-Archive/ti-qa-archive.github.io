A problem with the mail example

Hello everybody,

I try to test this mail example using Titanium IDE (using the SDK 2.1.0) and the Android Emulator 2.2:

var myWindow = Titanium.UI.createWindow({

    title:"My Email Test",
    backgroundColor:"#CCFF00",
    exitOnClose:true
});


var myLabel = Titanium.UI.createLabel({
    text:"Your text will be here",
    width:"auto",
    height:"auto"
});

var myButton = Titanium.UI.createButton({

    title:"Contact us",
    bottom:50,
    height:45,
    width:145
});

var eMailDialog = Titanium.UI.createEmailDialog({
    //html:true,
    toRecipients:["x.xxx@gmail.com"],
    subject:"Message test",
    messageBody:"Hello from the device",

});


myButton.addEventListener("click",function(e){
    eMailDialog.open()
});

eMailDialog.addEventListener("complete",function(e){

    if(e.result == eMailDialog.SENT){
        myLabel.text = "Message sent !";
    }
    else{
        alert("Statuts"+e.result),
        myLabel.text = "Message not sent !";
    }

});

myWindow.add(myLabel);
myWindow.add(myButton);
myWindow.open();

Unfortunately, I got this error message:

no applications can perform this action

Could you tell me why this didn't works.

    Test on a DEVICE or side load a mail app for testing (outside the scope of this Q&A)
