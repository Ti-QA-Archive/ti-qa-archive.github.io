  • Hi Mauro,

    Titanium 3.1.1 and here's code snippet. Works beautifully on iOS, not so much on Android

    for (var i = 0; i < Vars.exhibits[index].exhibits.length; i++) {
        var tableRow = Ti.UI.createTableViewRow({
            className : "exhibitdetails",
            height : rowHeight,
            selectionStyle : Ti.UI.iPhone.TableViewCellSelectionStyle.NONE,
            windowHeader : Vars.exhibits[index].exhibits[i].title,
            media : Vars.exhibits[index].exhibits[i].media,
            format : Vars.exhibits[index].exhibits[i].format,
        });

        var contentView = Ti.UI.createView({
            height : Ti.UI.FILL,
            width : Ti.UI.FILL,
        });

        var exhibitNumberView = Ti.UI.createView({
            top : 0,
            left : 0,
            height : Ti.UI.FILL,
            width : "80dp",
        });

        var exhibitGradientTop = Ti.UI.createView({
            top : 0,
            left : 0,
            height : "50%",
            width : Ti.UI.FILL,
            backgroundColor : Vars.exhibits[index].startColour,
        });

        var exhibitGradientBottom = Ti.UI.createView({
            bottom : 0,
            left : 0,
            height : "50%",
            width : Ti.UI.FILL,
            backgroundGradient : {
                type : "linear",
                startPoint : {
                    x : 0,
                    y : 0
                },
                endPoint : {
                    x : 0,
                    y : "100%"
                },
                colors : [{
                    color : Vars.exhibits[index].endColour
                }, {
                    color : Vars.exhibits[index].startColour
                }]
            },
        });

        var exhibitNumber = new Label("#FFFFFF", "Swiss 721", rowHeight / 2 + "dp", Vars.exhibits[index].exhibits[i].index);
        var exhibitImage = new Image(Vars.exhibits[index].exhibits[i].thumbnail);
        var tableContent = new Label("#FFFFFF", "Swiss 721", "", Vars.exhibits[index].exhibits[i].title);

        exhibitNumberView.add(exhibitGradientTop);
        exhibitNumberView.add(exhibitGradientBottom);
        exhibitNumberView.add(exhibitNumber);
        contentView.add(exhibitNumberView);
        contentView.add(exhibitImage);
        contentView.add(tableContent);

        exhibitNumber.height = Ti.UI.FILL, exhibitNumber.width = Ti.UI.FILL, exhibitNumber.zIndex = 1;
        exhibitNumber.textAlign = Ti.UI.TEXT_ALIGNMENT_CENTER;
        exhibitNumber.verticalAlign = Ti.UI.TEXT_VERTICAL_ALIGNMENT_CENTER;

        exhibitImage.top = 0, exhibitImage.left = "80dp", exhibitImage.height = "50%", exhibitImage.width = Ti.UI.FILL;

        tableContent.bottom = 0, tableContent.height = "50%", tableContent.width = Ti.UI.FILL, tableContent.left = "80dp";
        tableContent.backgroundColor = "#1A1819";
        tableContent.textAlign = Ti.UI.TEXT_ALIGNMENT_CENTER;
        tableContent.verticalAlign = Ti.UI.TEXT_VERTICAL_ALIGNMENT_CENTER;

        tableRow.add(contentView);
        tableData.push(tableRow);
    };
