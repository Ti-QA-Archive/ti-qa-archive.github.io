Using filesystem.resourceDirectory

How would I get a random image to load on a window from a folder using the filesystem.resourceDirectory? Also how would I get my event so when I clicked something the image would change to a different one?

  • Hello Richard,

    Try this one :

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});
var counter = 1;

var dirFullPath = Ti.Filesystem.resourcesDirectory + Ti.Filesystem.separator + 'images' + Ti.Filesystem.separator;
var dir = Titanium.Filesystem.getFile(dirFullPath);
var dirItems = dir.getDirectoryListing();

Ti.API.info('>>>>>>>> Length >>>' + JSON.stringify(dirItems));

var image = Ti.UI.createImageView({
    height : Ti.UI.SIZE,    //  you set also fixed 
    width : Ti.UI.SIZE,     //  you set also fixed
    image : '/images/' + dirItems[0]
});

var button = Ti.UI.createButton({
    top : 25,
    title : 'Change Image',
    height : 50,
    width : 200
});

button.addEventListener('click',function(e){
    image.setImage('/images/' + dirItems[counter]);
    counter++;
    if(counter>=dirItems.length)
    {
        counter = 0;
    }
});
win.add(image);
win.add(button);
win.open();

    Thanks. :)
