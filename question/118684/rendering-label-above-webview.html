Rendering label above webview?

I have created a label and webview that are both added to a window. The label is positioned toward the top of the window and the webview is below using the top property. The label also has a higher zIndex value than the webview…

Problem is, the label doesn't render above the webview, nor overlayed on it… Any ideas why it's not showing up?

Here's the code:
Ti.API.debug('Creating closeLabel');
closeLabel = Ti.UI.createLabel({
textAlign: 'center',
font: {
fontWeight: 'bold',
fontSize:22
},
color:'#900',
text: 'Close me (X)',
top: 5,
height: 14,
zIndex:2
});

 Ti.API.debug('Creating webView');

webView = Ti.UI.createWebView({
url: pUrl,
top: closeLabel.height + closeLabel.top,
width: 400,
height: view.height - closeLabel.height - closeLabel.top - view.borderWidth * 4,
scalesPageToFit:true,
zIndex:1,
autoDetect:[Ti.UI.AUTODETECT_NONE]
});

    closeLabel.show();
    window.add(webView);
    window.add(closeLabel);

    window.open();

    First, I suggest posting code using the tilda block for formatting purposes.
    Secondly, I don't see where you created the view component (which is being used to determine the webview's height and top properties).

    Also, here is some code that will make a label at the top and a webview under it. Not sure if it is what you are looking for, but there it is.

    var window = Titanium.UI.createWindow({  
    backgroundColor:'#fff'
});

var view = Ti.UI.createView({
    width:'100%', height:'100%', 
    borderColor:'#222222', borderWidth:2
});
window.add(view);

closeLabel = Ti.UI.createLabel({ 
    textAlign: 'center', 
    font: { fontWeight: 'bold', fontSize:18 }, 
    color:'#900', 
    text: 'Close me (X)', 
    top: 8, 
    height: 14, 
    zIndex:2 
});
view.add(closeLabel);


var webView = Ti.UI.createWebView({ 
    url: 'http://www.google.com', 
    top: (10 + closeLabel.top) + (closeLabel.size.height || closeLabel.height), 
    width: 400, 
    height: '100%', 
    scalesPageToFit:true, 
    zIndex:1, 
    autoDetect:[Ti.UI.AUTODETECT_NONE] 
});
view.add(webView);

window.open();

    Blessings!
