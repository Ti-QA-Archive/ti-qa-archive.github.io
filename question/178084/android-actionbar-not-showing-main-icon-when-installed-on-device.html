  • Since you asked nicely,

    This particular code is called from my controller file

    Which would indicate an alloy project. Yet, your main window is not actually using the $. before it as one would expect.

    In this example, using dk.napp.drawer in our project for the drawer ($.drawer).

    $.drawer.addEventListener('open', onNavDrawerWinOpen);

    function onNavDrawerWinOpen(evt) {
        this.removeEventListener('open', onNavDrawerWinOpen);

        if (this.getActivity()) {
            var actionBar = this.getActivity().getActionBar();

            if (actionBar) {
                Alloy.Globals.actionBar = actionBar;
                // Now we can do stuff to the actionbar  
                actionBar.setTitle('My App Title');
                // toggle the left window when the home icon is selected
                actionBar.setOnHomeIconItemSelected(function() {
                    $.drawer.toggleLeftWindow();
                });
            }
        }
    }

    That being said, all new projects are using com.tripvi.drawerlayout module, as its more native on android than dk.napp.drawer is.

    Other useful tools for your box: https://github.com/ricardoalcocer/tiactionbarhelper, https://github.com/ricardoalcocer/actionbarextras

    • Example that isnt using a drawer, so doesn't need the menu icon, but uses actionbarextras:

      $.timerWindow.addEventListener('open', function(e) {
        var activity = $.timerWindow.getActivity();
        if (activity) {
            abx.title = "My App Titler"
            activity.onCreateOptionsMenu = function(e) {
                var menu = e.menu;
                var MenuItemEvents = menu.add({
                    title: 'Events',
                    showAsAction: Ti.Android.SHOW_AS_ACTION_ALWAYS,
                    icon: '/images/dark-events.png'
                });

                MenuItemEvents.addEventListener('click', function(e) {
                    var events = Alloy.createController('eventListWindow').getView();
                    events.open();

                });

                var MenuItemSettings = menu.add({
                    title: 'Settings',
                    showAsAction: Ti.Android.SHOW_AS_ACTION_ALWAYS,
                    icon: '/images/dark-gear.png'
                });

                MenuItemSettings.addEventListener('click', function(e) {
                    var settings = Alloy.createController('settings').getView();
                    settings.open();

                });
            }
        }
