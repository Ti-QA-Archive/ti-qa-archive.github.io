  • hi,

    if you want to play more than one sound then instead of sound you should use audioPlayer to play sounds but as you want to play a sound at a time then you need to stop previous sound to listener current sound.

    sound1.addEventListener("touchstart", function(e){
    if(pickaxe){
        pickaxe.stop();
    }
    player = Ti.Media.createSound({url:"media/Hit19.wav"});
    player.play();
    currsound = 1;
    Ti.App.Properties.setInt("currsound", 1);
});

rock.addEventListener("touchstart", function(e) {
    Ti.API.info(currsound);
    if(player){
        player.stop();
