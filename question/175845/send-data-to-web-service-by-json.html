Send data to web service by JSON

I am using a form to send a data to a webservice URL, but it's always returning that error

on error: {"type":"error","source":{"cache":false},"code":415,"error":"HTTP error","success":false}

here is my code

button1.addEventListener('click', function(e) {

    var postData = {
        claimantType : "2",
        ClmDsc : "abcd",
        clmSite : "abcd",
        clmSubj : "abcd",
        latitude : "0",
        longitude : "0",
        persName : "abcd",
        siteCode : "1",
        regNo : "abcd",
        phone : "0506346779",
        mobile : "0506346779",
    };

    var xhr = Titanium.Network.createHTTPClient({
        onload : function(e) {
            Ti.API.info('----on load: ' + JSON.stringify(e));
        },
        onerror : function(e) {
            Ti.API.info('----on error: ' + JSON.stringify(e));
        }
    });

    xhr.open('POST', 'http://89.237.149.36/ws/resources/clm/newclm');
    xhr.send(postData);

});

I don't know what's going wrong, I need help please

  • When I send JSON to a web service, I use the following, sending the JSON object as a string:

    ...
xhr.open('POST', 'http://89.237.149.36/ws/resources/clm/newclm');
xhr.setRequestHeader('Content-Type', 'application/json; charset=utf-8');
xhr.send(JSON.stringify(postData));
...

    but it might depend on the web service and how it is expecting the data
