  • your code is returning the right value it just in milliseconds you need to convert that to days put this inside your button's click event 

    
b4.addEventListener('click', function(){
     Ti.API.info('P1 : ' + picker1.value);
    Ti.API.info('P2 : ' + picker2.value);

    var res = Math.abs((picker2.value) - (picker1.value));
    Ti.API.info('In milliseconds ' + res);
    var _24Hours = 1000*60*60*24;
    Ti.API.info('In days ' + Math.round(res/(_24Hours)));
});
