    You should create a file for your styles like this:

    common_styles.js

    exports.commonStyles = function() {
    return {    
        window: {width: '100%', height: '100%', backgroundColor: 'white'},

        //view's style    
        view: {
            container: { width: 934, height: 702, top: 46, left: 90, backgroundColor: '#cfd0d2', layout: 'vertical' },
            fullscreen: { width: 1024, height: 748, top: 0, left: 0 },
        },

        checkbox: { width: 19, height: 19, backgroundImage: 'images/common/checkbox.png', backgroundImageOn: 'images/common/checkbox-on.png', top: 11, left: 10 },

        scrollView: { contentWidth:'auto', contentHeight:'auto', showVerticalScrollIndicator:true, showHorizontalScrollIndicator:true },

        button: { width: 205, height: 50, font: { fontFamily: 'Helvetica Neue', fontSize: 21, fontWeight: 'bold' }, backgroundImage: 'images/common/btn-blue.png', backgroundLeftCap: 30, backgroundTopCap: 15 } 
    };
};

    Then Use it:

    var commonStyle = require('path/to/common_styles').commonStyles();
var button = Ti.UI.createButton(commonStyle.button);
