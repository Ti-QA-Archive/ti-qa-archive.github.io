function randomXToY(minVal,maxVal){ 
   var now = new Date();
   var seed = now.getSeconds();
   var randVal = minVal+(Math.random(seed)*(maxVal-minVal)); 
   return Math.round(randVal);
}

var randomnumber = randomXToY(1,5);

var bannerImage = Titanium.UI.createImageView({
    top:420,
    height:60,
    width:320,
    image:bannerImg[randomnumber]
});

bannerImage.addEventListener('click',function(e){
    Ti.Platform.openURL(bannerLink[randomnumber]);
});

So I've got this far in finding out how to rotate banner ads - It's not working. Any suggestions? Any workable code for rotating any images, even sequentially will be fine. Thanks for your help in advance. This is the last hurdle we have for our app!
