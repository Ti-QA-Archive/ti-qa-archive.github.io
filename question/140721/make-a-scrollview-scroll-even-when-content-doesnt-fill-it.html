Make a ScrollView scroll even when content doesn't fill it

Hey, I have a scroll view, with some views within it, but they don't reach the bottom of the scroll view. Is there a way I can make the scrollview still scroll (so I can pull down from the top, and pull up from the bottom, and it snaps back into it's place)?

in menu.js

function Menu( items ) {

    var _self = null;

    _self = new ui.ScrollView({ width:"100%", height:"100%", top:0, layout:"vertical" });

    for (x in items) {

        var _menurow = new MenuRow( items[x]['label'] );
        _self.add(_menurow);

    }

    return _self;

}

module.exports = Menu;

in MenuRow.js

function MenuRow( label ) {

    // private vars
    var _self = null;

    _self = new ui.View({ width:"100%", height:50, top:1, layout:"horizontal", backgroundColor:"#11ffffff" });
    return _self;

}

module.exports = MenuRow;
