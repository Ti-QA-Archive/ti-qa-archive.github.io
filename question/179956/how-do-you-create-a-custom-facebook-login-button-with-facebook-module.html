  • Hi Alan Young,

    You should create a standard Titanium button. Titanium.UI.createButton. Then using an event listener for 'click' on your button fire Titanium.Facebook.authorize();

    var win = Ti.UI.createWindow({
    backgroundColor : 'white',
    layout : 'vertical'
});
var fb = require('facebook');
fb.appid = "APPID";
fb.permissions = ['publish_stream', 'read_stream'];

var myButton = Ti.UI.createButton({
    backgroundImage : 'KS_nav_ui.png',
    width : 100,
    height : 30,
    top : 30,
    toggle : false //customproperty
});

myButton.addEventListener('click', function(e) {
    fb.authorize();
    if (e.source.toggle == false) {
        fb.addEventListener('login', function(e) {
            if (e.success) {
                alert('Logged in');
                myButton.toggle = true;
                myButton.backgroundImage = 'KS_nav_views.png';
            }
        });

    } else {
        fb.addEventListener('logout', function(e) {
            alert('Logged out');
            myButton.toggle = false;
            myButton.backgroundImage = 'KS_nav_ui.png';
        });

    }

});

win.add(myButton);

win.open();
