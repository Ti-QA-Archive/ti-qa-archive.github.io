      function ApplicationWindow() {
    //load component dependencies

    var MainView = require('MainView');
    //create component instance

    var self = Ti.UI.createWindow({
        backgroundImage : 'images/background.png',
        fullscreen : true,
        navBarHidden : true,
        exitOnClose : true
    });

    //construct UI
    var mainView = new MainView();
    self.add(mainView);

    return self;
