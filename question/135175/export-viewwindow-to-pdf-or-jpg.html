    Thank you Miguel, this example demonstrate the toImage().

    var win1 = Titanium.UI.createWindow({
    title : 'Tab 1',
    backgroundColor : '#fff'
});

var label1 = Titanium.UI.createLabel({
    color : '#999',
    text : 'Click Change Color',
    font : {
        fontSize : 20,
        fontFamily : 'Helvetica Neue'
    },
    textAlign : 'center',
    width : 'auto',
    top : 100
});

label1.addEventListener('click', function(e) {
    if(this.color == "red") {
        this.color = "blue"
    } else {
        this.color = 'red'
    }
});

var btn = Ti.UI.createButton({
    height : 35,
    width : 100,
    title : "View Image",
    top : 200
});
btn.addEventListener('click', function(e) {
    var windowImage = win1.toImage();
    setTimeout(function(ev) {
        var tempview = Ti.UI.createView({
            backgroundColor : "gray",
            top : 20,
            bottom : 20,
            right : 20,
            left : 20
        });
        tempview.addEventListener('click', function(ev) {
            win1.remove(tempview);
        });
        win1.add(tempview);
        var image = Ti.UI.createImageView({
            image : windowImage
        });
        tempview.add(image);
    },500);
});
win1.add(btn);
win1.add(label1);
win1.open();
