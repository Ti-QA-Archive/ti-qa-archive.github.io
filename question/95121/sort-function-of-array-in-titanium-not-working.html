    I m using same concept for Soring th array. Its working fine for me.

    I m using Titanium SDK 3.0.2GA.

    my code Snippet : 

    function sortByKey(array, key) {
   return array.sort(function(a, b) {
       var x = a[key]; var y = b[key];
           return ((x < y) ? -1 : ((x > y) ? 1 : 0));
   });
}

var sampleArray = sortByKey(sampleArray , 'name');
