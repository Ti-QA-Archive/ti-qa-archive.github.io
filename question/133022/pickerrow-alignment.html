pickerRow alignment

Once I created a pickerRow and assign some title to it, is there a way to align the title to the middle of the picker?
for example -
var data = [];
data[0] = Ti.UI.createPickerRow({title:"row number 1"});

I know I can use a label instead of the title, but then when I try to add some listener to print me the label text i get "undefined" :(

hope some one can guide me, thanks!

    Try this..

    
        var picker = Ti.UI.createPicker({
            selectionIndicator: true
        });

        var data = ['', '0000', '0031'],
        rows = [];

        for (var i = 0; i < data.length; i++){
            var row = Ti.UI.createPickerRow({
                title: data[i]
            });
            label = ti.Label({
                font:{
                    fontSize: 24,
                    fontWeight:'bold'
                },
                text: data[i],
                textAlign:'center'
            });

            row.add(label);

            rows.push(row);

        }
        picker.add(rows);
