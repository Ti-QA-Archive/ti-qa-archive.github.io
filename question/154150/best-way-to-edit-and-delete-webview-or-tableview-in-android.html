    deletebutn.addEventListener("delete", function(e) {
    db = Titanium.Database.open('cvapp');
    db.execute("DELETE FROM cv WHERE id=?",e.row.id);
    alert(tf.value, 'Cv er blevet slettet');
    db.close();
});

    deletebutn.addEventListener("click", function(e) {
    db = Titanium.Database.open('cvapp');
    db.execute("DELETE FROM cv WHERE id=?",e.row.id);
    alert(tf.value, 'Cv er blevet slettet');
