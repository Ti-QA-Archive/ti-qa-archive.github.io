// ******************************************************************
function creerHTMLcontenu(texteSize) {
    var strHTML = "<html><body><table>";
    var texteArticle = winArticle.contenuArticle;
    var indexIFrame = texteArticle.indexOf("<iframe");
    if (indexIFrame != -1) { // il y a une video a la fin de l'article
        var strTexteSansVideo = texteArticle.substring(0, indexIFrame);
        var indexDebutURL = texteArticle.indexOf("http", indexIFrame);
        var strQuote = texteArticle.substring(indexDebutURL-1, indexDebutURL);
        var indexFinURL = texteArticle.indexOf(strQuote, indexDebutURL);
        var strURLvideo = texteArticle.substring(indexDebutURL, indexFinURL);
        strHTML += "<tr><td style='font-size: " + texteSize + "px'>" + strTexteSansVideo + "</td></tr>";
        strHTML += "<tr><td><br/></td></tr><tr><td>";
        strHTML += '<iframe src="' + strURLvideo + '?showinfo=0" allowfullscreen="" frameborder="0" width="260"></iframe>';
        strHTML += "</td></tr>";
    }
    else {
        strHTML += "<tr><td style='font-size: " + texteSize + "px'>" + winArticle.contenuArticle + "</td></tr>";    
    }
    strHTML += "</table></body></html>";
    return strHTML;
}

function creerImageSeparateur(topBottomParam, isTop) {
    var view = Ti.UI.createView({
        left: 5, right: 5,
        height: 2,
        backgroundImage: 'separateur_article_alaUne.png'

    if(isTop) { view.top = topBottomParam; }

}
