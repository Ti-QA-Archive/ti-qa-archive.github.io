Window Open Transition Issue

I have a problem when I want to open a window with a transition.

When I click the button the window opens first then the transition happens.

So what you get is:

  • Window A is the one that open when the app starts
  • Click button
  • Window B opens
  • Window B then does a transition to Window B

What should happens is:

  • Window A is the one that open when the app starts
  • Click button
  • Window A then does a transition to Window B

See the video here: https://rapidshare.com/files/10272794/app_rec_1 - Wi-Fi.m4v

SDK: 2.1.3.v20120913140112 or 2.1.1.GA

var win = Ti.UI.createWindow({
    backgroundColor: '#FF0000',
});

var label = Ti.UI.createLabel({
    text: 'A',
    font: {
        fontSize: 35,
    }
});
win.add(label);

var btn = Ti.UI.createButton({
    title: 'Click',
    height: 50,
    width: 200,
    bottom: 0,
});
btn.addEventListener('click',function(){
    var cancelBtn = Ti.UI.createButton({
        title: 'Cancel',
    });
    cancelBtn.addEventListener('click',function() {
        win2.close();
    });        
    var win2 = Ti.UI.createWindow({
        backgroundColor: 'yellow',
        leftNavButton: cancelBtn,
    });
    var label = Ti.UI.createLabel({
        text: 'B',
        font: {
            fontSize: 35,
        }
    });
    win2.add(label);

    win2.open({
        modal: true,
        modalTransitionStyle: Ti.UI.iPhone.MODAL_TRANSITION_STYLE_FLIP_HORIZONTAL,
        modalStyle: Ti.UI.iPhone.MODAL_PRESENTATION_FORMSHEET,
    });
});
win.add(btn);

win.open();
