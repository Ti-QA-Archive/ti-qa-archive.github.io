  • when you click on the button you disappear the text

    button.addEventListener('touchstart',function(e){
            button.title = "";
        });

        button.addEventListener('touchend',function(e){
            button.title = "your_value_of_button";
        });

    • if you want you can increase the text of the title or even the size of the button

      button.addEventListener('touchstart',function(e){
             button.font = {fontSize : (font_of_normal_button)+10(for example),fontWeight:'bold'};
        button.height = (height_of_normal_button)+20(for example);
        button.width = (width_of_normal_button)+20(for example);
});

        button.addEventListener('touchend',function(e){
            button.font = {fontSize : (font_of_normal_button);
        button.height = (height_of_normal_button);
        button.width = (width_of_normal_button);
});

      var btn = Ti.UI.createButton({

        backgroundImage : "normal.png",
        backgroundSelectedImage : "clicked.png",
    });
