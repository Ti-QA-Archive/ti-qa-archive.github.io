    • I have it!

      We need to introduce a callback function to your loadInfo function - solves the whole thing.

      function loadInfo(obj) {   
    // Create our HTTP Client and name it "loader"   
    var loader = Titanium.Network.createHTTPClient();
    loader.open("GET","http://www.lacunaadmin.com/lacunapublications/json/hiscox-startup-ChannelData.json");  

    // Send the HTTP request   
    loader.send();

    // Runs the function when the data is ready for us to process  
    loader.onload = function () {   
        // Evaluate the JSON   
        var info = JSON.parse(loader.responseText);
        //Ti.API.info("Array: " + JSON.stringify(info));

        var tableData = [];
        for (var i = 0; i < info.rss.channel.item.length; i++) {
            var title  = info.rss.channel.item[i].title; 
            var link   = info.rss.channel.item[i].link;  
            var description = info.rss.channel.item[i].description; 
            var pubDate = info.rss.channel.item[i].pubDate;   
            tableData.push({title:title,link:link,desc:description,pubDate:pubDate});
        }
        Ti.API.info("Length: " + tableData.length);
        if (obj.sucess) {
            obj.success(tableData)
        }
    }
}

      and 

      Ti.include('ProcessJSON.js');        
loadInfo({
    success: function (data) {
        var tableData = data;
        Ti.API.info("Length: " + tableData.length); // 0                
    }
});

      This allows you to run code ONLY when the data has actually returned!

      Let me know how it goes.
