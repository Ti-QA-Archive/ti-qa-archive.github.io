    ok,
    try to open new window first and close window after like this..

    function next(e) {
    Alloy.createController('G2').getView().open();
    $.G.close();    
}

function back(e) {
    Alloy.createController('F').getView().open();
    $.G.close();    
}
$.G.open();

    function back(e) {

    Alloy.createController('F').getView().open();
    }
