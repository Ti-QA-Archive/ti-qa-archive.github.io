    easier to not use anonymous function

    Ti.App.addEventListener(&#39;blah_blah&#39;, function (e) {alert(&#39;1&#39;) });

    try this

    function aFunction(e) {alert('1') }
Ti.App.addEventListener('blah_blah', aFunction);
Ti.App.removeEventListener('blah_blah', aFunction);
