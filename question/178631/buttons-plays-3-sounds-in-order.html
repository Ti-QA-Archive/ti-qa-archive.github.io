Buttons plays 3 sounds in order

Hey all, i want to make a button play 3 sounds in the order that i choose, with the code that i have below,they all play at the same time.

note:each sound has a button attached to it and when the sound is playing the button's width increases by (this part i dont need help with, im just giving you some extra info)

//Rooster to play first
var roostersound =  Ti.Media.createSound({url:"sounds/rooster.wav"});
roostersound.addEventListener('complete', function() {
    roosterButton.width = 100;  
});
//Goat to play second 
var goatsound =  Ti.Media.createSound({url:"sounds/goat.wav"});
goatsound.addEventListener('complete', function() {
    goatButton.width = 100; 
});
//Lion to play third
var lionsound =  Ti.Media.createSound({url:"sounds/lion.wav"});
lionsound.addEventListener('complete', function() {
    lionButton.width = 100; 
});
//Start button
start1Button = Ti.UI.createButton ({
    backgroundImage:"buttons/start.png",
    top:450, left:20, height: 130, width: 230,
});
start1Button.addEventListener('click', function (e){
    goatsound.play();
    if (goatsound.STATE_PLAYING) {
              goatButton.width = 110; 
     };
     roostersound.play();       
     if (roostersound.STATE_PLAYING) {
                roosterButton.width = 110; 
    };
    lionsound.play();
     if (lionsound.STATE_PLAYING) {
                lionButton.width = 110; 
    };
});

  • Move the play command to the complete event of the previous sound:

    //Rooster to play first
var roostersound =  Ti.Media.createSound({url:"sounds/rooster.wav"});
roostersound.addEventListener('complete', function() {
    roosterButton.width = 100;  
    lionsound.play();
    if (lionsound.STATE_PLAYING) {
              lionButton.width = 110; 
    };
});
//Goat to play second 
var goatsound =  Ti.Media.createSound({url:"sounds/goat.wav"});
goatsound.addEventListener('complete', function() {
    goatButton.width = 100; 
    roostersound.play(); 
    if (roostersound.STATE_PLAYING) {
            roosterButton.width = 110; 
    };
});
//Lion to play third
var lionsound =  Ti.Media.createSound({url:"sounds/lion.wav"});
lionsound.addEventListener('complete', function() {
    lionButton.width = 100; 
});
//Start button
start1Button = Ti.UI.createButton ({
    backgroundImage:"buttons/start.png",
    top:450, left:20, height: 130, width: 230,
});
start1Button.addEventListener('click', function (e){
    goatsound.play();
    if (goatsound.STATE_PLAYING) {
            goatButton.width = 110; 
    };

});
