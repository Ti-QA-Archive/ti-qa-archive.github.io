    But if we are all in a guessing game here goes my guess.

    Add the following property to the window creation function where you DO not want the tabGroup to be visible.

    tabBarHidden: true,

    So use like this;

    var win = Ti.UI.createWindow({
    height: Ti.UI.FILL,
    tabBarHidden: true,
    title: 'Test',
    width: Ti.UI.FILL
});

    • Here is a full code example so you can see how to do this, to test simply create new default project and replace the contents of the app.js with this code;

      var win = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Answers'
});
var tab = Ti.UI.createTab({
    icon: 'KS_nav_views.png',
    title: 'Answers',
    window: win
});
var btn = Ti.UI.createButton({
    title: 'Win'
});
btn.addEventListener('click', function (e) {
    var newWin = Ti.UI.createWindow({
        height: Ti.UI.FILL,
        tabBarHidden: true,
        title: 'New Window',
        width: Ti.UI.FILL
    });
    tab.open(newWin, { animated: true });
});
win.setRightNavButton(btn);

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab);
tabGroup.open();

      You will notice the main window HAS the tabBar shown but the new window HIDES the tabBar. When you go back the original window HAS the tabBar still.
