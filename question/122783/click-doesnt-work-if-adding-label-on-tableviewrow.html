  • Check on the click event if the source is label. For example if you have 2 labels on your tableviewrow you can give each label a property, lets say "name:label1" and the other label "name:label2"

    Now you can check which you clicked by requesting the name of the source.

    tableview.addEventListener('click', function(e){
    if (e.source.name == "label1"){
        alert("clicked on label 1");
    } else if (e.source.name == "label2"){
        alert("clicked on label 2");
    } else {
        alert("not clicked on label");
    };
}

    Hope this helps!

    Tjeu

  • It's hard to tell why "click" event is not triggered on label added to the table view row, since you haven't provided any code.

    So, here is simple code that works with SDK 1.7.1:

    var win = Ti.UI.createWindow({ modal: true, backgroundColor: '#000' });

var tableView = Ti.UI.createTableView();

var tableViewRow = Ti.UI.createTableViewRow({ layout: 'horizontal', height: 200 });

var label1 = Ti.UI.createLabel({ text: 'Label1', width: 100, height: 200 });
var label2 = Ti.UI.createLabel({ text: 'Label2', width: 100, height: 200 });

label1.addEventListener('click', function() {
    Ti.alert('Label1');
});
label2.addEventListener('click', function() {
    Ti.alert('Label2');
});

tableViewRow.add(label1);
tableViewRow.add(label2);

tableView.setData([tableViewRow]);

win.add(tableView);

win.open();
