    function loginUser () {

   var Cloud = require('ti.cloud');
   Cloud.Users.login({
       login:$.txtUserName.value,
       password:$.txtPwd.value
   }, function(result){
       if(result.success){

       showProducts();
   }
   else{
       alert('login failed');
   }

});
}
$.index.open();


function showProducts(e) {  
    var w=Alloy.createController('products').getView();
    w.open();
}

$.index.open();
