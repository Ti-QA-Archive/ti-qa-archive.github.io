Click x and y coordinates on VideoPlayer are always zero

Hello,

How can I get the x and y coordinates from a click event on a VideoPlayer?

I'm trying this code:

var activeMovie = Titanium.Media.createVideoPlayer({
    url : contentURL,
    backgroundColor : 'blue',
    movieControlMode : Titanium.Media.VIDEO_CONTROL_NONE,
    scalingMode : Titanium.Media.VIDEO_SCALING_ASPECT_FILL,
    fullscreen : true,
    autoplay : true
});

activeMovie.addEventListener('click', function(e) {
    alert(e.x);
    activeMovie.hide();
    activeMovie.release();
    activeMovie = null;
});

But the x and y coordinates are always zero.

I'm using Titanium SDK 3.1.3.GA, on a Galaxy Note 2 device.

Thanks

Alexandre

  • I just had a play on the simulator (so caution!) and if i add a transparent view over the video player I can get the clicks off that. Does that help ?

    var w = Ti.UI.createWindow({
    backgroundColor : 'white'
});

var v = Ti.UI.createView({
    backgroundColor : 'transparent'
});

var activeMovie = Titanium.Media.createVideoPlayer({
    url : 'http:????',
    backgroundColor : 'blue',
    movieControlMode : Titanium.Media.VIDEO_CONTROL_NONE,
    scalingMode : Titanium.Media.VIDEO_SCALING_ASPECT_FILL,
    fullscreen : true,
    autoplay : true
});

v.addEventListener('click', function(e) {

    alert(e.x);

    activeMovie.hide();
    activeMovie.release();
    activeMovie = null;
});

w.addEventListener('open', function() {

    activeMovie.show();
    activeMovie.add(v);
});

w.open();
