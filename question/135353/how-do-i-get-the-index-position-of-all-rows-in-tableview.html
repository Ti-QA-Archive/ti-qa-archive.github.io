How do I get the index position of all rows in tableview?

  • Application type: mobile
  • Titanium SDK: 1.8.2
  • Deploy Target: iPhone

In an editable tableview once I moved the row I can only get the current index (e.index) of the row I'm moving. The question is whether it is possible to obtain the index of the other rows as well.


var tabGroup = Titanium.UI.createTabGroup();

var win = Titanium.UI.createWindow({  
    title:'Tab 1', backgroundColor:'#fff'
});

var tab1 = Titanium.UI.createTab({  
    title:'Tab 1', window:win
});

var data = [
    {title:"Row 0", editable: true},
    {title:"Row 1", editable: true},
    {title:"Row 2", editable: true},
    {title:"Row 3", editable: true}
];

var table = Titanium.UI.createTableView({data:data, editable:true, moveable:true, editing:true});

table.addEventListener('move',function(e) {

    Ti.API.info(e.index);

});

win.add(table);

tabGroup.addTab(tab1);  
tabGroup.open();

I've been trying different options interacting with the tableview but I can not get the index values ??of the other rows:

table.addEventListener('move',function(e) {

    for(var i = 0; i < table.data[0].rows.length; i++) {

        // get title OK
        // Ti.API.info(table.data[0].rows[i].title);

        // get index KO
        // Ti.API.info(table.data[0].rows[i].index); // <null>
        // Ti.API.info(table.data[0].rows[i].parent); // [object TiUITableViewSection]
        // Ti.API.info(table.data[0].rows[i].parent.index); // <null>
        // Ti.API.info(table.data[0].rows[i].parent.parent); // [object TiUITableView]
        // Ti.API.info(table.data[0].rows[i].parent.parent.index); // <null>

    }

});

  • hi Emil,

    you can set custom value for the row and get back…

    var data = [
    {title:"Row 0", editable: true, id : 0},
    {title:"Row 1", editable: true, id : 1},
    {title:"Row 2", editable: true, id : 2},
    {title:"Row 3", editable: true, id : 3}
];

table.addEventListener('move',function(e) {

    for(var i = 0; i < table.data[0].rows.length; i++) {

        // get title OK
         Ti.API.info(table.data[0].rows[i].id); 
    }

});

    table.addEventListener('move',function(e) {

        Ti.API.info("Title: " + table.data[0].rows[index].title + " Index: " + index);

    }

});
