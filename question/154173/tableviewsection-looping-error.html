  • Problem solved for item2. I replaced the part inside event listener below:

    //error code:

 tableview1.addEventListener("click",function(e){
                if(e.rowData.hasChild){
                    var newWin=Ti.UI.createWindow({
                        url:e.rowData.url,
                        title:e.rowData.title
                    });         
                }
                newWin.open({modal:true});
        });

    
//correct code:

 tableview1.addEventListener("click",function(e){
                if(e.rowData.hasChild){
                    var newWin=Ti.UI.createWindow({
                        url:e.row.url,
                        title:e.row.title
                    });         
                }
                newWin.open({modal:true});
        });

    summary: i replaced the e.rowData.url into e.row.url. Now, it goes to the correct pages. This is for future reference information for others who might encounter same problem.
