Urban Airship Module Not Doing Anything (Android)

I'm following the instructions to use the UA module in my app. I have installed the module correctly and get no errors in the log when built. When I try and register a new device nothing happens, no callbacks, nothing. Code below.

function eventError(e) {
    alert(e.error);
}

function eventSuccess(e) {
    alert(e.deviceToken);
}

function eventCallback(e) {
    alert(e.message);
}

// load urban airship module
var ua = require('ti.urbanairship');

var options = {
    PRODUCTION_APP_KEY:'XXX',
    PRODUCTION_APP_SECRET:'XXX',
    DEVELOPMENT_APP_KEY:'',
    DEVELOPMENT_APP_SECRET:'',
    IN_PRODUCTION:true,
    TRANSPORT:'helium',
};

ua.options = options;
ua.showOnAppClick = true;
ua.soundEnabled = false;

ua.takeOff({
    success: eventSuccess,
    error: eventError,
    callback: eventCallback
});

Any ideas why nothing is happening?
