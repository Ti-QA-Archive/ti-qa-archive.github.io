  • According to this Filesystem guide, you can only have below options for storing images on Android device:

    • applicationDataDirectory
    • externalStorageDirectory
    • tempDirectory
    var imageDir = Ti.Filesystem.getFile(Ti.Filesystem.externalStorageDirectory, 'user_images');
if(!imageDir.exists()) {
    imageDir.createDirectory();
}

// .resolve() provides the resolved native path for the directory.
var imageFile = Ti.Filesystem.getFile(imageDir.resolve(), myImageID + '.jpg');
Ti.API.info("ImageFile path is: " + imageFile.resolve());
if(imageFile.write(myImageData) === false) {
    // handle write error
}
// dispose of file handles
imageFile = null;
imageDir = null;
