      /////////////////////////////////////////////////////////////////////////
function ViewCell(img,lbl,desc){
    this.image = img;
    this.label = lbl;
    this.description = desc;
    this.show = function(){
        var img = Ti.UI.createImageView({
            left : 2,
            right : 2 ,
            image : this.image,
            width : 64,
            height : 64,
        });
        var lb = Titanium.UI.createLabel({
            text : this.label,
            left : 70 ,
            top : 0,
            font:{
                fontFamily:'Helvetica Neue',
                fontSize:17
            },
            color : 'blue'
        });
        var desc = Titanium.UI.createLabel({
            text : this.description,
            left : 70 ,
            top : 20,
            font:{
                fontFamily:'Verdana',
                fontSize:12
            },
            color : 'red'
        });
        var tvr = Titanium.UI.createTableViewRow({
            hasChild:false,width : 450,height : 100
        });
        tvr.add(img);
        tvr.add(lb);
        tvr.add(desc);
        return tvr;
    }
}
