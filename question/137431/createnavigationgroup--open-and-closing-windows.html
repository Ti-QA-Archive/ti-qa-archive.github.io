  • To get a better idea.

    FirstWindow:

    exports.FirstWindow = function(args){

    var SecondWindow = require('ui/SecondWindow').SecondWindow;

    var rootWindow = Ti.UI.createWindow({
        modal:true
    });

    var firstWindow = Ti.UI.createWindow({
        title:'SPH Tracker'
    });

    var navGroup = Ti.UI.iPhone.createNavigationGroup({
        window:firstWindow
    });
    rootWindow.add(navGroup);

    var myButton - Ti.UI.createButton({
        title:'Go to second window'
    });
    myButton.addEventListener('click', function(){
        navGroup.open(SecondWindow(), { animated:true });
    });

    return rootWindow;
}

    SecondWindow:

    exports.SecondWindow = function(args){

    var secondWindow = Ti.UI.createWindow({
        backButtonTitle:'Cancel'
    });

    var myButton = Ti.UI.createButton({
        title:'Save & Go back'
    });
    myButton.addEventListener('click', function(){
        // TODO: Code to do something
        secondWindow.close()
    });

    return secondWindow;
}

    In fact, what this does is to close the secondWindowleaving the screen black without moving back to the firstWindow. I say this because it leaves the top navigation bar with the Back button, and when I press it the it goes back to the firstWindow.
