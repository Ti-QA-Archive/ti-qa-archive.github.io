    Hi

    What you are looking for is the popover, this allows you to create a window like view that can contain other views even navigation style drill downs and tableViews but also point to the "thing" that initiated the window.

    I have provided an example below that will show you one working

    var win = Ti.UI.createWindow();

var button = Ti.UI.createButton({title: 'Open Popover!'});
button.addEventListener('click', function(e){
    popover.show({ view: button });
});
win.add(button);

var view = Ti.UI.createView({backgroundColor: 'green'});
view.add(Ti.UI.createLabel({text: "It's not easy being green."}));

var rightButton = Ti.UI.createButton({title: 'Robin'});
rightButton.addEventListener('click', function(e){
    alert("But green's the color of spring.");
});

var popover = Ti.UI.iPad.createPopover({
    width: 250,
    height: 100,
    title: 'Kermit',
    rightNavButton: rightButton
});
popover.add(view);

win.open();

    The arrow is automatically position for you (you do not manually set this) by this specific proprty;

    popover.show({
  view: button // this
});

    You simply set the view property to the button, view, label, whatever fired the popover to appear.
