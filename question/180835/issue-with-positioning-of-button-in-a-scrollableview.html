  • Hi

    Try wrapping your textFields inside a view with the layout property of "vertical". Leave your Button outside of this view. This should allow you to position the button where you want it while achieving a centred view with no hassle.

    here is my example:

    <Alloy>
    <Window id="win" >
        <ScrollView id="scrollView">
            <View id="view">
                <TextField id="textField1"/>
                <TextField id="textField2"/>
            </View>
            <Button id="button" />
        </ScrollView>
    </Window>
</Alloy>

    "#win": {
    backgroundColor:"orange"
},
"#view": {
    layout: 'vertical',
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
},
"#textField1": {
    borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    color: "#336699",
    height: 50,
    top: 30,
    width: 250
},
"#textField2": {
    borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    color: "#336699",
    height: 50,
    top: 30,
    width: 250
},
"#button": {
    backgroundColor: "blue",
    bottom: 30,
    color: '#fff',
    height: 50,
    width: 250
}

    Hope this helps.
