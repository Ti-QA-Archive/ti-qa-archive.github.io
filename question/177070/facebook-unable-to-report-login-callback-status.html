Facebook unable to report login callback status

Titanium Studio 3.2.3
iPhone IOS 7

Here is my code, a very simple copy and paste code from the documentation.

var fb = require('facebook');

fb.appid = '746587995866227';
fb.permissions = ['publish_stream'];
fb.forceDialogAuth = true;

fb.addEventListener('login', function(e) {
    if (e.success) {
        alert('Logged In');
    } else if (e.error) {
        alert(e.error);
    } else if (e.cancelled) {
        alert("Canceled");
    }
});
fb.authorize();

the login dialog pops up and the dialog closes after pressing the login button but the eventlistener doesn't fire and no alert dialog is displayed after loggin in. No error, nothing.
