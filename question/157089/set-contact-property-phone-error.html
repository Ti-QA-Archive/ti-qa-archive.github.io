Set Contact Property Phone Error

I'm trying to change an number of the contacts list but I'm getting this error when I tried it:

Failed to set contact property phone: The operation couldnt be completed. ( ABAddressBookErrorDomain error 0.) at menu.js ( line 340 )

I search on google and found that is something about authorization but on my code I already do the Adress Book authorization.

/////
var iniciaApp = function(){
    ////////
    var novoContato = Ti.Contacts.getPersonByID( uniquiID[i].id );        

    novoContato.phone = {
        'iPhone' : ['00000']        //     <----  LINE 340
    };

    Ti.Contacts.save( novoContato );
}

var autorizar = function(){
    if (Ti.Contacts.contactsAuthorization == Ti.Contacts.AUTHORIZATION_AUTHORIZED){

            iniciaApp();

    } else if (Ti.Contacts.contactsAuthorization == Ti.Contacts.AUTHORIZATION_UNKNOWN){

        Ti.Contacts.requestAuthorization(function(e){
                    if (e.success) {
                        iniciaApp();
                    } else {
                        acessoNegado();
                    }
            });

    } else {
            acessoNegado();
    };
};
autorizar();
