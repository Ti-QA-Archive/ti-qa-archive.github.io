Table_view_layout3 : the right button stay behind a halo..

Hi,

I have some code extracted form the KS app.
This code is based on the Table_view_layout.js example.

An interesting thing is that the right image i have put (on the place of the Plus button image of the example code) refuse to appear properly… The button seems to appear behind a halo of light. This halo disappear in the KS app but not on my code when i want to show() the button.

I don't understand why.

App type : Mobile
Titanium Mobile SDK Version: 1.8.2
module_apiversion=2
Mobile SDK Timestamp: 02/23/12 17:46
Mobile SDK Build Number: 59b3a90

var win1 = Titanium.UI.createWindow({  
    title:'test',
});
var data =[];

var section = Ti.UI.createTableViewSection();
data.push(section);

// ROW 1
var row1 = Ti.UI.createTableViewRow();

row1.height = 50;
row1.addEventListener('click', function(e) {

    if(add1.getVisible() === true) {
        add1.hide();
        Ti.API.info("unchecked");
        }
    else {
        add1.show();
        Ti.API.info("checked");
    }
});


var item1 = Ti.UI.createLabel({
    text:'Burger',
    color:'#fff',
    font:{fontSize:20, fontWeight:'bold'},
    top:3,
    left:10,
    height:30,
    width:100
});
row1.add(item1);

var add1 = Ti.UI.createButton({
    backgroundImage:'/images/checkblue.png',
    height:16,
    width:16,
    top:15,
    right:10
});

row1.add(add1);
section.add(row1);
add1.hide();



var tableview = Titanium.UI.createTableView({
    data:data,

  • How about to use Ti.UI.createView in replace of Ti.UI.createButton?

    var add1 = Ti.UI.createView({
    backgroundImage:'/images/checkblue.png',
    height:16,
    width:16,
    top:15,
    right:10
});

    You can bind events to view like button.
