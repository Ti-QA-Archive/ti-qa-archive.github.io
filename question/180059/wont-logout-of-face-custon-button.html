  • Hi,

    I'm also facing the same problem, maybe it's a bug. I'm not sure but if it is, then you can create a JIRA ticket for this issue. Here is my test code,

    var win = Ti.UI.createWindow({
    backgroundColor : '#fff',
});

var facebook = require('facebook');
facebook.appid = 'APP_ID';
facebook.permissions = ['publish_stream', 'read_stream'];

var label = Ti.UI.createLabel({
    text : 'Logged In = ' + facebook.loggedIn,
    font : {
        fontSize : 14
    },
    height : 'auto',
    top : 10,
    textAlign : 'center'
});
win.add(label);

function updateLoginStatus() {
    label.text = 'Logged In = ' + facebook.loggedIn;
}

var fbLogin = Ti.UI.createButton({
    top : 50,
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
    backgroundColor : '#3b5998',
    title : 'Login',
});

win.add(fbLogin);

fbLogin.addEventListener('click', function(e) {
    facebook.authorize();
    if (facebook.loggedIn == false) {
        facebook.addEventListener('login', updateLoginStatus);
        fbLogin.title = 'Logout';
    }
    else {
        alert("In");
        facebook.addEventListener('logout', updateLoginStatus);
        fbLogin.title = 'Login';
    }
});

win.open();
