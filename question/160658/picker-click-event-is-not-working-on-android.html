  • Hi Ahmed Salah,

    Your question is so intelligence.

    I have used the bellow code to find your solution. So please check this code and I hope that it will help you a little bit for solving your question. 

    Ti.UI.backgroundColor = 'black';

var win = Ti.UI.createWindow({
  exitOnClose: true,
  layout: 'vertical'
});
var picker= Ti.UI.createPicker({
  top:30,
  useSpinner:true
});
var lbl = Ti.UI.createLabel({
    text : 'click',
    color:'white',
    font : {fontSize:15},
    height : 'auto',
    width : 'auto',
    top:10,
});
win.add(lbl);

lbl.addEventListener('click',function(e){
    pickerSetting();
});

var data = [];
data[0]=Ti.UI.createPickerRow({title:'Bananas'});
data[1]=Ti.UI.createPickerRow({title:'Strawberries'});
data[2]=Ti.UI.createPickerRow({title:'Mangos'});
data[3]=Ti.UI.createPickerRow({title:'Grapes'});


picker.addEventListener('change',function(e){
    alert(e.row.title);
});

win.open();

function pickerSetting()
{
picker.add(data);
picker.selectionIndicator = true;

win.add(picker);
}
