Animation problem.

Hi,
I'm making a view, with buttons, to "animate in" when scrolled to the end of the tableview.
The when I start scrolling again there is a "animate out" event starting.

There are a couple of problems though:

  • Warning that says "New layout set while view [object TiUIView] animating: Will relayout after animation.

  • The scrollend event doesnt seem to start the first time you scroll to the end.

  • At IPHONE5 it seems that the view, when doing the "animate in" for the second time, is over the row above.

Codesnippet:

// View with send & save - buttons.
    var sendBarView = ui.view({                                                                                 
        backgroundImage: "img/form/bar.png",
        bottom: -70,
        width: 320,
        height: 80
    }); 
    sendBarView.add(submit, save);

   // Animation IN
   t.addEventListener('scrollend', function(e) {                                                            
        if (e.contentOffset.y + e.size.height + 100 > e.contentSize.height) {
            sendBarView.Seen = true;
            w.add(sendBarView);
            var animationUp = Titanium.UI.createAnimation();
            animationUp.bottom = 0;
            animationUp.duration = 130;
            sendBarView.animate(animationUp);
            w.open(sendBarView);
        }
    });

   //Animation OUT
   t.addEventListener('scroll', function(e) {
        if (sendBarView.Seen = true) {
            var animationDown = Titanium.UI.createAnimation();
            animationDown.bottom = -70;
            animationDown.duration = 70;
            sendBarView.animate(animationDown);
            w.open(sendBarView);
        }
    });

Thanks in advance!
