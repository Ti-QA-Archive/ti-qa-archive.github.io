    //Create a window

var win = Ti.UI.createWindow({
  backgroundColor: 'black',
  fullscreen: false,
  title: 'Window',
  layout:'vertical'

var db=Ti.Database.open('amb');
db.execute('CREATE TABLE IF NOT EXISTS student(s_id INTEGER PRIMARY KEY,s_number TEXT ,s_name TEXT, s_password TEXT, s_username TEXT)');

db.execute('INSERT INTO student (s_number,s_name,s_password,s_username) VALUES (?,?,?,?)', '1','am', '123', 'amlog'); 
var lastID = db.lastInsertRowID; // `city` has an auto-increment column

 var studentRS = db.execute('SELECT s_number,s_name,s_password,s_username FROM student'); 
var data=[];

while (studentRS.isValidRow()) 
{ 

data.push({
"s_name": studentRS.fieldByName('s_name'),   
"s_number": studentRS.fieldByName('s_number'),   
 "s_password": studentRS.fieldByName('s_password')  
});  
studentRS.next(); 
}
 studentRS.close();

for(var a=0;a<data.length;a++)
{
var nameLabel = Titanium.UI.createLabel({
text:data[a].s_name,
color:'#fff',
 font:{
    fontSize:25,
    fontFamily:'Helvetica Neue'
       },
 top:'220', left:'20',
 width:'auto'
 });

 win.add(nameLabel);
}

win.open();
