  • Here's an example of a TableViewRow with a top border, bottom border, and background gradient like this:

    function createMyRow() {
  var result = Ti.UI.createTableViewRow({
    className: 'myrow',
    height: 40,
    width: 'auto',
    backgroundGradient: {
      type: 'linear',
      colors: ['#f9f9f9','#d7d7d7'],
      startPoint: {x:0,y:0},
      endPoint:{x:0,y:37},
      backFillStart:false},
  });

  // top border
  result.add(Ti.UI.createView({
    top: 0,
    height: 1,
    backgroundColor: '#ccc',
  }));

  // bottom border
  result.add(Ti.UI.createView({
    bottom: 0,
    height: 1,
    backgroundColor: '#ccc',
  }));
}
