    Hi Abhishek,

    Onclick of first button it gives you 0 index not 1 and on second button click event it gives 1 index not 2 index.
    try this code for countriesAlertDialog.Best luck

    countriesAlertDialog.addEventListener('click', function(e) {
    // var index = e.
    if (e.index === 0) {
        // countriesAlertDialog.close();
        Titanium.UI.currentTab.open(win, {
            animated : true
        });
    } else if (e.index === 1) {
        // countriesAlertDialog.close();
        Titanium.UI.currentTab.open(win, {
            animated : true
        });
    }
});
