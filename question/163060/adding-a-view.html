Adding a View

I have been struggling with concept. I would like to add another view or window to this scroll view. When added I get an error stating In Unidentified object error. Given the following how would I be able to create this.

var winView1 = Ti.UI.createScrollView({
    top : 0,
    height : '85%'

});

winView1.add('myotherview.js')

Any help with this is greatly appreciated.

    var winView1 = Ti.UI.createScrollView({
    top : 0,
    height : '85%'

});

var otherView = Ti.UI.createView();

winView1.add(otherView)

    OR

    myotherview.js

    
function createMyView(){

    var v = Ti.UI.createView();
    return v;

}

exports.createMyView = createMyView

    and then

    
var winView1 = Ti.UI.createScrollView({
    top : 0,
    height : '85%'

});

var myView = require('myotherview');

var otherView = myView.createMyView();

winView1.add(otherView)
