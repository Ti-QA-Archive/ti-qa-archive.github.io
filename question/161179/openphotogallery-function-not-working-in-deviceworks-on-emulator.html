  • Hi Rama krishna!!!

    Please try this below code. It may solve your problem

    var images;
btnPhoto.addEventListener('click', function(e){
       Titanium.Media.openPhotoGallery({
           success:function(event)
           {
               Ti.API.info('Types: '+event.mediaType);
               if(event.mediaType == Ti.Media.MEDIA_TYPE_PHOTO)
               {
                   images=event.media;
               }
           },
           error:function(err)
           {
               Ti.API.info(err);
           },
       });
   });
