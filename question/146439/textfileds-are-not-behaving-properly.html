    Hi Gaurav,

    this is a known issue for Android TextFields in TableView rows (see Android: tableView textField focus is lost while typing).

    As described in Jira, it will probably never be fixed, because it's a bad practice on Android to use input TextFields in TableView rows.

    But there is a workaround though, which i implemented in the code below.

    The trick is to set windowSoftInputMode : Ti.UI.Android.SOFT_INPUT_ADJUST_PAN for the Window, to prevent the window from resizing if the keyboard appears.

    Next time, would you please provide a runnable code snipped, so it's much easier to analyse a problem, thanks.

    Also i would recommend, to use a simpler implementation of that logon dialog. Why don't you just use labels, textfields and buttons without that tableView and row stuff?

    var signUpWin = Ti.UI.createWindow({
    title : "Sign Up",
    backgroundColor : 'white',
    // Workaround as described in https://jira.appcelerator.org/browse/TIMOB-8332
    // http://docs.appcelerator.com/titanium/2.0/#!/api/Titanium.UI.Window-property-windowSoftInputMode
    // http://docs.appcelerator.com/titanium/2.0/#!/api/Titanium.UI.Android-property-SOFT_INPUT_ADJUST_PAN
    windowSoftInputMode : Ti.UI.Android.SOFT_INPUT_ADJUST_PAN
});

var tbl_data = [];
// var mainBview = appce.createBackgroundView({});
var mainBview = Ti.UI.createView({});
signUpWin.add(mainBview);
var mainView = Ti.UI.createTableView({
    top : 10,
    /*
     separatorStyle : appce.os({
     iphone : Ti.UI.iPhone.TableViewSeparatorStyle.NONE,
     android : ''
     }),
     */
    separatorColor : 'transparent',
    height : '90%',
    width : '95%',
    backgroundColor : 'transparent',
    allowsSelection : false
});

signUpWin.add(mainView);
// mainSview.add(mainView);
//var row = appce.tableViewRow({
var row = Ti.UI.createTableViewRow({
    hasChild : false,
    height : 55,
    backgroundColor : 'transparent'
});

//var uName = appce.createLabel({
var uName = Ti.UI.createLabel({
    text : 'Username : ',
    left : 20
});
// var uNameField = appce.createTextField({
var uNameField = Ti.UI.createTextField({
    hintText : 'Username',
    right : 20
});
row.add(uName);
row.add(uNameField);
tbl_data.push(row);

var rowPass = Ti.UI.createTableViewRow({
    hasChild : false,
    height : 55,
    backgroundColor : 'transparent'
});
var uPass = Ti.UI.createLabel({
    text : 'Password : ',
    left : 20
});
var uPassField = Ti.UI.createTextField({
    hintText : 'Password',
    right : 20
});
rowPass.add(uPass);
rowPass.add(uPassField);
tbl_data.push(rowPass);
mainView.data = tbl_data;

mainBview.add(mainView);

signUpWin.open();

    Hope that helps.

    Greetings,

    Matthias
