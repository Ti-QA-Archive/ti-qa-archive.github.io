    • Hi try this code, this will work for you:

      
var selectedItem1 = Ti.UI.createLabel({
    color: "black",
    text: 0,
    width: 80,
    height: 35,
    top: "0%",
    left:'15%'
});
var testDec = Ti.UI.createButton({
    title: "-",
    color: "black",
    top: "0%",
    left:'25%'
});
var testInc = Ti.UI.createButton({
    title: "+",
    color: "black",
    top: "0%",
    left:'35%'
});
testDec.addEventListener('click', function (e) {
       dec();
    });
    testInc.addEventListener('click', function (e) {
       inc();
    });
function dec() {
    var testDec = selectedItem1.text;
    testDec--;
    selectedItem1.text = testDec;
}
function inc() {
    var testInc = selectedItem1.text;
    testInc++;
    selectedItem1.text = testInc;
}

overlay.add(selectedItem1);
overlay.add(testDec);
overlay.add(testInc);
