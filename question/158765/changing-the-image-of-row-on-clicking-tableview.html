Changing the image of row on clicking tableView

Hi

I am implementing a picker using titanium tableView.I have added an image and label in row.Now what i want to do is that on clicking the row the image of row changed from checkbox unchecked to checkbox checked.I have been baging my head on it from one hour and has tried every solution suggested but no good.here is my code

var self = Ti.UI.createWindow({
        backgroundColor : 'white'
    });
    // Create an ImageView.
    var imageTopIllness = Ti.UI.createImageView({
        image : '/images/illnessTop.png',
        width : Ti.UI.FILL,
        height : '80dp',
        top : 0,
    });
    self.add(imageTopIllness);

    var tableData = [];
    var data = ['Head', 'Chest', 'Abdomen', 'Pelvis', 'Legs', 'Children', 'Back', 'Skin'];
    var tableView = Ti.UI.createTableView({
        top : '80dp',
    });
    self.add(tableView);

    for ( i = 0; i < data.length; i++) {
        var row = Ti.UI.createTableViewRow({
            height : '45dp',
        });
        var labIllness = Ti.UI.createLabel({
            textAlign : 'left',
            text : data[i],
            font : {
                fontSize : '16dp',
                fontWeight : 'bold'
            },
            left : '75dp',
            height : '20dp',
            width : Ti.UI.FILL,
        });
        row.add(labIllness);
        var imageCheckBox = Ti.UI.createImageView({
            image : '/images/checkboxUnchecked.png',
            left : '15dp',
            width : '20dp',
            height : '20dp'
        });
        row.add(imageCheckBox);
        tableData.push(row);
    }
    tableView.setData(tableData);
    tableView.addEventListener('click', function(e) {
        e.row.imageCheckBox.image = 'images/checkboxChecked.png';
    });
self.open()

Thanks

  • Accepted Answer

    var imageCheckBox = Ti.UI.createImageView({
            image : '/images/checkboxUnchecked.png',
            left : '15dp',
            width : '20dp',
            height : '20dp'
        });
    row.imageCheckBox=imageCheckBox;
        row.add(imageCheckBox);
        tableData.push(row);

    You may also need to 

    tableView.addEventListener('click', function(e) {
        e.rowData.imageCheckBox.image = 'images/checkboxChecked.png';
    });
