Listview custom properties

Hi everyone,
I need to append custom proprieties for each item in my listview and get them back when I click.
How can I do like this with Ti.UI.createListView????


for (var i = 0; i < array.length; i++){
    var row = Ti.UI.createTableViewRow({
        prop1 : "abcd"
        prop2: "efgh"
        prop3: "lmno"
    });
    row.addEventListener('click',function(e){
        alert(e.source.prop1);
    });

}

  • It is a bad idea to bind a event on every row, but anywhere:

    var rows = [];
for (var i = 0; i < array.length; i++){
    rows.push(Ti.UI.createTableViewRow({
        myprops : {1:"abcd",2:"efgh"},
        hasChild : true,
    // other properties
     }));
}
tableview.setData(rows);
tableview.addEventListener('click',function(e) {
    console.log(e.rowData.myprops);
})

  • Hi Better you use it like:

    var table = Ti.UI.createTableView();
var row = Ti.UI.createTableViewRow({
        prop1 : "abcd"
        prop2: "efgh"
        prop3: "lmno"
    });

    table.data = [row];
    table.addEventListener('click',function(e){
        alert(e.row.prop1);
    });
