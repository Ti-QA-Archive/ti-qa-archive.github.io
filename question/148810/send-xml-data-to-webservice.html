    Have a read of the HTTPClient docs

    var dataString = "<tblgeo><accuracyInMeters>555</accuracyInMeters><fkUserid><userid>2</userid></fkUserid><latitude>58.072483</latitude><longitude>8.248353</longitude><reverseGeocode>Beispieltzext64</reverseGeocode><timestmp>12121234</timestmp></tblgeo>";

// Create your HTTP Client
var xhr =  Titanium.Network.createHTTPClient({
    username: "user",
    password: "pwd",
    onload: function() {
        Ti.API.info(this.responseText); 
    },
    onerror: function() {
        alert("Danger, Will Robinson!"); 
    }
});

// Set the request type and the url
xhr.open("post", "http://serveradress/webresources/entities.tblgeo");

// Set the content-type header
xhr.setRequestHeader("Content-type", "application/xml; charset=utf-8");

// Send the request
xhr.send(dataString);

    • Meister,

      I do this type of web service operations every day, so I'm sure they work. I want to emphasize how easy it is with the XHR library:

      var XHR = require("/lib/xhr");
var xhr = new XHR();
xhr.post(yourURL,yourPayload,onSuccessCallback, onErrorCallback);

      onSuccessCallback and onErrorCallback are simply functions where you specify what you want to do with the response, and what you want to do in case of an error.

      It's really that simple :)
