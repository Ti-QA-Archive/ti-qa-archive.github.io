//**************************** Create tableview with section
function GetTableView(MyJSONData) {

    var tableAttendeeData = [];
    var sectionAttendeeData = [];
    if (MyJSONData.length != null) {

        for (var i = 0; i < MyJSONData.length; i++) {

            var viewHeader = Ti.UI.createView({
                backgroundColor : '#eee',
                height : Alloy.Globals.GetHeight(28)
            });

            var labelHeaderName = Ti.UI.createLabel({
                color : '#576996',
                font : {
                    fontSize : Alloy.Globals.GetFontSize(4)
                },
                text : MyJSONData[i][0].StrFirstCharacterOfName,
                left : Alloy.Globals.GetHeight(5)
            });

            viewHeader.add(labelHeaderName);

            var sectionHeader = Ti.UI.createTableViewSection({
                headerView : viewHeader,
                height : Alloy.Globals.GetHeight(30)

            });

            if (MyJSONData[i].length != null) {
                for (var j = 0; j < MyJSONData[i].length; j++) {

                    var rowAttendeeList = Ti.UI.createTableViewRow({
                        className : 'row' + j,
                        selectedBackgroundColor : '#FFF',
                        rowIndex : j,
                        height : Alloy.Globals.GetHeight(90),
                        title : MyJSONData[i][j].listUserInfo[0].StrName
                    });

                    var labelAttendeeId = Ti.UI.createLabel({
                        color : '#576996',
                        font : {
                            fontSize : Alloy.Globals.GetFontSize(2),
                        },
                        text : MyJSONData[i][j].listUserInfo[0].StrUserInfoId,
                        visible : false
                    });
                    rowAttendeeList.add(labelAttendeeId);            
                    sectionHeader.add(rowAttendeeList);
                }
            }
            sectionAttendeeData.push(sectionHeader);
            tableAttendeeData.push(sectionAttendeeData);
        }
    }
    return tableAttendeeData;

}
