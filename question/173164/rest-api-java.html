public String obtieneIdSession() throws Exception{
    URL url = null;
    String idSession=null;
    String totalCookie = "";
    try {

        CookieManager manager = new CookieManager();
        manager.setCookiePolicy(CookiePolicy.ACCEPT_ALL);
        CookieHandler.setDefault(manager);

         url = new URL(URL_ACS+"users/login.json?key="+API_KEY+"&login="+API_USR+"&password="+API_PAS+"");
         HttpURLConnection conn = (HttpURLConnection) url.openConnection();

         conn.setRequestMethod("POST");
         if (conn.getResponseCode() != 200) {
           throw new Exception(conn.getResponseMessage());
         }

         CookieStore cookiejar = manager.getCookieStore();
         List<HttpCookie> cookiesList = cookiejar.getCookies();

         for(HttpCookie cookiel: cookiesList){
            totalCookie += cookiel+"; ";
         }
         totalCookie = totalCookie.substring(0, totalCookie.length()-1);

         InputStream is = conn.getInputStream();
         BufferedReader rd = new BufferedReader(
             new InputStreamReader(is));
         StringBuilder sb = new StringBuilder();
         String line;
         while ((line = rd.readLine()) != null) {
           sb.append(line);
         }
         rd.close();

         conn.disconnect();

    System.out.println("ÏD sesion: " + totalCookie);
    idSession=totalCookie;
    } catch (Exception e) {
    throw new Exception("Problema de session "+e);
    }
    return idSession;

    public void enviaMensaje(String nombreAlert, String titulo,String session_id) throws Exception{
    URL url = null;

         post.put("to_ids","everyone");
         post.put("version", "v1"); 
         post.put("action", "notify"); 
         post.put("controller", "push_notification"); 
         post.put("format", "json"); 

         String ruta=URL_ACS+"push_notification/notify.json?key="+API_KEY;

          url=new URL(ruta);
         uc = (HttpsURLConnection) url.openConnection();
         uc.setRequestMethod("POST");
         uc.setDoInput(true);
         uc.setDoOutput(true);
         uc.setAllowUserInteraction(false);
         uc.setRequestProperty("Content-Type", "application/json; charset=utf-8");
         uc.setRequestProperty("Accept", "application/json");
         uc.setRequestProperty("Cookie",session_id);

public static void archivoXML(String archivo, List<String> listaRenglones, boolean bandera){

    System.out.println("Escribe archivo "+archivo);
    OutputStreamWriter bw=null;
    try{
        bw = new OutputStreamWriter(new FileOutputStream(RUTA+archivo),"UTF-8");
        if(bandera){
            bw.append("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
            bw.write("\n");
        }
        for(String renglon:listaRenglones){
            bw.append(renglon);
            bw.write("\n");
        }
    }catch(Exception e){
        System.out.println(e);
    }finally{
        try {
            bw.flush();
        } catch (IOException e1) {
            e1.printStackTrace();
        }
        try {
            bw.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

public static String encodeFile(String escudo) throws Exception {
     ByteArrayOutputStream ous = null;

     String source = RUTA+"/escudos/"+escudo+".png";
     String transform=null;
     InputStream ios = null;
     try{

         File sourceImage = new File(source);
         byte[] buffer = new byte[4096];
         ous = new ByteArrayOutputStream();
         ios = new FileInputStream(sourceImage);
         int read = 0;
         while ( (read = ios.read(buffer)) != -1 ) {
             ous.write(buffer, 0, read);
         }
         byte[] encodedBytes = Base64.encodeBase64(ous.toByteArray());
         transform=new String(encodedBytes);
     }catch(Exception ex){
         ex.printStackTrace();
     }finally{
         try {
             if ( ous != null ) 
                 ous.close();
        } catch ( Exception e) {
        }

        try {
             if ( ios != null ) 
                  ios.close();
        } catch ( Exception e) {
        }
     }

     return transform;
 }
