function LoadRemoteImage(obj, url) {
    var xhr = Ti.Network.createHTTPClient();
    xhr.onload = function() {
    //    Ti.API.info('image data=' + this.responseData);
        obj.image = this.responseData;
    };
    // open the client
    xhr.open('GET', url);
    // send the data
    xhr.send();
}
