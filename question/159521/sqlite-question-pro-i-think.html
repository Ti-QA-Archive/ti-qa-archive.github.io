    Perhaps you can adapt the code referenced in this StackOverflow answer specifically the SQLfiddle it references http://sqlfiddle.com/#!7/e03a4/3.

    SELECT
  substr( name, 1, pos-1) AS first_name,
  substr( name,    pos+1) AS last_name
FROM (
      SELECT
        author.name,
        nnn.x AS pos
      FROM       author
      INNER JOIN nnn
      WHERE substr( author.name, nnn.x, 1) = ' '
      GROUP BY author.name
     ) AS a
ORDER BY last_name;
