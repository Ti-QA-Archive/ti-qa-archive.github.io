    Hi stephane,

    Try this one.
    main.xml

    <Alloy>
    <View id="Container">
        <ScrollableView id="ScrollableView">
        </ScrollableView>
    <Button id="btn_yes"/>
    </View>
</Alloy>

    main.js

    $.agenda = Alloy.createController('agenda');
$.sortis = Alloy.createController('sortis');
$.profil = Alloy.createController('profil');
$.evenement = Alloy.createController('evenement');
var mesVues = [$.agenda.getView(),$.sortis.getView(),$.profil.getView(),$.evenement.getView()];
//ChangeView.agenda(mesVues,function(e){});

$.ScrollableView.views = mesVues;
$.ScrollableView.currentPage = 0; 
var ar = $.ScrollableView.getViews();
$.bt_yes.addEventListener('click',function(e){

if (($.ScrollableView.currentPage + 1) >= ar.length) {
            $.ScrollableView.scrollToView(0);
        } else {
            $.ScrollableView.scrollToView($.ScrollableView.currentPage + 1);
        }    
});
