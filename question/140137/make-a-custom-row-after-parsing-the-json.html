    renderV2(JSON.parse(this.responseText.replace(/\\n/gi,"")));

function renderV2(e){
    var venues=e;
    var data=[];
    var sorted_suburb = [];
    var sTitle=[];
    for(var i=0; i<venues.length;i++){
        sorted_suburb.push(venues[i].places_suburb);
    }
    sorted_suburb.sort();

    for (var c = 0; c < venues.length; c++) {
       var sTitle = venues[c].places_name;
        var row = Ti.UI.createTableViewRow({
            height : 100,
            id:c
        });
        var view = Ti.UI.createView({
            width:310,
            height:90,
            backgroundColor:'#fff',
            borderColor:'#a9abae',
            borderRadius:10,
            borderWidth:1,
            id:c
        });
        row.add(view);

        var label = Ti.UI.createLabel({
            text : sTitle,
            left : 10,
            top : 10,
            color : '#4a4a4a',
            height : 20,
            width : 220,
            font:{
                fontSize:15,
                fontWeight:'bold'
            },
            id:c
        });

        var suburb_label = Ti.UI.createLabel({
            text : sorted_suburb[c],
            left : 10,
            top : 50,
            font : {
                fontSize : 11
            },
            color : '#808285',
            height : 20,
            width : 220,
            id:c
        });
        view.add(suburb_label);
        view.add(label);

        data.push(row);
    }

    tableview.setData(data);
}




function requestServer(e) {

    var httpClient = Ti.Network.createHTTPClient();
    httpClient.onload = e.success;
    httpClient.onerror = e.failed;
    httpClient.open(e.method, e.url);
    httpClient.setTimeout(50000)
    if (checkInternetConnection()) {
        //progress.show();
        httpClient.send();
    } else {
        alert('Please check your cellular connection ')
    }
}

function checkInternetConnection() {
    return Ti.Network.online;
}
