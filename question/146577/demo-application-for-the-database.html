  • please follow simple step to install database using titanium
    step 1: create database using sqlite database browser and give the name of your table and add columns to it.

    step 2: var dbInstall = Titanium.Database.install(Titanium.Filesystem.resourcesDirectory+&#39;yourdatabase&#39;,&#39;xyz&#39;);
    step 3: 

    function insertData(data)
{
    try
    {
        dbFiles.db =Titanium.Database.open('xyz');
        dbFiles.db.execute('INSERT INTO loginDetails(username,password) values(?,?)',data['username'],data['password']);    
    }
    catch(e)
    {
        Ti.API.info('Error'+e);
    }
    finally
    {
        dbFiles.db.close();
    }
}
