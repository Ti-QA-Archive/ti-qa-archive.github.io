Textarea and textfiel blur on android

When triggered the blur event of a textfield or textarea, remains the colored border (as if somehow they were still focused).
Is there any way to completely lose the focus?

var win = Ti.UI.createWindow();
var textArea = Ti.UI.createTextArea({
    top : 10,
    width : 200,
    height : 150
})
var button = Ti.UI.createButton({
    top : 180,
    title : "OK"
})

button.addEventListener("click", function(){            
    Ti.UI.Android.hideSoftKeyboard();
    textArea.blur();
});
textArea.addEventListener("return", function(){            
    Ti.UI.Android.hideSoftKeyboard();
    textArea.blur();
});    
textArea.addEventListener("android:back", function(){
    Ti.UI.Android.hideSoftKeyboard();
    textArea.blur();
})


win.add(textArea);
win.add(button)

win.open();

  • If this may help, I found a solution to work around the problem:

    textArea.addEventListener("blur", function(){         
    if(Ti.Platform.osname === "android"){
        textArea.setFocusable(false)
        setTimeout(function(){
            textArea.setFocusable(true)
        },100);    
    }    
});
