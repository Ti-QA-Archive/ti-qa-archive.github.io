    hi,

    You can add a 'click' event listener on table and get the 'title' of the row clicked to update the textfield.

    var win = Ti.UI.currentWindow;
var textfield = Titanium.UI.createTextField({
    height:35,
    top:0,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});
var data = [{title:"Row 1"},{title:"Row 2"}];
var table = Titanium.UI.createTableView({data:data, top:50});

table.addEventListener("click", function(e){
textfield.value = e.rowData.title;
});

win.add(textfield, table);
