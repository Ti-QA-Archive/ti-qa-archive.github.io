Titanium4j 3.1 is here !

Hello Folks,

We are very exitecd to announce the general availability of Ti4j 3.1
Ti4j is our Java binding for Titanium that allows developers to write Titanium applications in Java.
Promiment feature in this release are :

1)
Support for Titanium SDK 3.1

2) Cleaner and more intuitiv API

package com.emitrom.ti4j.starter.client;

//imports here

/**
 * Entry point classes define <code>onModuleLoad()</code>.
 */
public class Ti4jMobile extends TiEntryPoint {

    @Override
    public void onStart() {
        // Let's hide the status bar on the iphone/ipad for neatness
        if (Platform.get().isIPhone() || Platform.get().isIPad()) {
            IPhone.get().setStatusBarHidden(true);
        }

        Window window = new Window();
        window.setBackgroundColor(Color.BISQUE);

        Button button = new Button("Tap Me");
        button.addClickHandler(new ClickHandler() {
            @Override
            public void onClick(ClickEvent event) {
                Titanium.alert("Emitrom", "Hello world from Ti4j Mobile");
            }
        });
        window.add(button);
        window.open();
    }
}

3) Support for GWT UI Binder, which enables layouting the UI in a declarative way.

<!DOCTYPE ui:UiBinder SYSTEM "http://dl.google.com/gwt/DTD/xhtml.ent">
<ui:UiBinder xmlns:ui="urn:ui:com.google.gwt.uibinder"
xmlns:ti="urn:import:com.emitrom.ti4j.mobile.client.ui"
xmlns:ios="urn:import:com.emitrom.ti4j.mobile.client.ui.ios"
 >
 <ui:import field="com.emitrom.ti4j.mobile.client.ui.iphone.SystemButton.*" />
 <ui:import field="com.emitrom.ti4j.mobile.client.ui.iphone.SystemButtonStyle.*" />
 <ui:import field="com.emitrom.ti4j.core.client.Color.*" />


  <ti:Window  backgroundColor="gray">
      <ti:View>
        <ti:Label  left="0" top="0" text="Hello World" color="white" />
        <ti:Button  left="100" title="Click Me"/>
      </ti:View>
      <ios:Toolbar top="0">
           <ti:Button systemButton="{SYSTEM_BUTTON_FLEXIBLE_SPACE}"/>
           <ti:Button  systemButton="{SYSTEM_BUTTON_CAMERA}" />
           <ti:Button title="Actions" style="{SYSTEM_BUTTON_STYLE_DONE}" />
      </ios:Toolbar>
  </ti:Window>    
</ui:UiBinder>
