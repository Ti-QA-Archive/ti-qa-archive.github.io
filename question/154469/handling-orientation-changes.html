    • I have used them as well but no luck with that.

      if(Titanium.Gesture.orientation == Titanium.UI.PORTRAIT || Titanium.Gesture.orientation == Titanium.UI.UPSIDE_PORTRAIT){
      detailView.animate({right:0,left:0});
      win.barImage = null;
      win.barImage = 'ui/images/IPADNavBGPortrait.png';
      }else if(Titanium.Gesture.orientation == Titanium.UI.LANDSCAPE_LEFT || Titanium.Gesture.orientation == Titanium.UI.LANDSCAPE_RIGHT){
      detailView.animate({right:0,left:0});
      win.barImage = null;
      win.barImage = 'ui/images/IPADNavBGPortrait.png';
      }
