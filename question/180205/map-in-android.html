  • Hi Shahaab Aslam Khan,

    Thanks for your concern.

    Please add the Google API key and permissions to the tiapp.xml file and add latitude, longitude to your map view like the below code and see if it works for you.

    var Map = require('ti.map');
var win = Ti.UI.createWindow();

var annotations = [
    Map.createAnnotation({
        latitude: 32.8245525,
        longitude: -117.0951632,
        title: 'Somewhere',
        animate: true,
        pincolor: Map.ANNOTATION_GREEN,
        draggable: true
    }),
];
var mapview = Map.createView({
    mapType: Map.NORMAL_TYPE,
    region: {
        latitude:32.8245525,
        longitude:-117.0951632,
        latitudeDelta:.05,
        longitudeDelta:.05
    },
    animate:true,
    regionFit:true,
    userLocation:false,
    annotations: annotations
});

win.add(mapview);
win.open();

    Hope it helps.

    Thanks
