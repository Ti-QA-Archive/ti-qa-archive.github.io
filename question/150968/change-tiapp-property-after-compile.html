  • You can do this with a CLI hook. In your project directory, create the directories "plugins/myplugin/1.0/hooks". In this hooks folder, create a JavaScript file with whatever name you want such as "myhook.js", then add this code:

    var fs = require('fs'),
    path = require('path');

exports.cliVersion = '>=3.0.25';

exports.init = function (logger, config, cli) {
    cli.addHook('build.pre.compile', function (build, finished) {
        logger.info('The build is about to start!');

        // do whatever you want before the build starts

        finished();
    });

    cli.addHook('build.post.compile', function (build, finished) {
        logger.info('The build is complete!');

        // parse the tiapp.xml and update whatever you want
        var tiappFile = path.join(cli.argv['project-dir'], 'tiapp.xml');
        fs.writeFileSync(
            tiappFile,
            fs.readFileSync(tiappFile).toString().replace(
                /(serverUrl[^>]*\>)([^<])*/,
                '$1http://www.google.com/'
            )
        );

        finished();
    });
};

    Then you need to activate the plugin in the tiapp.xml:

    <plugins>
    <plugin>myplugin</plugin>
</plugins>

    Next, build your app and look for the log output.

    Note: this code is not tested and has no warranty. Be adventurous at your own risk. :)
