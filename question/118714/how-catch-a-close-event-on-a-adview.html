How catch a 'close' event on a AdView

Hello,

i'm building my first app with appcelerator and I have an issue with an AdView.
I put my ad in th top of of windo and move the current view dow with an animation in order to haven't any overlaps with my content and the add on a 'Load' event. But after 60seconds, the Ad desapear but I'm not able to catch this 'close Ad' event.

Does anybody succeed to fix this ?

Here is my code :

var globalView = Titanium.UI.createScrollView({
backgroundColor : '#000',
top: 0,
contentWidth:'250',
contentHeight:'800',
showVerticalScrollIndicator:true,
showHorizontalScrollIndicator:true
}) ;

  if (parseFloat(Titanium.Platform.version) >= 3.2){
         var iad = Ti.UI.iOS.createAdView({width: 'auto',height: 'auto', top: 0});
        leRecap.add(iad);
        adViewAnimationOpen = Titanium.UI.createAnimation({top:50, duration:750});
        adViewAnimationClose = Titanium.UI.createAnimation({top:0, duration:750});
        iad.addEventListener('load', function(e)
            {
                globalView.animate(adViewAnimationOpen);
            })
        iad.addEventListener('close', function(e)
                      {
                          globalView.animate(adViewAnimationClose);
                      })

    }
