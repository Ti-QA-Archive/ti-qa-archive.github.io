  • Hi GAURAV jain

    Please try the bellow code.I hope, it will fulfill your need.The bellow code updates the second picker with the use of first picker

    picker1.addEventListener('change',function(e)
{
    var data=[];
    operatorService= e.row.title;
    set=e.row.custom_item;
    //alert(set);
    //picker.add(data);
    switch(set)
    {
        case "1":

            data[0]=Ti.UI.createPickerRow({title:'Airtel'});
            data[1]=Ti.UI.createPickerRow({title:'Idea'});
            data[2]=Ti.UI.createPickerRow({title:'Reliance'});
            picker.add(data);
            calltoRemove(data);
               break;
        case "2":
            data[0]=Ti.UI.createPickerRow({title:'Vodafone'});
            data[1]=Ti.UI.createPickerRow({title:'Bsnl'});
            data[2]=Ti.UI.createPickerRow({title:'Aircel'});
            picker.add(data);
            calltoRemove(data);
            break;
        case "3":

            data[0]=Ti.UI.createPickerRow({title:'Tata'});
            picker.add(data);
            calltoRemove(data);
            break;

    }
});

function calltoRemove(data)
{
        var numRows=picker.children[0].rowCount;
        for(var a=0;a<data.length;a++)
        {
            picker.children[0].children[a].title=data[a].title;
        }
        for(var b=numRows-1;b>=data.length;b--)
        {
            picker.children[0].removeRow(picker.children[0].children[b]);
        }
}
