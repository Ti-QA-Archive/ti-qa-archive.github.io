  • You're not very specific, but this may give you an idea:

    var the_top = 0
var the_left= 0

the_btn.addEventListener('click', function(){

    var the_img = Ti.UI.createImageView({
        image:'path_to_image.jpg',
        width:100,
        height:100,
        opacity:0.75,
        top:the_top,
        left:the_left
    });

    the_window.add(the_img);

    the_top = the_top+20;
    the_left = the_left+20;

});

    of course you will have to adapt the code to your needs
