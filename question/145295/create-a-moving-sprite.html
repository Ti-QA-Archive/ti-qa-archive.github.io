Create a moving sprite.

Hi all,

I was trying to make a cloud move across the screen towards the right and then restart back at the left.

var cloud = Titanium.UI.createView ({
top: 100,
left:100,
height: 'auto',
width:'auto',
backgroundImage = 'cloud.png'
});

Unfortunately, using cloud.animate doesn't seem to work…
Any ideas on how to fix this?

  • Try this:

    var parentView = win; // Whatever your containing view is, like a window.

var resetLeft = function() {
    cloud.left = 0;
}
var leftToRightAnimation = Ti.UI.createAnimation({
    left : 900, // Parent view width + cloud.width
    duration : 3000,
    curve : Ti.UI.ANIMATION_CURVE_LINEAR
});
leftToRightAnimation.addEventListener('complete', function() {
    resetLeft();
    cloud.animate(leftToRightAnimation);
});
var cloud = Ti.UI.createView({
    top : 100,
    left : 100,
    height : 30, // Do not use 'auto'
    width : 30, // Do not use 'auto'
    backgroundImage : 'cloud.png'
});
cloud.animate(leftToRightAnimation);
parentView.add(cloud);
