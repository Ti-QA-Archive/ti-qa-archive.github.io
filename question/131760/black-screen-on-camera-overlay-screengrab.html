black screen on camera overlay screengrab

Hi all,
I am hoping you can help me.

I want to start the camera, overlay an image on top of the camera and then save out a screenshot of the newly compiled image . Basically, the user can see what they look like wearing a hat.

Anyway, overlay etc all works well, but the screenshot shows a black background where I was expecting to see the camera output. So all I see is the hat on a black square. Grrr. Can anyone give me pointers as to where I'm going wrong?

Thanks,
Lee

// create window
var win = Titanium.UI.createWindow();


// add hat image
var imageHat = Ti.UI.createImageView({
                left : 0,
                top : 0,
                image : 'hat.png',  
                });


var takePhotoButton = Ti.UI.createButton({
    title:'> Press To Take Photo <',
    width: 200,
    height:50, 
    bottom:1,
});             


// create the overlay view with hat and button
var overlay = Titanium.UI.createView();
overlay.add(imageHat);
overlay.add(takePhotoButton);



// call camera and add overlay
Titanium.Media.showCamera({
    overlay:overlay,
    showControls:false, 
    mediaTypes:[Ti.Media.MEDIA_TYPE_PHOTO]
});



takePhotoButton.addEventListener('click', function(){

        // hide the button
        takePhotoButton.hide();

        //do screengrab
        Titanium.Media.takeScreenshot(function(event)
        {
            // set blob on image view
            screenGrab.image = event.media;
        });

        // switch off camera
        Ti.Media.hideCamera();

});


// open window
win.open();

  • Hi L F,
    Try this 

    
var cameraView = Ti.UI.createView({
    width : 320,
    height : 460
});

var viewButtons = Ti.UI.createView({
    bottom : 0,
    left : 0,
    width : 320,
    height : 48,
    backgroundImage : 'imageview.png'
});

var btnCancel = Ti.UI.createButton({
    title : 'cancel',
    bottom : 7,
    left : 10,
    width : 141,
    height : 32
});

btnCancel.addEventListener('click', function(e) {
    Ti.Media.hideCamera();
});

viewButtons.add(btnCancel);

var btnClick = Ti.UI.createButton({
    title : 'save',
    bottom : 7,
    right : 10,
    width : 141,
    height : 32
});

btnClick.addEventListener('click', function(e) {
    Ti.Media.takePicture();
});
viewButtons.add(btnClick);

cameraView.add(viewButtons);

    use this code on your camera overlay

    Titanium.Media.showCamera({
    mediaTypes : Ti.Media.MEDIA_TYPE_PHOTO,
    showControls : false, 
    overlay : cameraView
});
