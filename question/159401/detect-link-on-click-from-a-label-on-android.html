Detect link on click from a label on Android

Hi, html is available for android in labels, but how is it possible to detect a click on the links in the label and in order to make the user go to that webpage?

    var win = Ti.UI.createWindow({
    backgroundColor : 'black'
});

var lbllink1 = Ti.UI.createLabel({
    top : 50,
    color : 'white',
    text : 'Click: http://www.google.com',
    autoLink : Ti.UI.Android.LINKIFY_WEB_URLS //use auto link 
});

var lbllink2 = Ti.UI.createLabel({
    top : 150,
    color : 'white',
    html : "<a href='http://www.google.com'>Click me</a>", //you can use html this way
});

lbllink2.addEventListener('click', function() {
    Ti.Platform.openURL('http://www.google.com');
});
win.add(lbllink1);
win.add(lbllink2);
win.open();
