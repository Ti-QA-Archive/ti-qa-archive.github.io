    • in app.js set this 

      Ti.App.tabGrp = tabGroup;

      and in you first tab's window
      set this

      var win = Ti.UI.currentWindow;
var tabGroup = Ti.App.tabGrp;

win.addEventListener('open',function(e){
    tabGroup.animate({
        bottom : 400,
        duration : 100
    });
});
