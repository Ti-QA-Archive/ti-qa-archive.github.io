A question regarding the NavigationController.home of Forging Titanium Ep. 2

I noticed that the home method is implemented like this:

//go back to the initial window of the NavigationController
exports.NavigationController.prototype.home = function() {
    //store a copy of all the current windows on the stack
    var windows = this.windowStack.concat([]);
    for(var i = 1, l = windows.length; i < l; i++) {
        (this.navGroup) ? this.navGroup.close(windows[i]) : windows[i].close();
    }
    this.windowStack = [this.windowStack[0]]; //reset stack
};

What's the purpose of storing this.windowStack in the local windows variable rather than performing the operations directly on the windoStack, e.g.:

//go back to the initial window of the NavigationController
exports.NavigationController.prototype.home = function() {
    for(var i = 1, l = this.windowStack.length; i < l; i++) {
        (this.navGroup) ? this.navGroup.close(this.windwStack[i]) : this.windowStack[i].close();
    }
    this.windowStack = [this.windowStack[0]]; //reset stack
};

Is it only for improved readability?
