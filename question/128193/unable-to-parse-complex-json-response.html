    No, no, no …

    You don't need to parse XML because "JSON.parse" returns JavaScript object:

    var parsedObject = { 
    "status": "OK", 
    "message": "success", 
    "serverTime": 1320759235, 
    "0": { 
        "songId": "112885", 
        "songName": "SAN SAN SAN WOH ", 
        "albumName": "Kaagaz Ke Phool" 
    },
     "1": { 
        "songId": "112885", 
        "songName": "SAN CHALI HAWA", 
        "albumName": "Kaagaz Ke Phool" }, 
    "2": { 
        "songId": "112886", 
        "songName": "TITLE MUSIC -", 
        "albumName": "Kaagaz Ke Phool" 
    } 
};

    You can use it as any other JavaScript object.

    Ti.API.info(parsedObject["0"]["songName"]);

    The problem you have is in data format. It would be better if JSON string would look like:

    '{ "status": "OK", "message": "success", "serverTime": 1320759235, "data": [ 0: { "songId": "112885", "songName": "SAN SAN SAN WOH ", "albumName": "Kaagaz Ke Phool" }, 1: { "songId": "112885", "songName": "SAN CHALI HAWA", "albumName": "Kaagaz Ke Phool" }, 2: { "songId": "112886", "songName": "TITLE MUSIC -", "albumName": "Kaagaz Ke Phool" }] }'

    because "data" property would contain array of song data. Then you could use it:

    var songs = parsedObject['data'];
Ti.API.debug(songs);

    But, you can extract that data simply by deleting unnecessary things:

    delete parsedObject['status'];
delete parsedObject['message'];
delete parsedObject['serverTime'];

var songs = [];

for (var p in parsedObject) {
    songs.push(parsedObject[p]);
}

Ti.API.debug(songs);
