How to open several window one after another?

This is my situation, I have created a tab group of five tabs in app.js:

// create tab group
var tabGroup = Titanium.UI.createTabGroup();

// create tabs and windows
//Access
var accessWin = Titanium.UI.createWindow({  
    url:'access.js',
    navBarHidden:true,
    backgroundColor:'white'
});
var accessTab = Titanium.UI.createTab({  
    icon:'images/acceso.png',
    title:'Acceso',
    window:accessWin
});

...etc.

After that, in access.js I create a new window. This window creates fine with its navbar, title and everything. It creates outside of the tab and it covers the tab group:


buttonTest.addEventListener('click', function()
{
    var w = Titanium.UI.createWindow({
        backgroundColor:'white',
        title:'Pruébenos 15 días',
        barColor:'black',
        url:'demoRegister.js',
    });
    var b = Titanium.UI.createButton({
        title:'Cerrar',
        style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN
    });
    w.setLeftNavButton(b);
    b.addEventListener('click',function()
    {
        w.close();
    });
    w.open({modal:true});
});

Finally, in demoRegister.js I need to create a third window and this is when I have problems. I try to create a window the same way I created the second window in access.js but for some reason, the second window closes and the third window creates over the first window and without navBar:


    var win = Titanium.UI.createWindow({
        backgroundColor:'white',
        title:'Condiciones del Servicio',
        barColor:'black',
        navBarHidden:false,
        url:'serviceConditions.js'
    });
    var b = Titanium.UI.createButton({
        title:'Cerrar',
        style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN
    });
    win.setLeftNavButton(b);
    b.addEventListener('click',function()
    {
        win.close();
    });
    win.open({modal:true});

Somebody can explain this behaviour? How can I create a third window with its navBar and over the second window?
