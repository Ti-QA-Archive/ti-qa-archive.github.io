How to lock screen updates when closing multiple windows in a NavigationGroup

I can close all windows currently in the "stack" of the navGroup using a similar technique as described in many other threads, but I get each window closed to be displayed for a short time.

var numWindowsToClose = selectConditionNavWinStack.length;
for (ix = 0; ix < numWindowsToClose; ix++) {
    var win = _.last(selectConditionNavWinStack);
    win.removeEventListener('close', windowCloseListener);
    this.close(win, {animated : false});
    selectConditionNavWinStack.pop();
}

"this" in the code is the navGroup.
How can I lock the screen while closing the windows?

  • hi,

    if you want to show a window for a short time then instead of using for loop you can use interval method.

    var tm = setInterval(function(e) {
    if (selectConditionNavWinStack) {
        var win = _.last(selectConditionNavWinStack);
        win.removeEventListener('close', windowCloseListener);
        this.close(win, {
            animated : false
        });
        selectConditionNavWinStack.pop();
    }else{
        selectConditionNavWinStack = null;
        clearInterval(tm);
    }

}, 10);
