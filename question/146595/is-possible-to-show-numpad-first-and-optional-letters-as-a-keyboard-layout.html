  • Look in the kitchen sink. Under controls->textfield-keyboards there is an example of how to assign different keyboards to certain textfields.

    Here is an example of a textfield when focused shows a number pad:

    var win = Ti.UI.createWindow();

    win.backgroundColor = "black";

    var tf1 = Titanium.UI.createTextField({
        color:'white',
        height:50,
        top:10,
        left:10,
        width:250,
        keyboardType:Titanium.UI.KEYBOARD_NUMBER_PAD,
        returnKeyType:Titanium.UI.RETURNKEY_DEFAULT,
        borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED

    });
    tf1.addEventListener('return', function()
    {
        tf1.blur();
    });

    win.add(tf1);

win.open();

    Hope this helps,

    Carter
