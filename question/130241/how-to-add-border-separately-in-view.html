  • This is the code I'm using

    function BorderBottom () {
    var brd = Ti.UI.createView({
        backgroundColor: '#aaaaaa',
        height: 1,
        bottom: 0
    });

    return brd;
}

function BorderTop () {
    var brd = Ti.UI.createView({
        backgroundColor: '#aaaaaa',
        height: 1,
        top: 0
    });

    return brd;
}

exports.BorderBottom = BorderBottom;
exports.BorderTop = BorderTop;

  • another solution: add a big view with background color, and the children of it with 1 px distance

    var container = Ti.UI.createView({
    backgroundColor: '#000',
    width: 300,
        layout: 'vertical'
});

for (var i = 0; i < 100; i++) {
    var child = Ti.UI.createView({
        height: 30,
        width: '100%',
        backgroundColor: '#fff',
            top: 1
    });
    container.add(child);
}
