    Hello Hund,

    Thanks for your inquiry, Here is a sample app for play/pause/resume/reset….. audio. 

    Ti.UI.setBackgroundColor('white');
var win = Titanium.UI.createWindow({
    title : 'Sound',
    layout: 'vertical'
});

var timob7502fix = (Ti.version >= '3.0.0');

var sound = Titanium.Media.createSound();
sound.url = '/bugsbunny1.wav';

//
// PLAY
//
var play = Titanium.UI.createButton({
    title : 'Play',
    height : 40,
    width : 145,
});

play.addEventListener('click', function() {
    sound.play();
    if (timob7502fix) {
        pb.max = sound.duration * 1000;
    } else {
        pb.max = sound.duration;
    }
});
win.add(play);

//
// PAUSE
//
var pause = Titanium.UI.createButton({
    title : 'Pause',
    height : 40,
    width : 145,
});
pause.addEventListener('click', function() {
    sound.pause();
});
win.add(pause);

//
// RESUME
//
var resume = Titanium.UI.createButton({
    title : 'Resume',
    height : 40,
    width : 145,
});
resume.addEventListener('click', function() {
    sound.play();
});
win.add(resume);

//
// RESET
//
var reset = Titanium.UI.createButton({
    title : 'Reset',
    height : 40,
    width : 145,
});
reset.addEventListener('click', function() {
    sound.reset();
    pb.value = 0;

});
win.add(reset);

//
// STOP
//
var stop = Titanium.UI.createButton({
    title : 'Stop',
    height : 40,
    width : 145,
});
stop.addEventListener('click', function() {
    sound.stop();
    pb.value = 0;
});
win.add(stop);

//
// LOOPING
//
var looping = Titanium.UI.createButton({
    title : 'Looping (false)',
    height : 40,
    width : 145,
});
looping.addEventListener('click', function() {
    sound.looping = (sound.looping === false) ? true : false;
    looping.title = 'Looping (' + sound.isLooping() + ')';
});
win.add(looping);

//
// EVENTS
//
sound.addEventListener('complete', function() {
    pb.value = 0;
});

//
//  PROGRESS BAR TO TRACK SOUND DURATION
//

var pb = Titanium.UI.createProgressBar({
    min : 0,
    value : 0,
    width : 200
});

win.add(pb);
pb.show();

var i = setInterval(function()
    {
        if (sound.isPlaying())
        {
            Ti.API.info('time ' + sound.time);
            pb.value = sound.time;

        }
    },500);

win.open();

    Hope this helps you. Thanks.
