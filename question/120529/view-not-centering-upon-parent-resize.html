View not centering upon parent resize

Hi,

I have a view that I am animating height on. Inside the view is a textField. At the same time as the parent view's animation, the textField's opacity animates. 

var down = function() {
    searchContainer.animate({height:50, duration:250});
    search.animate({opacity:0, duration:250});
    search.hidden = true;
};

var up = function() {
    search.hidden = false;
    searchContainer.animate({height:75, duration:250});
    search.animate({opacity:1, duration:250});
};

For some reason, the textField centers itself only when its parent view is shrinking.

Any ideas? Thanks
