[ERROR] Script Error = 'undefined' is not an object (evaluating 'Ti.Database.open') at db.js (line 4)

Any idea why I am getting this ?

Following is the code

//bootstrap database
var database_name = 'productsManager';

var db = Ti.Database.open(database_name);
db.execute('CREATE TABLE IF NOT EXISTS products(product_id INTEGER PRIMARY KEY, store_id INTEGER, purchase_date TEXT, return_date TEXT, has_returns INTEGER, remind INTEGER);');
db.execute('CREATE TABLE IF NOT EXISTS Stores(store_id INTEGER PRIMARY KEY, store_name TEXT);');
db.execute('CREATE TABLE IF NOT EXISTS productImages(image_id INTEGER PRIMARY KEY, product_id INTEGER, image BLOB);');
db.close();

exports.add_new_store = function(_store_name) {
var db = Ti.Database.open(database_name);
db.execute("INSERT INTO Stores(store_name) VALUES(?)",_store_name);
db.close();

exports.stores_list = function() {
var storesList = [];
var db = Ti.Database.open(database_name);

storesList.push({
