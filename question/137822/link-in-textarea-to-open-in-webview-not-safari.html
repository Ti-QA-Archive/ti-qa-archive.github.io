ta1.addEventListener('click',function(e){
    var v = ta1.value;

    if(v.indexOf('http') != -1){
        // open new window with webview
        var tubeWindow = Ti.UI.createWindow({ 
        modal: true,
        barColor: '#050505',
        backgroundColor: '#000' 
    }); 

        var temp = v.split('http');
        var temp1 = temp[1].split(" ");
        var Url = "http"+ temp1[0];

        var linkview = Ti.UI.createWebView({
                url:  'http://' + temp1[0],
                barColor: '#050505',
                backgroundColor: '#050505'              
            });


             var htmlHack = '';

htmlHack += 'var head = document.getElementsByTagName("head")[0];';
htmlHack += 'head.appendChild(element);';

linkview.addEventListener('load', function(e) {
    linkview.evalJS(htmlHack);
});




     tubeWindow.add(linkview);

    });
