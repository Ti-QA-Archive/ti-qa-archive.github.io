  • Hi

    You need to structure you code and the order in which you define the httpClient method better, following this structure which is cleaner and includes your code correctly referenced.

    var url = url; // this seems weird
var xhr = Ti.Network.createHTTPClient({
    onload: function(e) {
        var json = JSON.parse(this.responseText);
        var lblModelYear = Titanium.UI.createLabel({
            text: json.ModelYear,
            color: '#fff'
        });
        view_redeem2.add(lblModelYear);

        var lblMake = Titanium.UI.createLabel({
            text: json.Make,
            color: '#fff'
        });
        view_redeem2.add(lblMake);

        var lblModel = Titanium.UI.createLabel({
            text: json.Model,
            color: '#fff'
        });
        view_redeem2.add(lblModel);

        var lblBodyType = Titanium.UI.createLabel({
            text: json.BodyType,
            color: '#fff'
        });
        view_redeem2.add(lblBodyType);

        // and more
    },
    onerror: function(e) {
        // this function is called when an error occurs, including a timeout
        Ti.API.debug(e.error);
        alert('error');
    },
    timeout:5000  /* in milliseconds */
});
xhr.open("GET", url);
xhr.send();  // request is actually sent with this statement

    Note that your use of json.modelyear would not have worked as your JSON response has capitalisation that MUST be honoured. So it becomes ModelYear.
