  • This will create a webview, then when it is done loading execute some javascript to disable the touchmove event.

    var webView = Ti.UI.createWebView({url: 'http://www.youtube.com'});
webView.addEventListener('load', function(evt){
this.evalJS('document.addEventListener("touchmove", function(e){ e.preventDefault(); });');
});
