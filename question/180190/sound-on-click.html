  • No library is required. Just add the following code.

    var play = Titanium.UI.createButton({ title:'Play', height:40, width:145, left:10, top:10 });
play.addEventListener("touchstart", function() { 
var sound = Ti.Media.createSound({url:'cricket.wav', preload:true});
sound.play();
});
win.add(play);
