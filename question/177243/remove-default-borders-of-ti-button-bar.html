    Hey Adnan,

    you don't have to add a eventlistener for every button, you can also use on the parent view, and identify the child clicks by an id. This works because Titanium bubbles the events from the children to the parent views.

    see this example:

    var buttonBar = Ti.UI.createView({height:50});
var leftButton = Ti.UI.createButton({clickId:"leftButton", title:"left", left:10});
var rightButton = Ti.UI.createButton({clickId:"rightButton", title:"right", right:10});
buttonBar.add(leftButton);
buttonBar.add(rightButton);

function buttonClicked(e) {
    if (!e.source.clickId) {
        return;
    }
    switch(e.source.clickId) {
        case "leftButton": 
            Ti.API.info("leftButton clicked");
            break;
        case "rightButton":
            Ti.API.info("rightButton clicked");
            break;
    }
}

buttonBar.addEventListener('click', buttonClicked);
