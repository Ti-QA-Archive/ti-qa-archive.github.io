    • This method allows you to pass the parameters into the object and if you do not have a value it ignores the line and only uses those that are required - leaving no gaps.

      function DetailScreen(obj) {
    var self = Ti.UI.createWindow({
        backgroundColor: 'white'
    });
    var scroll = Titanium.UI.createScrollView({
        contentWidth: Ti.UI.SIZE,
        contentHeight: Ti.UI.SIZE,
        top: '2%',
        layout: 'vertical',
        showVerticalScrollIndicator: true,
        showHorizontalScrollIndicator: false 
    });
    self.add(scroll);
    var myfontsize = (Titanium.Platform.displayCaps.platformHeight * 4) / 100;
    if (obj.story) {
        var labStory = Ti.UI.createLabel({
            text: obj.story,
            font: {
                fontSize: myfontsize,
                fontWeight: 'bold'
            },
            height: Ti.UI.SIZE,
            left: '2%',
            top: 0
        });
        scroll.add(labStory);
    }
    if (obj.newwords) {
        var labNewWords = Ti.UI.createLabel({
            text: obj.newwords,
            font: { 
                fontSize: myfontsize * 0.7
            },
            top: 0,
            height: Ti.UI.SIZE,
            left: '2%'
        });
        scroll.add(labNewWords)
    }
    if (obj.exp) {
        var labExp = Ti.UI.createLabel({
            text: obj.exp,
            font: {
                fontSize: myfontsize * 0.6
            },
            top: 0,
            height: Ti.UI.SIZE,
            left: '2%'
        });
        scroll.add(labExp)
    }
    return self;
}
exports.DetailScreen = DetailScreen; // personal preference

      Usage

      // provide all three
var Scroll = require('/scroll');

    story: 'One',
    newwords: 'Two',
    exp: 'Three'
});
win.add(scroll);

or

    story: 'One',

    • This example handles JUST the three line view in a single method so you can re-use it anywhere else in your app.

      ui.js

      function detail(obj) {
    var myfontsize = (Titanium.Platform.displayCaps.platformHeight * 4) / 100;
    var view = Titanium.UI.createView({
        height: Ti.UI.SIZE,
        layout: 'vertical',
        left: '2%',
        right: '2%',
        top: '2%',
        width: Ti.UI.SIZE
    });
    if (obj.story) {
        var labStory = Ti.UI.createLabel({
            font: {
                fontSize: myfontsize,
                fontWeight: 'bold'
            },
            height: Ti.UI.SIZE,
            left: 0,
            text: obj.story,
            top: 0,
            width: Ti.UI.FILL
        });
        scroll.add(labStory);
    }
    if (obj.newwords) {
        var labNewWords = Ti.UI.createLabel({
            font: { 
                fontSize: myfontsize * 0.7
            },
            height: Ti.UI.SIZE,
            left: 0,
            top: 0,
            text: obj.newwords,
            width: Ti.UI.FILL
        });
        scroll.add(labNewWords)
    }
    if (obj.exp) {
        var labExp = Ti.UI.createLabel({
            font: {
                fontSize: myfontsize * 0.6
            },
            height: Ti.UI.SIZE,
            left: 0,
            top: 0,
            text: obj.exp,
            width: Ti.UI.FILL
        });
        scroll.add(labExp)
    }
    return view;
}
exports.detail = detail;

      Usage

      var UI = require('/ui');
var detail = UI.detail({
    story: 'One',
    newwords: 'Two',
    exp: 'Three'
});
win.add(detail);
