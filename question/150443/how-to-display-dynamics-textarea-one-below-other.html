    hi,

    you can do that with setting view's layout property to vertical

    var view = Ti.UI.createView({
    backgroundColor: '#EEEEEE',
    layout : 'vertical'
});

scrollView.add(view)
//
//
//
var textoDescricao = Ti.UI.createTextArea({
    borderWidth : 3,
    borderColor : '#bbb',
    borderRadius : 5,
    color : 'black',
    backgroundColor : '#EEEEEE',
    font : {
        fontSize : 18,
        font : 'Noteworthy-Light'
    },
    textAlign : 'left',
    value : 'Sobre: ' + '    ' + descricao + '                                                           sdghd                                           gdfh                                    tgkgkuy g                                                   ygoyg',
    top : 20,
    width : 300,
    height : 'auto',
    editable : false
})
view.add(textoDescricao)
//
//
//
var textoLabel = Ti.UI.createTextArea({
    borderWidth : 3,
    borderColor : '#bbb',
    borderRadius : 5,
    color : 'black',
    backgroundColor : '#EEEEEE',
    font : {
        fontSize : 18,
        font : 'Noteworthy-Light'
    },
    textAlign : 'left',
    value : 'Genero: ' + ' ' + genero,
    width : 300,
    height : 'auto',
    editable : false
})
view.add(textoLabel)
