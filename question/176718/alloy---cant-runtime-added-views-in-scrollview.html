    • Yes, you are requiring it and creating a separate instance of it. So, what you want to do is call an exported method on your Required object to set the values. ie

      index.js

      $.index.open();
$.composer.load(message);

      composer.js

      exports.load = function(args) {
    if (args) {
        loadMsg(args);
        $.trigger("messageLoaded", {});
    }
};

  • Tnx a lot for the quick answer, that's pretty much what i've done.
    So now my code - sorry for all the code but maybe it;s something small and i don't want to miss anything.
    The traces shows that $.scrollRegions has 2 children and that the texts are correct. just can't see it on screen
    index.xml:

    <Alloy>
    <NavigationWindow id="index" class="main" platform="ios">
        <Require type="view" src="composer" id="composer"/>
    </NavigationWindow>
</Alloy>

    index.js (when message is an object):

    Alloy.createController('composer', {message: message});

$.index.open();

    composer.xml:

    <Alloy>
    <Window title="Composer">
        <ScrollView id="regionsScroll"></ScrollView>
       </Window>
</Alloy>

    composer.js:

    var args = arguments[0] || {};

function addTextRegion(region) {
    var ctrl = Alloy.createController('textRegion', {
        data: region.data
    });
    var txtRegionView = ctrl.getView();
    Ti.API.info("addTextRegion - txtArea text: " + ctrl.getText());
    $.regionsScroll.add(txtRegionView);

    Ti.API.info("regionsScroll num children: " + $.regionsScroll.children.length);
}

function loadMsg(msg) {
    Ti.API.info("composer controller - loadMsg - message: " + JSON.stringify(msg));
    _.each(message.regions, function(region){
        switch (region.type) {
            case "text":
                addTextRegion(region);
                break;
        }
    });
    Ti.API.info("composer controller - loadMsg - done");
}

var message = args.message;
if (message) {
    loadMsg(message);
    $.trigger("messageLoaded", {});
}

    textRegion.xml:

    <Alloy>
    <View class="txtRegionContainer">
        <TextArea id="txtArea" class="txtAreaClass"></TextArea>
    </View>
</Alloy>

    textRegion.js:

    var args = arguments[0] || {};

var data = args.data;
Ti.API.info("textRegion controller: " + JSON.stringify(data));
$.txtArea.value = data.html;

exports.getText = function(){
    return $.txtArea.value;
};
