Blur not working on textfield

Following code returns "Enter was pressed" but does not hide the keyboard:

var field = Ti.UI.createTextField({
        color: 'blue',
        height:34,
        left:10,
        width:470,
        hintText:'hint',
        borderStyle:Titanium.UI.INPUT_BORDERSTYLE_NONE,
        color:'#336699'
    });

    field.addEventListener('return', function(e) {
        field.blur();
        Ti.API.info('Enter was pressed');
    });

Any idea why the keyboard is not disappearing?

    after posting i thought it would have been clever to paste the full code. Sorry for that:

    • I have a window (win)
    • pressing a button (button_plus) opens a modal window
    • Modal window (window) contains a grouped table (table) with one row (row1)
    • This row holds the textfield (field)
    var button_plus = Ti.UI.createButton({
    backgroundImage: './images/button_plus.png',
    height: 29,
    width: 29,
});

var win = Ti.UI.currentWindow;
win.barImage = './images/navigation_bg.png';
win.barColor = 'white';
win.title = 'Folders';
win.rightNavButton = button_plus;

button_plus.addEventListener('click',function(e) {
    var closeButton = Titanium.UI.createButton({
        systemButton:Titanium.UI.iPhone.SystemButton.CANCEL
    });

    var window = Titanium.UI.createWindow({
        title: 'Create New Folder',
        leftNavButton: closeButton
    });

    window.open({
            modal:true,
            modalTransitionStyle: Ti.UI.iPhone.MODAL_TRANSITION_STYLE_COVER_VERTICAL,
            modalStyle: Ti.UI.iPhone.MODAL_PRESENTATION_FORMSHEET
    });

    var row1 = Ti.UI.createTableViewRow({
        title:'',
        selectionStyle:Ti.UI.iPhone.TableViewCellSelectionStyle.NONE
    });

    var field = Ti.UI.createTextField({
        color: 'blue',
        height:34,
        //top:10,
        left:10,
        width:470,
        hintText:'hint',
        borderStyle:Titanium.UI.INPUT_BORDERSTYLE_NONE,
        color:'#336699'
    });

    row1.add(field);
    var data = [row1];
    var table = Titanium.UI.createTableView({
        style:Titanium.UI.iPhone.TableViewStyle.GROUPED,
        data:data
    });
    window.add(table);

    closeButton.addEventListener('click',function(e) {
        window.close();
    });

    field.addEventListener('return', function(e) {
        field.blur();
        Ti.API.info('Enter was pressed');
    });
});
