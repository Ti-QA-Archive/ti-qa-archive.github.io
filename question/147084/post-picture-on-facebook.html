  • Hi Subhan,

    god news: it is a standard task. First you go to Faceebook and you decide what you want: sending to own profile, to page … Now you know endpoint, parameters …

    Now you can use Facebook-API from Titanium to speak with FBAPI.

    Here an example to get all likes of user:

    exports.get = function(_id, _callback) {
    function getFriendLikes() {
        var path = '/' + _id + '/likes?locale=de_DE';
        Ti.Facebook.requestWithGraphPath(path, {}, 'GET', function(_fb) {
            if (_fb.success) {
                //////  work with datas;
                _callback(res);
            } else {
                _callback(null);
            }
        });
    }
    Ti.Facebook.appid = '2029:::::504444';
    Ti.Facebook.permissions = ['read_friendlists', 'user_likes', 'friends_likes'];
    if (Ti.Facebook.loggedIn == false) {
        Ti.Facebook.authorize();
        Ti.Facebook.addEventListener('login', function(_e) {
            if (_e.success == true) {
                getFriendLikes();
            }
        });
    } else
        getFriendLikes();
};
