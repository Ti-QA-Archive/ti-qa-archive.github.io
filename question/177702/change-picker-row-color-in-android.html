    You can change the PickerRow backgroundColor with a custom Theme

    Something like this…

    <style name="MyCustomDropDownItemStyle"> parent="@android:style/Widget.AppCompat.Light.DropDownItem.Spinner">
    <item name="android:background">@color/white</item>
</style>

    If you're not familiar with defining custom Android styles and themes, you may want to use the Android Action Bar Style Generator which is also able to style the Spinner appearance (by Popup color).

  • A simple solution, you can use TableView instead of picker and customize it according to your need as Titanium provides a variety of way to do the task.
    use below code in app.js(copy/paste it) and test how it works, it's a TableView but works as a picker.

    Ti.UI.backgroundColor = 'white';
var win = Ti.UI.createWindow();
var tableData = [ {title: 'Apples',backgroundColor :'#fff'}, {title: 'Bananas',backgroundColor :'#fff'}, {title: 'Carrots',backgroundColor :'#fff'}, {title: 'Potatoes',backgroundColor :'#fff'} ];

var table = Ti.UI.createTableView({
  data: tableData,
  visible : false,
  width : '220dp',
  height : 'auto',
  top : '50dp',
  zIndex : 1
});
win.add(table);
var button = Titanium.UI.createButton({
   title: 'Select',
   top: '100dp',
   width: '100dp',
   height: '50dp',
   isVisible : false
});
win.add(button);
button.addEventListener('click',function(e)
{
   if(e.source.isVisible == false){
        table.visible = true;
        e.source.isVisible = true;
   }
   else{
        table.visible = false;
        e.source.isVisible = false;
   }
   Titanium.API.info("You clicked the button");
});

table.addEventListener('click',function(e)
{
   if(e.rowData.title){
       button.title = e.rowData.title;
       table.visible = false;
       button.isVisible = false;
   }
});

win.open();
