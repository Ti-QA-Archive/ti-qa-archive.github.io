  • Hi ryan tola

    Thanks for your concern. I create custom row using view and add that view in scrollView . You can try this code

    ###Test Code:

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});
function createRow(i) {
    var row = Ti.UI.createView({
        backgroundColor : 'white',
        borderColor : '#bbb',
        borderWidth : 1,
        width : '100%',
        height : 30,
        top : 0,
        left : 0
    });

    var aLabel = Ti.UI.createLabel({
        text : 'test',
        color : 'black',
        height : 'auto',
        width : 'auto',
        textAlign : 'center'
    });
    row.add(aLabel);
    return row;
}

var scrollView = Ti.UI.createScrollView({
    bottom : 10,
    contentHeight : 'auto',
    layout : 'vertical'
});

for (var i = 0; i <= 20; i++) {
    var row = createRow(i);
    row.index = i;
    scrollView.add(row);
}
win.add(scrollView);

scrollView.addEventListener('scroll', function(e) {
    //alert(e.source.index);
    Ti.API.info('e    '+ JSON.stringify(e));
     var r = e.source;
     if (r.height == 30) {
     r.height = 60;
     } else {
     r.height = 30;
     }

});

win.open();

    ###Test Environment:
    Ti SDK 3.4.0.GA
    Ti Studio 3.4.1
    Ti CLI 3.4.1
    Mac OS X 10.9.5
    Android 4.1.2 & 4.4.2
