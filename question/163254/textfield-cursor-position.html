  • Hi
    Try this code

    var win=Ti.UI.createWindow({
    layout:'vertical',
    backgroundColor:'white'
});

var maxLen=8;

// Create a TextField.
var txtField = Ti.UI.createTextField({
    height : 35,
    top : 10,
    left : 40,
    width : 240,
    hintText : 'This is hint text',

});

// Listen for return events.
txtField.addEventListener('return', function(e) {
    if(txtField.value.length>maxLen){
        var result='';
        for( i=0;i<8;i++){
            result=result+txtField.value[i];
        }
    }
    txtField.value=result;

});

// Add to the parent view.
win.add(txtField);

win.open();

  • Hi,

    You can add change event on the text filed, this will automatically drop digits according to your requirement

    For example:

    
txtbox.addEventListener('change', function(e){

    if (e.value.length > 6) {
        var txtboxValue = e.value;
        txtboxValue = txtboxValue.substring(0, 6);
        txtbox.value = txtboxValue;
    }
});
