  • Hi Javier,

    Try this code.

    var navView = Ti.UI.createView({
    height : 38,
    width : 320,
    top : 0,
    left : 0,
    backgroundImage : 'NavBar.png',
    backgroundColor:'transparent'
});
win.add(navView);

var navLabel = Ti.UI.createLabel({
    height : 30,
    width : 100,
    left : 135,
    top : 5,
    backgroundColor : 'transparent',
    text : "My Title"
});
navView.add(navLabel);
