  • My code:

    .....
    var windows = Ti.UI.currentWindow;
    windows.layout = 'vertical';
.....
    var scrollView = Titanium.UI.createScrollView({
        contentWidth : 'auto',
        contentHeight : 'auto',
        top : 0
    });

    var w = Ti.UI.createView({
        backgroundColor : '#336699',
        borderRadius : 2,
        width : 'auto',
        height : 'auto',
        layout : 'vertical'
    });

    scrollView.add(w);
    windows.add(scrollView);

.....
    var picker = Ti.UI.createPicker({
        useSpinner : true,
        visibleItems : 5,
        type : Ti.UI.PICKER_TYPE_PLAIN,
        columns : [column1, column2]
    });

    var pickerValue = ' ';
    picker.addEventListener('change', function(e) {
        showStatus(e.selectedValue[0] + " " + e.selectedValue[1]);
        pickerValue = picker.getSelectedRow(0).title + " " + picker.getSelectedRow(1).title;

    });

    w.add(picker);

    If i insert picker in normal windows it's work.
