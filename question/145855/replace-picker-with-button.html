  • I've done something similar using textField as a control which will open date/time picker on click.

    pickerField.addEventListener("click", function(e) {

                var picker = Ti.UI.createPicker();

                    picker.type = Ti.UI.PICKER_TYPE_DATE;
                    picker.showDatePickerDialog({
                        callback : function(ee) {
                            if (ee.cancel) {
                                Ti.API.info('user canceled dialog');
                            } else {
                                var date = new Date(ee.value);
                                e.source.value = String.formatDate(date);
                            }
                        }
                    });
                });

    pickerField is textField
