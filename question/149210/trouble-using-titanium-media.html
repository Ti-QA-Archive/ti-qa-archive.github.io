    Hi Erik Daniel ,

    If you like to take a photograph in titanium then please check below code :

    This below code is only for taking picture and gave the native path of the captured photo

    Titanium.Media.showCamera({
    success : function(event) {
        var image = event.media;
        var filename = Titanium.Filesystem.applicationDataDirectory + "/" + 'camera_photo' + new Date().getTime() + ".png";
        var f = Titanium.Filesystem.getFile(filename);
        if (f.exists()) {
            Ti.API.info('The file exist , trying to delete it before using it :' + f.deleteFile());
            f = Titanium.Filesystem.getFile(filename);
        }
        f.write(image);
        Ti.API.info("native Patah : " + f.nativePath);
    },
    cancel : function() {
    },
    error : function(error) {
    },
    allowEditing : true,
    autohide : false,
    showControls : true,
    mediaTypes : Ti.Media.MEDIA_TYPE_PHOTO,
});

    and if you upload this image to server then please use this link :
    http://developer.appcelerator.com/question/88371/image-upload-directly-from-camera
