<TableView id="subModelTableAtoZ" dataCollection = "vehicleSubModel">
                <TableViewRow id="subModelTableRow" className="aToZTable" rowId="{id}" modelID="{model_id}" subModelCategoryID="{submodel_category}" subModelName="{submodel_name}" subModelThumbnail="{submodel_thumbnail}" >
                    <View id="leftImageContainer">
                        <ImageView id="thumbnail" ></ImageView>
                    </View>
                    <View class="verticalBorder" left="19.8%"></View>
                    <View id="centerView">
                        <Label id="subModelName" text="{submodel_name}"></Label>
                    </View>
                    <ImageView id="rightArrow" image="/images/rightArrow.png"></ImageView>
                </TableViewRow>
            </TableView>

and .js file


//----------------------------------*** For Each Table Row ***--------------------------------------
    for(var i=0; i<$.subModelTableAtoZ.data[0].rows.length; i++){
