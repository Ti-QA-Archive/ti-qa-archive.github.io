    Create a view .. set the layout to 'vertical'.
    Add each textfield to the parent view while pushing those fields to an array that keeps track of each. When you click a button, loop through that array and set the value of each of those fields to ""

  • or without the need for an extra array;

    var win = Ti.UI.createWindow({
    layout : 'vertical'
});

for (var x = 0; x < 8; x++) {
    win.add(Ti.UI.createTextField({
        top : 5,
        left : 5,
        width : 100,
        height : 40,
        backgroundColor : 'white'
    }));
}

var btn = Ti.UI.createButton({
    title : 'Click me'
});

btn.addEventListener('click', function(e) {

    for (var x = 0; x < win.children.length; x++) {
        if(win.children[x].toString().toLowerCase().indexOf('textfield') != -1) {
            win.children[x].value='';
        }

    }
});

win.add(btn);
win.open();
