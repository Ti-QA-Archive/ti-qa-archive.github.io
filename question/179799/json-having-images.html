    Hi Chankya,

    You can create using scrollview..

    i don't have complete code for that but i give some hint how to create it

    e.g

    win.xml

    <Alloy>
    <Window id="winIndex" class="container">
        <ScrollView id="scrollView"></ScrollView>
    </Window>
</Alloy>

    win.tss

    "#scrollView":{
    width : "100%",
    height : "100%",
    contentWidth : "100%",
    layout : "horizontal"
}

    win.js

    for(var i=0;i<10;i++){
    var tView = Ti.UI.createView({
        top : 10,
        left : "2%",
        width : "30%",
        height : 100,
        backgroundColor : "red"
    });

    var imgView = Ti.UI.createImageView({
        width : "100%",
        height : "80%",
        backgroundColor : "green",
        top : 0
    });
    tView.add(imgView);

    var lbl = Ti.UI.createLabel({
        width : "100%",
        height : "20%",
        bottom : 0,
        backgroundColor : "yellow"
    });
    tView.add(lbl);
    $.scrollView.add(tView);
}

    hop this is helps you :)
    Thanks
