  • Hi Cristian,

    You can use below code for custom checkboxes implement. It will support multiple items selection, you can customize it.

    var row = Titanium.UI.createTableViewRow();

var tickBox = Ti.UI.createButton({
     backgroundImage:'../images/tick_unchecked.png',
     width:23,
     height:25,
     checkBok:"tick",
     customValue:"use your custom value",
     checked:false,
     left:10,
     top:20
});
row.add(tickBox);

tableView.addEventListener('click',function(e){
    if(e.source.checkBok){
        if(e.source.checked){
            e.source.backgroundImage = '../images/tick_unchecked.png';
            e.source.checked = false;
        }else{
            e.source.backgroundImage = '../images/tick_checked.png';
            e.source.checked = true;
            alert(e.source.customValue);
        }
    }
});
