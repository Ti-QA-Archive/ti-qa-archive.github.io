Call a funtion in 2nd controller within from 1st controller.

Hi.

I've 2 contollers. How to call a function in second controller from first controller?
My code is like
controller1

function openChats (e) {

    model_id = e.index;

    $.chatwin.open();

    threadID = e.rowData.accessibilityLabel;
    threadName =  e.rowData.title;

    sendreqfunction();
    // sendreqfunction(); does displaying of messages.
}

Now i want to call this function from 2nd controller.

controller2

var window =Alloy.createController('controller1', args);
var subView = window.getView();
subView. openChats();

    HI you can also do it like this …

    Controller 1

    this.openChats = function (e) {

    model_id = e.index;

    $.chatwin.open();

    threadID = e.rowData.accessibilityLabel;
    threadName =  e.rowData.title;

    sendreqfunction();
    // sendreqfunction(); does displaying of messages.
};

    Controller 2

    var window =Alloy.createController('controller1', args);

window.openChats();
