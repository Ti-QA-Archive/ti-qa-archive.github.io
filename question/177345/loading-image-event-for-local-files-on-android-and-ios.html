Loading image event for local files on Android and IOS

Hi,

Seems that "load" event only works with remote images. I'm trying to load local images into a view using customView on annotation map and sometimes the image is blank. I tried to use "load " event, but it's not firing.

Example:

    var map = require('ti.map');

    for (var i=0;i < 30;i++) {

    var myImage = Ti.UI.createImageView({
        width : 35,
        height : 35,
        image : '/icons/test.png',
        bottom : 0
    });

    var thisView = Ti.UI.createView({
        width : 100,
        height : 55
    });

    var labelView = Ti.UI.createView({
        width : Ti.UI.SIZE,
        height : 20,
        top : 0,
        backgroundColor : 'black',
    });

    var labelText = Ti.UI.createLabel({
        color : 'white',
        text : 'test' + i,
        top : 0,
        right : 5,
        bottom : 5,
        left : 5,
        width : Ti.UI.SIZE,
        height : 20
    });

    labelView.add(labelText);
    thisView.add(labelView);
    thisView.add(myImage);

    var ant = map.createAnnotation({
        title : ' ',
        latitude : {some latidude},
        longitude : {some longitude},
        customView : thisView,
    });

    mapview.addAnnotation(ant);
}

Any ideias?

Thanks.
