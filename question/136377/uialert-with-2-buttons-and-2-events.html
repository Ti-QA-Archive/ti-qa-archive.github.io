UIalert with 2 buttons and 2 events

Hi, in my iPad app i need to show an alert with 2 buttons and associate these 2 buttons with 2 different events, how can i do?

    AlertDialog would help:

    var dialog = Ti.UI.createAlertDialog({
    cancel : 2,
    buttonNames : ['Confirm', 'Help', 'Cancel'],
    message : 'Would you like to delete the file?',
    title : 'Delete'
})

dialog.show();

dialog.addEventListener('click', function(e) {
    Ti.API.info( JSON.stringify(e) );

    switch (e.index) {
        case 0: // Confirm
            alert('Comfirm clicked');
            break;

        case 1: // Help
            alert('Help clicked');
            break;    

        case 2: // Cancel
            alert('Cancel clicked');
            break;
    }

});
