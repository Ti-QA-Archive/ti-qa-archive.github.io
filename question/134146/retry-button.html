    hi Gurkhan,

    create a button and add it in your window then hide it by button.hide();
    like..

    
var button = Ti.UI.createButton({
    title :'Reload'
});

button.addEventListener('click',function(e){
    callWebService();
});

button.hide();

win.add(button);

callWebService(){

 }

    try{
        json_data = eval('('+this.responseText+')');

        alert('Check network connection!');
    button.show();
    }
}
}
