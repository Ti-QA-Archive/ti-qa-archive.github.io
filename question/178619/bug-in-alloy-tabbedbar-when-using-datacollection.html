  var transform6 = Alloy.createModel('preferenceDisplay');

  transform6.set("rowHeight","85dp");
  transform6.set("controlVisible",true);
  transform6.set("categoryLabelVisible",false);
  transform6.set("preferenceLabelVisible",true);

  // we build up our newPreferences collection and once finished assign the models in this to our view collection ($.preferences) 
  // otherwise, the multiple additions each kick off a collection event on $.preferences, which in turn causes the GUI to refresh

  // screen draws just the once
  $.preferences.reset(newPreferences.models);
}
