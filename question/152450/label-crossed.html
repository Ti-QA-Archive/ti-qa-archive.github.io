  • Hello Djmahel,

    Try this Code : 

    var win = Ti.UI.createWindow({
 backgroundColor : 'white',
 height : "100%",
 width : "100%"
});

var view = Ti.UI.createView({
 top : "20dp",
 height : "50dp",
 width : "100%"
});
win.add(view);

var label = Ti.UI.createLabel({
 top : "5dp",
 text : "jigar m maheshwari",
 font : {
  fontSize : "17dp",
  fontWeight : "bold"
 }
});
view.add(label);

var cropline = Ti.UI.createView({
 backgroundColor : 'blue',
 top : '18dp',
 height : "1dp",
 width : "320dp",
 zIndex : '10dp'
});
view.add(cropline);

win.open();

    Let me know if it works.

    Jigar
