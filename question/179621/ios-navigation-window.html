    Hi

    You have a number of issues here - the first and most important one - that all of your file and variables are too simplicity named. Whilst I imagine this is because you wanted some simplicity for speed - it confuses things, both the reader (me) and you after a while.

    The problem with the code is a simple one - you open windows in a navigation window by using the navigation window to open windows. Your example for subsequent windows - simply calls the open method of the new window - and has no reference to the navigation window - which it requires. Finally the last window explicitly states it does not want the navigation bar - this would remove the back button option even if the code was referenced correctly.

    The code example provided by Kosso is good - however it is designed for a single file containing all the variables. Your code is broken up into different modules, I prefer this module (although Alloy is much simpler), so the code is not as helpful.

    {appname}/resources/app.js

    var mainWindow = require('mainWin').createNavigationWindow({
var nav = require('nav').createNavigationWindow({
    window: mainWindow 
});
nav.open();

    {appname}/resources/nav.js

    var nav;
var createNavigationWindow = function (obj) {
    nav = Ti.UI.iOS.createNavigationWindow({
        window: obj.window
    });
    return nav;
};
var getNav = function () { return nav; };
exports.createNavigationWindow = createNavigationWindow;
exports.getNav = getNav;

    {appname}/resources/mainWin.js

    var createWindow = function () {
    var win = Ti.UI.createWindow({
        backgroundColor: 'red',
        title: 'Main Window'
    });
    var btnOpenWindow = Ti.UI.createButton({
        title: 'Open Window'
    });
    btnOpenWindow.addEventListener('click', function (e) {
        var childWin = require('childWin').createWindow();
        var nav = require('nav').getNav();
        nav.openWindow(childWin, { animated: true });

        // note the use of the referenced 'nav' require
    });
    win.add(btnOpenWindow);
    return win;
}
exports.createWindow = createWindow;

    {appname}/resources/childWin.js

    var createWindow = function () {
    var win = Ti.UI.createWindow({
        backgroundColor: 'green',
        title: 'Child Window'
    });
    var btnOpenWindow = Ti.UI.createButton({
        title: 'Open Another Window'
    });
    btnOpenWindow.addEventListener('click', function (e) {
        /*
        var childWin = require('childWin').createWindow();
        var nav = require('nav').getNav();
        nav.openWindow(childWin, { animated: true });
        */

        // repeat the process comment section again
        // but with a new 'childWin.js'
        // for each child
    });
    win.add(btnOpenWindow);
    return win;
}
exports.createWindow = createWindow;

    • I use a commonJS module to open all windows.
      For iOS I create the window I need, then use createWindow on the root navigationWindow.

      For Android I simply create the new window then use new_win.open({animated:true, theme: &#39;Theme.NoActionBar&#39;});

      btw: The theme I use refers an xml file called mytheme.xml in platform/android/res/value/ containing:

      <?xml version="1.0" encoding="utf-8"?>
<resources>
    <style name="Theme.NoActionBar" parent="@style/Theme.AppCompat">
        <item name="windowActionBar">false</item>
        <item name="android:windowNoTitle">true</item>
    </style>
</resources>

      That because I use fullscreen windows throughout, on iOS and Android. (in the app I'm using the examples from)
