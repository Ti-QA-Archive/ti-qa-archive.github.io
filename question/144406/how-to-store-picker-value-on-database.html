  • var pickerdate = e.value;
    var day = pickerdate.getDate();
    var month = 1 + pickerdate.getMonth();
    var year = pickerdate.getFullYear();
    newdate = day + "-" + month + "-" + year;

    then in your data base make a field with name newdate in text formate and store it.
