  • have a look at my sample code:

    var win = Ti.UI.createWindow({
    fullscreen : 'true',
    backgroundColor : 'white'
});

var table = Ti.UI.createTableView({
    height : Ti.UI.SIZE,
    width : '100%',
    backgroundColor : 'yellow',
    top : '42dp'
});

var ratingAttributes = ["Trust", "Respectfulness", "Fairness", "Caring", "Citizenship"];
var attributeColors = ["#F23D3D", "#0ED718", "#42B4E6", "#D9E610", "#FB9A32"];
var data = [];

var slideOutAn = Ti.UI.createAnimation({
    left : '50%',
    duration : 1000
});
var slideInAn = Ti.UI.createAnimation({
    left : '98%',
    duration : 1000
});
var row;
var label;

for ( i = 0; i < ratingAttributes.length; i++) {
    row = Ti.UI.createTableViewRow({
        backgroundColor : 'white'
    });
    label = Ti.UI.createLabel({
        text : ratingAttributes[i],
        color : 'black',
        left : '10dp',
        height : '50dp',
        width : 'auto'
    });
    row.add(label);
    var view = Ti.UI.createView({
        width : '50%',
        backgroundColor : 'red',
        left : '98%',
        height : '40dp',
        type : 'slider',
        layout : 'horizontal'
    });
    var chotaview = Ti.UI.createView({
        left : '0dp',
        width : '4%',
        height : '40dp',
        type : 'slider',
        backgroundColor : attributeColors[i]
    });
    view.add(chotaview);
    var badaview = Ti.UI.createView({
        // left:'4%',
        height : '40dp',
        backgroundColor : 'black',
        type : 'slider',
        width : '96%',
        layout:'horizontal'

    });



    view.add(badaview);
    row.add(view);
    data.push(row);
}
table.data = data;
table.addEventListener('click', function(e) {
    if (e.source.type !== 'slider') {
        Ti.API.info('not slider');
        Ti.API.info('children left '+e.row.children[1].left);
        if (e.row.children[1].left === '98%') {
            Ti.API.info('98% tha');
            e.row.children[1].animate(slideOutAn, function() {
                e.row.children[1].left = '50%';
            });
        } else {
            Ti.API.info('50% tha');
            e.row.children[1].animate(slideInAn, function() {
                e.row.children[1].left = '98%';
            });
        }
    } else {
        alert(e.source + '');
    }

});

win.add(table);
win.open();
