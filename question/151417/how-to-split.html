    var hintArr;
hintArr = '{"results" : [{"pair" : "3, 6"}, {"pair" :"5, 7"}, {"pair" :"2, 8"}]}';
obj = JSON.parse(hintArr);
var ar = [];
for (var r = 0; r < obj.results.length; ++r) {
    Ti.API.info(obj.results[r].pair);
    var s = obj.results[r].pair.split(',');
    s.forEach(function(el) {
        ar.push(el.trim());
    });
}

Ti.API.info('Array: '+ ar);
Ti.API.info(ar.join(','));
