  • Don't know exactly what you want but i think you are looking for

    var win = Ti.UI.createWindow({
    backgroundColor : 'white',
});

var matrix1 = Ti.UI.create2DMatrix({
    rotate : 180 //angle for rotate
});

var animte1 = Ti.UI.createAnimation({
    transform : matrix1,
    duration : 100,
    anchorPoint : {
        x : 0.5,
        y : 0.5
    }
});

// Create an ImageView.
var anImageView = Ti.UI.createImageView({
    image : 'location',
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
});
anImageView.addEventListener('swipe', function() {

    anImageView.animate(animte1);

});
// Add to the parent view.
win.add(anImageView);

win.open();
