Is there any way to get X,Y coordinates of the scrollview in Android (without eventListener) ?

  • I know you asked for a solution that did not involve handling events, but here's an example that does the job. I'll continue investigating to see if there is a way to do it without the listener.

    var win = Ti.UI.createWindow({
    backgroundColor:'#ffffff'
});
var scroll = Ti.UI.createScrollView({
    contentWidth:'auto',
    contentHeight:'auto',
    top:0,
    showVerticalScrollIndicator:true,
    showHorizontalScrollIndicator:true,
    layout:'vertical'
});
scroll.addEventListener('scroll', function(e) {
    // here's the magic
    Ti.API.info(e.x + ',' + e.y);
});
for (var i = 0; i < 30; i++) {
    scroll.add(Ti.UI.createView({
        backgroundGradient: {
            type:'linear',
            colors:['#000000','#0000ff'],
            startPoint:{
                x:0,
                y:0
            },
            endPoint:{
                x:'100%',
                y:0
            }
        },
        width:1000,
        height:60,
        top:8    
    }));    
}
win.add(scroll);

win.open();
