    • 
var window = Titanium.UI.createWindow({
    backgroundColor : 'white',
});
var textField = Ti.UI.createTextField({
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    color : "#336699",
    top : 50,
    left : 10,
    width : 150,
    hintText : 'Name',
    height : 25
});
var button = Ti.UI.createButton({
    title : 'Show name',
    top : 70, //right:10,
    width : 200,
    height : 100
});
var label = Ti.UI.createLabel({
    color : '#999',
    text : "Your name is ",
    font : {
        fontSize : 20,
        fontFamily : 'Times New Roman'
    },
    textAlign : Ti.UI.TEXT_ALIGNMENT_CENTER,
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE
});
window.add(label);
window.add(button);
window.add(textField);
window.open();
button.addEventListener('click', function(e) {
    if (textField.value == '') {
        alert('Please, type your name!');
    } else {
        alert('Hi, so you are ' + textField.value + '!?');
        Ti.API.info("My name is " + textField.value);
        label.text = "Your name is " + textField.value;
    }
});
