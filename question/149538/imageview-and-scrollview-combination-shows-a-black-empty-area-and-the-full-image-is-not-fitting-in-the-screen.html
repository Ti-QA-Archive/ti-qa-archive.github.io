    hi,

    first thing is you should set scrollview's height and width only. and if image_path is a remote url then it does not show image with backgroundImage property.

    var image_view = Titanium.UI.createImageView({
        top:40,
        image : image_path,
        backgroundColor : 'black',
        height:Ti.UI.SIZE,
        width:Ti.UI.SIZE,
        canScale: true,
});

var image_scroll_view = Titanium.UI.createScrollView({
        top:40,
        backgroundImage : image_path,
        height : Titanium.Platform.displayCaps.platformHeight - 40,
        width : Titanium.Platform.displayCaps.platformWidth,
        contentHeight : Ti.UI.SIZE,
    contentWidth : Ti.UI.SIZE,
        maxZoomScale : 10,
        minZoomScale : 1.0,
        backgroundColor : "black",
    });
image_scroll_view.add(image_view);

    • thanks Mitul, this code is working for me

      var image_view = Titanium.UI.createImageView({
        image : image_path,
        backgroundColor : 'black',
        canScale : true,
    });

    var height  = Titanium.Platform.displayCaps.platformHeight - 44;
    var width = Titanium.Platform.displayCaps.platformWidth;

    var image_scroll_view = Titanium.UI.createScrollView({
        top : 44,
        height : height,
        width : width,
        contentHeight : height,
        contentWidth : width,
        maxZoomScale : 10,
        minZoomScale : 1.0,
        backgroundColor : "black",
    });

    image_scroll_view.add(image_view);

    self.add(image_scroll_view);
