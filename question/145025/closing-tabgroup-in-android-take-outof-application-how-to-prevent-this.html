closing tabgroup in android take outof application how to prevent this

In Android from app.js there is button and i open tabgroup on that button click while closing that Tabgroup it close application not take me back to app.js
how to prevent this closing of App in android

//App.js

var win = Ti.UI.createWindow({
backgroundColor : 'teal',
});

var btn = Ti.UI.createButton({
title : 'click me',
left : 10,
top : 10,
height : 30,
width : 70,
color : 'black'
});
win.add(btn);

btn.addEventListener('click', function(r) {

var customTabGroup = Titanium.UI.createTabGroup();
Ti.App.customTabGroup = customTabGroup;

var winnew = Ti.UI.createWindow({
    url : '/ui/winnew.js',
    navBarHidden : true,
    modal : true

});
var tab1 = Titanium.UI.createTab({
    window : winnew,
    title : 'winnew',
    icon : 'KS_nav_ui.png'
});
customTabGroup.addTab(tab1);
customTabGroup.open();

});

win.open();

//winnew.js

var win = Ti.UI.currentWindow;

win.backgroundColor = 'white';

var barImage = Ti.UI.createView({
backgroundColor : 'black',
height : 43,
width : 320,
top : 0
});

var infoLabel = Ti.UI.createLabel({
text : 'winnew',
height : '30dp',
width : '190dp',
color : 'white',
textAlign : 'center',
font : {
fontSize : 20,
fontWeight : 'bold'
}
});
barImage.add(infoLabel);

var backBtnProgmmaPlatte = Ti.UI.createButton({
height : 30,
width : 55,
left : 10,
title : 'back',
color : 'white',

});

backBtnProgmmaPlatte.addEventListener('click', function(e) {

    Ti.App.customTabGroup.close();

});

barImage.add(backBtnProgmmaPlatte);
win.add(barImage);
