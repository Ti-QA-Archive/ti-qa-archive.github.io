  • you can override default navigation button by your own

    var myBtn = Ti.UI.createButton({
    title: "edit",
    backgroundImage: "edit.png" //just change this property
});

win.setRightNavButton(myBtn);

    you cannot have 2 left navigation buttons at a time

    Hello Cristian Alexandru,

    You cant add two button in navigation bar but you can add ButtonBar. Please check my sample code. I hope it help you.

    var win2 = Titanium.UI.createWindow({
    backgroundColor: 'red',
    title: 'Red Window'
});

var win1 = Titanium.UI.iOS.createNavigationWindow({
   window: win2
});


 var bb1 = Titanium.UI.createButtonBar({
    labels:['One', 'Two', 'Three'],
    backgroundColor:'#336699',
    top:50,
    style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
    height:25,
    width:Ti.UI.SIZE
});
win2.setRightNavButton(bb1);

 var bb2 = Ti.UI.createButton({ 
       height:25,
         width:25,
    backgroundImage: "icon.png" //just change this property
});


win2.setLeftNavButton(bb2);


win1.open();
