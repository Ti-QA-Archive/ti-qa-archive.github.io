ScrollView not scrolling on iPhone but works on Android

The following code displays properly on the Android Emulator and iPhone Simulator. The user can scroll on the Android but not the iPhone… What am I missing?

I am using SDK 1.8.0.v20110906212948

Ti.UI.setBackgroundColor('#000');

var win1 = Titanium.UI.createWindow({title:'Window 1'});

var TheScrollView=Ti.UI.createScrollView({contentWidth:'auto',
                                          contentHeight:'auto',
                                          height:'auto', 
                                          backgroundColor:'black',
                                          showVerticalScrollIndicator:true,
                                          showHorizontalScrollIndicator:false,
                                          scrollType:'vertical',
                                          layout:'vertical',
                                          touchEnabled:true});
win1.add(TheScrollView);

function CreateNewRow(_Caption)
{
  var TheCell=Ti.UI.createView({height:'auto',
                                width:'auto',
                                touchEnabled:true});
  var TheRow=Ti.UI.createView({height:'225dp',
                               touchEnabled:true});
  var TheLabel=Ti.UI.createLabel({text:_Caption,
                                  color:'white', 
                                  font:{fontSize:'18dp',fontFamily:'Helvetica Neue'},
                                  textAlign:'left',
                                  width:'auto',
                                  height:'auto',
                                  touchEnabled:true});
  TheScrollView.add(TheRow);
  TheRow.add(TheCell); 
  TheCell.add(TheLabel);
}; 

CreateNewRow('Row 1');
CreateNewRow('Row 2');
CreateNewRow('Row 3');

win1.open();

    In case of iPhone you should provide height of scrollview to specific number not "auto"

    so it should be like

    height:480

    Try this .. It will work.
