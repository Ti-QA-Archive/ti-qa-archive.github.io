  • same better formatted:

    var module = require("mod");
var tableL = [];
var db = Titanium.Database.install('MyDB.db', 'MyDB');
module.loadDB(tableL,db);
var listbox = Titanium.UI.createTableView({
    data: tableL,
    backgroundColor: '#EEEEFF', 
    top: 50,
    left: 0,
    height: 240,
    font: {fontSize: 20},
    separatorColor: '#333',
});

    exports.loadDB = function (tableL,db,rID) {
    var db = Titanium.Database.install('MyDB.db', 'MyDB');
    var rows = db.execute('SELECT * FROM MyDB');
    var section1 = Ti.UI.createTableViewSection({
        headerTitle: 'title'
    });
    while (rows.isValidRow()){
        var row1 = Ti.UI.createTableViewRow({
            title: ''+rows.field(0),
            font:{fontSize:21,fontWeight:'bold', fontFamily:'Arial'},
        });
        section1.add(row1);
        rows.next();
    }
    tableL.push(section1);
};
