  • I'm replacing my window title with a progressBar. Here's some code, shadow isn't supported yet, as far as I know, but I've kept the options in case it ever is:

    var progressBar = Ti.UI.createProgressBar({min: 0, max: 1, value: 0, message: "Fetching...", width: 200, color: "#fff", shadowColor: "#64696e", shadowOffset: {x:0, y:-1}, font: {fontSize: 14, fontWeight: "bold"}});
if (iPad) {
    progressBar.color = "#64696e";
}
progressBar.style = Ti.UI.iPhone.ProgressBarStyle.BAR;
progressBar.show();

window.setTitleControl(progressBar);

xhr.ondatastream = function(event) {
    if (event.progress > 0) {
        progressBar.message = "Fetching...";
        progressBar.value = event.progress;
    }
};
