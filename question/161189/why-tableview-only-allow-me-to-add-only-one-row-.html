function CreateMyRow(fid,name,lat,lon, min)
{
    var id=Ti.UI.createLabel({
        text:fid,
        height:hi,
        width:hi
    });
    var names=Ti.UI.createLabel({
        text:name,
        height:hi,
        width:hi
    });

    var lats=Ti.UI.createLabel({
        text:lat,
        height:hi,
        width:hi
    });

    var lons=Ti.UI.createLabel({
        text:lons,
        height:hi,
        width:hi
    });

    var mins=Ti.UI.createLabel({
        text:min,
        height:hi,
        width:hi
    });
    var row=Ti.UI.createTableViewRow({layout:'horizontal'});
    row.add(id);
    row.add(names);
    row.add(lats);
    row.add(lons);
    row.add(mins);

    return row;
}
