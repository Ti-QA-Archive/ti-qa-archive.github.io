    Hi

    Unfortunately not, nothing built-in to achieve that.

    If you want to have the images maintain their original size without stretching them - something more like this;

    var pWidth = Ti.Platform.displayCaps.platformWidth;
var toolBarView = Titanium.UI.createView({
    bottom: 0,
    height: 50,
    width: pWidth
});
var viewLeft = Ti.UI.createView({
    left: 0,
    width: parseInt(pWidth / 3, 10)
});
toolBarView.add(viewLeft);
var viewMiddle = Ti.UI.createView({
    width: parseInt(pWidth / 3, 10)
});
toolBarView.add(viewMiddle);
var viewRight = Ti.UI.createView({
    right: 0,
    width: parseInt(pWidth / 3, 10)
});
toolBarView.add(viewRight);
var buttonFacebook = Ti.UI.createImageView({
    image: '/HTML/images/facebook.png',
    top: 5
});
var buttonTwitter = Ti.UI.createImageView({
    image: '/HTML/images/twitter.png',
    top: 5
});
var buttonPintrest = Ti.UI.createImageView({
    image: '/HTML/images/pinterest.png',
    top: 5
});
viewLeft.add(buttonFacebook);
viewMiddle.add(buttonTwitter);
viewRight.add(buttonPintrest);

    This works by creating 3 internal views and making each 1/3 of the width of the screen. The left one OS anchored to the left, the right anchored to the right and the middle one defaults to the centre. Each image is added into the correct; left, middle or bottom.

    • Assuming you will have each image react to a touch event this would be a more efficient way to achieve that without having to have an event listeners for each button.

      var pWidth = Ti.Platform.displayCaps.platformWidth;
var toolBarView = Titanium.UI.createView({
    bottom: 0,
    height: 50,
    width: pWidth
});
var viewLeft = Ti.UI.createView({
    left: 0,
    width: parseInt(pWidth / 3, 10)
});
toolBarView.add(viewLeft);
var viewMiddle = Ti.UI.createView({
    width: parseInt(pWidth / 3, 10)
});
toolBarView.add(viewMiddle);
var viewRight = Ti.UI.createView({
    right: 0,
    width: parseInt(pWidth / 3, 10)
});
toolBarView.add(viewRight);
var buttonFacebook = Ti.UI.createImageView({
    action: 'facebook',
    image: '/HTML/images/facebook.png',
    top: 5
});
var buttonTwitter = Ti.UI.createImageView({
    action: 'twitter',
    image: '/HTML/images/twitter.png',
    top: 5
});
var buttonPintrest = Ti.UI.createImageView({
    action: 'pintrest',
    image: '/HTML/images/pinterest.png',
    top: 5
});
viewLeft.add(buttonFacebook);
viewMiddle.add(buttonTwitter);
viewRight.add(buttonPintrest);
var handler = function (e) {
    select case (e.action) {
    case 'facebook'
        // do something
        break;
    case 'twitter'
        // do something
        break;
    case 'pintrest'
        // do something
        break;
    }
};
toolBarView.addEventListener('click', handler);

      This works as all event bubble up unless you tell them not to. So the click event on each image is passed to the location view and then the toolbar view. This then responds to events at that level.

    • I had removed the layout property - which makes sense for 3 items but more do need it - so good idea to bring it back.

      To divide the width of something into equal chunks do this;

      var pWidth = Ti.Platform.displayCaps.platformWidth;
chunkWidth = parseInt(pWidth / 3, 10); // 3 equal
chunkWidth = parseInt(pWidth / 4, 10); // 4 equal
chunkWidth = parseInt(pWidth / 5, 10); // 5 equal

      parseInt makes sure the returns number is an integer - which is recommended when layout things out.

      So your code would be;

      var pWidth = Ti.Platform.displayCaps.platformWidth;
var imageViewPaddingFacebook = Ti.UI.createView({ width: parseInt(pWidth / 4, 10) });
var imageViewPaddingTwitter = Ti.UI.createView({ width: parseInt(pWidth / 4, 10) });
var imageViewPaddingPinterest = Ti.UI.createView({ width: parseInt(pWidth / 4, 10) });
var imageViewPaddingInstagram = Ti.UI.createView({ width: parseInt(pWidth / 4, 10) });

      Tips;

      • top:'5' should be top: 5, there is no need to turn a number into a string - the colour coding of true numbers also helps you to read your code
      • the only time you should use a string to represent numbers is if you have to suffix the number with; dp, px, % etc
      • Titanium can be shortened to Ti.
      • make your folders and filenames lowercase - trust me - there will be a time when you get a problem you cannot solve for ages - it will be a filename case issue.

      Also the single event listener is a much better plan - reducing the need for event listeners at anyone time will help to reduce memory leaks - if you do not know what these are now - they will be something you get interested in when your app scales.

    • FYI: here is my finished and tested code in case it helps anyone.

      
var pWidth = Ti.Platform.displayCaps.platformWidth;


var toolBarView = Titanium.UI.createView({
    bottom:0,
    height:'50',
    width: pWidth,
    layout:'horizontal',
    //backgroundColor:'#777'
});


var imageViewPaddingFacebook = Titanium.UI.createView({
    width: parseInt(pWidth / 4, 10)
});
var imageViewPaddingTwitter = Titanium.UI.createView({
    width: parseInt(pWidth / 4, 10)
});
var imageViewPaddingPinterest = Titanium.UI.createView({
    width: parseInt(pWidth / 4, 10)
});
var imageViewPaddingInstagram = Titanium.UI.createView({
    width: parseInt(pWidth / 4, 10)
});



var buttonFacebook = Ti.UI.createImageView({
    action:'facebook',
    image: '/images/showtime/facebook.png',
    top:5,
    width:40,
    height:40
});
var buttonTwitter = Ti.UI.createImageView({
    action:'twitter',
    image: '/images/showtime/twitter.png',
    top:5,
    width:40,
    height:40
});
var buttonPinterest= Ti.UI.createImageView({
    action:'pinterest',
    image: '/images/showtime/pinterest.png',
    top:5,
    width:40,
    height:40
});
var buttonInstagram= Ti.UI.createImageView({
    action:'instagram',
    image: '/images/showtime/instagram.png',
    top:5,
    width:40,
    height:40
});



var handlerSocialMedia = function (e) {
    switch(e.source.action) {
    case 'facebook':
        Ti.Platform.openURL('https://www.facebook.com/');
        break;
    case 'twitter':
        Ti.Platform.openURL('http://twitter.com/');       
        break;
    case 'pinterest':
        Ti.Platform.openURL('http://pinterest.com/');
        break;
    case 'instagram':
        Ti.Platform.openURL('http://www.instagram.com);
        break;
    }
};


buttonFacebook.addEventListener('click', handlerSocialMedia);
buttonTwitter.addEventListener('click', handlerSocialMedia);
buttonPinterest.addEventListener('click', handlerSocialMedia);
buttonInstagram.addEventListener('click', handlerSocialMedia);

imageViewPaddingFacebook.add(buttonFacebook);
imageViewPaddingTwitter.add(buttonTwitter);
imageViewPaddingPinterest.add(buttonPinterest);
imageViewPaddingInstagram.add(buttonInstagram);


toolBarView.add(imageViewPaddingFacebook);
toolBarView.add(imageViewPaddingTwitter);
toolBarView.add(imageViewPaddingPinterest);
toolBarView.add(imageViewPaddingInstagram);



self.add(toolBarView);
