    Hi Tim

    I have just created app.js added this code in it and changed the DB name which you gave me in mail.

    var win = Ti.UI.createWindow({
    backgroundColor: '#fff'
});

var favoriteButton = Titanium.UI.createButton({
    title:'Add to Favorites',
   // image:'../images/chat.png',
    width:200,
    height:40,
    top:60
});


favoriteButton.addEventListener('click', function()
{
    // Setup the variables
    var favorite = 1;
    var id = 1;
    var comments = 'Test Comments';

    //opens the database.
    var db = Ti.Database.open('ProductsDB');

    //execute query..
    var dataRS = db.execute("SELECT id FROM products WHERE id = " + id);
    if(dataRS.rowCount > 0) {
        // Update value in products
        db.execute('UPDATE products SET favorite = ? WHERE id = ?', favorite, id);    
        alert("Record Updated..");
    }
    else {
        // Update value in products
        db.execute('INSERT INTO products (comments,favorite) VALUES (?,?)', comments, 0);
        alert("Record Inserted..");
    }
    dataRS.close();
    db.close();
});

win.add(favoriteButton);

//Initialize Databse and create one if not exists...
initDatabase = function()
{
    //opens the database.
    var db = Ti.Database.open('ProductsDB');

    /* creates table to store bus related data. */
    db.execute('CREATE TABLE IF NOT EXISTS products (id INTEGER PRIMARY KEY, comments varchar(100), favorite INTEGER)');

    //closes database
    db.close();
}();

win.open();

    It works fine in android so hope this will also work for you now.

    Nikunj
