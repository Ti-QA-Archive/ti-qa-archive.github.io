touchmove limits

I have problem with a touchmove eventListener. I have an imageView that I want to scroll up and down. It containes other images inside, kind or like a gallery. But when it gets to the top I want it to stop and the same on the bottom.

How can I do that??

Here is the code:

// Create a parent window
//Change the layou to landscape 
Titanium.UI.orientation = Titanium.UI.LANDSCAPE_LEFT;

//Create a window 
var win1 = Ti.UI.createWindow({
    backgroundColor:'#ddd'
});

//Create a new view inside
var side_nav = Ti.UI.createImageView({
    top:0,
    right:0,
    width:120,
    height:600,
    backgroundColor:'#aaa'
});

//Create an array for the thumbnails
var thumbnails = [];

//Initial values for positioning the thumbnails
var initialPos = 20;
var spacing = 20;

//Loop for loading images 
for (i=0; i<8; i++){
    thumbnails[i] = Ti.UI.createImageView({
    top:initialPos,
    left:32,
    height:65,
    width:65,
    image:"images/thumb_" + i + ".png"

    });
    side_nav.add(thumbnails[i]);
    initialPos += 65 + spacing;
};


// Add view to the window
win1.add(side_nav);


//
//This is where things get messy 
//

side_nav.addEventListener("touchmove",function(e){

    side_nav.top += e.y/4;


});


win1.open();
