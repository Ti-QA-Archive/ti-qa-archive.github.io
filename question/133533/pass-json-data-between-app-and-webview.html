    var calView = Ti.UI.createWebView({url:'calendar3/default.html'});
    calendarViewWin.add(calView);        

    calView.addEventListener('load',function(e){      
        Ti.App.fireEvent('webPageReady', { 
            jeson: json
        });
    });

    default.html :

    $(document).ready(function() {

        Ti.App.addEventListener('webPageReady', function(e) {

            alert("TEST :"+e.jeson.length);

            var dataSum = [];

           for(var i=0; i<e.jeson.length; i++){
                var date = new Date(e.jeson[i].startTime);
                var d = date.getDate();
                var m = date.getMonth();
                var y = date.getFullYear();

                dataSum[i] = {title:e.jeson[i].name,start: new Date(y, m, d)};


            $('#calendar').fullCalendar({
                editable: false,
                events: dataSum
            });
        });
});
