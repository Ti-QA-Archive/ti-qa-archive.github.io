Problem when adding a label to button in Android

Hi,

I have created a function like this for creating buttons on a window :

function createButton(labelCaption, link) {
    var b = Titanium.UI.createButton({
        color:'#fff',
        backgroundImage:'../images/normal.png',
        backgroundSelectedImage:'../images/pushed.png',
        backgroundDisabledImage: '../images/disabled.png',
        top:vertSpace + posY * (btnHeight + vertSpace),
        left:horSpace + posX * (btnWidth + horSpace),
        width:64,
        height:64
    });

    var bl = Titanium.UI.createLabel({
        color:'blue',
        highlightedColor:'red',
        backgroundColor:'transparent',
        top:64,
        width:120,
        height:'auto',
        //right:5,
        textAlign:'center',
        font:{fontSize:14},
        text:labelCaption
    });

    b.add(bl);

    b.addEventListener('click', function()
     {
            var w = Ti.UI.createWindow({title:labelCaption, url:link});
            Titanium.UI.currentTab.open(w,{animated:true});
     });
win.add(b);
}

it work fine in iPhone simulator, but in Android Simulator it doesn't show the label.
Android 2.3 Titanium SDK 1.5.0, under osx

Any idea ?

here the label is added to the button like this : b.add(bl);

    You can't build compound widgets out of buttons. Use a View to contain both controls.
