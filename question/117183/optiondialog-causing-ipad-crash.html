optionDialog causing iPad Crash?

I am using Titanium Developer 1.3.0 and Mobile SDK 1.6.1.

My code is pretty basic:

            var cancel = Ti.UI.createButton({
                systemButton:Ti.UI.iPhone.SystemButton.STOP,
                style:Ti.UI.iPhone.SystemButtonStyle.BAR
            });
            var flexSpace = Ti.UI.createButton({
                systemButton:Ti.UI.iPhone.SystemButton.FLEXIBLE_SPACE
            });
            var send = Ti.UI.createButton({
                systemButton:Ti.UI.iPhone.SystemButton.ACTION,
                style:Ti.UI.iPhone.SystemButtonStyle.BAR
            });

            cancel.addEventListener('click', function() {
                itemImgPage.close();
            });

            send.addEventListener('click', function() {
                var optionDialog = Ti.UI.createOptionDialog({
                    options:['Open in Safari','Cancel'],
                    title:jsoncat[index].name+' Photo',

                optionDialog.addEventListener('click', function(e) {
                    if (e.index == 0) {
                        Ti.Platform.openURL('http://localhost/'+jsoncat[index].image);

                });

                optionDialog.show();

            var itemImg_Toolbar = Ti.UI.createToolbar({

                translucent:true,

      });
