Alloy - sync Adapter question

I need to store my settings in Ti.App.Properties. Surely, it is working but with a generated id like 

setting-ad07a98e-5249-2c2a-13bb-cf68fc668810 //Get it from Ti.App.listProperties()

Is the key supposed to be 1.1 ? because I set idAtrribute to version with default value 1.1

In controller file

var m = Alloy.createModel("setting");
m.save({
    version : "1.1",
    eftpos : {
        tyro : {
            mode : "Emulator"
        }
    }
});

setting.js

exports.definition = {
    config : {
        columns : {
            "version" : "String PRIMARY KEY",
            "eftpos" : "Object"
        },
        defaults : {
            "version" : "1.0",
            "eftpos" : {
                tyro : {
                    mode : 'Live'
                }
            }
        },
        adapter : {
            "type" : "properties",
            "idAttribute" : "version",
            "collection_name" : "setting"
        }
    },
    extendModel : function(Model) {
        _.extend(Model.prototype, {
            // extended functions and properties go here                    
        });

        return Model;
    },
    extendCollection : function(Collection) {
        _.extend(Collection.prototype, {
        });

        return Collection;
    }
};

    The "idAttribute" in the model config file is currently only supported by the sql adapter. The properties adapter only supports the current auto-id'ing.
