    Hi Markus,

    Try this : 

    var win = Ti.UI.currentWindow;

var Cloud = require('ti.cloud');


var tableView = Ti.UI.createTableView({
    top : '50dp',
    bottom : '50dp',
    width : '320dp'
});
win.add(tableView);

function getData() {
    var data = [];
    Cloud.Objects.query({
        classname : 'Kristianstad'
    }, function(e) {
        if (e.success) {
            for (var i = 0; i < e.Kristianstad.length; i++) {
                var gas = e.Kristianstad[i];
               var row = Ti.UI.createTableViewRow({
                    title : gas.bolag,
                    haschild : true,
                    height : '70dp',
                    backgroundColor : '#fff'
                });
                data.push(row);
            }
            tableView.data = data;
        } else {
            alert('Error:\\n' + ((e.error && e.message) || JSON.stringify(e)));
        }
    });

};
getData();
