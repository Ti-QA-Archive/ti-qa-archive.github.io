    titleAttributes: {
    color: '#fff'
}

  • Its a bit ugly but you can control the color within alloy by defining the titleControl properties in the xml definition file as below

    <Alloy>
    <NavigationWindow platform="ios">
        <Window id="index" class="container" titleControl="Ti.UI.createLabel({ text: 'White window title', color: 'white', width: Ti.UI.SIZE}) ">
        </Window>
    </NavigationWindow>
</Alloy>

  • It's platform-specific. titleControl() only works on ios, mobile web, and tizen. So, like the other answers, I'll post what I'm using right now in a project.

    if (Alloy.CFG.os === 'ios') {
    var titleLabel = Titanium.UI.createLabel({
        color : '#fff1bd',
        height : 18,
        width : 210,
        top : 10,
        text : "Home",
        textAlign : 'center',
        font : {
            fontSize : '28dp',
            fontWeight : 'bold'
        }
    });
    $.home.setTitleControl(titleLabel);
}
