    Surely you mean to do this? :

    
var currentLanguage = Titanium.Locale.getCurrentLanguage();
var taValue = ''; 

if(currentLanguage == "ja"){
    taValue = (CASE FOR JAPANESE);
} else {
    taValue = (CASE FOR ENGLISH);
};
