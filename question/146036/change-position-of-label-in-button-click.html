  • Hello Sharmila,

    See this…

    
var txtMiddleName = Ti.UI.createLabel({
    left : "10%",
    top : 50,
    height : 40,
    width : 300,
    text : 'Enter Middle Initial',
});

win.add(txtMiddleName);

var btn = Ti.UI.createButton({
    top : 120,
    left : 100,
    right : 100,
    height : 40,
    title : 'value'
});

btn.addEventListener('click', function(e) {
    txtMiddleName.left = '2%';
});

win.add(btn);

    It's working great for me, i hope it will help you.
