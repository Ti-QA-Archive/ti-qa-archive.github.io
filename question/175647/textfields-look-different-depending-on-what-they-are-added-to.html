Textfields look different depending on what they are added to

On some smartphones textfields sometimes looks different depending on what they are added to - window or scrollwindow. Has anyone experienced anything like this? (screenshot)

var login = Titanium.UI.createWindow({
       title:'Login Utente',
     height:'auto',
        tabBarHidden:true,
        layout : 'vertical',
    backgroundColor:'white'
});

var loginScrollView = Ti.UI.createScrollView({
    width: '100%',
    height:'auto',
    tabBarHidden:true,
    backgroundColor:'white',
    layout:'vertical'
});

var email = Titanium.UI.createTextField({
    color: 'black',
    top:'8%',
    left:'25%',
    width:'50%',
    height:'auto',
    font:{fontWeight:'bold',fontSize:'14sp'},
    hintText:'email',
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});
login.add(email); //ADD TO WINDOW

var password = Titanium.UI.createTextField({
    color: 'black',
     top:'2%',
    left:'25%',
    width:'50%',
    height:'auto',
    font:{fontWeight:'bold',fontSize:'14sp'},
    //passwordMask:true,
    hintText:'password',
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});
loginScrollView.add(password); //ADD TO SCROLLVIEW

login.add(loginScrollView);
login.open();
