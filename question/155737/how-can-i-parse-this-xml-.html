    Dear Hennrich,
    Use below code .I tested it at my end.

    var puxaTempo = Ti.Network.createHTTPClient({
    onerror:function(e){
    alert( e.erro )
    },

            onload:function(e){
            var ManifestXML = this.responseText;
            var xml_as_dom = Titanium.XML.parseString(ManifestXML);
            var my_items = xml_as_dom.getElementsByTagName("localtime").item(0).text;

            Ti.API.info( my_items ); //this is localtime value
        }  
    })
    puxaTempo.open( 'GET', 'http://www.earthtools.org/timezone-1.1/-23.548842/-46.638686' )
    puxaTempo.send();

    you need to add .item(0).text if there is only one tag <localtime> in your xml
    else you need to perform looping for multiple <localtime>.
    Like below code

    var puxaTempo = Ti.Network.createHTTPClient({
    onerror:function(e){
    alert( e.erro )
    },

            onload:function(e){
            var ManifestXML = this.responseText;
            var xml_as_dom = Titanium.XML.parseString(ManifestXML);
            var my_items = xml_as_dom.getElementsByTagName("localtime");
            var localtimeArr = [];
        for(var i=0; i<my_items.length; i++){
                  localtimeArr[i] = my_items.item(i).text;
                  Ti.API.info( localtimeArr[i] );//this is your local time value
             }

        }  
    })
    puxaTempo.open( 'GET', 'http://www.earthtools.org/timezone-1.1/-23.548842/-46.638686' )
    puxaTempo.send();

  • An other (shorter) way is to use a XML2JSON converter. For iOS you can use a native module from Omorandi and if you want to support also android you can use a javascript based module from David Bankier. Usage is very simple:

    onload: function() {
    var XMLTools = require("XMLTools");
    var xml = new XMLTools(this.responseText);
    var my_object = xml.toObject();
}

    Perhaps you use try/catch to prevent errors.
