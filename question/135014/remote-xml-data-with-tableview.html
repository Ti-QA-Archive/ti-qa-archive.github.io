  • Thank you for the help guys. I will post my code here for you to see what I'm doing. When the user then selects a row, I want the related XML data like: city, contact etc to be displayed in a new view.

    var win = Ti.UI.currentWindow;

//create an array to hold the remote xml data
var data = []; //empty data array

//declare the http client object
var xhr = Ti.Network.createHTTPClient();

//define search bar to attach to tableview
var searchBar = Ti.UI.createSearchBar({
    showCancel: true,
    height: 43,
    top: 0
});

//print out searchbar value whenever it changes
searchBar.addEventListener('change', function(e){
    //search the tableview as user types
    Ti.API.info('user searching for: ' + e.value);
});

//when the return key is hit, remove focus from searchbar
searchBar.addEventListener('return', function(e){
    searchBar.blur();
});

//when cancel button is tapped, remove focus from searchbar
searchBar.addEventListener('cancel', function(e){
    searchBar.blur();
});

//create the table view
var tblEmployees = Ti.UI.createTableView({
    height: 366,
    width: 320,
    top: 0,
    left: 0,
    rowHeight: 70,
    search: searchBar,
    filterAttribute: 'filter' //search filter which appears in TableRowView
});
win.add(tblEmployees);

//this method will process the remote data
xhr.onload = function(){

    var xml = this.responseXML;
    //get the item nodelist from response xml object
    var items = xml.documentElement.getElementsByTagName("employee");

    //loop each item in the xml
    for (var i = 0; i < items.length; i++){

        //create a table row
        var row = Ti.UI.createTableViewRow({
            hasChild: true,
            className: 'employee-row',
            filter: items.item(i).getElementsByTagName("firstName").item(0).text + ' ' + items.item(i).getElementsByTagName("lastName").item(0).text
        });

        //title label
        var titleLabel = Ti.UI.createLabel({
            text: items.item(i).getElementsByTagName("firstName").item(0).text + ' ' + items.item(i).getElementsByTagName("lastName").item(0).text,
            font: {fontSize: 14, fontWeight: 'bold'},
            left: 65,
            top: 10,
            height: 20,
            width: 210
        });
        row.add(titleLabel);

        //position label
        var positionLabel = Ti.UI.createLabel({
            text: items.item(i).getElementsByTagName("title").item(0).text,
            font: {fontSize: 10, fontWeight: 'normal'},
            left: 65,
            top: 15,
            height: 40,
            width: 200
        });
        row.add(positionLabel);

        //add icon to the left of the row
        var iconImage = Ti.UI.createImageView({
            image: 'http://localhost/mobile/pics/' + items.item(i).getElementsByTagName("picture").item(0).text,
            width: 50,
            height: 50,
            left: 10,
            top: 10
        });
        row.add(iconImage);

        //add the table row to our data[] object
        data.push(row);

    }

    //set the data property of the tableView to the data[] object
    tblEmployees.data = data;

};

//this method will fire there's an error in accessing the remote data
xhr.onerror = function(){
    //log the error to Titanium studio console
    Ti.API.error(this.status + ' _ ' + this.statusText);
};

//open up the employees xml feed
xhr.open('GET', 'http://localhost//mobile/employees.xml');

//execute the call to the remote feed
xhr.send();
