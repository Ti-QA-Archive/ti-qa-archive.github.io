    Hello Krishna

    Try this :

    var view = Ti.UI.createView({
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,
});

var img = Ti.UI.createImageView({
    height : '100dp',  // as you need
    width : '100dp',  // as you need
    image : 'image.png'  // your image
});

var lbl = Ti.UI.createLabel({
    zIndex : 1,
    text : 'Your Text'   // latitude longitude value,
    color : 'red'  //  as you need,
    top : '10dp'  // set as you required in image
});

view.add(img);
view.add(lbl);

var mergeImage = view.toImage();

    as above code is write Latitude and Longitude in image. you can send mergeImage as in your code. thanks.

  • This worked for me ….

    For merging the image and text, it should be 

    var mergeImage = captureView.toImage().media;

    Added the eventlistener for load in imageView

    imageView.addEventListener('load', imageLoaded);

    And the imageLoaded function

    function imageLoaded(e) {
    var mergeImage = captureView.toImage().media;
    var f = Ti.Filesystem.getFile(Ti.Filesystem.externalStorageDirectory + 'imageName.png');
    f.write(mergeImage);
    Ti.Media.Android.scanMediaFiles([f.nativePath], null, function(e) {
    });

}

    This will save the image, once it is fully loaded and hence we will get the full image with the text on top of it.
