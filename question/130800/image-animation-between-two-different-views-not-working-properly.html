Image Animation between two different views not working properly

Hi,
I have two views viz., lower_view & view0. I want the image 'right_logo' located in the lower view to first scale on click event and then transition to the center of view0. I have tried the below code but it fails to transition the image to center of other view.

var center1 = null;
var scaled1 = false;
var win = Titanium.UI.createWindow({  
     url:'crestor.js',
     orientationModes:[Ti.UI.LANDSCAPE_LEFT, Ti.UI.LANDSCAPE_RIGHT],
     backgroundColor:'white'

});
win.open();
var view0 = Ti.UI.createView({
    backgroundColor : '(149,201,236)',
    borderRadius : 10,
    width : 1030,
    height : 600,
    top : 0
});
win.add(view0);
var lower_view=Titanium.UI.createView({
    backgroundImage:'lower_background.png',
    height:150,
    top:600,
    width:1030
});
win.add(lower_view);
var right_logo = Titanium.UI.createImageView({
    url:'right_logo.png',
    height : 108,
    right : 50,
    top : 30,         
    width : 112
}); 
lower_view.add(right_logo);
right_logo.addEventListener('click', function()
{
    var t = Titanium.UI.create2DMatrix();
    if (!scaled1)
    {
        t = t.scale(3.0);
        center1 = right_logo.center;
        right_logo.animate({transform:t,center:view0.center,
zIndex:10,duration:500});
        scaled1 = true;
    }
    else
    {
        image1.animate({transform:t,center:center1,zIndex:1,
duration:500});
        scaled1 = false;
    }

});

Could anyone please suggest a solution as i need to do the job asap?
Thanks.

    try to add the right_logo to the win and not to the lower_view because the coordinates are relative to the parent container.
    You will have to change the right_logo coordinates relative to the win.
