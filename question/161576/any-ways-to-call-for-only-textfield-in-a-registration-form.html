  • I assume you added your registration form items to a scrollview. The solution is to assign a custom property to each item in your form. In the example below i have named the property "viewType", but this is arbitrary.
    Then loop through all children of your form/scrollview and push only those views with a specific "viewType" into a new array.
    You could do something like this.

    // begin simple form
var scrollView = Ti.UI.createScrollView();

var label1 = Ti.UI.createLabel({id: 1. viewType: "label"});
var textField1 = Ti.UI.createTextField({id: 1, viewType: "label"});
var label2 = Ti.UI.createLabel({id: 2, viewType: "label"});
var textField2 = Ti.UI.createTextField({id:2, viewType: "label"});
scrollView.add(label1);
scrollView.add(textField1);
scrollView.add(label2);
scrollView.add(textField2);
// end simple form



var allTextFields = [];

scrollView.children.forEach(function(view) {

    if (view.viewType == "textField") {
        allTextFields.push(view);
    }    
});

// Basic example to set all values

allTextFields.forEach(function(textField) {
    if (textField.value == null) {
        textField.value = 'test';
    }
});

Ti.API.info("allTextFields: "+JSON.stringify(allTextFields, undefined, 2);

    you can also use lodash.js/underscore.js and use this one liner ;)

    var allTextFields = _(scrollView.children).filter({viewType: 'textField'}).value();
