  • the method is pretty easy.
    at first, create a tabgroup within your main app page (app.js)

    var tabGroup = Titanium.UI.createTabGroup();
    add your main page as a tab:

    //create main page window link (your main page will be inside mainPage.js)

var mainPage = Titanium.UI.createWindow({
    title: '_mainPage_Title',
    url: "mainPage.js"
});


//create a tab for the main page and add it to the tabGroup
var mainPageTab = Titanium.UI.createTab({  
    title: 'mainPage',
    window: mainPage
});

tabGroup.addTab(mainPageTab); 

//you can set it as the default tab
tabGroup.setActiveTab(mainPageTab);

tabGroup.open();

    now you open up links inside you mainPage.js which will automatically have a leftside back button:
    Titanium.UI.currentTab.open(page,{animated:true});
