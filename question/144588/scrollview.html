var win = Ti.UI.createWindow({
    height : 200,
    width : 320,
    backgroundColor : 'white'
});

var scroll = Ti.UI.createScrollView({
    height : 250,
    top : 0,
    width : 320,
    scrollType : 'vertical'
});

win.add(scroll);

var button1 = Ti.UI.createButton({
    top : 150,
    height : 20,
    width : 90,
    title : 'Get Direction'
});
scroll.add(button1);
button1.addEventListener('click', function(e) {
    alert(scroll.getScrollType());
})

win.open();

    This code will give you the direction of scroll view . Please mark the answer as permanent if you problem is solved
