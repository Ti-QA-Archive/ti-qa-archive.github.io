  • Hi Ali,

    You cannot use the center value to calculate the center of a view as it is used for specifying where the view should be placed and does not indicate the position.

    However the code I have provided below will give you the center values of the view.

    var win = Ti.UI.createWindow({
    fullscreen: true,
    height: Ti.Platform.displayCaps.platformHeight,
    width: Ti.Platform.displayCaps.platformWidth,
    backgroundColor: 'green'
});

var view1 = Ti.UI.createView({
    top : 50,
    left : 50,
    width : 50,
    height : 50,
    backgroundColor: 'black'
});

var view2 = Ti.UI.createView({
    top : 100,
    left : 100,
    width:50,
    height:50,
    backgroundColor: 'red'
});

win.add(view1);
win.add(view2);
win.open();

var viewCenterY = view2.top + view2.height / 2;
var viewCenterX = view2.left + view2.width / 2;

Ti.API.info('view2s location is :'+ viewCenterX + ',' + viewCenterY);
