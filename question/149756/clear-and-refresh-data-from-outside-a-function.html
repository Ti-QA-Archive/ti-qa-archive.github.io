Latest Titanium SDK
iOS Application
Trying to get it working on iOS 4.3 and above.
Working off a Mac, OSX Mountain Lion.

                 var locationCallback = function(e)
                {
                    //Mobileweb seems to be not firing window event for some odd reason.
                    //Forcing a window open and focus event.
                    if(mainWin.openedflag == 0 ){
                        Ti.API.info('firing open event');
                        mainWin.fireEvent('open');
                    }
                    if(mainWin.focusedflag == 0){
                        Ti.API.info('firing focus event');
                        mainWin.fireEvent('focus');
                    }
                    if (!e.success || e.error)
                    {
                        coor.text = 'error:' + JSON.stringify(e.error);

            });
