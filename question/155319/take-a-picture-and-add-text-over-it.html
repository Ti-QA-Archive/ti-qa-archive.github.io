    • Here is a working example app so you can see it working;

      To test this as a new app - create anew classic default app and replace the contents of the app.js with the following code, or pick and choose as you see fit.

      var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
var view = Ti.UI.createView({
    height: Ti.UI.SIZE,
    width: Ti.UI.FILL
});
var img = Ti.UI.createImageView({
    height: Ti.UI.SIZE,
    image: 'example.jpg', // change this to suit
    top: 0,
    width: Ti.UI.SIZE
});
var lbl = Ti.UI.createLabel({
    color: '#000',
    height: Ti.UI.SIZE,
    font: {
        fontSize: 24
    },
    text: 'Hello',
    textAlign: 'center',
    width: Ti.UI.SIZE
});
win.add(view);
view.add(img);
view.add(lbl);
view.addEventListener('click', function (e) {
    var combined = Ti.UI.createImageView({
        height: Ti.UI.SIZE,
        image: view.toImage(null, true),
        top: 0,
        width: Ti.UI.SIZE
    });
    view.hide();
    win.add(combined);    
});
win.open();

    • Here you go, for the purposes of testing the code use the gallery rather than the camera works the same way.

      var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
var view = Ti.UI.createView({
    height: Ti.UI.SIZE,
    width: Ti.UI.FILL
});
var img = Ti.UI.createImageView({
    height: Ti.UI.SIZE,
    //image: 'example.jpg',
    top: 0,
    width: Ti.UI.SIZE
});
var lbl = Ti.UI.createLabel({
    color: 'red',
    height: Ti.UI.SIZE,
    font: {
        fontSize: 24
    },
    text: 'Hello',
    textAlign: 'center',
    width: Ti.UI.SIZE
});
win.add(view);
view.add(img);
view.add(lbl);
var btn = Ti.UI.createButton({
    bottom: 20,
    title: 'Gallery',
    width: 150
});
btn.addEventListener('click', function (e ){
    Ti.Media.openPhotoGallery({
        success: function (e) {
            img.setImage(e.media.imageAsThumbnail(640));
        },
        cancel: function () {
            Ti.API.info('Gallery.Open: ' + 'Cancelled');
        },
        error: function (e) {
            Ti.API.info('Gallery.Open: ' + 'Error');
        },
        mediaTypes: [Ti.Media.MEDIA_TYPE_PHOTO]
    });
});
win.add(btn);
view.addEventListener('click', function (e) {
    var combined = Ti.UI.createImageView({
        height: Ti.UI.SIZE,
        image: view.toImage(null, true),
        top: 0,
        width: Ti.UI.SIZE
    });
    view.hide();
    win.add(combined);    
});
win.open();
