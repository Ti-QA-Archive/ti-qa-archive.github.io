    Hello Muhammad,

    Try like this…

    
var mapAnnotation = Titanium.Map.createAnnotation({
    latitude:-32.707578, 
    longitude:71.797028,
    pincolor: Titanium.Map.ANNOTATION_RED,
    animate:true
});

var map = Titanium.Map.createView({
    top : 0,
    mapType: Titanium.Map.STANDARD_TYPE,
    region:{latitude:-32.707578, longitude:71.797028, latitudeDelta:0.5, longitudeDelta:0.5},
    animate:true,
    regionFit:true,
    userLocation:true,
    annotations:[mapAnnotation]
});
win.add(map);
