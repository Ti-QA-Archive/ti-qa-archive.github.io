Rotating an image using 'touchmove'

Hi guys,

I've been looking for a script to help me rotate an image by touching and dragging it and I've come up with nothing that gets the job done - so I decided to try and make one myself (using the bits and pieces I've found).

Basically, what I want is something like a turntable, where you touch one part of an image and rotate it depending on where your finger is.

The code below is what I've got so far, and I think I'm on the right track, but I can't seem to get my mind around what else I could be missing.

Any suggestions would be awesome.

Cheers.

window = Titanium.UI.createWindow({
    backgroundColor:'#47a',
});

window.orientation = Titanium.UI.PORTRAIT;
Ti.Gesture.addEventListener('orientationchange',function(e){
    window.orientation = e.orientation;
});

//load test image from the appcelerator web site
var image = Titanium.UI.createImageView({
   image:'http://www.appcelerator.com/wp-content/themes/appcelerator/img/HERO_iPh_social.png',
   backgroundColor:'transparent',
   height:'auto',
   width:'auto'
});


var startx;  // touchstart
var starty;  // touchstart
var endx; // touchmove current coordinates 
var endy; // touchmove current coordinates 
var distance;
var isTouched = false;
var rotation = 1;

// Spin the image
var t = Ti.UI.create2DMatrix();
var spin = Titanium.UI.createAnimation();
t = t.rotate(rotation);
spin.transform = t;
spin.duration = 10;

image.addEventListener('touchstart', function(e) 
{
    isTouched = true; 
    startx = e.x;
    starty = e.y;
});

image.addEventListener('touchend', function(e) 
{
    isTouched = false; 
});

image.addEventListener('touchmove', function(e) 
{ 
    endx = e.x;
    endy = e.y;

    if (isTouched)
    {
        if (endx > startx)
        {
            rotation += 4;
        }
        else
        {
            rotation -= 4;
        }
        startx = endx;
        image.animate(spin);

    }
});


window.add(image);
window.open({fullscreen:true});

  • I think you're on the right track with regard to the X positioning as the user drags the wheel left and right - but you're not taking into account that they may have their finger above or below the mid point of the dial - ie at the 12 o'clock position or the 6 o'clock position. Going left at the 12 o'clock position is anti-clockwise, but going left at the 6 o'clock position is clockwise.

    Also - because the changes in X at the 9 or 3 o'clock will be mimimal, (and likewise the changes in Y at the 12 or 6 o'clock positon) you might want to think about using a threshold to limit rotation in case the X (or Y) changed by just one pixel.

    Here is a crude example for checking the change in X (you would still need to add something to check the change in Y). Obviously, I'm not an efficient coder but hopefully it does show conceptually what you need to take into account.

    You'll need to define the 'Y_midpoint_of_dial' value - and you'll need to build similar code for the Y changes (in the up/down direction).

    You might want to replace your 'if (endx > startx)' statement with something like I posted below… I used a threshold of 5, you might want to experiment to see what works best in real life:

    //X: user scrolling L to R
  if (endx - startx > 5) {
    //is user touching closer to 6 or 12 o'clock? 
    //ie: are they touching above or below the middle of the dial
    if (starty > Y_midpoint_of_dial) {
      //finger is below middle of dial - so going anti-clockwise
      rotation -= 4;
    } else {
      //finger is above middle of dial - so going clockwise
      rotation += 4;
    }
  }

//X: user scrolling R to L
  if (startx - endx > 5) {
    //is user touching closer to 6 or 12 o'clock? 
    //ie: are they touching above or below the middle of the dial
    if (starty > Y_midpoint_of_dial) {
      //finger is below middle of dial - so going clockwise
      rotation += 4;
    } else {
      //finger is above middle of dial - so going anti-clockwise
      rotation -= 4;
    }
  }

    I hope this is helpful…

  • Okay, here's what I have so far. I've (theoretically) divided the image into quadrants and identified where the touch began and where it's headed. Now while I think I have everything right, the problem seems to lie in the way the image rotates. For lack of a better phrase, it's a little wonky.

    I'm not sure whether or not this is a problem in the animation itself, or if it's in the way touchmove updates in Titanium.

    Any ideas?

    Here's the code:

    window = Titanium.UI.createWindow({
    backgroundColor:'#47a',
    exitOnClose: true
});

// load test image from the appcelerator web site
var image = Titanium.UI.createImageView({
   image: 'http://www.appcelerator.com/wp-content/themes/appcelerator/img/HERO_iPh_social.png',
   backgroundColor:'transparent',
   height:'auto',
   width:'auto'
});

var startx;  // touchstart
var starty;  // touchstart
var endx; // touchmove current coordinates 
var endy; // touchmove current coordinates 
var distance;
var isTouched = false;

function spinClockwise()
{
    var t = Ti.UI.create2DMatrix();
    var spin = Titanium.UI.createAnimation();
    t = t.rotate(distance/1.5);
    spin.transform = t;
    spin.duration = 100;

    image.animate(spin);
}

function spinCClockwise()
{
    var t = Ti.UI.create2DMatrix();
    var spin = Titanium.UI.createAnimation();
    t = t.rotate(-distance/1.5);
    spin.transform = t;
    spin.duration = 100;

    image.animate(spin);
}

image.addEventListener('touchstart', function(e) 
{
    isTouched = true; 
    startx = e.x;
    starty = e.y;
});

image.addEventListener('touchend', function(e) 
{
    isTouched = false; 
});

image.addEventListener('touchmove', function(e) 
{ 
    endx = e.x;
    endy = e.y;

    // compute and set the distance variable
    // dx squared + dy squared = distance squared
    distance = Math.sqrt( (Math.abs(endx-startx) * Math.abs(endx-startx)) + (Math.abs(endy-starty) * Math.abs(endy-starty)) );

    if (isTouched)
    {
        // Check where the touch began and what direction it's moving to in a quadrant.
        // Assuming the image center is (0,0) Quadrant A would be (+,+), Quadrant B would be (-,+),
        // Quadrant C would be (-,-), and Quadrant D would be (+,-) 

        if (startx > image.animatedCenter.x && starty > image.animatedCenter.y) // touch began in quadrant A
        {
            if (endx < startx) // touch moved left
            {
                spinCClockwise(); // rotate counter clockwise    
            }
            if (endy < starty) // touch moved down
            {
                spinClockwise(); // rotate clockwise
            }

        }
        else if (startx < image.animatedCenter.x && starty > image.animatedCenter.y) // touch began in quadrant B
        {
            if (endx > startx) // touch moved right
            {
                spinClockwise(); // rotate clockwise
            }    
            if (endy < starty) // touch moved down
            {
                spinCClockwise(); // rotate counter clockwise     
            }
        }
        else if (startx < image.animatedCenter.x && starty < image.animatedCenter.y) // touch began in quadrant C
        {
            if (endx > startx) // touch moved right
            {
                spinCClockwise(); // rotate counter clockwise
            }    
            if (endy > starty) // touch moved up
            {
                spinClockwise(); // rotate clockwise     
            }
        }
        else if (startx > image.animatedCenter.x && starty < image.animatedCenter.y) // touch began in quadrant D
        {
            if (endx < startx) // touch moved right
            {
                spinClockwise(); // rotate clockwise
            }    
            if (endy > starty) // touch moved up
            {
                spinCClockwise(); // rotate counter clockwise     
            }
        }

        // reset start variable to current position
//        startx = endx;
//        starty = endy;
    }
});

window.add(image);
window.open({fullscreen:true});
