android option menu triggers all events on open

Hello all.
I added an option menu to my app like this:

var activity = Ti.Android.currentActivity;
activity.onCreateOptionsMenu = function(e){
    var menu = e.menu;
    var reply = menu.add({});
    var share = menu.add({});
    var facebook = menu.add({});

    reply.setIcon('mail_reply.png');
    share.setIcon('mail-replied.png');
    facebook.setIcon('facebook.png');

    reply.addEventListener('click', emailReply());
    share.addEventListener('click', emailPublish());
    facebook.addEventListener('click', FBpublish());
};

the problem is that once i click the options button on my phone all the events trigger instead of just wait for the button click.
Could it be because I'm adding empty objects? i don't want any titles or anything like that.
after I discard all the events that triggered the menu opens but it doesn't respond to clicks any more.

thanks for the help…
