  • Hi Simone,

    Try this code : 

    function facebook() {
    Titanium.Facebook.appid = Titanium.Locale.getString('fb_id');
    Titanium.Facebook.permissions = ['publish_stream'];
    Titanium.Facebook.forceDialogAuth = true;
    if (Ti.Facebook.loggedIn) {
        //Already Logged in
    } else {
        Titanium.Facebook.addEventListener('login', function(e) {
            if (e.success) {
                //Success
            } else if (e.error) {
                //Error
            }
        });
        //call the facebook authorize method to login
        Titanium.Facebook.authorize();
    }
}
