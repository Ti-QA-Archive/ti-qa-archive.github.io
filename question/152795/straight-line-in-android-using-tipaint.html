  • There is no need to use paint module to draw line,square circle etc, you can draw it with view itself:

    Here is code below:

    
var win = Ti.UI.createWindow();
win.backgroundColor = "white";
var lineView = Ti.UI.createImageView({
    top:50,
    width:100,
    height:2,
    backgroundColor:'red',

    });

win.add(lineView);

var circleView = Ti.UI.createImageView({
    top:200,
    width:100,
    height:100,
    borderRadius:50,
    backgroundColor:'red',

    });

win.add(circleView);

var squareView = Ti.UI.createImageView({
    top:350,
    width:100,
    height:100,
    backgroundColor:'red',

    });

win.add(squareView);

win.open();

    Let me know in case of any query.
