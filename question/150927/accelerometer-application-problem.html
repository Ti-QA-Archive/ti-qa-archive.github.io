Accelerometer Application problem

Hi,

I created an accelerometer application, it's working but my object is passing through the screen width and height , this is my code:



var win = Titanium.UI.createWindow({  
    title:'Accelerometer',
    backgroundColor:'#fff'
});


var dot = Titanium.UI.createView({
    backgroundColor: "#FF0000",
    backgroundImage:'KS_nav_views.png',
    width: 70,
    height: 70,
    left: Titanium.Platform.displayCaps.platformWidth/2,
    top: Titanium.Platform.displayCaps.platformHeight/2
});


win.add(dot);
win.open();

var lastTime = new Date().getTime();


var offset = 50;

var filter = 1.0;


var last_x = 0;
var last_y = 0;


Titanium.Accelerometer.addEventListener('update',function(e)
{

    var now = new Date().getTime();

    if(lastTime + offset < now )
    {

        last_x = e.x * filter + last_x * (1 - filter);
        last_y = e.y * filter + last_y * (1 - filter);


        dot.left -= (last_x*10);
        dot.top += (last_y*10);


        lastTime = now;
    }
});


if (Ti.Platform.osname == 'android'){
    Ti.Gesture.addEventListener('orientationchange', function(e) {
        var curAct = Ti.Android.currentActivity;
        curAct.setRequestedOrientation(Ti.Android.SCREEN_ORIENTATION_PORTRAIT);
    });
}

Thanks in advance.
