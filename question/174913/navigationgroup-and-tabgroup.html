navigation.prototype.openWindow=function(e){
    var win=Alloy.createController(e.source.destino.trim()).getView();
    Ti.API.info(e.source.destino);

        if (Ti.Platform.osname=='iphone'){
            var leftButton=Ti.UI.createButton({
                title:"Close",
                style:Ti.UI.iPhone.SystemButtonStyle.BORDERED,
                color:"#990000",
                backgroundColor:"blue"
            });
            leftButton.addEventListener('click',function(e){
                win.close();
            });

            win.setLeftNavButton(leftButton);
            self.navegacion.openWindow(win);
        }
        else{


            win.open();

        }
};

    Why are you adding a navigationGroup to a tabGroup? Tabs are by themselves navigationGroups :)

    tab_one.open(win)

    Will have the same effect like you'd have a navigationGroup.
