  • Hello andres guerrero,

    Thanks for your sample code. Here I have added sample code for your issues. In my sample code I add an Event Listener in table view and also I add another check for button event. Please test my code in your environments. I hope it solves your problem.

    Check attachment

    Test Code:

    
var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

var dataMessages = [];
for (var i = 0; i < 10; i++) {
    //var idMessage = dataMessages[i].idMessage;
    var row = Ti.UI.createTableViewRow({
        //title:idMessage,
        backgroundColor : '#FFF',
        height : '60dp',
        backgroundSelectedColor : '#009fcd',
        zIndex : 2
    });
    var indicatorStatus = Ti.UI.createView({
        left : '5dp',
        width : '12dp',
        height : 'auto',
        borderWidth : 2,
        borderColor : '#DAD6DA'
    });

    var checkMessage = Ti.UI.createButton({
        width : '50dp',
        height : '20dp',
        left : '20dp',
        title : 'Button',
        center : '0dp',
        value : false,
        borderRadius : 2,
        backgroundColor : '#DAD6DA',
        id : 'checkMessage',
    });

    var titleMessage = Ti.UI.createLabel({
        color : '#727072',
        objName : 'label',
        //text: dataMessages[i].titleMessage,
        touchEnabled : false,
        left : '55dp',
        width : '150dp',
        height : '20dp',
        top : '7dp',
        font : {
            fontSize : '18dp',
            fontFamily : 'ABeeZee-Regular',
            fontWeight : 'bold'
        }
    });
    var bodyMessage = Ti.UI.createLabel({
        color : '#727072',
        objName : 'label',
        text : 'Row ' + (i + 1),
        touchEnabled : false,
        left : '55dp',
        width : '150dp',
        height : '15dp',
        bottom : '8dp',
        font : {
            fontSize : '12dp',
            fontFamily : 'ABeeZee-Regular',
        }
    });
    var dateMessage = Ti.UI.createLabel({
        color : '#727072',
        objName : 'label',
        //text: dataMessages[i].dateMessage,
        touchEnabled : false,
        right : '25dp',
        width : '80dp',
        width : 'auto',
        font : {
            fontSize : '12dp',
            fontFamily : 'ABeeZee-Regular'
        }
    });
    row.add(indicatorStatus);
    row.add(checkMessage);
    row.add(titleMessage);
    row.add(dateMessage);
    row.add(bodyMessage);
    dataMessages.push(row);
}

// now assign that array to the table's data property to add those objects as rows
var listMessages = Titanium.UI.createTableView({
    data : dataMessages,
    top : '92dp',
    width : Ti.Platform.displayCaps.platformWidth,
    separatorColor : '#DAD6DA',
    backgroundColor : '#FFF',
    scrollabe : true
});

listMessages.addEventListener('click', function(e) {
    if (e.source.name == "label") {
        alert(e.rowData.title);
    } else if (e.source.id == "checkMessage") {
        alert(e.source.title);
    }
});

win.add(listMessages);
win.open();

    TableViews-Customrows
