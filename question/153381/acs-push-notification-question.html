    you have the api to send push.

    function notifiy(){
Cloud.PushNotifications.notify({
    channel: 'friend_request',
    payload: 'Welcome to push notifications'
}, function (e) {
    if (e.success) {
        alert('Success');
    } else {
        alert('Error:\n' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});
}

    you can have a code like this :

    setTimeout(notifiy, 10000);

    send notification in 10 seconds…for example.
