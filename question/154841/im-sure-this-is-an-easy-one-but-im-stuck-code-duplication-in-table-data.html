    function getTableViewRow(data){
    var row = Ti.UI.createTableViewRow({
    backgroundColor:'transparent',
        height:47, 
    color: '#FFFFFFF',
        font:{
    fontFamily:'Helvetica Neue',fontSize:16
    },
    title:data.title,
    url:data.url,
    leftImage:data.image
});

return row;

}

var data = [{title:'Title', image:'/image.png', url:'/path/to/url'}]; /// etc

var rows = [];
for (var x = 0;x<data.length;x++){
    rows.push(getTableViewRow(data[x]));
}

tableView.setData(rows});

    • Sure, Matt

      We declared an array called data that we are holding our tableViewRow information in

      var data = [];

      The number of items in the array can be counted with length, so

      data.length

      tells us how many items we have in the array.

      We can now set up a loop in javascript to iterate through each item in the array

      for (var x = 0; x< data.length;x++){
}

      This declares a variable x starting at zero and repeats the code within the { } until the value of x is no longer less than the number of items in the array - In other words repeat this code for each item in the array. After each iteration of the loop we add 1 to the value of x

      We can now pass the individual items in our array to the getTableViewRow function 

      getTableViewRow(data[x])

      and we add it to our second array rows

      rows.push(getTableViewRow(data[x]));

      Which we can now use to add the tableViewRows we've created to the tableView
