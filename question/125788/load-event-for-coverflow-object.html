    var view = Titanium.UI.createCoverFlowView({

// SHOW ACTIVITY INDICATOR HERE

while (group_images_rows.isValidRow())
{   
    images.push({image:'../images/'...});
    imgno++;

}

view.images = images;

  • i would create and open the view without the images attached, load the images, count the rows to get a number count of images, set a var with an image count and make it increment when each image is loaded to the array.
    When the imgno is the same number as the rowcount it has loaded the last image, and you can hide the activity indicator.

    Example… (not tested, just typed as an example)

    var view = Titanium.UI.createCoverFlowView({
    backgroundColor:'#000'
});
win.add(view);

// SHOW ACTIVITY INDICATOR HERE

var count = group_images_rows.getRowCount());
var imgno = 0;

while (group_images_rows.isValidRow())
{   
    images.push({image:'../images/'...});
    imgno++
    group_images_rows.next();
}
if (imgno == count){
// HIDE ACTIVITY INDICATOR
}

view.images = images;
