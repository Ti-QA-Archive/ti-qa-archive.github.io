    If you want to track whole application screen history 

    var global_list = Titanium.App.Properties.getList('globalScreen_list',[]);
alert(global_list.length);

global_list.push('Screen Name In which you currently enter');
Titanium.App.Properties.setList('globalScreen_list', global_list);

var global_list2 = Titanium.App.Properties.getList('globalScreen_list',[]);
alert(global_list2.length);
