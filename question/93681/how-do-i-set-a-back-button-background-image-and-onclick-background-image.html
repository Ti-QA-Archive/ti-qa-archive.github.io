    Just create a button and set that as leftNavButton, assign an action to close the window, done.

    var btnBack = new Ti.UI.createButton({
        backgroundImage:'image.png',
        backgroundSelectedImage:'image2.png',
        width:80,height:25
});
var win = new Ti.UI.createWindow({
    url:'something.js',
    leftNavButton:btnBack
});
btnBack.addEventListener('click', function(){
    win.close();
});
