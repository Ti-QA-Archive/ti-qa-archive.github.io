navigationGroup - windows controls overlat (previous window images shown)

Ti.App.addEventListener('profileClick', function(e) {
        var ProfileWin = require('ui/profile/userProfileWindow');
        var profileWin = new ProfileWin(model.driver);
        //profileWin.open();
        var navController = Ti.UI.iPhone.createNavigationGroup({
            window : profileWin,
        });
        ui.add(navController);
        profileWin.nav = navController;

    });

Window A contains some images with zIndex 1
