    cancel:function()
    {

    },

        // create alert

            a.setMessage('Device does not have video recording capabilities');

  • In webview you can have limited access to some Ti functionalities. To communicate between webview & Titanium JS code:

    • Webview html Code: 

      Ti.App.fireEvent('showCamera');

    • Titanium JS Code: 

    Ti.API.addEventListener('showCamera', function() {
    // Show camera here
});

      function Init() {
    // var FirstView = require('windows/win3.js');
    var left = document.getElementById("left");

    left.addEventListener('click', function(e) {
        alert("2");    
        Ti.App.fireEvent('showCamera');
    });            
}
