    message: '',

function xmlrpc(url, params, callback)
{
    var xhr = Titanium.Network.createHTTPClient();
    xhr.open("POST",url);
    xhr.onload = callback;
    xhr.send(params);
}

function rpc_cb()
{
        my_alert.message = '['+this.responseText+']';

var rpc_url = "http://127.0.0.1/lotto/xxx.php";

  • Make sure from both simulators you have access to 127.0.0.1. The simulator runs on a different network area. I'd suggest to use a "public endpoint" to test and make sure that the problem is a network problem and not a code one.
