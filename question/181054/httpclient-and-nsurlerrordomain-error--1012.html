HTTPClient and NSURLErrorDomain error -1012

Hi,

i get an error (-1012) with the HTTPClient in Titanium.

    var xhr = Ti.Network.createHTTPClient({
            onload : function() {
                    if (xhr.responseText != '' && xhr.responseText != null) {

                        var res = null;
                        try { res = JSON.parse(xhr.responseText); }
                        catch(e) { ... }

                        if (res != null) {
                            ....
                        } else {
                            ....
                        }

                    } else {
                        ....
                    }
            },
            ondatastream : function(e) {

            },
            onreadystatechange : function(e) {
                    ...  
            },
            onerror : function(e) {
                    Ti.API.error('Error: '+JSON.stringify(e));
            },
            timeout : 20000,
            cache : false

    });

    xhr.open('GET', "http://"+username+":"+password+"@"+hostname+":"+port, true);
    xhr.send();

Error Message (Username, Passwort, Host and Port is replaced!):

{"success":false,"code":-1012,"source":{"method":"GET","async":true,"timeout":20000,"cache":false,"url":"http://USERNAME:PASSWORD@HOSTNAME:PORT"},"type":"error","error":"The operation couldn’t be completed. (NSURLErrorDomain error -1012.)"}

When i type the URL in Browser, it works perfectly and i get the JSON response.
What is the problem here?
