Window size and tableview

Hi, i am doing a split view app for iPad and i need to modify the size of my window in relationship with a tableview (example:my tableview has 3 rows, my window's size must be of 3 rows and no other white rows) how i can do? thanks to everyone

  • hi Raffaele,

    try this

    var win = Ti.UI.createWindow({
    // backgroundColor : '#666666'
    backgroundColor : 'white',
    // backgroundImage : 'img/1.jpg'
});

var data = [];

data[0] = Ti.UI.createTableViewRow({
    hasChild : true,
    title : 'Row 1'
});
data[1] = Ti.UI.createTableViewRow({
    hasDetail : true,
    title : 'Row 2'
});
data[2] = Ti.UI.createTableViewRow({
    hasCheck : true,
    title : 'Row 3'
});

var footerView = Ti.UI.createView({
    height : 1
});

// create table view
var tableview = Titanium.UI.createTableView({
    data : data,
    rowHeight  : 40,
    footerView : footerView,
    scrollable : false
});

var height = tableview.data[0].rows.length * 40;
win.height = height;
// add table view to the window
win.add(tableview);

win.open();
