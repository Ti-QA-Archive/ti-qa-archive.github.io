Labels in TableViewRow

SDK 1.8.2.

I am trying to add labels to TableViewRows in a grouped TableView. When I place a label in the row with top:0 it is positioned very low. So I have to use -60 to place it on top. I have an offset of 60 …

Any idea?

code:

function HomeWindow(){

    var self = Ti.UI.createWindow({
        title:'Mijn gegevens',
        backgroundColor:'white',
        fullscreen:true
    });
    var dataTable = Ti.UI.createTableView({
        style: Titanium.UI.iPhone.TableViewStyle.GROUPED,
        touchEnabled: false
    });    

    var tableData = new Array();
    /*
     * NAW section
     */
    var nawSection = Ti.UI.createTableViewSection();
    nawSection.add(Ti.UI.createTableViewRow({
            title: member.firstname + ' ' + member.lastname        // this cell is ok...
        }));

    var addressRow = Ti.UI.createTableViewRow({height:80}); // layout vertical doesn't work?!?

    addressRow.add(Ti.UI.createLabel({text:member.address , top:-60, left:10}));
    addressRow.add(Ti.UI.createLabel({text: member.zipcode , top:0,left:10}));
    nawSection.add(addressRow);

    tableData.push(nawSection);



    dataTable.data = tableData;

    self.add(dataTable);



    return self;
}

  • Try setting the height of the labels. By default, they are expanding to fill the row. 

        addressRow.add(Ti.UI.createLabel({text:member.address , height:20, top:10, left:10}));
    addressRow.add(Ti.UI.createLabel({text: member.zipcode , height:20, top:40,left:10}));
