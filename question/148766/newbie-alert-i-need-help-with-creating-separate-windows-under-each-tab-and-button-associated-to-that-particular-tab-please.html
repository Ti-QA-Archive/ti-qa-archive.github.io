    var tabGroup=Ti.UI.createTabGroup();

var win1=Ti.UI.createWindow();

var tab1=Ti.UI.createTab({
title:'Your Title',
window:win1,
});

tabGroup.addTab(tab1);
tabGroup.active=tab1;
tabGroup.open();

    pls see this example this is for one tab if you want more tabs and windows you can create more windows and tabs like i did and add windows to tabs and tabs to the tab group. Please note that you shud have max 5 tabs.
