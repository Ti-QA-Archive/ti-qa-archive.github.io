  • Hi,

    Thanks for writing. I have tested your issue with Android Device 4.2.2.Its working without any error. Here is my sample code. It may help to you. Can you please test my code in your environments?

    Test Code:

    

var win = Ti.UI.createWindow({backgroundColor: 'white'});

// Create button

var Btn = Ti.UI.createButton({
    title:'Capture',
    width:100,
    height:100,
    backgroundColor:'pink'
});

// Add click event to capture button

 var selectedImg;
Btn.addEventListener('click', opencamera); 


function opencamera() {


    var opengallery = Ti.UI.createOptionDialog({
    title:'Image Saved',
    options:['Open Camera', 'Close']
    });

    opengallery.addEventListener('click', function(e) {
        Titanium.Media.showCamera({

            success : function(event) {
                var selectedImg= event.media;
                if(event.mediaType == Ti.Media.MEDIA_TYPE_PHOTO) {
                    var imgView = Titanium.UI.createImageView({
                        left: 10,width: 250,height: 250,image:selectedImg  
               });
               win.add(imgView);
              }
           }
       });
  });
  opengallery.show();
};

  // Add to parent

win.add(Btn);

win.open();

    Thanks
