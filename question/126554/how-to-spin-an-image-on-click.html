  • Spin the image

    //
//
var win1 = Titanium.UI.createWindow({
    title : 'Tab 1',
    backgroundColor : '#fff',
    layout : "vertical"
});

//
var view5 = Titanium.UI.createView({
    backgroundColor : '#336699',
    top : 60,
    height : 200,
    width : 200,
    anchorPoint : {
        x : 0.5,
        y : 0.5
    }
});
win1.add(view5);

var spin = Titanium.UI.createButton({
    title : 'spin',
    height : 40,
    width : 200,
    left : (320 - 200) / 2,
    top : 100
});

var rotate = function(degrees) {
    var t = Ti.UI.create2DMatrix();
    t = t.rotate(degrees);

    var a = Titanium.UI.createAnimation();
    a.transform = t;
    a.duration = 5;
    view5.animate(a);

    setTimeout(function() {
        rotate(++degrees);
    }, 5);
};

spin.addEventListener('click', function() {
    rotate(0);
});
win1.add(spin);
win1.open();
