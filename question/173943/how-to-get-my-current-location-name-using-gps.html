How to get my current location name using GPS

Hi,
i want to show my current location name where i am now. i can get my current coordinate(latitude, longitude) but not the current location name. is it possible to get the exact location name where i am staying now ? i tried like below - 

function GeoLocation() {
    Titanium.Geolocation.accuracy = Titanium.Geolocation.ACCURACY_BEST;
    Titanium.Geolocation.distanceFilter = 10;
    Titanium.Geolocation.getCurrentPosition(function(e) {
        if (e.error) {
            alert('HFL cannot get your current location');
            return;
        } else {
            var lon = e.coords.longitude;
            var lat = e.coords.latitude;
            Ti.API.info(lon + '   ' + lat);
            // i am getting here latitude & longitude, but i want the location name of this latitude and longitude
        }
    });
}

module.exports = GeoLocation;
