  • Hi

    Here is another approach, it lets you have logic on a row by row basis. You can build out the logic to suit.

    Simple example, to test it properly - create a new default project and replace the contents of app.js with the code below.

    Ti.UI.setBackgroundColor('#fff');

var win1 = Ti.UI.createWindow({
    backgroundColor: '#FFF',
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});

function rowUpdate(obj) {
    obj.color
}
function addRow(obj) {
    var strColor = obj.color || '';
    var row = Ti.UI.createTableViewRow({
        backgroundColor: '#fff',
        data: obj,
        height: Ti.UI.SIZE,
        width: Ti.UI.FILL
    });
    var lblTitle = Ti.UI.createLabel({
        color: '#000',
        font: {
            fontSize: 16
        },
        height: Ti.UI.SIZE,
        left: 10,
        text: obj.title || '',
        touchEnabled: false,
        width: Ti.UI.SIZE
    });
    row.add(lblTitle);
    var lblColor = Ti.UI.createLabel({
        color: '#666',
        font: {
            fontSize: 16
        },
        height: Ti.UI.SIZE,
        right: 10,
        text: strColor,
        textAlign: 'right',
        touchEnabled: false,
        width: 120
    });
    row.add(lblColor);
    Ti.App.addEventListener('rowupdate', function (e) {
        if (e.oldColor === strColor) {
            lblColor.setText(e.newColor);
        }
    });
    return row;
}

var data = [
    { id: 1, title: 'One', color: 'green' },
    { id: 2, title: 'Two', color: 'blue' },
    { id: 3, title: 'Three', color: 'yellow' },
    { id: 4, title: 'Four', color: 'blue' },
    { id: 5, title: 'Five', color: 'yellow' },
    { id: 6, title: 'Six', color: 'green' }
];
var intRow = 0, intRows = data.length, row, rows = [];
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    row = addRow(data[intRow]);
    rows.push(row);
}
var tbl = Ti.UI.createTableView({
    data: rows,
    height: Ti.UI.FILL,
    minRowHeight: 44,
    width: Ti.UI.FILL
});
tbl.addEventListener('click', function (e) {
    Ti.App.fireEvent('rowupdate', {
        oldColor: 'yellow',
        newColor: 'red'
    });
});
win1.add(tbl);

win1.open();

    I am just using labels but you can do this with anything you can add into a row.
