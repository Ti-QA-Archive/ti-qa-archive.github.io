  • Ok, here's some code.

    Titanium.UI.setBackgroundColor('#000');
var tabGroup = Titanium.UI.createTabGroup();

// Tab 1
var win1 = Titanium.UI.createWindow({
    url : 'win1.js'
});
var tab1 = Titanium.UI.createTab({
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});

var win1 = Titanium.UI.createWindow({
    title:'Tab 1',
    backgroundColor:'#fff',
    layout : 'vertical'
});
var tab1 = Titanium.UI.createTab({
    icon:'KS_nav_views.png',
    title:'Text Box 1',
    window:win1
});
// Add validation function
tab1.validateForm = function() {
    if(!txt1.value || txt1.value == '') {
        alert('Fill out Text Box 1');
        return false;
    }
    return true;
}
var lbl1 = Ti.UI.createLabel({
    text : 'Tab 1'
});
win1.add(lbl1)
var txt1 = Ti.UI.createTextField({
    width : 200
});
win1.add(txt1);

// Tab 2
var win2 = Titanium.UI.createWindow({
    title:'Tab 2',
    backgroundColor:'#fff',
    layout : 'vertical'
});
var tab2 = Titanium.UI.createTab({
    icon:'KS_nav_ui.png',
    title:'Tab 2',
    window:win2
});
var lbl2 = Ti.UI.createLabel({
    text : 'Tab 2'
});
// Add validation function
tab2.validateForm = function() {
    if(!txt2.value || txt2.value == '') {
        alert('Fill out Text Box 2');
        return false;
    }
    return true;
}
win2.add(lbl2)
var txt2 = Ti.UI.createTextField({
    width : 200
});
win2.add(txt2);

tabGroup.addTab(tab1);
tabGroup.addTab(tab2);

// Validation
tabGroup.addEventListener('focus', function(e) {
    if( !e.previousTab.validateForm() ) {
        tabGroup.setActiveTab(e.previousTab);
    }
});
tabGroup.open();
