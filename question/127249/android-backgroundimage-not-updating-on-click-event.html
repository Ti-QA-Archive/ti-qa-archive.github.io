Android backgroundImage Not updating on click event

I'm using the following code to change a button background when it is clicked (tried buttons, views and image view). The code works in iOS simulator, but does not in Android emulator.

This was tested on sdk 1.7.3 and the 1.8.0.v20111021103013

var rightButton = Titanium.UI.createImageView({
    width : 24,
    height : 24,
    top : 10,
    right : 5,
    backgroundImage : ( editMode ? 'images/remove.png' : 'images/next.png' ),
    bookid : libraryBook.bookid
});
tvrow.add(rightButton);

rightButton.addEventListener('click', function(e) {
    Ti.API.info('rightButton event fired');
    if(editMode) {
        var data = {
            bookid : e.source.bookid,
            row : rowIndex
        }
        if(rightButton.backgroundImage == 'images/remove.png') {
            Ti.API.info('remove');
            rightButton.backgroundImage = 'images/accept.png';
            data.remove = true;
        } else {
            Ti.API.info('dont remove');
            rightButton.backgroundImage = 'images/remove.png';
            data.remove = false;
        }
        Ti.App.fireEvent('libraryBookDelete', data);
    } else {
        rightButton.backgroundImage = 'images/next.png';
        trackClickHandler(e.source.bookid);
    }
});

This is an exported module (part of an export.function).
I've tried using 

rightButton.backgroundImage = '/images/accept.png';

and

rightButton.backgroundImage = '../images/accept.png';

but I still have the same problem.

Any ideas what's wrong?
