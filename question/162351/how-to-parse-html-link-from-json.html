  • Hello Amaresh!

    You will need to use regexp to get the URL out of the server's reply. There is a nice article about regexp here, check the first answer. With that, you will be getting something like this:

    Code

    var win = Ti.UI.createWindow({
    backgroundColor:'white',
});

var text = '<a href="http://google.com">';

function extractURL(text) {
    //var urlRegex = /(http?:\/\/[^\s]+)/g;
    var urlRegex = /(http|ftp|https):\/\/[\w-]+(\.[\w-]+)+([\w.,@?^=%&amp;:\/~+#-]*[\w@?^=%&amp;\/~+#-])?/;
    var URL = text.toString().match(urlRegex);

    if (URL !== null) {
        alert( URL[0] );
        Ti.API.info( URL );

    } else {
        alert('No URL found');
        return;
    }

}

win.open();

extractURL(text);

    How to use the example

    1. Create a new mobile project (classic) titanium
    2. Paste the code sample into app.js
    3. Save it and run it.

    Let us know how it goes!
