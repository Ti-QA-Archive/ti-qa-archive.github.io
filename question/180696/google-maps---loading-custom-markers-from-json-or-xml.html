    Hi Marko,

    yes it is possible,

    win.js

    for (var i = 0, len = arr.length; i < len; i++) {  //arr isList of coordinate which is comes from web service
    var annotation = Alloy.Globals.Map.createAnnotation({
        latitude : arr[i].latitude,
        longitude : arr[i].longitude,
        title : arr[i].Title,        //Title of annotation pop up
        subtitle : arr[i].Address, //subTitle of annotation pop up
        animate : true,
        image : "/images/yellowPin.png",
        // rightButton : "/images/rightArrow.png", // Right image of annotation pop up
        doc : arr[i],    //Custom Property for particular annotation
    });
    arrAnnotation.push(annotation);
}
$.mapView.addAnnotations(arrAnnotation);

    win.xml

    <Alloy>
    <Window>
        <View id="mapView" ns="Alloy.Globals.Map"></View>
    </Window>
</Alloy>

    alloy.js

    Alloy.Globals.Map = require('ti.map');
