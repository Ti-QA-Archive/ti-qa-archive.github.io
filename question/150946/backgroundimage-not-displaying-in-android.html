backgroundImage not displaying in Android

I know this is a very common topic, but so far I have not been able to figure out why it will not show my background images in Android. It works perfect in iOS.

This is the code I am using (app.js file):

var tabAmount = 4;

var tab1 = Ti.UI.createView({
    width: Ti.Platform.displayCaps.platformWidth / tabAmount - 1,
    height: 50,
    bottom: 0,
    left: 0,
    backgroundImage: 'tabBar.png',
    backgroundRepeat: true
});
var tab2 = Ti.UI.createView({
    width: Ti.Platform.displayCaps.platformWidth / tabAmount - 1,
    height: 50,
    bottom: 0,
    left: Ti.Platform.displayCaps.platformWidth / tabAmount,
    backgroundImage: 'tabBar.png',
    backgroundRepeat: true
});

var tab3 = Ti.UI.createView({
    width: Ti.Platform.displayCaps.platformWidth / tabAmount - 1,
    height: 50,
    bottom: 0,
    left: (Ti.Platform.displayCaps.platformWidth / tabAmount) * 2,
    backgroundImage: 'tabBar.png',
    backgroundRepeat: true
});

var tab4 = Ti.UI.createView({
    width: Ti.Platform.displayCaps.platformWidth / tabAmount,
    height: 50,
    bottom: 0,
    left: (Ti.Platform.displayCaps.platformWidth / tabAmount) * 3,
    backgroundImage: 'tabBar.png',
    backgroundRepeat: true
});

var tabLabel1 = Ti.UI.createLabel({
    text: 'Tab 1',
    color: '#fff',
    font: {
        fontSize: 12
    },
    top: 30
});

var tabLabel2 = Ti.UI.createLabel({
    text: 'Tab 2',
    color: '#fff',
    font: {
        fontSize: 12
    },
    top: 30
});

var tabLabel3 = Ti.UI.createLabel({
    text: 'Tab 3',
    color: '#fff',
    font: {
        fontSize: 12
    },
    top: 30
});

var tabLabel4 = Ti.UI.createLabel({
    text: 'Tab 4',
    color: '#fff',
    font: {
        fontSize: 12
    },
    top: 30
});

var tabContainer = Ti.UI.createView({
    width: Ti.Platform.displayCaps.platformWidth,
    height: 51,
    left: 0,
    bottom: 0,
    backgroundColor: '#000'
});

var main = Ti.UI.createWindow({
    backgroundColor: '#fff'
});

tab1.add(tabLabel1);
tab2.add(tabLabel2);
tab3.add(tabLabel3);
tab4.add(tabLabel4);

tabContainer.add(tab1);
tabContainer.add(tab2);
tabContainer.add(tab3);
tabContainer.add(tab4);

main.add(tabContainer);
main.open();

I am trying to make a custom set of tabs at the bottom of the page as a navigation, but the .png image I am trying to use for the background of each tab won't display in android.

I am currently using: Titanium SDK: 3.0.2.GA and Google APIs Android 2.2 [armeabi] and I also tried it in Google APIs Android 4.2.2 [armeabi-v7a] but no luck.

I have also tried changing the URL of the image to:

backgroundImage: '/tabBar.png'

I know the backgroundImage is linking to the correct place (Image in in Resources folder), because it displays fine on iOS.

Any help would be greatly appreciated.
