    • var win1 = Ti.UI.createWindow({backgroundColor:'#ffffff'});
var tableview = Ti.UI.createTableView();
var data = [];
for(var i=0;i<=20;i++){
    data.push(createRow('Row ' + i));
}
tableview.setData(data);
tableview.addEventListener('click', function(e){
    alert('You chose ' + e.rowData.title);
});

win1.add(tableview);
win1.open()

function createRow(title){
    var tvr = Ti.UI.createTableViewRow({title:title});
    //comment next two lines for it to work correctly.
    var vw = Ti.UI.createView();
    tvr.add(vw);
    return tvr;
}
