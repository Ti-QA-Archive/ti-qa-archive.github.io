    Try this:

    var right_button_view = Ti.UI.createView({
     width:80,
     height:45,
     backgroundColor:'red',
     borderRadius:6
});

var right_button_label = Ti.UI.createLabel({
    text:'A BUTTON',
    color:'white',
    width:80,
    height:14,
    textAlign:'center',
    font:{fontSize:14,fontWeight:'bold'},
    touchEnabled:false
});

right_button_view.add(right_button_label);


right_button_view.addEventListener('touchstart',function(){
     right_button_view.backgroundColor = 'blue';
});
right_button_view.addEventListener('touchend',function(){
     right_button_view.backgroundColor = 'red';
});

right_button_view.addEventListener('click',function(){
    Ti.API.info('the fake right button was clicked in the navbar');
});


// now set this to be the right nav button

currentWin.setRightNavButton(right_button_view);

    Don't use a Ti button - create a 'fake' button using views / labels etc.

    This should help ;)

    This is not a full solution for a badge you need, but you can always create a label and add/attach it to the view

    eg: Make a 20 x 20 size label, with a red background and a 10 pixel borderRadius, which will make it round to start with. The text of the label could be the 'badge' value.
