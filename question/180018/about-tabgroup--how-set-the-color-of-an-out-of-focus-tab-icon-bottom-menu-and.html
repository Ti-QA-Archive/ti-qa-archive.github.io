  • Dear Xavier,

    The simplest way to customize the tabgroup for IOS, just add a view on the tabgroup it overlaps the tabgroup and customize the view as per your requirement.

    You can use below code to achieve this.

    1.Create a javascript file ApplicationTabGroup.js and put below code in that file.also create five windows HomeWin.js, ControlWin.js, ShareWin.js, AboutWin.js, LocWin.js and put all the files in same folder. 

    function ApplicationTabGroup() {
    var self = Ti.UI.createTabGroup({
            backgroundColor: '#000'
        }),
        BaseUIWindow = require('HomeWin'),
        ControlsWindow = require('ControlWin'),
        PhoneWindow = require('ShareWin'),
        PlatformWindow = require('AboutWin'),
        MashupsWindow = require('LocWin');

    var baseUIWin = new BaseUIWindow(),
        controlsWin = new ControlsWindow(),
        phoneWin = new PhoneWindow(),
        platformWin = new PlatformWindow(),
        mashupsWin = new MashupsWindow();

    var baseUITab = Ti.UI.createTab({
        icon: 'home.png',
        window: baseUIWin
    });
    baseUIWin.containingTab = baseUITab;
    self.addTab(baseUITab);

    var controlsTab = Ti.UI.createTab({
        icon: 'home.png',
        window: controlsWin
    });
    controlsWin.containingTab = controlsTab;
    self.addTab(controlsTab);

    var phoneTab = Ti.UI.createTab({
        icon: 'home.png',
        window: phoneWin
    });
    phoneWin.containingTab = phoneTab;
    self.addTab(phoneTab);

    var platformTab = Ti.UI.createTab({
        icon: 'home.png',
        window: platformWin
    });
    platformWin.containingTab = platformTab;
    self.addTab(platformTab);

    var mashupsTab = Ti.UI.createTab({
        icon: 'home.png',
        window: mashupsWin
    });
    mashupsWin.containingTab = mashupsTab;
    self.addTab(mashupsTab);
    //this is view which needs to be add on tabgroup, you can change it to any color,height and width.
    var fakeTabs = Ti.UI.createView({
        backgroundColor: '#000',
        bottom: 0,
        height: 50,
        layout: 'horizontal',
        left: 0,
        width: Ti.UI.FILL
    });
    self.add(fakeTabs);
    var homeButton = Ti.UI.createButton({
        width: 64,
        height: 50,
        backgroundImage: 'home.png',
        backgroundSelectedImage: 'home_active.png',
        zIndex: 1
    });
    fakeTabs.add(homeButton);
    var alarmbutton = Ti.UI.createButton({
        width: 64,
        height: 50,
        backgroundImage: 'home.png',
        backgroundSelectedImage: 'home_active.png',
        zIndex: 1
    });
    fakeTabs.add(alarmbutton);
    var arrowButton = Ti.UI.createButton({
        width: 64,
        height: 50,
        backgroundImage: 'home.png',
        backgroundSelectedImage: 'home_active.png',
        zIndex: 1
    });
    fakeTabs.add(arrowButton);

    var sbutton = Ti.UI.createButton({
        width: 64,
        height: 50,
        backgroundImage: 'home.png',
        backgroundSelectedImage: 'home_active.png',
        zIndex: 1
    });
    fakeTabs.add(sbutton);
    var locbutton = Ti.UI.createButton({
        width: 64,
        height: 50,
        backgroundImage: 'home.png',
        backgroundSelectedImage: 'home_active.png',
        zIndex: 1
    });
    fakeTabs.add(locbutton);
    //focus first tab as a selected tab
    self.setActiveTab(0);
    homeButton.backgroundImage = 'home_active.png';
    //capture first tab event
    homeButton.addEventListener('touchstart', function(e) {

    });
    //capture second tab event
    alarmbutton.addEventListener('touchstart', function(e) {});
    //capture third tab event
    arrowButton.addEventListener('touchstart', function(e) {});
    ////capture fourth tab event
    sbutton.addEventListener('touchstart', function(e) {});
    //capture fifth tab event
    locbutton.addEventListener('touchstart', function(e) {});
    return self;
};
module.exports = ApplicationTabGroup;

    2.execute below code from your app.js

    //this will open the tabgroup with 5 windows.
var ApplicationTabGroup = require('ApplicationTabGroup');
var theTabGroup = new ApplicationTabGroup();
theTabGroup.open({
    transition: Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
});
