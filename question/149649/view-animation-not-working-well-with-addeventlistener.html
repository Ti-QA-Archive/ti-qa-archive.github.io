View animation not working well with AddEventListener?

I am a newbie to appcelerator,

By running this code, i am expecting it will keep switching the view, image1 -> image2 -> image1 -> image2 -> ….

But when run it on ipad simulator, it only switch from image1-> image2 -> image1 then no longer respond to the click.

Below are the code, did i do something wrong?

//FirstView Component Constructor
var BookView = function() {
    var win = Ti.UI.createWindow();

    view = Ti.UI.createView({
        backgroundColor:'#fff'
    });
    win.add(view);

    var image1 = Titanium.UI.createView({
        backgroundImage:'/images/smallpic1.jpg',
        height:75,
        width:75,
        borderWidth:0,
        borderColor:'#777'
    });

    var image2 = Titanium.UI.createView({
        backgroundImage:'/images/smallpic2.jpg',
        height:75,
        width:75,
        borderWidth:0,
        borderColor:'#777'
    });

    image1.addEventListener('click', function()
    {
        view.animate({view:image2,transition:Ti.UI.iPhone.AnimationStyle.CURL_UP});
    });

    image2.addEventListener('click', function()
    {
        view.animate({view:image1,transition:Ti.UI.iPhone.AnimationStyle.CURL_UP});
    });


    view.add(image1);
    view.add(image2);

    return win;
};

module.exports = BookView;
