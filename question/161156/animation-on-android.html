function moveMe(){

 matrix2d = matrix2d.scale(1.5); // scale to 1.5 times original size
var a = Ti.UI.createAnimation({
    transform: matrix2d,
    duration: 1000,
    autoreverse: true,

$.loadingImgView.animate(a);

setTimeout(function(){
