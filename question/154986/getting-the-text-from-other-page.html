  • hi,

    you need to use click event of button and on that click create new win and pass button title to that win.

    
button.addEventListener('click',function(e){

    var winDetail = Ti.UI.createWindow({
        backgroundColor : '#fff'    
    });

    var lbl = Ti.UI.createLabel({
        width : Ti.UI.SIZE,
        height : Ti.UI.SIZE,
        text : e.source.title
    });

    winDetail.add(lbl);

    winDetail.open({
        modal : true
    });
});

      var winDetail = Ti.UI.createWindow({

      var win = Ti.UI.currentWindow;
