  • The problem was to fire an event from iframe. The solution in iframe:

    var ctrl = {};  // in top windwow
ctrl.fireEvent2Tit() = function{
    Ti.App.fireEvent('popuphasnewdatas',data);
}

if (window.top != self) {  // in iframe
    ctrl.fireEvent2Tit(data);
}
