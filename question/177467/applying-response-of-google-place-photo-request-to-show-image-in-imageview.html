  • Ok figured it out. I needed to set the request to the backgroungImage or the image property itself. 

        var imageV = Ti.UI.createView({
        backgroundImage :'https://maps.googleapis.com/maps/api/place/photo?maxwidth=400&photoreference=CnRnAAAAyO6idDlnmed9KQWB7lPbksg_cRHu7bpewAkyIPlfRgxt9aHxvIFLJyqr82GvT9Luo8fs-V2sP5GrK1nBf4HJw9S4926wJ5QbO24ZI9cdiaOM2WSJ2VBv2TYsBYjXDmkx73oOJiZpkYziRpEVcZ_UMRIQuQENnkVxNBCB1tEV6Q253RoUEJMmx6ADo9peNhkGxmS-4WAQ53U&key=MYKEY'
    });
