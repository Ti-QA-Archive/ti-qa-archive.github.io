  • My improved version:

    module.exports = function() {
    var table = Ti.UI.createTableView({
        height : "auto", // better is Ti.UI.SIZE
        width : "30%",
        color : "gray",
        left : '0'
    });
    var arr = ["Home", "features", "Plans and pricing", "About us", "Contact us"];
    var data = [];
    for (var i = 0; i < 5; i++) {
        var row = Ti.UI.createTableViewRow();
        row.add(Ti.UI.createLabel({
            left : '2',
            text : "hihi"
        }));
        data.push(row);
    }
    table.setData(data);
};

    $.menu.addEventListener('click',require('myfile'));

    This is your issue:

    table.add(x.add(lable));

    You cannot cascade this adding.
