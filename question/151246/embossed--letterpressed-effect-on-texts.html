  • Here is what i did to get a LetterPressed(not very good) effect, until now.
    If someone find a better way to do this, please let me know.

    ps: I'm using alloy.

    ".MainHeaderTitle": {
    color:"#8b1f22",
    text:"TEXT",
    textAlign:Ti.UI.TEXT_ALIGNMENT_CENTER,
    font:{
        fontWeight:'bold',
        fontSize:'20px'    
    },            
    top:0,
    height:50,
    width: Ti.UI.FILL    
}

"#mainHeaderTitleDrop": {
    shadowColor:'#dd595a',
    shadowOffset:{x:'0.5', y:'0.5'},
    opacity:0.7        
}

"#mainHeaderTitleInner": {
    shadowColor:'#6e0e11',
    shadowOffset:{x:'-0.5', y:'-0.5'},
    opacity:0.8        
}

  • What if you did something like …

    function bezelMe(l) {
        l.zIndex   = 1;
        var outter = l;
            outter.zIndex = 2;
        var inner  = l;
            inner.zIndex = 3;

        outter.shadowColor    = '#dd595a';
        outter.shadowOffset    = {x:'0.5', y:'0.5'};
        outter.opacity        = 0.7;

        inner.shadowColor    = '#6e0e11';
        inner.shadowOffset    = {x:'-0.5', y:'-0.5'};
        inner.opacity        = 0.8;

        self.add(outter);
        self.add(inner);
}

var self = Ti.UI.createWindow({backgroundColor:'#FFF'});

var label = Ti.UI.createLabel({
        text   : 'BEZEL',
        font   : {fontSize:48,fontWeight:'bold'},
        height : 'auto',
        width  : 'auto'
    });

    self.add(label);

    // Add the effect to the label.
    bezelMe(label);

return self;
