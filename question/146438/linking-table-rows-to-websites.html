    • and one more thing on the click event of tableview you can show webview with the selected link.

      like in your case

      
tableview.addEventListener('click',function(e){
    alert(e.rowData.link);
    //here you can set this link to your webview 

    var winWeb = Ti.UI.createWindow();

    var webView = Ti.UI.createWebView({
        url : e.rowData.link
    });

    winWeb.add(webView);

    winWeb.open({
        modal : true
    });
});
