function wz_rp_favlist(){
    var self=Ti.UI.createView(myUI.rp_list);
    var common_function = require('/model/common');
    var tb_name;
    self.add(require('/ui/header').wz_header('????',false));
    tb_name = 'tb_history';

    var list1 = Ti.UI.createTableView(myUI.list1);//??

    var btnL = Ti.UI.createButton(myUI.btn_l);
    var btnM = Ti.UI.createButton(myUI.btn_m);
    var btnR = Ti.UI.createButton(myUI.btn_r);
    btnL.setLeft((Ti.Platform.displayCaps.platformWidth-common_function.wz_2px(240))/2);
    btnM.setLeft(btnL.left+common_function.wz_2px(80)+1);
    btnR.setLeft(btnM.left+common_function.wz_2px(80)+1);

    btnL.addEventListener('click',function(){
        if (list1.data.length==0){return false};
        f_refresh(true);
    });
    btnM.addEventListener('click',function(){
        if (list1.data.length==0){return false};
        f_refresh(false);
    });
    btnR.addEventListener('click',function(){
        if (list1.data.length==0){return false};
        for(var i=0;i<list1.data[0].rows.length;i++){
            if(list1.data[0].rows[i].children[1].value){
                Ti.App.myDBL.execute('delete from '+tb_name+' where id='+list1.data[0].rows[i].children[1].v_id);
            };
        };
        f_refresh(false);
    })

    self.add(list1);
    self.add(btnL);self.add(btnM);self.add(btnR);

    self.addEventListener('refreshList',function(){
        f_refresh(false);
    });
    f_refresh(false);

    return self;

    function f_refresh(_checked){
        var data=[];list1.data=[];
        var rs_id='',t_id='';
        var rows=Ti.App.myDBL.execute('select * from '+tb_name+' order by id desc');
        while (rows.isValidRow()){
            if (rows.fieldByName('c')==0){//c?0?RS?1?T
                rs_id = rows.fieldByName('d');t_id = '';
            }else{
                t_id = rows.fieldByName('d');rs_id = '';
            };
            var tt=f_makeitemP(rs_id,t_id,rows.fieldByName('a')+'  ?'+rows.fieldByName('b')+'?',rows.fieldByName('id'),_checked);
            data.push(tt);
            rows.next();
        };
        list1.data = data; data=[];
    };
    function f_makeitemP(p_rs_id,p_t_id,p_rymc,p_id,p_checked){
        var self = Ti.UI.createTableViewRow(myUI.list_item);
        self.v_rs_id=p_rs_id;
        self.v_t_id=p_t_id;
        var tt_label=Ti.UI.createLabel(myUI.list_item_label);
        tt_label.text=p_rymc;
        var tt_check = Titanium.UI.createSwitch(myUI.list_item_check);
        tt_check.v_id=p_id;
        tt_check.value = p_checked;
        self.add(tt_label);self.add(tt_check);
        self.addEventListener('click',f_clickitemP);
        return self;
    };
    function f_clickitemP(e){
        if(e.source.v_tag==1){return false};
        self.fireEvent('showDetail',{rs_id:this.v_rs_id,t_id:this.v_t_id})
    };

}
