  • You could set the default to be opaque in your tiapp.xml, then on the window you want it translucent you can add the following code:

    Titanium.UI.iPhone.statusBarStyle = Titanium.UI.iPhone.StatusBar.TRANSLUCENT_BLACK;

  • This code makes the statusbar and a toolbar transclucent (the trick is to set top:-20 (for the statusbar) and bottom:-49 (for the toolbar). 

    Titanium.UI.iPhone.statusBarStyle = Titanium.UI.iPhone.StatusBar.TRANSLUCENT_BLACK;

var win = Titanium.UI.createWindow({  
    title:'test',
    top:-20,
    bottom:-49,
    backgroundImage:'Abstract.jpg',
});

var send = Titanium.UI.createButton({
    title: 'Send',
    style: Titanium.UI.iPhone.SystemButtonStyle.DONE,
});

var cancel = Titanium.UI.createButton({
    systemButton: Titanium.UI.iPhone.SystemButton.CANCEL
});

var flexSpace = Titanium.UI.createButton({
    systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

var toolbar = Titanium.UI.iOS.createToolbar({
    items:[flexSpace, send, cancel, flexSpace],
    bottom:49,
    borderTop:true,
    borderBottom:false,
    translucent:true,
    barColor:'#000',
    opacity:0.9,
}); 
win.add(toolbar);

win.open();
