  • Simply set the event's source yourself when you fire the event.

    ...
myComponent.fireEvent('click', { source: whatever_you_want_to_pass });

  • Hi Alexander,

    exports.create = function(options) {
    var self = Ti.UI.createView();
    self.mylabel = Ti.UI.createLabel();
    self.add(self.mylabel);
    self.mylabel.addEventListener('click',function(){
        self.fireEvent('clicked');
    })
    return self;
}
