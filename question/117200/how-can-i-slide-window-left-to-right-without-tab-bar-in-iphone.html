  • // New window (homeWin) open at 320 px and move to 0
homeWin.left = 320
var slide_it_left = Titanium.UI.createAnimation();
slide_it_left.left = 0;
slide_it_left.duration = 300;
homeWin.open(slide_it_left);

// Existing window move from 0 to -320 / -640
var slide_it_right = Titanium.UI.createAnimation();
slide_it_right.left = -320; // or -640
slide_it_right.duration = 300;
win.animate(slide_it_right); // Existing showing window.
