    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

var b1 = Ti.UI.createButton({
    top : 10,
    width : 200,
    height : 50,
    backgroundColor : 'green'
});

var b2 = Ti.UI.createButton({
    top : 200,
    width : 200,
    height : 50,
    backgroundColor : 'red'
});

b1.addEventListener('click', function() {

    b2.backgroundColor = 'yellow';
});
win.add(b1);
win.add(b2);
win.open();

    works for me

var Using_ME = false;
var Using_XWRIS = false;

var CommentButton = Ti.UI.createButton({
    title : 'CommentButton',
    height : 'auto',
    width : 'auto',
    top : 50,
    left : 50
});

// Listen for click events.
CommentButton.addEventListener('click', function() {
    Using_ME = true;
    colorChange();

});

// Add to the parent view.
win.add(CommentButton);

var ME_Button = Ti.UI.createButton({
    title : 'ME_Button',
    height : 'auto',
    width : 'auto',
    top : 250,
    left : 50
});

// Listen for click events.
ME_Button.addEventListener('click', function() {
    Using_XWRIS = true;
    colorChange();
});

// Add to the parent view.
win.add(ME_Button);

function colorChange() {

    if (Using_ME == true) {
        Using_ME = false;
        Using_XWRIS = true;
        ME_Button.backgroundColor = "#80FF80";
    } else {
        CommentButton.backgroundColor = "#FF8080";
        Using_XWRIS = false;
        Using_ME = true;
    }
}

win.open();
