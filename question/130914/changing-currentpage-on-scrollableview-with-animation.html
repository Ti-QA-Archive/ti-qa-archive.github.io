  • Try following:

    var view1 = Ti.UI.createView({
 height: '100%',
 width: '100%'
});
var view2 = Ti.UI.createView({
 height: '100%',
 width: '100%'
});
var view3 = Ti.UI.createView({
 height: '100%',
 width: '100%'
});
var scrollView = Ti.UI.createScrollableView({
 views: [view1, view2, view3],
 showPagingControl: ture
});
randomPage = Math.floor(Math.random()*3); //Will get random value between 0 to 2
scrollView.scrollToView(scrollView.views[randomPage]);
