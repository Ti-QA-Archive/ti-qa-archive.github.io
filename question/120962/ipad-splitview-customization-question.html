iPad SplitView Customization Question

MainWindows.navigation = Ti.UI.createWindow({
    url:'master_sched.js',
    backgroundImage:'../images/side.jpg',
    width: 300,
    borderWidth: 0
});
MainWindows.userFeed = Ti.UI.createWindow({
    backgroundImage: '../images/dirt.jpg',
