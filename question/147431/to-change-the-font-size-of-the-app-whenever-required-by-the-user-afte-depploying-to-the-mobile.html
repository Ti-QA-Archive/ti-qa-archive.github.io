  • For every label you use a module:

    exports.create = function(_text,_color,_font) {
    var self = Ti.UI.createLabel({text:_text,color:_color:font:_font});
    T.App.addEventlistener('app:changefontsize',function(ratio){
        var font = self.getFont();
        font.fontSize *= ratio;
        self.setFont(font);
    })
    return self;
}
