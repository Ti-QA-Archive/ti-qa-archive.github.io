  • Hi Alan,

    You also can calculate value of screen size and set object at desired position like below example where 'value' is the screen height for idel condition. like 320x480,

    function GetHeight(value) {
    var temp = (value * 100) / 480;
    return parseInt((screenHeight * temp) / 100);
}

    also you can set value for width also. This is the basic one.

    • Try this one in all screen resolution.

      var win = Ti.UI.createWindow({
    title : 'Screen Resolution'
});

var screenWidth =  Ti.Platform.displayCaps.platformWidth;
var screenHeight = Ti.Platform.displayCaps.platformHeight;

var lblTop = Ti.UI.createLabel({text : "Top Label", top : GetHeight(20), left : GetWidth(20),height : 25, width : 150});
var lblBottom = Ti.UI.createLabel({text : "Bottom Label", bottom : GetHeight(20), right : GetWidth(20),height : 25, width : 150});


function GetHeight(value) {
    var temp = (value * 100) / 480;
    return parseInt((screenHeight * temp) / 100);
}

function GetWidth(value) {
    var temp = (value * 100) / 320;
    return parseInt((screenWidth * temp) / 100);
}

win.add(lblTop);
win.add(lblBottom);

Ti.API.info("Screen Size :" + screenWidth + "x" + screenHeight);
Ti.API.info("Values of top :" + lblTop.top + " " + lblTop.left);
Ti.API.info("Values of Bottom :" + lblBottom.top + " " + lblBottom.right);

win.open();
