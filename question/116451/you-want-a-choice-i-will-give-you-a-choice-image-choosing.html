How would I go about giving my users the option to selecting a picture from there photo gallery OR taking one on there camera before they upload it?

        var dialog = Titanium.UI.createOptionDialog({
        options: ['Take Photo or Video','Choose Existing', 'Cancel'],
        cancel:2
    });

    dialog.addEventListener('click', function(e) {
        if (e.index == 0) {
            picFromCamera();
        } else if (e.index == 1) {
            picFromGallery();
        }
    });
    dialog.show();
