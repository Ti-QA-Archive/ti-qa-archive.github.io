Hello,

I'm kinda new to titanium so I don't know how to do the following..
I want to create a vertical scrollView with set of views, for every view I want to drag it horizontally

If I dragged it to left, then the view will be deleted, and replaced by the next view.

and If I clicked it for now an alert will appear.

I tried to use Ti.draggable but views didn't work properly inside scrollView + Ti.draggable views for some reason are not clickable, I don't know why!!.

So, I tried to make it myself, but I couldn't even drag view inside scrollView properly..


var scrView = Ti.UI.createScrollView({
    backgroundColor : 'transparent',
    height : Ti.UI.SIZE,
    contentHeight : "auto",
    contentWidth : Ti.UI.FILL,
    layout : "vertical",
    showHorizontalScrollIndicator : false,
    showVerticalScrollIndicator : true,
    scrollType : "vertical",
    width : Ti.UI.FILL,
});
var selected = {
    x : -1,
    //y : -1
};

for (var i = 0; i < 6; i++) {
    var view = Ti.UI.createView({
        width : Ti.UI.FILL,
        layout : 'vertical',
        height : 120,
        top:5,
        backgroundColor:"red"
    });

    view.addEventListener('click', function(e) {
        alert("here");
    });
    var txtLLbl = Ti.UI.createLabel({
        left :  10,
        top : 10,
        width :  "50%",
        height : "50%",
        text : i,
        textAlign : 'center',
        font : {
            fontSize : 50
        },
        color : 'white',
        backgroundColor : '#fff'
    });

    view.addEventListener('touchstart', function(e) {
        selected.x = e.x;
        //selected.y = e.y;
    });

    view.addEventListener('touchmove', function(e) {
            e.source.left += e.x - selected.x;
            //e.source.top += e.y - selected.y;
            selected.x = e.x;
            //selected.y = e.y;
    });

    view.addEventListener('touchend', function(e) {
        selected.x = -1;
        //selected.y = -1;
    });
    view.add(txtLLbl);
    scrView.add(view);
}

$.main.add(scrView);

$.index.open();

So please if someone can help me with this, thanks :)
