Facebook login even called twice using authorize()

Hello.

I am trying to login with authorize() method for facebook and adding event listener for login event in facebook. but the event login called twice.

Titanium.Facebook.appid = Titanium.Locale.getString('Facebook_AppID');
Titanium.Facebook.permissions = ['publish_stream'];
Titanium.Facebook.forceDialogAuth = true;

Titanium.Facebook.addEventListener('login', function(e) {
    alert('called a ' + e.uid + ' ' + e.type + ' ' + e.success + ' ' + e.source + ' ' + e.error + ' ' + e.data + ' ' + e.bubbles + ' ' + e.cancelBubble);
});

Titanium.Facebook.addEventListener('logout', function(e) {
    alert('called a ' + e.uid + ' ' + e.type + ' ' + e.success + ' ' + e.source + ' ' + e.error + ' ' + e.data + ' ' + e.bubbles + ' ' + e.cancelBubble);
});

function recommandFacebook() {
    alert('h');
    Titanium.Facebook.authorize();
}

while executing above code the alert 'h' called once then it showing facebook login. Once the login is successful. The 'login' even is called twice.

      function facebook() {
    Titanium.Facebook.appid = Titanium.Locale.getString('fb_id');
    Titanium.Facebook.permissions = ['publish_stream'];
    Titanium.Facebook.forceDialogAuth = true;
    if (Ti.Facebook.loggedIn) {
        //Already Logged in
    } else {
        Titanium.Facebook.addEventListener('login', function(e) {
            if (e.success) {
                //Success
            } else if (e.error) {
                //Error
            }
        });
        //call the facebook authorize method to login
        Titanium.Facebook.authorize();
    }
}
