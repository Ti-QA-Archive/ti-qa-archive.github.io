  • Hi

    A quick example, even in Alloy this is done in actual code.

    // assumes $.nav is associated with main window this attached to
$.nav.activity.onCreateOptionsMenu = function (e) {
    var actionbar = $.nav.activity.actionBar;
    actionBar.backgroundImage = '/images/actionbar.png';
    actionBar.displayHomeAsUp = false;
    actionBar.title = 'Title';
    actionBar.onHomeIconItemSelected = function () { 
        alert('hello');
    };
    var menu = e.menu;
    var menuFav = menu.add({
        title: 'menu',
        icon: '/images/something.png',
        showAsAction: Ti.Android.SHOW_AS_ACTION_IF_ROOM
    });
    menuFav.addEventListener("click", function (e) {
        alert('world')
    });
};
