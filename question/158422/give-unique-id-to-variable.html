Give unique id to variable

I am trying to give a unique id to a variable every time an image view is clicked on. I then want to take that same id and reference it to play a certain video based on the id for the array I am giving.

Here is my code. I am having no luck getting the first imageview of videoOne to work. Any help is appreciated.

var vid;

videoOne.addEventListener('click', function(e) {
    vid.id = 0;
    Ti.App.fireEvent('videoClick',{
        x:e.id
    });
});


Ti.App.addEventListener('videoClick', function(e) {

    var videos = new Array();
    videos[0] = "Lighting.mp4";
    videos[1] = "Mechanical-Drive-Transmission.mp4";
    videos[2] = "Cab.mp4";
    videos[3] = "Flex-Frame.mp4";
    videos[4] = "Traction.mp4";
    videos[5] = "Booms.mp4";
    videos[6] = "Fill-Station.mp4";
    videos[7] = "Crop-Clearance.mp4";

    var videoPlayer = Titanium.Media.createVideoPlayer({
        url: "/videos/" +videos[vid.id],
        width:2560,
        height:1600,
        autoplay:true,
        keepScreenOn:true,
        mediaControlStyle:Titanium.Media.VIDEO_CONTROL_NONE,
        scalingMode:Titanium.Media.VIDEO_SCALING_ASPECT_FIT
    });

    var vidClose = Ti.UI.createButton({
         title:"Stop Video",
         bottom:250,
         right:50,
         height:75,
         width:400,
         backgroundColor:'#E93D42',
         color:'#ffffff',
         backgroundImage:'none',
         borderRadius:7,
         borderColor:'#000000',
         zIndex: 999
    });

    vidClose.addEventListener('click',function(){
        videoPlayer.hide();
        videoPlayer.release();
         vidPlay.close();
    });

    var vidPlay = Ti.UI.createWindow({
        width:2560,
        height:1600
    });

    vidPlay.open({
        modal:true,
        modalTransitionStyle:Ti.UI.iPhone.MODAL_TRANSITION_STYLE_PARTIAL_CURL,
        modalStyle:Ti.UI.iPhone.MODAL_PRESENTATION_PAGESHEET
    });

    vidPlay.add(vidClose);
});
