Change imageView image animated

I have a imageView with a backgroundColor and an image.

So far I can change the backgroundColor animated

imageView.animate({
    duration: 500,
    backgroundColor: '#000'
});

I thought, I could do the same with the image parameter

imageView.animate({
    duration: 500,
    image: '/images/myNewImage.png'
});

But this won't work. Also I have found no examples for this. I don't want an animated image sequence. Just exchange the image

  • My solution:

    Using this commonJS module: animator.js

    var self = Ti.UI.createWindow();

var image1 = Ti.UI.createImage({
    image: FIRSTURL
});
self.add(image1);

var image2 = Ti.UI.createImage({
    opacity:0,
    image: FIRSTURL
});
self.add(image2);

var Animator = require('animator');

self.addEventListener('changeImage', function(e){
    image2.image = e.image;
    new Animator().fade({
        view: image2,
        value: 1,
        onComplete: function () {
            image1.image = e.image;
            image2.opacity = 0;
        }
    });
});
