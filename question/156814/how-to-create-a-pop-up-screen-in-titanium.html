    Hi ravi sharma,

    Just you can follow this code and simply run this in app.js file, i think will be helpful for you.

    var win = Ti.UI.createWindow({
    backgroundColor : 'white',
    layout : 'vertical'
});

// Create a Button.
var btn = Ti.UI.createButton({
    title : 'Show Window',
    height : 60,
    width : 100,
    top : 20,
    left : 50
});
win.add(btn);
// Listen for click events.
btn.addEventListener('click', function() {
    var infoWin = Titanium.UI.createWindow({
        layout : 'vertical',
        height : Titanium.UI.SIZE,
        width : Titanium.UI.SIZE
    });

// Create a Label.
var name = Ti.UI.createLabel({
    text : 'name',
    color : '#000',
    font : {fontSize:15},
    height :Titanium.UI.SIZE,
    width : Titanium.UI.SIZE,
    top : 10,
    left : 10,
    textAlign : 'center'
});

// Add to the parent view.
infoWin.add(name);

    // Create a TextField.
    var userName = Ti.UI.createTextField({
        height : Titanium.UI.SIZE,
        top : 10,
        width : Titanium.UI.FILL,
        hintText : 'Add your user name',
        softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_DEFAULT_ON_FOCUS, // Android only
        keyboardType : Ti.UI.KEYBOARD_DEFAULT,
        returnKeyType : Ti.UI.RETURNKEY_DEFAULT,
        borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED
    });

    infoWin.add(userName);

    // Create a Button.
    var save = Ti.UI.createButton({
        title : 'save',
        height : 60,
        width : 100,
        top : 10,
        right : 10
    });

    // Listen for click events.
    save.addEventListener('click', function() {
        infoWin.close();
    });

    var cancel = Ti.UI.createButton({
        title : 'cancel',
        height : 60,
        width : 100,
        top : 10,
        right : 10
    });

    cancel.addEventListener('click', function() {
        infoWin.close();
    });

    // Add to the parent view.
    infoWin.add(save);
    infoWin.add(cancel);

    infoWin.open({
        modal : true
    });

});
win.open();
// Add to the parent view.

  • Hi Ravi,

    1. Create a modalWindow on a click of button.
    2. Make the size of the modal window(pop over) as you want.
    3. Open the modal window(pop over) with animation.
    4. Create one label and one text field and two button cancel or ok and add them to the modal window.
    5. Add the listeners to the cancel and ok button:
      button_cancel.addEventListener('click', function(e){
       modal_win.close();
      });

    • Here is a small example f what you required. you need to add and modify elements accordingly:

      //Creating main window
var win = Titanium.UI.createWindow({
    backgroundColor : 'white'
});

var button = Titanium.UI.createButton({
    title : 'Hello',
    top : 10,
    width : 200,
    height : 150,
    font : {
        fontSize : '50'
    },
});

win.add(button);
win.open();

// creating a popup
var view = Titanium.UI.createView({
    backgroundColor : 'red',
    top : 200,
    left : 100,
    width : 500,
    height : 300,
    visible : false
});
win.add(view);
button.addEventListener('click', function() {

    view.visible = 'true';
    var button_cancel = Titanium.UI.createButton({
        title : 'Cancel',
        bottom : 20
    });
    view.add(button_cancel);
    button_cancel.addEventListener('click', function() {
        view.visible = 'false';
    });

});
