  • Hello Isaac ,

    Try below code. i set 3 view change below code as you need.

    var win = Ti.UI.createWindow({
    backgroundColor : "white"
});

var scrollView = Ti.UI.createScrollView({
    top : "20dp",  //  for iOS7  or  0 for other
    bottom : "0dp",
    width : "100%",
    layout : "vertical"
});

for ( var i = 0; i < 10; i++) {
    var viewContain;
    if (i % 3 == 0 || i == 0) {
        viewContain = Ti.UI.createView({
            top : '5dp',
            left : '0dp',
            height : Ti.UI.SIZE,
            right : '0dp',
            layout : 'horizontal',
            backgroundColor : 'red'
        });
        var view = Ti.UI.createView({
            backgroundColor : 'gray',
            top : '0',
            left : '10dp',
            height : '93dp',
            width : '95dp',
            layout : "vertical"
        });
        var icnImage = Ti.UI.createImageView({
            top : '10dp',
            height : Ti.UI.SIZE,
            width : Ti.UI.SIZE,
            image : 'appicon.png'
        });
        var lblTitle = Ti.UI.createLabel({
            text : "Title",
            top : '5dp',
            font : {
                fontSize : '10dp'
            }
        });
        view.add(icnImage);
        view.add(lblTitle);
        viewContain.add(view);
        scrollView.add(viewContain);
    } else {
        var view = Ti.UI.createView({
            backgroundColor : 'gray',
            top : '0',
            left : '10dp',
            height : '93dp',
            width : '95dp',
            layout : "vertical"
        });
        var icnImage = Ti.UI.createImageView({
            top : '10dp',
            height : Ti.UI.SIZE,
            width : Ti.UI.SIZE,
            image : 'appicon.png'
        });
        var lblTitle = Ti.UI.createLabel({
            text : "Title",
            top : '5dp',
            font : {
                fontSize : '10dp'
            }
        });
        view.add(icnImage);
        view.add(lblTitle);
        viewContain.add(view);
    }
}
win.add(scrollView);
win.open();

    i hope this helps. thanks.

    • ###see below code with addEventListener for all view.

      var win = Ti.UI.createWindow({
    backgroundColor : "white"
});

var scrollView = Ti.UI.createScrollView({
    top : "20dp",  //  for iOS7  or  0 for other
    bottom : "0dp",
    width : "100%",
    layout : "vertical"
});

for ( var i = 0; i < 20; i++) {
    var viewContain;
    if (i % 4 == 0 || i == 0) {
        viewContain = Ti.UI.createView({
            top : '5dp',
            left : '0dp',
            height : Ti.UI.SIZE,
            right : '0dp',
            layout : 'horizontal',
        //    backgroundColor : 'red'
        });
        var view = Ti.UI.createView({
        //    backgroundColor : 'gray',
            top : '0',
            left : '4%',
            height : Ti.UI.SIZE,
            width : '20%',
            layout : "vertical"
        });
        var icnImage = Ti.UI.createImageView({
            top : '10dp',
            height : Ti.UI.SIZE,
            width : Ti.UI.SIZE,
            image : 'appicon.png',
            touchEnabled : false
        });
        var lblTitle = Ti.UI.createLabel({
            text : "Title" + i,
            top : '5dp',
            font : {
                fontSize : '10dp'
            },
            color : "black",
            touchEnabled : false
        });
        view.lblTitle = lblTitle;
        view.add(icnImage);
        view.add(lblTitle);
        viewContain.add(view);
        view.addEventListener("click",function(e){
            alert(e.source.lblTitle.text);
        });
        scrollView.add(viewContain);
    } else {
        var view = Ti.UI.createView({
        //    backgroundColor : 'gray',
            top : '0',
            left : '4%',
            height : Ti.UI.SIZE,
            width : '20%',
            layout : "vertical",
        });
        var icnImage = Ti.UI.createImageView({
            top : '10dp',
            height : Ti.UI.SIZE,
            width : Ti.UI.SIZE,
            image : 'appicon.png',
            touchEnabled : false
        });
        var lblTitle = Ti.UI.createLabel({
            text : "Title" + i,
            top : '5dp',
            font : {
                fontSize : '10dp'
            },
            color : "black",
            touchEnabled : false
        });

        view.lblTitle = lblTitle;

        view.add(icnImage);
        view.add(lblTitle);
        viewContain.add(view);

        view.addEventListener("click",function(e){
            alert(e.source.lblTitle.text);
        });
    }
}
win.add(scrollView);
win.open();
