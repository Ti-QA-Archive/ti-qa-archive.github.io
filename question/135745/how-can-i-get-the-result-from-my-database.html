  • var db = Ti.Database.install('../project.sqlite', 'project');
var query = db.execute('SELECT * FROM student');

var names = [];

// Loop through ResultSet
while(query.isValidRow()) {
     // Use query.fieldByName(FieldName) or query.field(FieldIndex)
    names.push( query.fieldByName('name') );
    query.next();
}

// Close ResultSet & DB
query.close();
db.close();
