  • I am not sure if Titanium provides an animation for the slider, but if not I would use a quick timer to animate the slider:

    
var slider_target_pos = 123;

// SET THE INTERVAL TO 100 EM OF SECOND
var interval = setInterval(function(){

    //CHECK IF THE VALUE IS DIFFERENT THAN THE TARGET VALUE
    if(mySlider.value != slider_target_pos)
    {
        // CHECK IF WE NEED TO GO DOWN OR UP WITH THE SLIDER
         if(mySlider.value < slider_target_pos)
         { 
            //ADD 
            ++ mySlider.value
          }
         else
         {
            // SUBSTRACT
            -- mySlider.value;
         }
    }
    // VALUES ARE EQUAL SO WE CLEAR THE INTERVAL
    else
    {
        clearInterval(interval);
    }
}, 100);
