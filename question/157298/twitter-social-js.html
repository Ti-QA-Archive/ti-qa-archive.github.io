  • Social.js has been developed to help share on social networks like twitter and linkedin.

    The only methods returned are:

    • isAuthorized
    • deauthorize
    • authorize
    • share

    If you really want to reuse socialjs, you should add new methods to the adapter.

      var authorization = 'Bearer ' + "XXXXXX" //XXXXXX - Twitter bearer token
var url = "https://api.twitter.com/1.1/statuses/user_timeline.json?screen_name=XXXXXXXXX";
var xml = Ti.Network.createHTTPClient({
    onload : function(e) {
        json = JSON.parse(this.responseText);

    },
    onerror : function(e) {
        //error
    },
    timeout : 50000 /* in milliseconds */
});
xml.open("GET", url);
xml.setRequestHeader("Authorization", authorization);
xml.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
xml.send();
