var backgroundImage = Ti.UI.createImageView({
    image:"images/backgroundImage.png",

    top: 0
});

var logo = Ti.UI.createImageView({
    image:"images/logo.png",

    right: 20

var data=[
    {img:"images/123.png",  dest:"aboutUs.js"},
    {img:"images/news.png",  dest:"news.js"},
    {img:"images/upload.png",  dest:"share.js"}
        ];

        var data2=[];
        for(var i=0; i<data.length; i++){

        var buttons=Ti.UI.createImageView({
            image: data[i].img,
            left:20
        });

            top:5
        });
