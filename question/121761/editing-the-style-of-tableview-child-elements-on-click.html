Editing the style of tableView child elements on click

How would I go about editing the style of views/labels within a tableViewRow when the editing option of the tableView is set to true?

I would have thought something like this would work but it doesn't:

data = [];

this_row = Ti.UI.createTableViewRow({
    width:320,
    height:50,
    selectedBackgroundColor:'transparent'
});

row_view = Ti.UI.createView({
    height:42,
    width:'100%',
    backgroundColor:'#e1e1e1',
    backgroundGradient:{
        type:'linear',
        colors:[{color:'#f1f1f1',position:0.0},{color:'#d1d1d1',position:1.0}]
    },
    borderRadius:6,
    top:4,
    width:300
});

this_row.add(row_view);
data.push(this_row)

tableView = Ti.UI.createTableView({
    data:data,
    backgroundColor:'transparent',
    separatorColor:'transparent',
    width:320
});

win.add(tableView);

var editButton = Ti.UI.createButton({title:'Edit'});

win.rightNavButton = editButton;

editButton.addEventListener('click',function(e){
    tableView.editing = true;
    row_view.width = '80%';
    row_view.left = 40;
});

I've also tried something like this:

editButton.addEventListener('click',function(e){
    tableView.editing = true;
    tableView.children[0].width = '80%';
    tableView.children[0].left = 40;
});

  • Try animating it:

    editButton.addEventListener('click', function() {
    tableView.editing = true;
    row_view.animate({
        width:'80%',
        left:40,
        duration:1000
    });
});
