Controls fall under the Navigation Bar after rotation

Has anyone come across this issue and have a solution or does this need to be reported as a bug?

Example: Gist

var wnd = Ti.UI.createWindow({
    title: "Title Bar",
    modal: true,
    backgroundColor: "#FFF", 
    orientationModes: [
        Titanium.UI.PORTRAIT,
        Titanium.UI.UPSIDE_PORTRAIT,
        Titanium.UI.LANDSCAPE_LEFT,
        Titanium.UI.LANDSCAPE_RIGHT
    ]
});
var view = Ti.UI.createView({backgroundColor: "#F00", height: 200});
var view2 = Ti.UI.createView({backgroundColor: "#00F", height: 216});
var text = Ti.UI.createTextArea({backgroundColor: "#0F0", suppressReturn: false});


view.add(text);
wnd.add(view);
wnd.add(view2);
wnd.open();

Ti.Gesture.addEventListener("orientationchange", function (event) {
    view.height = 200;
    view2.height = 216;
    if (event.orientation === Ti.UI.LANDSCAPE_LEFT || event.orientation === Ti.UI.LANDSCAPE_RIGHT)
    {
        view.height = 106;
        view2.height = 162;
    }
});

When the iPhone simulator is rotated, in the example above, the first view containing the textarea, is pushed under the windows navbar.

  • After your comment on the vertical layout, I am also seeing the behavior you describe. I've filed this JIRA ticket to track the bug. My suggestion to work around in this case is to avoid using hard coded heights/widths, use percentages, and dock the two elements of the view to the top and bottom of the window. This seems like it might get close to what you're looking for, and would eliminate the need (potentially) for orientation change handling.

    var wnd = Ti.UI.createWindow({
    title: "Title Bar",
    modal: true,
    backgroundColor: "#FFF", 
    orientationModes: [
        Titanium.UI.PORTRAIT,
        Titanium.UI.UPSIDE_PORTRAIT,
        Titanium.UI.LANDSCAPE_LEFT,
        Titanium.UI.LANDSCAPE_RIGHT
    ]
});
var view = Ti.UI.createView({backgroundColor: "blue", height: '50%', top:0});
var view2 = Ti.UI.createView({backgroundColor: "red", height: '50%', bottom:0});
var text = Ti.UI.createTextArea({backgroundColor: "green", suppressReturn: false});


view.add(text);
wnd.add(view);
wnd.add(view2);
wnd.open();

  • Until this bug is fixed, the solution is to offset the view by the height of the navigation bar. 44 for portrait and 32 for landscape.

    function portrait(event)
{
    return event.orientation == Ti.UI.PORTRAIT || event.orientation == Ti.UI.UPSIDE_PORTRAIT;
}

var wnd = Ti.UI.createWindow({
    title: "Title Bar",
    backgroundColor: "#FFF",
    modal: true
});

var viewA = Ti.UI.createView({backgroundColor: "#F00", top: 0, height: 200});
var viewB = Ti.UI.createView({backgroundColor: "#00F", bottom: 0, height: 216});
var text = Ti.UI.createTextArea({backgroundColor: "#FAF", value: "This is some text", top: 0});

viewA.add(text);
wnd.add(viewA);
wnd.add(viewB);

wnd.open();

Ti.Gesture.addEventListener("orientationchange", function (event) {
    // landscape
    viewA.top = 32;
    viewA.height = 106;
    viewB.height = 162;

    //portrait
    if (portrait(event))
    {
        viewA.top = 44; // offset the view
        viewA.height = 200;
        viewB.height = 216;
    }
});
