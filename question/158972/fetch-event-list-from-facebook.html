    This code represents the facebook event list with some important properties 

    var fbmod2=require('facebook');
fbmod2.appid = '416805508435668';
fbmod2.permissions = ['read_stream','user_photos','friends_photos','friends_status','user_events'];

fbmod2.requestWithGraphPath('me/events', {fields:'picture.width.height,name,start_time'}, 'GET', function(e) {
    if (e.success) {

        var evs=JSON.parse(e.result);
        for(var a=0;a<evs.data.length;a++)
        {

            var evrow=Ti.UI.createTableViewRow({
                layout:'horizontal',
                evid:evs.data[a].id
            });

            var evimage=Ti.UI.createImageView({
                image:evs.data[a].picture.data.url,
                width:60,
                height:60
            });

            var evview=Ti.UI.createView({
                layout:'vertical',
                width:210
            });

            var evlevel=Ti.UI.createLabel({
                text:evs.data[a].name,
                left:10
            });
            evview.add(evlevel);
            var evDate=Ti.UI.createLabel({
                text:evs.data[a].start_time,
                left:10
            });
            evview.add(evDate);
            evrow.add(evimage);
            evrow.add(evview);
            eventListTable.appendRow(evrow);

        }
    } else if (e.error) {
        alert(e.error);
    } else {
        alert('Unknown response');
    }
});
