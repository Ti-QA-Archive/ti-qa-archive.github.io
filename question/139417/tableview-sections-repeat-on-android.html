Tableview sections repeat on Android

Titanium Studio 2.0.2
Android 2.3.3 (on device)

I expect the sections to have a header labeled "Test title 0", "Test title 1", "Test title 2", etc. in sequential order.

Instead the labels are randomly "Test title 0" and "Test title 1" (updates each time they get redrawed after scrolling)

How can i prevent this, so the tiles can be shown correctly?

Sample code below:

var self = Ti.UI.createWindow();
var view = Ti.UI.createView();
self.add(view);

var table = Ti.UI.createTableView();
view.add(table);

var sections = [];

for (var i = 0; i < 10; i++)
{
    var header = Ti.UI.createView({
        height : 'auto',
        width : '100%',
        backgroundColor: '#FFFFFF'
    });

    var titleLabel = Ti.UI.createLabel({
        font : {
            fontSize : 20,
            fontWeight : 'bold'
        },
        text : "Test title " + i,
        top : 0,
        left : 10,
        width : '100%',
        height : 20
    });
    header.add(titleLabel);

    var descriptionLabel = Ti.UI.createLabel({
        font : {
            fontSize : 11,
        },
        text : "Description " + i,
        left : 10,
        top : 20,
        width : '100%',
        height: 'auto'
    });
    header.add(descriptionLabel);

    var section = Ti.UI.createTableViewSection({
        headerView: header,
    });

    for (var ii = 0; ii < 10; ii++)
    {
        section.add(Ti.UI.createTableViewRow({
            height : 38,
            className : 'simpleRow',
        }));
    }

    sections.push(section);
}

table.setData(sections);

self.open();
