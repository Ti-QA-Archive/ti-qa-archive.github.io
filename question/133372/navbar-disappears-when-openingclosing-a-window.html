  • if you simply use open method to open window it does not works so try to modal the window.. try this one

    var myWindow = Titanium.UI.createWindow({
    id : 'myWindowID',
    title : 'test',
    backgroundColor : '#FFFFFF',
    barColor : '#414444',
    navBarHidden : false

    // fullscreen: false
});

myWindow.open({
    modal : true
});

myWindow.close();
setTimeout(function(e) {
    myWindow.open({
        modal : true
    });
}, 1000);
