  • Hello Raju,
    Not sure whether it will help you or not, i used below code in my apps to display the loader.user can't perform any action if it's displayed.

    //create a file and name it activityInd.js, put below code inside that file
/**
 *  @Description function checks is device platform android or not
 *  @Param none
 *  @Return true if android platform
 */
function isAndroid() {
    return Ti.Platform.osname == 'android';
}

/**
 *  @Description function create activity indicator and also show/hide the activity indicator
 *  @Param none
 *  @Return activity indicator object
 */
exports.activityIndicator = function() {
    var container;

    if (isAndroid()) {

        container = Ti.UI.createView({
            width : '100%',
            height : '100%',
            top : 0,
            zIndex : 100000,
            left : 0,
            visible : false
        });

        container.add(Ti.UI.createView({
            width : '100%',
            height : '100%',
            zIndex : 100000,
            top : 0,
            left : 0,
            backgroundColor : '#000',
            opacity : 0.5,
            zIndex : 0
        }));

        var ai = Ti.UI.createActivityIndicator({
            style : Ti.UI.ActivityIndicatorStyle.DARK,
            color : '#fff',
            zIndex : 100000
        });
        container.add(ai);
        ai.show();
        ai = null;
    } else {
        container = Ti.UI.createView({
            width : '100%',
            height : '100%',
            top : 0,
            zIndex : 100000,
            left : 0,
            visible : false
        });

        container.add(Ti.UI.createView({
            width : '100%',
            height : '100%',
            zIndex : 100000,
            top : 0,
            left : 0,
            backgroundColor : '#000',
            opacity : 0.5,
            zIndex : 0
        }));

        var ai = Ti.UI.createActivityIndicator({
            style : Ti.UI.iPhone.ActivityIndicatorStyle.BIG,
            color : '#fff',
            zIndex : 100000
        });
        container.add(ai);
        ai.show();
        ai = null;
    }

    container._isShowing = false;

    container._show = function(params) {
        if (this._isShowing) {
            return;
        }


            this.children[1].message = params.message;


        if (params.timeout) {
            this._myTimeout = setTimeout(function() {
                exports.manager.get('ai')._hide();

                if (params.timeoutMessage) {
                    var alertDialog = Ti.UI.createAlertDialog({
                        title : 'Update Timeout',
                        message : params.timeoutMessage,
                        buttonNames : ['OK']
                    });
                    alertDialog.show();
                }
            }, params.timeout);
        }

        this._isShowing = true;
        this.show();
    };

    container._hide = function() {
        if (this._myTimeout !== undefined) {
            clearTimeout(this._myTimeout);
            delete this._myTimeout;
        }
        if (this._isShowing) {
            this._isShowing = false;
            this.hide();
        }
    };

    return container;
}; 

//activityInd.js code ends here

/*
use below code in any window to show/hide the loader
*/

var ai = require('activityInd').activityIndicator();
//add this to window object.
win.add(ai);
//use below method to show the loader
ai._show({message : 'Loading...'});

//use below code to hide the loader
ai._hide();
