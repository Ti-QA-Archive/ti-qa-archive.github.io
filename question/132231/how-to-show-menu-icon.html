  • you can't just say icon: something.

    you need to use the "setIcon" function like this:

        Ti.Ui.currentWindow.activity.onOptionsMenuCreate = function(e){
    var menu = e.menu;
    var item = menu.add({title: 'HOME1'});
    item.setIcon('url-to-png');
    };
