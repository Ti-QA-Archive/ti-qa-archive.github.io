  • i got it using this function

    function getLong(x,zoom) {
    var longitude = ((x * 360) / (256 * Math.pow(2, zoom))) - 180;
    while (longitude > 180.0)
    {
        longitude -= 360.0;
    }
    while (longitude < -180.0)
    {
        longitude += 360.0;
    }
    return longitude;

}

function getLat(y,zoom) {
    efactor = Math.exp((0.5 - y / 256 / Math.pow(2, zoom)) * 4 * Math.PI);
    latitude = Math.asin((efactor - 1) / (efactor + 1)) * 180 / Math.PI;
    if (latitude < -90.0) {
        latitude = -90.0;
    } else if (latitude > 90.0) {
        latitude = 90.0;
    }
    return latitude;
}
