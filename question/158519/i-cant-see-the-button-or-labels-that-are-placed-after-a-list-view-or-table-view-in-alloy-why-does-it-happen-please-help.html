    Hi Jinto,

    try following code.

    
<View class="v_container" bottom="20" top="10">
    <ListView id="listview" class="v_container" defaultItemTemplate="template" height="Ti.UI.SIZE">
        <Templates>
            <ItemTemplate name="template">
                <View width="Ti.UI.FILL" height="Ti.UI.SIZE" layout="vertical">
                    <ImageView bindId="image" id="image" height="40px"/>
                    <Label bindId="desc" id="desc" width="Ti.UI.FILL" left="2%" right="2%"/>
                </View>
            </ItemTemplate>
        </Templates>
        <ListSection>
            <ListItem desc:text="Label1" image:image="/KS_nav_views.png" />
            <ListItem desc:text="Label1" image:image="/KS_nav_ui.png" />
        </ListSection>
    </ListView>
</View>
<Button title="submit" width="TI.UI.SIZE" left="1%" right="2%" onclick="submit" top="20" bottom="20" zIndex="100"/>
