  • follow this tutorial to add a custom font to your project, then add the fontWeight property to your label

    var label1 = Titanium.UI.createLabel({
   color:'#000',
   text:'I am Window 1',
   font:{
      fontSize:40,
      fontFamily: customFont
      fontWeight:"bold"
   },
   textAlign:'center',
   width:'auto'
});
