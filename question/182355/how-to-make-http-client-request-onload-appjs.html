var xhr = Ti.Network.createHTTPClient();
xhr.onload = function  () {

  var res = JSON.parse(this.responseText);
  var collectionData = [];

  _.each(res,function(item,index){
      var model = Alloy.createModel('fugitives', item);
        collectionData.push(model);
        Alloy.Collections.fugitives.add(model);

  });
  Alloy.Collections.fugitives.add(collectionData);
};
xhr.open('GET', 'http://bountyhunterapp.appspot.com/bounties');
xhr.send();
Alloy.Collections.fugitives.fetch();

    It will load data from server to your app in a collection
