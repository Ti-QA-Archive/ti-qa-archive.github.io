    • You can draw a circle or a square like this … 

      var circle = Ti.UI.createImageView({
        width: 100,
        height: 100,
           borderRadius: 50,
     borderColor: '#1a0033',
    backgroundColor:'#296EC1'
});

var square = Ti.UI.createView({
    top:180,
      backgroundColor:'#296EC1',
    height:50,
      width:400
