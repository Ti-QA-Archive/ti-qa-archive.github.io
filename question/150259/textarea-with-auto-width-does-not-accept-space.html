TextArea with auto width does not accept space

I need a text area with automatic sizes. Unfortunately it is not possible to append a space to the current value. Another strangeness: if I insert the space into the current value, then I can append too. 

var win = Ti.UI.createWindow({
    backgroundColor: 'white',
    fullscreen: true,
});

var textArea = Ti.UI.createTextArea({
    top: 20,
    font: {fontSize: 30},
    height: Ti.UI.SIZE,
    width: Ti.UI.SIZE,
    suppressReturn: false,
    scrollable: false,
    value: 'my name is',
    borderWidth: 1,
    autocorrect: false,
})

win.add(textArea);

win.open();

target: iOS, Titanium SDK 3.0.2
