  • var view1=Ti.UI.createView({});
    var view2=Ti.UI.createView({backgroundColor:'#81F79F'});
    var view3=Ti.UI.createView({backgroundColor:'#9AFE2E'});

    var scroll=Ti.UI.createScrollableView({
    views:[view1,view2,view3],
    showPagingControl:true,
    pagingControlTimeout:5000
    });
    win1.add(scroll);
