  • Hi Eliza,
    Try out this to go back to root window.

    Root window :

    var Rootwin = Ti.UI.createWindow({
        url : 'next.js',
        _parent : Titanium.UI.currentWindow,
        navGroup : win.navGroup
    });

    Next window :

    var win = Ti.UI.currentWindow;
var Thirdwin = Ti.UI.createWindow({
        url : 'Third.js',
        _parent : Titanium.UI.currentWindow,
        navGroup : win.navGroup,
        rootWindow : win
    });

    Third Window :
    on button ClickEvent put this code. that will close all window and go to root window.

        win.navGroup.close(win.rootWindow);
    win.navGroup.close(win._parent);
    win.navGroup.close(win);
