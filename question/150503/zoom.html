  • Hello Devyani,

    Try below sample code : 

    //Zoom a view and dissipate it
function zoom(view, callback) {
    var matrix = Ti.UI.create2DMatrix({ 
        scale:1.5 
    });

    view.animate({ 
        transform:matrix, 
        opacity:0.0, 
        duration:250 
    }, function() {
        callback && callback();
    });
};
