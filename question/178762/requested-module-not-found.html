Requested module not found

In the main view of the application I have a view that - when clicked - directs to some other view. When my Samsung Galaxy s3 goes to sleep, and I wake it up after about 10 minutes, I get this exception:

Location: [280,9] ti:/module.js
Message: Uncaught Error: Requested module not found: alloy/controllers/null
Source: throw new Error("Requested module not found: " + request);

So what's the solution? My event handler for clicking the aforementioned view is like this:

$.profileButton.addEventListener("click", utils.debounce(function() {
    Alloy.createController("profile/profile").getView().open();
}));

debounce function is defined like that: 

exports.debounce = function (func, delayed, wait) {
    var timeout;
    var immediate = delayed === undefined;

    return function debounced () {
        var obj = this, args = arguments;

        function delayed () {
            if (!immediate) {
                func.apply(obj, args);
            }

            timeout = null; 
        };

        if (timeout) {
            clearTimeout(timeout);
        } else if (immediate) {
            func.apply(obj, args);
        }

        timeout = setTimeout(delayed, wait || 300); 
    };
};
