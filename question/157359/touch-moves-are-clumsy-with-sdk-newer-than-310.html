Touch moves are clumsy with SDK newer than 3.1.0

This is the sample code I used to move a view. With SDK 3.1.0 (and probably older) the movement is smooth. If I run the same code on a newer SDK (3.1.1, 3.1.2) the move is not fluid.

Did you noticed the same effect and found any workaround to work better even on the latest SDKs?

var win = Ti.UI.createWindow({fullscreen: true, backgroundColor: '#fff'});

var touchMoveBase = {
    set: function(point) {
        this.x = point.x;
        this.y = point.y;
    }
};

function touchStart(e) {
    var convPoint = e.source.convertPointToView({x: e.x, y: e.y}, win);
    touchMoveBase.set(convPoint);
}

function touchMove(e) {
    var convPoint = e.source.convertPointToView({x: e.x, y: e.y}, win);
    e.source.top += convPoint.y - touchMoveBase.y; 
    e.source.left += convPoint.x - touchMoveBase.x;

    e.source.animate({
        top: e.source.top, 
        left: e.source.left,
        duration: 1, 
    });

    touchMoveBase.set(convPoint);
}

var view = Ti.UI.createView({width: 80, height: 80, top: 600, left: 200, backgroundColor: 'yellow'});
win.add(view);

view.addEventListener('touchstart', touchStart);
view.addEventListener('touchmove', touchMove);

win.open();
