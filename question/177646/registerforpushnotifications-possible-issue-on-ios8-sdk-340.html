      Here's the Ti code I generally use

      function receivePush(e) {
    Ti.API.info(e);
    //Custom push actions
    //lib_push.doNextAction(e);
}

function deviceTokenSuccess(e) {
    Ti.API.info(e);
    var Token = e.deviceToken;
    //Internal token management
    //UpdatePushToken(Token);
    subscribeToChannel('alerts', Token);
    subscribeToChannel('geoinfo', Token);
}

function deviceTokenError(e) {
    Ti.API.info(e);
}

function subscribeToChannel(Channel, Token) {
    Cloud.PushNotifications.subscribeToken({
        device_token : Token,
        channel : Channel,
        type : Ti.Platform.name == 'android' ? 'gcm' : 'ios'
    }, function(e) {
        Ti.API.info(e);
    });
}

function registerPushNotifications(e) {
    Ti.App.iOS.registerUserNotificationSettings({
        types : [Ti.App.iOS.USER_NOTIFICATION_TYPE_ALERT, Ti.App.iOS.USER_NOTIFICATION_TYPE_SOUND, Ti.App.iOS.USER_NOTIFICATION_TYPE_BADGE]
    });
    Ti.Network.registerForPushNotifications({
        success : deviceTokenSuccess,
        error : deviceTokenError,
        callback : receivePush
    });
};
