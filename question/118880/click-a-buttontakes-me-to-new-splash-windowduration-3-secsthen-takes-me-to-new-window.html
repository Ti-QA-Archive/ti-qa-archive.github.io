    You can use a timeout function so after 3 seconds open a new window. See my quick example below

    //create current window
var win1 = Titanium.UI.createWindow({ backgroundColor:'#fff'});

// create button
var btn1= Titanium.UI.createButton({  
title:"Button1",
top:100, 
width: 200,
height: 50 
});
win1.add(btn1);

//create splash window
var win2 = Titanium.UI.createWindow({ 
modal:true,
backgroundColor:'000',
title:"splash" });

//create new window
var win3 = Titanium.UI.createWindow({
backgroundColor:'000fff',
title:"new window" 
});

//button EventListener
btn1.addEventListener("click", function(e){ 

    //open splash window
    win2.open();
    //timeout function for 3 seconds 
    setTimeout(function()
    {
     //open new window
     win3.open();
    },3000);    
});

win1.open();
