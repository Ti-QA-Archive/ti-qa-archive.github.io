  • you'll need to attach the click event to the annotations

    <View ns="Ti.Map" id="map"  >
        <Annotation onClick="doClick" title="Microsoft" latitude="37.47" longitude="-122.12" id="a1" pincolor="red"/>
        <Annotation onClick="doClick" id="a2" latitude="12.390749" longitude="-10" pincolor="green" title="yahoo" />
    </View>

    and check which annotation triggered the action

    function doClick(e) {

    if (e.source.id === 'a1')
        alert('Microsoft');
    else
        alert('Yahoo');
}
