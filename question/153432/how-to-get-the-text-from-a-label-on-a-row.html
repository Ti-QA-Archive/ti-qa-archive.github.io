How to get the text from a label on a row?

Hello i'm looking for a help. I'm getting some data from an XML and i'm creating a table with rows. when a user clicks on a row i want to send the data to another class. I have used static text and i saw that is working but i tries to send the data that has each row but is not working. can anyone help me?

self.add(table);
table.addEventListener('click', function(e) {
    //Here is the issue
    self.fireEvent('itemSelected', { title: e.row.titleview, image: "test", decription: "test" });
});

Here is my all of the code

var createRssRow = function(item) {
var tablerow = Ti.UI.createTableViewRow({
height: "auto",
link: item.link5,
backgroundColor: '#fff',
hasChild: true,
width: Ti.UI.FILL
});
var imageview = Ti.UI.createImageView({
image: item.image,
height: "42dp",
width: "68dp",
left: "15dp",
top: "15dp",
bottom: "10dp"
});
var titleview = Ti.UI.createLabel({
text: item.title,
height: Ti.UI.SIZE,
color: '#000',
top: "10dp",
bottom: "10dp",
font: {
fontSize: 14
},
left: "100dp",
right: "5dp"
});
var smalldescription_view = Ti.UI.createLabel({
text: item.description,
textAlign: 'center',
color: '#444',
font: {
fontSize: 12
},
top: "100dp",
left: "15dp",
right: "15dp",
bottom: "10dp"
});
tablerow.add(imageview);
tablerow.add(titleview);

return tablerow;

};

function View() {
var self = Ti.UI.createView({
backgroundColor:'#fff'
});

var table = Ti.UI.createTableView();
self.add(table);
table.addEventListener('click', function(e) {
    //Here is the issue
    self.fireEvent('itemSelected', { title: e.row.titleview, image: "test", decription: "test" });
});

self.refreshRssTable = function(data) {
    if (Object.prototype.toString.apply(data) === '[object Array]') {
        var rows = [];
        for (var i = 0; i < data.length; i++) {
            rows.push(createRssRow(data[i]));
        }
        table.setData(rows);
    }
};

return self;

}
