    Hello Chung ,

    Try Something like this :

    var arr = [];
    for(var i=1; i<=5;i++)
    {
        var view = Ti.UI.createView({
            bottom : 0,
            height : 200,
            width : 150
        });

    var label = Ti.UI.createLabel({
            top : 5,
            height : Ti.UI.SIZE,
            width : Ti.UI.SIZE,
            text : ('Logo' + i),     //   Your Display Text  
            color : 'red',
            zIndex : 5
        });

        // IF you Display logo  then use imageView instead of label 

        var img = Ti.UI.createImageView({
            height : '100%',
            width : '100%',
            image : '/images/item.jpeg'    // Remote URL
        });
        view.add(label);
        view.add(img);
        arr.push(view.toImage());
    }

    var myCarouselView = Ti.UI.iOS.createCoverFlowView({
        images : arr 
    });
