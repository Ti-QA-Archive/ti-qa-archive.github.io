How to create check box on off in property file

Hi Friends,

I created check box button in my tableview. if i click the check box the check box is "on" then store into the property file. whenever i see my list the check box is "on".

Please help me urgently needed:

Here below sample code:

checkBoxImageOff[i] = Ti.UI.createImageView({
            right : 05,
            id : i,
            name : 'checkOff',
            zIndex : 99,
            height : 40,
            width : 50,
            visible : true,
            touchEnabled : true,
            image : getImageResource('radio-unselected'),
        });

       checkBoxImageOn[i] = Ti.UI.createImageView({
            right : 05,
            id : i,
            name : 'checkOn',
            zIndex : 999,
            height : 40,
            width : 50,
            visible : false,
            touchEnabled : true,
            image : getImageResource('radio-selected'),

        });

        row.addEventListener('click', function(e) {

            if (checkedRow.length == 0) {
                checkBoxImageOn[e.source.id].setVisible(true);
                checkBoxImageOff[e.source.id].setVisible(false);
                checkedRow.push(e.source.id);
            } else {
                var isID = checkedRow[0];
                checkBoxImageOn[isID].setVisible(false);
                checkBoxImageOff[isID].setVisible(true);
                checkedRow.splice(0, 1);
                checkBoxImageOn[e.source.id].setVisible(true);
                checkBoxImageOff[e.source.id].setVisible(false);
                checkedRow.push(e.source.id);
            }
        });
