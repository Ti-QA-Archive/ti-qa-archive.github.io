    Yes you can do in same way 

    var myIndex = e.index;

if(myIndex == 0){
var win = Titanium.UI.createWindow({
            url: 'window.js',
            backgroundColor: "black",
            //modal: true,
        });
       win.open();

}else if(myIndex == 1){


}else if(myIndex == 2){

}
});

    There will be no problem at all.
