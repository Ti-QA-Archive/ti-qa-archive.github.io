  • Some Code in commonJS format:

    share.js

    exports.createmenu = function(event){
    var menu = event.menu;
    var menuGoogle = menu.add({title: 'Share on Google+'});
    var menuFacebook = menu.add({title: 'Share on Facebook'});
    var menuTwitter = menu.add({title: 'Share on Twitter'});
    var menuEmail = menu.add({title: 'Share via Email'});
}

    app.js

    var myWindow = Ti.UI.createWindow({title: ''Sweet Menus!',backgroundColor: '#262626', barColor: 'red', modal:false});
var menu = require('/share');
var activity = myWindow.activity;
activity.onCreateOptionsMenu = function(event) {
     menu.createmenu(event);
};

myWindow.open();
