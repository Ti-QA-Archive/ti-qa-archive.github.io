custom image picker refresh not working on tabs

Hi, i have a 2 column picker with custom images and want to refresh the second column with the information on the first column. To try, i do the following and works great on a simple window, but if i put the same code in a tab, when i move the first picker column, the second refreshes with a white space (no image is showed). Is this a bug? help please:

var win = Titanium.UI.createWindow();
win.backgroundColor = '#C8CFD5';

//-- Create the 1st column and set the data
var column1 = Titanium.UI.createPickerColumn();

var row = Ti.UI.createPickerRow();
var img = Ti.UI.createImageView({image:'images/logo.png',width:40,height:40});
row.add(img);
column1.addRow(row);

var row = Ti.UI.createPickerRow();
var img = Ti.UI.createImageView({image:'images/facescontact.png',width:40,height:40});
row.add(img);
column1.addRow(row);                        


//-- Create the 2nd column with nothing
var column2 = Titanium.UI.createPickerColumn();


//-- Create the picker
var picker = Titanium.UI.createPicker({
    top:20,
    selectionIndicator: true,
    columns: [column1, column2]
});

picker.addEventListener('change', function(e){
   Ti.API.log("Inside the picker"); 

    //-- Detect which column has fired the event
    if(e.columnIndex === 0)
    {
         column2.rows = null;

        var row = Ti.UI.createPickerRow();
        var img = Ti.UI.createImageView({image:'images/logo.png',width:40,height:40});
        row.add(img);
        column2.addRow(row);

        picker.reloadColumn(column2);

    }
});

win.add(picker);   

win.open();
