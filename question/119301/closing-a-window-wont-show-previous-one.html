closing a window won't show previous one

Hello all, I have this code:

mainMenu.js


var win = Titanium.UI.currentWindow;

var mainMenuView = Titanium.UI.createView({width:320,height:480});
win.add(mainMenuView);

var newGameButton = Titanium.UI.createButton({top:10,left:10,width:200,height:40,title:"Options"});
newGameButton.addEventListener("click", function(e) {
    var optionsWindow = Titanium.UI.createWindow({url:"optionsMenu.js"});
    optionsWindow.open({transition:Ti.UI.iPhone.AnimationStyle.CURL_UP});
})
mainMenuView.add(newGameButton);

optionsMenu.js


var win = Titanium.UI.currentWindow;

var optionsView = Titanium.UI.createView({width:320,height:480});
win.add(optionsView);

var backButton = Titanium.UI.createButton({width:200,height:40,title:"Back"});
backButton.addEventListener("click", function(e) {
    win.close({transition:Ti.UI.iPhone.AnimationStyle.CURL_DOWN});
})
optionsView.add(backButton);

clicking on "Back" in the options window does close itself, but the mainMenu window is not shown anymore…

how can I switch between windows easily?

thanks!
