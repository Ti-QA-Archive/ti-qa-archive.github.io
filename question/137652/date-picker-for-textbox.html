Here is my code:

var birthday = Ti.UI.createTextField({

    height:40,
    hintText:'birthday',
    editable:false,  
    borderStyle:Ti.UI.INPUT_BORDERSTYLE_ROUNDED

scrollView.add(birthday); 

birthday.addEventListener('click',function(e){
    var picker = Ti.UI.createPicker({

            birthday.setValue('' + " col1:" + picker.getSelectedRow(0)

        })

    var scrollView = Ti.UI.createScrollView({
    height : 'auto',

});
var birthday = Ti.UI.createTextField({
    color : '#336699',

scrollView.add(birthday);

birthday.addEventListener('click', function(e) {
    var picker = Ti.UI.createPicker({
        type : Ti.UI.PICKER_TYPE_DATE,
        //useSpinner for Android Only
        useSpinner : true,
        minDate : new Date(2009, 0, 1),
        maxDate : new Date(2014, 11, 31),
        value : new Date(2014, 3, 12),
        top : 70,

    var set = Ti.UI.createButton({
        title : 'Set',
        top : 240,
    })

    set.addEventListener('click', function(e) {
        birthday.setValue('' + " col1:" + picker.value );
    })

    win.add(picker);
    win.add(set);
}); 
win.add(scrollView);
