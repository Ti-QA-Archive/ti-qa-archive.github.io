Internal server error. The Cocoafish Team has just been notified of this problem.

Hi, I'm trying to search PhotoCollections via the Cloud services. But I always get the error message:

"Internal server error. The Cocoafish Team has just been notified of this problem."

This is my code:

Cloud.PhotoCollections.search({
        user_id: config.admin_user_id
    }, function (e) {
        if (e.success) {
          Ti.API.info('API::getOperationsKategorien -> success = ' + JSON.stringify(e));
            var operationen = [];
            for (var i = 0; i < e.collections.length; i++) {
                var collection = e.collections[i];
                Ti.API.info('API::getOperationsKategorien -> collection = ' + JSON.stringify(collection));
                if (typeof(collection.custom_fields) != "undefined" && typeof(collection.custom_fields.operationen) != "undefined" && collection.custom_fields.operationen==true)
                {
                  operationen.push(collection);
                }
            }
            return operationen;
        } else {
            Ti.API.error('Error:\\n' +
                ((e.error && e.message) || JSON.stringify(e)));
        }
    });
