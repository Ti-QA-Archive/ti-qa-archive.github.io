  • HI,amri hichem..For Facebook permissions use this code

    Ti.Facebook= require('facebook');
Ti.Facebook.appid = "FB_APP_ID";
Ti.Facebook.permissions = ['publish_stream']; // Permissions your app needs
Ti.Facebook.forceDialogAuth = true;

    Then Add

    var fbLoginBtn = Titanium.UI.createButton({
       title: '     Login With Facebook',
       width: '80%',
       height: '11%',
});
var flag=null;
fbLoginBtn.addEventListener('click',function(e)
{ 
        Ti.Facebook.addEventListener('login', function(e) {
            if (e.success) {
                flag= 1;
                //alert('Logged In');

            } else if (e.error) {
                flag= 0;
                //alert(e.error);

            } else if (e.cancelled) {
               // alert("Canceled");
               flag= 0;
            }

            if(flag==1)
            {
                Ti.Facebook.logout();
                alert('Logged In');    
                flag=0;            
            }
            else
                alert("Cancelled");


        });
        Ti.Facebook.authorize();

});
