Center Image in scrollView while scrolling.

How to place image at center position while scrollview zoom is increases and user moves image by touch.
image place to be center when user moves image in the scrollview.

for example:

var imageScrollView = Ti.UI.createScrollView({
    minZoomScale:1.0, maxZoomScale:2.0, zoomScale:1.0,
    height:300, width:300, left:0, top:0,
    contentHeight:'auto', contentWidth:'auto', x:0, y:0
});

var imageView = Ti.UI.createImageView({
    image:image, 
    width:300, height:300
});
imageScrollView.add(imageView);

imageScrollView.addEventListener('scale', function(e){
        zoom = e.scale;
});
