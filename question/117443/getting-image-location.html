I am having difficulties obtaining the x,y coordinates of an imageview that changes dynamically.

I can't find anything in the documentation that mentions this. Any one know how to get the coordinates or relative position?

  • var win = Titanium.UI.currentWindow;


var dynamicTop = 0;
var dynamicLeft = 0;

var myImageView = Titanium.UI.ImageView({
  top:dynamicTop, 
  left:dynamicLeft
});

function getMyImageRelPosition(object_to_evaluate){

    alert('my object is placed at '+
             object_to_evaluate.top+
             ' from the top and '+
             object_to_evaluate.left+
             ' from the left' );

}

win.addEventListener('get_position', 
    function(){ 
         getMyImageRelPosition(myImageView) 
    });

win.fireEvent('get_position');
