Used maxZoomScale property,but image is not zooming,why

This is my Code

var win = Ti.UI.currentWindow;


win.orientationModes = [ Ti.UI.PORTRAIT ];

var isFullscreen = false;
var isGesture = true;
var images = win.windowInfo;
var myID = win.myID;
var viewArray = [];


var imageLocation = '/images/';

var photosView = Ti.UI.createLabel({
    backgroundImage : imageLocation + images[myID].filename,
    width : 320,
    height : 470,
    top : 0,

});


var pview =  Titanium.UI.createScrollView({
                          contentWidth:'auto',
                          contentHeight:'auto',
                          top:0,
                          maxZoomScale:1.0,
                          minZoomScale:1.0,
                          showVerticalScrollIndicator:true,
                          showHorizontalScrollIndicator:true
});



pview.add(photosView);
win.add(pview);

I have used "maxZoomScale" property of createScrollView, but my image is not zooming. please somebody figure out why it is not happening. Thank you:)

  • maxZoomScalemust be bigger than minZoomScale. For set up initial zoom, use zoomScaleproperty:

    var pview =  Titanium.UI.createScrollView({
                          contentWidth:'auto',
                          contentHeight:'auto',
                          top:0,
                          maxZoomScale:5.0,
                         zoomScale:3.0,
                          minZoomScale:1.0,
                          showVerticalScrollIndicator:true,
                          showHorizontalScrollIndicator:true
});
