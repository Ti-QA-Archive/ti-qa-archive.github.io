    Hi Cédric, try with this

    index.xml

    <Alloy>
    <Window>
        <View id="view1" onClick="init"/>
        <View id="view2" onClick="init"/>
        <Button id="btnclickMe" onClick="clickMe"/>
    </Window>
</Alloy>

    index.tss

    "Window": {http://www.appcelerator.com/developers
    backgroundColor:"green"    
},

"#view1":{
    backgroundColor:"blue",
    height: 100,
    width: 200,
    top: 20,
    left: 0,
    visible: false
},

"#view2":{
    backgroundColor:"red",
    height: 100,
    width: 200,
    top: 140,
    left: 0,
    visible: false
},

"#btnclickMe": {
    title: "Click here!"
}

    index.js

    var cycler,
    aBoolean = true;

function clickMe(){
    $.btnclickMe.enabled = false;
    $.view1.setVisible(true);
    cycler = setInterval(function(){
        aBoolean = !aBoolean;
        $.view1.setVisible(aBoolean);
        $.view2.setVisible(!aBoolean);
    }, 5000);
}

function init(){
    clearInterval(cycler);
    $.btnclickMe.enabled = true;
    $.view1.setVisible(false);
    $.view2.setVisible(false);
    aBoolean = true;
}

$.index.open();
