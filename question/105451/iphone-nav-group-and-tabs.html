iPhone Nav Group and Tabs

Hi guys,

I have a weird glitch. I have a tabgroup and when the app fires up it calls the js script with a nav group in it. 

app.js code

var tabGroup = Titanium.UI.createTabGroup();

var win = Titanium.UI.createWindow({  
    title:'Welcome',
    url:'welcome.js'
});
var tab = Titanium.UI.createTab({  
    icon:'116-controller.png',
    title:'Tab 1',
    window:win
});

tabGroup.addTab(tab);  
tabGroup.open();

navgroup.js Code

var w = Ti.UI.currentWindow;
var win = Titanium.UI.createWindow({top:0});
var win1 = Titanium.UI.createWindow({
    backgroundColor:"red",
    title:"Red Window"
});

win1button.addEventListener('click', function(e)
{

var nav = Titanium.UI.iPhone.createNavigationGroup({

    backgroundColor:"blue",
    title:"Blue Window"
});

var win2button = Ti.UI.createButton({width:20,height:20,title:'next'});
win2button.addEventListener('click', function(e)
{
    nav.open(win3,{animated:true})

win2.setRightNavButton(win2button);

/* Win 3 */
var win3 = Titanium.UI.createWindow({
