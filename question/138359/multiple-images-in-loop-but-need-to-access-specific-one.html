    You could do something like this:

    for (var i = 0; i < myArr.length; i++ ) {

    var mycat = myArr[i];

    var myImage = Ti.UI.createImageView({
        height:42,
        width: 53,
        top: 10,
        // bottom:10,
        left:5,
        image: mycat.catImageUrl
    });

     (function(catImage, catImageHoverUrl, catID, catName) {

        catImage.addEventListener('click',function(e) { 

            this.image = catImageHoverUrl;

            Ti.API.info('Cat Id:' + catID);
            Ti.API.info('Cat Name:' + catName);
            Ti.API.info('Cat Image Hover :' + catImageHoverUrl);

        });

    })(myImage, mycat.catImageHoverUrl, mycat.catID, mycat.catName);

    scrollView.add(myImage);

}

    I would also advise against adding your own properties to UI elements, sometimes it can cause behavior that is hard to debug.
