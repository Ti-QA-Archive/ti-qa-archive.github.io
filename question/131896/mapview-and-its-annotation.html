MapView and it's annotation

I have a map view, I add some annotations. When I click on map view I get an annotation A but on map view, annotation B show it's popover. So that, I set mapView.selectAnnotation(A) in the end of event 'click' of mapView but I get a loop between annotation A and B after that my app is crash. How to fix this? Please help me !!! I use Titanium 1.7.0 and this problem only happen with annotations have near location

var selectedAnnotation;
mapview = Titanium.Map.createView({
    mapType: Titanium.Map.STANDARD_TYPE,
    region: { latitude: 50.55735542801804, longitude: 4.469935999999968, latitudeDelta: 0.02, longitudeDelta: 0.02 },
    animate: true,
    regionFit: true,
    userLocation: true
});
mapview.addEventListener('click', function (evt) {   
    if (evt.annotation != undefined) {
       selectedAnnotation = evt.annotation;
       mapview.selectAnnotation(selectedAnnotation );
    }
});
