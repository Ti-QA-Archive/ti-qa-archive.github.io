Cannot play songs in queue list

My problem is: although i set playlist in music player but the music player just play one song, cannot use skipToNext() method to play next song.

var player = Titanium.Media.appMusicPlayer;

var playlist = [];
var songs = ['All I Have To Give', 'All of My Love'];

for (var i=0; i < songs.length; i++) {
    var query = {};
    query['title'] = { value: songs[i], exact: true };

    var result = Ti.Media.queryMusicLibrary(query);

    if (result.length > 0) {
        playlist.push(result[0]);
    }
}

function playSong() {
    player.setQueue(playlist);
    player.play();
}

function nextSong() {
    player.skipToNext();
}

    This took me a good while to figure out

    Change 

            player.setQueue(picked);

    To

            player.setQueue({items: picked});
