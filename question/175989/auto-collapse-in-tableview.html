  • It is a simple programming task, like accordion in jQuery: 

    var rows = require('ui/rows')();
var table = Ti.UI.createTableView({activerowndx:null,data:rows});

table.addEventListener('click',function(e){
    if (table.activerowindex != null) {
        // here works the magic 
        // i.e.:
        rows[able.activerowindex].setHeight(50);
        table.activerowindex = e.index; // index of row
        rows[able.activerowindex].setHeight(100);    
    }    
})
