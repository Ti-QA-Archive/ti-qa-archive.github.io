Can not assign callback function to view

When I add a numeric property to a view I can see it in debug log. But when I add a function as property it is not added. Why?

See last code block addPlayers.js, the output of "addPlayers currentWindow2:" contains gameId, but gameStatus disappears.

mainScreen.js


navigation.openWindowWithData('addPlayers',tableData[index].status);

navigation.js



if (OS_IOS)
    navWindow = Titanium.UI.iOS.createNavigationWindow();

var windows=new Array();

exports.openWindowWithData =function(controllerName,userData) {
    var view = Alloy.createController(controllerName,{userData: userData}).getView();
    view.userData=userData;
    view.gameStatus=[];
    view.gameId=0;
    if (OS_IOS) 
        navWindow.openWindow(view);
    else 
        view.open();
    windows.push(view);
    return view;
};

exports.currentWindow=function() {
    return windows[windows.length-1];
};
