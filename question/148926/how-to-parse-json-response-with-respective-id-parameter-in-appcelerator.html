      Ti.UI.backgroundColor = '#dddddd';

      var win = Ti.UI.createWindow();

      var table = Ti.UI.createTableView();

      var hotdealjson = '{"parties":{"2012-02-21":[{"name":"test","ID":"1","eventdate":"2012-02-21","time":"22:00:00","place":"Le rafiot","club_id":"7"}],"2012-02-22":[{"name":"test1","ID":"2","eventdate":"2012-02-21","time":"22:00:00","place":"Le rafiot","club_id":"7"}],"2012-02-28":[{"name":"test2","ID":"3","eventdate":"2012-02-21","time":"22:00:00","place":"Le rafiot","club_id":"7"}]}}';

      var parsedData = JSON.parse(hotdealjson);

      var tableData = [];

      var parties, i, row, nameLabel, nickLabel;

      for (i = 0; i < parsedData.parties.length; i++) {
      row = Ti.UI.createTableViewRow({
      height:'60dp'
      });

          nameLabel = Ti.UI.createLabel({
        text:parsedData.parties[i].name,
        font:{
            fontSize:'24dp',
        fontWeight:'bold'
    },

    height:'auto',
    left:'10dp',
    top:'5dp',
    color:'#000',
    touchEnabled:false
    });

    nickLabel = Ti.UI.createLabel({
    text:parsedData.parties[i].eventdate,
    font:{
        fontSize:'16dp'
    },

    height:'auto',
    left:'15dp',
    bottom:'5dp',
    color:'#000',
    touchEnabled:false
    });

    row.add(nameLabel);
    row.add(nickLabel);
    tableData.push(row);
    }        
table.setData(tableData);

table.addEventListener('click', function(e) {
    if (e.rowData)

    }
});

      win.add(table);
