  • Here is some example code:

    var win = Ti.UI.createWindow({
    fullscreen:true,
    backgroundColor:'white'
});

var view = Titanium.UI.createScrollView({
    width:'100%',
    height:'100%',
    contentHeight:1000,
    backgroundColor:'white'
});

var someElement = Titanium.UI.createLabel({
    text:'Hi, I\'m a label',
    top:0,
    height:40,
    right:5,
    left:5
});

var otherElement = Titanium.UI.createLabel({
    text:'Hi, I\'m a label (2)',
    bottom:10,
    height:40,
    right:5,
    left:5
});

var tableView = Titanium.UI.createTableView({
    width:'100%',
    height:300,
    left:0,
    right:0
});

var rows = [];
for(var i=0; i<50; i++){
    rows.push(Titanium.UI.createTableViewRow({
        title:'This is row ' + String(i)
    }));
};
tableView.setData(rows);

view.add(someElement);
view.add(tableView);
view.add(otherElement);
win.add(view);
win.open();

    Try this on iPhone simulator…as you can see, it's possible to scroll the tableView and the scrollview. If you try this on an Android device or emulator you can see that is impossible to scroll the table.

    • Hi Libero,
      Just set fix height of scrollview that solve your problem.

      var view = Titanium.UI.createScrollView({
    width:'100%',
    height:'400',
    contentHeight:1000,
    backgroundColor:'white'
});
