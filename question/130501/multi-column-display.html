  • You can use something like below to make a grid. And just make it a 2 column grid.

    var cellWidth = 45,
        cellHeight         = 45,
        xSpacer         = 5,
        ySpacer         = 12,
        xGrid             = 5,
        yGrid             = 4,
        tableData         = [],
        cellIndex         = 0;
        for (var y=0; y<yGrid; y++){
            var thisRow = Ti.UI.createTableViewRow({
                className: "grid",
                layout: "horizontal",
                height: cellHeight+xSpacer,
                selectedBackgroundColor:"transparent"
            });
            for (var x=0; x<xGrid; x++){
                var temp = cellIndex + 1,
                thisView = Ti.UI.createView({
                    objName:"grid-view",
                    objIndex:temp.toString(),
                    left: ySpacer,
                    height: cellHeight,
                    width: cellWidth,
                    border:1,
                    borderColor:'#333'
                }),
                thisLabel = Ti.UI.createLabel({
                    color:'#333',
                    font:{fontSize:14,fontWeight:'bold'},
                    text:temp.toString(),
                    touchEnabled:false,
                    textAlign:'center'
                });
                //-> Control level select based on HnR.LEVEL
                if (temp > HnR.LEVEL) {
                    thisView.borderColor = '#CCC';
                    thisLabel.color = '#CCC';
                    thisView.touchEnabled = false;
                }
                thisView.add(thisLabel);
                thisRow.add(thisView);
                cellIndex++;
            }
            tableData.push(thisRow);
        }
        var tableview = Ti.UI.createTableView({
            data:tableData,
            height:200,
            width:300,
            scrollable:false,
            separatorStyle:'none'
        });
        return tableview;

    Above you will see yGrid and xGrid variables. Just play around with it.
