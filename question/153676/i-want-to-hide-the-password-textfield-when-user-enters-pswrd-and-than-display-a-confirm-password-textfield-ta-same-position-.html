    Hello Nabeel,

    Try below Code : 

    //<<<<<<<<<<<<<<<<<<<<PASSWORD>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>
vae win = Ti.UI.createWindow({
    height : "100%",
    width : "100%"
});

var pass_register = Titanium.UI.createLabel({   
    text:'PASSWORD:',
    font: {fontSize:29,fontFamily :"Helvetica"},
    backgroundColor:'white',
    color:'#666666',
    height:'auto',
    width:'auto',
    left: '5%'
});
win.add(pass_register);

var passtext_register = Titanium.UI.createTextField({   
    borderRadius:3,
    borderColor: '#99CCFF',
    keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
    //softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_SHOW_ON_FOCUS,
    font: {fontSize:24,fontFamily :"Helvetica"},
    left:'5%',
    height:'35dp',
    width:'70%',
    color:"#99CCFF",
    passwordMask: true
});
win.add(passtext_register);

passtext_register.addEventListener('blur', function(e){
    pass_register.text = "CONFIRM PASSWORD";
    passtext_register.value = "";
});

win.open();

    Jigar

var pass_register = Titanium.UI.createLabel({

    color:'#666666',
    height:'50dp',
    width:'auto',

});

var passtext_register = Titanium.UI.createTextField({  
    top:'50dp', 
    borderRadius:3,
    borderColor: '#99CCFF',
    keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
    //softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_SHOW_ON_FOCUS,
    font: {fontSize:24,fontFamily :"Helvetica"},
    left:'5%',
    height:'35dp',
    width:'70%',

});
login_scroll.add(passtext_register);


//<<<<<<<<<<<<<<<<<<<< Cnfirm PASSWORD>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>

var pass2_register = Titanium.UI.createLabel({ 
    top:0, visible:false,
    text:'CONFIRM PSWORD:',
    font: {fontSize:29,fontFamily :"Helvetica"},
    backgroundColor:'white',
    color:'#666666',
    height:'50dp',
    width:'auto',
    left: '5%'
});
login_scroll.add(pass2_register);

var passtext2_register = Titanium.UI.createTextField({  
    top:'50dp',visible:false,
    borderRadius:3,
    borderColor: '#99CCFF',
    keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
    font: {fontSize:24,fontFamily :"Helvetica"},
    left:'5%',
    height:'35dp',
    width:'70%',
    color:"#99CCFF",
    passwordMask: true
});
login_scroll.add(passtext2_register);


passtext_register.addEventListener('blur', function(){
    pass_register.visible = false;
    passtext_register.visible=false;
    pass2_register.visible = true;
    passtext2_register.visible = true;
});



login_scroll.open();
