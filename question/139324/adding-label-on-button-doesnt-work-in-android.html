Adding label on button doesn't work in android.

Hi,

Adding a Label in Button works fine in iphone, but not in android. Please help.

var btnMoreAppoinments = Titanium.UI.createButton({
    backgroundImage: '/images/popup_btn.png',
    width: '157dp',
    height: '31dp',
    borderWidth: 1,
    borderRadius: 2,
    top: '14dp',
    left: '15dp',
});
var lblMoreAppoinments = Titanium.UI.createLabel({
    text: 'Appointments',
    width: 'auto',
    height: 'auto',
    font: {
        fontFamily: 'Arial',
        fontWeight: 'bold',
        fontSize: '12sp'
    },
    color: '#FFFFFF',
    textAlign:'center'
});
btnMoreAppoinments.add(lblMoreAppoinments);

Regards
Sreekumar

  • In android adding button to view doesn't support , so you have to use view as an alternative in case of android.

    var viewMoreAppoinments = Titanium.UI.createView({
    backgroundImage: '/images/popup_btn.png',
    width: '157dp',
    height: '31dp',
    borderWidth: 1,
    borderRadius: 2,
    top: '14dp',
    left: '15dp',
});
var lblMoreAppoinments = Titanium.UI.createLabel({
    text: 'Appointments',
    width: 'auto',
    height: 'auto',
    font: {
        fontFamily: 'Arial',
        fontWeight: 'bold',
        fontSize: '12sp'
    },
    color: '#FFFFFF',
    textAlign:'center'
});
viewMoreAppoinments.add(lblMoreAppoinments);
