  • This is working for me on iOS and 2.1.3GA, it loops until I call "stopAudio" function:

    
var audioBackground;    
var startAudio = function(url){
        if (audioBackground){
            if (audioBackground.isPlaying()){
                //Ti.API.info('already playing... go away..');
                return;
            }
        }else{
            audioBackground = Ti.Media.createSound({ 
                url: url,
                allowBackground: false,
                looping: true
            });             
        }
        Titanium.Media.audioSessionMode = Titanium.Media.AUDIO_SESSION_MODE_AMBIENT;

        audioBackground.play();           
}
var stopAudio = function(){
        if (audioBackground){
            if (audioBackground.isPlaying()){        
                audioBackground.pause();
            }
        }
}

startAudio('sounds/xxx.mp3');
