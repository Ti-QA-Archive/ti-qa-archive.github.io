Geolocation: How to removeEventListener correctly?

Hello,
I've been struggling with this and really don't know what I'm doing wrong here.

Here's some code example:

index.xml:

<Alloy>
    <Window class="container">
        <Label id="a" top="10">Start</Label>
        <Label id="f" top="20">f</Label>
        <Label id="g" top="30">g</Label>
        <Label id="b" bottom="10">Stop</Label>
    </Window>
</Alloy>

index.js:

var lala = function(e) {
    Ti.Geolocation.ACCURACY_HIGH;
    Ti.Geolocation.preferredProvider = Ti.Geolocation.PROVIDER_GPS;
    Ti.Geolocation.addEventListener('location', function(e) {
        if (e.error) {
            alert('Please enable GPS');
        } else {
            alert("yeah!");
            $.f.value = e.coords.latitude;
            $.g.value = e.coords.longitude;
        }
    });
};

$.a.addEventListener('click', lala);
//This starts the GPS services correctly on device.


$.b.addEventListener('click', function(){
    Ti.Geolocation.removeEventListener('location', lala);
});
$.index.open();

I've also tried(inside index.js):

$.b.addEventListener('click', function(){
    $.a.removeEventListener('click', lala);
});

But none of these actually removes GPS Location Service.

Thanks a lot for the help.

    Hi,

    you need to pass same function to remove listener.

    try this one..

    
var getLocation = function(e){
    if (e.error) {
            alert('Please enable GPS');
        } else {
            alert("yeah!");
            $.f.value = e.coords.latitude;
            $.g.value = e.coords.longitude;
        }
}

var lala = function(e) {
    Ti.Geolocation.ACCURACY_HIGH;
    Ti.Geolocation.preferredProvider = Ti.Geolocation.PROVIDER_GPS;
    Ti.Geolocation.addEventListener('location', getLocation);
};

$.a.addEventListener('click', lala);
//This starts the GPS services correctly on device.


$.b.addEventListener('click', function(){
    Ti.Geolocation.removeEventListener('location', getLocation);
});
$.index.open();
