  • Hi Super,

    Here is the code for rows having different background color for rows which are empty. You can add you own logic for searching items and iterate the results in for loop.

    
    var self = Ti.UI.createView();
    var tableView = Ti.UI.createTableView({
        separatorStyle : Ti.UI.iPhone.TableViewSeparatorStyle.NONE,
        backgroundColor : '#2b2b2b',
    });
    self.add(tableView);

    var _data = []
    var _return = [];
    var view = [];
    var caption = [];
    for(var i=0; i<3;i++) {
        _return[i] = Ti.UI.createTableViewRow({
            height : 45,
            backgroundColor : '#FFF',
            id : i,
        });

        view[i] = Ti.UI.createView({
            height : "100%",
            width : "100%",
            backgroundColor : '#FFF',
            borderColor : '#2b2b2b',
            borderWidth : 1
        });
        caption[i] = Ti.UI.createLabel({
            text : "item " + i,
            font : {
                fontSize : 20,
                fontWeight : 'bold'
            },
            left : 20,
            width : 270,
            touchEnabled : false,
            color : '#5a5a5a',
        });
        view[i].add(caption[i]);
        _return[i].add(view[i]);
        _data.push(_return[i]);
    }
    tableView.data = _data;

    Let me know, if it works

  • Hi Kapil, sure no problem.

    Here is the code from my controller (additional unrelated logic removed)

    //Data is passed in from calling controller
var args = arguments[0] || {};

loadTable(args.people);

function loadTable(e) {
  var data = [];
  var index = [];
  for (var i = 0; i < e.meta.total_count; i++) {
    var person = e.objects[i];
    var row = Alloy.createController('page_friendsRow', person).getView();    
    row.username = person.username;
  }

  $.tableView.setData(data);

  var search = Ti.UI.createSearchBar();
  $.tableView.setSearch(search);
  $.tableView.searchHidden = true;
}
