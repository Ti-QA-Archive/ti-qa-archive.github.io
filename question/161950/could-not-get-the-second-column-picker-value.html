    Hi

    Please check the following code. I hope, it will solve your problem a little bit.

    Thanks 

    var first="";
var second="";
picker.addEventListener('change',function(e){
    if(e.columnIndex==0)
    {
        first=e.row.title;
    }
    else
    {
        second=e.row.title;
    }
     alert(first+'  '+second);
});
