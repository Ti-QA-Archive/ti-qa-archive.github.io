    Hi

    I have put a quick test app together for you, create a new basic project and replace the contents of app.js with the following;

    Ti.UI.setBackgroundColor('#000');

var tab1, tab2, tab3;
(function () {
    var win = Ti.UI.createWindow({  
        backgroundColor: '#fff',
        title: 'Tab 3',
    });
    tab3 = Ti.UI.createTab({  
        icon: 'KS_nav_views.png',
        title: 'Tab 3',
        window: win
    });

    var v_text = Titanium.UI.createView({
        backgroundColor: '#fff',
        borderRadius: 10,
        bottom: 90,
        left: 10,
        right: 10,
        top: 100
    });

    var cur_lang = 'eng';

    var lbl_text = Titanium.UI.createLabel({
        backgroundColor: 'orange',
        borderRadius: 10,
        bottom: 0,
        color: '#224',
        font: {
            fontSize: 20,
            fontFamily: 'Helvetica Neue'
        },
        left: 10,
        right: 3,
        text: 'On',
        textAlign: 'left',
        top: 0,
        width: 200
    });

    v_text.add(lbl_text);
    win.add(v_text);

    lbl_text.addEventListener('singletap',function (){
        lbl_text.setText('Single Tapped');
    });

    lbl_text.addEventListener('swipe',function(){
        lbl_text.setText('Swiped');
    });
})();

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab3);
tabGroup.open();

    This app uses some of your code but is simplified for the purposes of the demo.

    This works correctly on both iOS and Android. If you tap the label it will show single tap if you swipe it will show swipe.

    I have tried it on Android 2.2 specifically. BTW 2.1.4 is the latest stable version, it is worth considering updating.
