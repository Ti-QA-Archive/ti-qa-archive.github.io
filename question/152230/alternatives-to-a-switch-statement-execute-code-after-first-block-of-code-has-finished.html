Alternatives to a switch statement? Execute code after first block of code has finished.

This is a simplified version of my event listener, the code works fine. But, I have 16 cases. Is there a way that I can do this with out using switch statements, a more efficient way? The count down and alert fire at the beginning of each case. 


var one = rows.fieldByName('time');
var two = rows.fieldByName('time1');
var three = rows.fieldByName('time2');
var one1 = rows.fieldByName('time3');
var two1 = rows.fieldByName('time4');
var three1 = rows.fieldByName('time5');

Ti.App.addEventListener('finished', function() {

             --timerRepeat;

            switch (timerRepeat) {
                case 3:
                    countdown.countdownSetTime({
                        hours : 0,
                        minutes : 0,
                        seconds : one
                    });

                    alert( one1);
                    countdown.countdownStart();
                    break;
                case 2:
                countdown.countdownSetTime({
                        hours : 0,
                        minutes : 0,
                        seconds : two
                    });

                    alert( two1);
                    countdown.countdownStart();
                    break;
                case 1:
                countdown.countdownSetTime({
                        hours : 0,
                        minutes : 0,
                        seconds : three
                    });

                    alert( three1);
                    countdown.countdownStart();
                    break;
        }

    });

    How about:

    var rows = [
     rows.fieldByName('time'),
     rows.fieldByName('time1'),
     rows.fieldByName('time2'),
     rows.fieldByName('time3'),
     rows.fieldByName('time4'),
     rows.fieldByName('time5')
]

Ti.App.addEventListener('finished', function() {
    countdown.countdownSetTime({
        hours : 0,
        minutes : 0,
        seconds : rows[timerRepeat]
    });

    alert(rows[timerRepeat]);
    countdown.countdownStart();
});
