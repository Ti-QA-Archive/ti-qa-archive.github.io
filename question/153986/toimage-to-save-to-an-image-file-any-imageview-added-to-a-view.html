toImage to save to an image file any imageView added to a view

Hi, in a View I have three images and I want to save them as it was only one, using toImage method:

var imgViewA1 = Ti.UI.createImageView({image:"imgs/a1.png"});
var imgViewA2 = Ti.UI.createImageView({image:"imgs/a2.png"});
var imgViewA3 = Ti.UI.createImageView({image:"imgs/a3.png"});

var tempView = Ti.UI.createView({layout:"vertical",width:320,height:480});
tempView.add(imgViewA1);
tempView.add(imgViewA2);
tempView.add(imgViewA3);
self.add(tempView);

var blob = tempView.toImage();
var filename = Titanium.Filesystem.tempDirectory + "/blob_img.png";
var f = Titanium.Filesystem.getFile(filename);
f.write(blob);
Titanium.Media.saveToPhotoGallery(f);

Problem is it saves only the first image and ignores the other two. Is there any way to save ANY image added to a view?

  • Accepted Answer

    var imgViewA1 = Ti.UI.createImageView({image:"imgs/a1.png", top:0});
var imgViewA2 = Ti.UI.createImageView({image:"imgs/a2.png", top:0});
var imgViewA3 = Ti.UI.createImageView({image:"imgs/a3.png", top:0});
