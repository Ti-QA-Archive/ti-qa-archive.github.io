How to resend confirmation mail using ACS to User if his/her Activation/Confirmation mail is lost.

I referred the docs to find out a solution for this, and I came to know that it could be done using Ti.Cloud.Users.resend_confirmation. Here is my code:

function resendConfirmation (emailId) {

var Cloud = require('ti.cloud');
Cloud.debug = true;
Cloud.Users.resend_confirmation({
        email    : emailId,
    }, function(e) {

        if (e.success)     {
            //var user = e.users[0];
            alert('success');
        }
        else
        {
            alert('Login Error:\n' +
                ((e.error && e.message) || JSON.stringify(e)));
        }
});

}

But it is throwing me following error:
[ERROR] Script Error = 'undefined' is not a function (evaluating 'Cloud.Users.resend_confirmation')

  • Hi Rob,

    Try This : 

    var Cloud = require('ti.cloud');
Cloud.debug = true;
Cloud.Users.login({
    login : 'push123',
    password : 'push123'
}, function(e) {
    if (e.success) {
        Cloud.Users.resend_confirmation({
            email : emailId,
        }, function(e) {
            if (e.success) {
                //var user = e.users[0];
                alert('success');
            } else {
                //error
            }
        });
    } else {
        alert('Error: ' + ((e.error && e.message) || JSON.stringify(e)));
    }
});
