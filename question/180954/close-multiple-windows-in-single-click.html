  • Once you reached new window close the previous window. Below is the sample code,

    
var win = Ti.UI.createWindow();

var btnSubmit = Ti.UI.createButton({
    title: 'Submit'
});

btnSubmit.addEventListener('click', function() {
    var win1 = Ti.UI.createWindow();
    win1.open();
    win1.addEventListener('postlayout', function() {
        win.close();
    });

});

win.add(btnSubmit);

win.open();

    Hope this helps.

  • Hi Balakrishna,

    You can try this too.

    
var winroot = Ti.UI.createWindow({title:'Root Window', backgroundColor:'gray'});

var btn = Ti.UI.createButton({
    title: 'Go to form window'
});

winroot.add(btn);




btn.addEventListener('click', function() {
    var formwindow = Ti.UI.createWindow({title:'Form Window', backgroundColor:'red'});

var btnSubmit = Ti.UI.createButton({
    title: 'Submit'
});
btnSubmit.addEventListener('click', function() {
    var win3rd = Ti.UI.createWindow({backgroundColor:'white', title:'Third window'});
    var lbl = Ti.UI.createLabel({text:'I am 3rd window, press back buton to go back to root window.',color: 'green'});
    win3rd.add(lbl);
    win3rd.open();
    formwindow.close();
    });


formwindow.add(btnSubmit);
formwindow.open();
});

 winroot.open();
