var playButton = Titanium.UI.createButton({
    title: "",
    width: "13%",

    backgroundImage: "play_100x100[white] (1).png",

var outer = Ti.UI.createView({

Why can't I add the button to the view? Every time I try I am getting a runtime. The message is 

Uncaught Error: Invalid value, expected type Object.

  • Accepted Answer

    Well, if I run the following code, I encounter no issue at all.

    var win = Ti.UI.createWindow();

var playButton = Titanium.UI.createButton({
    title: "",
    width: "13%",
    height: "8.5%",
    bottom: "14%",
    left: "42%",
    id: "playButton",
    backgroundImage: "play_100x100[white] (1).png",
    zIndex:1
});

var outer = Ti.UI.createView({
    backgroundColor: 'purple',
    contentHeight: Ti.UI.SIZE,
    contentWidth: Ti.UI.FILL,
    height: Ti.UI.FILL,
    layout: 'vertical',
    showHorizontalScrollIndicator: false,
    showVerticalScrollIndicator: true,
    width: Ti.UI.FILL
});

outer.add(playButton);
win.add(outer);

win.open();

    My advice to you would be to clean your project using the Project -> Clean… menu.

    Hope this helps.
