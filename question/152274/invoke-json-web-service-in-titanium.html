  • First off, you would need to add an event listener on the button for the click event as follows:

        btn.addEventListener('click', function(e) {
         // Web service code
    });

    The code to call a rest webservice to return a json object would be as follows:

    
var xhr = Titanium.Network.createHTTPClient();
xhr.onload = function(e) {
    // this is where you would process the returned object.
    if (this.responseText != null) {
        var jsObj = JSON.parse(this.responseText);
        // Do something with the object
    } else {
       // Webservice returned nothing
    }
};

xhr.onerror = function(e) {
    // This is where you would catch any errors thrown from calling the webservice. 
    // e.error holds the error message
}

xhr.open('GET', 'http://www.somewebserviceurl.com/methodname');
xhr.setRequestHeader('Content-Type', "application/json; charset=utf-8");
xhr.send();

    Hope that helps.
