Controler/view communication

Hi *,

here my controler:

ctrl.references = ( function() {
    var api = {};
    api.getAll = function(url,callback) {
        var xhr = Titanium.Network.createHTTPClient({});
        xhr.onload = function() {
            try {
                var data = JSON.parse(this.responseText);
                callback(data);
            } catch(e) {
            };
        };
        xhr.open('GET', url);
        xhr.setRequestHeader("Content-Type", "application/json; charset=utf-8");
        xhr.send();
    };
    return api;
}());

and in this view I try to read data, but data is undefined:

view.references = ( function() {
    var api = {};
    api.getWindow = function() {
        var w = Ti.UI.createWindow({});
        var tv = Ti.UI.createTableView({top:40});

            Ti.UI.log(data);  // here come the problem

        })
        w.add(tv);
        return w;
    };
    return api;
}());

Whats going wrong?

  • Here my solution to link own apps:

    if(! typeof (view) == 'object') {
    var view = {};
}
view.references = ( function() {
    var api = {};
    api.getWindow = function() {
        var w = Ti.UI.createWindow({});
        w.addEventListener('close', function() {
            w = null;
        });
        var tv = Ti.UI.createTableView({
            top : 40
        });
        tv.addEventListener('click', function(e) {
            Ti.Platform.openURL(e.rowData.appstore);
            Ti.API.log(e.rowData.appstore);
        });
        ctrl.references.getAll('http://webmasterei.com/fileadmin/references.json', function(data) {
            var rows = [];
            for(var i = 0; i < data.length; i++) {
                var item = data[i];
                var row = Ti.UI.createTableViewRow({
                    hasChild : true,
                    height : 'auto',
                    appstore : item.link
                });
                var title = Ti.UI.createLabel({
                    font : {
                        fontSize : 18,
                        fontWeight : 'bold'
                    },
                    top : 5,
                    left : 80,
                    height : 20,
                    text : item.title
                });
                var descr = Ti.UI.createLabel({
                    font : {
                        fontSize : 14,
                    },
                    top : 25,
                    left : 80,
                    height : 'auto',
                    text : item.descr
                });
                row.add(descr);
                var img = Ti.UI.createImageView({
                    image : item.icon,
                    width : 60,
                    height : 60,
                    top : 5,
                    left : 5
                });
                row.add(img);
                row.add(title);
                rows.push(row);
            }
            tv.setData(rows);

        })
        w.add(tv);
        return w;

    };
    return api;
}());
