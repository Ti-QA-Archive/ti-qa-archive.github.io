    It would be better to save the files in a folder, assign a name and reference the file in the database instead.

var btnSort2 = Ti.UI.createButton({
    height: 40,
    right: 10,
    title: 'Test 2',
    top: 10,
    width: '40%'
});
win.add(btnSort2);

btnSort1.addEventListener('click', function (e) {
    var imageFile = Ti.Filesystem.getFile('color.png');
    var oneImage = imageFile.read();
    var db = Ti.Database.open('images');
    db.execute("Drop Table if exists pics;");
    db.execute("CREATE TABLE IF NOT EXISTS pics (id INTEGER, image TEXT);");
    db.execute("INSERT INTO pics (id,image) VALUES(?,?);", 1, Ti.Utils.base64encode(oneImage));
    db.close();
    Ti.API.info('Test 1');
});
btnSort2.addEventListener('click', function (e) {
    var db = Ti.Database.open('images');
    var rs = db.execute("SELECT id,image FROM pics");
    var imgData;
    if (rs.isValidRow()) {
        imgData = rs.fieldByName('image');
    }
    Ti.API.info('imgData: ' + imgData);
    rs.close();
    db.close();

    var img = Ti.UI.createImageView({
        height: Ti.UI.SIZE,
        image: Ti.Utils.base64decode(imgData),
        width: Ti.UI.SIZE
    });
    win.add(img);
    Ti.API.info('Test 2');
});

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab);
tabGroup.open();
