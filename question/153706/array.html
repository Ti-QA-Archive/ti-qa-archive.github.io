  • Hi,
    If supposed you are talking from one window to 2nd window. You can do it as below:

    firswin.js

    
var array =  ['one', 'two', 'three'];

var win = Ti.UI.createWindow({
    backgroundColor: 'white',
    url: 'secwin.js'
    title: 'Click button to test'
    myarray: array   
});

var button = Ti.UI.createButton({
    backgroundImage: '/images/custom-slider-right.png',
    backgroundSelectedImage:'/images/custom-slider-left.png',
    title: 'Click me!',
    top: 10,
    width: 300,
    height: 200
});
button.addEventListener('click',function(e){
    Ti.API.info("You clicked the button");
});
win.add(button);
win.open();

    And here is 2nd win code

    secwin.js

    var win = Ti.UI.currentWindow;

 var getarray = win.myarray // In this way you can get into the second window

    // not getarray will contain the passed array

    Hope that will help you.
