  • Here I found the solution, so I would like to share it

    var _storage = Ti.Filesystem.applicationDataDirectory;
if(Ti.Filesystem.isExternalStoragePresent()){
    alert("Externam storage ");
    _storage = 'file:///sdcard/';                       
};

var folder = Ti.Filesystem.getFile(_storage,'Download/images.jpeg');
if (folder.exists()){
        alert("Exist");
}
else
{
alert("not exist");
}
