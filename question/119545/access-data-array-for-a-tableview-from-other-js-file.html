Access data array for a TableView from other .js file

I am willing to have the same TableView in several tabs of my project (Titanium mobile SDK, ipad/iphone). I want to write the data array for that tableView in only one of the windows (in a .js file) and access it from other windows (from other .js files). Let´s see the (simple) code to clarify it:

app.js

var tabGroup = Titanium.UI.createTabGroup({ });
var win0 = Titanium.UI.createWindow({
    url:'js/main.js',
    fullscreen:true
    });
var tab0 = Titanium.UI.createTab({window:win0});
var win1 = Titanium.UI.createWindow({
    url:'js/main2.js',
    fullscreen:true
    });
var tab1 = Titanium.UI.createTab({window:win1});
tabGroup.addTab(tab0);
tabGroup.addTab(tab1);
tabGroup.setActiveTab(0); 
tabGroup.open({});

main.js

var win = Ti.UI.currentWindow;
var dataTable = [{title:"Jane"},{title:"Mary"},{title:"Sally}];
var table = Titanium.UI.createTableView({data:dataTable});
Titanium.UI.currentWindow.add(table);

main2.js

var win = Ti.UI.currentWindow;
// I want to access the data array from main.js for this table
var table = Titanium.UI.createTableView({data:dataTable});
Titanium.UI.currentWindow.add(table);

Any suggestions? Thanks.
