mobile-scollableView: 'Titanium.UI' [undefined] after return from a sub page

See simplified code below! Description: "singletap" on scrollView opens a new window with "Back"-button..
"click" on "Back"-putton closes the sub page. After that "singletap" on scrollView doesn't work any more and console shows:

[WARN] Exception in event callback. {
    ...
    line = 19;
    message = "Result of expression 'Titanium.UI' [undefined] is not an object.";
    ...
}

Tab-Code:

var win = Titanium.UI.currentWindow;

var pic = Ti.UI.createView({
    backgroundColor:'black',
    height:245,
    width:229
});

var scrollableView = Ti.UI.createScrollableView({
    height:320,
      views:[pic],
      showPagingControl:true,
      pagingControlColor:'transparent',
      top:90
});

scrollableView.addEventListener("singletap",function(e){
    var w = Titanium.UI.createWindow({
        title:'',
        backgroundColor:"transparent",
        url:'/pages/subpage.js'
    });

    Ti.UI.currentTab.open(w,{animated:true});
});

win.add(scrollableView);

Subpage-Code:

//Create a pointer to the current window context
var win = Titanium.UI.currentWindow;

var backButton = Ti.UI.createButton({
    title:'Back'        
}); 

backButton.addEventListener('click', function(){
    win.close();
});

win.add(backButton);

System:
