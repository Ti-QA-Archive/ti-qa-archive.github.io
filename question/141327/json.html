Json

Hello,

i'm trying to read a json file with the following content:

{
    "biography": [
        {
            "content": "<html>test test test test test test test test test test test test test test test test test test </html>"
        }
    ]
}

in titanium i am writing this:


var url = 'http://localhost/example.json';
var xhr = Ti.Network.createHTTPClient({});
    xhr.onload = function() {
            json = JSON.parse(this.responseText);
            myHtml = json.biography.content;
        };
    xhr.open("GET", url);
    xhr.send();

    Ti.API.info(json);

    var webView = Titanium.UI.createWebView({
        html: myHtml,
        height:'100%',
        width:'100%',
        backgroundColor:'#ff0',
        color:'#000'
    });

the result of Ti.API.info(json); is null
please help

    That is because the Ti.info is out side the load function and since the xhr is not synchronous you get a null try this

    var url = 'http://localhost/example.json';
var xhr = Ti.Network.createHTTPClient({});
    xhr.onload = function() {
            json = JSON.parse(this.responseText);
            myHtml = json.biography.content;

            Ti.API.info(json);

            var webView = Titanium.UI.createWebView({
                   html: myHtml,
                height:'100%',
                width:'100%',
                backgroundColor:'#ff0',
                color:'#000'
            });

        };
    xhr.open("GET", url);
    xhr.send();
