Activuty Indicator not working on modal window

Hi am using 3.1.3.GA sdk, Alloy and 2.3.4 Android Emulator and also tested in 4.1 Android, I created a modal window to load places from server, it takes sometime to load so I decided to add ActivityIndicator to my modal window, but not showing, My code is,

index.xml

var activityIndicator;
function activityIndicatorShow(){
    var style;
    if(osname == 'iphone'){
        style = Ti.UI.iPhone.ActivityIndicatorStyle.DARK;
    }
    else{
          style = Ti.UI.ActivityIndicatorStyle.DARK;
    }
    activityIndicator = Ti.UI.createActivityIndicator({
          color: 'green',
          font: {fontFamily:'tt0001m_0', fontSize:26, fontWeight:'bold'},
          message: 'Loading...',
          style:style,
          top:10,
          left:10,
          height:Ti.UI.SIZE,
          width:Ti.UI.SIZE
    });
    $.placeChooser.add(activityIndicator);
}
activityIndicatorShow();
loadPlaces();

function loadPlaces(){
    var data = [];
    var placesList = Alloy.Globals.locationData;
    if(placesList.length <= 0){
        loadlocationsFromOnline();
    }else{
                  activityIndicator.show();
        for(var i = 0, j = placesList.length;i < j;i++){
            data.push(Alloy.createController('location/placeSuggestor', {
                articleUrl: placesList[i].id+'~'+placesList[i].value,
                places: placesList[i].value
            }).getView());                    
        }
        $.placeSuggestions.setData(data);
             activityIndicator.hide();
    }    
}
