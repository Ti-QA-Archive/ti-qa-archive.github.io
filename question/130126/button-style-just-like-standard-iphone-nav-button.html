  • You can get the "nav bar" style button by creating a ButtonBar with a single label object. Here is a working example:

    var win = Titanium.UI.createWindow({  
    title:'Button test',
    backgroundColor:'#fff'
});

var button1 = Ti.UI.createButtonBar({
    labels:[{title:"Blue Button", width:100}],
    style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
    backgroundColor:'#1b67a5',
    top:100,
    zIndex:110,
    height:30,
    width:'auto'
});

var button2 = Ti.UI.createButtonBar({
    labels:[{title:"Orange Button", width:100}],
    style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
    backgroundColor:'#c67621',
    top:200,
    zIndex:110,
    height:30,
    width:'auto'
});

win.add(button1);
win.add(button2);

win.open();
