    Hi Aniket,

    This may help you : 

    var loginReq = Titanium.Network.createHTTPClient();
var params = {
 username : uname,
 password : pwd
};
loginReq.onload = function() {
 var response = JSON.parse(this.responseText);
 alert(response);
}
loginReq.onerror = function() {
 alert(this.responseText);
}
loginReq.open("POST", "http://mysite.com/get_init.php"); 
loginReq.send(JSON.stringify(params)); //or Try this - loginReq.send(params);

    Thanks SP
