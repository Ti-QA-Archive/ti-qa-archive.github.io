  • Hello VARUN,

    Suppose you have a button Replace and you can set event as like

    Replace.addEventListener('click', function() {

// where aLabel and aTextField will be the same position
    view.remove(aLabel);
    aLabel = null;
    view.add(aTextField);
});

    #app.js

    var win = Titanium.UI.createWindow({
    backgroundColor : 'red',
    title : 'Red Window',
});

// Create a Button.
var Replace = Ti.UI.createButton({
    title : 'Repless',
    height : 100,
    width : 200,
    top : 10,

});

// Listen for click events.
Replace.addEventListener('click', function() {
    win.remove(aLabel);
    aLabel = null;
    win.add(aTextField);
});

// Add to the parent view.
win.add(Replace);

// Create a Label.
var aLabel = Ti.UI.createLabel({
    text : 'This is label',
    color : '#000',
    font : {
        fontSize : 20
    },
    height : 50,
    width : 250,
    top : 200,

});

// Add to the parent view.
win.add(aLabel);

// Create a TextField.
var aTextField = Ti.UI.createTextField({
    height : 50,
    top : 200,
    width : 250,
    hintText : 'This is hint text',
    //softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_DEFAULT_ON_FOCUS, // Android only
    keyboardType : Ti.UI.KEYBOARD_DEFAULT,
    returnKeyType : Ti.UI.RETURNKEY_DEFAULT,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});

// Listen for return events.
aTextField.addEventListener('return', function(e) {
    aTextField.blur();
    alert('Input was: ' + aTextField.value);
});

win.open();

    If you have any question, then let me know.

    Thanks.

  • I made the following changes and it worked:

    Just put the label to be replaced into another view(unitcontainer in this case ) and remove the label and add textfield to that view.

    xml

    <View id="HeaderView" layout="vertical" >                   
    <View id="Horizontal" layout="horizontal" >
       <View id="unitcontainer">
     <Label class="bold" id="unit" text ="sometext"/>
    </View>
        <Label class="fbold" id="unitCode" text="someText"/>
    </View>
</View>

    js

    if(args.unit ===' '){
    $.unitcontainer.remove($.unit);
    $.unit=null;
    var unitTextfield =  Ti.UI.createTextField({
       borderStyle:Titanium.UI.INPUT_BORDERSTYLE_BEZEL,
        height:"80%",
        width:Titanium.UI.FILL,
        top:0,
        font:{
        fontFamily: 'Arial',
        fontWeight: 'bold',
        fontSize: 14
    });
    $.unitcontainer.add($.unitTextfield);
}
