    var imagen        = Ti.UI.createImageView();
imagen.image         = '/images/apart.png';
imagen.image         = redimensionarImg('images', 'apart.png', 500, 0);


function redimensionarImg(folder, img, newWidth, newHeight){
    var f = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory, folder, img);
    var blob = f.read();
    var ratio = blob.width / blob.height;

    if (newWidth == 0){
        var w = newHeight * ratio;
        var h = newHeight;
    } else if (newHeight == 0){
        var w = newWidth;
        var h = newWidth / ratio;
    } else {
        var w = newWidth;
        var h = newHeight;
    }
    return ImageFactory.imageAsResized(blob, { width:w, height:h});
}
