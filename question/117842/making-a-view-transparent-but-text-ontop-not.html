Making a view transparent, but text ontop not

            var charsView = Ti.UI.createView({
                bottom:5,
                left:15,
                zIndex:2,
                opacity:0.25,
                width:'auto',
                height:'auto',
                backgroundColor:'#fff'
            });
            var charsName = Ti.UI.createLabel({
                text:"Text",
                top:2,
                left:5,
                right:5,
                bottom:2,
                zIndex:3,
                opacity:1.0,
                color:'#000',
                width:'auto',
                height:'auto',
                font:{fontSize:14,fontWeight:'bold'}
            });

No matter what I do, the entire thing (charsName included) is at opacity 0.25, when I want the VIEW to be 0.25 and the text to be completely opaque. Anyone know how to make this possible?

  • Try something like this (I haven't test it, I've just write it directly on my browser)

    var wrapper = Ti.UI.createView({
  backgroundColor: 'transparent',
  height: 'auto',
  width:'auto'
});

var charsView = Ti.UI.createView({
                bottom:5,
                left:15,
                zIndex:2,
                opacity:0.25,
                width:'auto',
                height:'auto',
                backgroundColor:'#fff'
            });
var charsName = Ti.UI.createLabel({
                text:"Text",
                top:2,
                left:5,
                right:5,
                bottom:2,
                zIndex:3,
                opacity:1.0,
                color:'#000',
                width:'auto',
                height:'auto',
                font:{fontSize:14,fontWeight:'bold'}
            });

wrapper.add(charsView);
wrapper.add(charsName);

onEvent('yourEvent', function () {
  charsView.opacity = 0.25;
})

  • In order to make a small window pop up (win) with a transparent background everywhere else (winBG) I had to do the following:

    var winBG = Titanium.UI.createWindow({
    backgroundColor:'#000',
    opacity:0.60
});
var win = Titanium.UI.createWindow({
    height:300,
    top:10,
    right:10,
    left:10,
    borderColor:'#FFF',
    borderWidth:3,
    backgroundColor:'#FF0000',
    borderRadius:3.0
});
winBG.open();
win.open();

    Whatever elements you don't want to be transparent, add to the win and the winBG should show up behind it as a transparent overlay. To get the win to center you'll have to modify it, I happened to want it near the top.

    I'm using Titanium Mobile 1.6.1 and Android 1.6.

  • You need to add your label, charsName, as a child view of your parent view, charsView. To make it really layered out, specify your parent view's layout to composite

    var charsView = Ti.UI.createView({
                bottom:5,
                left:15,
                zIndex:2,
                opacity:0.25,
                width:'auto',
                height:'auto',
                backgroundColor:'#fff',
        /* try adding this if you want the label to be literally on top of this parent view */
        layout: "composite"
            });
            var charsName = Ti.UI.createLabel({
                text:"Text",
                top:2,
                left:5,
                right:5,
                bottom:2,
                zIndex:3,
                opacity:1.0,
                color:'#000',
                width:'auto',
                height:'auto',
                font:{fontSize:14,fontWeight:'bold'}
            });
/* This label needs to be a child view of the parent view. 
  Try this first before specifying the layout to composite for the parent view. */
charsView.add(charsName);
