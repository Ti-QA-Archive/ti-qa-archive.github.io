Longpress problem

Hello!
I am trying to implement longpress for a tableview (can't believe that's not in the API), but am running into some trouble.
I've tried using some code that i found here on the Q & A written by Jason B:

var touchable = function(control) {
        var tolerance = 2;
        var longpressDuration = 500;
        var _touchTimeout = null;
        var _longpressInterval = null;
        var reset = function() {
            if (_longpressInterval) clearTimeout(_longpressInterval);
        };
        var longpressFire = function() {
            control.fireEvent('longpress');
        };
        control.addEventListener('touchstart', function(evt) {
            reset();
            _longpressTimeout = setTimeout(longpressFire, longpressDuration);
        });
        control.addEventListener('touchmove', function (e) {
                  if (_longpressInterval) clearTimeout(_longpressInterval);
        });
        control.addEventListener('touchcancel', reset);
        control.addEventListener('touchend', function (end) {
            reset();
        });
        return control;
};

But it seems to trigger no matter how long the touch is :/
Also, i wan't to get which row it is that triggers longpress. How do i do that?
