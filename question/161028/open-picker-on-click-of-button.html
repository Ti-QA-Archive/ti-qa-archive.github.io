  • Try this:

    $.win.open();
var picker = Ti.UI.createPicker({
          top:50,
     visible : false
});
picker.add([Ti.UI.createPickerRow({title:'Bananas'}),Ti.UI.createPickerRow({title:'Strawberries'}),Ti.UI.createPickerRow({title:'Mangos'}),Ti.UI.createPickerRow({title:'Grapes'}]);
picker.selectionIndicator = true;
win.add(picker);
picker.setSelectedRow(0, 2, false); 2, false);      
 $.img.addEventListener("click", function() {
    picker.show();    
});
