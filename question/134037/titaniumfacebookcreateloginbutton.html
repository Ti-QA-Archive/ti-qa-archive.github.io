Titanium.Facebook.createLoginButton

Hello, i want to change background image of facebook login button from default to customize image. please help me out.

  • create a custom button,

    var button = Titanium.UI.createButton({ 
    title: 'FB Login', 
    backgroundImage:'test.png' // custom background image,
    width:100,
    height:30,
    top:45,
    left:150
});

    add event listener to custom button, and call Facebook.authorize() method

    button.addEventListener('click', function (e) {
   Titanium.Facebook.authorize(); // Facebook authorization(login)
});

    then add a facebook login event listener

    Titanium.Facebook.addEventListener('login', function(e) {
   //handle login response here
});

    Hope this help you :)

  • If you want to have a dinamic button that change the facebook logged image for another one when it is loggedin. 

    var facebookBtn = Titanium.UI.createButton({ 
    backgroundImage:'/images/facebook_connect.png',
    width:150,
    height:30,
    top : '72%',
});

facebookBtn.addEventListener('click', function (e) {


    if(fb.getLoggedIn())
    {
        fb.logout();
    }
    else 
    {

           fb.authorize();
    }
});


fb.addEventListener('login', function(e) {
             facebookBtn.setBackgroundImage('/images/facebook_connect_lout.png');

    });

fb.addEventListener('logout', function(e) {
            facebookBtn.setBackgroundImage('/images/facebook_connect.png');

        });
