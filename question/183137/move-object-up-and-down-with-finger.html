  • SOLUTION:

    var viewSize = 100, viewX = 0, viewY = 30;

// grey box
var view = Titanium.UI.createView({
    height:viewSize,
    width:viewSize,
    backgroundColor:'#7f8c8d',
    opacity:0.3,
    top:150
});

// red box
var view2 = Titanium.UI.createView({
    height:viewSize,
    width:viewSize,
    backgroundColor:'#c0392b',
    top:150
});

view2.addEventListener('touchmove', function(e) {

    view2.setTop(e.y + viewY - (viewSize/2));

});

view2.addEventListener('touchend', function(e) {

    viewY = view.getTop();;

    view2.setTop(view.getTop());

});

$.camera.add(view);
$.camera.add(view2);
