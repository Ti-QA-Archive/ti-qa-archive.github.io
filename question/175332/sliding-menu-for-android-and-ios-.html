  • Hi Khodour,

    try this.

    Ti.UI.setBackgroundColor('#fff');
var win=Ti.UI.createWindow({
    backgroundColor: 'white',
    navBarHidden:true
   // zIndex: 0
});
var leftView=Ti.UI.createView({
    height: '100%',
    width:'75%',
    top: 0,
    right: 0,
    backgroundColor: 'black',
   // zIndex: 1
});
win.add(leftView);
var data=[{ title:'Settings'},
    { title:'Contact Us'}];
var tableView=Ti.UI.createTableView({
    left: 0,
    top: 0,//'45dp',
    bottom :0,
    scrollable: false,
    //backgroundColor: 'blue',
    width: '100%',
    data:data
});
leftView.add(tableView);

var mainView=Ti.UI.createView({
     top: 0,
   // zIndex: 2,
    height: '100%',
    width : '100%',
    right : 0,
    backgroundColor: 'white'
});
win.add(mainView);
var btnList=Ti.UI.createButton({
     image: ' btnImage.png',  //  put your image path here
     right: "10dp",
    top: "14dp",
    height: Ti.UI.SIZE,
    width: Ti.UI.SIZE,
    style: 'PLAIN',
    backgroundColor: 'transparent',
   // title:'hello'
});
mainView.add(btnList);
var animateToLeft = Ti.UI.createAnimation({
    right : '75%',
    curve : Ti.UI.ANIMATION_CURVE_EASE_OUT,
    duration : 500,
});

var animateToRight = Ti.UI.createAnimation({
    right : '0',
    curve : Ti.UI.ANIMATION_CURVE_EASE_OUT,
    duration : 500,
});
var position = "right";
btnList.addEventListener('click',function(e){
    if (position == "right") {
        mainView.animate(animateToLeft);
        position = "left";
    } else {
        mainView.animate(animateToRight);
        position = "right";
    }
});
win.open();
