Changing of title or image in buttonBar not work

Hi,
after clicking on buttonBar I want to change title or image. It doesnt work ;-((

var buttonBar = Titanium.UI.createButtonBar({
    labels : [{
        title : 'Start Radio CML',
        width : 120,
        enabled : true
    }, {
        image : Ti.Filesystem.resourcesDirectory + '_clm.png',
        width : 36
    }, {
        title : 'Panoramio Map',
        width : 120,
        enabled : true
    }, {
        image : Ti.Filesystem.resourcesDirectory + '_panoramio.png',
        width : 36
    }],
    style : Ti.UI.iPhone.SystemButtonStyle.BAR,
    height : 36,
    width : 'auto'
});
buttonBar.addEventListener('click', function(e) {
    if(e.index == 0) {
        if(streamer.playing) {
            streamer.stop();
            buttonBar.labels[0].title = 'Start Radio CLM';
        } else {
            streamer.start();
            buttonBar.labels[0].image = 'clm.png';
            buttonBar.labels[0].title = 'Stop Radio CLM';
            buttonBar.labels[1].image = Ti.Filesystem.resourcesDirectory + 'clm.png',
        }
    }
});

Any ideas?

Rainer

  • If you scroll down to the bottom in the Kitchen Sink example where it shows a "custom buttonbar" you'll see that you should keep an external array of your labels, don't try to access the one you assign when you create buttonbar. Try something like this:

    var labels = [{
        title : 'Start Radio CML',
        width : 120,
        enabled : true
    }, {
        image : Ti.Filesystem.resourcesDirectory + '_clm.png',
        width : 36
    }, {
        title : 'Panoramio Map',
        width : 120,
        enabled : true
    }, {
        image : Ti.Filesystem.resourcesDirectory + '_panoramio.png',
        width : 36
}];
var buttonBar = Titanium.UI.createButtonBar({
    labels : labels,
    style : Ti.UI.iPhone.SystemButtonStyle.BAR,
    height : 36,
    width : 'auto'
});
buttonBar.addEventListener('click', function(e) {
    if(e.index == 0) {
        if(streamer.playing) {
            streamer.stop();
            labels[0].title = 'Start Radio CLM';
        } else {
            streamer.start();
            labels[0].image = 'clm.png';
            labels[0].title = 'Stop Radio CLM';
            labels[1].image = Ti.Filesystem.resourcesDirectory + 'clm.png',
        }
    }
});
