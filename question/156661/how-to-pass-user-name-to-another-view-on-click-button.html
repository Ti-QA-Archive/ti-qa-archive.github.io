  • Hi

    You can pass parameters into the CommonJS modules you create and also gain information back from inside using call backs. Using these techniques allows you to create and use CommonJS that passing information between each other.

    Example CommonJS showing the techniques;

    function TestView(o) {
    // allows this function to be called without error
    // if no parameter is passed
    var obj = o || {};
    var self = Ti.UI.createView({
        layout:"vertical"
    });
    var nameText=Ti.UI.createLabel({
        // example of something passed in
        text: obj.title || '',
        top: 50
    });
    var nameTextField = Ti.UI.createTextField({
        borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
        height: 35,
        top: 100
        width: 140,
    }); 
    nameTextField.addEventListener('change', function (e) {
        // example of call back function
        if (obj.change) {
            obj.change({
                name: e.value
            });
        }
    });
    self.add(nameTextField);
    return self;
}
module.exports = TestView;

    Usage

    var view = require('TestView').TestView({
    change: function (e) {
        alert(JSON.stringify(e));
    },
    title: 'Shows on label'
});
