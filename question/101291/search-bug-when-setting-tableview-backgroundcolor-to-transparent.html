  • Hi! It was me who started this thread. What I experienced was actually no bug, but bad coding! When I searched, I created a new tableview wich placed itself on top of the first one. As all the tableview background was transparent I could still see all the tableviews stacked on top of each other. The correct way is to replace the data of the first tableview rather than creating new ones… A beginner mistake I made a year ago, hope this helps someone who is having the same problem now =)

    • I made a quick sample and this works for me.
      SDK 1.8.2.

      Tell me if i missed something =)

      Titanium.UI.setBackgroundColor('#000');

var win = Titanium.UI.createWindow({  
    backgroundColor:'#ccc'
});

var data = [];

data[0] = Ti.UI.createTableViewRow({title:'Horse'});
data[1] = Ti.UI.createTableViewRow({title:'Cat'});
data[2] = Ti.UI.createTableViewRow({title:'Dog'});
data[3] = Ti.UI.createTableViewRow({title:'Gandalf'});

var search = Titanium.UI.createSearchBar({
    barColor:'#385292',
    showCancel:false,
    hintText:'search'
});
search.addEventListener('change', function(e)
{
    Ti.API.info(e.value);
});
    search.addEventListener('return', function(e)
{
    search.blur();
});
    search.addEventListener('cancel', function(e)
{
    search.blur();
});

var tableView = Titanium.UI.createTableView({
    data:data,
    search:search,
    backgroundColor:'transparent',
    searchHidden:true
});
win.add(tableView);

win.open();
