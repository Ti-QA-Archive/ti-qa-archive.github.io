ex. 

// Recycle no-longer-visible pages 
    for (MyImageView *page in visiblePages) {

            [page removeFromSuperview];
        }
    }

    OK, than you need to use JavaScript arrays.

    //define array
var views = [];
//push Ti View to array
views.push(Ti.UI.createView({ backgroundColor: '#000' }));
//one more :)
views.push(Ti.UI.createView({ backgroundColor: '#fff' }));

//loop
for (var i = 0; i < views.length; ++i) {
    Ti.API.debug(views[i].backgroundColor);
}
