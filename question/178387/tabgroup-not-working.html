    • You need to require the class and then create an instance of it so it would be something like this
      app.js

          var tabGroup = require('tabGroup');
    var tabGroupWindow = tabGroup();
    tabGroupWindow.open();

      tabGroup.js

      function tabGroup(){
    self = Ti.UI.createTabGroup();
    var window1 = require('Window1');
    var win1 = new window1();

    var tab1 = Ti.UI.createTab({
               window: win1,
                icon:'KS_nav_ui.png'
        });


    return self;
}
module.exports = tabGroup;
