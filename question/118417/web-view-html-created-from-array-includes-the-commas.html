    think you want to use "join"

    https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/Array/join

    var instructionHtml = (function() {
    var arr =[];
    for(var i=0,j=instrArr.length;i<j;i++) {
        arr.push('<p>'+instrArr[i]+'</p>');
    }

    return arr.join(); // <-- this
})();
