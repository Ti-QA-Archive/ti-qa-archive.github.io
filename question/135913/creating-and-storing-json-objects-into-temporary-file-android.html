creating and storing json objects into temporary file (android)

Hi, I am currently working on android platform and developing small application.In my application I need one temporary file which contains my all json objects. I want to create new temp file and access to that temp file.What is the proper way to create file and storing data in it.Need help… Thank you

    Try this:

    // Write JSON object to temp file
var file = Ti.Filesystem.getFile(Ti.Filesystem.tempDirectory, 'tmp.json');
var jsonObj = {
    a: 1,
    b: 2,
    c: {d: 3}
};

file.write( JSON.stringify(jsonObj) );

// Read JSON from temp file
var jsonStr = file.read().text;
var json = JSON.parse(jsonStr);
alert(json.c.d);

    • function write(obj, id) {
    var file = Ti.Filesystem.getFile(Ti.Filesystem.tempDirectory, id + '.txt');
    file.write( JSON.stringify(obj) );
}

function read(id) {
    var file = Ti.Filesystem.getFile(Ti.Filesystem.tempDirectory, id + '.txt');
    var jsonStr = file.read().text;
    return JSON.parse(jsonStr);
}

var jsonObj1 = {
    id: 'adfd2332',
    data: {a: 1, b: 2}
};
var jsonObj2 = {
    id: 'adfd1112',
    data: {a: 3, b: 4}
};

write(jsonObj1, jsonObj1.id);
write(jsonObj2, jsonObj2.id);

// Write JSON object to temp file
var obj1 = read(jsonObj1.id);
var obj2 = read(jsonObj2.id);

alert(obj1.data.a);
alert(obj2.data.a);
