    Hi

    Please check the following code.I hope, it will help to you

    index.xml

    <Alloy>
    <Window id="win" backgroundColor="white" exitOnClose="true" fullscreen="false" title="ScrollView Demo">
        <View id="mainNav" backgroundColor="green" height="60"  top="0">
          <Button id="buttonMenuLeft"  left="0">Left</Button>
          <View id="innerView">
              <Button id="buttonTimeline" >Time</Button>
              <Button id="loadViewMap" >Map</Button>
          </View>
          <Button id="buttonMenuRight" right="0">Right</Button>

        </View>
    </Window>
</Alloy>

    index.js

    "#innerView":
{
layout:"horizontal",
width:Ti.UI.SIZE,
height: Ti.UI.SIZE
}
