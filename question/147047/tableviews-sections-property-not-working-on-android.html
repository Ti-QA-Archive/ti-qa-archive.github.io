    • Ok, I just created a new project, opened app.js, cleared out the code Appcelerator puts in there by default, pasted the following code, ran in Apple iPhone simulator, worked fine. Are you getting a particular error?

      Ti.UI.backgroundColor = 'white';
var win = Ti.UI.createWindow();

var sectionFruit = Ti.UI.createTableViewSection({ headerTitle: 'Fruit' });
sectionFruit.add(Ti.UI.createTableViewRow({ title: 'Apples' }));
sectionFruit.add(Ti.UI.createTableViewRow({ title: 'Bananas' }));

var sectionVeg = Ti.UI.createTableViewSection({ headerTitle: 'Vegetables' });
sectionVeg.add(Ti.UI.createTableViewRow({ title: 'Carrots' }));
sectionVeg.add(Ti.UI.createTableViewRow({ title: 'Potatoes' }));

var table = Ti.UI.createTableView({
  // Prior to Release 3.0, add sections using the data property:
  // data: [sectionFruit, sectionVeg]
  sections: [sectionFruit, sectionVeg]
});

win.add(table);
win.open();

var sectionFish = Ti.UI.createTableViewSection({ headerTitle: 'Fish' });
sectionFish.add(Ti.UI.createTableViewRow({ title: 'Cod' }));
sectionFish.add(Ti.UI.createTableViewRow({ title: 'Haddock' }));

// Prior to Release 3.0, you can only add and remove sections by setting the data property:
// table.data = [ sectionFish, sectionFruit, sectionVeg ];
table.insertSectionBefore(0, sectionFish);

    • Marco… Sorry, brain cramp…

      Took a look and it seems that on Android the sections: [sectionFruit, sectionVeg] is not setting the sections property of the table. Change the code above to use the data property which will accept the sections and it works. Looks like a bug. 

      var table = Ti.UI.createTableView({
  // Prior to Release 3.0, add sections using the data property:
  data: [sectionFruit, sectionVeg]
  //sections: [sectionFruit, sectionVeg]
});
