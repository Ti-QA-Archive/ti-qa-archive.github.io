Mobileweb layout

This code does not render correctly on mobile web, how can I fix it ?

var win = Ti.UI.createWindow({
        title : 'Welcome',
//        modal : true,
        backgroundColor : "#c92"
    });
    var label = Ti.UI.createLabel({
        text : 'Enter your mobile number',
        textAlign : 'center',
        right:0, left:0,
        height : 40,
        top : 20
    });

    var txt = Ti.UI.createTextField({
        hintText : 'Your Mobile Number',
        top : 80,
        height : 40,
        right:0, left:0
    });

    var btn = Ti.UI.createButton({
        title : 'Send Code',
        top : 140,
        height : 40,
         width: Ti.UI.Fill
    });


    var txt2 = Ti.UI.createTextField({
        hintText : 'Enter authentication code',
        top: 200 , width: Ti.UI.Fill, height: 40
    });

    var btn2 = Ti.UI.createButton({
        title : 'Authenticate',
        top: 260, height: 40, width: Ti.UI.FILL
    });

    win.add(label);
    win.add(txt);
    win.add(btn);
    win.add(txt2);
    win.add(btn2);
    win.open();

    There is a typo in your code. Ti.UI.Fill is not a real constant. It needs to be Ti.UI.FILL in all uppercase letters.
