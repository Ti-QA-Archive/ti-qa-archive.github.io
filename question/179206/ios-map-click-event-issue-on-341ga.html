iOS: Map click event issue on 3.4.1.GA

I found that click on a map anntation's title or subtitle will not fire a click event.

Titanium SDK: 3.4.1.GA

ti.map: 2.2.2

Platform: iOS8

var win = Titanium.UI.createWindow({
    backgroundColor: '#fff'
});

var region = {
    latitude: 23.5,
    longitude: 121,
    latitudeDelta: 0.006,
    longitudeDelta: 0.006
};

var Map = require('ti.map');
var map = Map.createView({
    region: region,
    animate: true,
    userLocation: false
});


var annotation = Map.createAnnotation({
    title: 'title',
    subtitle: 'subtitle',
    latitude: 23.5,
    longitude: 121,
});

map.addAnnotation(annotation);

map.addEventListener('click', function(e) {
    Ti.API.info(e);
});

win.add(map);

win.open();

  • After full testing, I found the not only title or subtile has the issue.

    Here is the code:

    var win = Titanium.UI.createWindow({
    backgroundColor: '#fff'
});

var region = {
    latitude: 23.5,
    longitude: 121,
    latitudeDelta: 0.006,
    longitudeDelta: 0.006
};

var Map = require('ti.map');
var map = Map.createView({
    region: region,
    animate: true,
    userLocation: false
});


var annotation = Map.createAnnotation({
    title: 'title',
    subtitle: 'subtitle',
    latitude: 23.5,
    longitude: 121,
    leftView: Ti.UI.createView({
        width: 30,
        height: 30,
        backgroundColor: 'red'
    }),
    rightView: Ti.UI.createView({
        width: 30,
        height: 30,
        backgroundColor: 'green'
    }),
});

map.addAnnotation(annotation);

map.addEventListener('click', function(e) {
    Ti.API.info(e);
    switch (e.clicksource) {
        case 'pin':
            // corrent
            Ti.API.info('pin clicked');
            break;

        case 'annotation':
            // incorrect, user deselects the annotation by clicking elsewhere in the map view, clicksource shuild be null not annotation.
            Ti.API.info('annotation clicked');
            break;

        case 'title':
            // incorrect, never fire
            Ti.API.info('title clicked');
            break;

        case 'subtitle':
            // incorrect, never fire
            Ti.API.info('subtitle clicked');
            break;

        case 'leftPane':
            // incorrect, never fire
            Ti.API.info('leftPane clicked');
            break;

        case 'rightPane':
            // incorrect, never fire
            Ti.API.info('rightPane clicked');
            break;

        case 'infoWindow':
            // incorrect,never fire
            Ti.API.info('infoWindow clicked');
            break;

        case 'null':
            // incorrect,never fire
            Ti.API.info('nothing clicked');
            break;
    }
});

win.add(map);

win.open();
