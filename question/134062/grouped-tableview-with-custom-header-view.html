Grouped Tableview with custom header view

Hello,

Is there any way to add a custom header view for a Grouped tableview? I can see how I can add a simple text-label but cant really figure out how I can add more than that.

  • Grouped Tableview doesn't support custom header view as far as i know. But you can archive this by creating table with custom headerView for each header. 

    var createCustomView = function(title) {
    var view = Ti.UI.createView({
        backgroundColor: 'blue',
        height: 40,
    });
    var text = Ti.UI.createLabel({
        text: title,
        left: 20,
        color: '#fff'
    });
    view.add(text);
    return view;
};


var win = Ti.UI.createWindow();

// Table 1
var inputData1 = [
    {title:'row 1'},
    {title:'row 2'},
    {title:'row 3'}
];
var tableView1 = Titanium.UI.createTableView({
    data:inputData1,
    style:Titanium.UI.iPhone.TableViewStyle.GROUPED,
    headerView: createCustomView('Header 1'),
});

// Table 2
var inputData2 = [
    {title:'row 4'},
    {title:'row 5'},
    {title:'row 6'}
];
var tableView2 = Titanium.UI.createTableView({
    data:inputData2,
    style:Titanium.UI.iPhone.TableViewStyle.GROUPED,
    headerView: createCustomView('Header 2'),
    top: 200
});

win.add(tableView1);
win.add(tableView2);
win.open();
