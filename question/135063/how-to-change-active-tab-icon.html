  • hi Hitesh,

    i tried the code and works well…can you try this..?

    in app.js

    var tabGroup = Titanium.UI.createTabGroup();

//
// create base UI tab and root window
//

var tab1 = Titanium.UI.createTab({
    icon : 'KS_nav_views.png',
    title : 'Tab 1',
    window : win
});

tabGroup.addTab(tab1);

// open tab group
tabGroup.open();

Ti.App.addEventListener('changeTabIcon', function(e) {
    tabGroup.tabs[e.index].icon = 'img/' + e.name;
    //i set the image from img folder
});

    in 1.js

    var win = Ti.UI.currentWindow;

//custom tabGroup

var btn = Ti.UI.createButton({
    title : 'Change',
    top : 10,
    left : 10,
    width : 40,
    height : 30
});

btn.addEventListener('click', function(e) {
    Ti.App.fireEvent('changeTabIcon', {
        index : 0,
        name : '1.jpg'
    });
});

win.add(btn);
