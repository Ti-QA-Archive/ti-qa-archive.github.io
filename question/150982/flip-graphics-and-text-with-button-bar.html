Flip graphics and text with button bar

I am trying to flip graphics and text between two buttons but one, the text keeps appearing on top of each other instead of disappearing and replacing and; two, how do I set the text to automatically appear in button one? I am coding for iOS and using the latest updates

    var contactswin = Titanium.UI.createWindow({
        backgroundColor: 'FFF',
    });    

    var contactsscrollView = Titanium.UI.createScrollView({
    contentWidth:'auto',
    contentHeight:'auto',
    top:44,
    showVerticalScrollIndicator:true,
    showHorizontalScrollIndicator:true
    });

    var bb = Titanium.UI.createButtonBar({
    labels:['X', Y',],
    backgroundColor:'#336699',
    top:50,
    style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
    height:30,
    width:250
});
    contactsscrollView.add(bb);

    var contactsdesc = Ti.UI.createLabel({
    color:'#000',
    font:{fontSize:16},
    left:10,
    top:145,
    right:10,
    textAlign:'left',
    text:'X stuff.',
    width:'auto'
    });

    var contactsdescy = Ti.UI.createLabel({
    color:'#000',
    font:{fontSize:16},
    left:10,
    top:145,
    right:10,
    textAlign:'left',
    text:'Y stuff.',
    width:'auto'
    });

    bb.addEventListener('click', function(e) {
    if (e.index === 0) {
    contactsscrollView.add(contactsdesc);
    } else if (e.index === 1) {
    contactsscrollView.add(contactsdescy);
}
});
    contactswin.add(contactsscrollView);
    contactswin.open();    
});
