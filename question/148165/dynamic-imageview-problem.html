  • Hi

    Try this;

    var dinamik = [], img;
for (var i=0; i < 10; i++){
    img = Ti.UI.createImageView({
        image: '/images/Binek.png',    
    });
    dinamik.push(img);
}
var scrollView = Titanium.UI.createScrollableView({
    views: dinamik,
    showPagingControl: true,
    pagingControlHeight: 30,
    maxZoomScale: 2.0,
    currentPage:0
});
win.add(scrollView);
