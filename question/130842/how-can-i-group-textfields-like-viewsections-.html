    Here is a way of doing that
    create a view with a vertical layout and add texboxes to it. Make the top property of the textboxes = 0

    var container = Ti.UI.createView({
        layout: 'vertical',
        height: 'auto',
        width: 'auto',
        borderRadius: 10,
        borderColor: '#000'
    });
var textFieldDefaults1 = {
    height: '40dp',
    width: '250dp',
    top: 0,
    color: '#222',
    backgroundColor: 'white',
    borderColor: '#000'
};
var textFieldDefaults2 = {
    height: '40dp',
    width: '250dp',
    top: 0,
    color: '#222',
    backgroundColor: 'white',
    borderColor: '#000'
};
container.add(textFieldDefaults1);
container.add(textFieldDefaults2);
