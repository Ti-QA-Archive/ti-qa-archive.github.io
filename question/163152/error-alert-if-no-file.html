     var win = Ti.UI.createWindow({

       // Create a Label.

           font : {fontSize:12},
           height : Ti.UI.SIZE,
           width : Ti.UI.SIZE,
           textAlign : 'center'
       });
               // Add to the parent view.
       errView.add(error);


// Create a Button.
var submit = Ti.UI.createButton({
