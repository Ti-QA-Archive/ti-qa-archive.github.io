    //in your app.js
var win = Titanium.UI.createWindow({
    title: 'welcome',
    backgroundColor: '#ccc'
});

var newWin = Titanium.UI.createLabel({
    text: 'New Window',
    font :{fontSize:17,fontWeight:'bold',fontFamily:'Helvetica Neue'},
    width: 100,
    textAlign: 'left',
    top: 10,
    left: 10,
    height: 25,
    color: '#000'
});
win.add(newWin);

newWin.addEventListener('click', function(e){ 
    var category = Ti.UI.createWindow({
        title: 'category',
        animated:true,
        modal: true
    });
    category.open();
});

win.open();
