  • I'm using the code below with 1.7.2 SDK, TiStudio 1.0.4, Android emulator 2.2, on Mac OSX Lion and it is working as expected. Try this code in your environment and see how it works. 

    Titanium.UI.setBackgroundColor('#000');

var win = Titanium.UI.createWindow({
    title: "Test",
    backgroundColor: "#000"
});

var theImage = Titanium.UI.createImageView({
        width : "100dip",
        height : "100dip",
        right : "10dip",
        image : "http://developer.appcelerator.com/assets/img/DEV_titmobile_image.png"
    });
    win.add(theImage);

win.open();
