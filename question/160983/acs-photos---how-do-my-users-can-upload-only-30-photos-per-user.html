    • Since you're working in Titanium, I assume, why not use the cloud functions built in, like this:

      var Cloud = require('ti.cloud');

    Cloud.Photos.query({
        page : 1,
        per_page : 100,
        where : {
            user_id : userID //not sure if you are using the built in user or a custom property, adjust as needed
        }
    }, function(e) {
        if (e.success) {
            if (e.meta.total_results > 30) {
                //Checking for more than 30... 
                //do what you need
            } else {
                //Less than 30, so let them upload a new one
            }
            //alert('Success:\n' + 'Count: ' + e.photos.length);
        } else {
            alert('Error:\n' + ((e.error && e.message) || JSON.stringify(e)));
        }
    });

      Whether you use the built in or the JSON call, the return property you are looking for is meta.total_results. That will tell you how many items match your query.
