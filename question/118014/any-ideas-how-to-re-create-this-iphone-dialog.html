  • Hi there,

    You need to set the 'movable' and 'editable' attributes set to true in the desired tableview.

    I'll show you an example. How to recreate the iPhone/Settings/Keyboards with titanium :)

    //-- Create the main window
var win = Titanium.UI.createWindow();

//-- Create a navGroup
var win1 = Titanium.UI.createWindow({
    backgroundColor:'#336699',
        title:"Keyboard"
});
var nav = Titanium.UI.iPhone.createNavigationGroup({
   window: win1
});
win.add(nav);

//-- Just some data 
//-- in this case, the same data presented in the image 
var data1 = [
    {title:'English (UK)', id:'uk', hasChild:true},
    {title:'Chinese - Simplified', id:'ch', hasChild:true}];

var data2 = [{title:'Add New Keyboard', id:'key', hasChild:true}]; 


//-- Create our editable and moving table
var tableView1 = Titanium.UI.createTableView({
    style: Titanium.UI.iPhone.TableViewStyle.GROUPED,
    data:data1,
    editable:true, 
    moveable:true
});

//-- Dummy table
var tableView2 = Titanium.UI.createTableView({
    style: Titanium.UI.iPhone.TableViewStyle.GROUPED,
    top: 120,
    data:data2
});

//--  create edit/cancel buttons for nav bar
var edit = Titanium.UI.createButton({
    title:'Edit'
});

var cancel = Titanium.UI.createButton({
    title:'Cancel',
    style:Titanium.UI.iPhone.SystemButtonStyle.DONE
});

//-- This event is going 
//-- to transform our editable tableView
edit.addEventListener('click', function()
{
    win1.setRightNavButton(cancel);

    //-- This does all the magic 
    //-- and says that our tableview1 is editable :)
    tableView1.editing = true;

    //-- Hide the 2nd tableview
    tableView2.visible = false;
});

cancel.addEventListener('click', function()
{
    win1.setRightNavButton(edit);

    //-- This does the exact opposite :)
    tableView1.editing = false; 

    //-- show the 2nd tableview
    tableView2.visible = true;
});


//-- The events for the tableview1 
//-- When a row is deleted
tableView1.addEventListener('delete', function(e){
    Titanium.API.info(e);
    Titanium.API.info(e.source);
    Ti.API.info('The '+e.index+' with '+e.rowData.id+' id row was deleted');
});
//-- When a row is moved
tableView1.addEventListener('move', function(e){
    Titanium.API.info(e);
    Titanium.API.info(e.source);
    Ti.API.info('The '+e.index+' with '+e.rowData.id+' id row was moved');
});

//-- Add everything to our main window
win1.setRightNavButton(edit);
win1.add(tableView1);
win1.add(tableView2);
win.open();

    For more information, see the documentation about tableviews, especially, the moving/movable and editing/editable attributes.

    Oh and don't forget to check the examples in the kitchensink ;)
    BaseUI - Views - Table Views

    Hope it helps you
