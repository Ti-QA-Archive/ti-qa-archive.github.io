    Ok solved it now. Here is my solution:

    var win = Titanium.UI.currentWindow;

var data = [];


 var tableview = Titanium.UI.createTableView({
    data:data,
    style:Titanium.UI.iPhone.TableViewStyle.PLAIN,
    backgroundColor:'yellow',
    //headerView:headerView,
    //footerView:footerView,
    maxRowHeight:100,
    minRowHeight:100,
    separatorStyle: Ti.UI.iPhone.TableViewSeparatorStyle.NONE
});


        var resourcesDir = Titanium.Filesystem.applicationDataDirectory+'../'+'/Library'+'/Application%20Support'+'/database';
        var dir = Titanium.Filesystem.getFile(resourcesDir);
        var dir_files = dir.getDirectoryListing();
        for (var i=1;i<dir_files.length;i++){ 
           data.push({title:'' + dir_files[i].toString() + ''});
        // set the array to the tableView
            tableview.setData(data);
        }




win.add(tableview);
