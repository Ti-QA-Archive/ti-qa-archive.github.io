    hi,

    you can set your custom variable to imageview and on the click event you can know that which image is clicked. like..

    for(var i=0; i < 10; i++){

var img = Ti.UI.createImageView({
    c_id : i
});

img.addEventListener('click',function(e){
    alert(e.source.c_id);
})

}
