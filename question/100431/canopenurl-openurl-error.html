canOpenURL, openURL error

I am trying to build a simple app for the Android platform using the Mobile SDK. Calling canOpenURL or openURL throws the following error in the emulator and in on the device

TypeError: Cannot call method "canOpenURL" of undefined(app://app.js#28)

I have pasted the app.js contents.

Please help! I am stumped!

Sujit

Ti.UI.setBackgroundColor('#000');

var rootWindow = Ti.UI.createWindow({
id: 'rootWindow',
title:'Quick Phone List',
backgroundColor: '#ff0',
fullscreen:'false'
});

var myData = [
{ title: 'Contact One', customData: '8889997777' },
{ title: 'Contact Two', customData: '8889997777' },
{ title: 'Contact Three', customData: '8889997777' },
{ title: 'Contact Four', customData: '8889997777' }
];

var rootTableView = Ti.UI.createTableView({
id:'rootTableView',
data:myData
});

rootTableView.addEventListener('click', function (e) {
var url = 'tel:' + e.rowData.customData;

if (Titanium.Platform.canOpenURL(url)) {
    Titanium.Platform.openURL(url);
}
else {
    alert('Cannot open URL');
}

});

rootWindow.add(rootTableView);

rootWindow.open();
