    • You are the absolute best! Thank you for taking the time. For future searchers who are curious, here is a copy of the code that ended up working (self is the tableView in question):

      self.addEventListener('scroll', function(e){
        var curRows = this.data[0].getRows();
        var loadIndex = [];
        Ti.API.warn('There are this many total rows: ' +curRows.length);
        for(var x in curRows) {
            if (curRows[x].loadRow === true) {
                loadIndex.push(x);
                Ti.API.warn('The index of the row in question is: ' +x);
            }
        }
        //THIS CODE ONLY WORKS FOR ANDROID - NEED TO DO SOMETHING ELSE FOR IOS!!!!!!!!
        //!!!!!!!
        var lastVisibleItem;
        for (var x in loadIndex) {
            lastVisibleItem = e.firstVisibleItem + e.visibleItemCount;
            Ti.API.warn('The index of the first visible item is: ' +e.firstVisibleItem );
            Ti.API.warn('The index of the last visible item is: ' +lastVisibleItem );
            if (loadIndex[x] >= e.firstVisibleItem && loadIndex[x] <= lastVisibleItem) {
                Ti.API.warn('The loadingRow is VISIBLE!!!!');
                break;
            }
        }
    })
