Don't focus the searchbar wehen opening a window

Hi, i've a problem with the searchbar and a tableview.
I have a window with a tableview and a searchbar at the top.
Now when i open the window, the searchbar gets the focus und the softkeyboard pops up.
On iPhone it works fine. I've only have this problem with android.
Kitchen Sink have the same problem. Have a look at Base UI -> Views -> Table Views -> Table Search.
How can i fix that? I don't want to focus the searchbar!
I found a solution with a timer and the blur method:

win.addEventListener('open' function(e) {
setTimeout(function(e) {
searchBar.blur();
}, 500);
});

But that's not very nice. The user recognizes that the keyboard is shown and hidden automatically. It isn't a noice behaviour.

Maybe someone can help me

    • Ok, i have a sample to reproduce it.

      ApplicationWindow.js

      //Application Window Component Constructor
function ApplicationWindow() {

    // create main tabgroup control
    var tabGroup = Ti.UI.createTabGroup();

    // create main tab for tabgroup
    var mainTab = Ti.UI.createTab({
        title: 'maintabcaption'
    });
    tabGroup.addTab(mainTab);

    // create login window
    var loginWindow = Ti.UI.createWindow({
        title: 'login',
        backgroundColor: '#fff',
        tabBarHidden: true
    });

    // button for login
    var btnLogin = Ti.UI.createButton({
        title: 'dologin'
    });
    btnLogin.addEventListener('click', function() {
        var SearchBar = require('ui/SearchBar');
        var searchBar = SearchBar();
        mainTab.open(searchBar);
    });
    loginWindow.add(btnLogin);

    mainTab.setWindow(loginWindow);

    return loginWindow;
};

//make constructor function the public component interface
module.exports = ApplicationWindow;

      SearchBar.js

      //Application Window Component Constructor
function SearchBar() {

    //create component instance
    var self = Ti.UI.createWindow({
        backgroundColor:'#ffffff'
    });

    //construct UI
    var searchBar = Titanium.UI.createSearchBar({
        barColor:'#000', 
        showCancel:true,
        height:43,
        top:0,
    });
    searchBar.hide();

    // create main tableview
    var listView = Ti.UI.createTableView({
        scrollable: true,
        search: searchBar
    });

    self.add(listView);

    //show
    self.addEventListener('open', function(e) {
        setTimeout(function(e){
            searchBar.show();
        }, 1000) ;
    });

    return self;
}

//make constructor function the public component interface
module.exports = SearchBar;
