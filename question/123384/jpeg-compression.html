    • Sorry didn't see the comment system.

      This is the code I am using to shoot and upload an image on an Android phone. It results in a file of 613kb in size on the server. The same image, saved from an image program with jpeg compression at 50% results in a file of 85kb. Which is usable over a 2G cellular network. So I guess there must be some way of applying jpeg compression on the handset before it is uploaded ?

      Thanks again.

      var win = Ti.UI.currentWindow;

      var button1 = Titanium.UI.createButton({
      title:'Capture',
      textAlign:'center',
      width:'auto',
      top:85,
      left:20
      });

      var filename = "test.jpg";

      button1.addEventListener('click',function(e) {

      Titanium.Media.showCamera({
    success:function(event)
    {
        var note = Titanium.UI.createNotification({
            duration: 3500,
            message: "Attempting to upload photo -- Please wait."
        });
        note.show();

      var image = event.media;

      var resize = Titanium.UI.createImageView ({
            width:1024,
            height:768,
            image:image
        }); 

      var image=resize.toImage().media;

      var payload = {
      "media": image,
      "name": filename
      };

      xhr = Ti.Network.createHTTPClient();
      xhr.open("POST","http://myurl/upload.php");
      xhr.send(payload);
      xhr.onload = function() {

      Ti.API.info(this.responseText);

      };
      },
      error:function(error)
      {
      if (error.code == Titanium.Media.NO_CAMERA)
      {
      alert('Device does not have video recording capabilities');
      }
      else
      {
      alert('Unexpected error: ' + error.code);
      }
      }
      });
      });

      win.add(button1);
