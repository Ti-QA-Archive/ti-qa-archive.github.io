Animation on Android

I've got an animation issue - working on iOS not on Android . On Android the animation doesn't work until I trigger it twice, at which time it happens twice. iOS it works fine.

Here is my code:

    detailsIcon.addEventListener('click',function(e){
        var addlInfo = additionalInfo(placeObj);
        addlInfo.zIndex=999;
        addlInfo.left = util.getScreenWidth();
        placeWin.addlInfo = true;
        placeWin.addlInfoView = addlInfo;
        addlInfo.addEventListener('postlayout',slideIn);

        function slideIn(){
            Ti.API.info("Slide In called");        
            addlInfo.animate({left:0,duration:300},function(){
                addlInfo.left = 0;
                Ti.API.info("Slide In Animation complete");           //on android this only occurs the second time I click detailsIcon - and then it happens twice
            });
            addlInfo.removeEventListener('postlayout',slideIn);
            Ti.API.info("SlideIn function complete");
        };

        placeWin.details.add(addlInfo);

    });
