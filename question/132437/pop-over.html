    This simply mean treat window as dialog window … hope this idea works for you.

    • Call popup.open(); on some event . Follow the code as an example.

      var popup = Titanium.UI.createWindow({

      width:200,
height: 200,
backgroundImage: 'images/popup_bg.png'

      });
      var label = Titanium.UI.createLabel({

      top:0,
text:'Information',
height:'auto',
width:'auto',

textAlign:'center'

      });

      var btndone = Titanium.UI.createButton({

      top: 150,
width: 97,
height: 47,
title :'OK'

      });

      popup.add(btndone);
      popup.add(label);
      btndone.addEventListener('click', function(){

      // do code here on popup close
popup.close();

      });
