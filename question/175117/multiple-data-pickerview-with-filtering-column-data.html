example:

column #1 : fruit, vegetable, animal

every i choose fruit on column #1, only mango and banana will show in column #2, and so on

    Hi,

    Thank you for your concern.

    I have tried to solve your problem by using the following code. I hope, it will help you. Please check the below code.

    index.xml

    <Alloy>
    <Window backgroundColor="white" layout="vertical" exitOnClose="true">
        <Picker id="picker" top="50" selectionIndicator="true" useSpinner="true">
            <Column id="column1">
                <Row title="Fruit"/>
                <Row title="Vegetable"/>
                <Row title="Animal"/>
            </Column>
            <Column id="column2">
                <Row title="Bananas"/>
                <Row title="Strawberries"/>
            </Column>
        </Picker>
    </Window>
</Alloy>

    index.js

    var fruit = [ 'Bananas', 'Strawberries' ];
var vegetable=['spinach', 'coleslaw'];
var animal=['cat', 'dog'];

$.picker.addEventListener('change',function(e){
    if(e.columnIndex===0)
    {
        if(e.rowIndex===0)
        {
            setRow(fruit);
        }
        else if(e.rowIndex===1)
        {
            setRow(vegetable);
        }
        else if(e.rowIndex===2)
        {
            setRow(animal);
        }
    }
});

$.picker.setSelectedRow(0, 0, false); 

function setRow(value)
{
    for(var i=0, ilen=$.column2.rowCount; i<ilen; i++){
        $.column2.rows[i].title= value[i];
    }
    $.picker.reloadColumn($.picker.columns[1]);
}

$.index.open();

    • Hi,

      You can also use switch….case to solve the problem.

      $.picker.addEventListener('change',function(e){
    if(e.columnIndex===0)
    {
        switch(e.rowIndex)
        {
            case 0:
                setRow(fruit);
                break;
            case 1:
                setRow(vegetable);
                break;

            case 2:
                setRow(animal);
                break;
        }
    }
});
