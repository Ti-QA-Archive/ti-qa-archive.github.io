  • Darius,

    I know it's a loooong time since you posted your need, but I just read your thread and wrote a very simple Webview control that may help whoever need it. You probably found a solution for this problem already, but I think this may help a feel "rookies" like myself. :)

    My best regards!

    //Set your WebView.top to '10%' and add this funtion to you your Window 'yourWindow(WebNavBar(webview));' and you're golden.

function WebNavBar(webview)
{
    var barraNav = Ti.UI.createView({
            top:'0%',
            height:'10%',
            width: '100%',
            backgroundImage: '/images/WebNav/webview_navbar_background.png',
            backgroundColor:'#fff', //Branco
            borderRadius: 3,
            borderWidth: 1,
            zIndex: -1,
            borderColor: '#575c58'
        });

        var btn_back = Ti.UI.createButton({
            title: 'Back',
            color: "#fff",
            textAlign : Ti.UI.TEXT_ALIGNMENT_RIGHT,
            font : { 
                fontSize: '10dip', 
                font: 'sans-serif',
                fontWeight:'bold'
            },
            image: '/images/WebNav/backButton.png',
            left: '2%',
            right: '85%',
            top: '15%',
            bottom: '15%'
        });
        barraNav.add(btn_back);

        btn_back.addEventListener('click', function(e){
            if (webview.canGoBack()) {
                   webview.goBack();
               }
        });

        var btn_foward = Ti.UI.createButton({
            title: 'Fwd',
            color: "#fff",
            textAlign : Ti.UI.TEXT_ALIGNMENT_LEFT,
            font : { 
                fontSize: '10dip', 
                font: 'sans-serif',
                fontWeight:'bold'
            },
            image: '/images/WebNav/fowardButton.png',
            left: '18%',
            right: '69%',
            top: '15%',
            bottom: '15%'
        });
        barraNav.add(btn_foward);

        btn_foward.addEventListener('click', function(e){
            if (webview.canGoForward()) {
                webview.goForward();
            }
        });

        var logo = Ti.UI.createImageView({
            url:'/images/logo_dzai.png',
            left: '40%',
            top: '10%',
            height:'90%',
            width:'25%'
        });
        barraNav.add(logo);


        var btn_refresh = Ti.UI.createButton({
            image: '/images/WebNav/reload.png',
            right: '2%',
            left: '85%',
            top: '10%',
            bottom: '10%'
        });
        barraNav.add(btn_refresh);

        btn_refresh.addEventListener('click', function(e){
            webview.reload();
        });

        return barraNav;
}
