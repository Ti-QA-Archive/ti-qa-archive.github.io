buttonCoto.addEventListener('click', function(e){

    Hi Eric, Anthony is right, Change your code with this one. 

    var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});

var coto = 0;
var buttonCoto = Titanium.UI.createButton({ 
title:'Cotoletta',
 font:{fontSize:20},
 verticalAlign: 'bottom',
 backgroundImage:'coto.jpg',
 width:'200',
 height:'200',
 top: 50, 
left: 10
 });
buttonCoto.addEventListener('click', function(e){
 coto = coto + 1 ;
 labelCoto.text = 'Cotoletta:'+coto;
});
var labelCoto = Ti.UI.createLabel({
 color:'#999',
 text:'Cotoletta:' + coto, 
font:{fontSize:40,fontFamily:'Helvetica Neue'},
width:'auto',
 top: 50, 
left: 10
 });

win1.add(buttonCoto);
win1.add(labelCoto);
win1.open();

    It is tested.. Hope it will help..
