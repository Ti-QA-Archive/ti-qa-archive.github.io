  • hi,

    as Sarafaeaz said you can not use both events together but i have a work around for that. just try below code.

    you can use single tap event or touchend event instead of click event

    function getType(e) {
    alert(e.type);
}

var win = Ti.UI.createWindow({

});

// Create a WebView
var aWebView = Ti.UI.createWebView({
    backgroundColor : 'red'
});

//use touchend OR singletap event.
//aWebView.addEventListener('touchend', getType);

aWebView.addEventListener('singletap', getType);
aWebView.addEventListener('longpress', getType);

// Add to the parent view.
win.add(aWebView);

win.open();
