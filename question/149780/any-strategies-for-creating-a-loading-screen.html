  • Hi Jigar ,

    if you want to have only the label of loading screen, the "Titanium.UI.ActivityIndicator" is sufficient, but if you want to put a view below the label, for make it more readable , this is the way : 

    var actIndView = Ti.UI.createView({
    height:175,
    width : 150 ,
    backgroundColor:'#000',
    opacity:0.60
})  

var actInd = Titanium.UI.createActivityIndicator({
    color   :'white',
    style : Ti.UI.ActivityIndicatorStyle.BIG,
    message :'Attendere...',
});

actIndView.add(actInd)
