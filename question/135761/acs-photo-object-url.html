  • "urls" is an object literal, AKA a dictionary. Use a key accessor, such as dict.key or dict['key'], to get the URLs out of it.

    Cloud.Photos.query(function (e) {
    if (e.success) {
        if (e.photos.length == 0) {
            alert('No photos!');
        }
        else {
            for (var i = 0, l = e.photos.length; i < l; i++) {
                alert(e.photos[i].filename + ': '
                    + e.photos[i].urls.square);
            }
        }
    }
    else {
        alert(e);
    }
});
