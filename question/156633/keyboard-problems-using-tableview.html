    • nope still having the same problem , my fields need the default keyboard layout with option to add numbers.
      here's my code.

      <code>

      var TheTable = Titanium.UI.createTableView({
      separatorStyle : 'none',
      backgroundColor : "transparent",
      top : 55,
      style : Ti.UI.iPhone.TableViewStyle.GROUPED,
      scrollable : true,
      selectionStyle : Ti.UI.iPhone.TableViewCellSelectionStyle.NONE

      });

      var data = [];
      var textfield = [];
      var label = [];

      for ( i = 0; i < formdata.length; i++) {

      var row = Titanium.UI.createTableViewRow({

    height : 50
    //        customtest : formdata[i].id

});

label[i] = Ti.UI.createLabel({
    font : {
        fontFamily : 'Arial Hebrew',
        textAlign : 'right',

        fontSize : 18,
        fontWeight : 'bold'
    },

    //color : '#004281'

    text : formdata[i].name,
    right : 7
});


    textfield[i] = Titanium.UI.createTextField({
        color : '#336699',
        //top : 10,
        right : '110',
        width : '200',

        backgroundColor : 'none',

        textAlign : 'right',
        font : {
            fontSize : 20,
            //        fontWeight : 'bold',
            textAlign : 'right',
            fontFamily : 'Helvetica Neue'

        },

        hintText : formdata[i].label,
        //        passwordMask:true,

      keyboardType : Titanium.UI.KEYBOARD_DEFAULT,
      returnKeyType : Titanium.UI.RETURNKEY_DONE,
      borderStyle : Ti.UI.INPUT_BORDERSTYLE_NONE,
      autocapitalization: Titanium.UI.TEXT_AUTOCAPITALIZATION_NONE

          });

      textfield[i].autocorrect = false;

      textfield[i].addEventListener('focus', function() {

      TheTable.scrollable = true;

      });

      textfield[i].addEventListener('blur', function() {

      TheTable.scrollable = false;

      });

      row.add(label[i]);
row.add(textfield[i]);

data.push(row);

      };

      var sendbutton = Titanium.UI.createButton({
      backgroundImage : '../images/main/SEND.png',

      top : 285,
width : 150,
height : 35

      });

      TheTable.setData(data);

      contact.add(TheTable);
