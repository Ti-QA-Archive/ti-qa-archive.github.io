    hi Raffaele,

    what you say about this…..?

    var patt =new RegExp('^(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$','gi');
var txt = Ti.UI.createTextField({
    top : 20,
    left : 10,
    width : 200,
    height : 40,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED 
});

txt.addEventListener('return',function(e){
    var v = txt.value;
    var m = v.match(patt);
    if(m != null){
        alert(' << SUCCESS >> ');
    }
});


win.add(txt);
