  • mRow = Titanium.UI.createTableViewRow({
               height:'20%',
               width:'96%',                             
            });

    after building the row i'm putting a view on it,

     view = Titanium.UI.createView({
                backgroundColor:'#ffffff',
                borderRadius:5,               
                width:'96%',               
                height:'94%',
                left:'2%',
                top:'3%',                              
            });
mRow.add(view);
