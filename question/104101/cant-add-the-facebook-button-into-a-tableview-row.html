  • I got this working in iOS. Here's how i'm adding it. I'm not sure the main differences from yours, but follow it like mine to see if helps.

    var fbutton = Titanium.Facebook.createLoginButton ({
    'style':'wide',
    'apikey':facebook_consumer_key,
    'secret':facebook_consumer_secret,
    'perms':'read_stream, publish_stream, manage_pages'
    });
    fbutton.visible = true;

    var Frow = Ti.UI.createTableViewRow({header:'Facebook Login'});
    Frow.add(fbutton);
    tableview.appendRow(Frow);
    win.add(tableview);
