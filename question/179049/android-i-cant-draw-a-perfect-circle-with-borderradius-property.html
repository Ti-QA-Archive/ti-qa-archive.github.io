    backgroundColor: 'black',
    borderRadius: 150,
    width: 300,
    height: 300
});
w.add(v);
w.open();

  • Hi, i think this is an elegant solution for create a circle-view for Android in Titanium:

    //Create a view to wrap the content
var container = Ti.UI.createView({
    width: 256,
    height: 256
});

//Create a circle-view, with borderWidth == borderRadius    
container.add(Ti.UI.createView({
    width: 256,
    height: 256,
    borderRadius: 128, //Half of width 
    borderColor: "#ffffff",
    borderWidth: 128 //Half of width 
}));

//Add the content, with same width and height. It will be label, button, image, etc
container.add(Ti.UI.createLabel({
    width: 256,
    height: 256,
    font:{
        fontFamily: 'FontAwesome',
        fontSize: 128
    },
    color: "#000000",
    text: String.fromCharCode(0xf022),
    textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER
}));

//Remember to add to your window or view
yourWindowOrView.add(container);

    You will see this:

    Now, check the difference with those others codes:

    var container = Ti.UI.createLabel({
    width: 256,
    height: 256,
    font:{
        fontFamily: 'FontAwesome',
        fontSize: 128
    },
    color: "#000000",
    text: String.fromCharCode(0xf022),
    textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER,
    borderRadius: 128,
    backgroundColor: "#ffffff"
});

//Remember to add to your window or view
yourWindowOrView.add(container);

    And if you add to the last code the borderWidth property (In this case, borderWidth: 20) you will see something like this:
