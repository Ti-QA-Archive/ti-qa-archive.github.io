    Hi,

    if you want to store globally then you can do as following..

    function doClick(e) { 
    alert("in");
    Alloy.Globals.credentials ={
        username : username,
        password : password
    };
     var getwin = Alloy.createController("getvalue").getView().open();
    }

    so when you click , it will store as json object and then you can access anywhere in your app.

  • Hello Narasimha ,

    You can pass argument like this :

    var  username = $.un.value;
var  password = $.ps.value;

var args = {
    username : username,
    password : password
};
function doClick(e) { 
    alert("in");
     var getwin = Alloy.createController("getvalue",args).getView().open();
    } 

$.index.open();

    ###getValue.js

    var args = arguments[0];

$.getun.text = args.username;
$.getps.text = args.username;

    i hope this helps. thanks.
