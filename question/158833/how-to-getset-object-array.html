    Hi

    JavaScript does not support the traditional class as you may have used in the past. Because of this there is more than one way to define a class;

    • Using a function
      • Methods defined internally
      • Methods defined with prototype
    • Using object literals
    • Singleton using a function

    The following looks closest to what you are used to so I have changed your example to use the "Using a function - Methods defined with prototype" approach;

    // class
function Car(obj) {
    this.id = obj.id || 0;
    this.brand = obj.brand || '';
    this.series = obj.series || '';
    this.type = obj.type || '';
    this.color = obj.color || '';
} 
Car.prototype.print = function () {
    return this.id + ', ' + this.brand + ', ' + this.series + ', ' + this.type + ', ' + this.color;
};

// usage;
var arrCar = [];
var car1 = new Car({
    id: 1,
    brand: 'Honza',
    series: 'Cixic',
    type: 'suv',
    color: 'black'
});
arrCar.push(car1);

arrCar[0].series = 'sedan';

var str = '', intCar = 0, intCars = arrCar.length;
for (intCar = 0; intCar < intCars; intCar++) {
    str += arrCar[intCar].Print() + '\n\r'; //print with enter
}
alert(str);

    See how that goes.
