Connection Refused with HTTPClient

I've used this many times before…not sure what is going on now. I keep getting a connection refused error when I call open() on an xml file.

app.js

var win = Ti.UI.createWindow({
    url:'main_windows/main.js'
});
win.open();

main.js

function loadXML()
{
    xhr = Ti.Network.createHTTPClient();
    xhr.open('GET','test2.xml');

    xhr.onload = function()
    {
        switch (this.status)
        {
            case 200:
                parseXMLInfo(this.responseXML);
                break;
            case 404:
                alert('XML File not found');
                break;
            default:
                alert('Something happened :( ' + this.status);
        }
    };

    xhr.onerror = function(event)
    {
        alert(event);
    };
    xhr.send();
}
loadXML();

This is the error I am getting.

error = "Error Domain=ASIHTTPRequestErrorDomain Code=1 \"A connection failure occurred\" UserInfo=0x693a620 {NSUnderlyingError=0x693a230 \"The operation couldn\U2019t be completed. Connection refused\", NSLocalizedDescription=A connection failure occurred}";
source = "[object TiNetworkClient]";
type = error;
}

test2.xml is in the same folder as main.js…I'm not doing anything different than how I would normally do it. I am using 1.5.1 sdk and 4.2 iphone sdk
