var url = "http://192.168.1.171/Api/UserLogin?version=1.0&datatype=Json";
var client = Ti.Network.createHTTPClient({
onload : function(e) {
var json = JSON.parse(this.responseText);
json = JSON.stringify(json);
alert(json);
},
onerror : function(e) {
alert(e.error);
},
autoEncodeUrl: true,
async: true,
timeout : 10000
});

client.setRequestHeader('Content-Type', "application/json");
client.open("POST", url);
client.send(loginPost);
}

public string UserLogin(string type, TPersonInfo model)
{
if (string.IsNullOrEmpty(type))
{
type = "Json";
}
