Create ImageView with Functions/Methods

Dear readers,

How can I create an ImageView with own defined Functions/Methods? I have a app.Designer.js file that I use in app.js by specifying var Designer = require(&#39;ui&#x2F;app.Designer&#39;);. But I don't get the own defined Functions to work.

app.Designer.js

exports.ImageLogo = function() {
    var self = Titanium.UI.createImageView();

    self.image = '/images/Logo.jpg';
    self.right = 0;
    self.top = 0;

    self.Redraw = function() {
        var ratio = 500 / 127;

        if(guiHelper.TitOrientation() === 'portrait'){
            self.width = 0.75 * guiHelper.TitWidth();
            self.height = 0.75 * guiHelper.TitWidth() / ratio;
        }
        else if (guiHelper.TitOrientation() === 'landscape')
        {
            self.width = 0.5 * guiHelper.TitWidth();
            self.height = 0.5 * guiHelper.TitWidth() / ratio;
        }
    }
    return self;
};

app.js

var Designer = require('app.Designer');

var ImageLogo = Designer.ImageLogo(); //This works and I get an ImageView
ImageLogo.Redraw(); //This gives me an error "has no method 'Redraw'

Help is greatly appreciated!

Best regards,
Rémy Samulski

  • Hello Remy,

    You need to use Redraw as a function.

    exports.ImageLogo = function() {
    var self = Titanium.UI.createImageView();

    self.image = '/images/Logo.jpg';
    self.right = 0;
    self.top = 0;

    return self;
};


    exports.Redraw = function(image) {
        var ratio = 500 / 127;

        if(guiHelper.TitOrientation() === 'portrait'){
            image.width = 0.75 * guiHelper.TitWidth();
            image.height = 0.75 * guiHelper.TitWidth() / ratio;
        }
        else if (guiHelper.TitOrientation() === 'landscape')
        {
            image.width = 0.5 * guiHelper.TitWidth();
            image.height = 0.5 * guiHelper.TitWidth() / ratio;
        }
    }

    Let me know if it works
