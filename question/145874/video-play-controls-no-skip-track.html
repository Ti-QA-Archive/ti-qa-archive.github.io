Video Play controls no skip track

Hi, I can't seem to get any callback on the skip button of the fullscreen controls of the video player. I want to skip the current playing video and move onto the next (see code). With the controls for fullscreen the skip button is clickable but does not fire anything.

// Application Window Component Constructor
function MovieAll() {
    var mp4_array = ['12345.m4v', 'GrandOldDuke.m4v', 'HeyDiddleDiddle.m4v', ];

    var current_index =  0;

    var self = Ti.UI.createWindow({
        modal : true,
        navBarHidden: true
    });

    var activeMovie = Titanium.Media.createVideoPlayer({
        backgroundColor:'#111',
        mediaControlStyle:Titanium.Media.VIDEO_CONTROL_FULLSCREEN,
        allowsAirPlay:true,
        keepScreenOn: true,
        contentURL: 'video/'+ mp4_array[0],
        autoplay: true,
        mediaTypes: Titanium.Media.VIDEO_MEDIA_TYPE_VIDEO
    });

    self.add(activeMovie);

    activeMovie.show();
    activeMovie.play();

    activeMovie.addEventListener('complete', function(e) {
        current_index++;
        if(current_index < mp4_array.length) {
            // play the next mp4 in the array
            activeMovie.stop();
            activeMovie.contentURL = 'video/' + mp4_array[current_index];
            activeMovie.play();
            Ti.API.info('Videos, Next, ' + current_index);

        } else {
            Ti.API.info(JSON.stringify(e));
            activeMovie.stop();
            Ti.UI.iPhone.hideStatusBar();
            self.close();
            self.visible = false;
        }
    });

    return self;
}

module.exports = MovieAll;

It seems that clicking "done" is the only way to skip the video like in my example above, but this is NOT ideal.
