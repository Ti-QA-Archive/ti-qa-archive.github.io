    Put your request code into a function, like this;

    var url = "http://103.21.171.61/pakarga/topup.svc/getfaqjson?account=";

var win = Ti.UI.createWindow();

var scrollView = Ti.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    showVerticalScrollIndicator : true,
    showHorizontalScrollIndicator : false,
    height : '100%',
    width : '100%',
    top : 0
});


var table = Ti.UI.createTableView();

scrollView.add(table);

win.add(scrollView);


function getInvoice(invoiceId) {

    var tableData = [];
    //variables
    var json, faq, varfaq, i, row, nameLabel, nickLabel;

    //xhr
    var xhr = Ti.Network.createHTTPClient({

        onload : function() {
            // Ti.API.debug(this.responseText);
            json = JSON.parse(this.responseText);

            //for loop
            for ( i = 0; i < json.faq.length; i++) {
                varfaq = json.faq[i];

                //Row
                row = Ti.UI.createTableViewRow({
                    height : 'auto'
                });

                var nameLabel = Ti.UI.createTextArea({
                    value : varfaq.question,
                    font : {
                        fontSize : 15,
                        fontFamily : "Gotham"
                    },
                    color : '#000',
                    shadowColor : '#aaa',
                    shadowOffset : {
                        x : 0,
                        y : 1
                    },
                    backgroundColor : '#dcdcdc',
                    editable : false,
                    width : "auto",
                    height : "200dp",
                    left : "10dp",
                    top : "5dp"
                });

                var nickLabel = Ti.UI.createTextArea({
                    value : '"' + varfaq.answer + '"',
                    font : {
                        fontSize : 15,
                        fontFamily : "Gotham"
                    },
                    color : '#000',
                    shadowColor : '#aaa',
                    shadowOffset : {
                        x : 0,
                        y : 1
                    },
                    backgroundColor : '#dcdcdc',
                    editable : false,
                    width : "auto",
                    height : "100dp",
                    left : "10dp",
                    bottom : "5dp"
                });

                row.add(nameLabel);
                row.add(nickLabel);
                tableData.push(row);
            }

            table.setData(tableData);
        },

        //on error
        onerror : function(e) {
            Ti.API.debug("STATUS: " + this.status);
            Ti.API.debug("TEXT: " + this.responseText);
            Ti.API.debug("ERROR: " + e.error);
            alert('There was an error retrieving the remote data. Try again.');
        },
        timeout : 5000

    });

    xhr.open("GET", url + invoiceId);
    xhr.send();

}

win.open();

getInvoice('PK10001');
getInvoice('PK10002');
