    "I want to print length of only key's for key1, key2"

    for (var i=0; i < keyarray.length; i++) {
    for(key in keyarray[i]){
        Ti.API.info('key: ' + key + ' length: ' + keyarray[i][key].length);
    }
}

  • Hi narasimha,

    you have to stringify the object with JSON.stringify().

    And your keyarray has no yGridkeyarray, so testvar will be undefined.

    var colorSet = [
                {id : 'array_key', image : '/images/p2.jpg', datum : ' Dec 2012'}, 
                {id : 2, image : '/images/p3.jpg', datum : ' Jun 2012'}, 
                {id : 3, image : '/images/cafe.jpg', datum : ' Feb 2013'}
               ];
var colorSet2 = [
                {id : 'array_key', image : '/images/cafe.jpg', datum : ' Dec 2012'}, 
                {id : 2, image : '/images/p2.jpg', datum : ' Jun 2012'}, 
                {id : 3, image : '/images/cafe.jpg', datum : ' Feb 2013'}
                ];
var keyarray = [];

keyarray = [
            {key1 : colorSet}, 
            {key2 : colorSet2}
           ];

var yGrid = keyarray.length;
var yGridkey = keyarray[0].key1.length;

for (var i = 0; i < yGrid; i++) {

    var yGridkeyarray = keyarray[i];

    // Ti.API.info("yGridkeyarray:" + yGridkeyarray);
    Ti.API.info("yGridkeyarray:" + JSON.stringify(yGridkeyarray));

    // keyarray has no yGridkeyarray!!
    var testvar = keyarray[i].yGridkeyarray;

    Ti.API.info("testvar:" + testvar);
}
