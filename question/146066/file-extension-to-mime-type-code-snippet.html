    Function to return the mime type for a given file extension (including the '.'):

    var mimeTypes = null; //only fill once
function getMimeType(extension) {

    if (mimeTypes == null) {
        var getMimeTypes = require('/mimetypes');
        mimeTypes = new getMimeTypes();
    }

    if (mimeTypes[extension]) {
        return mimeTypes[extension];
    }
}

  • Hi Mark

    A tweak on your code, this keeps all the functionality and logic of mime types inside the CommonJS module.

    mimetypes.js

    var mimeType = {
    ".123": "application\/vnd.lotus-1-2-3",
    ".3dml": "text\/vnd.in3d.3dml",
    ".3g2": "video\/3gpp2"
    //... (object too big to paste here :( };
};
function exists(strExt) {
    return ((typeof (mimeType[strExt]) !== 'undefined') ? true : false);
}
function get(strExt) {
    return mimeType[strExt];
}
exports.exists = exists;
exports.get = get;

    You then call the functions like this;

    var Mime = require('/mimetypes');
alert(Mime.exists('.123')); // returns true
alert(Mime.exists('.pdf')); // returns false
alert(Mime.get('.123')); // returns "application/vnd.lotus-1-2-3"
alert(Mime.get('.pdf')); // returns undefined (nothing in alert)

    This way all the logic can be neatly stowed away as well as the lookup.
