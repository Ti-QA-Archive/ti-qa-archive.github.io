    You could use like below.

    var imageView = Titanium.UI.createImageView();
imageView.addEventListener("load",getImageSize);
imageView.setImage("http://developer.appcelerator.com/assets/img/devlink/profile_badges/tcad_tab_small.png");
window.add(imageView);

function getImageSize() {
    var blob = imageView.toBlob();
    if (blob === null) {
        Ti.API.info("unable to get image size");        
    } else {
        Ti.API.info(blob.width + "x" + blob.height);
    }
}
