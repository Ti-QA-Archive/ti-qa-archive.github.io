    Hi

    Animations need to be performed using an animation object, something like will work for you;;

    var aniHide = Ti.UI.createAnimation({
    top: -100,
    curve: Ti.UI.ANIMATION_CURVE_EASE_IN_OUT,
    duration: 400
});
var aniShow = Ti.UI.createAnimation({
    top: 0,
    curve: Ti.UI.ANIMATION_CURVE_EASE_IN_OUT,
    duration: 400
});
home.ui.txtUserPassword.addEventListener('focus', function() {
    home.ui.LoginScreen.animate(aniHide);
});
home.ui.txtUserPassword.addEventListener('blur', function() {
    home.ui.LoginScreen.animate(aniShow);
});
