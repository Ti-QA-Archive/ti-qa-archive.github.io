  • Hello Djamel,

    you can get the ScrollView ContentHeight by last object of your scrollview

    Just try this Demo

    var win = Ti.UI.currentWindow;

var scrollview = Ti.UI.createScrollView({
    height : 300,
    width : 320,
    top : 0,
    left : 0,
    contentHeight : 'auto',

});

// Create a Button.
var GetHeight = Ti.UI.createButton({
    title : 'GetHeight',
    height : 25,
    width : 150,
    top : 310,
    left : 120
});

// Listen for click events.
GetHeight.addEventListener('click', function(e) {

    var img = scrollview.children[(scrollview.children.length - 1)];
    var ht = img.top + img.toImage().height;
    alert('ScrollView ContentHeight is > ' +ht);

});

// Add to the parent view.
win.add(GetHeight);

var imgTop = 2; 
for (var i = 0; i < 10; i++) {
    var imgView = Ti.UI.createImageView({
        top : imgTop,
        bottom : 2,
        height : 'auto',
        width : 'auto',
        image : 'KS_nav_ui.png'
    });
    scrollview.add(imgView);

    imgTop = imgTop + imgView.toImage().height;
}

win.add(scrollview);
