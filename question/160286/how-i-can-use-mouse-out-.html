  • var w = Ti.UI.createWindow();
var t = Ti.UI.createTextField({
    width : 200,
    height : 50,
    backgroundColor : 'white',
    color : 'black'
});

w.addEventListener('click', function() {
    t.blur();
});
w.add(t);

w.addEventListener('open', function() {

    t.focus();
});
w.open();
