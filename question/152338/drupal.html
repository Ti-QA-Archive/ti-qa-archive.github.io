  • Here is a login into an api using titanium and i think it would work for you too

    if (username.value !== '' && pass.value !== '') {
            activityIndicator.show();
            var xhr = Ti.Network.createHTTPClient({
                onload : successHandler,
                onerror : errorHandler,
                timeout : 1000 * 15
            });
            var url = 'http://www.gmail.com';
            var authstr = 'Basic ' + Ti.Utils.base64encode(username.value + ':' + pass.value);
            xhr.open('GET', url);
            authstr = authstr.replace(/(\r\n|\n|\r)/gm, "");
            xhr.setRequestHeader('Authorization', authstr);
            xhr.setRequestHeader("Content-Type", "text/html; charset=utf-8");
            xhr.send();
        } else {
            alert("User not defined")
}
