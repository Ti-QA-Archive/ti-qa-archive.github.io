textfield value and app crash

Hi, in my android app i have a textfield with this code, can you tell me why my app crashes?

var piu_prima_cifra= Ti.UI.createButton({
    top:100,
    height:200,
    width:200,
    title:'+',
});
win.add(piu_prima_cifra);
var number_prima_cifra=1;
var primacifra= Ti.UI.createTextField({
    top:300,
    height:200,
    width:200,
    value:1,
    textAlign:'center',
    font:{fontSize:40, fontStyle:'italic'},

});
win.add(primacifra);

piu_prima_cifra.addEventListener("click",function(e){
    number_prima_cifra++;
    primacifra.setValue(number_prima_cifra);
    //HERE MY APP STOPS
});

    You have to convert your integer to a string before setting the text filed value:

    primacifra.setValue(String(number_prima_cifra));
