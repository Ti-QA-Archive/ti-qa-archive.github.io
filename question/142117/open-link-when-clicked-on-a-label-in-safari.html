    //create label (or button etc)
    var label = Ti.UI.createLabel({
    text:'http://www.google.com.pk/', /instead of google give your link/
    backgroundColor:'pink'
    });

    //add event listener for label click
    label.addEventListener('click', function(e)
    {
    //open link in safari - application will close
    Titanium.Platform.openURL('http://www.soft4u.be');
    });
    win.add(label);

    • Just copy past this code in app.js file

      var tabGroup= Ti.UI.createTabGroup();
      var win1= Ti.UI.createWindow({
      //modal:true,
      title:'Open link in Safari'
      //url:'test.js'
      })

      //create label (or button etc)
      var label = Ti.UI.createLabel({
      / give your link instead og google here/
      text:'http://www.google.com.pk/',
      backgroundColor:'pink'
      });

      //event listener
      label.addEventListener('click', function(e)
      {
      //open link in safari - application will close
      Titanium.Platform.openURL('http://www.soft4u.be');
      });
      win1.add(label);

      var tab1= Ti.UI.createTab({
      title:'Label link',
      window:win1,
      });
      tabGroup.addTab(tab1);
      tabGroup.open();
