ApplicationTabGroup.js (open windows with tabGroup)

function ApplicationTabGroup(Window) {
    //create module instance
    var self = Ti.UI.createTabGroup();

    //create app tabs
    var home_view = require('ui/common/HomeView'),
        novedades_view = require('ui/common/novedadesView');
        prueba_view = require('ui/common/pruebasView');
        favoritos_view = require('ui/common/FavoritosView');
        settings_view = require('ui/common/SettingsView');

    //create app tabs
    var homeWin = new home_view({title:'Home',tabGroup:self}),
        novedadesWin = new novedades_view({title:'Novedades',tabGroup:self}),
        pruebasWin = new prueba_view({title:'Prueba',tabGroup:self}),
        favoritosWin = new favoritos_view({title:'Favoritos',tabGroup:self});
        settingsWin = new settings_view({title:'Settings',tabGroup:self});

    // Tab HOME
    var homeTab = Ti.UI.createTab({
        title: 'Home',
        icon: '/images/logo.png',
        window: homeWin
    });
    homeWin.containingTab = homeTab;

    // Tab NOVEDADES
    var novedadesTab = Ti.UI.createTab({
        title: 'Novedades',
        icon: '/images/aside_menu.png',
        window: novedadesWin
    });
    novedadesWin.containingTab = novedadesTab;

    // Tab PRUEBAS
    var pruebasTab = Ti.UI.createTab({
        title: 'Prueba',
        icon: '/images/aside_menu.png',
        window: pruebasWin
    });
    pruebasWin.containingTab = pruebasTab;

    // Tab FAVORITOS
    var favoritosTab = Ti.UI.createTab({
        title: 'Favoritos',
        icon: '/images/star.png',
        window: favoritosWin
    });
    favoritosWin.containingTab = favoritosTab;

    // Tab SETTINGS
    var settingsTab = Ti.UI.createTab({
        title: 'Settings',
        icon: '/images/preferencias.png',
        window: settingsWin
    });
    settingsWin.containingTab = settingsTab;

    self.addTab(homeTab);
    self.addTab(novedadesTab);
    self.addTab(pruebasTab);
    self.addTab(favoritosTab);
    self.addTab(settingsTab);

    // Auto-inicia en el primer tab-bar (HOME) al iniciar la app
    self.setActiveTab(0);
    self.model = Ti.Platform.model;
    return self;
};

module.exports = ApplicationTabGroup;
