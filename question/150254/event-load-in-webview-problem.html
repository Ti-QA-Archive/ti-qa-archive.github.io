 * Win

web.addEventListener('load', function(e){

});  

/*
 * Beforeload

web.addEventListener('beforeload', function(e){

win.add(web);
win.open();

  • Hi, Alexey Mezhevikin

    Event load is called only one times.

    This code is working perfectly for me.

    var win = Ti.UI.createWindow();

var url = 'http://habrahabr.ru';

var web = Ti.UI.createWebView({
    top: 0,
    width: 320,
    height:480,
        url: url
});

win.add(web);

web.addEventListener('load', function(e){
    Ti.API.info('load');
});  

web.addEventListener('beforeload', function(e){
     Ti.API.info('before');
}); 

 win.open();
