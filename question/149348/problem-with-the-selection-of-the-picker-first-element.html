Problem with the selection of the picker first element

Hello evrybody,

I've created a picker using this code:


var data = []; 

data[0] = Titanium.UI.createPickerRow({title:"25", val:"25"});
data[1] = Titanium.UI.createPickerRow({title:"28", val:"28"});
data[2] = Titanium.UI.createPickerRow({title:"30", val:"30"});
data[3] = Titanium.UI.createPickerRow({title:"33", val:"33"});

mypicker.add(data);

var mypicker = Titanium.UI.createPicker({
    bottom:250
});

I got the selected item like this:



mypicker.addEventListener("change",function(e){

    total=e.row.val;
});

The problem is that when I selected the first element, I've got always undefined !!!

What is the problem with my code?

Application type: mobile
Titanium SDK: 3.0.2
Platform & version: Android 4.0.3
Device:physical device model

    i think you want to get the first when row value when user does not change picker so you can set first value to total value like below.

    var data = []; 

data[0] = Titanium.UI.createPickerRow({title:"25", val:"25"});
data[1] = Titanium.UI.createPickerRow({title:"28", val:"28"});
data[2] = Titanium.UI.createPickerRow({title:"30", val:"30"});
data[3] = Titanium.UI.createPickerRow({title:"33", val:"33"});

mypicker.add(data);

var mypicker = Titanium.UI.createPicker({
    bottom:250
});

setTimeout(function(e){
    total = mypicker.getSelectedRow(0).title;
},200);
