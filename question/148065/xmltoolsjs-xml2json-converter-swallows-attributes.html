XMLTools.js (xml2json converter) swallows attributes

Hello,

I try to use XMLTools.js to convert. Here XML and here JSON, I'am missing attributes. There are swallow.

var traverseTree = function(node) {
    var textOnly = true;
    var part = {};
    if (node.hasChildNodes()) {
        for (var ch_index = 0; ch_index < node.childNodes.length; ch_index++) {
            var ch = node.childNodes.item(ch_index);
            if (ch.nodeName == '#text' && ch.textContent.replace(/\n/g, '').replace(/ /g, '') == "")
                continue;
            //skip blank text element
            if (ch.nodeType == 3) {//Text Node
                return ch.textContent;
            } else {
                part = addToObject(part, ch.tagName, traverseTree(ch));
            }
        }
        textOnly = false;
    }
    if (node.hasAttributes()) {
        for (var att_index = 0; att_index < node.attributes.length; att_index++) {
            var att = node.attributes.item(att_index);
            part = addToObject(part, att.nodeName, att.nodeValue);
            part[att.nodeName] = att.nodeValue;
        }
        textOnly = false;
    }
    return part;
};

Especially I'am missing 'identifier'-property of 'simpleChoice' of 'choiceInteraction',

Any ideas?
