    change your label like this, you have it positioned too high, that is why it is not visible

    var label = Titanium.UI.createLabel({
    color:"#363636",
    text:"Pull down and release to refresh!",
    font:{fontSize:14,fontWeight:'bold'},
    width:'auto',
    textAlign:'center',
    bottom:15, // <---- NEW VALUE
    height:14
});
