  • I find a solution but I think that is not optimal, here is the code 

    var rows = db.execute('SELECT * FROM Multimedia');
    var dataArray = [];
    var donnees=[];
    var imgs=[];

    while (rows.isValidRow())  
    {  
        dataArray.push(''+rows.fieldByName('id_multimedia')); 
        imgs.push(rows.fieldByName('data_multimedia'));
        rows.next();
    };
    rows.close();
    for (var i =0; i<dataArray.length;i++){
        Titanium.API.info('DATA ARRAY : '+dataArray[i]);
        Titanium.API.info('DATA IMAGE : '+ imgs[i]);
        var imgView = Ti.UI.createImageView({
                 left : 20,
                image : imgs[i],
                   width : 'auto',
                height : 'auto'
        });
        var tvr = Titanium.UI.createTableViewRow({
            title:dataArray[i],hasChild:true,width : 200
            });
        tvr.add(imgView);
        donnees.push(tvr);
    };
