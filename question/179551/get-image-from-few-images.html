    Here is the code to get the current image.

    
var cur_index = 0;
$.image.images = [
    "/promotion1.jpg",
    "/promotion2.jpg",
    "/promotion3.jpg"
];

$.image.duration= 3000;
$.image.start();

$.image.addEventListener('change', function(e){
    cur_index = e.index;
});


$.image.addEventListener('click', function(e){
    Ti.API.info(e.source.images[cur_index]);
        // display the image which user clicked
});

    Hope this helps.
