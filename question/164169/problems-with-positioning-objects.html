    It's correct to use DPs, however this: 54.67dp is not ok :) Use integers.

    You should look in tiapp.xml and see if you have

        <property name="ti.ui.defaultunit">dip</property>

    or set it yourself, then all the dimensions will be in dp, without the need to write top:&#39;10dp&#39; but top:10 will be enough and the compiler will take care of it for you.

  • Hi andres guerrero, thanks for your question. You can try the code below. I've tested it and works fine for all devices.

    First create a user define module ui.js and use the code below

    var base = {
    width : 320,
    height : 480
};
var orgWidth = Titanium.Platform.displayCaps.platformWidth;
var orgHeight = Titanium.Platform.displayCaps.platformHeight;
var divide = base['width'] / orgWidth;
exports.fit= function(px) {
    return Math.round(px / divide);
};

    Then in your every window use it like,

    var ui = require('ui');
Titanium.UI.setBackgroundColor('#000');

var win= Ti.UI.createWindow({
 backgroundImage: '#000'
});

var anImageView = Ti.UI.createImageView({
    image : '/img.jpg',
    width : ui.fit(320),
    height : ui.fit(240),
    top : ui.fit(50),
    left : ui.fit(50)
});
win.add(anImageView);

win.open();

    Hope it'll help you to solve your problem.
