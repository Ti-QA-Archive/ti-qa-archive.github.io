  • Hi,

    thanks for the hint, this was exactly what I've been trying first, without any succsess. This code does not work in my present situation.

    But I found a solution for the probleme:

    moving the code from the .open() function to the createWindow() function solved the issue. So this works fine:

    var win =  Titanium.UI.createWindow(
{animated:false, 
title:"test",  
fullscreen:true,
 navBarHidden: false,
    exitOnClose: true,  
    activity : { 
    onCreateOptionsMenu : function(e) {
    var actionBar = win.activity.actionBar; 
    actionBar.onHomeIconItemSelected = function () { 
    }     
    var menu = e.menu;
    var menuItem1 = menu.add({ title: "Menu 1",  icon :  Ti.Filesystem.resourcesDirectory + "images/tab_home.png", showAsAction : Ti.Android.SHOW_AS_ACTION_IF_ROOM});
     var menuItem4 = menu.add({ title: "Menu 2",  icon :  Ti.Filesystem.resourcesDirectory + "images/tab_fav.png", showAsAction : Ti.Android.SHOW_AS_ACTION_IF_ROOM});
      var menuItem3 = menu.add({ title: "Menu 3",  icon :  Ti.Filesystem.resourcesDirectory + "images/tab_faq.png", showAsAction : Ti.Android.SHOW_AS_ACTION_IF_ROOM});
    var menuItem = menu.add({ title: "Menu 4",  showAsAction : Ti.Android.SHOW_AS_ACTION_IF_ROOM, icon :  Ti.Filesystem.resourcesDirectory + "images/tab_set.png", showAsAction : Ti.Android.SHOW_AS_ACTION_IF_ROOM});
    var menuItem2 = menu.add({ title: "Premium",  icon : Ti.Filesystem.resourcesDirectory + "images/tab_dmd.png"});    
     var menuItem5 = menu.add({ title: "Weitere Infos"});


    menuItem.addEventListener("click", function(e) {
    });

    menuItem1.addEventListener("click", function(e) {
    });

    menuItem2.addEventListener("click", function(e) {
    });

    menuItem3.addEventListener("click", function(e) {
    });

    menuItem4.addEventListener("click", function(e) {
    });

    menuItem5.addEventListener("click", function(e) {
    });
        }
        }
        });

win.open();
