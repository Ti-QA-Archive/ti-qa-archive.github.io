Auto update view.

Hi. I'm trying with a chat app. I'd fetched the previous chat history and able to sent the new chat messages to server. Now how can I update the view so that the view gets reloaded automatically when I submit a new chat message? When I restart the app i get the new messages..
here goes my code..

Messages.js


var message =args.message;
$.row.title = message;

function submit (e) {
    var msg = $.test.value;
    if(msg!== '')
    {
        $.test.value = '';
        var chat = Alloy.createModel('sendChatModel');
        chat.save({ "patient_id":args.patient_id,"message":msg,"cookie":args.cookie,"thread_id":args.thread_id });
        var details = chat.fetch({
                success: function(collection, response, options) {
                    if(response.data.published){
                        $.newmsg.text = msg;
                    }
                },
                error: function(collection, loginReq, options) {
                    alert("error");
                }
            });         

    }
    else{
        $.test.value ="Enter text";

    }

}
$.win.open();
