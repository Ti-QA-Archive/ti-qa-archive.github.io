    This is untested, but you'll want something more like:

    // module.js
exports.getUserData = function(callback){
 Ti.Facebook.requestWithGraphPath('me', {}, 'GET', function(e) {
    if (e.success) {
        // alert(e);
        callback(e.result);
        // alert(data);
    } else if (e.error) {
        // alert(e.error);
        callback(false);
    } else {
        // alert('Sem resposta');
        callback(false);
    }
 });
});

    then:

    facebook.getUserData(function(teste){
 Ti.API.log(teste);
});
