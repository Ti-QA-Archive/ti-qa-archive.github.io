  • Accepted Answer

    Hello Carah,
    You can put this code into your project and change the photo name/sizes and it should get you the result you're looking for.

    Dave 

    var mainWindow = Ti.UI.createWindow({
    backgroundColor:"#fff",
    title:"My App",
    width:320,
    height:640
});
mainWindow.open();

var controlView = Titanium.UI.createView({
    backgroundColor:"#333",
    height:350,
    width:280,
});
mainWindow.add(controlView);    // controlView gets added to the main window

var infoView = Ti.UI.createView({
    height:350,
    width:280,
    backgroundColor:'#999'
});

var info = Ti.UI.createLabel({ text: "This photo rocks!" });
infoView.add(info);

var photoView = Titanium.UI.createImageView({
    image:'photo.jpg',
    height:350,
    width:280
});
controlView.add(photoView);    // only add the photoView to controlView because the infoView will be added during the animate below

infoView.addEventListener('click', function() {
    controlView.animate({view:photoView,transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT});
});
photoView.addEventListener('click', function() {
    controlView.animate({view:infoView,transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});
});

  • Hi Carah,
    if your target is iPhone, try tweaking with:

    var infoView = Ti.UI.createView({});
var descrLabel = Ti.UI.createLabel({text: 'description'});
infoView.add(descrLabel);
Ti.UI.currentWindow.animate({view: infoView, transition: Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT});

    where infoView is the view containing the textual information on the current photo (see descrLabel).
