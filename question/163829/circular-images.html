  • You can create a kind of circular image (not perfect) by setting a borderRadius on your image

    var w = Ti.UI.createWindow();
var i = Ti.UI.createImageView({
    image : 'http://cache2.asset-cache.net/gp/460688619.jpg?v=1&c=IWSAsset&k=3&d=ciff01xWH27%2biVaV9T6%2bkx6mwf2omM2YGXVxH9j1RE7TQMN7b9KQzmf9XlsCaXRK',
    borderWidth : 1,
    borderRadius : 90,
    borderColor:'white'
});

w.add(i);
w.open();

    • You must set image settings (height, width, borderRadius, etc.) in contactImg directly, not using another imageview inside success event.

      For example:

      var newContact = {
    contactImg : Ti.UI.createImageView({
        height : 80,
        width : 80,
        borderRadius : 40,
        borderWidth : 5
    })
};

Titanium.Media.showCamera({
    success : function(event) {
        if (event.mediaType == Ti.Media.MEDIA_TYPE_PHOTO) {                    
            newContact.contactImg.image = event.media;
        }            
    },
    cancel : function() {
     },
    error : function(error) {
        //error happend, create alert 
        if (error.code == Titanium.Media.NO_CAMERA) {
            alert('Device does not have camera');
        } 
        else {
            alert('Unexpected error: ' + error.code);
        }
    },
    allowImageEditing : true,
    saveToPhotoGallery : true
});
