  • Accepted Answer

    var win = Ti.UI.createWindow();

var tv = Ti.UI.createTableView();

var tvr = Ti.UI.createTableViewRow({title:'Click Me'});

tvr.addEventListener('click', function(){

    var newwin = Ti.UI.createWindow({backgroundColor:'green'});
    var backBtn = Ti.UI.createButton({top:0, left:0, title:'Back'});
    backBtn.addEventListener('click', function(){
        newwin.close()
    });

    newwin.add(backBtn);
    newwin.open();

});

tv.data=[tvr];

win.add(tv);
win.open();
