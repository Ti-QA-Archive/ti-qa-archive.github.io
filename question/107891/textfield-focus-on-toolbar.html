Textfield focus on toolbar

Hi guys,
Great API.
Question: I have a window (on iPad) with a toolbar child added; on the toolbar are several items: 1 textfield, 3 buttons. I would like, when the window opens, to force focus on the textfield (and have the keyboard popup). Unfortunately, this does not work.
Here is my code below:

var barcodeField = Titanium.UI.createTextField({
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
    height: 32,
    width: 175
});

var flexSpace = Titanium.UI.createButton({
    systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

var menuButton = Titanium.UI.createButton({
    style: Titanium.UI.iPhone.SystemButtonStyle.DONE,
    title: 'Menu'
});

var caseButton = Titanium.UI.createButton({
    style: Titanium.UI.iPhone.SystemButtonStyle.DONE,
    title: 'Case'
});

var logoutButton = Titanium.UI.createButton({
    style: Titanium.UI.iPhone.SystemButtonStyle.BORDERED,
    title: 'Logout'
});

var toolbar = Titanium.UI.createToolbar({
    items: [barcodeField, flexSpace, menuButton, caseButton, logoutButton],
    top: 0
});

var window = Titanium.UI.createWindow({});
//window.add(barcodeField);
//window.add(menuButton);
//window.add(caseButton);
//window.add(logoutButton);
window.add(toolbar);
window.addEventListener('open', function() {
    barcodeField.focus();
});

window.open();

It seems though if I add the textfield and buttons to the window itself, it does work. Any help would be greatly appreciated.
