  • Hi choi jong, Sarafaraz Babi have given you the right answer please accept it.

    So you might be using such property following this:

    Ti.UI.setBackgroundColor('#fff');
var TablePlusWin = Ti.UI.createWindow();

var countryData = [];
//countryData is set.

for (var i = 0; i < 15; i++) {
    var country = {
        title : "Title " + i
    };
    countryData[i] = country;
}

var header = Ti.UI.createView({
    backgroundColor : '#1583AF',
    height : Ti.UI.SIZE
});

var headerLabel = Ti.UI.createLabel({
    height : '30',
    font : {
        fontSize : 18,
        fontWeight : 'bold'
    },
    text : 'ISO Countries of World',
    color : '#fff',
    textAlign : 'center'
});

header.add(headerLabel);

var countryTable = Ti.UI.createTableView({
    data : countryData,
    headerView : header
});
TablePlusWin.add(countryTable);
TablePlusWin.open();
