  • Hello PRASHANT SHUKLA,

    Are you try to images slider? If you want then use my following code for this. I hope this help you. Or also you can share your code. 

    var win = Titanium.UI.createWindow({

    backgroundColor : '#fff',
});

var images = [];

images.push('/images/1.png');
images.push('/images/2.png');
images.push('/images/3.png');
images.push('/images/4.png');

var imageView = Titanium.UI.createImageView({
    images : images,
    duration : 100,
    repeatCount : 2,
    top : 0
});



imageView.addEventListener('load', function(e) {

    imageView.start();
});



win.add(imageView);
win.open()
