    This has been discussed in other posts. The map click event only registers when an annotation is clicked. Try this and click the annotation to generate the event:

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

var mapview = Titanium.Map.createView({
    mapType : Titanium.Map.STANDARD_TYPE,
    region : {
        latitude : '51.51098606916964',
        longitude : '-0.1338958740234375',
        latitudeDelta : 0.04,
        longitudeDelta : 0.04
    },
    animate : true,
    regionFit : true,
    userLocation : true
});

mapview.addAnnotation(Ti.Map.createAnnotation({
    animate : true,
    pincolor : Titanium.Map.ANNOTATION_RED,
    title : 'annotation',
    latitude : '51.51098606916964',
    longitude : '-0.1338958740234375',
}));

win.add(mapview);
mapview.addEventListener('click', function(evt) {
    alert('clicked');
});

win.open();
