TableViewRow hasCheck property not getting toggled on click event

Im trying to toggle the hasCheck property of TableViewRow using click event but its not behaving properly as expected. I found it working properly in Android but not working in iOS

Application type: mobile

Titanium SDK: 3.1.1 GA and 3.1.0 GA

Platform & version: iOS 6.1

Device: iOS simulator

Host Operating System: OSX 10.8.4

Titanium Studio: build: 3.1.0.201304151600

Please find the sample code attached below.


function TestView() {    
    var data = [];
    var self = Titanium.UI.createTableView({        
        data : data,
    });    
    for (i = 0; i < 5; i++) {
        var section = Titanium.UI.createTableViewSection({headerTitle: "section"+i});        
        for (j = 0; j < 5; j++) {
            var row = Ti.UI.createTableViewRow({
                title : "row"+j
            });
            section.add(row);            
        }
        self.appendSection(section)
    }
    self.addEventListener('click', function(e) {
        e.row.hasCheck = !e.row.hasCheck;
    });
    return self;
}
module.exports = TestView;
