Is the best way to do this by calculating the left (or right) offsets dynamically? Or is there an easier way?

  • Posted for posterity. Answer 1 is correct, but I wanted to contribute a more complete example.

    See the toolbar example in KitchenSink. As Dan indicates, the secret is use of FLEXIBLE_SPACE.

    The following adds three ButtonBars on the same vertical line evenly spaced horizontally. If you don't want them to "stick" to the left and right sides, simply add the flexSpace as the first and last elements in the items array.

    Note - this is for an iPad app. The dimensions won't work well in an iPhone format :)

    
        var bbNavigation = Ti.UI.createButtonBar({ //create the main navigation toolbar
            labels:[
                     {title:'People',width:80,enabled:false},
                    {title:'Places',width:80,enabled:false},
                    {title:'Things',width:80,enabled:true},
                    {title:'Favorites',width:80,enabled:false},
                    {title:'Saved',width:80,enabled:false}
                    ],
            backgroundFocusedColor:'#3366699',
            //top:50,
            style:Titanium.UI.iPhone.SystemButtonStyle.BAR
        });

        var bbDataView = Ti.UI.createButtonBar({ //create the left navigation toolbar
            labels:[
                     {title:'A',width:40,enabled:false},
                    {title:'B',width:40,enabled:false},
                    {title:'C',width:40,enabled:true}
                    ],
            backgroundFocusedColor:'#3366699',
            //top:50,
            style:Titanium.UI.iPhone.SystemButtonStyle.BAR
        });

        var bbDataView2 = Ti.UI.createButtonBar({ //create the right navigation toolbar
            labels:[
                     {title:'1',width:40,enabled:false},
                    {title:'2',width:40,enabled:false},
                    {title:'3',width:40,enabled:true}
                    ],
            backgroundFocusedColor:'#3366699',
            //top:50,
            style:Titanium.UI.iPhone.SystemButtonStyle.BAR
        });

        /*
         * ********** LISTENERS **********
         */
        bbNavigation.addEventListener('click', function(e){
            //Ti.API.info('You clicked: '+e.source.labels[e.index].title);
            Ti.App.fireEvent('navigation',{
                'index':e.index,
                'title':e.source.labels[e.index].title
            });
        });

        bbDataView.addEventListener('click', function(e){
            //Ti.API.info('You clicked: '+e.source.labels[e.index].title);
            Ti.App.fireEvent('dataView',{
                'index':e.index,
                'title':e.source.labels[e.index].title
            });
        });

        bbDataView2.addEventListener('click', function(e){
            //Ti.API.info('You clicked: '+e.source.labels[e.index].title);
            Ti.App.fireEvent('dataView',{
                'index':e.index,
                'title':e.source.labels[e.index].title
            });
        });

        var flexSpace = Titanium.UI.createButton({
            systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
        });
        var bar = Ti.UI.createToolbar({ //create the base toolbar
            bottom:'base',
            items:[bbDataView,flexSpace,bbNavigation,flexSpace,bbDataView2]
        });
