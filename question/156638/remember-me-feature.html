    Hi

    You have not added a switch or other view that would allow the user the choice to have their details "remembered", you need to add one.

    Something like;

    var remember = Ti.UI.createSwitch({
    // if there was something stored then true
    value: Ti.App.Properties.getString('email', '') !== '' 
});
self.add(remember);

    You then check the state of this control during a successful login and set the properties that should be remembered or reset if permission is denied. During the creation of the email and password fields you read back the values of the properties.

    if (response.logged == true) {
    if (remember.getValue() === true) {
        Ti.App.Properties.setString('email', email.getValue());
        Ti.App.Properties.setString('password', password.getValue());
    } else {
        Ti.App.Properties.setString('email', '');
        Ti.App.Properties.setString('password', '');
    }
...

    Tweak your field creations;

    //create a textfield
var email = Titanium.UI.createTextField({
    hintText : 'Email ID',
    value: Ti.App.Properties.getString('email', '')
});
//add to the parent window
self.add(email);

//create a textfield
var password = Ti.UI.createTextField({
    hintText : 'Password',
    value: Ti.App.Properties.getString('email', '')
});
//add to the parent view
self.add(password);
