    You can use Custom Property for this like :

    var row = Titanium.UI.createTableViewRow({
    height : Ti.UI.SIZE,
    className : 'row',
    color : 'black',
    layout : 'horizontal'
}); 


for (var x = 0; x < switchesperrow; x++) {

     var target2 = rasgosEsp[count];
     row.switchName = target2; 
    // Use of Custom property  add above line in your code here
     row.add(target2);

     count++;

};

tablerows2.push(row);

    now your code switch(e.source.switchName) is working and you can get the id of selectedrow's switch. i hope this is solved your problem. thanks.
