Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT to and from the same Window

A good example of this is in the iPod Music app on the iPhone. In the top right of the navigation bar when a song is playing you can toggle between the songs titles and the album cover.

I'm not going for this exactly, but similar in that it is a back and forth Animation using the FLIP_FROM_LEFT and FLIP_FROM_RIGHT.

In my photo Gallery module, I have a grid view with thumbnails of the images. I've managed to code it correctly and it works. The window that I'm opening is called "_imageWin".Basically, it's the ImageView.

// Attach click listener to each thumb
        thumb.addEventListener('click', function(e) 
        {
            // Create a new window and show the image selected
            _imageWin = Ti.UI.createWindow({
                backgroundColor : VARS.rowColor,
                translucent:true,
                barColor : VARS.barColor,
                title : e.source.id + 1 + ' of ' + photos.length,
            });
 // Transition animation for each thumbnail clicked
 _imageWin.open({transition: Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT});

However, I can't figure out how to trigger the FLIP_FROM_RIGHT animation back to the grid view on the back button in the top navigation bar.

Using the resources on this Q&A I'm using this:

  var backButton = Ti.UI.createButton({
              title:'Back',
              image:'Resources/iphone/back@2x.png',
              style: Ti.UI.iPhone.SystemButton.ADD
            });    

            backButton.addEventListener('click',function(e){
                 _imageWin.close({transition:Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});
            });
               _imageWin.leftNavButton = backButton;

This renders correctly, however, no animation occurs at all.

I've also tried simply this:

 _imageWin.close({transition: Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});

I've also tried to change the name of the window _imageWin because the Gallery module I'm using has many views within views. I've tried them all and get errors for all I've tried.

From what I've read, it should work, but it isn't.

    here is a small demo of what you want so compare this with your code and tell me if any problem,

    var self = Ti.UI.createWindow({
    // backgroundColor : '#666666'
    backgroundColor : 'white',
    // layout : 'vertical'
    // backgroundColor : 'stripped',
    // titleImage : 'img/1-Star.png',
    // url : 'pinchZoom.js'
});

var lbl = Ti.UI.createLabel({
    text : 'WIN 1',
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE
});

self.add(lbl);
self.addEventListener('click', function(e) {
    var win = Ti.UI.createWindow({

    });

    var lbl = Ti.UI.createLabel({
        text : 'WIN 2',
        width : Ti.UI.SIZE,
        height : Ti.UI.SIZE
    });

    win.add(lbl);

    win.addEventListener('click', function(e) {
        win.close({
            transition : Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT
        });
    });

    win.open({
        transition : Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
    });
});

self.open();
