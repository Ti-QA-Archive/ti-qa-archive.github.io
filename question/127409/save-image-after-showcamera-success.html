Save image after showCamera success

Hello!
I am trying to save an image that's been taken using showCamera, but with no success. The folder Filters doesn't even get created, what is wrong?

Titanium.Media.showCamera({
    success:function(event)
    {
        var path = Titanium.Filesystem.isExternalStoragePresent() ? Titanium.Filesystem.getExternalStorageDirectory() : Titanium.Filesystem.getApplicationDataDirectory();
        Ti.API.info(path);
        var folder = 'Filters';
        var newFolder = Titanium.Filesystem.getFile(path,folder);
        if (!newFolder.exists())
        {
            newFolder.createDirectory();
        }

        var file = 'Camera/identify.jpg';
        var f = Titanium.Filesystem.getFile(path,file);
        f.write(event.media);
    },
    cancel:function()
    {
    },
    error:function(error)
    {
    },
    allowImageEditing:false,
    saveToPhotoGallery:false
});
