Delay in window rendering incase of android smart phones and tablets

Hi I am using Titanium to develop an application. In which I am creating 4 tabs and a window for each tab. The problem is the window rendering is getting delayed a fraction of second for the first time when tab gets switched from one tab to another tab. This is happening for the first time only. Whenever I switch back to same window again I am not facing this rendering issue. My window consists of table view and five rows. Can anyone share any possible solution why the delay in rendering is happening. Below is my code…

function More() {

var moreWindow = Titanium.UI.createWindow({

    title : 'More',        

    orientationModes : [Titanium.UI.PORTRAIT]
});

if (Ti.platform.osname === 'iphone') {

    moreWindow.barColor = 'blue';

}    

var moreData = [{
    label : 'Terms'
}, {
    label : 'Claims'
}, {
    label : 'Privacy'
}, {
    label : 'Additional Details'
}, {
    label : 'Logout'
}];

var moreDataInput = [];

for (var i = 0; j = moreData.length, i < j; i++) {

    var moreTableRow = Ti.UI.createTableViewRow({

        title : moreData[i].label,

        font : {

            fontSize :12

        },

        color : 'orange',            

        selectedColor : 'white',

        backgroundColor : 'grey',

        selectedBackgroundColor : 'red',

        backgroundSelectedColor : 'red'
    });

    if (Ti.platform.osname === 'android')
        moreTableRow.left = '3%';

    moreDataInput.push(moreTableRow);
}

var tableViewOptions = {
    data : moreDataInput,
    separatorColor : 'transparent',
    backgroundImage : 'none',
    minRowHeight : 50

};
if (Ti.platform.osname === 'iphone') {
    tableViewOptions.footerTitle = '';
}
var moreTable = Ti.UI.createTableView(tableViewOptions);

moreWindow.add(moreTable);


return moreWindow;

}

module.exports = More;

I am opening this window in tab. Like this I am opening four windows in four tabs. The first tab widnow is opening fine without any delay in rendering. But the other windows are getting delayed in rendering for the first time when the tab gets switched. Kindly share a possible solution.

Thanks in Advance
