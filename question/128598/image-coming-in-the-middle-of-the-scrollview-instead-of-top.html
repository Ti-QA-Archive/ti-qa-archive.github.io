Image coming in the middle of the scrollview instead of top

Hi, I have a view on a scroll view, and on the view there is an image view. Why my image is always coming in the middle of the scrollview. i want my image on the top. Whats going wrong in my code?(for iphone)

var win=Titanium.UI.currentWindow;

var scrollView = Titanium.UI.createScrollView({
    contentWidth:'auto',
    contentHeight:'auto',
    top:0,
    showVerticalScrollIndicator:true,
    showHorizontalScrollIndicator:true
});
var view = Ti.UI.createView({
    backgroundColor:'#336699',
    borderRadius:10,
    width:300,
    height:2000,
    top:10
});


var image=Titanium.UI.createImageView({
    image:'images/pic1.png',
    top:0
});

scrollView.add(view);

view.add(image);


win.add(scrollView);
