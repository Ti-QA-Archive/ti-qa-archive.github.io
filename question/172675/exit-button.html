  • Hello Amri,

    you can use below for android. in iOS you can not close app.

    var win = Titanium.UI.createWindow({
    backgroundColor : '#fff',
    exitOnClose: true,   //   set this property 
});

var exit =Titanium.ui.createButton({});
exit.addEventListenner('click',function(){

// only for android
        var alert = Ti.UI.createAlertDialog({
            title : "Message",
            message : "Are you sure you want to close the application?",
            buttonNames : ["OK", "Cancel"]
        });
        alert.addEventListener('click', function(e) {
            if (e.index == 0) {
        //  if window's  exitOnClose property is true than use this 
            win.close();  // it close app


        //  if window's  exitOnClose property is false than use this 
                Ti.Android.currentActivity.finish();
            }
        });
        alert.show();
    });

    i hope this helps. thanks.
