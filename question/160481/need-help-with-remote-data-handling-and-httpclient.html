<WebEnv>

http:&#x2F;&#x2F;eutils.ncbi.nlm.nih.gov&#x2F;entrez&#x2F;eutils&#x2F;efetch.fcgi?db=pubmed&amp;query_key=&quot; + QueryKey + &quot;&amp;WebEnv=&quot; + WebEnv + &quot;&amp;rettype=xml&amp;retmax=10
