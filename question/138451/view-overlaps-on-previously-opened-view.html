    Basically you have a container view, in pseudo code:

    var lastMainView = null;
var setMainView = function(view) {
    if(lastMainView) {
        window.remove(lastMainView);
    }

    window.add(view);
    lastMainView = view;
}

btn.addEventListener('click', function() {
    var view = createMyView();
    setMainView(view);
});

    where btn is the one of your 5 images.
