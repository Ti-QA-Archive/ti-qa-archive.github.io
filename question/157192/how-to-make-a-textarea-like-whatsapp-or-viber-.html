  • Hello Douglas Hennrich,

    I have added one example code for you. Please check this. I hope that will help you. In my example, I remove scroll View layout property. And I also divided two part of this scroll View, One for text area and other one. 

    
var scrollView = Ti.UI.createScrollView({
    backgroundColor : 'blue',
    bottom : 0,
    height : 380
});
loginWin.add(scrollView);

var view = Ti.UI.createView({
    height : 280,
    backgroundColor : '#eee',
});
scrollView.add(view);

var escreverTexto = Ti.UI.createTextArea({
    borderRadius : 10,
    borderColor : '#000',
    //backgroundImage : 'imagens/chatArea.png',
    backgroundColor : '#eee',
    hintText : 'Type',
    width : 280,
    height : 'auto',
    font : {
        fontSize : 15
    },
    suppressReturn : true,
    bottom : 5,
    left : 5,
    paddingLeft : 10,
    paddingRight : 10,
    appearance : Ti.UI.KEYBOARD_APPEARANCE_ALERT
});
scrollView.add(escreverTexto);

escreverTexto.addEventListener('change', function(e) {
    if (escreverTexto.rect.height > 80) {
        escreverTexto.height = 80;
        Ti.API.info('escreverTexto.height' + escreverTexto.rect.height);
        view.height = 200;
    }
});
