Tableview contents not clickable inside SplitWindow

I have created a splitWindow and in my masterView i have added a tableview.
Below is the code:- 

var masterWindow = Ti.UI.createWindow({
    title:'Master',
    backgroundColor: '#fff'
});
masterWindow.add(folderData());

var detailWindow = Ti.UI.createWindow({
    title:"Detail",
    backgroundColor: "#fff"
});

var navWindow =Ti.UI.iPad.createSplitWindow({
    detailView: detailWindow,
    masterView: masterWindow,
    showMasterInPortrait:true
});
navWindow.open();

function folderData(){

var folderNames = [];
    for(var key in folderStructure)
    {
        folderNames.push(folderStructure[key]+","+key);
    }

        var data = [];
        for(var j=0; j<folderNames.length; j++)
        {
            var a = folderNames[j].split(",");

            var    row = Ti.UI.createTableViewRow({height:80,backgroundColor:'#ececec',selectedBackgroundColor:'#dddddd',note:a[0], folderid:a[2]}); 

                var photo = Ti.UI.createImageView({
                    image:'images/folder.jpg',
                    top:5,
                    left:10,
                    width:50,
                    height:50,
                    clickName:'photo'
                });
                row.add(photo);

                var file_name = Ti.UI.createLabel({
                        text: a[0],
                        color: '#232329',
                        textAlign:'left',
                        width:600,
                        wordWrap:false,
                        left:70,
                        top:10,
                        font:{fontWeight:'normal',fontSize:18}
                    });
                row.add(file_name);

                var dateTime = Ti.UI.createLabel({
                    text: a[1],
                        color: '#232329',
                        textAlign:'left',
                        wordWrap:false,
                        left:70,
                        top:50,
                        font:{fontWeight:'normal',fontSize:18}
                });
                row.add(dateTime);

                data.push(row);
        }

        var folderTable = Ti.UI.createTableView({
            data : data,
            separatorColor:'#626262',
            top:40
        });

        return folderTable;
};

I went through titanium documentation and searched everywhere else to find if there is any issue with tableview inside Splitwindow. But I could not find.

I would appreciate help with this.TIA.
