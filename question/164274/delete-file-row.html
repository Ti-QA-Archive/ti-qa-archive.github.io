  • Hi,

    Please run the below code. I hope, it will help you

    var win=Titanium.UI.createWindow({ 
    backgroundColor:'black',
    title:'Testing', 
    layout:'vertical' 
}); 

var f = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory,'three.txt');
var contents = f.read();
var str=contents.toString();
if(str=='')
{
    for (var i=0; i < 8; i++) {
          f.write('Test'+i+'\n',true);
    };
    contents = f.read();
    str=contents.toString();    
}
var res = str.split("\n");
var tableData = [];
for (var i=0; i < res.length; i++) {
    tableData.push({
        title:res[i],
    });
};

var table = Ti.UI.createTableView({
   data: tableData
});
table.addEventListener('click',function(e){

    setFile(e.index);
    table.deleteRow(e.index);
});

win.add(table);
win.open();

function setFile(index)
{
    var newData="";
    if(res.length>=index)
    {
        res.splice(index,1);
        alert(res.length);
    }
    for (var i=0; i < res.length; i++) {
        if(res[i]!="")
            newData=newData+res[i]+'\n';
    }
    Ti.API.info(JSON.stringify(res));
    f.write(newData);
}
