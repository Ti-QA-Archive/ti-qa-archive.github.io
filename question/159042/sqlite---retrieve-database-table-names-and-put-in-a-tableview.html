Hi there

I would just like to retrieve a list of table names from a database and then present these in a list (tableview I assume but open to suggestions?).

This is how I would get the actual row details and I can then put these in a table:

var db = Ti.Database.install('catalogue.sqlite','photos');

var rows = db.execute('SELECT * FROM photos GROUP BY rowid');

But how to do the same but get the actual tables from the database?

  • using type= ' table' return tables and sequence_tables. For Getting only database tables,use below query

    
var currentWin = Titanium.UI.createWindow({
    backgroundColor : '#fff',
    fullscreen : true
});



var data = [];
var db = Titanium.Database.open('photos');
var result = db.execute('SELECT name FROM sqlite_master WHERE type  IN (?,?) AND name NOT LIKE ? UNION ALL SELECT name FROM sqlite_temp_master WHERE type IN (?,?) ORDER BY 1','table','view','sqlite_%','table','view') ;

while (result.isValidRow()) {

    Ti.API.info('result' +result.fieldByName('name'));

    data.push({
        title: result.fieldByName('name'),
    });
    result.next();
}
result.close();
db.close();


var aTableView = Ti.UI.createTableView();

aTableView.setData(data);

currentWin.add(aTableView);

currentWin.open();

  • Hello JC Woodsman,

    For retrieve all table name for database please use my following code. I hope this help to you. 

    

var currentWin = Titanium.UI.createWindow({
    backgroundColor : '#fff',
    fullscreen : true
});



var data = [];
var db = Titanium.Database.open('photos');
var result = db.execute('SELECT * FROM sqlite_master WHERE type=?', 'table');

while (result.isValidRow()) {

    Ti.API.info('result' +result.fieldByName('name'));

    data.push({
        title: result.fieldByName('name'),
    });
    result.next();
}
result.close();
db.close();


// Create a TableView.
var aTableView = Ti.UI.createTableView();

// Populate the TableView data.

aTableView.setData(data);

currentWin.add(aTableView);

currentWin.open();
