tabGroup re-click selected tab

Hi,

Im trying to find out if a user has clicked on the current tab in a tabGroup so that i can reset the view. A similar functionality is found in many apps(ebay,spotify to name a couple)

The focus, blur and click events only fire when you change tabs.

tabGroup.addEventListener('focus', function(e) {
    alert(e)
}

The singletap event fires but it fires across the whole view not just the tabgroup

var selectedTab = "Home"
tabGroup.addEventListener('singletap', function(e) {
    clickedTab=this.getActiveTab()
    if(clickedTab.title==selectedTab){
        alert("reset")
    }
    selectedTab=clickedTab.title
});

I could get the Y position of the singletap, make sure its at the bottom (in the tabGroup) and the execute my reset but it seems a bit ugly.

Any ideas?

  • Try this:

    var tg = Ti.UI.createTabGroup();

var win1 = Ti.UI.createWindow({backgroundColor: 'white'});
var t1 = Ti.UI.createTab({title: 'Tab 1', window: win1});

var win2 = Ti.UI.createWindow({backgroundColor: 'red'});
var t2 = Ti.UI.createTab({title: 'Tab 2', window: win2});

var win3 = Ti.UI.createWindow({backgroundColor: 'blue'});
var t3 = Ti.UI.createTab({title: 'Tab 3', window: win3});

var win4 = Ti.UI.createWindow({backgroundColor: 'green'});
var t4 = Ti.UI.createTab({title: 'Tab 4', window: win4});

var win5 = Ti.UI.createWindow({backgroundColor: 'yellow'});
var t5 = Ti.UI.createTab({title: 'Tab 5', window: win5});

tg.addTab(t1);
tg.addTab(t2);
tg.addTab(t3);
tg.addTab(t4);
tg.addTab(t5);
tg.open();


var activeTabTitle = tg.activeTab.title;
tg.addEventListener('singletap', function(e) {
    setTimeout(function(){
        if (activeTabTitle == tg.activeTab.title) {
            alert('Reclick Tab - Do your stuff here');
        }

        activeTabTitle = tg.activeTab.title;
    }, 200);
});

  • Turns out its a problem with my custom tabGroup module. You can achieve the desired effect with Minh's code above using the following function

    tg.addEventListener('blur', function(e) {
   if(e.index==e.previousIndex){
       alert("do stuff here")
   }
});
