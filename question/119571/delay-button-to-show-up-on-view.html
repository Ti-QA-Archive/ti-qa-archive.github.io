  • Found the answer. Added:

    var btnAd = Ti.UI.createButton({
    backgroundImage: 'mypicture.png',
    height: 50,
    width: 320,
    style: Ti.UI.iPhone.SystemButtonStyle.PLAIN,
    top: 412,
    });

    btnAd.addEventListener('click', function() {

    Titanium.Platform.openURL('http://yourwebsite');

    });

    setTimeout(function() {

    tvCategories.animate({
        height: 412 - 50,
        duration: 500,
    }, function(e) {

        btnAd.visible = false;

        winBrowse.add(btnAd);

        btnAd.visible = false;
        btnAd.top = 412-50;
        btnAd.visible = true;

    }
);

    }, 3000);
