    You are store data in MyArray2 which is local variable and Add your var myArray2=[]; into var myArray = []; and check it. try this code…

    var arbo = Ti.XML.parseString(xmltext).documentElement;
var mesItemACA = arbo.getElementsByTagName("monitem");
var mesItem = arbo.getElementsByTagName("categorie");
var l = mesItem.length;

var myArray = [];

for (var i = 0; i < l; i++) {
    var item = mesItem.item(i);
    var name_item = item.getElementsByTagName("nom").item(0).text;
    myArray.push(name_item);

    var item_cats = item.getElementsByTagName("type");
    l2 = item_cats.length;

    var myArray2 = [];

    for (var j = 0; j < l2; j++) {
        try {
            var typecat = item.getElementsByTagName("type").item(j).text;
            myArray2.push(typecat);
        } catch(E) {
            alert(E);
        }
        myArray.push(myArray2);
    }
};
