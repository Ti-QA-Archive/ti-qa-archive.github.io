ImageView not loading changed image

If i create an ImageView (or View for that matter) and add it to the result of Ti.UI.createWindow
and then delete all program references to that image, I then, say, download a different image with same dimensions and filename, and then try to create an ImageView (or View) with it as its backgroundImage then the old image is shown !!

Is this a bug ? or a known "FEATURE" ?

running on Titanium 1.0.8.201201262211
Using either 1.8.0.1 OR 1.8.2 SDK
on MAC OS 10.7.3
to an iPad 5.0 Simulator
However I have seen this happening on an iPad hardware device with 5.1

  var mainWin   = null,
      imageViewTest = null;



function doImageViewTest(  ) {

  if( imageViewTest == null )
  {
    var libraryDirectory   =  Titanium.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, '../Library/Caches').nativePath;
    var SVTImagesDirectory =  Titanium.Filesystem.getFile(libraryDirectory, 'images');
    var SVTImagesPath      =  SVTImagesDirectory.nativePath;

    var imagePathName = Titanium.Filesystem.getFile( SVTImagesPath, 'asset_P1.jpg' ).nativePath;
    imageViewTest = Titanium.UI.createImageView({
        backgroundImage: imagePathName,
        width: 768,
        height: 1024,
        top: 0,
        left: 0
    });
    mainWin.add( imageViewTest )
  }
  else
  {
    mainWin.remove( imageViewTest );
    delete imageViewTest;
    imageViewTest = null;
  }
};



(function () {
  mainWin = Ti.UI.createWindow({
                  backgroundColor:'#000'
                });

  var doIVButton = Titanium.UI.createButton({
                   top: 200,
                   left: 100,
                   width: 200,
                   height: 50,
                   title:'Toggle ImageView',
                   zIndex: 50
                 });


  //Pick up the 'click' event on the buttons
  doIVButton.addEventListener( 'click', function(e){
    Ti.API.debug( "bootstrap.js::doIVButton.click event" );
    doImageViewTest();
  });


  // Add UI elements to main window
  mainWin.add( doIVButton );

  // Show main window
  mainWin.open();
})();
