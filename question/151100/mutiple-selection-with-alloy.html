    • Hi Zaki,

      Here is the code by using tableView. Change the for loop with your item list. It should work like charm. And add 2 images with unselected check-box and selected check-box in resource folder

      
var self = Ti.UI.createView();
    var tableView = Ti.UI.createTableView();
    self.add(tableView);

    var _data = []
    var _return = [];
    var checkBoxImageOn = [];
    var checkBoxImageOff = [];
    var caption = [];
    for(var i=0; i<3;i++) {
        _return[i] = Ti.UI.createTableViewRow({
            height : 45,
            backgroundColor : '#FFF',
            id : i,
        });

        caption[i] = Ti.UI.createLabel({
            text : "item " + i,
            font : {
                fontSize : 20,
                fontWeight : 'bold'
            },
            left : 20,
            width : 270,
            touchEnabled : false,
            color : '#5a5a5a',
        });
        _return[i].add(caption[i]);

        checkBoxImageOff[i] = Ti.UI.createImageView({
            right : 10,
            id : i,
            name : 'checkOff',
            zIndex : 99,
            height : 32,
            width : 32,
            visible : true,
            touchEnabled : true,
            image : 'radio_btn_off.png' ,
        });
        _return[i].add(checkBoxImageOff[i]);

        checkBoxImageOn[i] = Ti.UI.createImageView({
            right : 10,
            id : i,
            name : 'checkOn',
            zIndex : 999,
            height : 26,
            width : 26,
            visible : false,
            touchEnabled : true,
            image : 'radio_btn_on1.png' ,
        });
        _return[i].add(checkBoxImageOn[i]);

        _return[i].addEventListener('click', function(e) {

            if (checkBoxImageOn[e.source.id].getVisible() == false) {
                checkBoxImageOn[e.source.id].setVisible(true);
                checkBoxImageOff[e.source.id].setVisible(false);
            } else {
                checkBoxImageOn[e.source.id].setVisible(false);
                checkBoxImageOff[e.source.id].setVisible(true);
            }
        })

        _data.push(_return[i]);
    }
    tableView.data = _data;
