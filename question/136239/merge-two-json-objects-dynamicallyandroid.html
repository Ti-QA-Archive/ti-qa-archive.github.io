Merge two json objects dynamically(android)

Hi, I am developing android application. In my application I need to merge two json objects.
My json objects looks like:

var object1 = {
                name:'nilesh',
                friends:
                [{
                    id:1,
                    id:2
                }]
            }

var object2 = {
                creator: {name: 'nilkash',id:1},
                users:
                [{ name : 'xyz', id:5},    
                 { name: 'pqr', id:7}    
                ] 
            }

I want to modify object1 which looks like

var object1 = {
                name:'nilesh',

                friends:
                [{
                    id:1,
                    id:2
                }],

                creator: {name: 'nilkash',id:1},

                users:
                [{ name : 'xyz', id:5},    
                 { name: 'pqr', id:7}    
                ] 
            }

Is there any way to solve this problem. I want to add some attributes from object2 to object1. Thank you…..

  • or you can do something like this

    var obj1 = {
    name : 'nilesh',
    friends : [{
        id : 1,
        id : 2
    }]
}
var obj2 = {
    creator : {
        name : 'nilkash',
        id : 1
    },
    users : [{
        name : 'xyz',
        id : 5
    }, {
        name : 'pqr',
        id : 7
    }]
}

var obj3 = {};
for(var attrname in obj1) {
    obj3[attrname] = obj1[attrname];
}
for(var attrname in obj2) {
    obj3[attrname] = obj2[attrname];
}

Ti.API.info(JSON.stringify(obj3));
