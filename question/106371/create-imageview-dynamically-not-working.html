Create ImageView Dynamically Not Working....

I am trying to create an ImageView dynamically. When I create a single ImageView it works just fine. I can also create a dynamic ImageView okay, but not with the ability to drag and drop it. When I place the "olt" arguments on to the ImageView the compiling fails.

Working Code…

var imageView2 = Titanium.UI.createImageView({
        image:'KS_nav_views.png',
        width:261,
        height:178,
        top:20 }),
olt = Titanium.UI.create3DMatrix(),
        curX, curY;
imageView2.addEventListener('touchstart',function(e){
        curX = e.x;
        curY = e.y;
});
imageView2.addEventListener('touchmove',function(e){
        var deltaX = e.x - curX,
                deltaY = e.y -curY;
        olt = olt.translate(deltaX,deltaY,0);
        imageView2.animate({transform: olt, duration: 100});
});
win1.add(imageView2);

Dynamic create code fails with "olt":

for (i=0;i<=1;i++)
{
    varLeft = i*20;
    var image[i] = Titanium.UI.createImageView({image:'KS_nav_ui.png', width:46, height: 43, left: varLeft}), 
        olt = Titanium.UI.create3DMatrix(),
        curX, curY;
    image[i].addEventListener('touchstart',function(e){
        curX = e.x;
        curY = e.y;
    });
    image[i].addEventListener('touchmove',function(e){
        var deltaX = e.x - curX,
                deltaY = e.y -curY;
        olt = olt.translate(deltaX,deltaY,0);
        imageView2.animate({transform: olt, duration: 100});
    });
    win1.add(image[i]);
}

All I get as a result of the failing code is:

[INFO] Compiling JavaScript…one moment
[WARN] JavaScript compiler reported "Missing semicolon." at app.js:96
[WARN] JavaScript compiler reported "Result of expression 'left.left' [undefined] is not an object." at app.js:2224

Any help would be much appreciated!

  • I figured this out. I had to declare the variable individually to get the first issue resolved. Then, the next issue was calling animate on the touchmove event where it couldn't find the imageV[i] object. I switched this to use the source of the object and voila!!!

    for (i=0;i<=1;i++)
{
    varLeft = i*30;
    imageV[i] = Titanium.UI.createImageView({image:'KS_nav_ui.png', width:46, height: 43, left: varLeft});
     var olt = Titanium.UI.create3DMatrix();
    var curX, curY;
    imageV[i].addEventListener('touchstart',function(e){
            curX = e.x;
            curY = e.y;
    });
    imageV[i].addEventListener('touchmove',function(e){
            var deltaX = e.x - curX,
                deltaY = e.y -curY;
            olt = olt.translate(deltaX,deltaY,0);
            //imageV[i].animate({transform: olt, duration: 100});
            e.source.animate({transform: olt, duration: 100});
    });
    imageV[i].addEventListener('touchend',function(e){
            // do something
    });
    win1.add(imageV[i]);
}

    Enjoy!
