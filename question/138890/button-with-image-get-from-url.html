    okey then use this code.

    var imgView = Ti.UI.createImageView({
    image : 'https://dl.dropbox.com/u/66829596/player.png',
    height : 160,
    width : 280,
});
Titanium.UI.currentWindow.add(imgView);

var bt_play_vid = Titanium.UI.createButton({
    width : 280,
    height : 160,
    left : 20,
    top : 20,
});

imgView.addEventListener('load', function(e) {
    bt_play_vid.image = imgView.toImage();
    imgView.hide();
    // or you can use 
    //====Titanium.UI.currentWindow.remove(imgView);
});

Titanium.UI.currentWindow.add(bt_play_vid);
