  • What I do (on iOs) is capture orientationchange event and reconfigure the visual components I need to.

        Ti.Gesture.addEventListener('orientationchange',function(e){
        // If is in Landscape
        if(Ti.UI.orientation == Ti.UI.LANDSCAPE_RIGHT || Ti.UI.orientation == Ti.UI.LANDSCAPE_LEFT){
            for(var i = 0, j = introSlide.views.length; i < j; i++){
                introSlide.views[i].right = 250;
            }
        // If is in Portrait
        }else{
            for(var i = 0, j = introSlide.views.length; i < j; i++){
                introSlide.views[i].right = 0;
            }
        }
    });
