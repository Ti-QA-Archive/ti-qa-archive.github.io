  • Hi You can try this also

    /*
 * Add a method to the date Class to format the date in DD/MM/YY format
 */

Date.prototype.formatDate_DD_MM_YY = function(inputFormat) {
    function pad(s) {
        return (s < 10) ? '0' + s : s;
    }

    var d = new Date(inputFormat);
    return [pad(d.getDate()), pad(d.getMonth() + 1), d.getFullYear()].join('/');
};
