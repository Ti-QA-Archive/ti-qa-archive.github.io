How to show the label/Title of DashboardView Item?

Hi, my question might be silly, but I cannot figure it out.
I am creating a very simple dashboard view with 1 item there. Here's my code. Is there a way to get the label or title of each item shown? Titanium.UI.DashboardItem doesn't have any properties like text, label, or title, but I saw other people using them. Yet mine doesn't work.

Titanium.UI.setBackgroundColor('#0c0c0c');

// create tab group
var tabGroup = Titanium.UI.createTabGroup();

var data = [];

data[0] = Titanium.UI.createDashboardItem({
    image:'KS_nav_ui.png',
    selectedImage:'KS_nav_views.png',
    canDelete: false,
    label:'Apple',
    title:'APPLE'
});


var dashboard = Titanium.UI.createDashboardView({
    title:'Dashboard View',
    data:data,

});

var win1 = Titanium.UI.createWindow({  
    title:'Window 1',
    backgroundColor:'#fff'
});

var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab1',
    window:win1
});


win1.add(dashboard);

win1.open();

tabGroup.addTab(tab1);    

tabGroup.open();
