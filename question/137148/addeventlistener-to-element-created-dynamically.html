addEventListener to element created dynamically

So, I have a tableView and I'm adding rows to it dynamically.
I get as many rows as answers I receive from a query I made.
I want to add an EventListener to each one of this rows so that, when pressed, a new window would opened.

P.S.: also, my row with the title : 'Sem ofertas para essa pesquisa', it is not showing up when I get no answers from my query and I have no idea why…

var respostas = Titanium.UI.createTableView({
    top : '150 px',
});

xhr.onload = function() {
            var addresposta = [];
            var response = this.responseText;
            var array_resp = response.split("|");

            if(array_resp.length > 0) {

                var answers = [];
                for(var i = 0; i <= array_resp.length - 2; i++) {

                    addresposta.push(Titanium.UI.createTableViewRow({
                        title : array_resp[i],
                        className : 'className',
                        height : '70 px',
                        backgroundImage : 'tabelrow_bkg.png',
                    }));

                }
            } else {
                addresposta.push(Titanium.UI.createTableViewRow({
                    title : 'Sem ofertas para essa pesquisa',
                    className : 'className',
                }));
                }

            respostas.setData(addresposta);
        };

  • Hi Nelson,

    add rowId property to your row and assign the id value, you can use the value of your 'i' for this:

    rowId: i,

    then add an eventlistener to tableview:

    respostas.addEventListener('click', function(e){
     alert(e.rowData.rowId);
});

  • Hi Nelson,

    You need to pass only url name on rows property to open a window and add click event on tableview so when user click on rows so you can open window dynamiclly.

    for example:

    addresposta.push(Titanium.UI.createTableViewRow({
      title : array_resp[i],
        className : 'className',
        height : '70 px',
        backgroundImage : 'tabelrow_bkg.png',
    windowToOpen : "url/filename.js",
    windowTitle : "you new window title"
}));

tableview.addEventlistener("click",function(e){
    if(e.row.windowToOpen !=undefined && e.row.windowToOpen != null){
        // open new window here
    }
});
