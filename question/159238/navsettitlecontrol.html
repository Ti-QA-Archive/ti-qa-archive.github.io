  • yup i solved it by changing calender.js file like below -

    function showCalendars(_title, onComplete) {

    homeWindow = Titanium.UI.createWindow({
        backgroundColor : '#36A9E1',
        layout : 'vertical',
        title : 'The Horse Management Network',
        statusBarStyle : Titanium.UI.iPhone.StatusBar.TRANSLUCENT_BLACK
    });
    var navLabel = Titanium.UI.createLabel({
        color : '#fff',
        height : Ti.UI.SIZE,
        width : Ti.UI.SIZE,
        top : 15,
        text : 'The Horse Management Network',
        textAlign : 'right',
        font : {
            fontFamily : 'Trebuchet MS',
            fontWeight : 'bold',
            fontSize : 14
        }
    });

    var navView = Titanium.UI.createView({
        //backgroundColor : 'green',
        width : Ti.UI.FILL,
        height : Ti.UI.FILL
    });

    navView.add(navLabel);

    var navImage = Ti.UI.createImageView({
        image : "/images/Horse_120.png",
        width : 70,
        height : 25
    });

    homeWindow.setLeftNavButton(navImage);
    homeWindow.setTitleControl(navView);

    // associate label to title
    //homeWindow.setTitleControl(titleLabel);

    return homeWindow;
}

module.exports = showCalendars;
