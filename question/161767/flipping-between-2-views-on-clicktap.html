Flipping between 2 views on click/tap

Hi folks,

Quick one here, and I've got myself in a right old mess!

OK, so here is what i need my app to do…..

  1. Shows an initial 'view' with a text label attached, say 'View 1'

  2. User can tap that view, which flip animates in a second view and shows a label on that view, say 'View 2'

  3. User then taps again and it returns back to View 1, and so on, and so on.

I have the animation flip working fine, but the View 2 label never shows.

Any ideas? Code below.

iOS with Studio 3.2.0

var window = Titanium.UI.currentWindow;

var firstView = Ti.UI.createView({
    backgroundColor:'#FFF',
    top:165,
    width:300,
    height:140,
    opacity:0.7
});
window.add(firstView);

var flippedView = Titanium.UI.createView({
    backgroundColor:'#000',
    top:165,
    width:300,
    height:140,
    opacity:0.7,
    visible:false
});
window.add(flippedView);

var label1 = Ti.UI.createLabel({
    text:'Initial View'
});

firstView.add(label1);

var label2 = Ti.UI.createLabel({
    text:'2nd View'
});

flippedView.add(label2);


firstView.addEventListener('click', function(e) {
    firstView.add(flippedView);

    firstView.animate({
        view:flippedView,
        transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
    });
    flippedView.visible = true;
});

flippedView.addEventListener('click', function(e) {
    flippedView.add(firstView);

    flippedView.animate({
        view:firstView,
        transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
    });
    flippedView.visible = false;
});

Many thanks!

Simon

    var state = true;
var window = Titanium.UI.createWindow({
    backgroundColor: '#ffffff',
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});

var view = Ti.UI.createView({
    backgroundColor:'green',
    top:165,
    width:300,
    height:140
});
window.add(view);

var front = Ti.UI.createView({
    backgroundColor:'red',
    top:0,
    left:0,
    width:300,
    height:140,
    touchEnabled: false
});
var label1 = Ti.UI.createLabel({
    text:'Initial View'
});
front.add(label1);
view.add(front);

var back = Titanium.UI.createView({
    backgroundColor:'blue',
    top:0,
    left:0,
    width:300,
    height:140,
    touchEnabled: false
});
var label2 = Ti.UI.createLabel({
    text:'2nd View'
});
back.add(label2);

view.addEventListener('click', function(e) {
    switch (state) {
    case true:
        Ti.API.info('true');
        view.animate({
            view: back,
            transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
        });
        break;
    case false:
        Ti.API.info('false');
        view.animate({
            view: front,
            transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
        });
        break;
    }
    state = !state;
});

window.open();

    I added the views to a containing view, put the eventListener on the containing view and handled the animation with a switch.

    Also note that I didn't add the back view to the container, it was added to the view parameter of the animation.

    Hope this helps!
