    • Stephen how do you iterate through the results

      It depends. Sometimes we cache the results to a file first:

      var Parse = require('/lib/parse');
var Caching = require('/lib/caching');

exports.myObjects = function(_callback){
    var f = Ti.Filesystem.getFile(Caching.cachedPath('data'),'myObjects.json');
    function callback(success, data){

        if (success){
            Ti.API.info('SUCCESS: ');
            f.write(data);
        } else {
            Ti.API.error('ERROR: '+ data);
            f.write('{}');
            alert('We had an error fetching the list of myObjects.  Long press the tab bar below to refresh when you have Internet access.');
        }
    _callback();
    }

    // Call Parse
    Parse.getObjects(myObjects', callback);

}
