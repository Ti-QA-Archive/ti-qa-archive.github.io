var password = Titanium.UI.createTextField
({
    width:150,
    height:50,
    top:150,
    left:5,
    hintText:'Pass',
    paddingLeft:8,
    paddingRight:8

});

var buttonSub = Titanium.UI.createButton
({
    width:100,
    top:130,
    height:120,
    left:200,
    title:'Sub'
});

buttonSub.addEventListener('click',function()
{
    if(name.value=='')
    {
        alert('Enter your name, please!');
    }
    if(password.value =='')
    {
        alert('Enter your password');
    }

    if(name.value!=''&& password.value!='')

        load();
    }

});

function load() {
    var data_send = {
        name : name.value,
        password : password.value,
    }
    var xhr = Ti.Network.createHTTPClient();
    xhr.onload = function() {
        var json = JSON.parse(this.responseText);

        alert('name' + name.value);
        alert('password' + password.value);
    }

    xhr.onerror = function() {

    }

    xhr.open('POST','http://bilel216.0fees.net/insert.php');

    xhr.send(JSON.stringify(data_send));

}
