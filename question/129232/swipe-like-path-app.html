  • hii

    this is for table row,

    var tableView;
    var data = [];

    // create the rest of the rows
    for (var c=1;c<3;c++)
    {
    var row = Ti.UI.createTableViewRow();
    row.backgroundImage = 'images/list_bk.png';
    row.selectedBackgroundImage = 'images/list_bk.png';
    row.height = 58;

    var title = Ti.UI.createLabel({
    color   : '#fff',
    font    : { fontSize:14,fontWeight:'bold', fontFamily:'Helvetica' },
    left    : 10,
    top     : 11,
    height  : 15,
    width   : 200,
    text    : 'Title '+c
}); 
row.add(title);

var address = Ti.UI.createLabel({
    color   : '#fff',
    font    : { fontSize:10, fontFamily:'Helvetica' },
    left    : 10,
    top     : 30,
    height  : 12,
    width   : 200,
    text    : 'Address'
}); 
row.add(address);

var distance = Ti.UI.createLabel({
    color   : '#fff',
    font    : { fontSize:14,fontWeight:'bold', fontFamily:'Helvetica' },
    left    : 250,
    top     : 21,
    height  : 15,
    width   : 80,
    text    : '1.1'
}); 
row.add(distance);

row.addEventListener('click',function(e){
if( e.row.backgroundImage == 'images/list_bk.png' ) {
    e.row.backgroundImage = 'images/list_swipe.png';
    address.hide();
} else {
    e.row.backgroundImage = 'images/list_bk.png';
    address.show();
}
});

data.push(row);

    }

    //
    // create table view
    //
    tableView = Ti.UI.createTableView({
    data : data,
    minRowHeight : 58,
    separatorStyle : Ti.UI.iPhone.TableViewSeparatorStyle.NONE
    });

    win.add(tableView);
