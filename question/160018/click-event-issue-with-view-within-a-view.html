Click event issue with View within a view.

Hi all,

I am writing a code do generate a click event on a image,
so I used this code

        var meView = Ti.UI.createView({
            left: 10
            id: "Item"
        });

        var meImage = Ti.UI.createImageView({
            image: "img/img.png",
            height : 120,
            top:2,
            left:"2%",
            width : 120,
            borderRadius : 10,
        });
        meImage.addEventListener('click', function(e) {
            var controller = Alloy.createController("Next_window");                 
        });
        meView.add(meImage);
        var label = Ti.UI.createLabel({
            font : {
                fontSize : 14,
                fontWeight : "bold",
            },
            text : "Main Label",
            width : "65%",
            top:20,
            left:"130",
        });

        label.addEventListener('click', function(e) {
            var controller = Alloy.createController("Next_window");
        });
        var labelView = Ti.UI.createView({
            layout: "vertical"
        });
        labelView.add(label);
        var brLabel = Ti.UI.createLabel({
            font : {
                fontSize : 12,
                fontWeight : "bold"
            },
            color: "#696969",
            text : "Sub Label", 
            width : "65%",
            top:20,
            left:"70%"
        });
        labelView.add(brLabel);
        meView.add(labelView);

I am adding all these components to the window at the last, But I am not getting the the click event on image,
I am able to generate the event on Main Label, but not on image

TIA.

Maruti
