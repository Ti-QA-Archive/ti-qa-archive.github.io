Show/Hide TableViews

Hello! I must be confused on how to show/hide views. Specifically a table view. I want to have an iOS tabbed bar that can show a corresponding table view and hide the other. Here's the code I'm using (the entirety of app.js in my test project):

mainWin = Ti.UI.createWindow({layout: 'vertical'});


var tableList1 = [{title: 'items'}, {title: 'for'}, {title: 'table'}, {title: '1'}];
var tableList2 = [{title: 'items'}, {title: 'for'}, {title: 'table'}, {title: '2'}];

var table1View = Ti.UI.createTableView({data: tableList1, id: 'table1'});
var table2View = Ti.UI.createTableView({data: tableList2, id: 'table2'});


var bb = Ti.UI.iOS.createTabbedBar({
    labels: ['ONE', 'TWO'],
    backgroundColor:'#336699',
    top:10,
    style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
    index: 0
});

bb.addEventListener('click', function(e) {
    if(e.index == 0)
    {
        table2View.setVisible(false);
        table1View.setVisible(true);
    }
    else
    {
        table1View.setVisible(false);
        table2View.setVisible(true);
    }    
});

mainWin.add(bb);
mainWin.add(table1View);
mainWin.add(table2View);

mainWin.open();

When running the app, the tabbed bar displays and the table1View is shown. When clicking the "TWO" button in the tabbed bar, I am given a blank screen where the table view should be (background color shows through). CLicking "ONE" restores it. ANy thoughts?

Thanks!

    Hi Mike,

    A Tableview is a heavy component, instead of using two of these monsters, reset the data of one Tableview.

    var tableList1 = [{title: 'items'}, {title: 'for'}, {title: 'table'}, {title: '1'}];
var tableList2 = [{title: 'items'}, {title: 'for'}, {title: 'table'}, {title: '2'}];
..
// remove the other
var tableView = Ti.UI.createTableView({data: tableList1});
...
bb.addEventListener('click', function(e) {
    if(e.index == 0)
    {
        tableView.setData(tableList1);
    }
    else
    {
        tableView.setData(tableList2);
    }   
});

    • to do that, place your views in a scrollableView then you can scroll to a speciifc view by a click on your tabbedbar

      like that: 

      var win = Ti.UI.createWindow();

var tableView = TI.UI.createTableView();
var webView = TI.UI.createWebView();
var view = Ti.UI.createView();

var bb = Ti.UI.iOS.createTabbedBar({
    labels: ['table', 'web', 'view'],
    backgroundColor:'#336699',
    top:10,
    style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
    index: 0
});

var scrollableView = Ti.UI.createScrollableView({
    views:[tableView, webView, view], 
    contentHeight:Ti.UI.SIZE,
    contentWidth: Ti.Platform.displayCaps.platformWidth
});
win.add(scrollableView);

bb.addEventListener('click', function(e) {
   scrollableView.currentPage = e.index ;
});

win.open();
