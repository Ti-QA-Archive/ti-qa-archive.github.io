    Hello Jezz,

    Try this….

    
var data = []; 

data[0]=Titanium.UI.createPickerRow({title:'Two'}); 
data[1]=Titanium.UI.createPickerRow({title:'Three'});
data[2]=Titanium.UI.createPickerRow({title:'Four'});
data[3]=Titanium.UI.createPickerRow({title:'Five'});  
data[4]=Titanium.UI.createPickerRow({title:'Six'});

var picker = Titanium.UI.createPicker({
    top: 200,
    selectionIndicator: true
});

// picker event listener
picker.addEventListener('change',function(k){
   var currentRow = k.rowIndex;
    if (currentRow == 0) {
        windowTabGroup.activeTab = twoPlayersTab;
    } else if (currentRow == 1) {
        windowTabGroup.activeTab = threePlayersTab;
    } else if (currentRow == 2) {
        windowTabGroup.activeTab = fourPlayersTab;
    } else if (currentRow == 3) {
        windowTabGroup.activeTab = fivePlayersTab;
    } else if (currentRow == 4) {
        windowTabGroup.activeTab = sixPlayersTab;
    } 
});

pickerWindow.add(picker);
picker.add(data);
