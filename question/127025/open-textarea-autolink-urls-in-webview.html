    • You could use a webView instead.

      
function linkify(text) {
    return text.replace(/(https?:\/\/\S+)/gi, function(s) {
        return '<a href="' + s + '">' + s + '</a>';
    });
}

var webView = Ti.UI.createWebView({
    html: linkify(body)
});

      The linkify function will convert any links in the text to html links.
