  • Hi

    Please check the below code. This code may one kind of example for your problem if you like.

    var scrollView = Ti.UI.createScrollView({
  layout:'vertical',
  contentWidth: 'auto',
  contentHeight: 'auto',
  showVerticalScrollIndicator: true,
  showHorizontalScrollIndicator: true,
  height: '60%',
  width: '80%'
});
var view = Ti.UI.createView({
  backgroundColor:'green',
  borderRadius: 10,
  top: 10,
  height: 200,
  width: '100%'
});
scrollView.add(view);

innerView=Ti.UI.createView({
  backgroundColor:'#000',
  borderRadius: 10,
  top: 10,
  height: 100,
  width: '100%'
});
scrollView.add(innerView);

var click = Ti.UI.createButton({
    title : 'click',
});

click.addEventListener('click', function() {
    scrollView.remove(innerView);
});

scrollView.add(click);

win.add(scrollView);
