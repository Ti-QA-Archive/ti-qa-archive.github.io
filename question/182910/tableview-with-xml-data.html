Tableview with XML data

Hi!

So I am using the code below the get data from an XML file into my tableView, but now I want to put some extra data from the XML in some kind of "detailView" that opens when I press a tableRow, can not get it to work, any suggestions ?

var win = Titanium.UI.createWindow({  
    title:"XHR",
    backgroundColor:"#FFFFFF",
    exitOnClose:true
});


var data = [];
var tableView = Titanium.UI.createTableView({
    top: '5%',
    data:data
});

function errorMessage(){
    alert("NOT WORKING");
}

function renderXML(){
    var breakfast_menu = this.responseXML.documentElement;
    var food = breakfast_menu.getElementsByTagName("food");


    for(var i=0; i<food.length;i++){
        var item = food.item(i);
        var title = item.getElementsByTagName("name").item(0).text;

        var row = Titanium.UI.createTableViewRow({
            title:title,
            className:"tableRow",
        });
        data.push(row);
    }

    tableView.setData(data);

}

var xhr = Titanium.Network.createHTTPClient({
    onload: renderXML,
    onerror: errorMessage
});

win.add(tableView);

xhr.open("GET","http://www.w3schools.com/xml/simple.xml");
xhr.send();




win.open();
