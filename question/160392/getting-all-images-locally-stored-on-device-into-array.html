    • I got curious and DID try it. It seems to work perfectly on iOS7.

      var w = Ti.UI.createWindow();
var s = Ti.UI.createScrollView({
    layout : 'vertical'
});

var module_assetlibrary = require('de.marcelpociot.assetlibrary');

module_assetlibrary.assets({
    group : "all",
    load : function(e) {
        var image = e.image;
        var thumbnail = e.thumbnail;
        var view = Ti.UI.createImageView({
            image : image,
            top : 5
        });
        s.add(view);
    }
});

w.add(s);
w.open();
