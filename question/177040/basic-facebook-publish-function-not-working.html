Basic Facebook Publish function not working

Hi there, I've been trying to write a simple android app as a test, which posts a simple status update to facebook. The problem is, that even though I set the fb.permissions to "publish_stream" or "publish_actions" (tried both), the login (authorize) dialog only asks for the profile info, which I didnt even ask, and does NOT ask for the publish permissions I've set. Content and variable names are in spanish, sorry for that, but i guess it doesnt matter.

It only has the default index and this is the code: (index.js)

var fb = require("facebook");
fb.appid = "MYAPPIDHERE";
Ti.API.debug("Creando fb");
fb.permissions = ["publish_actions"];

fb.addEventListener("login", function(e){
    Ti.API.debug("Se disparo el evento");
    if (e.success){
        Ti.API.debug("logueado");        
        fb.requestWithGraphPath('me/feed', {message: "Probando API Graph de Facebook"}, 
                 "POST", function(e) {
            if (e.success) {
                alert("Estado publicado con exito: " + e.result);
            } else {
                if (e.error) {
                    alert(e.error);
                } else {
                    alert("Resultado desconocido");
                }
            }
        });            
    } else if (e.error) {
        alert(e.error);
    } else {
        alert("Error desconocido");
    }
});

$.ventana.add(fb.createLoginButton({
    top : "50%",
    style : fb.BUTTON_STYLE_WIDE
}));

$.ventana.open(); //ventana is the name of the only window in the alloy xml markup

Thanks!
