    • This code is a direct duplicate of your example image with all the functionality that image implies. This took a few minutes to put together - you understand all the copy easily.

      I have included an initial "Edit" button which when pressed starts the edit mode with delete action as the default, you then get an add button - press it - it works, the rows can be deleted, you can finish by pressing the done button

      To test this you can either; create a new default project and replace the contents of app.js with the code below, or you can pick and choose as you see fit.

      var win = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Answers'
});
var tab = Ti.UI.createTab({
    icon: 'KS_nav_views.png',
    title: 'Answers',
    window: win
});

var data = [
    { title: 'No' },
    { title: 'Yes' },
    { title: 'Maybe' },
    { title: 'Try again' },
];
var tbl = Ti.UI.createTableView({
    data: data,
    editable: true,
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
win.add(tbl);

var intRow = 0;
var btnEdit = Ti.UI.createButton({ title: 'Edit' });
var btnDone = Ti.UI.createButton({ title: 'Done', style: Ti.UI.iPhone.SystemButtonStyle.DONE });
var btnAdd = Ti.UI.createButton({ title: '+', systemButton: Ti.UI.iPhone.SystemButton.ADD });
win.setRightNavButton(btnEdit);
btnAdd.addEventListener('click', function (e) {    
    intRow = intRow + 1;
    tbl.appendRow({
        title: 'New Row ' + intRow
    });
});
btnEdit.addEventListener('click', function (e) {    
    tbl.setEditing(true);
    win.setLeftNavButton(btnAdd);
    win.setRightNavButton(btnDone);
});
btnDone.addEventListener('click', function (e) {
    tbl.setEditing(false);
    win.setLeftNavButton(undefined);
    win.setRightNavButton(btnEdit);
});

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab);
tabGroup.open();

      For a bonus if you tweak this code;

    • Hi

      I have made my earlier example code a cross platform version. This provides the same methods and visuals as before for iOS and uses the Android equivalents to achieve the same.

      A few notes;

      • Whilst the Android table does not directly support the exact same delete implementation as iOS it does have its own convention, I used it in the example below
      • You can also use the swipe event instead of the longclick, I used my preference - you use yours
      • There is no native moveable equivalent for the Android table - may be they just do not like it :) it was not part of your original question so should not be an issue.
      • You can choose to add it specifically for iOS if you want for a platform specific feature if you want - I do.
      • It is not designed to look good as that is for you to meld it to your apps need.
      • The code is not best practice coding and can be neater - but that was not my aim and you have speed of delivery as a factor.
      • It works, it works well, enjoy.

      Here we go, again this can be used as a stand-alone app for testing and review or pick and choose. Putting some of this in a CommonJS module would be a good plan.

      var win = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Answers'
});
var tab = Ti.UI.createTab({
    icon: 'KS_nav_views.png',
    title: 'Answers',
    window: win
});

var data = [
    { title: 'No', color: '#000' },
    { title: 'Yes', color: '#000' },
    { title: 'Maybe', color: '#000' },
    { title: 'Try again', color: '#000' },
];
var tbl = Ti.UI.createTableView({
    data: data,
    editable: true,
    height: Ti.UI.FILL,
    //moveable: true,
    minRowHeight: 60,
    width: Ti.UI.FILL
});

if (Ti.Platform.getOsname() === 'android') {
    tbl.addEventListener('longclick', function (e) {
        var dialog = Ti.UI.createAlertDialog({
            buttonNames: [
                'Delete',
                'Cancel'
            ],
            cancel: 1,
            message: 'Would you like to delete this row?',
            title: 'Delete'
        });
        var intRow = e.index;
        dialog.addEventListener('click', function (e){
            if (e.index !== e.source.cancel){
                tbl.deleteRow(intRow);
            }
        });
        dialog.show();
    });
}
win.add(tbl);

var intRow = 0;
var btnEdit = Ti.UI.createButton({ title: 'Edit' });
var btnDone = Ti.UI.createButton({ title: 'Done', style: Ti.UI.iPhone.SystemButtonStyle.DONE });
var btnAdd = Ti.UI.createButton({ title: '+', systemButton: Ti.UI.iPhone.SystemButton.ADD });

if (Ti.Platform.getOsname() === 'android') {
    var footer = Ti.UI.createView({
        backgroundColor: '#eee',
        bottom: 0,
        height: 60,
        width: Ti.UI.FILL
    });
    tbl.setBottom(60);
    var lbl = Ti.UI.createLabel({
        color: '#000',
        font: {
            fontSize: 14
        },
        text: 'Press and hold to delete',
        textAlign: 'center',
        visible: false
    });
    footer.add(lbl);
    win.add(footer);
    btnAdd.applyProperties({
        height: 50,
        left: 5,
        visible: false,
        width: 100
    });
    footer.add(btnAdd);
    btnEdit.applyProperties({
        height: 50,
        right: 5,
        width: 100
    });
    footer.add(btnEdit);
    btnDone.applyProperties({
        height: 50,
        right: 5,
        visible: false,
        width: 100
    });
    footer.add(btnDone);
}

btnAdd.addEventListener('click', function (e) {    
    intRow = intRow + 1;
    tbl.appendRow({
        color: '#000',
        title: 'New Row ' + intRow
    });
});
btnEdit.addEventListener('click', function (e) {    
    if (Ti.Platform.getOsname() === 'iphone') {
        tbl.setEditing(true);
        tbl.setMoveable(tbl.getMoveable());
        win.setLeftNavButton(btnAdd);
        win.setRightNavButton(btnDone);
    } else {
        btnEdit.hide();
        btnDone.show();
        btnAdd.show();
        lbl.show();
    }
});
btnDone.addEventListener('click', function (e) {
    if (Ti.Platform.getOsname() === 'iphone') {
        tbl.setEditing(false);
        tbl.setMoveable(false);
        win.setLeftNavButton(undefined);
        win.setRightNavButton(btnEdit);
    } else {
        lbl.hide();
        btnDone.hide();
        btnAdd.hide();
        btnEdit.show();
    }
});
if (Ti.Platform.getOsname() === 'iphone') {
    win.setRightNavButton(btnEdit);
}

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab);
tabGroup.open();
