  • this is a continuation from above comment… where do you close the window that you created in the button click event?

    row.addEventListener('click', function(e) {
    a=e.rowData.id; // get id

    var vidwin = Ti.UI.createWindow({
        backgroundColor:'black',        
    });
    var closeButton = Ti.UI.createButton({});
    vidwin.add(closeButton);


    var activeMovie = Titanium.Media.createVideoPlayer({});

    closeButton.addEventListener('click', function() {
        activeMovie.hide();
        activeMovie.release();
        activeMovie = null;
        // CLOSE THE WINDOW
        vidwin.close();
    });

    vidwin.add(activeMovie);
    activeMovie.play();

});

    this code is not tested or complete, was meant to provide a guide to how I would restructure what you did above, kind of a starting point for you
