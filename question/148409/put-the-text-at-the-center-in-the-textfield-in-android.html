  • Here is your solution always give the value in dp instead of absolute

    var snameText = Ti.UI.createTextField({
    width : '100dp',
    height : '50dp',
    verticalAlign : Ti.UI.TEXT_VERTICAL_ALIGNMENT_BOTTOM,
    left :'10dp',
    backgroundColor : '#000000',
    softKeyboardOnFocus : Titanium.UI.Android.SOFT_KEYBOARD_SHOW_ON_FOCUS,
    color : '#6D6D6C',
    font : {
        fontSize : 12
    },
    paddingLeft : 3,
    borderWidth : 0,
    borderRadius : 4
});

    You could increase the height and width according to your need
