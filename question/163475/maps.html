  • Hello Kusuma,

    Try this one :

    var MapModule = require("ti.map");
var win = Ti.UI.createWindow({
    fullscreen : false
});
var map = MapModule.createView({
    userLocation : true,
    mapType : MapModule.NORMAL_TYPE,
    animate : true,
    region : {
        latitude : -33.87365,
        longitude : 151.20689,
        latitudeDelta : 0.02,
        longitudeDelta : 0.02
    }, //Sydney
    top : '30%'
});
win.add(map);

// route A  coordinates 
var coord1 = {
    latitude : -33.87469,
    longitude : 151.20689
};

// route B coordinates
var coord2 = {
    latitude : -33.87375,
    longitude : 151.20589
};
var route1 = [coord1, coord2];

var route = MapModule.createRoute({
    points : route1,
    color : "blue",
    width : 5.0
});

map.addRoute(route);

win.open();

    i hope this helps. thanks.

  • Hi kusuma jammula,

    I could give you a example of how to use map in titanium. Just have a look of the below code.

    First of all add ti.map module in your tiapp.xml and use Emulator Target as Google APIs(Google Inc.) when you create a new AVD from AVD Manager or Edit it.

    Now in app.js try this,

    var homeWin = Titanium.UI.createWindow({  
    backgroundColor:'#fff',
    navBarHidden:true,
    height:'100%',
    width:'100%',
});

var mapBtn = Titanium.UI.createButton(btn());
mapBtn.title='Map';
mapBtn.top='100dp';
homeWin.add(mapBtn);

// Listen for click events.
mapBtn.addEventListener('click', function() {
    var mapPage = Ti.UI.createWindow({
            url: '/map.js',
            backgroundColor:'#DDDEDE',
            navBarHidden:true,
            modal:true,
        });
        mapPage.open();
});

homeWin.open();

function btn()
{
    var proper = {
            width:'60dp',
            height:'40dp',
            borderRadius: 4,
            borderWidth: 2,
            borderColor: '#fff',
            backgroundGradient: {
                type: 'linear',
                colors: [
                    '#b5babd',
                    '#4a0000'
                ]   
            }
          };

    return proper;
}

    And in map.js,

    var mapPage = Titanium.UI.currentWindow;

//var map=require('ti.map');
var annotations = Ti.Map.createAnnotation({
        latitude: 23.7272512,
        longitude: 90.39666650000004,
        animate: true,
        pincolor: Ti.Map.ANNOTATION_RED,
    });

var mapView = Ti.Map.createView({
    mapType: Ti.Map.HYBRID_TYPE,
    region: {
        latitude:23.7272512,
        longitude:90.39666650000004,
        latitudeDelta:.01,
        longitudeDelta:.01
    },
    animate:true,
    regionFit:true,
    userLocation:true,
    annotations: [annotations]
});
mapPage.add(mapView);

    Hope it'll be a little bit of help to start solving your problem.

    Thanks
