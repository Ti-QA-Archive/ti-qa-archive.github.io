Facebook authorize crash

Im trying to post something to Facebook from iPhone with the titanium api.
Here is my code:

  function shareViaFacebook() {
    jg.info('FACEBOOK: share');
    Titanium.Facebook.appid = $.facebook.appid;
    Titanium.Facebook.permissions = ['publish_stream'];

    if (Titanium.Facebook.loggedIn) {
      jg.info('FACEBOOK: logged in');
      postFacebookFeed();
    } else {
      jg.info('FACEBOOK: not logged in');
      Titanium.Facebook.addEventListener('login', afterFacebookLogin);
      Titanium.Facebook.authorize();
      jg.info('FACEBOOK: after authorize');
    }
  }

  function afterFacebookLogin() {
    jg.info('FACEBOOK: afterFacebookLogin');
    postFacebookFeed();
    Titanium.Facebook.removeEventListener('login', afterFacebookLogin);
    jg.info('FACEBOOK: after afterFacebookLogin');
  }

  function postFacebookFeed() {
    jg.info('FACEBOOK: postFacebookFeed');

  }
