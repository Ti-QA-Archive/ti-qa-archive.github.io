facebook login button

When i do this code below;

// Don't forget to set your appid and requested permissions, else the login button
// won't be effective.
var win = Ti.UI.createWindow({backgroundColor: 'white'});
var fb = require('facebook');
fb.appid = FACEBOOK_APP_ID;
fb.permissions = ['publish_stream'];
fb.addEventListener('login', function(e) {
    if (e.success) {
        alert('Logged in');
    }
});
fb.addEventListener('logout', function(e) {
    alert('Logged out');
});

// Add the button.  Note that it doesn't need a click event listener.
win.add(fb.createLoginButton({
    top : 50,
    style : fb.BUTTON_STYLE_WIDE
}));
win.open()

the facebook login and logout button appears on the same window instead of the logout to appear in the tabgroup(where users get re-directed to after login process). I then decided to create seperate login and logout buttons but my question now is; is it possible for me to add facebook button style to a normal login button? Thanks.

    Hi Mark,

    You can not set facebook button style in  normal button. you can set image of normal button.Download facebook login logout button image and set it in normal button like this.

    
var normalBtn = Ti.UI.createButton({
    top : 50,
    image : 'fbbtnlogin.png'
});
win.add(btn);

normalBtn.addEventListener('click',function(e){
    if (!fb.loggedIn) {
        fb.authorize();
    }
});
if (fb.loggedIn) {
    normalBtn.image = 'fbbtnlogout.png';
}
else
{
    normalBtn.image = 'fbbtnlogin.png';
}
