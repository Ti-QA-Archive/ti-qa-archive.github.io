  • hi,

    i have work around for ios….

    create another label with height 1 and put that label on the text label that can at least solve your problem for ios.

    test this example

    var lbl = Ti.UI.createLabel({
    bottom : 10,
    text : 'Hi, how are you..?',
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
    color : 'red'
});

win.add(lbl);

var lbl1 = Ti.UI.createLabel({
    bottom : (lbl.bottom + lbl.toImage().height)/2,
    width : lbl.toImage().width,
    height : 1,
    backgroundColor : 'red'    
});

win.add(lbl1);
