  • There is lot of mistakes in your code. I have corrected it. Please find below the code,

    
var win = Ti.UI.createWindow({
    backgroundColor:'white'
});

var ma = Ti.UI.createTextField({
    top:'32',
    left:'19%',
    height:'48',
    width:'200',
    borderColor: 'black',
    borderWidth: '2'
});

var img = Ti.UI.createImageView({
    image:'gatito.png',
    width:'64',
    height:'64',
    top:'128',
    backgroundColor: 'red'
});
var button = Ti.UI.createButton({
    title:'ACEPTAR',
    width:'128',
    height:'38',
    color:'orange',
    top:'256',
    borderRadius:'10'
});

button.addEventListener('click',function(){
    //Aqui sin los numeros ingresados son menor a noventa
    if (parseInt(ma.value) <= 90 && parseInt(ma.value) >= -90  && ma.text !== "") {
          var img_rotation = Ti.UI.create2DMatrix().rotate(parseInt(ma.value));
          img.animate({
                      transform:img_rotation,
                      duration:2000
          });
          alert('el valor es correcto');                     
    }
    else {  
        alert('el valor es incorrecto');
    }
});

win.add(ma);
win.add(img);
win.add(button);
win.open();

    Hope this helps you.
