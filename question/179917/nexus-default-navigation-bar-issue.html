    • Hi,

      I don't know any quick solution for your problem and i'm not facing any problem like yours. As you are saying that, problem occurs for your homescreen only. So you can change the code for that window like below. It'll get the devices height and width and set that height and width to other components according that device.

      Test Code:

      var win = Ti.UI.createWindow({
    backgroundColor:'#fff',
});

var screenWidth = Ti.Platform.displayCaps.platformWidth;
var screenHeight = Ti.Platform.displayCaps.platformHeight;

function GetHeight(value) {
    var temp = (value * 100) / 480;
    return parseInt((screenHeight * temp) / 100);
}

function GetWidth(value) {
    var temp = (value * 100) / 320;
    return parseInt((screenWidth * temp) / 100);
}

var anImageView = Ti.UI.createImageView({
    image : '/image.png',
    width : GetWidth(320),
    height : GetHeight(240),
    top : GetHeight(50),
    left : GetHeight(50)
});
win.add(anImageView);
win.open();

      Hope this will resolve your problem.

      Thanks
