  • You could also create a configuration array and still use a loop to construct your buttons. I like using a loop so that you don't have to keep repeating the code that sets the width, height, and other common properties. You can also use a single function as an eventListener for all four buttons.

    var config = [
    {
        image: '/images/Image1.png',
        title: 'title 1',
        clickurl: 'http://www.example.com/link1'
    },
    {
        image: '/images/Image2.png',
        title: 'title 2',
        clickurl: 'http://www.example.com/link2'
    },
    {
        image: '/images/Image3.png',
        title: 'title 3',
        clickurl: 'http://www.example.com/link3'
    },
    {
        image: '/images/Image4.png',
        title: 'title 4',
        clickurl: 'http://www.example.com/link4'
    }
];

var onbtnclick = function (e) {
    alert (e.source.clickurl);
};


for (var i=0;i<4;i++) {
    button[i] = Ti.UI.createButton({ 
           backgroundColor:'#D8D8D8',
           borderWidth:1,
           backgroundImage: config[i].image,
           title: config[i].title
           width: Ti.Platform.displayCaps.getPlatformWidth()/2,
           height:Ti.Platform.displayCaps.getPlatformWidth()/2,
           top:0,
           left:0
    });
    button[i].clickurl = config[i].clickurl;

    button[i].addEventListener ('click', onbtnclick);
    SV.add(button[i]);
}
