  • Hello Gopinath,

    Try this…

    var win = Ti.UI.currentWindow;
var scrollView =  Ti.UI.createScrollView({
    contentWidth: 'auto',
    contentHeight: 'auto',
    layout: 'horizontal'
});

for (var i = 0;  i < 7;  i++) {

    var view = Ti.UI.createView({
        height: 200,
        width: 310,
        top: 10,
        left: 10,
        backgroundColor: '#CCC'
    });

    scrollView.add(view);
};
win.add(scrollView);
