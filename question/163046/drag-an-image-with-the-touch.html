  • Hello Victor,

    Try this :

    
var win = Ti.UI.createWindow({
    backgroundColor : "white"
});

var img = Ti.UI.createImageView({
    image : "appicon.png",      // Your IMage path
    top : 25,
    left : 25,
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE
});

win.add(img);

var curX, curY, deltaX, deltaY;

img.addEventListener('touchstart', function(e) {
    curX = e.x;
    curY = e.y;
});

img.addEventListener('touchmove', function(e) {

    var coords = img.convertPointToView({
        x : e.x,
        y : e.y
    }, win);

    deltaX = coords.x - curX;
    deltaY = coords.y - curY;
    img.left = deltaX;
    img.top = deltaY;

});

win.open();

    i hope this helps. thanks.
