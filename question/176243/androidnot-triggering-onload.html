Android_not triggering onload

Onload never happens. Developing for android on Windows7.

var text = "";
var artist = [];
var data = [];
var url = "http://www.rockwerchter.be/en/line-up";
var tableLineup = Ti.UI.createTableView({
    data : [],
    top : 50,
    headerTitle : 'Artists',
    font : {
        fontSize : 14
    }
});
var httpCLient = Ti.Network.createHTTPClient({
    timeout : 5000,
    onerror : function(e) {
        Ti.API.error("ERROR " + e.error);
    },

    onload : function() {
        text = this.responseText;
        var regex = new RegExp(".*box-title box-title-h2.*", "g");
        artist = text.match(regex);

        for (var i = 0; i < artist.length; i++) {
            artist[i] = artist[i].replace("<h2 class=\"box-title box-title-h2\">", "");
            artist[i] = artist[i].replace("</h2>", "").trim();
            data.push(Titanium.UI.createTableViewRow({
                title : artist[i]
            }));
        }    
    }
});
httpCLient.open("GET", url);
httpCLient.send();
