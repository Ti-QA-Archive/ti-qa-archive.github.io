  • alert3 also need an eventListener:

    var b1 = Ti.UI.createButton({
    title:'Add Contact',
    width:100,
    height:40,
    bottom:20
});

b1.addEventListener('click', function(e){

    var alert3 = Titanium.UI.createAlertDialog({
      title:'Contacts', 
      message:'Hello',
      buttonNames: ['Save','Cancel'],
      cancel:1
    });

    alert3.show();

    alert3.addEventListener('click', function(e){
        //put here your stuff
            if(e.index === 0){
                saveContact();
            }

    });

});     //Closed EventListener
