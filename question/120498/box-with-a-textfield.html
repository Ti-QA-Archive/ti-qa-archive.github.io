  • In android: 

    var view = Ti.UI.createView({
    height: 60,
    top: 5,
    left: 5,
    right: 5        
});
var textField = Ti.UI.createTextField({
    left: 0,
    right: 0,
    height: 40,
    hintText: 'Enter you text'
});
view.add(textField);

var optionDialog = Ti.UI.createOptionDialog({
    androidView: view,
    options: ['Ok']
});
optionDialog.show();

    And in iphone u have to create custom alert box. 

    Var view = Ti.UI.createView({
    backgroundColor: 'transparent'
    left: 0,
    right: 0,
    top: 0,
    bottom: 0
});
Ti.UI.currentWindow(view);
view.hide();

var alertView = Ti.UI.createView({
    width: 300,
    height: 100,
    backgroundColor: '#3399FF'
});
var textField = Ti.UI.createTextField({
    left: 0,
    right: 0,
    height: 40,
    top: 5, 
    hintText: 'Enter you text'
});
var okBtn = Ti.UI.createButton({
    height: 40,
    width: 'auto'
    title: 'Ok',
    bottom: 5
});
alertView.add(okBtn);
alertView.add(textField);
okBtn.addEventListener('click', function(e){
    view.hide();
});

    when u want to show alert use 

    view.show();
