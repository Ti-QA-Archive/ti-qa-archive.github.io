  • I found the solution :

    var button = Ti.UI.createButtonBar({
    labels:[{title:'Click here to be a good guy'}],
    backgroundColor:'#CC0000',
    top:230,
    style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
    width:300,
    height:40,
    left:10,
    right:10
});

win.add(button);
