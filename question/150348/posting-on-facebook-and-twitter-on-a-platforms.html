Posting on facebook and twitter on a platforms

Hi friends of social media*,
I try to post something. For iOS6 I'am using native module of Marcel.
My problem is twitter solution given by alloy.

Here my code:

exports.create = function(_options) {
    this.post2facebook = function() {
    }
    this.post2twitter = function() {
    }
    var self = this;
    var w = Ti.Platform.displayCaps.platformWidth;
    var options = _options;
    options.error = function() {);
    };
    var socialButton = Ti.UI.createImageView({
    });
    socialButton.addEventListener('click', function(e) {
        options.service = (e.x < socialButton.getWidth() / 2) ? 'facebook' : 'twitter';
        if (Ti.Platform.name != 'android' && parseInt(Ti.Platform.version) > 5) {
            var module = require('de.marcelpociot.social');
            module.showSheet(options);
        } else {
            switch (options.service) {
                case 'facebook':
                    Ti.Facebook.appid = Ti.App.Properties.getString('facebook_appid');
                    Ti.Facebook.permissions = ['publish_actions'];
                    if (Ti.Facebook.loggedIn == false) {
                        Ti.Facebook.authorize();
                        Ti.Facebook.addEventListener('login', function(_e) {
                            if (_e.success == true) {
                                self.post2facebook();
                            }
                        });
                    } else
                        self.post2facebook();
                    break;
                case 'twitter':
                    var twitter = require('alloy/builtins/social.twitter').create({
                        consumerKey : Ti.App.Properties.getString('twitter.appid'),
                        consumerSecret : Ti.App.Properties.getString('twitter.appsecret')
                    });
                    if (!twitter.isAuthorized()) {
                        twitter.authorize(function(evt) {
                            self.post2twitter()
                        });

                    } else
                        self.post2twitter();
                    break;
            }
        }
    });
    return socialButton;
}

My problem is the twitter node: If I'am not logged in, the the dialog comes. After succesful login I have to re click the start button. How can I automatic start the posting?
