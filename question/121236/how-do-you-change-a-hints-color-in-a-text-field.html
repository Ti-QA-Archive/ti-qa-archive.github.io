  • Hey Gary,

    Just my two cents:

    Hint colors are not supported. TextArea does not accept them. I am using a workaround using labels and events to simulate a TextHint on TextArea. With this approach you could change the texthint color:

    // This is the 'Hint Text'
var LabelCommentHintText = Ti.UI.createLabel({
            text: 'Type your comment here (optional)',
// Here I disable the touch on this particular label
            touchEnabled: false,
            height: 100,
            top: 260,
            left: 17,
            width: 330,
// here you can change the color since this is a simple label
            color: 'blue',
            visible: true,
            font: {
                fontSize: 15,
                fontFamily: 'TrebuchetMS-Bold'
            }
        });
// This is the text area  
        var comment = Titanium.UI.createTextArea({
            value: '',
            height: 100,
            top: 260,
            left: 10,
            width: 330,
            autocorrect: false,
            autocapitalization: false,
            borderRadius: 10,
            font: {
                fontSize: 15,
                fontFamily: 'TrebuchetMS-Bold'
            }
        });
// Hide  HintText on TextArea focus
        comment.addEventListener('focus', function (e) {
            LabelCommentHintText.visible = false;
        });

    Cheers, hope it helps

    Daniel

  • Hi Gary Williams,

    You know that it is not possible to change HintText color yet. But you can customize your code so that it may look alike. Try this following code

    var win = Ti.UI.createWindow({
    backgroundColor : 'pink'
});

var hintText = Ti.UI.createLabel({
    text : 'Enter your name',
    color : 'green',
    font : {
        fontSize : 12
    },
    backgroundColor : 'transparent',
    height : 35,
    width : 240,

});

var textbox = Ti.UI.createTextField({
    height : 35,
    top : 10,
    left : 40,
    width : 240,
    color : hintText.color,

});
textbox.value = hintText.text;
var flag = false;
var count = true;

textbox.addEventListener('click', function(e) {

    if (textbox.value == hintText.text && e.source.color == 'green') {
        textbox.setSelection(0, 0);
        count = false;
    }

});

textbox.addEventListener('change', function(e) {

    if (!flag) {
        flag = true;
        textbox.color = 'black';
        var txt = textbox.value;
        var actual = textbox.value.length;
        var hint = hintText.text.length;
        textbox.value = "";
        for (var i = 0; i < actual - hint; i++) {
            textbox.value += txt[i];
        }
    }

    if (textbox.value.length == 0) {
        textbox.value = hintText.text;
        textbox.color = hintText.color;
        textbox.setSelection(0, 0);
        flag = false;
    }

});

textbox.addEventListener('return', function(e) {
    if (textbox.value == hintText.text) {
    } else {
        e.source.color = 'black';
    }
});

win.add(textbox);
win.open();
