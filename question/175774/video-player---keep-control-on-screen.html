    • Thank you for your answer. I have used
      mediaControlStyle : Titanium.Media.VIDEO_CONTROL_EMBEDDED
      , but it doesn't work and I couldn't keep controls unhidden. My code for video player:

      var videoPlayer = Titanium.Media.createVideoPlayer({
    top : 2,
    // autoplay : true,
    backgroundColor : 'blue',
    height : 100,
    width : Ti.UI.FILL,
    mediaControlStyle : Titanium.Media.VIDEO_CONTROL_EMBEDDED ,
    scalingMode : Titanium.Media.VIDEO_SCALING_ASPECT_FIT
});

videoPlayer.url = 'voices/test.mp3';
controlView.add(videoPlayer);
