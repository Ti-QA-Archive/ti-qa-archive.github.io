  • hi,

    this is just a demo of how to increase a label so you can take as a reference.

    var win = Ti.UI.createWindow();
var lbl = Ti.UI.createLabel({
    backgroundColor : 'red',
    width : 1,
    height : 1
});
win.add(lbl);
var hw = 1;
setInterval(function(e) {
    hw += 1;
    lbl.width = hw;
    lbl.height = hw;
}, 500);

win.open();
