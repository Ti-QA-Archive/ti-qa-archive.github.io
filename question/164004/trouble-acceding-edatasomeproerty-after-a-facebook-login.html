    Hello Eduard Punk,

    Thanks for writing. I have tested your issue in android device 4.2.2.

    Can you please test my sample code in your environments? It may help to you.

    Check the attachment

    Test Case:

    

var win = Ti.UI.createWindow({backgroundColor: 'white'});
var fb = require('facebook');
fb.appid =XXXXXXXXXXXXXXX;
fb.permissions = ['publish_stream','read_stream'];
fb.addEventListener('login', function(e) {
    if (e.success) {
        Ti.API.info('e.data ' + e.data);
        var t =JSON.parse(e.data);
        alert('first name is ' + t.first_name);
        Ti.API.info('first name is ' + t.first_name);


    }
});
fb.addEventListener('logout', function(e) {
    alert('Logged out');
});

// Add the button.  Note that it doesn't need a click event listener.
win.add(fb.createLoginButton({
    top : 50,
    style : fb.BUTTON_STYLE_WIDE
}));
win.open()
