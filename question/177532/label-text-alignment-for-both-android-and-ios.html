  • Use Layout and you would be good to go

    var win = Ti.UI.createWindow({
    backgroundColor : "white",
    layout : "vertical"
});
var meMsg = Titanium.UI.createLabel({
    text : "gdfgd dfhfgh gfdgdg fgdgd fgdg gdfgd dfhfgh gfdgdg fgdgd fgdg  tyryr fghfgh gdfgd dfhfgh gfdgdg fgdgd fgdg gdfgd dfhfgh gfdgdg fgdgd fgdg  tyryr fghfgh",
    color : "#444444",
    top : 20,
    height : Ti.UI.SIZE,
    width : "100%",
    textAlign : "left",
});
win.add(meMsg);
var meDate = Titanium.UI.createLabel({
    text : "Fri 6:20PM",
    color : "#6B8EBF",
    height : Ti.UI.SIZE,
    textAlign : "left",
    top : 5
});
win.add(meDate);
win.open();
