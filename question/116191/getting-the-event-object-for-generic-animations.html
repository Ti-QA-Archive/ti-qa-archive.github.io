Getting the event object for generic animations?

Hi,

I want to define some generic animation and need to find a way how to access the object the event was assigned to inside the callback.

var winAnimation = Titanium.UI.createAnimation();
winAnimation.transform = Titanium.UI.create2DMatrix().scale(1.5);
winAnimation.opacity = 0.5;
winAnimation.duration = 300;

winAnimation.addEventListener('complete', function(e) {
     <element the animation was assigned to>.animate({
        transform:Titanium.UI.create2DMatrix(),
        duration:200,
        opacity:1.0 });
});

I have tried '''this''', '''e''', '''e.source''' but I all of those wont give me the element. Is there any way?

Thanks for any hints.

  • var winAnimation = Titanium.UI.createAnimation();
winAnimation.transform = Titanium.UI.create2DMatrix().scale(1.5);
winAnimation.opacity = 0.5;
winAnimation.duration = 300;

// assign target window as a property of animation
winAnimation._target = aWindow;

winAnimation.addEventListener('complete', function(e) {
    var window = e.source._target;
     window.animate({
        transform:Titanium.UI.create2DMatrix(),
        duration:200,
        opacity:1.0
    });
});
