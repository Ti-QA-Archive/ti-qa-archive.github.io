Weird Problem with UI

Hello there,

I have a problem which looks really weird to me. I have the following code:

win = Titanium.UI.currentWindow;

var myView = Titanium.UI.createView({
width:"100%",
height:40,
top:20,
borderRadius:10,
borderColor:"#D4D7DB",
layout:"horizontal"
})

var myViewLeft = Titanium.UI.createView({
width:"60%",
height:40,
top:20,
borderColor:"Red"
})

var myViewRight = Titanium.UI.createView({
width:"40%",
height:40,
top:20,
borderColor:"Yellow"
})

myView.add(myViewLeft);
myView.add(myViewRight);
win.add(myView);

The above code works just perfect and does the obvious. However, when I come to change the width of myView into something, let's say 95% , it looks that myViewRight (the yellow view), disapears, i.e. it doesn't fit into myView.

AFAIK this shouldn't happen since myViewRight.width should be the 40% of the 95% which is the parent, right?

Can you please throw some light here?

thank you,
George

var win1 = Titanium.UI.createWindow({
    title : 'Tab 1',
    backgroundColor : '#fff'
});

// calc the with dont just use percentage
var myView = Titanium.UI.createView({
    width :  (Titanium.Platform.displayCaps.platformWidth * .50), 
    height : 40,
    top : 20,
    borderRadius : 10,
    borderColor : "#D4D7DB",
    layout : "horizontal"
})

myView.startLayout();
var myViewLeft = Titanium.UI.createView({
    width : "60%",
    height : 40,
    top : 20,
    borderColor : "Red"
})

var myViewRight = Titanium.UI.createView({
    width : '40%',
    height : 40,
    top : 20,
    borderColor : "Yellow"
})

myView.add(myViewLeft);
myView.add(myViewRight);
win1.add(myView);

// open tab group
win1.open();
