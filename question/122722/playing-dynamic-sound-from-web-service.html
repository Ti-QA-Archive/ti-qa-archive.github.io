Playing dynamic sound from web service

Hi *,

here I try to play a wav file. But nothing appears:

function text2wav() {
    var xhr = Ti.Network.createHTTPClient();
    var datas = {'foo':textInput.value,'p1':100,'p2':100,'s':'de1'};
    xhr.onload = function(e) {
        var wav = this.responseData;
        var tmp = Ti.Filesystem.createTempFile('caf');
        tmp.write(wav);
        player = Ti.Media.createSound({url:tmp});
        player.play();
        player.addEventListener('complete', function(){tmp.deleteFile();});
    };
    xhr.onerror = function(e) {
        Ti.API.log(e);
    };
    xhr.open('POST',URL);  // this URL give me wav files
    xhr.send(datas);
}

On the server runs the script fine and answered with wav (or mp3)
But the createSound item doesnt play.

Best regards
Rainer
