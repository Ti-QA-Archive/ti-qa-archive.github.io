  • Hi

    As Gergely said - Titanium does not have this built in, there is native support for this, but we do not get access to it.

    So you create your own tableViewRow with your own controls inside it.

    Something like this is a starting point;

    var data = [];

var row = Ti.UI.createTableViewRow({
    height: Ti.UI.SIZE,
    width: Ti.UI.FILL
});
var lbl = Ti.UI.createLabel({
    color: 'blue',
    height: Ti.UI.SIZE,
    left: 10,
    text: 'Caption',
    textAlign: 'left'
    width: Ti.UI.SIZE
});
row.add(lbl);
var txt = Ti.UI.createTextField({
    color: '#000',
    left: 80,
    right: 10,
    height: 50,
    width: Ti.UI.FILL
});
row.add(txt);
data.push(row);
var tbl = Ti.UI.createTableView({
    data: data,
    height: Ti.UI.FILL,
    minRowHeight: 60,
    width: Ti.UI.FILL
});
win.add(tbl);
