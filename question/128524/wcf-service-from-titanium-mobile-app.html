    I can give you the below example of code but i am not sure this will suit your service or not. Here i have developed service in WCF which takes and return JSON Data.

    
//This function returns the URL of the webservice.
function getWebserviceURL()
{
    //its just the dummy service its not exist online you can use your own here.
    return "http://wcfservice.wcfapp.net/sampleService.svc/";

}


function callservice()
{

        //fetching data from server.
        var xhr = Titanium.Network.createHTTPClient();
        xhr.setTimeout(1200);
        xhr.onerror = function(e)        /* on  error in getting data from server */
        {
            //check response status and act aaccordingly.
            if (xhr.status != 200) {
                alert("The service is currently unavailable. Please Try Again Later.");
                return;
            }
        };

        //on getting response from server.
        xhr.onload = function()
        {
            //alert(this.responseText);
            var response = JSON.parse(this.responseText);
        };

        //set the url for the service to the get student courses
        var request_url = getWebserviceURL() + "function_name_here";
        xhr.open('POST', request_url);
        xhr.send();
    }
}
