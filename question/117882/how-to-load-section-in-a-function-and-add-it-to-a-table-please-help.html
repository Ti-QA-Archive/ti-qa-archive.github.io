  • var data = [];

tableView = Titanium.UI.createTableView({
    top:320,
    height:400,
    scrollable: false,
    data:data,
    style:Titanium.UI.iPhone.TableViewStyle.GROUPED,
    touchEnabled : true,
    allowsSelection : false
});


var section_one = Ti.UI.createTableViewSection({top:0});
section_one.add(Ti.UI.createTableViewRow({
   title:'section_one_row_1', height:'auto'
}));
section_one.add(Ti.UI.createTableViewRow({
   title:'section_one_row_2', height:'auto'
}));
data.push(section_one);

var section_two = Ti.UI.createTableViewSection({top:0});
section_two.add(Ti.UI.createTableViewRow({
   title:'section_two_row_1', height:'auto'
}));
section_two.add(Ti.UI.createTableViewRow({
   title:'section_two_row_2', height:'auto'
}));
data.push(section_two);

tableView.setData(data);
