Calculating textfield text length

I am working on calculating the amount of characters in a text field. I am just doing this as a basic starter application after the hello world…

//FirstView Component Constructor
function FirstView() {
    //create object instance, a parasitic subclass of Observable
    var self = Ti.UI.createView();

    var textfield = Ti.UI.createTextField({
        width: 100,
        height: 40,
        top: 60
    });
    self.add(textfield);

    var label = Ti.UI.createLabel({
        top: 100,
        color: ('BLACK'),
        text:(textfield.length)
    });
    self.add(label);

    return self;
}

module.exports = FirstView;

it wont calculate the length. Infact, the label is blank. Any advice?
