btnthing2.addEventListener('click', function(){
         var filePath = '/sound/blue.mp3';
         playSound(filePath);
});

function playSound(filePath) {
    var player = Ti.Media.createSound();
    player.addEventListener("error", function(e) {
    alert("Error occured while playing file\n" + e.message);
    });

    if (player.playing) {
        player.stop();
    }
    player.url = filePath;
    player.play();
};

currentWin.open();
