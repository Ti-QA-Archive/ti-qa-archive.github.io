Android application black screen and not responding....

I have a android app that is a simple page with a scroll view and a few labels. When the emulator is on this screen and I hit the run button in Titanium, the emulator returns a black screen with the application title at the top. After a few moments, a box pops up and says that the application is not responding. When I kill the application, the emulator will then run the application like it should.

I've tried the resume, pause and exitOnClose functions. The screen still just hangs.

I've been able to recreate this problem on an actual phone too so it's not an emulator issue. Anyone have any ideas?

    var win = Titanium.UI.currentWindow;
win.backgroundColor = '#575757';

// data for tableview
var data = [{
    title:'ATM, Debit and Credit Cards',
    color:'#000000',
    font: {
        fontSize:16,
        fontWeight:'bold'
    },
    url: 'faqATMs.js'
}
];

var titleFAQs = Titanium.UI.createLabel({
    text:'Frequently Asked Questions',
    font:{
        fontSize:22,
        fontWeight:'bold'},
    textAlign:'center',
    top:'47',
    width:'320',
    height:'40',
    color:'#FFFFFF'
});

var tableView = Titanium.UI.createTableView({
    data:data,
    backgroundColor:'#FFFFFF',
    rowBackgroundColor:'white',
    top:90
});

tableView.addEventListener('click', function(e) {
    if (e.rowData.url) {
        var win = Titanium.UI.createWindow({
            url:e.rowData.url,
            title:e.rowData.title
        });
        Titanium.UI.currentTab.open(win, {
            animated:false
        });
    }
});
var wrapperView = Titanium.UI.createView({
    backgroundColor:'transparent',
    width:300
});

wrapperView.add(tableView);
win.add(titleFAQs);
win.add(wrapperView);

    //create the window 
var win1 = Titanium.UI.createWindow({
    top:'0',
    width: 'auto',
    height: 'auto'
});

var label1 = Titanium.UI.createLabel({
    color:'#999',
    text:'I am Window 1',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
});

win1.add(label1);

//finally, open the window to launch the app 
win1.open();
