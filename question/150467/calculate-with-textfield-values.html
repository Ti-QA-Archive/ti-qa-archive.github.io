    Try this : 

    var a = 0, b=0;

var small_nr = Ti.UI.createTextField({
    height : '20dp',
    left : '42dp',
    color : '#181818',
    width : '120dp',
    top : '163dp',
    hintText : 'Mät Start',
    keyboardType : Ti.UI.KEYBOARD_PHONE_PAD,
    paddingLeft : 10
});
mat_start.addEventListener('blur', function(e) {
    a = small_nr.value;
    e.source.value = e.value + ' st';
});
var big_nr = Ti.UI.createTextField({
    height : '20dp',
    right : '50dp',
    color : '#181818',
    width : '100dp',
    top : '163dp',
    hintText : 'Mät Stopp',
    keyboardType : Ti.UI.KEYBOARD_PHONE_PAD,
    paddingLeft : 10
});
mat_stopp.addEventListener('blur', function(e) {
    b = big_nr.value;
    e.source.value = e.value + ' st';
});
var result = Ti.UI.createTextField({
    height : '20dp',
    left : '55dp',
    color : '#181818',
    width : '100dp',
    top : '213dp',
    hintText : 'Antal Km',
    paddingLeft : 10
});
km.addEventListener('focus', function(e) {
    result.value = b - a
});
