  • You should be using a CommonJS model, like so:

    Business.js

    function Business(string1, string2) { // OR pass an object like Business(obj), and retrieve like so: obj.string1 & obj.string2

    var self = Ti.UI.createView({ backgroundColor:'#282828' });

    var string1Lbl = Ti.UI.createLabel({text: string1, top: 50});
    self.add(string1Lbl);

    var string2Lbl = Ti.UI.createLabel({text: string2, bottom: 50});
    self.add(string2Lbl);

    return self;
};

module.exports = Business;

    How to use, example in app.js

    var Business = require('/location/Business'); // location of Business.js without '.js'

// For simplification now, make sure you follow the sequence of the params
var newBusiness = new Business('some string', 'another string');  
// Otherwise pass in an object {string1:'some string', string2:'another string'} and retrieve it in Business.js

// add another Business.js module in this window
var anotherInstance = new Business('Blah...', 'another string');
