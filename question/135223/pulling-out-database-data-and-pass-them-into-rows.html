    try this one and comment this line in your code tableview.setData([section1,section2]);

    var data = [];
data.push(section1);
function setData() {

    var rows = db.execute('SELECT DISTINCT col_1 FROM bygrace');

    // create the array
    var dataArray = [];

    while(rows.isValidRow()) {
        var r = Ti.UI.createTableViewRow({
            title : '' + rows.fieldByName('col_1') + '',
            hasChild : true,
            path : 'areas.js'
        });
        section2.add(r);

        rows.next();
    };

    // set the array to the tableView
    section2.setData(dataArray);
    data.push(section2);
    tableview.setData(data);
};
