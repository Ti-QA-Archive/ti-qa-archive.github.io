  • hi cristian,

    you can create a file let's say database.js and paste this code into it after that you can include database.js file in your window 

    
var db = ( function() {

    var api = {};
    var conn = Ti.Database.open('test');

    conn.execute('CREATE TABLE IF NOT EXISTS test1 (id INTEGER, name TEXT)');

    api.all = function() {
        var results = [];
        var resultSet = conn.execute("SELECT * FROM test1");
        while(resultSet.isValidRow()) {
            results.push({
                id : resultSet.fieldByName('id'),
                name : resultSet.fieldByName('name'),
            });
            resultSet.next();
        }
        resultSet.close();
        return results;
    };
    return api;
}());

    you can access it by 

    var rows = db.all();

    good luck
