//Replace CurrentURL and NextURL with your song urls...

var win = Ti.UI.createWindow({
    backgroundColor : 'pink',
    layout : 'vertical'
});
var playbtn = Ti.UI.createButton({
    title : 'Play',
    height : 30,
    width : 100,
});
playbtn.addEventListener('click', function(e) {
    audioPlayer.start();
});
win.add(playbtn);
var stopbtn = Ti.UI.createButton({
    title : 'stop',
    height : 30,
    width : 100,
});
stopbtn.addEventListener('click', function(e) {
    audioPlayer.stop();
});
win.add(stopbtn);
var loopbtn = Ti.UI.createButton({
    title : 'loohttp://www.appcelerator.com/p',
    height : 30,
    width : 100,
});
loopbtn.addEventListener('click', function(e) {
    if (Titanium.App.Properties.getBool("loop") == true) {
        Titanium.App.Properties.setBool("loop", false);
        console.info('looping off');
    } else {
        Titanium.App.Properties.setBool("loop", true);
        console.info('looping on');
    }
});
win.add(loopbtn);
win.open();

var audioPlayer = MediaPlayer = Ti.Media.createAudioPlayer({
    allowBackground : true,
    url : "Current URL"
});

audioPlayer.addEventListener('change', function(e) {
    if (e.state == audioPlayer.STATE_STOPPED) {
        audioPlayer.stop();
        if (Titanium.App.Properties.getBool("loop") == true) {
            console.info('it works');
            audioPlayer.start();
        } else {
            console.info('Play next');
            audioPlayer.setUrl('Next URL');
            audioPlayer.start();
        }
    }
});
    Titanium.App.Properties.setBool("loop", false);
