  • Accepted Answer

    var view=Ti.UI.createView({
    height: 50,
    width: 100,
    backgroundColor: '#666'
});

var toggleSwitch=Ti.UI.createView({
    height: 50,
    width: 50,
    left: 0
});

view.add(toggleSwitch);

toggleSwitch.addEventListener('click',function(e){
    e.source.left = e.source.left < 50? 50:0;
});
