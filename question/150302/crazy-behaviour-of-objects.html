Crazy behaviour of objects

Hello,

this code in commonJS module I cannot understand:

var answer = {
    state : true,
};
var self = Ti.UI.createView({
    backgroundColor : 'white',
    answer : answer
});
console.log(self.answer.state);
self.answer.state = false;
console.log(self.answer.state);

On console is '1' und '1'.

    var answer = {

    backgroundColor : 'white',
    answer : answer
});

var d = self.answer;
d.state = false;
self.answer = d;

var dic = self.answer;

console.log(dic.state);
