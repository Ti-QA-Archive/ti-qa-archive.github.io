how to change the google chart default position in webView?

I use the google chart show in webView, but the google chart image default x , y position is 0 px.

I set HTML code in webView , but the google chart image is not change.

my code is below:

var webView = Ti.UI.createWebView(
        {
            url: google chart site, 
            top: 0 ,
            left: 0,
            width: '100%' ,
            height: '100%',
            scalesPageToFit: true
        }
    ); 

 var htmlCode = '';
    htmlCode += 'var element = document.createElement("meta");';
    htmlCode += 'element.name = "viewport";';
    htmlCode += 'element.style = " margin-left = 120  margin-top = 100";';
    htmlCode += 'var head = document.getElementsByTagName("head")[0];';
    htmlCode += 'head.appendChild(element);';

    webView.addEventListener('load', function(e) {
        webView.evalJS(htmlCode);
    });
    webView.enableZoomControls= true
    win.add(webView);
