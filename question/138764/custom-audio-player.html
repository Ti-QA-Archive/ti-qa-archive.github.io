  • Le problème avec Ti.Media.createAudioPlayer c'est qu'il n'est pas possible de connaître la durée d'un son.
    Le player fait uniquement Play ou Pause ou Stop. Par contre il sera pas possible de faire une barre de progression pour le player.

    Il y a deux solutions, soit utiliser SoundPlayer qui le permet ou d'utiliser un player vidéo et en faire un player audio.

    Dans la première solution, c'est la plus simple et elle marche très bien. Le seul problème est qu'il va précharger le fichier son avant son lancement (Avec un fichier mp3 de 10Mo c'est pas terrible.

    Perso j'utilise un player vidéo, et ça marche très bien.

    T'es obligé de faire une barre de progression?

  • Voilà un example complet. J'ai supprimé les parties inutiles et j'ai commenté les parties les plus importantes.

    //Déclaration de la fenêtre
var win = Ti.UI.currentWindow;

//Je passe une varible 'd' qui depuis la fenêtre précédentr
var url = 'http://www.monsite.ch/podcasts/'+d['fkcat']+'/'+d['id']+'.mp3';

//Je mets en place l'image de la pochette
var Cover = Titanium.UI.createImageView({  
        image: unescape(d['image']),
        top:40,  
        left:15,
        right: 15,
        defaultImage:'images/cover.png'
});    

win.add(Cover);

//Création de la barre ou il y aura le slider
var header = Titanium.UI.createView({
    height:35,
    top:0,
    left: 0,
    backgroundGradient:{
        type:'linear',
        colors:[myapp.theme.couleur1,myapp.theme.couleur2],
        startPoint:{x:0,y:0},
        endPoint:{x:0,y:35},
        backFillStart:false
    },
    borderTop:false,
    borderBottom:false,
    zIndex: 22
});
win.add(header);


//Création du slider
var tempsSlider = Titanium.UI.createSlider({
    min:0,
    max:100,
    value:0,
    left:55,
    right:55,
    height:11,
    top:5,
    zIndex: 30            
});
win.add(tempsSlider);


//On met en place deux Labels, un avec le temps initial, le second le temps restant
var temps_debut = Titanium.UI.createLabel({
    text:'00:00:00' ,
    color:'#000',
    font:{
        fontFamily:'Helvetica Neue',
        fontSize:11
    },
    top:10,
    left: 4,
    zIndex: 30
});

var temps_final = Titanium.UI.createLabel({
    text:'00:00' ,
    color:'#000',
    font:{
        fontFamily:'Helvetica Neue',
        fontSize:11
    },
    top:10,
    right: 4,
    zIndex: 30
});

win.add(temps_debut);
win.add(temps_final);


//On créé deux boutons, stop et play
var bt_play = Titanium.UI.createButton({
    backgroundImage:'images/play.png',
    width:30,
    height:30,
    bottom:40,
    left: 10,
    zIndex: 3
});

var bt_stop = Titanium.UI.createButton({
    backgroundImage:'images/stop.png',
    width:30,
    height:30,
    bottom:40,
    left: 8,
    visible: 'no',
    zIndex: 3
});

//On met en place un icon de chargement
var loading = Titanium.UI.createActivityIndicator({
    bottom:40,
    height:30,
    width: 30,
    left: 10,
    style:Titanium.UI.iPhone.ActivityIndicatorStyle.PLAIN,
    zIndex: 3
});

win.add(bt_play);
win.add(bt_stop);
win.add(loading);


//Initialisation du player vidéo
var sound = Titanium.Media.createVideoPlayer({});
    sound.contentURL = url;

//Lancement du son
bt_play.addEventListener('click', function()
{
    if(Ti.App.Properties.getBool('avertissement')==true)
    {
        if (Titanium.Network.networkType == Ti.Network.NETWORK_MOBILE ){
                Ti.UI.createAlertDialog({
                    title:'Attention',
                    message: "L'écoute d'un flux via une connexion EDGE/3G peut engendrer des frais importants sur votre abonnement mobile. Vous pouvez désactiver cette alerte dans les préférences de l'application."
                }).show();
        }
    }

    bt_play.hide();
    bt_stop.show();
    sound.play();        
});

//Mise en pause du son
bt_stop.addEventListener('click', function()
{
    bt_stop.hide();
    bt_play.show();
    sound.pause();
});        

//Fonction qui permet de définir le temps restant du fichier et de convertir l'heure en 00:00:00
function dateHMS(secs) {
    secs = secs % 86400;
    var t = new Date(1970,0,1);
    t.setSeconds(secs);
    var s = t.toTimeString().substr(0,8);
    if(secs > 86399)
        s = Math.floor((t - Date.parse("1/1/70")) / 3600000) + s.substr(2);

    if(secs=="")
        s = "00:00:00";

    return s;
}

//On met en place un écouteur sur le player vidéo
sound.addEventListener('playing',function(e)
{
    bt_play.hide();
    bt_stop.show();
    i = setInterval(function()
    {
        temps_debut.text = dateHMS(Math.round(sound.getCurrentPlaybackTime()/1000));
        temps_final.text = '-'+dateHMS(Math.round((sound.getDuration()-sound.getCurrentPlaybackTime())/1000));
        tempsSlider.max = Math.round(sound.getDuration()/1000);
        tempsSlider.value = Math.round(sound.getCurrentPlaybackTime()/1000);
    },1000);
});     

//Mise en place de l'écouteur sur le slider
tempsSlider.addEventListener('change', function(e)
{
    temps_debut.text = dateHMS(e.value);
    temps_final.text = '-'+dateHMS(Math.round(((sound.getDuration()/1000)-e.value)));
});

tempsSlider.addEventListener('touchstart', function(e)
{
    clearInterval(i);
});

tempsSlider.addEventListener('touchend', function(e)
{
    sound.stop();
    sound.initialPlaybackTime = Math.round(e.value)*1000;
    sound.play();
    bt_play.hide();
    bt_stop.show();
});

win.addEventListener('close', function()
{
    clearInterval(i);
    sound.pause();
    bt_stop.hide();
    bt_play.show();
});

sound.addEventListener('complete', function()
{
    sound.initialPlaybackTime = 0;
    tempsSlider.value = 0;
    bt_stop.hide();
    bt_play.show();
});

//Lancement du son à l'ouverture
bt_play.hide();
bt_stop.show();
sound.play();

    Au final vous avez un lecteur audio, avec une progression qui marche sur une fichier distant.

  • Extra ! Vous devez indiquer que l'application ai accès au player même en mode passive.

    Ouvrez en mode code le fichier tiapp.xml et indiquez le code suivant dedans :

        <iphone>
        <orientations device="iphone">
            <orientation>Ti.UI.PORTRAIT</orientation>
        </orientations>
        <backgroundModes>
            <mode>audio</mode>
        </backgroundModes>
    </iphone>

    En suite, dans le fichier app.js, indiquez la ligne suivante:

    Ti.Media.defaultAudioSessionMode = Ti.Media.AUDIO_SESSION_MODE_PLAYBACK;

    Ca doit être bon comme ça :)
