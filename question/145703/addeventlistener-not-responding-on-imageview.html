  • I think Malcolm's almost got the whole solution – but there's still a scoping problem. photo will be shared by all the ImageViews, so what will happen is that you'll be able to click on all the thumbnails, but they'll all bring up the same large image.

    You need to do something like

    function createThumbnailImageView (small, medium, parentView)
{
        var iv = Ti.UI.createImageView({
            image: small,
            width: '140dp',
            height: '140dp',
            borderRadius: 10,
            top:'60dp',
            left: '13dp'
        });

        iv.addEventListener('click', function (e) {
            var I_BigPic = Ti.UI.createImageView({
                image: medium,
                width: '100%',
                heigth: '100%',
                top: '20dp'
            });
            parentView.add(I_BigPic);
        });

    return iv;
}

Cloud.Photos.query({
    page: 1, 
    per_page: 100, 
    order: "created_at"
}, function (e) {

    for (var i = 0; i < e.photos.length; i++) {
        var photo = e.photos[i];
        Ti.API.info(photo.filename);
        var I_Post = createThumbnailImageView (photo.urls.small_240, photo.urls.medium_500, w_ProjectSub);

        v_Thumbnails.add(I_Post);
    }
});

    This will give you the scoping you need so that each imageview can have its own small and medium images.
