remove a friend doesn't work

os: mac os
titantium: 2.1.2
device: android i9100,

when i remove one friend from my friends list, the titanium says "friends remove successfully", but when i query friends again, the user is still in my friends list

exports.queryFriend = function(user, callback, failCallback) {
    Cloud.Users.query({
        page : 1,
        per_page : 10
    }, function(e) {
        if (e.success) {
            Ti.API.info(JSON.stringify(e.users));
            Ti.App.Properties.setList('AllUsers', e.users);
            callback();
        } else {
            alert('Error:\\n' + ((e.error && e.message) || JSON.stringify(e)));
            failCallback();
        }
    });
};

exports.deleteFriend = function(user, callback, failCallback) {
    Cloud.Friends.remove({
        user_ids : '5025ff96b68553743d0a547c'
    }, function(e) {
        if (e.success) {
            alert('Friend(s) removed');
            callback();
        } else {
            alert('Error:\\n' + ((e.error && e.message) || JSON.stringify(e)));
            failCallback();
        }
    });
};
