    • Yes. I created an app in AdMob and pasted the resulting publisher id in the code.

      Here's how I'm creating the ad view, maybe I'm missing something there:

      function createAdView(height)
{
    var adview = Ti.Admob.createView();
    Ti.Geolocation.accuracy = Ti.Geolocation.ACCURACY_BEST;
    Ti.Geolocation.distanceFilter = 0;
    Ti.Geolocation.purpose = 'To show you local ads, of course!';
    Ti.Geolocation.getCurrentPosition(function reportPosition(e) {
        if (!e.success || e.error) {
            // aw, shucks...
        }
        else 
        {
        //    adview.testing = true,
            adview.width = 320;
            adview.height = height;
            adview.publisherId = ADMOB_PUBLISHER_ID; // You can get your own at http: //www.admob.com/
            adview.keywords = '';
            adview.location = e.coords;
        }
    });    

    return adview;
}

      And, of course, before that I have:

      Ti.Admob = require('ti.admob');

var ADMOB_PUBLISHER_ID = "mypublishedid";

Ti.include("/scripts/admob.js");
