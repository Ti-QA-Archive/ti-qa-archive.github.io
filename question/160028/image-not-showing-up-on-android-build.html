Image not showing up on Android build

I am simply trying to get an image to show up and I am having no luck. The app I am working on has lots of other images that show up, but for whatever reason these images don't show. Here is my code to run through the loop I am using for my images. The visible false you will see in the code gets set to true when the tourWindow gets opened.


c = 0;
var image = [];

var startTouch = 0;
var endTouch = 0;
var nextInt = 50;

for (var i=0;i<16;i++)
{
    image[c] = Ti.UI.createImageView({
            image:'images/sprayer-'+(c+2)+'.png',
            visible:false,
            left: '50',
            bottom:0,
            width: Ti.UI.SIZE,
            height: Ti.UI.SIZE
    });

    image[c].addEventListener('touchstart',function(e){
            startTouch = e.x;
    });
    image[c].addEventListener('touchmove',function(e){
            endTouch = e.x;
            if (startTouch > endTouch)
            {
                var dir = 1;
            }
            else
            {
                var dir = 2;
            }
            var touchLength = Math.abs(startTouch-endTouch);
            //Ti.API.info("Touch Length: "+touchLength);

            if (touchLength >= nextInt)
            {
                nextInt += 50;
                Ti.API.info("Moving");
                if (dir == 2)
                    nextBtn.fireEvent('click');
                else
                    prevBtn.fireEvent('click');
            }
    });
    image[c].addEventListener('touchend',function(e){
            nextInt = 50;
    });

    tourWindow.add(image[c]);
    c++;
}
