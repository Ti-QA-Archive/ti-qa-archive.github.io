  • Hi

    Try this;

    var btn1 = Titanium.UI.createButton({
    title:'teste'
});
win.add(btn1);
var anim;
function completeEvent() {
    Ti.API.info('complete');
    anim.removeEventListener("complete", completeEvent);    
}
function startEvent() {
    Ti.API.info('start');
    anim.removeEventListener("start", startEvent);
}
btn1.addEventListener("click",function(){
    anim = Ti.UI.createAnimation();
    anim.setLeft(10);
    anim.setDuration(300);

    anim.addEventListener("complete", completeEvent);
    anim.addEventListener("start", startEvent);
    btn1.animate(anim);
})

    Strange as it may sound the different is where you place the creation properties;

    Yours;

    ...
var anim = Titanium.UI.createAnimation({left:10,duration:300});
...

    Revised;

    anim = Ti.UI.createAnimation();
anim.setLeft(10);
anim.setDuration(300);

    I declared the anim outside the function to allow the event listeners to be cleaned up.
