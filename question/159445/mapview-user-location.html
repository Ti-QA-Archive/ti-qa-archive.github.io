    Ti.Geolocation.preferredProvider = Titanium.Geolocation.PROVIDER_GPS;
    Ti.Geolocation.purpose = "Recieve User Location";

    Titanium.Geolocation.distanceFilter = 10;

    if (Titanium.Geolocation.locationServicesEnabled === false) {
        Ti.API.debug('Your device has GPS turned off. Please turn it on.');
        alert('Your device has GPS turned off. Please turn it on.');

    function updatePosition(e) {
        //var timestamp = e.coords.timestamp;
        //alert('timestamp:' +new Date(timestamp));
        if (!e.success || e.error) {
            alert("Unable to get your location.");
            Ti.API.debug(JSON.stringify(e));
            Ti.API.debug(e);
            return;
        }

        var region = {
            latitude : e.coords.latitude,
            longitude : e.coords.longitude,
            animate : true,
            latitudeDelta : 0.01,
            longitudeDelta : 0.01
        };

        mapView.setLocation(region);

        Ti.App.fireEvent("app:got.location", {
            "coords" : e.coords
        });
    };

    var mapView = Titanium.Map.createView({
        top : 24,
        left : 24,
        mapType : Titanium.Map.STANDARD_TYPE,
        //region : {

        //    longitude : 114.1659045,
        //    latitudeDelta : 0.001,
        //    longitudeDelta : 0.001
        //    },
        animate : false,
        regionFit : true,
        userLocation : true,
        hideAnnotationWhenTouchMap : true,
        width : 752,
        height : 880,
        zIndex : 0
    });

    Ti.App.addEventListener("app:got.location", function(d) {
        Ti.App.GeoApp.f_lng = d.longitude;
        Ti.App.GeoApp.f_lat = d.latitude;

        // you need to remove this listener, see the blog post mentioned above
        Ti.Geolocation.removeEventListener('location', updatePosition);
