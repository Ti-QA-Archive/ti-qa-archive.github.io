    • I know this manual. In this example we have simple construction.
      In my case:

      ui.loclist = ( function() {
    var api = {};
    api.getAlfaListWindow = function(item) {
        var w = Ti.UI.createWindow({
            title : 'Liste',
            barColor : blue
        });
        var search = Titanium.UI.createSearchBar();
        var locliste = Ti.UI.createTableView({
            search : search,
            height : '100%'
        });
        locliste.addEventListener('click', function(e) {
            var detailwindow = ui.tides.getDetail(e.rowData.item);
            win1.tab.open(detailwindow);
        });
        var locs = ctrl.stations.getAlfaList();
        var sections = [];
        var ndx = 0;
        for(var s in locs) {
            sections[ndx] = Ti.UI.createTableViewSection({
            });
            sections[ndx].headerTitle = s + ' …';
            var row;
            for(var i = 0; i < locs[s].length; i++) {
                row = Ti.UI.createTableViewRow({
                    height : 40,
                    item : locs[s][i],
                    hasChild : true,
                });
                var label = Ti.UI.createLabel({
                    text : locs[s][i].label,
                    height : 20,
                    textAlign : 'left',
                    color : '#555',
                    font : {
                        fontSize : 18,
                        fontWeight : 'bold'
                    },
                    left : 10

                });
                row.add(label);
                sections[ndx].add(row);
            }
            ndx++;
        }
        locliste.data = sections;
        w.add(locliste);
        return w;
    };
    return api;
}());
