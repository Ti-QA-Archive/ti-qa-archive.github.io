    Hi Mona

    You can also fetch data from json same like fetching from array
    e.g try this one 

    var response = '{"Themes":[{"id":"5478bdc32be23cc0d402b33e", "created_at":"2014-11-28T18:24:03+0000", "updated_at":"2014-11-28T23:20:16+0000", "name":"Cairo", "locked":true, "[CUSTOM_Levels]level_ids":[{"id":"5478d3cba2b2aac40402bb36", "created_at":"2014-11-28T19:58:03+0000", "updated_at":"2014-11-28T21:10:23+0000"} ]}]}';
    var res = JSON.parse(response);
    var objTheme = res.Themes;
    alert(objTheme[0]["[CUSTOM_Levels]level_ids"]);

    Here "response" is your response.
