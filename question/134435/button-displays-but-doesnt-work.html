Button displays but doesn't work

I defined two variables:

var item1, item2;

Next I called a function which runs the following lines of code

        item1 = Titanium.UI.createButton({
            image:"images/restart_2.png",
            backgroundSelectedImage:"images/restart_1.png",
            style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN,
            top:5,
            right:8
        });

       item1.addEventListener('click',function(e) { doThis(); });

        win.add(item1);

        item2 = Titanium.UI.createButton({
            image:"images/menu_2.png",
            backgroundSelectedImage:"images/menu_1.png",
            style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN,

        item2.addEventListener('click',function(e) { doThis2(); });

        win.add(item2);
