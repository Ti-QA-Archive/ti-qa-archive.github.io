    You dont have a variable called summary, it would be json.summary.total_count

    Eg

    var d = '{ "data": [ { "id": "597239594", "name": "Lars Eng" }, { "id": "648623169", "name": "Christoffer Karlsson" } ], "paging": { "cursors": { "after": "MTMwNjI5NDI3NQ==", "before": "NTk3MjM5NTk0" } }, "summary": { "total_count": 2 } }';

var json = JSON.parse(d);
alert(json.summary.total_count);
