  • Here goes the example for you

    

// create table view var tableview = Titanium.UI.createTableView({ data:data, editable:true });

var edit = Titanium.UI.createButton({ title:'Edit' });

edit.addEventListener('click', function() { win.setRightNavButton(cancel); tableview.editing = true; });

win.setRightNavButton(edit);

tableview.addEventListener('delete',function(e) { var s = e.section; Ti.API.info('rows ' + s.rows + ' rowCount ' + s.rowCount + ' headerTitle ' + s.headerTitle + ' title ' + e.rowData.title);

Titanium.API.info("deleted - row="+e.row+", index="+e.index+", section="+e.section + ' foo ' + e.rowData.foo);

});
