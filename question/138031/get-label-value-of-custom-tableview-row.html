Get Label Value of Custom TableView Row

Hi,

So, my problem is that I need to get the label value of the custom table view row selected in order to pass it to the child view. So, here is essentially what I have set up as my table:

    while (rows.isValidRow()) {

        var label = Ti.UI.createLabel({
            text : rows.fieldByName('entity_name'),
            color : '#666',
            left : '50dp',
            font : {
                fontSize : 22,
                fontWeight : 'bold',
                fontFamily : 'Arial'
            },
            backgroundColor : '#fff'
        });
        var image = Ti.UI.createImageView({
            left : '10dp',
            image : '/app/images/TblArrow.png'
        });
        var button = Ti.UI.createButton({
            right : '10dp',
            height : '30dp',
            width : '80dp',
            backgroundImage : '/app/images/TblRefer.png'
        });
        var row = Ti.UI.createTableViewRow();

        row.filter = label.text;
        row.add(image);
        row.add(label);
        row.add(button);
        tbl_data.push(row);
        rows.next();
    }
    mainTable.setData(tbl_data);
};

And I have an action listener for the table, and as you can see, it tries to pass the table label value, in this case, 'Test'. I need 'Test' to be replaced with the value of the label at the row, as shown above.

mainTable.addEventListener('click', function(e) {

    var nextWindow = Ti.UI.createWindow({
        url : '/test.js',
        top : '0dp',
        backgroundColor : '#fff',
        fullscreen : false,
        navBarHidden : true,
        companyName : 'Test'
    });
    nextWindow.open();
});

    alert(e.source.myText);
