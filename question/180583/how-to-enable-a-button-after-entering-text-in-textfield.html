how to enable a button after entering text in textfield?

I want the user to fill up a form with name and last name. when the window opens the button has to be disable, but when the user completes the form the button will be enable automatically. This will prevent to send the form without any information.

var textarea = Titanium.UI.createTextField({
    borderWidth : 2,
    borderColor : '#bbb',
    borderRadious : 5,
    font : {fontSize:20, fontWeight:'italic'},
    keyboardType: Ti.UI.RETURNKEY_GO,
    textAlign: 'left',
    top : 300,
    height : '40',
    width : '500',
    left : 150
});

var boton4 = Titanium.UI.createButton({
    title : 'Send Info.',
    top : 700,
    width : 250,
    height : 70,
    right : 100,
    backgroundColor : 'white',
    font : {fontSize: '30sp'}
});

    For boton4 set touchEnabled : false.

    Then, you need a listener:

    textarea.addEventListener('change',function(e){
    boton4.touchEnabled = (e.value.length);
})
