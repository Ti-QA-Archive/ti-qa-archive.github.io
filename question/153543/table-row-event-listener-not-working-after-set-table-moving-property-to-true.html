var testWin = Ti.UI.createWindow({
    backgroundColor: '#fff',
    title: 'test',

var contentArry = [];

    hideSearchOnSelection : false,
    moving : true,
    width : 320,
    left : 0,

    textAlign : 'center'
});

contentArry.push(containerRow);

containerTbl.data = contentArry;

containerRow.addEventListener('click', function(e) {
    alert("clicked")
});
testWin.add(containerTbl);
