    • I think this is what he means (app.js):

      //-- window 1
var win1 = Titanium.UI.createWindow({
    title : "window 1",
    });

//-- window 2
var win2 = Titanium.UI.createWindow({
    title : "window 2",
        });
var openWin3Button = Titanium.UI.createButton({
    title : "open"
});
openWin3Button.addEventListener("click", function() {
    tapgroupWindow.getActiveTab().open(win3);
});
win2.add(openWin3Button);

//-- window 3
var win3 = Titanium.UI.createWindow({
    title : "window 3",
});
var closeWin3Button = Titanium.UI.createButton({
    title : 'close',
});
closeWin3Button.addEventListener("click", function() {
    tapgroupWindow.getActiveTab().close(win3);
});
win3.add(closeWin3Button);

//-- tabgroup
var tapgroupWindow = Ti.UI.createTabGroup({});
var tab1 = Ti.UI.createTab({
    title : "tab 1",
    window : win1
});
tapgroupWindow.addTab(tab1);

var tab2 = Ti.UI.createTab({
    title : "tab 2",
    window : win2
});
tapgroupWindow.addTab(tab2);

//-- let's begin
tapgroupWindow.open();
