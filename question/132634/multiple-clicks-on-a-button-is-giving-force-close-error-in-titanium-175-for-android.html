multiple clicks on a button is giving force close error in titanium 1.7.5 for android

Android mobile project, Titanium 1.7.5, android emulator 2.3.3,

Hi,

In my application if I click on a button multiple times it is giving force close error(the application is crashing),

The code is as follows:

var win = Titanium.UI.createWindow({backgroundImage:'bg.jpg', modal:true, navBarHidden:true, fullscreen:false});

var actInd = Titanium.UI.createActivityIndicator({bottom : 10,height : 50,width : 10,message:' Loading...'});

var btnLogin = Titanium.UI.createButton({
        title : 'Login',
        top:'60%',
        width : '92dp',
        height :'40dp',
        backgroundImage:'images/btn-small.png',
        //backgroundColor:'#FD8300',
        font:{fontFamily:'calibri', fontSize:'12dp', fontColor:'#fff', fontWeight:'bold'},
        color:'#fff',
        //borderRadius:4,
        softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_HIDE_ON_FOCUS
    });

btnLogin.addEventListener('click', function(e) 
{
        actInd.show();
        Ti.include('TC.js');                 
        actInd.hide();
});

win.add(btnLogin);

Is there any way to avoid multiple clicks in titanium..?

Please help me…..

Thanks.

    Try the following

    btnLogin.addEventListener('singletap', function(e) 
{
        actInd.show();
        Ti.include('TC.js');                 
        actInd.hide();
});

    This will solve your problem
