    Hello Zain,

    ITs a sample code :

    var win = Titanium.UI.createWindow();

var tableView = Titanium.UI.createTableView();
win.add(tableView);
//Table View Data Array
var dataArray = [];

var searchReq = Titanium.Network.createHTTPClient();
searchReq.onload = function() {
    var json = this.responseText;
    var response = JSON.parse(json);
    if (response.product == true) {
        var rowData = [];
        var Name = response.p_name;
        var Manufacturer = response.p_manu;
        var Price = response.p_price;
        var Quantity = response.p_quantity;
        var Row = response.row_no;
        var Rack = response.rack_id;
        var Shelf = response.shelf_id;
        var row = Titanium.UI.createTableViewRow({
            height : 'auto'
        });

        var Name = Titanium.UI.createLabel({
            text : Name,
        });
        row.add(Name);

        var Manufacturer = Titanium.UI.createLabel({
            text : Manufacturer,
        });
        row.add(Manufacturer);

        var Price = Titanium.UI.createLabel({
            text : Price,
        });
        row.add(Price);

        var Quantity = Titanium.UI.createLabel({
            text : Quantity,
        });
        row.add(Quantity);

        var Row = Titanium.UI.createLabel({
            text : Row,
        });
        row.add(Row);

        var Shelf = Titanium.UI.createLabel({
            text : Shelf,
        });
        row.add(Shelf);

        var Rack = Titanium.UI.createLabel({
            text : Rack,
        });
        row.add(Rack);
        dataArray.push(row);
    } else {
        alert(response.message);
    }
};
searchReq.open("POST","http://192.168.0.53/remoteRead.php");
searchReq.send();
tableView.setData(dataArray);
win.open();
