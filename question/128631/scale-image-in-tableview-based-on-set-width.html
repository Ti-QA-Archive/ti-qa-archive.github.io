Scale image in tableView based on set width?

I'm hoping someone can help me out. I've created a tableView that contains rows consisting of only imageViews. I can currently specify a height and it will scale the width accordingly. What I'd like to do is specify the width and have it scale the height, which doesn't seem to be working.

Here is the code I'm currently using:

var win = Ti.UI.currentWindow;
win.title = 'Pictures';
win.backgroundImage = '../images/VGbackground.png';
win.barColor = '#800000';
win.tabBarHidden = 'true';

var images = [
    '../images/mbc1.jpg',
    '../images/mbc2.jpg',
    '../images/mbc3.jpg',
    '../images/mbc4.jpg',
    '../images/mbc5.jpg',
    '../images/mbc6.jpg',
];

var picTable = Ti.UI.createTableView({
    width:295,
    height:350,
    backgroundColor:'#000',
    borderRadius:5,
    editable:true,
    moveable:true
});
win.add(picTable);


for (var i in images) {
    var row = Ti.UI.createTableViewRow({
        className:'image',
        height:225
    });

    var imgView = Ti.UI.createImageView({
        image: images[i],
    });
    row.add(imgView);

    picTable.appendRow(row);
};
