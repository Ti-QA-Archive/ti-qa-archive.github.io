    • The escape function in javascript only decodes and/or decodes URIs per the link: decode URI component. For example, this function would make sure a link to a particular file can be read by a server and routed properly. It does not however transcode the html character entities to their corresponding values.

  • Since I only had a few characters to re-code, this is the function I created to deal with this problem. It would be nice if appcelerator came with this out of the box though: 

    function entityToHtml(xmlString) {

    //Ref: http://unicode-table.com/en/

    if (xmlString != null) {
        var entity_dict = [{
            key : "&#10;",
            value : "\n"
        }, {
            key : "&#60;",
            value : "<"
        }, {
            key : "&#62;",
            value : ">"
        }, {
            key : "&#34;",
            value : "\"",
        }];

        for (var i in entity_dict) {
            xmlString = xmlString.replace(new RegExp(entity_dict[i].key, "g"), entity_dict[i].value);
        }

        return xmlString;

    }
}
