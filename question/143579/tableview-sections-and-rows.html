TableView Sections and Rows

Is it possible to do this…

var section = Titanium.UI.createTableViewSection();
section.headerTitle = "Hello";
var row1 = Titanium.UI.createTableViewRow({title:"Hello 1"});
var row2 = Titanium.UI.createTableViewRow({title:"Hello 2"});
section.add(row1);
section.add(row2);
tableview.add(section);

But in a way like this?

var tabledata = [
        {title:'Company 1',  selectedBackgroundColor:'orange'},
        {title:'Company 2', selectedBackgroundColor:'orange'},
        {title:'Company 3', selectedBackgroundColor:'orange'},
        {title:'Company 4', selectedBackgroundColor:'orange'},

    ];

    var tableview = Titanium.UI.createTableView({
        data: tabledata,
        style:1

    });
