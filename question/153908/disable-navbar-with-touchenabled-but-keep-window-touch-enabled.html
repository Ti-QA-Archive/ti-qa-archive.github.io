    • Here is the code;

      var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    exitOnClose: true,
    height: Ti.UI.FILL,
    modal: true,
    title: 'OptionDialog',
    width: Ti.UI.FILL
});
var btnL = Ti.UI.createButton({
    title: 'Left'
});
var btnR = Ti.UI.createButton({
    title: 'Right'
});
btnL.addEventListener('click', function (e) {
    alert('left');
    var win = Ti.UI.createWindow({
        backgroundColor: '#99000000',
        height: Ti.UI.FILL,
        modal: true,
        navBarHidden: true,
        width: Ti.UI.FILL
    });
    win.addEventListener('click', function (e) {
        win.close({animated: false });
    });
    win.open({animated: false });
});
btnR.addEventListener('click', function (e) {
    alert('right');
});
win.setLeftNavButton(btnL);
win.setRightNavButton(btnR);
win.open();
