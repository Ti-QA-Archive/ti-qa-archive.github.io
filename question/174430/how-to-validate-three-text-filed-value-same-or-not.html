    Assuming the problem is trying to confirm that each field is unique from each other;

    Array.prototype.isUnique = function() {
    var a = [];
    var l = this.length;
    for(var i=0; i<l; i++) {
      for(var j=i+1; j<l; j++) {
        if (this[i] === this[j]) { j = ++i; }
      }
      a.push(this[i]);
    }
    a.sort();
    return a.length === 1;
};

alert([ceNumber.value, trptxt.value, validatetxt.value ].isUnique());
