how to hide a function/module?

titanium sdk 1.8.2

iphone sdk 4.3

Hi,

I created two windows using two different functions/modules and its ocurence is controlled by a play and pause button. When I pressed the button, the function gets loaded overlapping the play view, which is ok. Now my problem is when I click play again this creates a new function instead of just showing the previous play view, and the pause view is still visible. How can I hide the pause view function in this case?

This is my code:

app.js

var isPaused = false;

playbtn.addEventListener('click', function(){
    if (isPaused){
        //how to hide the pause view?
    }
    var goPlayBtn = require('ui/PlayView');
        var play = new goPlayBtn();    
    });

pausebtn.addEventListener('click', function(){
    isPaused = true;

    var pauseBtn = require('ui/PauseView');
    var pause = new pauseBtn();
});

PlayView.js:

function PlayView(){
    var playView = Ti.UI.createWindow({
        width: 320,
        height: 380,
        opacity: 0.2,
        backgroundColor: 'blue'
    });

    playView.open();
}

module.exports = PauseView;

PauseView.js:

function PauseView(){
    var fadeView = Ti.UI.createWindow({
        width: 320,
        height: 380,
        opacity: 0.2,
        top: 0,
        backgroundColor: 'red'
    });

    fadeView.open();
}

module.exports = PauseView;
