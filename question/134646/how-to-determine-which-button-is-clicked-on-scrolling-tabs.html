    hi Nash,

    i got the problem and the solution is something like below..

    
var test = function(e){
    alert(e.text);// here you can know that which lable is clicked
}

var l1 = Ti.UI.createLabel({
    text:'1',
    font:{fontSize:13},
    color:'#fff',
    width:'auto',
    textAlign:'center',
    height:'auto'
    touchEnabled: true // THIS DOES THE TRICK TO ACCEPT CLICK
});

l1.addEventListener('click', test);

var l2 = Ti.UI.createLabel({
    text:'2',
    font:{fontSize:13},
    color:'#fff',
    width:'auto',
    textAlign:'center',
    height:'auto'
    touchEnabled: true // THIS DOES THE TRICK TO ACCEPT CLICK
});

l2.addEventListener('click', test);

var l3 = Ti.UI.createLabel({

    color:'#fff',
    width:'auto',
    textAlign:'center',
    height:'auto'

l3.addEventListener('click', test);
