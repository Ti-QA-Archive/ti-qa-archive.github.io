    Hi,
    I just tried this which worked as expected. Triggered from a button (in the top right corner) click :

    
test_button.addEventListener('click',function(e){

    var popover = Ti.UI.iPad.createPopover({ 
        width:300, 
        height:250,
        title:'Test Popover',
        arrowDirection:Ti.UI.iPad.POPOVER_ARROW_DIRECTION_UP
    });

     var pop_view = Ti.UI.createView({
         width:300,
         height:200,
         backgroundColor:'#ff9900',
         top:0
    });

    var popSwitch = Titanium.UI.createSwitch({
        value:false,
        top:10,
        zIndex:20,
        left:20
    });
    pop_view.add(popSwitch);

    popover.add(pop_view);

    popover.show({
        view:e.source,
        rect:{x:0,y:0,width:0,height:0},
        animated:true
    });
});

    Let me know how you get on.
