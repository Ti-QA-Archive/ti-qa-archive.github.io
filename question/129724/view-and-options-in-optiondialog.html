    • Yeah, it's clear now :D

      Here is full working example:

      var win = Ti.UI.createWindow({
    navBarHidden: true,
    backgroundColor: '#000'
});

var button = Ti.UI.createButton({
    title: 'Open dialog'
});

button.addEventListener('click', function() {
    var view = Ti.UI.createView({
        layout: 'vertical',
        height: 200,
        width: 200
    });

    var column1 = Ti.UI.createView({
        height: 50,
        layout: 'horizontal'
    });
    var column2 = Ti.UI.createView({
        height: 50,
        layout: 'horizontal'
    });

    var text1 = Ti.UI.createLabel({
        text: 'Text 1'
    });

    var option1 = Ti.UI.createLabel({
        text: 'Option 1'
    });

    column1.add(text1);
    column1.add(option1);

    var text2 = Ti.UI.createLabel({
        text: 'Text 2'
    });

    var option2 = Ti.UI.createLabel({
        text: 'Option 2'
    });

    column2.add(text2);
    column2.add(option2);

    view.add(column1);
    view.add(column2);

    var dialog = Ti.UI.createOptionDialog({
        widht: 200,
        height: 10,
        //options: ['Option 1', 'Option 2', 'Option 3'],
        androidView: view,
        //cancel: 2,
        title: 'Options'
    });

    dialog.addEventListener('click', function(e) {
        Ti.API.info('Dialog click event!');
        Ti.API.debug(JSON.stringify(e));
    });

    dialog.show();
});

win.add(button);

win.open();
