Titanium Android tabGroup exitOnClose

I have tabGroup in my app.js. I would like to prevent the application from closing when user presses the back button. How can I use exitOnClose as there is no window defined in app.js

var tabGroup = Titanium.UI.createTabGroup({id:'tabGroup1'});

var win1 = Titanium.UI.createWindow({
    url:'about.js'
});

var tab1 = Titanium.UI.createTab({
    icon:'images/about.png',
    window:win1
});

var win2 = Titanium.UI.createWindow({
    url:'account.js'
});
var tab2 = Titanium.UI.createTab({
    icon:'images/account.png',
    window:win2
});

var win3 = Titanium.UI.createWindow({
    url:'settings.js'
});
var tab3 = Titanium.UI.createTab({
    icon:'images/settings.png',
    window:win3
});

tabGroup.addTab(tab1);
tabGroup.addTab(tab2);
tabGroup.addTab(tab3);

tabGroup.open();

Any other way to avoid closing the app?

Thanks

    • you have windows in your tabs and they can catch android:back (you have to listen for this event in each tab's window) :

      win1.addEventListener('android:back', function(){
    /*
    do nothing if you don't want any behavior 
    other than preventing the tabGroup to close
    */
});
win2.addEventListener('android:back', function(){
    /*
    still doing nothing
    */
});
win3.addEventListener('android:back', function(){
    /*
    would be nice to have this at tabGroup level ... 
    */
});
