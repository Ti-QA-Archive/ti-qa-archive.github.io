mapview just giving blank screen, no map

trying to follow the tutorials but when i try to do a mapview it is just giving me a blank screen… no map appears… Can anyone tell me what is going on.

hunter.gl.showCapturedLocation = function(JSONlocation){
    var win = Ti.UI.createWindow({    
        title: 'Captured At'
    });

    var annotationInfo = JSON.parse(JSONlocation);

    var locationAnnotation = Ti.Map.createAnnotation({
        latitude : annotationInfo.latitude,
        longitude : annotationInfo.longitude,
        title : annotationInfo.title,
        pincolor : Ti.Map.ANNOTATION_GREEN,
        animate : true
    });

    var mapView = Ti.UI.createView({
        mapType : Titanium.Map.STANDARD_TYPE,
        region : {
            lattitude : annotationInfo.latitude,
            longitude : annotationInfo.longitude,
            latitudeDelta : 0.1,
            longitudeDelta : 0.1        
        },
        animate : true,
        regionFit : true,
        userLocation : false,
        annotations : [locationAnnotation]
    });

    win.add(mapView);
    return win;
};
