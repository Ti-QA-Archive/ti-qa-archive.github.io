  • This is running for me

    var appmenu = Titanium.UI.createMenu();

var file = Titanium.UI.createMenuItem("File");

    file.addItem("Save file", function(evt)
        {
        if (evt.getType() == 'clicked')
                {

                        alert('Hello world!');
                }
        });
    file.addSeparatorItem();
    file.addItem("Close");
    appmenu.appendItem(file);
    Titanium.UI.setMenu(appmenu);
