    Hi

    Use this a starting point. The following code will;

    • Tab Group
    • 3 Tabs
    • Each Tab has a window
    • Each window has a label
    • The first Tab has a button
    • The button will open a new window within the tab
    • The new window automatically has a back button

    To use this code - create a new blank default project and replace the contents of app.js with the code below;

    var win1 = Ti.UI.createWindow({
    backgroundColor: '#fff',
    title: 'Window 1'
});
var tab1 = Ti.UI.createTab({  
    icon: 'KS_nav_ui.png',
    title: 'Tab 1',
    window: win1
});
var lbl1 = Ti.UI.createLabel({
    text: 'Label 1'
});
win1.add(lbl1);
var btn1 = Ti.UI.createButton({
    title: 'New Window'
});
btn1.addEventListener('click', function (e) {
    var win = Ti.UI.createWindow({
        backgroundColor: '#fff',
        title: 'New Window'
    });    
    tab1.open(win, { animated: true });
});
win1.setRightNavButton(btn1);

var win2 = Ti.UI.createWindow({
    backgroundColor: '#fff',
    title: 'Window 2'
});
var tab2 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 2',
    window: win2
});
var lbl2 = Ti.UI.createLabel({
    text: 'Label 2'
});
win2.add(lbl2);

var win3 = Ti.UI.createWindow({
    backgroundColor: '#fff',
    title: 'Window 3'
});
var tab3 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 3',
    window: win3
});
var lbl3 = Ti.UI.createLabel({
    text: 'Label 3'
});
win3.add(lbl3);

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);
tabGroup.addTab(tab2);
tabGroup.addTab(tab3);

tabGroup.open();
