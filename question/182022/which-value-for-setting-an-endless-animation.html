    Hi Geppoz,

    You can do it by using custom logic

    e.g

    var view = Ti.UI.createView({
    height : 100,
    width : 100,
    backgroundColor : "red"
});
$.winIndex.add(view);
var anim = Ti.UI.createAnimation({
    opacity : 0,
    autoreverse : true,
});
anim.addEventListener("complete",function(){
    view.animate(anim);
});
view.addEventListener("click", function() {
    view.animate(anim);
});
