Map Annotation not Rotating - Android

I'm trying to drop a custom pin on a map and rotate it to point in the direction of travel. The image will not rotate under Android however everything works on the IOS emulator. Is this broken under Android or is there some trick I don't know about.

The relevant code snippet is:

    var MapModule = require('ti.map');

    this.addPinToMap = function(lat, lng, rotation, colour) {

        var img = Ti.UI.createImageView({
            image : '/images/Pointer_RED.png',
            anchorPoint : {
                x : '0.5',
                y : '0.5'
            },
            transform : Ti.UI.create2DMatrix().rotate(rotation),
        });

        var aPin = MapModule.createAnnotation({
            latitude : lat,
            longitude : lng,
            //pincolor : colour,
            customView : img,
            myid : 1 
        });

        map.addAnnotation(aPin);
    };
