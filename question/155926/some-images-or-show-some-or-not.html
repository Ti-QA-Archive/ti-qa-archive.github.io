     });

          text : responseJSON.payload[i].country_name,
         fontSize : 20,
         color : 'black',
         left : 10,
         touchEnabled : false
     });

     var img = Ti.UI.createImageView({

          right:10,
          height:50,
          width:50
     })
         tablerow.add(img);
         tablerow.add(label);
         tabledata.push(tablerow);
 }

    tableView.setData(tabledata);
    }
