    It s actually pretty easy. Simply create a view(A) on top of your tableview.
    Now listen for 'scroll' event on the tableview.
    In that event compute the "decale" from last event (just on y).
    Then add that decale (which can be negative) to the A y coordinate.
    The only trick is to do A.y = Math.max(0 A.y+decale);

    that should be enough to create the effect of instagram.
    Now the custom header is a supplement. I you create a customheader of the same height as A, and correctly initialise A position, it will look like A is part of the tableview but stick to the top bar.

  • The same effect is in SpiegelOnline-App and in my app PauliTour: if the tableview is near its boundaries the additional views appears.

    mytableView.addEventListener('scroll', function(_e) {
                var height = _e.contentSize.height - _e.size.height;
                if(_e.contentOffset.y < 50 || _e.contentOffset.y + 50 > height) {
                    if(anims.gal_is_up == false) {
                        var goUp = Ti.UI.createAnimation({
                            bottom : 0,
                            duration : 500
                        });
                        galerie.animate(goUp);
                        anims.gal_is_up = true;
                        goUp.addEventListener('complete', function() {

                        });
                    }
                } else {
                    if(anims.gal_is_up) {
                        var goDown = Ti.UI.createAnimation({
                            bottom : -60,
                            duration : 500
                        });
                        galerie.animate(goDown);
                        anims.gal_is_up = false;
                        goDown.addEventListener('complete', function() {

                        });

                    }
                }
                if(_e.contentOffset.y < 100) {
                    radioContainer.animate(Ti.UI.createAnimation({
                        duration : 100,
                        opacity : 1
                    }));
                } else {
                    radioContainer.animate(Ti.UI.createAnimation({
                        duration : 100,
                        opacity : 0
                    }));
                }
            });
