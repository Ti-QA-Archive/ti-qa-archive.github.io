  • var table = Titanium.UI.createTableView();

var row = Titanium.UI.createTableViewRow();
var view = Titanium.UI.createView({backgroundColor:'red',width:20,height:20});
row.height = 'auto';
row.add(view);
table.setData(row);

var dialog = Titanium.UI.createOptionDialog({
    title: 'Hello',
    options: ['Option 1','Option 2'],
    cancel:1,
    // not sure if this is even possible! :D
    androidView: table
});
dialog.show();

    try this one if this doesnt work then create a transparent window having table view looking like option dialog and then add chack boxes in that table view.
