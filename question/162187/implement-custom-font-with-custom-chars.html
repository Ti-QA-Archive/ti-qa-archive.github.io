Hi guys,

    var button = Ti.UI.createButton({
        title: '\f20e',
        width: '25dp',
        height: '25dp',
        top: '35dp',
        left: '15dp',
        borderRadius: '13dp',
        borderColor: '#525252',
        color: '#525252',
        font: {
            fontSize: '13dp',
            fontFamily: (Ti.Platform.osname === 'android') ? 'ionicons.ttf' : 'ionicons'
        },
    });
