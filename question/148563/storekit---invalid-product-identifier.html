var Storekit = require('ti.storekit');
Storekit.receiptVerificationSandbox = true;
Storekit.receiptVerificationSharedSecret = "XXX";


function payments_upgrade()
{
    alert("uploading");
    requestProduct("PremiumMonthly",function(products){
        alert("Got the requested products");
        console.log(products);

    });
}

function requestProduct(identifier, success)
{
    showLoading();
    Storekit.requestProducts([identifier], function (evt) {
        hideLoading();
        if (!evt.success) {
            alert('ERROR: We failed to talk to Apple!');
        }
        else if (evt.invalid) {
            alert('ERROR: We requested an invalid product!');
        }
        else {
            success(evt.products);
        }
    });
}

I know that the Product ID is correct (http://cl.ly/image/3X030W321x1p), and I'm using the right bundle identifier.
