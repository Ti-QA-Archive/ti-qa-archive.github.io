Problem with setSelectedRow on Picker

hello to all guys.
I have a problem with the setting of the values of a picker.

this is a part of code of my picker

var picker = Ti.UI.createPicker({
            top : 43,
            backgroundColor : "red"
        });

        picker.selectionIndicator = true;

        var Hours = [];
        var Minits = [];

        //Ti.App.Hours = 0;
        //Ti.App.Minit = 0;

        for (var i = 0; i < 24; i++) {
            var str = '';
            if (i < 10) {
                str = '0' + i;
            } else {
                str = '' + i;
            }
            Hours.push(str);
        }

        for (var i = 0; i < 60; i++) {

            var str = '';
            if (i < 10) {
                str = '0' + i;
            } else {
                str = '' + i;
            }

            Minits.push(str);
        }

        var column1 = Ti.UI.createPickerColumn();

        for (var i = 0; i < Hours.length; i++) {
            var row = Ti.UI.createPickerRow({
                title : Hours[i]
            });
            column1.addRow(row);
        }


        var column2 = Ti.UI.createPickerColumn();

        for (var i = 0; i < Minits.length; i++) {

            var row = Ti.UI.createPickerRow({
                title : Minits[i]
            });
            column2.addRow(row);
        }

        picker.add([column1, column2]);

  • ????i think the are problems with the time change of the position of the picker , so i solved using a simple setTimeout

    setTimeout(function(){
                        picker.setSelectedRow(0,appOre,false);
                    },1);

setTimeout(function(){
                        picker.setSelectedRow(1,appMinuti,false);
                    },1);

    or

    picker.setSelectedRow(0,appOre,false);

 setTimeout(function(){
                        picker.setSelectedRow(1,appMinuti,false);
},1);
