create back button in android application functioning as android-back?

I want my own application to have a back button, that works just as the android back button does.
It is just an application user interface requirement, otherwise android-back button could be used to navigate to previous view.
Can anyone tell me how to give that same functionality to a button in application?

  • hi,

    so you can create a button eventlistner to close your window and just over ride android's back button event like

    var win1 = Ti.UI.createWindow({
    title : 'window 1',
    backgroundColor : 'red',
    fullscreen : false,
});
win1.addEventListener('click', function() {
    var win2 = Ti.UI.createWindow({
        title : 'window 2',
        backgroundColor : 'yellow'
    });

    win2.addEventListener('android:back',function(e){
    });

    var btn = Ti.UI.createButton({
        title : 'close window 2'
    });
    btn.addEventListener('click', function() {
        win2.close();
    });
    win2.add(btn);
    win2.open({
        modal : true
    });
});
win1.open({
    modal : true
});
