Distance between two coordinates

Hi all,

I am using the below code to find distance between two coordinates, the result i am getting from this code is not correct, for 0.16 KM the result return 3014Km


lat1 = '25.116436'; lon1 = '55.390416'; lat2 = '25.114105'; lon2 = '55.392219';

function distance(lat2, lon2, lat1, lon1, unit) {
        var radlat1 = Math.PI * lat1/180;
        var radlat2 = Math.PI * lat2/180;
        var radlon1 = Math.PI * lon1/180;
        var radlon2 = Math.PI * lon2/180;
        var theta = lon1-lon2;
        var radtheta = Math.PI * theta/180;
        var dist = Math.sin(radlat1) * Math.sin(radlat2) + Math.cos(radlat1) * Math.cos(radlat2) * Math.cos(radtheta);
        dist = Math.acos(dist);
        dist = dist * 180/Math.PI;
        dist = dist * 60 * 1.1515;
        if (unit=="K") { dist = dist * 1.609344; }
        if (unit=="N") { dist = dist * 0.8684; }
        var result = Math.floor(dist * 100) / 1000000;

        return result
   }

output : 3041Km , expected 0.16Km

can anybody suggest me right solution to find distance………..
