    Hello Adam,

    Simple Example :

    ###index.xml

    <Alloy>
    <Window backgroundColor="white">
        <Button id="btn" onClick="openWindow"/>
    </Window>
</Alloy>

    ###index.js

    $.index.open();

function openWindow(){
    var title  = "Hello";   // set value you need to pass in second window
    var win = Alloy.createController("test",title).getView();
    //  pass value in second window

    win.open();
}

    ###test.xml

    <Alloy>
    <Window id="winTest" backgroundColor="white" layout="vertical">
    <Label id="lbl" top="100"/>
        <Button id="btn" top="20"  onClick="closeWindow"/>
    </Window>
</Alloy>

    ###test.js

    var args = arguments[0] || {};    //  get the value from 1st window

$.lbl.text = args;    // set value here

function closeWindow(){
    $.winTest.close();  
}

    i hope this helps. thanks.
