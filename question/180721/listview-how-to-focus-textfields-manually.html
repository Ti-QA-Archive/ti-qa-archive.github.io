    index.js

    function addUI() {
    var items = [];

    for (var i = 1; i < 5; i++) {
        items.push({
            label : {
                text : "Label " + i
            },
            field : {
                hintText : "Input " + i
            }
        });
    }

    $.section.setItems(items);
}

function focusNextField(e) {
    var currentInput = e.section.getItemAt(e.itemIndex).field;

    if(e.itemIndex+1 == e.section.getItems().length) {
        Ti.API.warn("Last field reached -> Return");
        return;
    }

    var nextInput = e.section.getItemAt(e.itemIndex+1).field;

    Ti.API.warn(currentInput.hintText); // "Input 1"
    Ti.API.warn(nextInput.hintText); // "Input 2"

    // nextInput.focus(); // <- Not working: "undefined is not a function (evaluating 'nextInput.focus()')"

    // -> No methods can be applied on <ItemTemplate>-childs
}
