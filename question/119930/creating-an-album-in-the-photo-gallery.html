  • Just got it working on Android! :)
    Pseudocode, you will need to provide your own blob (e.g. just do var blob = your_view.toImage() )

    //set filename + dir name
var filename = "test.png";
var dir = Ti.Filesystem.externalStorageDirectory+"/Your Dir Name Which Will Be Album Name/";

//create dir if it doesn't exist
var filedir = Ti.Filesystem.getFile(dir);
if (!filedir.exists()) filedir.createDirectory();

//save file to dir with filename
var file = Ti.Filesystem.getFile(dir, filename);
file.write(blob.media);    

//scan/detect the saved file - this will make the Android Gallery aware of the files i the dir
Ti.Media.Android.scanMediaFiles([file.nativePath], ["image/png"], function(e){
    //e.path = Path to the media file that was scanned.
    //e.uri = URI to the file if it was scanned and added to the media library, or null if the file was not added.

    alert(JSON.stringify(e));
});
