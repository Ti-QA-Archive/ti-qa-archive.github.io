  • Hi

    The below code is a sample xml parse example. So please check the code and i hope it will help you to complete your actual task

    var win1 = Titanium.UI.createWindow({  
    title:'XML',
    backgroundColor:'#fff'
});

    var aTableView = Ti.UI.createTableView();
    win1.add(aTableView);
    var c=null;
    c = Titanium.Network.createHTTPClient();
    c.onload = function()
        {
            Ti.API.info('IN ONLOAD ');
            xml=this.responseXML.documentElement;
            doc=xml.getElementsByTagName("PLANT");

            for(var a=0;a<doc.length;a++)
            {
                var row1=Ti.UI.createTableViewRow({
                    layout:'vertical'
                });
                var aLabel1 = Ti.UI.createLabel({
                    text : doc.item(a).getElementsByTagName("COMMON").item(0).text,
                    color : 'red',
                    font : {fontSize:15},
                    height : Ti.UI.SIZE,
                    width : Ti.UI.SIZE,
                    textAlign : 'center'
                });
                row1.add(aLabel1);
                var aLabel2 = Ti.UI.createLabel({
                    text : doc.item(a).getElementsByTagName("PRICE").item(0).text,
                    color : 'green',
                    font : {fontSize:12},
                    height : Ti.UI.SIZE,
                    width : Ti.UI.SIZE,
                    textAlign : 'center'
                });
                row1.add(aLabel2);

                    var aLabel3 = Ti.UI.createLabel({
                    text : doc.item(a).getElementsByTagName("LIGHT").item(0).text,
                    color : 'blue',
                    font : {fontSize:12},
                    height : Ti.UI.SIZE,
                    width : Ti.UI.SIZE,
                    textAlign : 'center'
                });
                row1.add(aLabel3);
                aTableView.appendRow(row1);
            }
        };
    c.ondatastream = function(e)
        {
            // ind.value = e.progress ;
            // Ti.API.info('ONDATASTREAM1 - PROGRESS: ' + e.progress);
        };
    c.onerror = function(e)
        {
            Ti.API.info('XHR Error ' + e.error);
        };
    c.open('GET', 'http://www.w3schools.com/xml/plant_catalog.xml');
    c.send();
win1.open()
