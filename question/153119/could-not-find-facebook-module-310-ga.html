could not find module &#39;facebook&#39;

<modules>
    <!-- Add the appropriate line(s) to your modules section -->
    <module platform="iphone">facebook</module>
</modules>

app.js:

var fb = require('facebook')

fb.appid = xxxxxxxxxxx
fb.permissions = ['publish_stream']; // Permissions your app needs
fb.forceDialogAuth = true
fb.addEventListener('login', function(e) {

        Ti.API.INFO( e.uid )
         fbPhoto = "http://graph.facebook.com/"+e.uid+"/picture"
