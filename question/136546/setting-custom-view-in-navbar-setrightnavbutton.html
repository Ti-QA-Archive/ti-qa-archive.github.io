Setting custom view in navbar (setRightNavButton)

Hi guys,

I need a little help with the following:

var winObj = Titanium.UI.createWindow();
winObj.setRightNavButton(memit.ui.createAvatarIcon(), {
    animated : false
});

createAvatarIcon = function() {
    var view = Ti.UI.createView({
        width : 250,
        height : 40
    });
    var img = Ti.UI.createImageView({
        image : 'images/test.png',
        width : 30,
        height : 30,
        right : 7,
        hires : Ti.Platform.displayCaps.density === 'high'
    });
    img.addEventListener('click', function() {
        /// Event logic
    });
    var label1 = Ti.UI.createLabel({
        text : L('Welcome_salute'),
        textAlign : 'right',
        font : {
            fontFamily : memit.ui.theme.fontFamilyUI,
            fontSize : 10
        },
        top : 10,
        right : 44
    });
    var label2 = Ti.UI.createLabel({
        text : 'Text here',
        textAlign : 'right',
        font : {
            fontFamily : memit.ui.theme.fontFamilyUI,
            fontSize : 14
        },
        height : 14,
        right : 44,
        bottom : 4
    });
    view.add(img);
    view.add(label1);
    view.add(label2);
    return view;
};

That worked perfectly until Ti 2.0.1.GA2, displaying the composite view generated in createAvatarIcon function in the iOS navBar. Now, the labels are not rendered along with the imageview inside the composite view. Any ideas?
