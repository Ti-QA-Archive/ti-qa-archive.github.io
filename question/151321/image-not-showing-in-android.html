Image not showing in Android

This code works in iOS but not Android. I get no errors in Android, just white space where the image should be. Any ideas why this is happening?

function floorMap(_args) {
    var self = Ti.UI.createWindow({
        //title:title,
        backgroundColor:'white',
        barColor:'orange'
    });


var view1 = Ti.UI.createView({ backgroundImage:'images/showtime/floormap.png'})

                var scrollView = Titanium.UI.createScrollView({
                contentWidth:'auto',
                contentHeight:'auto',
                showVerticalScrollIndicator:false,
                showHorizontalScrollIndicator:false,
                top:0,left:0,width:'100%',height:'100%',
                minZoomScale:1,  
                maxZoomScale:15, 
                zoomScale:1,
                oldZoom:1

            });

            scrollView.addEventListener('pinch',function(e){

                        if (e.scale>1)
                        {
                            if (e.scale>scrollView.zoomScale)
                                {
                                    scrollView.zoomScale=e.scale;
                                }
                            else
                                {
                                    scrollView.zoomScale=scrollView.oldZoom + (e.scale-1);
                                }
                            }
                        else
                        if (e.scale<scrollView.zoomScale)
                          scrollView.zoomScale=scrollView.zoomScale - (1-e.scale);
            });

            scrollView.addEventListener('doubletap', function(e) {
                if (scrollView.ZoomScale == 5.0){

                    scrollView.ZoomScale = 1.0;
                }else{


                    //scrollView.setCenter(e.x,e.y);
                    //scrollView.scrollTo(newX, newY);
                    scrollView.ZoomScale = 5.0;

                        //zoomScale = true;
                };

            });

            scrollView.addEventListener('touchend',function(e){
                scrollView.oldZoom=scrollView.zoomScale;
            });

            scrollView.add(view1);

self.add(scrollView);

    return self;
};

module.exports = floorMap;

    • How do I increase the "VM budget"?

      That does seem like the easy answer, doesn't it…

      To strictly answer your question in your tiapp.xml add/modify:

      <property name="ti.android.threadstacksize" type="int">32768</property>

      However, often the emulator runs out of memory (less often with the new android sdk tools than it used to, and most often with images). Try to run your app on a device if you haven't already. How large is your image file?
