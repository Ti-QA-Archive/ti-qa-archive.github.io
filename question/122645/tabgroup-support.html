TabGroup Support

Grettings

In the most attentive, I want to ask about a problem with the implementation of Titanium Aplication.

The Question is this:

When I implemented a TabGroup, I can’t generate a homogeneous for Android and iPhone. When I compile in iPhone the TabGroup appears at the bottom (Standard IOS) and when I compile in Android the TabGroup appears at the top of the screen, the development application that has a design TabGroup located a the bottom for both platforms. How can we fix this?

P.D Had the same problem when deployed on BlackBerry with the same design?

    You can build a custom tabGroup, here is something I am working on, enjoy!

    /*
 *
 * Closure for secretSauce
 *
 */
(function() {
    var network = [];
    var animation =[];
    var uiElements =[];

    //porgram
    animation.tab1 = true;
    animation.tab2 = false;
    animation.tab3 = false;
    animation.tab4 = false;
//    animation.tab5 = false;

    animation.animationA = Titanium.UI.createAnimation({
        delay:500,
        height:65,
        opacity:1.0,
        zIndex:9,

        duration:900,

    });

    animation.animationB = Titanium.UI.createAnimation({
        zIndex:1,
        height:50,
        duration:500,
        //opacity:0.8,
    });
    animation.animationC = Titanium.UI.createAnimation({
        delay:500,
        opacity:1.0,
        duration:2000
    });
    //ui
    uiElements.win1 = Titanium.UI.createWindow({
        backgroundColor:'#fff'
    });

    uiElements.view1 = Ti.UI.createView({
        height:'100%',
        width:'100%'
    });

    uiElements.ad = Titanium.UI.createView({
        height:'10%',
        width:'100%',
        //left:0,
        bottom:0,
        backgroundColor:'black'
    });
    uiElements.mainArea = Titanium.UI.createView({
        height:'75%',
        width:'100%',
        //left:0,
        top:0,
        backgroundColor:'#CCC'
    });
    uiElements.addlabel = Ti.UI.createLabel({
        test:"ad",
        color:"white"
    });

    uiElements.bar1 = Titanium.UI.createView({
        height:'5%',
        width:'100%',
        //left:0,
        bottom:'20%',
        backgroundColor:'black'
    });
    uiElements.tab1 = Titanium.UI.createView({
        height:50,
        width:'25%',
        left:0,
        bottom:'10%',
        backgroundColor:'red',
        id:'test',
        focusable:true,
        anchorPoint: {
            x:0.0,
            y:0.0
        }
    });

    uiElements.tab2 = Titanium.UI.createView
    ({
        height:50,
        zIndex:0,
        width:'25%',
        left:"25%",
        bottom:'10%',
        backgroundColor:'green',
        focusable:true
    });
    uiElements.tab3 = Titanium.UI.createView({
        height:50,
        zIndex:0,
        width:'25%',
        left:'50%',
        bottom:'10%',
        backgroundColor:'blue',
        focusable:true
    });

    uiElements.tab4 = Titanium.UI.createView({
        height:50,
        zIndex:0,
        width:'25%',
        right:0,
        bottom:'10%',
        backgroundColor:'yellow',
        focusable:true
    });



    uiElements.table1 = Titanium.UI.createTableView({height:'100%',width:'100%',opacity:0.0,});
    uiElements.row1 = Titanium.UI.createTableViewRow({height:100});
    uiElements.rowView1 = Titanium.UI.createView({
        backgroundColor:'red',
        width:'100%',
        height:20
    });
    uiElements.row2 = Titanium.UI.createTableViewRow({height:100});
    uiElements.rowView2 = Titanium.UI.createView({
        backgroundColor:'blue',
        width:'100%',
        height:20
    });



    uiElements.row1.add(uiElements.rowView1);
    uiElements.row2.add(uiElements.rowView2);
    uiElements.table1.appendRow(uiElements.row1);
    uiElements.table1.appendRow(uiElements.row2);

    uiElements.view1.add(uiElements.mainArea);
    uiElements.ad.add(uiElements.addlabel);
    uiElements.view1.add(uiElements.ad);
    uiElements.win1.add(uiElements.view1);
    uiElements.view1.add(uiElements.bar1);
    uiElements.view1.add(uiElements.tab1);
    uiElements.view1.add(uiElements.tab2);
    uiElements.view1.add(uiElements.tab3);
    uiElements.view1.add(uiElements.tab4);
    //uiElements.view1.add(uiElements.tab5);

    uiElements.mainArea.animate(animation.animationC);
    function animateTabs() {
        if(animation.tab1 === true) {

            uiElements.tab1.animate(animation.animationA);

        } else {

            uiElements.tab1.animate(animation.animationB);

        }

        if(animation.tab2 === true) {

            uiElements.tab2.animate(animation.animationA);

        } else {

            uiElements.tab2.animate(animation.animationB);

        }

        if(animation.tab3 === true) {

            uiElements.tab3.animate(animation.animationA);

        } else {
            uiElements.tab3.animate(animation.animationB);

        }
        if(animation.tab4 === true) {

            uiElements.tab4.animate(animation.animationA);

        } else {
            uiElements.tab4.animate(animation.animationB);

        }


    }

    uiElements.tab1.addEventListener('click', function() {

        animation.tab1 = true;
        animation.tab2 = false;
        animation.tab3 = false;
        animation.tab4 = false;
    //    animation.tab5 = false;
        animateTabs();
    });
    uiElements.tab2.addEventListener('click', function() {

        animation.tab1 = false;
        animation.tab2 = true;
        animation.tab3 = false;
        animation.tab4 = false;
    //    animation.tab5 = false;
        animateTabs();
    });
    uiElements.tab3.addEventListener('click', function() {

        animation.tab1 = false;
        animation.tab2 = false;
        animation.tab3 = true;
        animation.tab4 = false;
    //    animation.tab5 = false;
        animateTabs();
    });
    uiElements.tab4.addEventListener('click', function() {

        animation.tab1 = false;
        animation.tab2 = false;
        animation.tab3 = false;
        animation.tab4 = true;
        //animation.tab5 = false;
        animateTabs();
    });

    animateTabs();

    uiElements.win1.open();

    function initApp(){
        uiElements.mainArea.add(uiElements.table1);
    }
    initApp();

})();

    You would have to create your own custom tab controller to obtain the effect you need.
