if (offlineversion < aaa){

    var dialog = Titanium.UI.createOptionDialog({
    options:['Ok, download please'+ 'Cancel'],
    cancel:1,
    title:'',

    dialog.show();

dialog.addEventListener('click',function(e)
{

    if (e.index == 0)
        {     // get new database }

    }

    Hi Adrian,

    there was a little mistake.

    options:['Ok, download please'+ 'Cancel'],

    should be

    options:['Ok, download please', 'Cancel'],

    This is working:

    var offlineversion = 1;
var aaa = 2;

if (offlineversion < aaa) {

    var dialog = Titanium.UI.createOptionDialog({
        options : ['Ok, download please', 'Cancel'],
        cancel : 1,
        title : ''

    });
    dialog.addEventListener('click', function(e) {

        if (e.index == 0) {
            // get new database
            Ti.API.info("DB download confirmed.")
        }

    });

    dialog.show();
}

    Greetings,

    Matthias
