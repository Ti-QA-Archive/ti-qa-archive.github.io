window.open() and window.close() too fast

I have an issue where my window starts stacking if I open and close it too fast. Its like the same window starts duplicating itself on top of each other. I am calling the open and close method through a custom event that lives in app.js:

var accountWindow = Titanium.UI.createWindow({
    title:'New Account',
    url:'main_windows/create.js',
    tabBarHidden:true
});

Ti.App.addEventListener('toggleAccount', function(event)
{
    if (event.open == true)
    {
        accountWindow.open({
            modal:true,
            modalTransitionStyle:Ti.UI.iPhone.MODAL_TRANSITION_STYLE_COVER_VERTICAL,
            modalStyle:Ti.UI.iPhone.MODAL_PRESENTATION_FULLSCREEN
        });
    }
    else
    {
        accountWindow.close();
    }
});

anyone experience this before?
