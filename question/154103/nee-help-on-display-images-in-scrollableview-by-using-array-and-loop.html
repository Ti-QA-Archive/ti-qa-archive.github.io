    • ok, as i said before that you can not add array of images to scrollview. you need to add a imageView in scrollView while looping array.

      var scrollViewHorizontalLatest = Titanium.UI.createScrollView({
    contentWidth:Ti.UI.SIZE,
    contentHeight:90,
    height:90,
    width:'100%',
    layout:'horizontal',
    top:15,
    left:0
});

for (var j = 0; j < arrImagesLatest.length ; j++) { 

var imgView = Ti.UI.createImageView({
    image : arrImagesLatest[j],
    height:15,
    width:55
}); 

scrollViewHorizontalLatest.add(imgView);
}
