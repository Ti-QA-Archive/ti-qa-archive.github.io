    Hello Davidson,

    try this :

    var win = Ti.UI.createWindow({
    backgroundColor : "white"
});

var lblEmail = Ti.UI.createLabel({
    left : "10dp",
    top : "20dp",
    text : "Email ID"
});

var txtEmail = Ti.UI.createLabel({
    left : "60dp",
    top : "20dp",
    width : "100dp",
    height : "40dp",
    borderColor : 'gray',
    borderWidth : "2dp",
});

var lblPass = Ti.UI.createLabel({
    left : "10dp",
    top : "100dp",
    text : "Password"
});

var txtPass = Ti.UI.createLabel({
    left : "60dp",
    top : "100dp",
    width : "100dp",
    height : "40dp",
    borderColor : 'gray',
    borderWidth : "2dp",
});

win.add(lblEmail);
win.add(txtEmail);
win.add(lblPass);
win.add(txtPass);

var btnSubmit = Ti.UI.createButton({
    height : "100dp",
    width : "100dp",
    bottom : "50dp",
    title : "Submit"
});

btnSubmit.addEventListener('click',function(e){
    if (txtEmail.value.trim() == '')
    {
        alert('enter email');
        return;
    }else if(txtPass.value.trim() == '')
    {
        alert('enter password');
        return;
    }else{
        // your open window code here..
    }
});

win.open();

    i hope it's help to you. thanks.
