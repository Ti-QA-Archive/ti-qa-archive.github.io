    NSString *urlString=[NSString stringWithFormat:@"%@",[request URL]];
    //check if callback url is specified
    if ([urlString rangeOfString:@"///oauthcallback"].location == NSNotFound)
    {
    //Do nothing if requested url is not callback url;

    } else
    {
    //Do authentication for request
    [self gotPin: nil];
    return NO;

    }
