    Here's your simplified, edited code. Note that I had to change the editable property of the data object to true to get 'swipe to delete` working in the iPhone simulator.

    var data =[{title:"Inserisci Nuovo", backgroundColor:'#AFEEEE', clickName:'New',editable:true}];

tableview.addEventListener('delete',function(e)
    data.splice(e.index,1);
    Titanium.App.Properties.setList('ladata',data);
});
