    Hi

    If you want to test if a specific date falls in-between two other dates then use this simple function;

    function isDateInRange(lower, upper, date) {
    return (now >= start && now <= stop) ? true : false;
}

    You can test this as follows - this will pass;

    var start = new Date('2013/01/01 12:00:00');
var stop = new Date('2013/02/01 12:00:00');
var now = new Date();
Ti.API.info('in range = ' + isDateInRange(start, stop, now));
// in range = true

    This will fail;

    var start = new Date('2012/01/01 12:00:00');
var stop = new Date('2012/02/01 12:00:00');
var now = new Date();
Ti.API.info('in range = ' + isDateInRange(start, stop, now));
// in range = false

    • You are right I did not understand you first time, but I have it now.

      So you need to make sure that NOW falls between two TIMES that are related to TODAY and possible TOMORROW.

      function addTime(strTime) {
    var time = strTime.split(':');
    var d = currentDate();
    d.setHours(time[0]);
    d.setMinutes(time[1]);
    d.setSeconds(time[2]);
    return d;
}
function isTimeInRange(strLower, strUpper) {
    var now = new Date();
    now.setHours(2);
    now.setDate(now.getDate() + 1);
    var lower = addTime(strLower)
    var upper = addTime(strUpper)
    if (lower > upper) {
        upper.setDate(upper.getDate() + 1);
    }
    return (now >= lower && now <= upper) ? true : false;
}
function currentDate() {
    var d = new Date();
    return new Date(d.getFullYear(), d.getMonth(), d.getDate());
}
Ti.API.info('isTimeInRange: ' + isTimeInRange('20:00:00', '06:00:00'));

      function currentDate() {
    var d = new Date();
    return new Date(d.getFullYear(), d.getMonth(), d.getDate());
}
function addTime(strTime) {
    var time = strTime.split(':');
    var d = currentDate();
    d.setHours(time[0]);
    d.setMinutes(time[1]);
    d.setSeconds(time[2]);
    return d;
}
function isTimeInRange(strLower, strUpper) {
    var now = new Date();
    //now.setHours(15);
    var lower = addTime(strLower);
    var upper = addTime(strUpper);

    var inRange = false;

    if (upper > lower) {
        // opens and closes in same day
        inRange = (now >= lower && now <= upper) ? true : false;
    } else {
        // closes in the following day
        inRange = (now >= upper && now <= lower) ? false : true;
    }

    Ti.API.info('lower: ' + lower.toTimeString());
    Ti.API.info('now: ' + now.toTimeString());
    Ti.API.info('upper: ' + upper.toTimeString());

    return inRange;
}
Ti.API.info('isTimeInRange: ' + isTimeInRange('20:00:00', '06:00:00'));

    function formatDate(date) {
    var d = new Date(date);
    var hh = d.getHours();
    var m = d.getMinutes();
    var s = d.getSeconds();
    var dd = "AM";
    var h = hh;
    if (h >= 12) {
        h = hh-12;
        dd = "PM";
    }
    if (h == 0) {
        h = 12;
    }
    m = m<10?"0"+m:m;

    s = s<10?"0"+s:s;

    /* if you want 2 digit hours:
    h = h<10?"0"+h:h; */

    var pattern = new RegExp("0?"+hh+":"+m+":"+s);

    var repalcement = h+":"+m;
    /* if you want to add seconds
    repalcement += ":"+s;  */
    repalcement += " "+dd;    

    return date.replace(pattern,repalcement);
}
