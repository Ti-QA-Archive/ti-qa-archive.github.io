How to link one page with another

Hi all,

How can i link one page with another page for android phone app? For example, clicking button A which links to another page.

Thank you. :)

    This can be done through the use of event listeners attached to the button.

    var win1 = Titanium.UI.createWindow({
    title : 'win 1',
    backgroundColor : '#fff',

});

var button = Titanium.UI.createButton({
    title : 'Button A',
    width : 140,
    height : 40
});
win1.add(button);

button.addEventListener('click', function() {
    //open new window when button clicked
    var win2 = Titanium.UI.createWindow({
        title : 'new window',
        backgroundColor : '#ff0000'
    });
    var label2 = Titanium.UI.createLabel({
        color : '#999',
        text : 'I am Window 2',
        textAlign : 'center',
        width : 'auto'
    });

    win2.add(label2);

    win2.open();
});

win1.open();

 var win = Ti.UI.createWindow(args);

 var button - Ti.UI.createButton({
    title:args.buttonTitle,
    width:140,
    height:40
 });
 win.add(button)

 button.addEventListener....//create event listener here

 //add other items to the default window

 return win;
});

      Then, when you create your window, you will use the new function:

      var win1 = createDefaultWindow({
    title:'win 1',
    backgroundColor:'#fff',
    buttonTitle:'click me'   //note how we pass custom properties to the
                //button via the args
});

win1.open();
