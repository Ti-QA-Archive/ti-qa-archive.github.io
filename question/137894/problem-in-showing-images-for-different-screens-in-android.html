  • Shashi, See this…

    
Ti.App.Properties.setInt('MaxWidth', Ti.Platform.displayCaps.platformWidth);
Ti.App.Properties.setInt('MaxHeight', Ti.Platform.displayCaps.platformHeight);

var screenWidth = Ti.App.Properties.getInt('MaxWidth');
var screenHeight = Ti.App.Properties.getInt('MaxHeight');

function GetHeight(value) {
    var temp = (value * 100) / 480;
    return parseInt((screenHeight * temp) / 100);
}

function GetWidth(value) {
    var temp = (value * 100) / 320;
    return parseInt((screenWidth * temp) / 100);
}

var win = Ti.UI.currentWindow;

var img = Ti.UI.createImageView({
    height :GetHeight(200),
    width : GetWidth(200),
    top :GetHeight(10),
    image : 'image.png'
});
win.add(img);
