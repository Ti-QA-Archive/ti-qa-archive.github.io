Post all field values

I've created an array of textfields which I would then like to submit to a remote mysql database, but struggling with how to get all the values inputed so I can post these to a php file.

Just while I'm trying to work out the submit button just brings up an alert, but I can only every seem to get the last value.

var win = Titanium.UI.createWindow({
    title :'test'});

var arrayLength =10;
var label = [];
var vsb = [];
var rs = [];

for( i =0; i < arrayLength; i++){

    label[i] = Titanium.UI.createLabel({
        id : 'label' + i,

        top : (i+1) *50,
        text : 'Till' + (i+1),
        width :60,
        height :40,
        font : {
            fontSize : 14,

        },
    });
    win.add(label[i]);

    vsb[i]= Ti.UI.createTextField({
        id : 'vsb' + i,
        height :40,
        width :60,
        hintText :'VSB',
        top : (i+1) *50,
        left : 80,
        value: '',
        keyboardType : Ti.UI.KEYBOARD_PHONE_PAD,
        returnKeyType : Ti.UI.RETURNKEY_DONE,
        borderStyle : Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
        });
    win.add(vsb[i]);

    rs[i]= Ti.UI.createTextField({
        id : 'rs' + i,
        height :40,
        width :60,

        top : (i+1) *50,
        left : 150,
        keyboardType : Ti.UI.KEYBOARD_PHONE_PAD,
        returnKeyType : Ti.UI.RETURNKEY_DONE,
        borderStyle : Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
        });
    win.add(rs[i]);

    };

    var btn = Ti.UI.createButton({
        title:'Submit',
        top:840,

    {
        var params = {

    });

win.open();

    btn.addEventListener('click',function(e)
    {
for (int i = 0;i<vsb.length ; i++){
        var params = {  
                 vsb: vsb[i].value,
                 rs: rs[i].value  
             };

    alert(params);
}
    });
