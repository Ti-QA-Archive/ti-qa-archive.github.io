  • Hi Nik Viradiya,

    I think you will get your answer drop down list with keyword searching like auto complete in this following code and you can also get help this link autoComplete

    var win = Ti.UI.createWindow({
    backgroundColor : 'white',
    layout : 'vertical'
});

//SEARCH BAR
var search = Titanium.UI.createSearchBar({
    barColor : 'red',
    height : 50,

    hintText : 'What\'s It Worth?',
    top : 0
});

var tableData = [ {title: 'Apples'}, {title: 'Bananas'}, {title: 'Carrots'}, {title: 'Potatoes'} ];

var autocomplete_table = Titanium.UI.createTableView({
    search : search,
    scrollable : true,
    top : 0,
    data : tableData
});
win.add(autocomplete_table);

win.open();
