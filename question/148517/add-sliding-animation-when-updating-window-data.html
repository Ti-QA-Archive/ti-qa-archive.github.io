Add sliding animation when updating window data

I have a window that displays data from a record retrieved from the database. When the user clicks a "next record" button I would like to update the window with the new data and then reveal the new data with a sliding animation like you get when you open a new window in a NavigationGroup. How can I achieve this?

  • hi,

    test this demo

    var indx = 0;
var self = Ti.UI.createWindow({
    // backgroundColor : '#666666'
    backgroundColor : 'white',
    // backgroundColor : 'stripped',
    // titleImage : 'img/1-Star.png',
    // url : 'mapView.js'
});
var arrViews = [];
for (var i = 0; i < 10; i++) {
    var view = Ti.UI.createView({
        top : 0,
        bottom : 0,
        width : 320
    });

    var lbl = Ti.UI.createLabel({
        text : 'page ' + (i + 1),
        width : Ti.UI.SIZE,
        height : 20
    });
    view.add(lbl);

    arrViews.push(view);
}

var scrollView = Ti.UI.createScrollableView({
    views : arrViews,
    scrollingEnabled : false,
    top : 0,
    bottom : 60,
    width : 320
});
scrollView.addEventListener('scroll', function(e) {
    indx = e.currentPage;
});
self.add(scrollView);
var btn = Ti.UI.createButton({
    title : 'Next',
    bottom : 10,
    width : 60,
    height : 40
});

btn.addEventListener('click', function(e) {
    scrollView.scrollToView(indx + 1);
});
self.add(btn);

self.open({
    modal : true
});
