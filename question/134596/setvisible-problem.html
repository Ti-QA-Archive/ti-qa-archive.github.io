  • Hi ilan, try this.

    var win = Titanium.UI.createWindow({backgroundColor : 'gray'});

win.open();

var t = Ti.UI.createButton({
    title : "Test",
    height : 35,
    width : 100,
    top : 100
});

t.addEventListener('click',function(e){
    // e.source.visible = false;
    // this.visible = false;
    t.visible = false;
})

win.add(t);
