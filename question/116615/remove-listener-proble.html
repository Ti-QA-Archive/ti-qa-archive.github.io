Remove listener proble

Hi everyBody!!

I got a problem to remove an eventListener and I don't know what i am doing wrong. Can you help me?


row.addEventListener('click',function(e) {
                //db.execute('DELETE FROM LISTE_DETAIL');
                var db = Titanium.Database.open('mydb');
                db.execute('INSERT INTO LISTE_DETAIL (LABEL) VALUES(?)',e.row.title);
                e.row.removeEventListener('click',function() {});
                db.close(); // close db 
});

    you need to declare the function separately

    function rowClickFunction(e) {
   //db.execute('DELETE FROM LISTE_DETAIL');
   var db = Titanium.Database.open('mydb');
   db.execute('INSERT INTO LISTE_DETAIL (LABEL) VALUES(?)',e.row.title);
   e.row.removeEventListener('click',rowClickFunction);
   db.close(); // close db 
});

row.addEventListener('click',rowClickFunction);
