Animation is not drawn on Android when the event is "open"

I use Titanium studio 2.1.2, Titanium SDK 2.1.3
I use Mac OS X 10.7.4 and Windows 7 x64.
Both OS'es make similar behaviour.
Also I use Android Phone Motorola Atrix 4g based on Tegra 2 SoC with 2.3.7 Android version OS.

I try to make animation like "slide-in from bottom".
Current behavior: yellow block appears in the end point, it doesn't slide.
The same behavior is in Android Emulator 4.1.2

Here is my code.

var win = Ti.UI.createWindow({
    fullscreen: true,
    hideNavBar: true,
    backgroundColor: '#ffffff'
});

var box = Ti.UI.createView({
    height          : '100px',
    width           : '300px',
    backgroundColor : '#ffff44',
    top             : '100%'
});
win.add(box);

win.addEventListener('open', function() {
    var anim = Ti.UI.createAnimation();
    anim.duration = 500;
    anim.top = '40%'; // move down 100px
    box.animate(anim);
});

win.open();

If i change 'open' event to 'click' event, animation is smooth.
If this question has been discussed and/or solved already, please, give me link to it.

    • Hi Dmytro,

      if % is not working on Android as expected, we can calculate pixel from %.

      This should be a very flexible solution for you.

      Also i realized, that 500 for that animation was way too fast for my Android device, so i changed ist to 1000.

      You may also read Titanium.Platform.DisplayCaps and Transitioning to the New UI Layout System.

      var win = Ti.UI.createWindow({
    fullscreen: true,
    hideNavBar: true,
    backgroundColor: '#ffffff'
});

var box = Ti.UI.createView({
    height          : '100px',
    width           : '300px',
    backgroundColor : '#ffff44',
    // top             : '100px' // '100%'
    top             : Ti.Platform.displayCaps.platformHeight
});
win.add(box);

win.addEventListener('open', function() {
//    alert("open");
    var anim = Ti.UI.createAnimation();
    anim.duration = 1000; // 500 was too fast on my device
    anim.top = '200px'; 
    anim.top = verticalPercentageToPixel(40);
    box.animate(anim);
});

// calculate % to pixel 
function verticalPercentageToPixel(percentage){
    var pixel = (Ti.Platform.displayCaps.platformHeight / 100) * percentage;
    Ti.API.info("platformHeight:"+ Ti.Platform.displayCaps.platformHeight);
    Ti.API.info("percentage:"+ percentage);
    Ti.API.info("pixel:"+ pixel);
    return pixel; 
};


win.open();

      Greetings,

      Matthias

  • I tested next code and it worked for me on phone. Didn't test it on emulator:

    var win = Ti.UI.createWindow({
    fullscreen: true,
    hideNavBar: true,
    height: '100%',
    backgroundColor: '#ffffff'
});

var box = Ti.UI.createView({
    height          : '100px',
    width           : '300px',
    backgroundColor : '#ffff44',
    top             : '100%'
});
win.add(box);

var func =  function() {
    setTimeout(function() {
        var anim = Ti.UI.createAnimation();
        anim.duration = 500;
        anim.top = '40%'; // move down 100px
        box.animate(anim);
    }, 1000);                            //100ms is enough for real device
}  

win.addEventListener('open', func);

win.open();
