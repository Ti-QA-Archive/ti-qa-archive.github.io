TableRows disappear when scrolling on Android

I have a tab that displays a tableview. I have a view for each row and when I scroll down and then back up, the contents of the rows disappear. It only affects 1.5, 1.4 works fine. Here is the code I'm using:

var data = new Array(1);
var screenWidth = Titanium.Platform.displayCaps.platformWidth;
for (var row = 0; row < 20; row++)
{
    var tableRow = Titanium.UI.createTableViewRow({
        width: screenWidth
        ,row: row
        ,height: 40
    });
    var testView = Titanium.UI.createView({
        width: screenWidth,
        height: 40,
        left: 0
    });
    var testName = Titanium.UI.createLabel({
        color: '#D1D1D1',
        text: 'A name',
        left: 22,
        top: 4,
        font:{fontSize:12},
        width: screenWidth - 22
    });
    testView.add(testName);
    tableRow.add(testView);
    data.push(tableRow);
}
var tableview = Titanium.UI.createTableView({
    data: data    
});

Titanium.UI.currentWindow.add(tableview);
