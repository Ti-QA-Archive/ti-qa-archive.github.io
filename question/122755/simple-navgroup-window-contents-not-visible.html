Simple NavGroup window contents not visible

Hi,

I am using a Navgroup with a scroll view. When navigating to the next window, the contents of window display the fist time, however the contents do not display again if navigating to the window in the future.

Adding the 'open' event to the partially visible window, setting visible to true, causes the contents to 'jerk' into view. Setting other properties can also cause the window do display again for example setting the scroller height.

Any ideas? I am using Ti API 1.7.1

Thanks,
Mike.

app.js – NOTE this is an iPad project with a Split View.


Titanium.UI.setBackgroundColor('#000');

//
var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});

var label1 = Titanium.UI.createLabel({
    color:'#999',
    text:'I am Window 1',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
});

win1.add(label1);

//
//
var win2 = Titanium.UI.createWindow({  
    title:'Window 2',
    backgroundColor:'#fff'
});

var label2 = Titanium.UI.createLabel({
    color:'#999',
    text:'I am Window 2 - Tap to Navigate to Window 3',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
});

win2.add(label2);


var win3 = Titanium.UI.createWindow({  
    title:'Window 3',
    backgroundColor:'#fff'
});

var scrollView3 = Ti.UI.createScrollView();

var label3 = Titanium.UI.createLabel({
    color:'#999',
    text:'I am Window 3',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
});

scrollView3.add(label3);
win3.add(scrollView3);

var navGroup = Titanium.UI.iPhone.createNavigationGroup(
    { window:win2}
);


var mainSplitView = Titanium.UI.iPad.createSplitWindow({
        masterView: win1,
        detailView: navGroup
    });


label2.addEventListener('click',function() {
    navGroup.open(win3);
});

mainSplitView.open();
