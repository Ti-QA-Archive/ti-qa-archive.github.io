  • Hi Gurkan,

    the content of your remote file should be:

    [ {
    "square":'images/gallery/square_1.png',
    "image":'images/gallery/1.jpg',
    "note:" '1 image note'
 },{
    "square":'images/gallery/square_1.png',
    "image":'images/gallery/1.jpg',
    "note:" '1 image note'
 },{
    "square":'images/gallery/square_1.png',
    "image":'images/gallery/1.jpg',
    "note:" '1 image note'
 }]

    In your controler you can read remote JSON in this manner:

    ctrl.getImages = function(params,callback) {
    var url = 'URL';
    var xhr = Ti.Network.createHTTPClient({
        timeout : 15000
    });
    xhr.onload = function() {
        try {
           var images = JSON.parse(this.responseText);
        } catch(e) {
            Ti.API.log(e);
        };
    };
    xhr.open('GET', url);
    xhr.send();
};

    In your view you can play:

    
var imageViews = [];
ctrl.getImages({},function(data) {
    for (var i =0;i<data.length;i++) {
    imageView[i] = Ti.UI.createImageView({image:data[i].image});
    }
}

    Hope it helps.

    Rainer
