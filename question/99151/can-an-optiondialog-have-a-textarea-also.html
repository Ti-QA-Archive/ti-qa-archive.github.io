  • you still need to deal with shifting the screen up when the textarea is clicked otherwise the keyboard will cover the area where the user is typing

    function addPopupWindow( win1 ) {

    if ( win1.popUp != undefined ) {
        win1.remove(win1.popUp);
    }

    var popup_view = Titanium.UI.createView({
        width:320,
        backgroundColor:'#999999',
        borderColor:'#333333',
        left:0,
        bottom:-240,
        height:200
    });
    win1.popUp = popup_view;
    popup_view._parent = win1;

    var closeBtn = Ti.UI.createButton({
        title:'Close Window', width:110, height:30, top:5,
        font:{fontSize:14,fontWeight : 'bold'},
        borderColor:'#666',
        borderWidth:2,
        borderRadius:10
    });
    var oneBtn = Ti.UI.createButton({
        title:'One Button', width:110, height:30, top:45,
        font:{fontSize:14,fontWeight : 'bold'},
        borderColor:'#666',
        borderWidth:2,
        borderRadius:10
    });
    var otherBtn = Ti.UI.createButton({
        title:'Cancel', width:110, height:30, top:85,
        font:{fontSize:14,fontWeight : 'bold'},
        borderColor:'#666',
        borderWidth:2,
        borderRadius:10
    });

    var textArea = Ti.UI.createTextArea({
        width:210, height:50, top:130,
        font:{fontSize:14,fontWeight : 'bold'},
        borderColor:'#666',
        borderWidth:1
    });
    var popup_animation = Titanium.UI.createAnimation();
    // this will set it in to view
    popup_animation.bottom = 0;
    // ... in half a second
    popup_animation.duration = 500;

    closeBtn.addEventListener('click', function(e) {
        closePopUpView(popup_view);
    });
    popup_view.add(closeBtn, otherBtn, oneBtn, textArea);

    // add it when you need to (it will still be 'off screen')
    win1.add(popup_view);

    // then when you need it to 'appear' / pop up ... use:
    popup_view.animate(popup_animation);
}

function closePopUpView(_view, _btn) {
    var popup_animation = Titanium.UI.createAnimation();
    if (_view != null) {
        // this will set it in to view
        popup_animation.bottom = -240;
        // ... in half a second
        popup_animation.duration = 500;
        _view.animate(popup_animation, function(){
            _view._parent.remove(_view);
        });

    }
}
