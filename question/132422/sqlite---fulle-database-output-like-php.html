    var rows = db.execute('SELECT rowid, name FROM people');

while(rows.isValidRow()) {
    Ti.API.info('Person --- ROWID: ' + rows.fieldByName('rowid') + ', name:' + rows.field(1));
    rows.next();
}
rows.close();
db.close();
