  • Hear a sample code:

    var win = Ti.UI.createWindow({
    backgroundColor : "#fff",
    layout : "vertical"
});

var clickLbl = Ti.UI.createLabel({
    top:0,
    text : "Click below for current screenshot",
    color : "#000"
});
win.add(clickLbl);

var image = Ti.UI.createImageView({
    width : 150,
    height : 150,
    backgroundColor : "#000",
    borderRadius : 10,
    top : 10
});
win.add(image);

var contentURL = 'http://assets.appcelerator.com.s3.amazonaws.com/video/media.m4v';
if (Ti.Platform.name == 'android')
    contentURL = "http://assets.appcelerator.com.s3.amazonaws.com/video/media.3gp";

var videoPlayer = Titanium.Media.createVideoPlayer({
    top : 20,
    width : 300,
    height : 300,
    url : contentURL,
    backgroundColor : "#aabbcc",
    movieControlMode : Titanium.Media.VIDEO_CONTROL_DEFAULT,
    scalingMode : Titanium.Media.VIDEO_SCALING_ASPECT_FIT,
    autoplay : false
});
win.add(videoPlayer)

image.addEventListener("click", function(){

        /**** decomment this for toImage() method ***        
        var imageBlob = videoPlayer.toImage();
        image.setImage(imageBlob)        
        *****/

        /**** decomment this for takeScreenshot() method ***/
        Ti.Media.takeScreenshot(function(e){
            image.setImage(e.media)
        });
        /****/

        Ti.API.info("Screenshot!");    

});

win.open();
