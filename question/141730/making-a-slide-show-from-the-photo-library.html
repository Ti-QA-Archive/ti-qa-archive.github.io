  • See the code below

    var css =      {
        backgroundColor : '#FFFFFF',
        type : 'ImageView',
        images : [],
        duration : 1000,
        repeatCount : 0
};

var slideShow = Ti.UI['create' + css.type](css);

var images = ['1.jpg,', '2.jpg'];
slideShow.images = images;

slideShow.start();
