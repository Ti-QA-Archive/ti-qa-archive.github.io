  • this oughta work (at first i thought it didn't work, but you cant see a white indicator on a white background :P )

    var window = Titanium.UI.currentWindow;

var data = [];
var indicators = [];
for (var i=0; i < 5; i++) {
    var row = Ti.UI.createTableViewRow({
        height:50
    });
    var actInd = Ti.UI.createActivityIndicator({
        height:50, width:50, right: 0,
        style:Titanium.UI.iPhone.ActivityIndicatorStyle.DARK    //DARK over white background, PLAIN over dark background... doigh
    });
    var label = Titanium.UI.createLabel({
        text: "indicator:",
        left: 0, bottom:0,  height:50,
    });
    row.add(label);
    row.add(actInd);

    data.push(row);
    indicators.push(actInd);        //array of indicators
};

var tableview = Titanium.UI.createTableView({
    data:data
});

tableview.addEventListener('click', function (evt) {
    indicators[evt.index].show();    //activate indicator on clicked row
})

window.add(tableview);
