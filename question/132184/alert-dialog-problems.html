    here is what you need to do. Hope it helps 

    Logout.addEventListener('click', function(e){


   var alertDialog = Titanium.UI.createAlertDialog({ title: 'Confirm', message: 'Sure to log out', buttonNames: ['OK','No!'] }); 
   alertDialog.show();

   alertDialog.addEventListener('click', function(e){

       if(e.index == 0)
       {
           alert('you are log out');
            // your additional code logic
       }
       else
       {
           alert('go back to your app');
           // your additional code logic
       }
   }) ;

});
