    you can add an eventlistener on the alert dialog so you know when it is clicked and what was clicked.

    var d = Titanium.UI.createAlertDialog({
   title: 'Nattklubb Tranan',
   message: 'Din kommentar skickades utan problem.',
   buttonNames: ['Ok'],
});
d.addEventListener('click',function(e){
});

d.show();
