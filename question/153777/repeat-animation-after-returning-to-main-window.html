Repeat animation after returning to main window

I have a window with 2 views in it. Each view has their own animation. When clicked on the view, it will open another window with transition.

Now, what I want is that when i click on back in the new window, it will return to the main window and the animation scene plays again. This is where I'm stuck.

I'm currently building for iPhone and using 3.0.0.

Here are some of my code:

var mainWin = Ti.UI.createWindow({backgroundColor:'white'});

var view1 = Ti.UI.createImageView({
    width:screenWidth,
    height:75,
    top:86,
    backgroundColor:'red',
    right:-320,
});
mainWin.add(view1);

var transform1 = Titanium.UI.create2DMatrix();
transform1 = transform1.translate(-480,0);

var animation1 = Titanium.UI.createAnimation({
    transform: transform1,
    duration: 1000,
    curve: Titanium.UI.ANIMATION_CURVE_EASE_IN_OUT,
});


var view2 = Ti.UI.createImageView({
    width:screenWidth,
    height:75,
    top:161,
    backgroundColor:'blue',
    left:-320,
});
mainWin.add(view2);

var transform2 = Titanium.UI.create2DMatrix();
transform2 = transform2.translate(250,0);

var animation2 = Titanium.UI.createAnimation({
    transform: transform2,
    duration: 800,
    curve: Titanium.UI.ANIMATION_CURVE_EASE_IN_OUT
});

view2.addEventListener('click', function(){
    var mainWin2 = Ti.UI.createWindow({backgroundColor:'white'});

    var view1 = Ti.UI.createImageView({
        width:screenWidth,
        height:75,
        top:86,
        backgroundColor:'red',
    });
    mainWin2.add(view1);    

    view1.addEventListener('click', function(){
        mainWin2.close({transition:Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT});
    });

    mainWin2.open({transition:Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});
});

mainWin.open();
