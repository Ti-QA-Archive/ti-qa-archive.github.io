Alloy : Model adapter properties

Hi

idAttribute and db_name seem not to work. Alloy 0.3.6 (I cleaned the app and wipe out all app-related settings)

nike.js

exports.definition = {
    config : {
        "columns" : {
            "descr" : "String",
            "year" : "String",
            "size" : "int",
            "image" : "String"
        },
        "defaults" : {
            "descr" : "N/A",
            "color" : "N/A",
            "size" : "N/A",
            "image" : "String"
        },
        "adapter" : {
            "type" : "sql",
            "collection_name" : "nike",
            "db_name" : "nikes",
            "idAttribute": "descr"
            //"db_file": "/nikes.sqlite",
        }
    },

    extendModel : function(Model) {
        _.extend(Model.prototype, {
            // Extend, override or implement Backbone.Model
        });

        return Model;
    },

    extendCollection : function(Collection) {
        _.extend(Collection.prototype, {
            // Extend, override or implement Backbone.Collection
        });

        return Collection;
    }
};

With the above adapter settings, I am still having [underscore]alloy[underscore].sql and another id column as the primary key.
Am I supposed to have nikes.sql and descr as primary key ?

  • Accepted Answer

    The "sql_new" adapter was introduced in 0.3.5, which includes all the brand new functionality, including what you are trying to do here. Change the adapter type to sql_new and it should work fine.

    NOTE: "sql_new" will replace "sql" in alloy 1.0.0. In 1.0.0, your code would work.

    actually this is a constant name define in

    &#x2F;usr&#x2F;local&#x2F;lib&#x2F;node_modules&#x2F;alloy&#x2F;Alloy&#x2F;lib&#x2F;alloy&#x2F;sync&#x2F;sql.js

    so if you want to set dynamic name then just reaplace below function in above sql.js file

    in this code i have used dbName attribute name for database name.

    function InitAdapter(config) {
    if (!db) {
        if (Ti.Platform.osname === 'mobileweb' || typeof Ti.Database === 'undefined') {
            throw 'No support for Titanium.Database in MobileWeb environment.';
        } else {
            var dbName = "_alloy_";

            if (config) {                
                if (config.adapter) {                    
                    if (config.adapter.dbName) {                        
                        dbName = config.adapter.dbName;
                    }
                }
            }

            db = Ti.Database.open(dbName);
        }
        module.exports.db = db;

        // create the migration table in case it doesn't exist
        db.execute('CREATE TABLE IF NOT EXISTS migrations (latest TEXT, model TEXT)');
    }
    return {};
}
