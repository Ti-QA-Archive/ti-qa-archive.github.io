    • something like that:

      var tabGroup = Titanium.UI.createTabGroup();
var win1 = Titanium.UI.createWindow({
    title:'Deals',
    backgroundColor:'#fff',
});
var logoutBtn = Titanium.UI.createButton({
    height:31,
    width:200,
    title:'logout',
    bottom:10
});
win1.add(logoutBtn);
var tab1 = Titanium.UI.createTab({
    icon:'images/dealsIcon.png',
    title:'DEALS',
    window:win1
});
tabGroup.addTab(tab1);
tabGroup.open();
tabGroup.hide();


var loginWin = Titanium.UI.createWindow({
    title:'Deals',
    backgroundColor:'#fff',
});
var loginBtn = Titanium.UI.createButton({
    height:31,
    width:200,
    title:'login',
    bottom:10
});
loginWin.add(loginBtn);
loginWin.open();
loginBtn.addEventListener('click', function(){
    loginWin.hide();
    tabGroup.show();

});
logoutBtn.addEventListener('click', function(){
    loginWin.show();
    tabGroup.hide();

});
