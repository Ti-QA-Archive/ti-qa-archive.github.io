    Hi mahmoud,

    try this code.

    var index = $.sliderViw.getViews();
setInterval(function(e) {
        if (($.sliderViw.currentPage + 1) >= index.length) {
            $.sliderViw.scrollToView(0);
        } else {
            $.sliderViw.scrollToView($.sliderViw.currentPage + 1);
        }
    }, 5000);
