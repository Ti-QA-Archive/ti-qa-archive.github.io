  • From window_navbar.js in Kitchen Sink:

    //
// TOGGLE LEFT NAV BUTTON
//
var b7 = Titanium.UI.createButton({
    title:'Left Nav',
    height:40,
    width:145,
    top:160,
    left:10
});

var leftnav=false;
b7.addEventListener('click', function()
{
    if (!leftnav)
    {
        var b = Titanium.UI.createButton({title:'Left Nav'});
        win.leftNavButton = b;
        leftnav = true;
    }
    else
    {
        win.setLeftNavButton(null);
        leftnav = false;
    }
});

win.add(b7);
