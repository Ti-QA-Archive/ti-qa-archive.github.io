    function randomquiz(number) {
    var nums = new Array();
    for (var i = 0; i < number; i++)
        nums[i] = i;
    nums.sort(function(a, b) {
        return Math.random() - 0.5;
    });
    return nums;
}

  • If your data is in a database, then use the random() function in your query. It will be faster than randomizing in JavaScript.

    SELECT * FROM table ORDER BY RANDOM() LIMIT 1;
