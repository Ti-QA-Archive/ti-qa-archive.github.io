  • Here is code to make xhr request 

    
var win = Ti.UI.createWindow({
  fullscreen: true
});

var activityIndicator = Ti.UI.createActivityIndicator({
  font: {fontFamily:'Helvetica Neue', fontSize:26, fontWeight:'bold'},
  message: 'Loading Data...',
  style:Ti.UI.iPhone.ActivityIndicatorStyle.DARK,
});

win.add(activityIndicator);
activityIndicator.show();

var url = "yourUrl";
var client = Ti.Network.createHTTPClient({
     onload : function(e) {
         // do what you got to do
         activityIndicator.hide();
     },
     // function called when an error occurs, including a timeout
     onerror : function(e) {
         // treat errors
         activityIndicator.hide();
     },
     timeout : 5000  /* in milliseconds */
 });
 // Prepare the connection.
 client.open("GET", url);
 // Send the request.
 client.send();

win.open();

  • Hi,

    forget ugly code of Moiz, it is an anti pattern. A good idea is to make the xhr request in a commonJS modul like this (it is without common error handling):

    exports.get = function(_url,_callback) {
    var client = Ti.Network.createHTTPClient({
         onload : function(e) {
             _callback(JSON.parse(this.responseText));
         },
         timeout : 5000  /* in milliseconds */
     });
    client.open("GET", _url);
     client.send();
}

    In view:

    exports.create = function() {
    var self = Ti.UI.createWindow();
    self.list = Ti.UI.createTableView();
    require('xhr').get(url,function(_data){
        for (var i=0;i<_data.length;i++) {
            self.list.appendRow(require('listitem').create(_data[i]));
        }
    });
    return self;
}

    and in listitenm.js you create table view and fill it with data.

    Hope this helps-

  • Best Practice:

    • first window has a table view with list of items,
    • second window contains detail view in a scroll view,
    • on focus of first window you start a xhr request, in callback you send back the object to the list,
    • the list renders title, subtitle, image and the object is pasted (sticked) on table view row
    • on click on table view you get _event.rowData.data and create second window and fill with data.
