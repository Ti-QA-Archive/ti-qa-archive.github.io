    • Hi Rainer, In reading xml we do like

      var xhr = Titanium.Network.createHTTPClient();
    xhr.open('GET', 'http://abc.com/mobileapp/data.xml');
    xhr.onload = function() {
        var xml = this.responseXML;
        domain = xml.getElementsByTagName('domain').item(0).text;
        var logo = xml.documentElement.getElementsByTagName("appLogo").item(0).text;
        g_logoView.backgroundImage = domain + logo;
    };
    xhr.send();
