Titanium desktop crashes with httpServer

Here a nasty crash with Exception Type: EXC_BAD_ACCESS (SIGSEGV) in logs

Here an isolated piece of code to reproduce it easily

// setup
    var httpServer = Titanium.Network.createHTTPServer();

    httpServer.bind(8081,function(request, response) {
     Titanium.API.debug("Got request for URI: " + request.getURI());
     var req = new XMLHttpRequest();
    req.open("GET", "http://google.com");

    req.onreadystatechange = function()
    {
      try{
        alert("state changed");
        response.setContentType("text/html");
        response.write("<h1>Hello!</h1>");
      }catch (exception) {
           alert("exception:" + exception); 
      }
    };
    req.send();

    });
