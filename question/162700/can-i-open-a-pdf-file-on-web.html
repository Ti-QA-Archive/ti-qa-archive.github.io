  • I used a module to accomplish this because in android you can't do this with titanium's itself.

    If android, load the module…

        if(osname=="android"){
        var pdfreader = require('com.pdfreader.my');
    }

    This is a custom event that I used to start. You could use any event you like.
    In here I download pdf file to local. Because we can't open remote pdf files in android.
    By the way, I hate android :)

        Ti.App.addEventListener('pdfOpen', function(data){ 
        Titanium.API.info("--> " + data.pdfLink);

        var xhr = Titanium.Network.createHTTPClient({
            onload: function() {
                var f = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory,'pdf.pdf');

                f.write(this.responseData);
                Ti.App.fireEvent('loaded', {filepath:f.nativePath});
            },
                timeout: 10000
            });

        xhr.open('GET', data.pdfLink);
        xhr.send();    

    });

    When download finishes, we will open the module for viewing the pdf file we downloaded.
    Note in here; we should use replace function to clean 'file://' statement in our link. Without this it won't work. I could have learned it after hours :)

    And a focus event for window for detecting close of pdfviewer. In this event I'm deleting the file we downloaded.

        Ti.App.addEventListener('loaded', function(e) {
        var tmpFile = e.filepath;
        pdfreader.ShowPdfFromFile(tmpFile.replace("file://",""));

        window.addEventListener('focus', function(e){
            var file = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory,'pdf.pdf');
            if (file.exists()) { file.deleteFile(); }
        });
    });

    That's it! I hope it helps to you too… :)

  • This is my solution (Thanx to Malcom):

    exports.createPDFViewer = function(_pdfurl, _progresswidget) {
    if (Ti.Android) {
        if (Ti.Filesystem.isExternalStoragePresent()) {
            var filenameBase = new Date().getTime();
            var tmpFile = Ti.Filesystem.getFile(Ti.Filesystem.tempDirectory, filenameBase + '.pdf');
            var appFile = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory, _pdfurl);
            tmpFile.write(appFile.read());
            if (tmpFile.exists()) {
                var intent = Ti.Android.createIntent({
                    action : Ti.Android.ACTION_VIEW,
                    type : "application/pdf",
                    data : tmpFile.nativePath
                });
                try {
                    Ti.Android.currentActivity.startActivity(intent);
                } catch(e) {
                    Ti.API.debug(e);
                    alert('No apps PDF apps installed!');
                }
            } else {
                Ti.API.info('starting intent tmpFile exists: ' + tmpFile.exists());
            }
        }
    } else {
        // this module is much better then simple webview:
        var quickpdfreader = require('com.kuchbee.quickpdf');
        quickpdfreader.openPDF({
            file : _pdfurl.replace(/\.pdf/,''),
            password : '',
            isMailEnabled : false,
            loadFileFrom : 'ressources'
        });
    }
    return;
};
