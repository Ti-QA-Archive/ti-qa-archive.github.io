Show / hide a TableView.headerView

I'd like to use the TableView.headerView to show a validation message for TableViewRow items.

Is there any way to hide / show the headerView as required? I can show a headerView, but subsequently hiding it leaves a blank space behind which is undesirable. And if I don't originally show the headerView then it doesn't seem to be possible to trigger it to show later on.

I highlight the current TableViewRow item being viewed so it would be obvious to the user which item the validation message relates to.

Any ideas? Cheers, Gavin

    You can hide tableview by set tableView.headerView = null; or show it by set tableView.headerView = view;

    Check this sample code:

    var createCustomView = function(title) {
    var view = Ti.UI.createView({
        backgroundColor: '#222',
        height: 40
    });
    var text = Ti.UI.createLabel({
        text: title,
        left: 20,
        color: '#fff'
    });
    view.add(text);
    return view;
};


var inputData = [
    {title:'row 1'},
    {title:'row 2'},
    {title:'row 3'},
    {title:'row 4', hasChild: true},
    {title:'row 5', hasChild: true},
];
var tableView = Titanium.UI.createTableView({
    data:inputData,
});

var win = Ti.UI.createWindow();
win.add(tableView);
win.open();

function toggleHeaderView(show) {
    if (show) {
        tableView.headerView = createCustomView('Header View');
    } else {
        tableView.headerView = null;    
    }
}

toggleHeaderView(true);

// Hide HeaderView after 1 sec
setTimeout(function(){
    toggleHeaderView(false);
    // toggleHeaderView(true);    
}, 1000);
