    • does ist work?
      if not, here is my code to get JSON data into tableview:

      Ti.UI.backgroundColor = '#ffffff';

var url = "http://www.myurl.com";
var win = Ti.UI.createWindow();
var table = Ti.UI.createTableView();
var tableData = [];
var json, posts, post, i, row, titleLabel, excerptLabel, image, postURL;
var xhr = Ti.Network.createHTTPClient({
    onload: function() {
        // Ti.API.debug(this.responseText);
        json = JSON.parse(this.responseText);

        for(i = 0; i < json.posts.length; i++) {
            post = json.posts[i];
            row = Ti.UI.createTableViewRow({
                height:'auto'
            });
            titleLabel = Ti.UI.createLabel({
                text:post.title,
                font:{
                    fontSize:'14dp',
                    fontWeight:'bold'
                },
                height:'auto',
                left:'75dp',
                top:'5dp',
                color:'#000'
            });

            excerptLabel = Ti.UI.createLabel({
            text:post.excerpt,
            font:{
                fontSize:'10dp'
            },
            height:'auto',
            left:'75dp',
            top:'45dp',
            color:'#000',
            });

            image = Ti.UI.createImageView({
                image:post.thumbnail,
                top:5,
                left:5,
                height:65,
                width:65
            });
            row.title = post.title;
            row.id = post.id;
            row.content = post.content;

            row.add(titleLabel);
            row.add(excerptLabel);
            row.add(image);
            tableData.push(row);
        }

        table = Titanium.UI.createTableView({data:tableData});
        win.add(table);
    },
    onerror: function(e) {
    Ti.API.debug("STATUS: " + this.status);
    Ti.API.debug("TEXT:   " + this.responseText);
    Ti.API.debug("ERROR:  " + e.error);
    alert('Fehler! Keine Daten empfangen. Versuchen Sie es erneut.');
    },
    timeout:5000
});

xhr.open("GET", url);
xhr.send();

win.open();
