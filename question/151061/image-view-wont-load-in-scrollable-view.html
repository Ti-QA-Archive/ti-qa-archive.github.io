Image view won't load in scrollable view

I have a scrollable view which is supposed to contain an image view for each page. The image is from a remote url. But when i run it, only the last photo is shown.

Here is my code:

for(var i = 0, j = landingData[0].bigImages.length; i<j; i++){
    Ti.API.warn('bb '+landingData[0].bigImages[i]);
    var pic = Ti.UI.createImageView({});

    var c = Titanium.Network.createHTTPClient();
    //c.setTimeout(10000);
    c.onload = function() {
        if(c.status == 200) {
            pic.image = this.responseData;
        }
    }

    c.open('GET', landingData[0].bigImages[i]);
    c.send();
    posterViewImages.push(pic);
}

var scrollView = Titanium.UI.createScrollableView({
    views:posterViewImages,
    showPagingControl:true,
    pagingControlHeight:15,
    maxZoomScale:2.0,
    currentPage:posterView.selected
});

I'm using the above code because my previous code doesn't load the image view at all.
Here is my previous code:

for(var i = 0, j = landingData[0].bigImages.length; i<j; i++){
    Ti.API.warn('bb '+landingData[0].bigImages[i]);
    var pic = Ti.UI.createImageView({image:landingData[0].bigImages[i]});
    posterViewImages.push(pic);
}

var scrollView = Titanium.UI.createScrollableView({
    views:posterViewImages,
    showPagingControl:true,
    pagingControlHeight:15,
    maxZoomScale:2.0,
    currentPage:posterView.selected
});

Even if I declare the width and height to be Ti.UI.SIZE, it still won't load.

Anyone can help? The reason why I didn't declare the width and height in the first code is so that the image shown will be the same size as retrieved from server.

In the end, this image view will be put inside a scrollview which will later be shown in the scrollable view.

I'm using 2.1.4 and am building for iPhone.

    hi,

    try this one…

    
function loadImages(url,pic){
    var c = Titanium.Network.createHTTPClient();
    //c.setTimeout(10000);
    c.onload = function() {
        if(c.status == 200) {
            pic.image = this.responseData;        
        }
    }

    c.open('GET', url);
    c.send();
}

for(var i = 0, j = landingData[0].bigImages.length; i<j; i++){
    Ti.API.warn('bb '+landingData[0].bigImages[i]);
    var pic = Ti.UI.createImageView({});

    loadImages(landingData[0].bigImages[i],pic);
    posterViewImages.push(pic);
}

var scrollView = Titanium.UI.createScrollableView({
    views:posterViewImages,
    showPagingControl:true,
    pagingControlHeight:15,
    maxZoomScale:2.0,
    currentPage:posterView.selected
});
