    • Here is the code sample you are requesting

      var module = require('com.techwithyou.ocrEng');

var win = Ti.UI.createWindow({
    height:"100%",
    width:"100%",
    backgroundColor:"white"
});

var btnCamera = Ti.UI.createButton({
    title:"Show Camera",
    // bottom:"100dp",
    color:"black",
    borderColor:"black",
    borderWidth:"1dp",
    borderRadius:"5dp",
    backgroundColor:"white"

});


function showCamera(){
    Ti.Media.showCamera({
        success : doOCR,
        cancel : cancelCamera
    });
}

function doOCR(e) {
    module.doOCR({
       image:e.media,
       didFinish:gotResult //This callback will be triggered once the image scanning is done
    });
}

function cancelCamera(){
    alert('User Canceled');
}

function gotResult(e){
    alert(e.text);
}

win.add(btnCamera);
win.open();


btnCamera.addEventListener('click',showCamera);
