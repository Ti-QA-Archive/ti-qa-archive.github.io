The related part of my code:

 var url = "https://gdata.youtube.com/feeds/api/videos?";

    postJSON(url,params,function($apantisi){

function postJSON($url, $params, $cb, $ncb) {
    if(Titanium.Network.online) {
        var xhr = Titanium.Network.createHTTPClient();
        xhr.setTimeout(20000);
        xhr.onload = function() {
            if(this.responseText) {
                try {
                    $cb(this.responseText);
                } catch(e) {
                    if($ncb){
                        alert("Wrong data");
                        Ti.API.error($url);
                        $ncb(e);
                    }else{
                        alert(e);
                    }
                }
            } else {
                Ti.API.error("postJSON failed");
                Ti.API.info(this.responseData);
            }

        };
        xhr.onerror = function(e) {
            alert(e);
            if($ncb) {
                $ncb.call(this);
            }
        };
        xhr.open('POST', $url);
        xhr.send($params);
    } else {
        alert(L('errornonetwork', "Geen netwerkverbinding"));
        Ti.API.info($ncb);
        if($ncb) {
            Ti.API.info('mpike');
            $ncb.call(this);
        }
    }
}

Any ideas how to pass the parameters in order to post a like/dislike?
