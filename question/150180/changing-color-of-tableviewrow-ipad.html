    In that we just changing row color. all other elements are remain same in the array data.I have tried this code and working nicely..

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

// Create a TableView.
var aTableView = Ti.UI.createTableView({
    editable : true
});

// Populate the TableView data.
var data = [{
    title : 'Row 1',
    hasChild : true,
    header : 'First'
}, {
    title : 'Row 2',
    hasDetail : true
}, {
    title : 'Row 3',
    hasCheck : true,
    header : 'Second'
}, {
    title : 'Row 4'
}, {
    title : 'Row 1',
    hasChild : true,
    header : 'First'
}, {
    title : 'Row 2',
    hasDetail : true
}, {
    title : 'Row 3',
    hasCheck : true,
    header : 'Second'
}, {
    title : 'Row 4'
}];

function ChangeRowColor() {
    for (var i = 0; i < data.length; i++) {
        if (i % 2) {
            data[i].color = 'red';
        } else {
            data[i].color = 'green';
        }

    };
    aTableView.setData(data);
}

ChangeRowColor();

aTableView.addEventListener('delete',function(e){
    data.splice(e.index, 1);
    ChangeRowColor();
});

// Add to the parent view.
win.add(aTableView);

win.open();
