  • createFavouriteBar function(userParms) {

        var MAX_LABELS = 5;

        var labels = [];
        var selectedColor = '#FFB900';
        var unselectedColor = '#FFF';

        var parms = {
            width : Ti.UI.SIZE,
            height : 40,
            backgroundColor : 'transparent',
            layout : 'horizontal'
        };


        var container = Ti.UI.createView(parms);

        container.value = 1;

        function labelClickHandler(e) {

            for (var x = 0; x < labels.length; x++) {

                if (labels[x].index <= e.source.index) {
                    if (labels[x].color != selectedColor) {
                        labels[x].color = selectedColor;
                    }

                } else {
                    if (labels[x].color != unselectedColor) {
                        labels[x].color = unselectedColor;
                    }

                }

                container.value = e.source.index + 1;

            }

        }

        for (var x = 0; x < MAX_LABELS; x++) {

            var label = Ti.UI.createLabel({
                font : {
                fontSize : 26,
                fontFamily : 'FontAwesome',
                fontWeight : 'bold'
                },
                text : '\uf005',
                index : x,
                color : x == 0 ? selectedColor : unselectedColor,
                left : 0,
                width : 30
            });

            labels.push(label);

            label.addEventListener('click', labelClickHandler);
            container.add(label);

        }

        return container;
    }
