    var win = Ti.UI.createWindow();

var buttonBar = Ti.UI.iOS.createTabbedBar({
    labels : ['About', 'List', 'Another List'],
    top : 0
});

win.add(buttonBar);

var view1 = Ti.UI.createView({
    top : 50,
    visible : false
});
var view2 = Ti.UI.createView({
    top : 50,
    visible : false

});
var view3 = Ti.UI.createView({
    top : 50,
    visible : false

});

view1.add(Ti.UI.createLabel({
    text : 'This is the label',
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
    color : 'white'
}));

view2.add(Ti.UI.createTableView({
    backgroundColor : 'green'
}));
view3.add(Ti.UI.createTableView({
    backgroundColor : 'red'
}));

win.add(view1);
win.add(view2);
win.add(view3);

buttonBar.addEventListener('click', function(e) {

    switch(e.index) {

        case 0:
            view2.visible = false;
            view3.visible = false;
            view1.visible = true;
            break;
        case 1:
            view1.visible = false;
            view3.visible = false;
            view2.visible = true;
            break;
        case 2:
            view1.visible = false;
            view2.visible = false;
            view3.visible = true;

            break;

    }

});

win.open();
