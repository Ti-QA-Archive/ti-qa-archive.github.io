TableView animation

If i open a modal window with the following code, how can i get a nice animation to display so that when i click on a parent row, it slides nicely to display my child row subdata?

var win = Ti.UI.currentWindow;

var cancel = Titanium.UI.createButton({systemButton:Ti.UI.iPhone.SystemButton.CANCEL});

var data = [{title:'row 1',hasChild:true,className:'parent'},
{title:'row 2',hasChild:true,className:'parent'},
{title:'row 3',hasChild:true,className:'parent'}];

var subdata = [{title:'A'}, {title:'B'}, {title:'C'}];

var tableView = Ti.UI.createTableView({data:data,style: Titanium.UI.iPhone.TableViewStyle.GROUPED});

win.add(tableView);

win.leftNavButton = cancel;

cancel.addEventListener('click', function(){win.close();});

tableView.addEventListener('click', function(e) {
if (e.rowData.className == 'parent') {
tableView.setData(subdata, {animate:true});
} else {
tableView.setData(data, {animate:true});
}
});

  • That's great, cheers. For anyone else interested, here's my new code

    
var win = Ti.UI.currentWindow;

var win2 = Ti.UI.createWindow({navBarHidden:true});
var win3 = Ti.UI.createWindow({navBarHidden:false});

var nav = Ti.UI.iPhone.createNavigationGroup({
    window:win2
});
win.add(nav);

var cancel = Titanium.UI.createButton({systemButton:Ti.UI.iPhone.SystemButton.CANCEL});

var data = [{title:'row 1',hasChild:true}, 
            {title:'row 2',hasChild:true}, 
            {title:'row 3',hasChild:true}];

var subdata = [{title:'A'}, {title:'B'}, {title:'C'}];

var tableView = Ti.UI.createTableView({data:data,style: Titanium.UI.iPhone.TableViewStyle.GROUPED});
var tableView2 = Ti.UI.createTableView({data:subdata,style: Titanium.UI.iPhone.TableViewStyle.GROUPED});

win2.add(tableView);
win3.add(tableView2);

win.leftNavButton = cancel;

cancel.addEventListener('click', function(){win.close();});

var parentRow;

tableView.addEventListener('click', function(e) {
    parentRow = e.rowData;
    nav.open(win3, {animated:true});
});

tableView2.addEventListener('click', function(e) {
    parentRow.title = e.rowData.title;
    nav.close(win3, {animated:true});
});
