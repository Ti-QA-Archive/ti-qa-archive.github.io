SDK 1.8.x - VideoPlayer properties locked

Hello,
As far as I know, on ipad, Since 1.8.x SDK Branch, I saw that VideoPlayer properties (top,left,width,height) are locked if you create it with "visible:false".

Here the code to test

// this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#fff');


var mainWin = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});

var    VideoPlayer=Ti.Media.createVideoPlayer({
    zIndex:2,
    top:0,
    left:0,
    url:"",
    backgroundColor:"#ffffff",
    movieControlMode:Titanium.Media.VIDEO_CONTROL_EMBEDDED,
    scalingMode:Titanium.Media.VIDEO_SCALING_ASPECT_FIT,
    width:300,
    height:300,
    visible:false
});
mainWin.add(VideoPlayer);

var btn1=Ti.UI.createButton({
    top:10,
    left:500,
    width:150,
    height:50,
    title:'video1'
});
btn1.addEventListener('click',function(){

    VideoPlayer.top=300;
    VideoPlayer.left=300;
    VideoPlayer.width=200;
    VideoPlayer.height=200;
    VideoPlayer.visible=true;
});
mainWin.add(btn1);


var btn2=Ti.UI.createButton({
    top:100,
    left:500,
    width:150,
    height:50,
    title:'video2'
});
btn2.addEventListener('click',function(){

    VideoPlayer.top=0;
    VideoPlayer.left=0;
    VideoPlayer.width=300;
    VideoPlayer.height=300;
    VideoPlayer.visible=true;
});
mainWin.add(btn2);



mainWin.open();

As you can see, btn1 and btn2 has no effect and videoplayer is stuck at top/left:0/0 even if you click on btn1 (which locate videoplayer to 300/300).

Now, try by creating VideoPlayer with "visible:true":

var    VideoPlayer=Ti.Media.createVideoPlayer({
    zIndex:2,
    top:0,
    left:0,
    url:"",
    backgroundColor:"#ffffff",
    movieControlMode:Titanium.Media.VIDEO_CONTROL_EMBEDDED,
    scalingMode:Titanium.Media.VIDEO_SCALING_ASPECT_FIT,
    width:300,
    height:300,
    visible:true
});

And now it works…:'(
