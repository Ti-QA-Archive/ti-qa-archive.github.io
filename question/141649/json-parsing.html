  • Your problem isn't with the JSON parsing, although I'd use JSON.parse instead of 'eval' You are seeing the 'exerpt' html displaying - that is what you are accessing as 'tweet'. Your author name wasn't showing cause it was white text (on a white background)

    var console = {
    log : function(str) {
        return Ti.API.info(str);
    }
};
// Create variable "win" to refer to current window
var win = Titanium.UI.createWindow({
    backgroundColor : '#000'
});
var tableView = Titanium.UI.createTableView();
// Function loadTweets()
function loadWordpress() {
    // Empty array "rowData" for our tableview
    var rowData = [];
    // Create our HTTP Client and name it "loader"
    var loader = Titanium.Network.createHTTPClient();
    // Sets the HTTP request method, and the URL to get data from
    loader.open("GET", "http://jarnvags.se/wordpress/?p=20&json=20");
    // Runs the function when the data is ready for us to process
    loader.onload = function() {
        var wordpress = JSON.parse(this.responseText);

        //for (var i = 0; i < wordpress.po st.length; i++) {

        Ti.API.info(wordpress.post.title);
        var tweet = wordpress.post.excerpt;
        // The tweet message
        var user = wordpress.post.author.name;
        // The screen name of the user
        var avatar = wordpress.post.user_avatar;
        // The profile image
        // Create a row and set its height to auto
        var row = Titanium.UI.createTableViewRow({
            height : Ti.UI.SIZE
        });

        // Create the view that will contain the text and avatar
        var post_view = Titanium.UI.createView({
            height : Ti.UI.SIZE,
            layout : 'vertical',
            top : 5,
            right : 5,
            bottom : 5,
            left : 5,
            backgroundColor : '#333'
        });
        // Create image view to hold profile pic
        var av_image = Titanium.UI.createImageView({
            url : avatar, // the image for the image view
            top : 0,
            left : 0,
            height : 48,
            width : 48
        });
        post_view.add(av_image);
        // Create the label to hold the screen name
        var user_lbl = Titanium.UI.createLabel({
            text : user,
            left : 54,
            width : 120,
            top : 10,
            bottom : 2,
            height : 16,
            textAlign : 'left',
            color : '#fff',
            font : {
                fontFamily : 'Trebuchet MS',
                fontSize : 14,
                fontWeight : 'bold'
            }
        });
        post_view.add(user_lbl);
        // Create the label to hold the tweet message
        var tweet_lbl = Titanium.UI.createLabel({
            text : tweet,
            left : 54,
            top : 0,
            bottom : 2,
            height : 'auto',
            width : 'auto',
            textAlign : 'left',
            color : '#bbb',
            font : {
                fontSize : 14
            }
        });
        post_view.add(tweet_lbl);
        // Add the post view to the row
        row.add(post_view);
        // Give each row a class name
        //row.className = "item" + i;
        // Add row to the rowData array
        rowData.push(row);
        //  }
        // Create the table view and set its data source to "rowData" array

        //Add the table view to the window
        tableView.setData(rowData);

    };
    // Send the HTTP request
    loader.send();
}

win.add(tableView);
win.addEventListener('open', function() {
    // better perceived performance to load the table
    // after the window opens
    loadWordpress();
});
win.open();
