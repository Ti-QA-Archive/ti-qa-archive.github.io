Reading JSON

Hi,

i'm trying to get a json online from this url: http://alupco.com/mobileCP/categories/json wish have the following content:

{
    "categories": [
        {
            "title": "ALUK",
            "id": "3",
            "icon": "sites/default/files/category_pictures/ALUK 1.png"
        },
        {
            "title": "Fabricator",
            "id": "1",
            "icon": "sites/default/files/category_pictures/1348562467_applications-engineering.png"
        },
        {
            "title": "Distributor",
            "id": "2",
            "icon": "sites/default/files/category_pictures/1348562631_deliverables.png"
        },
        {
            "title": "Alubond",
            "id": "4",
            "icon": "sites/default/files/category_pictures/alubond logo.png"
        }
    ]
}

in my app i'm writing the following code:

var categoryUrl = "http://alupco.com/mobileCP/categories/json";

    var json;
    var xhr = Ti.Network.createHTTPClient({});
    xhr.onload = function() {
        json = JSON.parse(this.responseText);
        Ti.API.info(this.responseText);
        for (var i = 0; i < json.categories.length; i++) {
            Ti.API.info(json.categories[i].icon);
        }
    };
    xhr.open("GET", categoryUrl);
    xhr.send();

but i am getting a different value from the one online, the icon value is printed without the file name, this is what i am getting for each entry: sites&#x2F;default&#x2F;files&#x2F;category_pictures&#x2F;

plz advice

thanks
