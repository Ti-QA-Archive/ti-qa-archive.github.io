  • var tabGroup = Ti.UI.createTabGroup({
    height : 40,
    width : 320,
    bottom : 0
});

var win = Ti.UI.createWindow({
    modal : true,
    title : 'First Window',
    backgroundColor : 'white',
});

var simpleButton = Ti.UI.createButton({
    title : 'Go To Next Window',
});

simpleButton.addEventListener('click', function(e) {
    var anotherWindow = Ti.UI.createWindow({
        modal : true,
        title : 'Another Window',
        url : 'another.js',
        backgroundColor : 'white'
    });
    tab1.open(anotherWindow)
});

win.add(simpleButton);

var tab1 = Ti.UI.createTab({
    window : win,
    title : 'Home'
});

tabGroup.addTab(tab1);
tabGroup.activeTab = tab1

tabGroup.open();

    its a simple example to show you how event listeners works the difference i have taken a button instead of image you can do same stuff with image also and remember to create a file named another.js in your resources
    and please mark as permanent if your querry is resolved
