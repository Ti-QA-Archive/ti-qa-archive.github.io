webView.addEventListener('load', function(e){
    actualHeight = e.source.evalJS("document.height;");
    e.source.height = textContentActualHeight;

  • Hi W C,

    Thanks for your concern.

    I've tested your issue and webview content height is is working as expected for me.

    Please check my test code and screenshot below.

    var win1 = Titanium.UI.createWindow({  
    backgroundColor:'#fff'
});

var tweetButton = Ti.UI.createButton({
    title:'Tweet',
    height:'auto',
    width:'auto',
    top:40
});

win1.add(tweetButton);

tweetButton.addEventListener('click', function(data) {
var tweetWindow = Titanium.UI.createWindow();

var webview = Titanium.UI.createWebView({
        url:'http://www.google.com'
    }); 
    tweetWindow.add(webview);
    tweetWindow.open({
        modal:true
    });

    webview.addEventListener('load', function(e){
        actualHeight = e.source.evalJS("document.height;");
        e.source.height = textContentActualHeight;
    });

    var closeTweetButton = Titanium.UI.createButton({
        title:'Close'
    });
    closeTweetButton.addEventListener('click', function() {
        tweetWindow.close();
    });
    tweetWindow.setLeftNavButton(closeTweetButton);
});

win1.open();

    Hope it'll help you to solve your problem.

    Thanks
