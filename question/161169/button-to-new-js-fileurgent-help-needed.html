  • Hi eugene yeo!!!

    I have read your question and solved it. So please check the below code which may fulfill your need.

    home.js

    /**
 * @author Amimul
 */
var self=Ti.UI.currentWindow;
self.backgroundColor='#ffffff';
self.navBarHidden=true;
self.exitOnClose=true;

//construct UI
//var FirstView = require('ui/common/FirstView');
//var firstView = new FirstView();

var button = Titanium.UI.createButton({ 
    title: 'My button', 
    top: 10, 
    width: 100, 
    height: 50 
});

var label1 = Ti.UI.createLabel({ 
    color: '#900', 
    font: { fontSize:48 }, 
    shadowColor: '#aaa', 
    shadowOffset: {x:5, y:5}, 
    text: 'A simple label', 
    textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER, 
    top: 80, 
    width: 'auto', 
    height: 'auto' 
});

var label2 = Ti.UI.createLabel({ 
    color:'blue', 
    text: 'A long label with\na few line breaks\nand unicode (UTF8)\nsymbols such as\na white chess piece \u2655\nand the euro symbol \u20ac\nlooks like this!\n', 
    textAlign: Ti.UI.TEXT_ALIGNMENT_LEFT, 
    top: 110, 
    width: 300, 
    height: 200 
});

self.add(button); 
self.add(label1); 
self.add(label2);
