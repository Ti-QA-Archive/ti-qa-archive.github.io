Movie handling on Tabchange

I've broken my head 2 days trying to handle stoping and starting a movie on tabchange.

here is my code for others having the same problems.

var win2 = Ti.UI.currentWindow;

win2.orientationModes = [
    Titanium.UI.PORTRAIT,
    Titanium.UI.UPSIDE_PORTRAIT,
    Titanium.UI.LANDSCAPE_LEFT,
    Titanium.UI.LANDSCAPE_RIGHT
];

win2.addEventListener('focus',function(){

    var contentURL = 'http://some.http.streaming.com/playlist.m3u8';
    var activeMovie = Titanium.Media.createVideoPlayer({
        url: contentURL,
        backgroundColor: '#111',
        movieControlMode: Titanium.Media.VIDEO_CONTROL_DEFAULT,
        scalingMode: Titanium.Media.VIDEO_SCALING_ASPECT_FIT
    });

    win2.add(activeMovie);
    activeMovie.play();
    iads = Ti.UI.iOS.createAdView({
        width: 'auto',
        height: 'auto',
        bottom: -100,
        borderColor: '#000000',
        backgroundColor: '#000000'
    });

    t1 = Titanium.UI.createAnimation({
        bottom: 0,
        duration: 750
    });
    iads.addEventListener('load',function() {
        iads.animate(t1);
    });
    win2.add(iads);

    win2.addEventListener('blur', function(){
        activeMovie.stop();
    })
})
