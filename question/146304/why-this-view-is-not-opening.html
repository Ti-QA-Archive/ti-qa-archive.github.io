    Hi

    If you were trying to add a view and open it - change your code to;

    circleB.addEventListener('click', function() {
    var secnView = Ti.UI.createView({
        backgroundColor : 'red'
        //visible : true
    });
    win.add(secnView);
});

    If you were trying to open a new window - change your code to;

    circleB.addEventListener('click', function() {
    var newWin = Ti.UI.createWindow({
        backgroundColor : 'red'
        //visible : true
    });
    // add any other views to newwin here
    newWin.open();
});

    If you were trying to open a new window and have it slide in (inside tabs) - change your code to;

    circleB.addEventListener('click', function() {
    var newWin = Ti.UI.createWindow({
        backgroundColor : 'red'
        //visible : true
    });
    // add any other views to newwin here
    tab.open(newWin, { animated: true });
    //assumes "tab" is the variable for your current tab
});

    All code assumes win is the variable assigned to the window, if yours is a diffirent variable name simply rename the win in the example to whatever you are using.
