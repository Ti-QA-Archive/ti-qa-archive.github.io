    • Here is something to start you off with, this is some very simple code that build an app for either Android or iOS.

      To test this, create a new default Project and replace the contents of app.js with the following code;

      Ti.UI.setBackgroundColor('#000');

var win1 = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Tab 1'
});
var tab1 = Ti.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});
var label1 = Ti.UI.createLabel({
    color: '#999',
    font: {
        fontSize: 20
    },
    text: 'Window 1',
    textAlign: 'center',
    top: 20,
    width: Ti.UI.FILL
});
win1.add(label1);
var btn1 = Ti.UI.createButton({
    title: 'Press Me',
    top: 100
});
win1.add(btn1);
btn1.addEventListener('click', function (e) {
    alert('Hello');
});

var win2 = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Tab 2'
});
var tab2 = Ti.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Tab 2',
    window:win2
});
var label2 = Ti.UI.createLabel({
    color: '#999',
    font: {
        fontSize: 20
    },
    text: 'Window 2',
    textAlign: 'center',
    top: 20,
    width: Ti.UI.FILL
});
win2.add(label2);
var btn2 = Ti.UI.createButton({
    title: 'Press Me',
    top: 100
});
win2.add(btn2);
btn2.addEventListener('click', function (e) {
    label2.setFont({
        fontSize: 50
    });
    label2.setColor('#c60000');
});

var win3 = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Tab 3'
});
var tab3 = Ti.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab 3',
    window:win3
});

var view1 = Ti.UI.createView({ backgroundColor:'red' });
var view2 = Ti.UI.createView({ backgroundColor:'green' });
var view3 = Ti.UI.createView({ backgroundColor:'blue' });

var scrollableView = Ti.UI.createScrollableView({
  views:[view1,view2,view3],
  showPagingControl:true
});

win3.add(scrollableView);

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);  
tabGroup.addTab(tab3);  
tabGroup.open();

      This creates three tabs;

      • Tab1; Label and Button, button has an event listener that lets you run code when the button is pressed
      • Tab 2; Label and Button; press the button to change the way the label looks
      • Tab 3; Scrollable View, there are three views and you can scroll them from side to side.
