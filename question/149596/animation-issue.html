Animation issue?

So I am trying to make an animation where a cloud floats across the screen from left to right, then resets at the left and starts animating again. However, it doesn't work.

var cloud = Ti.UI.createView({   //cloud view which contains cloud image
    top : 100,
    left : 100,
    height : 200, // Do not use 'auto'
    width : 300, // Do not use 'auto'
    backgroundImage : 'cloud.png'
});

var leftToRightAnimation = Ti.UI.createAnimation({  // animates across the screen
    left : width,    //width = screen width
    duration : 10000,
    curve : Ti.UI.ANIMATION_CURVE_LINEAR
});

var resetLeft = function() {          //resets the cloud's position to 0
    cloud.left = 0;
};

leftToRightAnimation.addEventListener('complete', function() {   //once the cloud 
    resetLeft();       // has animated from left to right, then it resets on the 
    cloud.animate(leftToRightAnimation);  //left and starts to animate again
});

cloud.animate(leftToRightAnimation);    //the actual call of cloud animation

What happens is that the cloud animates from left to right, then pops up at the left (because of the resetLeft function) and then just vanishes…

Can someone help?

    hi,

    i tried with lot of possibilities but at the end i got the solution so try this one

    var cloud = Ti.UI.createView({//cloud view which contains cloud image
    top : 100,
    left : 0,
    height : 200, // Do not use 'auto'
    width : 300, // Do not use 'auto'
    backgroundColor : 'red'
});

var leftToRightAnimation = Ti.UI.createAnimation({// animates across the screen
    left : 320, //width = screen width
    duration : 10000,
    repeat : 1000
});

cloud.animate(leftToRightAnimation);
// animateView();
//the actual call of cloud animation

self.add(cloud);
