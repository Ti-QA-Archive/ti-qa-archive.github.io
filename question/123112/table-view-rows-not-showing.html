Table View Rows Not Showing

I am creating a table view in which it's first cell will be different the the rest. I have the code below but the rows are not displaying. Can someone help me please?


var reviews = data.reviews;

var cellView = Ti.UI.createView({height: 'auto'});
var row = Ti.UI.createTableViewRow({ height:'auto'});

    for (var i = 0; i < reviews.length; i++) {

        var review = reviews[i];

        if (i == 0) {

            var reviewTitle = Titanium.UI.createLabel({
                text: 'Title',
                textAlign: 'right',
                font:{fontFamily:'Arial', fontSize:20, fontWeight:'normal'},
                color: '#32580c',
                width: 123,
                height: 20,
                top: 12,
                right: 15
            });
            cellView.add(reviewTitle);
        }

        else {

            // User Image
                var userImageView = Titanium.UI.createImageView({
                    image: review.user_img,
                    width: 54,
                    height: 54,
                    top: 0,
                    right: 20
                });
                cellView.add(userImageView);    
        }

        row.add(cellView);
        items.push(row);
    }

items is defined before that as var items = [];, and the tableview is defined and displayed after the code block above as:

var tableview = Titanium.UI.createTableView({
    data: items,
    minRowHeight: 49
});
win.add(tableview);
