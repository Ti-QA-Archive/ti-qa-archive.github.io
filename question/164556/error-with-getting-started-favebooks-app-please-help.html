  • The "addBook" event handler is missing. Could you try adding these to the index.js. 

    function addBook() {
    var addbookView = Alloy.createController("addbook", {}).getView();
    if(OS_IOS) {
        $.navGroupWin.openWindow(addbookView);
    } else if(OS_ANDROID) {
        addbookView.open();
    }
}
