});

var animateToRight = Ti.UI.createAnimation({
    left : '50%',
    curve : Ti.UI.ANIMATION_CURVE_EASE_OUT,
    duration : 500,
});

var position = "left";
var showLeftPanel = function(e) {
    if (position == "left") {

/*
    for right button

var positionR = "right";

});

var RanimateToRight = Ti.UI.createAnimation({

        positionR = "right";

        $.mainView.animate(RanimateToLeft);
        positionR = "left";
    }
};
