    Hi

    If all you are trying to achieve is a simple toggle state of one or a small number of items then properties are the way to go.

    // get setting and assume display if not set already
var boolButtonState = Ti.App.Properties.getBool('buttonState', true);
btn.addEventListener('click', function (e) {
    if (boolButtonState === true) {
        // now hide button
        // code
    } else {
        // now show button
        // code
    }
});
Ti.App.Properties.setBool('buttonState', !boolButtonState); // toggle

    • Here you go, tested on iPhone and Android - all working;

      var win1 = Ti.UI.createWindow({
    fullscreen: false,
    backgroundColor: Ti.App.Properties.getString('currentColor', 'black')
});
var button1 = Ti.UI.createButton({
    title: 'Press Me'
});
button1.addEventListener('click', function (e) {
    var strNewColor = '';
    if (win1.getBackgroundColor() === 'black') {
        strNewColor = 'red';
    } else {
        strNewColor = 'black';
    }
    Ti.App.Properties.setString('currentColor', strNewColor);
    win1.setBackgroundColor(strNewColor);
});

win1.add(button1);
win1.open();

      Instead of using a boolean (a true/false toggle) I have used a string to store the colour name instead - makes sense for your specific need here.
