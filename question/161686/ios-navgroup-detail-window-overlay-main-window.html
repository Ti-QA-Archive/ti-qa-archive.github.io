iOS navGroup detail window overlay main window

Hi! When I open detailView window, its overlay on mainView. I can't explain how it work with my english :( I made video, please help fix it.

In both window used tableview with labels.

Some code:

function ApplicationWindow() {

    var MasterView = require('ui/common/MasterView'),
    DetailView = require('ui/common/DetailView');

    var self = Ti.UI.createWindow();

    var masterView = new MasterView(),
        detailView = new DetailView();

    var masterContainerWindow = Ti.UI.createWindow({
            title:'????-????'
        });
    masterContainerWindow.add(masterView);    

    var detailContainerWindow = Ti.UI.createWindow({
        title:'???????'
    });
    detailContainerWindow.add(detailView);

    var navGroup = Ti.UI.iPhone.createNavigationGroup({
        window:masterContainerWindow
    });
    self.add(navGroup);

    masterView.addEventListener('itemSelected', function(e) {
        detailView.fireEvent('itemSelected',e);
        navGroup.open(detailContainerWindow);
    });
}

  • NavigationGroup has been removed from 3.2.0.GA and on.

    In February, Apple will require that submissions be built against iOS sdk 7.x

    3.2.0.GA is the latest release, and as the most stable iOS 7 support.

    Recommend you make the move to 3.2.0.GA.
