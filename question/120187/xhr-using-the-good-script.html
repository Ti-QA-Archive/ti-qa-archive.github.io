  • Here's a basic way to make an ajax call that returns json:

    function ajax (url, callback) {

        url = url;
        Ti.API.info("ajax call: " + url);
        var loader = Ti.Network.createHTTPClient();
        loader.setTimeout(0);
        loader.onload = callback;
        loader.onerror = someFuncToHandleErrors;
        loader.open("GET", url);
        loader.send();

    }

    You would call it like so:

    ajax("myurl", function() { //this.responseText contains the json result });
