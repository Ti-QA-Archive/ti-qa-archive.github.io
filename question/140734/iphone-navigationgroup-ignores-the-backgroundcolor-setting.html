  • It sounds like you are actually trying to change the nav bar color, and not the background color of the navgroup. The nav bar color is set at the window level using the barColor property:

    var win = Titanium.UI.createWindow();

var win2 = Titanium.UI.createWindow({
    barColor : '#ff0000',
    title : 'Red Window'
});

var button = Ti.UI.createButton({
    title : 'Open Blue',
    height : 70,
    width : 150
});
win2.add(button);

var nav = Titanium.UI.iPhone.createNavigationGroup({
    window : win2,
    backgroundColor : '#ffffff',
});

win.add(nav);
win.open();

var win3 = Titanium.UI.createWindow({
    barColor : 'blue',
    title : 'Blue Window'
});

button.addEventListener('click', function() {
    nav.open(win3, {
        animated : true
    });
});
