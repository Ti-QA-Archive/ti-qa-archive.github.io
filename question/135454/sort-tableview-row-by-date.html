var feedArray = [

    {title: 'Feed 3', desc: 'Desc 3', date: '2012-03-18'}

function findFeedByDate(d) {
    for(var i = 0; i < feedArray.length; i++) {
        if ( d == feedArray[i].date ) {
            return feedArray[i];
        }
    }
}

  • Simply use the sort method of arrays.

    var points = [40,100,1,5,25,10];
points.sort();

    The sort() method sorts numbers and words alphabetically. If you want to switch between ASC & DESC you can use the points.reverse() method.

    Now, if you want to define a custom sorting function you can use something like this:

    var points = [40,100,1,5,25,10];
 points.sort(sortfunc);

function sortfunc(a,b) {
 return b - a;
}

    Finally, you want to sort dates …! Hence, convert each date into the corresponding TimeStamp and then sort. You can find a ton of read functions for converting Date2Timestamp and vice versa in Google.

    Hope this helps
