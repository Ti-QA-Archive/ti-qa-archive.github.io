Animation flicker with createAnimation!!!

With a simple animation on a real ipod sometimes the animation flicker. Ideas?

var win = Ti.UI.createWindow({
    backgroundImage: 'img/fondo.png',
})


//
// CREATE IMAGE VIEW
//

var pez = Titanium.UI.createImageView({
    image:'img/pez.png',
    width: 'auto',
    height: 'auto',
    top: 50,
    left: 75,
});

win.add(pez);

var playerPez = Titanium.UI.createAnimation({
   top: 25,
   duration:2400,
   autoreverse: true
});                            

playerPez.addEventListener('complete',function() {

  var numero = Math.round(Math.random()*10);
   if (numero < 5) {
           imageView.start();
           imageView.show();       
   }

   pez.animate(playerPez);


});

pez.animate(playerPez);

win.open();

  • animation by default have an easing effect that will be obvious when chaining them. add the following to your animation to remove the easing and voila. Note that this is only true if the flicker is happening at the end when one animation ends and another starts.

    var playerPez = Titanium.UI.createAnimation({
   top: 25,
   duration:2400,
   autoreverse: true,
   curve: Ti.UI.ANIMATION_CURVE_LINEAR
});

  • Instead of using an ImageView, try just using a view and set the image using the backgroundImage property. Also try explicitly setting the height and width.

    var pez = Titanium.UI.createView({
    backgroundImage:'img/pez.png',
    width: 100,
    height: 100,
    top: 50,
    left: 75,
});
