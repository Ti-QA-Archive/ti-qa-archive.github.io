    Hi Nuibb,

    use this code and it should work.

    listView.addEventListener('itemclick', function(e) {
    Ti.API.info(e.itemId);

    //e.row.children[1].text = '';// This not working. How can i change this ?

    // get the item
    var item = e.section.getItemAt(e.itemIndex);
    // set new Text
    item.value.text = 'new Text';
    // update item
    e.section.updateItemAt(e.itemIndex, item);  

});

    You find docs here ListView itemclick Event

    • For row counting on listView section items.

      listView.addEventListener('itemclick', function(e) {
    Ti.API.info(e.itemId);

    //e.row.children[1].text = '';// This not working. How can i change this ?

    // row count on section items
    var rowCount = e.section.getItems().length;
    Ti.API.info('rows counted:' +rowCount);

    // get the item
    var item = e.section.getItemAt(e.itemIndex);
    // set new Text
    item.value.text = 'new Text';
    // update item
    e.section.updateItemAt(e.itemIndex, item);  
});
