Identify particular row of table view which is use in multiple views.

Hi,I am developing android application. In my application I am displaying one window. Window contains two view with similar layout but different data.I am using single layout file and show hide both views.It looks like

//   a.js
explorewin = Titanium.UI.createWindow(
{
    backgroundColor:'#f8f8f8'
});

Ti.include('b.js');// b.js contains UI for views

new_view = Ti.UI.createView(
{
    backgroundColor:'#f8f8f8',
});explorewin.add(explore_variables.new_view);

popular_view = Ti.UI.createView(
{
    backgroundColor:'#f8f8f8',
});explorewin.add(explore_variables.popular_view);

explorewin.open();

// for new_view hide popular_view
view_dis(new_view)//view_dis is function inside b.js

view_dis(popular_view)///view_dis is function inside b.js

//b.js
view_dis= function(display_view)
{
    feed_item_table = Ti.UI.createTableView(
    {

    });

    //display table view with some data...........
