  • Hi Anthony ,

    Try this once.

    Titanium.Gesture.addEventListener('orientationchange', function(e){

    if(Ti.Gesture.portrait==true){

        label.backgroundImage = 'images/Custom_SearchBar_Portrait_Dic_Double.png';
        search.left = getwidth(100);  // its better to use % than using a fixed pixels value, specially when the user turns the device screen
        search.right = getwidth(5); 
        close_btn.top = getheight(50);
        ChangeLang_btn.top = getHeight(50);
        close_btn.left = getWidth(150);
    }

    else{
        label.backgroundImage = 'images/Custom_SearchBar_Dic_Double.png'; 
        search.left = getwidth(100); 
        search.right = getwidth(5);; 
        close_btn.top = getHeight();;
        ChangeLang_btn.top = '4.5%';
        close_btn.left = '12%';
    }

})
