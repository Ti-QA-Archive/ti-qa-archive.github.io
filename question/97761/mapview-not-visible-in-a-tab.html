MapView not visible in a Tab

A Mapview does not render map (only grid) if it's placed on a tab. It works ok in a normal window.

Is this normal or am i missing somthing?

Here is the code im using:

var tg = Ti.UI.createTabGroup();

var win1 = Ti.UI.createWindow({
backgroundColor:'white'
});

var tab1 = Ti.UI.createTab({
title:'Map', window:win1
});

var win2 = Ti.UI.createWindow({
backgroundColor:'white'
});

var tab2 = Ti.UI.createTab({
title:'Tweets', window:win2
});

var mapview = Titanium.Map.createView({
mapType: Titanium.Map.STANDARD_TYPE,
region: {latitude:33.74511, longitude:-84.38993,
latitudeDelta:0.01, longitudeDelta:0.01},
animate:true,
regionFit:true,
userLocation:true
});

win1.add(mapview);

tg.addTab(tab1);
tg.addTab(tab2);

tg.open();

Im using Titanium Developer 1.2.2
Developing for Android using sdk 2.1

Thanks

  • Try the following code, which works for me using Titanium 1.7.1 (2011/06/17 00:13 293a6d…) and Android 2.1:

    Ti.UI.setBackgroundColor('#000');

var tabGroup = Ti.UI.createTabGroup();

var win1 = Ti.UI.createWindow({
  title:'Tab 1',
  backgroundColor:'#fff'
});
var tab1 = Ti.UI.createTab({
  title:'Tab 1',
  window:win1
});

var label1 = Ti.UI.createLabel({
  color:'#999',
  text:'I am Window 1'
});

win1.add(label1);

var win2 = Ti.UI.createWindow({
  title:'Tab 2',
  backgroundColor:'#fff'
});
var tab2 = Ti.UI.createTab({
  title:'Tab 2',
  window:win2
});

var mapview = Titanium.Map.createView({
  mapType: Titanium.Map.STANDARD_TYPE,
  animate:true,
  regionFit:true,
  userLocation:false
});
win2.add(mapview);

tabGroup.addTab(tab1);
tabGroup.addTab(tab2);

tabGroup.open();

    Hope this helps
