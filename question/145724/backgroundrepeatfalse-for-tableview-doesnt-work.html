  • yes you are right it seems that there is a bug in backgroundRepeat property .its always true in tableview .backgroundRepea:false has no effect and it repeats the backgroundImage … .

    if you want workaround then use a view for backgrounImage and add tableview in it,you can set transparancy (visibility of background view) by settin opacity property of tableView

    
var win = Ti.UI.createWindow();
var view = Ti.UI.createView({
      backgroundImage: "http://www.appcelerator.com/wp-content/uploads/2009/06/titanium_desk.png",
      backgroundRepeat:false    

});

var data=[];

    data[0] = Ti.UI.createTableViewRow({height: 40,width:80,left:30,
        backgroundColor:'white',title:'Row 1'})
    data[1] = Ti.UI.createTableViewRow({height: 40,width:100,left:30,
        backgroundColor:'red',title:'Row 2'}) 
    data[2] = Ti.UI.createTableViewRow({height: 40,width:100,left:30,
        backgroundColor:'blue',title:'Row 3'})

var table= Ti.UI.createTableView({
    top:10,
    data : data,
    maxRowHeight:30,
    opacity: 0.2

})

view.add(table);
win.add(view);

win.open();
