Preset textfield value without using hintText

I want a textfield value in window to be populated from database when I open a window, so that I can edit, is it possible in Titanium to do this without using hintText?

    Do like this

        var username = Ti.UI.createTextField({
        top : '60dp',
        //hintText:'email address',
        keyboardType : Titanium.UI.KEYBOARD_DEFAULT,
        width : '60%',
        height : '45dp',
        right : '2%',
        borderRadius : 5,
        borderColor : 'black',
        borderWidth : 1
    });
self.add(username)
username.value=dbvalue;
