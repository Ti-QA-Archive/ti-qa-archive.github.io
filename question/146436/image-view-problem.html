  • Hi Gaurav Sood!!

    Your code is correct. But for testing your code, I have to add a simple table and a row . I have no idea about your presented image so I have used a demo image but It works.

    So Please check the bellow code. I hope, it will solve your problem

    var win=Ti.UI.createWindow({
    layout:'vertical',
    backgroundColor:'#fff'
});

var aTableView = Ti.UI.createTableView();

win.add(aTableView);

var remember_me = Titanium.UI.createImageView({
          image:'KS_nav_ui.png',
          width:Ti.UI.SIZE,
          height:Ti.UI.SIZE,
          left:20
        });
var remember_me_var=false;

var row2=Ti.UI.createTableViewRow();
row2.add(remember_me);
aTableView.appendRow(row2);
// row2.add(remember_me_label);


        remember_me.addEventListener('singletap', function(e) {

            if(!remember_me_var)
            {
                remember_me_var = true;
                remember_me.image = 'KS_nav_views.png';
            }
           else
           {
                remember_me_var = false;
                remember_me.image = '';
                remember_me.image = 'KS_nav_ui.png';
           }
        });

win.add(remember_me);
win.open();
