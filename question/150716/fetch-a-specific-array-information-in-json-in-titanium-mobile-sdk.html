Fetch a specific array information in .JSON in Titanium Mobile SDK?

How to fetch the information of one array only instead of all from json url in Titanium Mobile SDK?

Here is my code in app.js

var url = "http://dashberry.com/mspikes/json/mainmenu.json";
var table = Ti.UI.createTableView();
var tableData = [];
var json, mainmenu, fighter, i, row, nameLabel, susunLabel;
function menu2()
{
    var win = Ti.UI.createWindow();

    var xhr = Ti.Network.createHTTPClient(
    {
           onload: function(e) 
           {
               json = JSON.parse(this.responseText);

            for (i = 0; i < json.length; i++) 
            {
                    fighter = json[i];
                row = Ti.UI.createTableViewRow(
                {
                    height:'120dp'
                });

                kodLabel = Ti.UI.createLabel(
                {
                    text:'"' + fighter.MenuKod + '"',
                    font:{ fontSize:'16dp'},
                    height:'auto',
                    left:'30dp',
                    top: '60dp',
                    bottom:'5dp',
                    color:'#000',
                    touchEnabled:false
                });

                nameLabel = Ti.UI.createLabel(
                {
                    text:fighter.MenuNama,
                    font:{fontSize:'24dp',fontWeight:'bold'},
                    height:'auto',
                    left:'10dp',
                    top:'5dp',
                    color:'#000',
                    touchEnabled:false
                });

                //alert(nameLabel)

                susunLabel = Ti.UI.createLabel(
                {
                    text:'"' + fighter.MenuSusunan + '"',
                    font:{ fontSize:'16dp'},
                    height:'auto',
                    left:'15dp',
                    bottom:'5dp',
                    color:'#000',
                    touchEnabled:false
                });

                //alert(susunLabel)

            row.add(susunLabel);
            row.add(nameLabel);
            row.add(kodLabel);
            tableData.push(row);

            }

            table.setData(tableData);

            //Ti.API.info("Received text: " + this.responseText);
            //json = JSON.parse(this.responseText);
            //Ti.API.debug(this.responseText);
            //alert('success');

           },

        onerror: function(e) 
        {
            Ti.API.debug("STATUS: " + this.status);     
            Ti.API.debug("TEXT:   " + this.responseText);     
            Ti.API.debug("ERROR:  " + e.error); 
            //Ti.API.debug(e.error);
            alert('error');
           },

        timeout:5000 
    });

    xhr.open("GET", url);
    xhr.send();  

    win.add(table);
    return win;
}
module.exports = menu2;
