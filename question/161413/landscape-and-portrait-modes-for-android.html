Landscape and portrait modes for android

platfrom : Android
SDK : 3.1.3GA
Hi Iam facing a problem when Iam designing for both modes in Android, I used this code for orientation.

Ti.Gesture.addEventListener('orientationchange',function(e) {
    Ti.App.fireEvent('orient', {portrait:e.source.isPortrait()});
});
Ti.App.addEventListener('orient', function(evt) {
        if(evt.portrait===true) {
        view1.height='8%';
} else {
        view1.height='10%';
}
    });

var view1 = Ti.UI.createView({
    top:'0%',
    height:(Titanium.UI.isPortrait()) ?'8%':'10%',
    width:'100%',
    //backgroundColor:'#5973b0'
    backgroundImage:'/images/header.png'

});
win1.add(view1);

Iam getting error in android Object UI has no method isPortrait(). its working in ios, but not in android, please help me guys.
Thanks
