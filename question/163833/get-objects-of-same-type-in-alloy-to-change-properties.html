Get objects of same type in Alloy to change properties

Hi all.

I'm trying to figure how I can access all the objects of same type in Alloy.

For example, I have those 5 elements

<View id="menu1" class="menuButton"/>
<View id="menu2" class="menuButton"/>
<View id="menu3" class="menuButton"/>
<View id="menu4" class="menuButton"/>
<View id="menu5" class="menuButton"/>

How can I get it in Alloy controller, in a loop, to change all they values, e.g., I want to change the left property for all, but I don't want to to do manually for each one.

for (var i=1; i<elem_class_menuButton.lenght; i++) {
    i.left = "10px";
}

  • Accepted Answer

    index.xml

    <Alloy>
    <Window id="win" title="Tab 1">
        <View id="menu1" class="menuButton" autoStyle="true"/>
        <View id="menu2" class="menuButton" autoStyle="true"/>
        <View id="menu3" class="menuButton" autoStyle="true"/>
        <View id="menu4" class="menuButton" autoStyle="true"/>
        <View id="menu5" class="menuButton" autoStyle="true"/>
        <Button id="btn" onClick="onClick"></Button>
    </Window>
</Alloy>

    index.tss

    "Window": {
    backgroundColor: "#fff",
    layout:"vertical"
},

".menuButton": {
    top:"5",
    backgroundColor:'green',
    height:"40",
    left:"50",
    right:"0"    
}

"#btn":{
    width:"150",
    height:"30",
    title:"Change",
    bottom:0
}

    index.js

    function onClick(e) {

    var s = $.createStyle({
        classes : 'menuButton',
        apiName : 'View',
        left : '0'
    });

    var ch = $.win.children;

    for (var x = 0; x < ch.length; x++) {

        var c = ch[x];

        var cl = c.classes;

        if (cl) {

            for (var y = 0; y < cl.length; y++) {

                if (cl[y] == 'menuButton') {

                    c.applyProperties(s);
                }
            }
        }

    }

}

$.win.open();
