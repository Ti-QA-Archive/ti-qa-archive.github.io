  • Without seeing your code, my guess is your win is linked to the tabGroup. If you close the tabgroup it will not display the windows. Most likely that is why you are seeing just the black app background.

    Instead of closing the tabgroup couldn't you use setTabBarHidden to hide the tabgroup? This would allow for you to hide the tabgroup content when in a detail window.

    • Ok, I see the app flow alittle better now.

      You could do something like this. Your first window (win) will show the tabgroup, but your second window photosWindow wouldn't. An example of this would be the twitter app. Hope this helps, but please remember there is a known issue with this as detailed here http://jira.appcelerator.org/browse/TIMOB-6889

      var win = Ti.UI.createWindow({
    backgroundColor:"black",
});

//Uncomment this if you don't want the tabbar to show
//win.tabBarHidden=true;

var b1 = Titanium.UI.createButton({
title:'Click me',
height:40,
width:200,
top:10
});

b1.addEventListener('click', function()
var photosWindow = Titanium.UI.createWindow({
    url: 'photoThumbs.js',
    navBarHidden:true,
    tabBarHidden:true
});
Titanium.UI.currentTab.open(photosWindow,{animated:true});
});
win.add(b1);



var tabGroup = Titanium.UI.createTabGroup();

var tab1 = Titanium.UI.createTab({
    icon:'KS_nav_views.png',
    title:'open photoswindow',
    window:win
});

tabGroup.addTab(tab1);

tabGroup.open({
transition:Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
});
