    var win = Ti.UI.currentWindow;

    var data = [];//array of data
    var lastRow = 4;//data per page
    var p = 0;

    for (var c=0;c<lastRow;c++)
    {
    data[c] = {title:"Row "+(c+1)};
    }
    var btnrow = Ti.UI.createTableViewRow();
    var next = Titanium.UI.createButton({
    title:'Next',
    color:'#fdb335',
    height:30,
    width:100
    });
    btnrow.add(next);
    data.push(btnrow);

    var tableView = Ti.UI.createTableView({
    data: data
    });

    win.add(tableView);

    var navActInd = Titanium.UI.createActivityIndicator();
    win.setRightNavButton(navActInd);

    var updating = false;
    //var loadingRow = Ti.UI.createTableViewRow({title:"Loading…"});

    function beginUpdate()
    {
    updating = true;
    navActInd.show();

    //tableView.appendRow(loadingRow);
tableView.deleteRow(lastRow,{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.NONE});


// just mock out the reload
setTimeout(endUpdate,2000);
//endUpdate();

    }

    function endUpdate()
    {
    updating = false;

    var xhr = Ti.Network.createHTTPClient();
//appel synchrone au WS non obligatoire

xhr.onload = function(){

    var rep     = JSON.parse(this.responseText);
    var maxData = rep.obj.length;

    //alert(maxData);

    // simulate loading
    for (var c=lastRow;c<lastRow+4;c++)
    {
        var name = rep.obj[c].name;
        tableView.appendRow({title:name},{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.NONE});
    }
    lastRow += 4;
    tableView.appendRow(btnrow);

    // just scroll down a bit to the new rows to bring them into view
    //tableView.scrollToIndex(lastRow-9,{animated:true,position:Ti.UI.iPhone.TableViewScrollPosition.BOTTOM});

    navActInd.hide();

}

xhr.onerror = function(){
    alert('Erreur de serveur !');    
}

xhr.open("GET",'url.php');
xhr.send();

    }

    next.addEventListener('click',function(e)
    {
    beginUpdate();
    });

var data    = [];//array of data
var lastRow = 4;//data per page
var p       = 0;

for (var c=0;c<lastRow;c++)
{
    data[c] = {title:"Row "+(c+1)};
}
var btnrow = Ti.UI.createTableViewRow();
var next = Titanium.UI.createButton({
        title:'Next',
        color:'#fdb335',
        height:30,
        width:100
});
btnrow.add(next);
data.push(btnrow);

var tableView = Ti.UI.createTableView({
    data: data
});

win.add(tableView);

var navActInd = Titanium.UI.createActivityIndicator();
win.setRightNavButton(navActInd);

var updating = false;
//var loadingRow = Ti.UI.createTableViewRow({title:"Loading..."});

function beginUpdate()
{
    updating = true;
    navActInd.show();

    //tableView.appendRow(loadingRow);
    tableView.deleteRow(lastRow,{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.NONE});


    // just mock out the reload
    setTimeout(endUpdate,2000);
    //endUpdate();
}

function endUpdate()
{
    updating = false;


    var xhr = Ti.Network.createHTTPClient();
    //appel synchrone au WS non obligatoire

    xhr.onload = function(){

        var rep     = JSON.parse(this.responseText);
        var maxData = rep.obj.length;

        //alert(maxData);

        // simulate loading
        for (var c=lastRow;c<lastRow+4;c++)
        {
            var name = rep.obj[c].name;
            tableView.appendRow({title:name},{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.NONE});
        }
        lastRow += 4;
        tableView.appendRow(btnrow);

        // just scroll down a bit to the new rows to bring them into view
        //tableView.scrollToIndex(lastRow-9,{animated:true,position:Ti.UI.iPhone.TableViewScrollPosition.BOTTOM});

        navActInd.hide();

    }

    xhr.onerror = function(){
        alert('Erreur de serveur !');    
    }

    xhr.open("GET",'url.php');
    xhr.send();

}

next.addEventListener('click',function(e)
{
    beginUpdate();
});
