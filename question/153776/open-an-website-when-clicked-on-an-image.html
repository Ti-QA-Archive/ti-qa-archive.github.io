  • Hi prashanth,

    Malcolm code is good but if window is open directly than there is problem to close so i just add close button if it is for iOS otherwise you need to add view on window and add that button.

    logo.addEventListener('click', function(e) {
    var webView = Ti.UI.createWebView({
        top : 0,
        url : 'http://www.google.com'
    });
    var btnClose = Ti.UI.createButton({
        title : 'close'
    });
    btnClose.addEventListener('click', function(evt) {
        window.close();
    });
    var window = Ti.UI.createWindow({
        backgroundColor : 'white',
        rightNavButton : btnClose
    });

    window.add(webView);
    window.open({
        modal : true
    });
});
