  Ti.API.info('We found a match!');
} else {
  Ti.API.info('No regex match found!');
}

    • Hey Andres, you could probably do something like …

          function validateNum(num) {
        return num.replace(/[^0-9\.]/g,"");
    }
    depDayField.addEventListener('change', function(e) {
        depDayField.value = validateNum(depDayField.value);
    });
