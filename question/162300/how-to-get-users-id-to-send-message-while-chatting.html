Cloud.Chats.create({
    to_ids: ids.join(','),
    message: 'Good morning'
}, function (e) {
    if (e.success) {
        for (var i = 0; i < e.chats.length; i++) {
            var chat = e.chats[i];
            alert('Success:\n' +

                'Updated: ' + chat.updated_at + '\n' +
