  • Hi Ray,

    Here is a code segment where view's background image not disappears when the swipe event fired.

    Ti.UI.backgroundColor='gray';
win = Titanium.UI.createWindow({
    fullscreen: true,
    layout: 'composite',
    backgroundColor: 'white',
});

win.open();

var inner = Ti.UI.createView({
    height: Ti.UI.SIZE,
    width: Ti.UI.SIZE,
    top: 50,
    backgroundColor:'transparent',
    backgroundImage :'/KS_nav_ui.png'
});
win.add(inner);
 lbl = Ti.UI.createLabel({
    height: Ti.UI.SIZE,
    width: Ti.UI.SIZE,
    text: 'This Label is HERE',
    font: {fontSize: 25},
    color: 'black',
});
inner.add(lbl);


 inner.addEventListener('swipe',function(){
     alert('swipe');
 });
