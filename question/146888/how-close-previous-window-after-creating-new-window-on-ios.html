  • Hello Ashish,

    Try like this…

    var win1 = Titanium.UI.createWindow({
    title : 'Window 1',
    backgroundColor : 'red'
});
win1.open();

var win2 = Titanium.UI.createWindow({
    title : 'Window 2',
    backgroundColor : 'blue'
});

var btn = Ti.UI.createButton({
    title : 'Click',
    top : 100,
    left : 100,
    right : 100,
    height : 40
});

btn.addEventListener('click', function(e) {

    win1.close();

    var win2 = Titanium.UI.createWindow({
        title : 'Window 2',
        backgroundColor : 'blue'
    });
    setTimeout(function(er) {
        win2.open({
            modal : true
        });
    }, 100);
});
win1.add(btn);
