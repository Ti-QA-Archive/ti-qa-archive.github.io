Question about the countdown timer

Hello everyone,

I am trying to test the following code of creating a countdown timer (referring to this web site: http://cssgallery.info/create-a-countdown-timer-with-titanium-appcelerator/)

I just would like to ask a question about the code:


var countDown =  function( m , s, fn_tick, fn_end  ) {
    return {
        total_sec:m*60+s,
        timer:this.timer,
        set: function(m,s) {
            this.total_sec = parseInt(m)*60+parseInt(s);
            this.time = {m:m,s:s};
            return this;
        },
        start: function() {
            var self = this;
            this.timer = setInterval( function() {
                if (self.total_sec) {
                    self.total_sec--;
                    self.time = { m : parseInt(self.total_sec/60), s: (self.total_sec%60) };
                    fn_tick();
                }
                else {
                    self.stop();
                    fn_end();
                }
                }, 1000 );
            return this;
        },
        stop: function() {
            clearInterval(this.timer);
            this.time = {m:0,s:0};
            this.total_sec = 0;
            return this;
        }
    };
};    

var my_timer = new countDown(5,30, 
        function() {
            display_lbl.text = my_timer.time.m+" : "+my_timer.time.s;
        },
        function() {
            alert("The time is up!");
        }
    );

how could we access the 's' and the 'm' attributes using (my_timer.time.m) and (my_timer.time.s)?

and what is the difference between time and timer?

Thanks in advance

  • If anyone would like to add a pause button to their timer here is the code to add after the stop button function: 

    pause: function() {
            clearInterval(this.timer)
            this.time = {m:this.total_min,s:this.total_sec};
            return this;
        }

    Then you simply have to call my_timerX.pause(); and when you are ready to restart my_timerX.start();

    Works great for me.
