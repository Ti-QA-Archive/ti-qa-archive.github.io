Uncaught typeerror cannot call method 'open' of null

Hi Everyone,

Every time I click on the tableview and trying to open a new tableview I'm getting this error message

uncaught typeerror cannot call method 'open' of null
Location: [58,27] detail.js
Source: Titanium.UI.currentTab.open(win,{animation:true});

Below is my syntax, I'm not sure what I'm doing wrong here. What do you think I'm doing wrong? Any help would be appreciated. Thanks everyone 

var win = Ti.UI.currentWindow;

var tableview = Ti.UI.createTableView({editable:true});

var id = win.id;

var db = Titanium.Database.open("test5");
var sql = db.execute("SELECT DISTINCT * FROM test WHERE author=?", id);

var data = [];

Ti.API.info("Total Row count is : " + sql.getRowCount());

while(sql.isValidRow()){

    var quoteID= sql.fieldByName('quote');

    data.push({
        title:quoteID, 
        hasChild:true, 
        id:quoteID,
        url:'list.js'    
    });

    sql.next();
}

tableview.setData(data);
win.add(tableview);


// Add event listeners to make the table work

tableview.addEventListener('click',function(e){

    if(e.rowData.url)
    {
        var win = Ti.UI.createWindow({

             id:e.rowData.id,
             title:e.rowData.title,
             backgroundColor:'#eee',
             url:e.rowData.url,
             bacColor:'#1e1e1e'

        });

      Titanium.UI.currentTab.open(win, {animation:true});

     }

});

animation:true
