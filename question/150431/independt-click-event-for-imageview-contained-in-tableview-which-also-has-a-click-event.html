Independt Click event for Imageview contained in Tableview which also has a click event

I have a tableview for which I have the 'click' event defined. In this I have rows and in one of the rows there is an image. I want to have a 'click' event for this imageview. However, when the click event of the imageview is fired, it also fires the click event of the tableview!! I moved the 'click' event from table to row view, but its the same thing! So how can I have independent click events for the tableview and the imageview contained inside the table view.

var menuTableView = Ti.UI.createTableView();

var section1 = Ti.UI.createTableViewSection({
        headerTitle :L('user')
    });

var row1 = Ti.UI.createTableViewRow({
    title:'Row 1'

);

section1.add(row1);
var row2 = Ti.UI.createTableViewRow({
    title:'Row 2'

);


var rowImage = Ti.UI.createImageView({
    image:'/images/notificiationalert.png'
});

row2.add(rowImage);

section1.add(row2);

menuTableView.setData([section1]);

menuTableView.addEventListener('click', function(e) {
        alert('Tableview click');

        }
    });
  • Application type: mobile
  • Titanium SDK: 2.1.3
  • Platform & version: iOS 5.0, Android 2.3
  • Device: iOS simulator, Android emulator as well as physical devices
  • Host Operating System: OSX

    hi,

    you can set a custom property to imageview and then check that in tableview's click event….

    for iphone, you can go with sunny's suggestion but if you want in both platform as you mentioned in you question you can use the below code.

    var row2 = Ti.UI.createTableViewRow({
    title:'Row 2'

);


var rowImage = Ti.UI.createImageView({
    image:'/images/notificiationalert.png',
    isImageClicked : true
});

menuTableView.addEventListener('click', function(e) {
    if(e.source.isImageClicked){
        alert('Image click');
    }else{
        alert('Tableview click');        
    }

});

    image:'/images/notificiationalert.png',
    cause: "button"//Custom property
});
row.add(button);

row.addEventListener('click', function(e) {
    if(e.source.cause == "image"){
        alert('Image clicked');
    }else if(e.source.cause == "button"){
        alert('button clicked..');        
    }

});
