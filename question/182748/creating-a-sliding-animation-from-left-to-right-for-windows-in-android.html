Creating a sliding animation from left to right for windows in android

Hi guys,
I have a view whenever somebody clicks it on it I want another window should come sliding from right to left. So how do I do that ? 

var view1= Ti.UI.createView({
    layout: "vertical",
    height: Ti.UI.SIZE,
    backgroundColor: "#fff",
    left: "1%",
    right: "1%",
    top: 5,
    id: Data[i].id
});

view1.addEventListener("click", function(e){
    Ti.API.info(e.source.id);
    var DetailsWindow = require('ui/handheld/DetailsWindow');
    new DetailsWindow("Listing Details",auth_token, e.source.id).open();
});
//ui/handheld/DetailsWindow.js
function DetailsWindow(title,auth_token,id){
    var detailsWindow = Ti.UI.createWindow({
        title: title
    });
        var animation = Titanium.UI.createAnimation();
    //animation.backgroundColor = 'black';
    animation.duration = 300;
    var animationHandler = function() {
      animation.removeEventListener('complete',animationHandler);
      //animation.backgroundColor = 'orange';      
      animation.left = "0";    
    };
    animation.addEventListener('complete',animationHandler);    
    detailsWindow.open(animation);
       return detailsWindow;
};
module.exports = DetailsWindow;
