  • Try this code you get very well what i say :

    var win = Titanium.UI.createWindow({
    backgroundColor:'gray',
    title:'Test Demo',
    layout : 'vertical'
});

var lbl = Ti.UI.createLabel({
    top : 10,
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,
    text : 'Font Size is 2',
    font : {
        fontSize : 2 
    }
});
var lbl1 = Ti.UI.createLabel({
    top : 3,
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,
    text : 'Font Size is 1.5',
    font : {
        fontSize : 1.5 
    }
});
var lbl2 = Ti.UI.createLabel({
    top : 3,
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,
    text : 'Font Size is 1',
    font : {
        fontSize : 1 
    }
});

var lbl3 = Ti.UI.createLabel({
    top : 3,
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,
    text : 'Font Size is 0.5',
    font : {
        fontSize : 0.5 
    }
});

win.add(lbl);
win.add(lbl1);
win.add(lbl2);
win.add(lbl3);
win.open();
