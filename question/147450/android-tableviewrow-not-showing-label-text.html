Android TableViewRow Not Showing Label Text

I am on the last issue with a project in which I developed for iPhone and Android and I'm having an issue on the Android side with labels in rows on Android.

Here's my code for dynamically showing data on a custom drilldown. Note: the category table has rows with IDs which then are the filtering component for the subcategory table data. Both sets of data are static and defined already so there's no external fetching of data. Again, this same code works for iOS with nothing that is platform specific. What's weird is that the correct number of tows show up, the ID is correct, and I can even print out what the titles should be from the creation for loop. They just aren't showing up!

view.setTableData = function(tbl,data){
        tbl.setData(data);
    }

var categoryTable = Ti.UI.createTableView({
        width:app.appWidth,
        height:viewHeight,
        top:0,
        left:0,
        data:catObj,
        backgroundColor:'#fff'
    });

var subcategoryTable = Ti.UI.createTableView({
        width:app.appWidth,
        height:viewHeight,
        top:0,
        left:app.appWidth,
        backgroundColor:'#fff'
    });

categoryTable.addEventListener('click',function(e){
        var selectedCat = e.rowData.id;
        var subcatArray = [];

        for(var i = 0; i < subcatObj.length; i++)
        {
            if(subcatObj[i].cat_id == selectedCat)
            {
                //Ti.API.info(subcatObj[i].title);
                var tempRow = Ti.UI.createTableViewRow({
                        hasChild:true,
                        id:subcatObj[i].id,
                        catName:e.rowData.catName,
                        subcatName:subcatObj[i].title,
                        height:80,
                        className:'subCatTable'
                    });

                var rowLabel = Ti.UI.createLabel({
                    font:{fontSize:app.fn.aspectRatio(60,'h')},
                    width:app.appWidth - app.fn.aspectRatio(40,'w'),
                    height:40,
                    text:subcatObj[i].title,
                    left:app.fn.aspectRatio(20,'w'),
                    color:'#000'
                });
                tempRow.add(rowLabel);

            }
            view.setTableData(subcategoryTable,subcatArray);
        }

        view.transition(1);
        _btn.currentView = 1;
        _btn.visible = true;
    });

Any help would be awesome.

    Hi

    I am going to have to make a bunch of assumptions as your code references objects and variables that are not provided.

    • app.appWidth equals half the screen.
    • subcatObj is defined somewhere
    • subcatObj has at least one entry
    • At least one row in categoryTable contains and id property suitable for the listener so this works - e.rowData.id

    In this loop you create tempRow and add rowLabel to it. but then you do nothing with tempRow, it is not added to an array to be added to a table later or added to any table directly.

    for(var i = 0; i < subcatObj.length; i++)
{
    if(subcatObj[i].cat_id == selectedCat)
    {
        //Ti.API.info(subcatObj[i].title);
        var tempRow = Ti.UI.createTableViewRow({
                hasChild:true,
                id:subcatObj[i].id,
                catName:e.rowData.catName,
                subcatName:subcatObj[i].title,
                height:80,
                className:'subCatTable'
            });

        var rowLabel = Ti.UI.createLabel({
            font:{fontSize:app.fn.aspectRatio(60,'h')},
            width:app.appWidth - app.fn.aspectRatio(40,'w'),
            height:40,
            text:subcatObj[i].title,
            left:app.fn.aspectRatio(20,'w'),
            color:'#000'
        });
        tempRow.add(rowLabel);

    }
    view.setTableData(subcategoryTable,subcatArray);
}

    You also seem to be settings the table with this line;

    view.setTableData(subcategoryTable,subcatArray);

    Yet at no point do you assign anything to the variable subcatArray for it to have anything to pass over. It simply stays as an empty array.

    I can only assume you have forgotten or deleted the array push and that the setTableData should be outside the loop. 

    for(var i = 0; i < subcatObj.length; i++)
{
    if(subcatObj[i].cat_id == selectedCat)
    {
        //Ti.API.info(subcatObj[i].title);
        var tempRow = Ti.UI.createTableViewRow({
                hasChild:true,
                id:subcatObj[i].id,
                catName:e.rowData.catName,
                subcatName:subcatObj[i].title,
                height:80,
                className:'subCatTable'
            });

        var rowLabel = Ti.UI.createLabel({
            font:{fontSize:app.fn.aspectRatio(60,'h')},
            width:app.appWidth - app.fn.aspectRatio(40,'w'),
            height:40,
            text:subcatObj[i].title,
            left:app.fn.aspectRatio(20,'w'),
            color:'#000'
        });
        tempRow.add(rowLabel);
        subcatArray.push(tempRow);
    }
}
if (subcatArray.length > 0) {
   view.setTableData(subcategoryTable,subcatArray);
}
