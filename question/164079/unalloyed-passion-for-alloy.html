    dashboard.js

    var icons = [];

// private
function addIcon(obj) {
    var intMargin = 5;
    var view = Ti.UI.createView({
 //      backgroundColor: 'orange',
        bottom: intMargin,
        height: Ti.UI.SIZE,
        id: obj.id,
        layout: 'vertical',
        left: intMargin,
        right: intMargin,
        top: intMargin,
        width: 80
    });
    if (obj.action) {
        view.addEventListener('click', function (e) {
            obj.action({
                id: obj.id,
                title: obj.title
            });
        });
    }
    var img = Ti.UI.createImageView({
        height: Ti.UI.SIZE,
        image: obj.icon,
        touchEnabled: false,
        width: Ti.UI.SIZE
    });
    view.add(img);
    var lbl = Ti.UI.createLabel({
        height: Ti.UI.SIZE,
        image: obj.icon,
        color: '#fff',
        text: obj.title,
        textAlign: 'center',
        top: 5,
        touchEnabled: false,
        width: Ti.UI.FILL
    });
    view.add(lbl);
    return view;
}

// public
function add(obj) {
    icons.push({
        id: obj.id,
        icon: obj.icon,
        title: obj.title
    });
}
function show(obj) {
    var view = Ti.UI.createView({
//        backgroundColor: '#FFF',
        height: Ti.UI.SIZE,
        layout: 'horizontal',
        top: '120dp',
        width: Ti.UI.SIZE
    });
    var intIcon = 0, intIcons = icons.length, icon;
    for (intIcon = 0; intIcon < intIcons; intIcon = intIcon + 1) {
        icon = addIcon({
            icon: icons[intIcon].icon,
            id: icons[intIcon].id,
            title: icons[intIcon].title,
            action: obj.action
        });
        view.add(icon);
    }
    return view;
}
exports.add = add;
exports.show = show;

    Hi

    You have "hardcoded" the icons in your controller and then iterated through them setting a single instance of a button to the values, therefore overwriting the previous one with the next and ending up with only one button - the last one.

    You might as well move the "hard coding" to the View XML as follows:

        <View id="iconView">
        <ImageView id="iconImage1" onClick="showTabGroup">
            <Label id="iconLbls1" class="iconLabel" text="Sermon"/>
        </ImageView>
        <ImageView id="iconImage2" onClick="showTabGroup">
            <Label id="iconLbls2" class="iconLabel"  text="Prayer" />
        </ImageView>
... Repeat for each one
        <ImageView id="iconImage9" onClick="showTabGroup">
            <Label id="iconLbls9"  class="iconLabel" text="Contact"/>
        </ImageView>
    </View>

    And then in the style TSS file, add the styling for the text, image and position for each ImageView accordingly

    "#iconImage1": {
      image: '/images/dashboard/sermon.png'
    , top: 0
    , left: 0
    , height: Ti.UI.SIZE
    , width: Ti.UI.SIZE 
}
"#iconImage2": {
      image: '/images/dashboard/prayer.png'
    , top: 0
    , left: 40
    , height: Ti.UI.SIZE
    , width: Ti.UI.SIZE 
}
//
// Repeat for each one
//
"#iconImage9": {
      image: '/images/dashboard/contact.png'
    , top: 120
    , left: 120
    , height: Ti.UI.SIZE
    , width: Ti.UI.SIZE 
}
".iconLabel": {
      height: Ti.UI.SIZE
    , color: '#fff'
    , textAlign: 'center'
    , top: 5
    , touchEnabled: false
    , width: Ti.UI.FILL
}

    Changing the top/left for each icon depending on the row/column they are in and the size of the icon and spacing in between.

    • Hurray!!!! I solved the dilemma by God's grace. The parent view (white-bordered rectangle) containing the icon views and icons was styled in in the index.tss with a vertical layout. With the new TI UI kayo out, composite layout rules, see Transitioning to the New UI Layout System in the Titanium Mobile Guides, each child view top is added to the preceding view bottom which was throwing my views off. To use vertical layout, I found myself having to subtract view heights from subsequent top positions (Not good).

      I switched to horizontal layout everywhere and defined only one iconView class and bing, it all works as desired now. See below.

     height: Ti.UI.SIZE,
 //    layout: 'vertical',
    layout: 'horizontal',
     left: 10,
     right: 10,
     top: Alloy.Globals.margin,
     width: 80
