Is there a way to make a label wrap around an image?

  • [SOLVED] Ok well not really, but this is the (kind of sloppy) way I am doing it for now at least:

        function truncate_at(text, length) {    

        if (text.length < length) return text.length;

        for (var i = length-1; text.charAt(i) != ' '; i--) {
            length--;
        }

        return length;
    }

    var text_value_all = 'Text string to separate';
    var truncate_at_var = truncate_at(text_value_all, 169);
    var text_vale1 = text_value_all.substr(0, truncate_at_var);
    var text_vale2 = text_value_all.substr(truncate_at_var, text_value_all.length);

    var tower_text = Titanium.UI.createLabel({
        color:'#000',
        text: text_vale1,
        width: 210,
        top:26,
        left:10,
        font:{fontSize:12},
        height:'auto'
    });
    tower.add(tower_text);

    if(text_vale2.length > 0) {
        var tower_text_two = Titanium.UI.createLabel({
            color:tower_text.color,
            text: text_vale2,
            width: 300,
            top:tower_text.top + 77,
            left:tower_text.left,
            font: tower_text.font,
            height:'auto'
        });
        tower.add(tower_text_two);
    }

    Please note that the truncate_at() length var will need to be manually calculated to the area, and that a separate section/label needs to be created for each area that needs to wrap… but it works for the app that I am building at least :)
