    Hi Abraham,

    This is very simple code. And now you are primary stage. So you have to learn and practices more. Here is the documentation link for learning. And also have some test code test it.

    var win = Ti.UI.createWindow({

});

var view = Ti.UI.createView();

// Create a Button.
var click = Ti.UI.createButton({
    title : 'Click',
    height : 60,
    width : 100,
    top : 50,

});

// Listen for click events.
click.addEventListener('click', function() {
    alert('\'aButton\' was clicked!');
});

// Add to the parent view.
view.add(click);

win.add(view);
win.open();
