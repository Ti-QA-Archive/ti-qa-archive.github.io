  • Hi Artur Sz,

    Thanks for your concern.

    I've tested your issue on android and works fine for me.

    For test case, i've used picker and few text fields. ScrollView and Picker works fine on my side.

    Please check my test sample code.

    
var window1 = Titanium.UI.createWindow({
    backgroundColor:'green',
    layout:'vetical'
});

var scrollView = Ti.UI.createScrollView({
  top:0,
  backgroundColor:'red',
  layout:"vertical",
  contentWidth: 'auto',
  contentHeight: 'auto',
  showVerticalScrollIndicator: true,
  showHorizontalScrollIndicator: true,
  height: '30%',
  width: '100%'
});
window1.add(scrollView);

var picker = Ti.UI.createPicker({
    top:5
});
var data = [];
data[0]=Ti.UI.createPickerRow({title:'Appcelerator'});
data[1]=Ti.UI.createPickerRow({title:'Titanium'});
picker.add(data);
scrollView.add(picker);
picker.setSelectedRow(0, 0, false);

picker.addEventListener('change', function(e) {
    alert(e.row.title);
});

var txtFld= Ti.UI.createTextField({
    top:5,
    height:50,
    width:200,
    hintText:'I\'m a text field.' 
});
scrollView.add(txtFld);

var txtFld2= Ti.UI.createTextField({
    top:5,
    height:50,
    width:200,
    hintText:'I\'m a text field.' 
});
scrollView.add(txtFld2);

var txtFld3= Ti.UI.createTextField({
    top:5,
    height:50,
    width:200,
    hintText:'I\'m a text field.' 
});
scrollView.add(txtFld3);

window1.open();

    Hope it will help you.

    Thanks
