    You can set the height of list items ("rows") in a ListView by setting their height property. Like this:

    <!-- in the template -->
<ItemTemplate name="template" height="60">...

<!-- a specific item -->
<ListItem height="100" ...

    /* in the TSS, use a class or ID selector that applies to ItemTemplate */
"#oddrows": {
    height: 100
}

    // in the controller
var items=[];
for(var i=0; i<5; i++){
    items.push({
        title: {text: 'Title one'}, 
        properties: {height: 100}
    });
};
$.mySection.setItems(items);
//
