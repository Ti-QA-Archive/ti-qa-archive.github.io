Passing a value from a modal window to the parent

Hi, If I have a variable with a value in a modal window and I want to pass it to the parent window,for example:

//In the parent I call the modal
w.open({modal:true,modalTransitionStyle:style,modalStyle:presentation,navBarHidden:false});

//once in the modal I assign a variable
var image='test';

before I make the close of the modal window, how I pass the value to the parent? thanks!

    • I have this parent window that call a modal:

      (parent)

      var image='';

      var signBtn = Ti.UI.createButton(
      {
      title: 'Firmas',
      style:Ti.UI.iPhone.SystemButton.BACK
      });

      signBtn.addEventListener('click',function(e)
      {
      var style = Ti.UI.iPhone.MODAL_TRANSITION_STYLE_FLIP_HORIZONTAL;
      var presentation = Ti.UI.iPhone.MODAL_PRESENTATION_PAGESHEET

      var w = Ti.UI.createWindow({
    url:'firmas.js',
    title:'Firmas'
});
w.backgroundImage = '../background.png';
w.open({modal:true,modalTransitionStyle:style,modalStyle:presentation,navBarHidden:false});

      });
      win.rightNavButton= signBtn;

      once on the modal the user make a screenshot, and I want to pass the screenshot photo back to the parent where I want to attach it to an email.

      I have this code on the modal window:

      doneButton.addEventListener('click',function()
      {
      Titanium.Media.takeScreenshot(function(event)
      {
      f=Ti.Media.saveToPhotoGallery(event.media);
      image = event.media;
      var f = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory,image);
      f.write(image);
      });
      alert ('Firmas Guardadas!');
      });

      var backBtn = Ti.UI.createButton(
      {
      title: 'Cerrar',
      style:Ti.UI.iPhone.SystemButton.BACK
      });

      backBtn.addEventListener('click', function()
{
 win.close();
});

      win.leftNavButton = backBtn;

      Back into the parent I want to attach the screenshot taked on the modal:
      (Parent)

      var f = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory,image);
      emailDialog.addAttachment(f);

      but the image variable is coming empty, any idea?
