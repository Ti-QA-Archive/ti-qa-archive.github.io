    for (var c=0; c<pk.length;c++)

    Try this in for loop:

    for(var c = 0; c < pk.length; c++) {
        ...

        Ti.API.info(pkgsID);

        if ( pkgsID.indexOf('PKG') == -1 && pkgsID.indexOf('PACKAGE') == -1 ) { // Ignore rows not contain PKG, PACKAGE 
            continue;
        }

        ...
}
