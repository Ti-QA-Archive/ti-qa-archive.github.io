  • This works:

    w.activity.onCreateOptionsMenu = function(_e) {
            var menu = _e.menu;
            var m1 = menu.add({
                title : 'Do you want logout?'
            });
            m1.setIcon(Titanium.Android.R.drawable.ic_menu_close_clear_cancel);
            m1.addEventListener('click', function(e) {
                tabGroup.close();
            });
        };
