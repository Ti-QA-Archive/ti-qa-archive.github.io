      var rideauView = Titanium.UI.createView({
    width: 250,
    height: 150,
    backgroundImage: 'images/rideau.png',
    top: 220,
    left: 35
});
win.add(rideauView);

var coverView = Titanium.UI.createView({
    width: 250,
    height: 115,
    top: 90,
    left: 35
});
win.add(coverView);

function displayImages(xmlstr) {
    var xmldoc = Titanium.XML.parseString(xmlstr);
    var doc = xmldoc.documentElement;
    var image = doc.getElementsByTagName("FILMS");

    var items = image.item(0).getElementsByTagName("FILM");
    var images = [];
    var datas = [];

    for (var c = 0; c < items.length; c++) {
        var item = items.item(c);
        var photo = item.getElementsByTagName("PHOTO").item(0).text;

        images[c] = 'http://.........../' + photo;
        datas[c] = item;
    }


    var ligneTitre = Ti.UI.createLabel({
        text: '',
        color: '#000',
        textAlign: 'center',
        //left: 60,
        height: 'auto',
        width: 'auto',
        top: 15,
        font: {
            fontWeight: 'bold',
            fontSize: 16
        }
    });

    var ligneTitreEmis = Ti.UI.createLabel({
        text: '',
        color: '#808080',
        textAlign: 'center',
        //left: 60,
        height: 'auto',
        width: 'auto',
        top: 47,
        font: {
            //fontWeight: 'bold',
            fontSize: 12,
            fontFamily: 'Verdana-BoldItalic'
        }
    });

    var ligneDate = Ti.UI.createLabel({
        text: '',
        color: '#808080',
        textAlign: 'center',
        //left: 135,
        height: 'auto',
        width: 'auto',
        top: 330,
        font: {
            fontWeight: 'bold',
            fontSize: 12
        }
    });



    // create coverflow view with images 

    var coverflow = Titanium.UI.createCoverFlowView({
        top: 0,
        images: images,
        defaultImage: 'images/openflow_default.png',
        backgroundColor: 'transparent'
    });

        coverflow.addEventListener('change', function(e) {

            var index;
            if (e.index == null) {
                index = 0;
            } else {
                index = e.index;
            }

            var titre = datas[index].getElementsByTagName("TITREPLAYER").item(0).text;
            var titreF = (titre.length > 20) ? titre.substring(0, 20) + "..." : titre;
            var titreFM = titreF.toUpperCase();
            var titreEmis = datas[index].getElementsByTagName("TITRE").item(0).text;
            var date = datas[index].getElementsByTagName("DATE_DEBUT").item(0).text;
            var mois = new Array("Janvier", "Février", "Mars", "Avril", "Mai", "Juin", "Juillet", "Août", "Septembre", "Octobre", "Novembre", "Décembre");
            var jours = new Array("Dimanche", "Lundi", "Mardi", "Mercredi", "Jeudi", "Vendredi", "Samedi");
            date = date.replace('.0', '');
            var tab = date.split(' '); // On sépare date (yyyy/mm/dd) et heure (hh:mm)
            var tab_date = tab[0].split('/'); // On récupère l'année, le mois et le jour dans un tableau
            var tab_time = tab[1].split(':'); // On récupère l'heure et les minutes dans un tableau
            date = new Date(tab_date[0], tab_date[1] - 1, tab_date[2], tab_time[0], tab_time[1]);
            var jour = (date.getDate() < 10) ? "0" + date.getDate() : date.getDate();
            var minutes = date.getMinutes();
            if (date.getMinutes() < 10) {
                minutes = "0" + date.getMinutes();
            }

            date = jours[date.getDay()] + " " + jour + " " + mois[date.getMonth()] + "  -  " + date.getHours() + "h" + minutes;

            ligneTitre.text = titreFM;
            ligneTitreEmis.text = titreEmis;
            ligneDate.text = date;

        });

        rideauView.add(ligneTitre);
        rideauView.add(ligneTitreEmis);
        win.add(ligneDate);

        coverView.add(coverflow);
        coverflow.fireEvent('change', null);

    }



filmsServices.callService("film", displayImages);
