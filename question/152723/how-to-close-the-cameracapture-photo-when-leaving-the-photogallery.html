How to close the camera/capture photo when leaving the photogallery.

I want to close the whole "camera capture view" when closing the photogallery. But I cant figure out how to do it.
Please see the code with the cameraview and overlay:

var gallery = Ti.UI.createButton({
    bottom:10, 
    width:120, 
    height:40,
    right: 10,
    title:'Gallery'
});


gallery.addEventListener("click", function(){
    changeImage = this.idx;
    Ti.Media.openPhotoGallery ({
        allowEditing: false,
        mediaTypes: Ti.Media.MEDIA_TYPE_PHOTO,
        showControls: false,
        overlay:overlay,
        autoHide: false,
        success: function(e){
            placeholder[changeImage].setImage(e.media.imageAsThumbnail(129,0,0));
            placeholder[changeImage].setBorderRadius(10);
            var aspectRatio = e.media.height / e.media.width;
            photos[changeImage] = imagefactory.compress(e.media.imageAsResized(640, 640*aspectRatio), 0.5);

        }

    });

});



var overlay = ui.view({

});

overlay.add(gallery);

    top: 80,
    left: 25,
