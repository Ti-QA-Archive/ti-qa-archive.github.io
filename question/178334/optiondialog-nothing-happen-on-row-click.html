OptionDialog nothing happen on row click

I created my OptionDialog and I can correctly open it but nothing happen when i select a row…

Here is my code: 

function OpenPopUp(regionArea) {
        var data = [];

        var v1 = Ti.UI.createView({
            backgroundColor : '#212429'
        });

        var tableView = Ti.UI.createTableView({
            separatorColor : '#111214'
        });
        v1.add(tableView);

        for(var i=0; i < regionArea.length; i++){
            var Row = Ti.UI.createTableViewRow({
                height : '50dp',
                touchEnabled : false,
                width : '100%',
                backgroundColor : '#212429'
            });

            var label = Ti.UI.createLabel({
                text : regionArea[i],
                color : 'white',
                touchEnabled : false,
                font : {fontSize : '18dp'}
            });
            Row.add(label);
            data.push(Row);
        }
        tableView.setData(data);

        var dialog  = Ti.UI.createOptionDialog({
            title:"Select your area:",
            androidView:v1,
            //cancel:1
            //buttonNames:['Cancel']
        });    

        dialog.addEventListener('click',function(e)
        {
            dialog.hide();
            alert('You selected ' + e.index);
        });

        dialog.show();
}

How can my function return the selected row?

Regards,
Matteo
