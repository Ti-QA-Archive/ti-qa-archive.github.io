  • Hi, thanks for reply :)
    This is the pdf generated with bencodings.pdf:
    http://www.mediafire.com/view/1blmrq7biyuekjc/_invoice2.pdf

    This is the html file used to generate the pdf:
    http://www.mediafire.com/view/fkae4vigv1eewg8/template.html

    And this is the code:

    var pdfview = Ti.UI.createWebView({width:'210mm',height:'297mm',html:htmlTemplate});
var win = Ti.UI.createWindow();
win.fullscreen = true;
win.navBarHidden = true;
win.add(pdfview);

pdfview.addEventListener('load', function(e) {
    var pdf = require('bencoding.pdf');
    var img2 = (Ti.Android ? pdfview.toImage().media : pdfview.toImage());
    var pdfBlob2 = pdf.createConverters().convertImageToPDF(img2,100);
    var pdfFile2 = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, 'invoice2.pdf');
    pdfFile2.write(pdfBlob2);
});
