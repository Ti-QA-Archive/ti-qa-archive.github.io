  • Hello Raffaele,

    See this…

    
var d = [];

var searchBar = Titanium.UI.createSearchBar({
    hintText : "Search Here",
    barColor : 'black',
});

searchBar.addEventListener('cancel', function(e) {
    searchBar.blur();
});

var tblView = Ti.UI.createTableView({
    search : searchBar
});

var arr = ['row', 'title', 'header', 'footer', 'testing', 'row1', 'title1', 'header1', 'footer1', 'testing1'];

for ( i = 0; i < arr.length; i++) {

    var row = Ti.UI.createTableViewRow({
        height : 60,
        title : arr[i]
    });
    d.push(row);

}
tblView.setData(d);
tblView.filterAttribute = 'filter';

win.add(tblView);
