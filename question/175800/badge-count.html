  • Hi Maria,

    Thanks for your concern.

    I've tested your issue to increment the Badge Count for local notification.

    Please check my test code below.

    var win = Ti.UI.createWindow({
    backgroundColor:'#fff'
});

//Ti.UI.iPhone.appBadge=1;
function getBadge(){
    var badgeCount = Ti.UI.iPhone.getAppBadge(); //Will return the app badges
    badgeCount = badgeCount + 1;                 //Incrementing the appbadge
    var result=Ti.UI.iPhone.setAppBadge(badgeCount); // Set the badge count
    return result;
}

var button = Ti.UI.createButton({
    title : 'Local Notification',
    height : 50,
    width : 160,
    top : 60,
});

button.addEventListener('click', function() {
    Ti.App.iOS.scheduleLocalNotification({
        alertBody:"App was put in background",
        alertAction:"Re-Launch!",
        userInfo:{"hello":"world"},
        badge : getBadge(),
        date:new Date(new Date().getTime() + 3000) // 3 seconds after backgrounding
    });
});

win.add(button);
win.open();

    You can also decrease your badge number similarly.

    Hope it'll help you to solve your problem.
