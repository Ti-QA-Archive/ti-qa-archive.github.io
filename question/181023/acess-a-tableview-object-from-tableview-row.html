      index.xml

      
<Alloy>
    <Window>
        <TableView id="tbl" onClick="clkTblView">
            <TableViewRow layout="horizontal">
                <Label>1</Label>
                <Label>2</Label>
                <Label>3</Label>
            </TableViewRow>
            <TableViewRow layout="horizontal">
                <Label>4</Label>
                <Label>5</Label>
                <Label>6</Label>
            </TableViewRow>
        </TableView>
    </Window>
</Alloy>

      index.js

      
$.index.open();

function clkTblView(e) {
    Ti.API.info('e ==> ' + JSON.stringify(e));
    Ti.API.info('Row swiped: ' + e.source);
    $.tbl.deleteRow(e.index);
}
