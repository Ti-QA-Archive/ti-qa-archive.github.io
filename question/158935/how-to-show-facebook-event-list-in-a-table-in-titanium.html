  • function show(){

    var EventlistTableView= Ti.UI.createTableView({
    top : 60,
    buttom : 40,
    minRowHeight : 100,

    });

    win2.add(EventlistTableView);

    fb.requestWithGraphPath('me/events', {fields:'picture,name,start_time'}, 'GET', function(e) {
    if (e.success) {

        var evs=JSON.parse(e.result);
    for(var a=0;a<evs.data.length;a++)
    {


        var evrow=Ti.UI.createTableViewRow({
            layout:'horizontal',
            evid:evs.data[a].id
        });

        var evimage=Ti.UI.createImageView({
            image:evs.data[a].picture.data.url,
            width:60,
            height:60
        });

        var evview=Ti.UI.createView({
            layout:'horizontal',
            width:210
        });

        var evlevel=Ti.UI.createLabel({
            text:evs.data[a].name,
            left:10
        });
        evview.add(evlevel);
        var evDate=Ti.UI.createLabel({
            text:evs.data[a].start_time,
            left:10
        });
        evview.add(evDate);
        evrow.add(evimage);
        evrow.add(evview);
        EventlistTableView.appendRow(evrow);

    }

} else if (e.error) {
    alert(e.error);
} else {
    alert('Unknown response');
}

    });

    }
