Transformed scrollable view on ANdroid doest work

Hi,
this simple code should rotate a gallery:

exports.create = function() {
    var container = Ti.UI.createView({
        transform : Ti.UI.create2DMatrix().rotate(90).scale(1.2),
        width : Ti.Platform.displayCaps.platformHeight,
        height :
        Ti.Platform.displayCaps.platformWidth
    });
    var views = [];
    for (var i = 0; i < 6; i++) {
        views.push(Ti.UI.createImageView({
            image : '/assets/fotos/' + i + '.jpg',
        }));
    }
    var gallery = Ti.UI.createScrollableView({
        views : views
    });
    container.add(gallery);
    return container;
}
