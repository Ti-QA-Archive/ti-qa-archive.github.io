Why can not I create a new window in addEventListener?

When I try to create a new window in an application crash addEventListener

app.js

var data = [
    {title: 'Test', view: 'z.js'},
];

var main = Titanium.UI.createWindow({
    title:'Main',
});

var tableView = Titanium.UI.createTableView({
    data:data,
    headerTitle: 'Test',
});

main.add(tableView);
main.open();

tableView.addEventListener('click', function(e)
{
    var new = Titanium.UI.createWindow({
        title: 'asdas',
        url: 'z.js',
    });
    new.open();
});

z.js

alert('z');

  • Accepted Answer

    You can't name a variable new, it is a reserved word in javascript. Change:

    var new = Titanium.UI.createWindow({
        title: 'asdas',
        url: 'z.js',
    });
    new.open();

    to 

    var win2 = Titanium.UI.createWindow({
        title: 'asdas',
        url: 'z.js',
    });
    win2.open();

    and everything works fine.
