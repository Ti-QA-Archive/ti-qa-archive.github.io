position properties (top, right, bottom, left) not updating after animation

animateSlideIn = Ti.UI.createAnimation({bottom:0});
animateSlideOut = Ti.UI.createAnimation({bottom:(pickerView.height * -1)});

toggleButton.addEventListener('click',function(e) {
    Ti.API.info('bottom before: ' + pickerView.bottom);
    if(pickerView.bottom == 0){
        pickerView.animate(animateSlideOut);
