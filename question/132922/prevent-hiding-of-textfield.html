  • Add your image view into ScrollView would help

    var win = Ti.UI.createWindow({backgroundColor : 'red'});
var iv = Ti.UI.createImageView({image: 'images/Default.png'});
var tf1 = Titanium.UI.createTextField({color:'#336699', height:35, top:200, left:10, width:250, borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED}); 
iv.add(tf1);
var tf2 = Titanium.UI.createTextField({color:'#336699', height:35, top:250, left:10, width:250, borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED}); 
iv.add(tf2);

var sv = Ti.UI.createScrollView();
sv.add(iv);
win.add(sv);
win.open();

tf2.addEventListener('blur', function(){
    sv.scrollTo(0, 0);
});
