GridLayout for ALL :)

Hello folks,

As I am learning the Titanium mobile.. i find it hard to acctually pinpoint every pixel and use the top left bottom right… does Titanium Mobile have gridlayout.. ?? They say.. it dont.. so good news :)
Just now i created a GridLayout that can help you perfectly align your objects in the window.. you are free to use or modify it depending in your needs..

I know its not so perfect like the JAVA gridlayout but i hope it can help you all who have problem in getting a starting layout.. so i hope this could help you :)..

Basically the simple idea behind this is the multiplication table that we do when we start making our program in java :).. so i just adapt the code like that.. its simple but i guess worth it..

exports.GridLayout = function(row, col, padding){
    var vview = Ti.UI.createView({
        height: 'auto',
        width: 'auto',
        layout: "vertical"
    });
    for(var x = 0; x < row ; x++){
        var hview = Ti.UI.createView({
            height: "auto",
            width: "auto",
            layout: "horizontal"
        });
        for(var y = 0; y < col; y++){
            var btn = Ti.UI.createButton({
                title: x+1,
                height: 150,
                width: 150,
                top: padding,
                left: padding
            });
            hview.add(btn);
        }
        vview.add(hview);
    }
    return vview;
}

SO again i assume you save the above code in gridlayout.js :)
So basically i jsut use a btn here :)… to clearly show how its working..
to use this.. in your app.js you can do this..

var win = Ti.UI.createWindow({
    backgroundColor: 'gray',
    fullscreen: true
});

var view = require("gridlayout").GridLayout(3,3,2);
win.add(mylayout);

win.orientationModes = [Ti.UI.PORTRAIT];
win.open();

    exports.GridLayout = function(row, col, padding,obj){
    var vview = Ti.UI.createView({
        height: 'auto',
        width: 'auto',
        layout: "vertical"
    });
    var tmp=0;
    for(var x = 0; x < row ; x++){
        var hview = Ti.UI.createView({
            height: "auto",
            width: "auto",
            layout: "horizontal"
        });
        for(var y = 0; y < col; y++){
            obj[tmp].top = padding;
            obj[tmp].left = padding;
            hview.add(obj[tmp]);
            tmp++;
        }
        vview.add(hview);
    }
    return vview;
}

  • Hello again.. i modify the GridLayout… give it more functionality…
    So far you can edit.. each object alignment to the grid box (left, right, top, bottom etc…), give you how many percent will it occupy in the window

    exports.GridLayout = function(row, col, padding,obj,space){

    var elemPad = (padding ==0) ? 0 : padding + (padding/2);    //to balance the width of each element
    var eviewWidth = (parseFloat(Math.floor(100 / col))/100) * Ti.Platform.displayCaps.platformWidth - elemPad; //the element width

    //grid container
    var vview = Ti.UI.createView({
        height: space + '%',
        width: '100%',
        layout: "vertical"
    });
    var ctr=0;    //counter
    for(var x = 0; x < row ; x++){
        //horizontal container
        var hview = Ti.UI.createView({
            height: Math.floor(100 / row) + '%',
            width: '100%',
            layout: "horizontal"
        });
        for(var y = 0; y < col; y++){
            //create element container
            var eview = Ti.UI.createView({
                backgroundColor: 'gray',
                height: '100%',
                width: eviewWidth,
                top: padding,
                left: padding
            });
            hview.add(eview);
            eview.add(obj[ctr]);
            require("alignment").setAlign(obj[ctr]); //set alignment of the object inside the element
            ctr++;
        }
        vview.add(hview);
    }
    return vview;
}

    save it again as gridlayout.js
    there is another file i added that set the alignment.. save the below code to alignment.js

    exports.setAlign =     function (obj){
        switch(obj.align){
            case 'center':
            break;
            case 'left':
                obj.left = 0;
            break;
            case 'right':
                obj.right = 0;
            break;
            case 'top':
                obj.top = 0;
            break;
            case 'bottom':
                obj.bottom = 0;
            break;
            case 'topleft':
                obj.left = 0;
                obj.top = 0;
            break;
            case 'topright':
                obj.right = 0;
                obj.top = 0;
            break;
            case 'leftbottom':
                obj.left = 0;
                obj.bottom = 0;
            break;
            case 'rightbottom':
                obj.right = 0;
                obj.bottom = 0;
            break;
        }
    return obj;
}

    • Well … I could help you a bit.

      It would be ideal if you could get particular element like this:

      var element = myGrid[2][3];

      To be able to do that you need two dimensional array, or you could say array of arrays.

      myGrid[2] //contains array of columns in second row

myGrid //contains array of rows in the grid

      The thing here is that "vview" is Titanium object so you cannot use it like an array, but you can use it like an object (which properties are numbers).

      So ..

      vview[x] = hview; //object that will contain x-th row's columns

vview[x][y] = btn; //reference to x-th row and y-th column element

      Note that it would be better if buttons were view, so you can easily add and remove components.

      Hope you understand (it's not complete solution, but you can use this idea to implement this behavior by yourself).
