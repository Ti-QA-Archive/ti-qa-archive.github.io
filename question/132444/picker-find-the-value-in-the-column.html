picker find the value in the column

Hi all,

I have a mutli-column picker and i would like to take information from the choice of the user and after a click on a button go to a other views to see the result.

`Ti.UI.backgroundColor = 'black';
var win = Ti.UI.createWindow({
exitOnClose: true,
layout: 'vertical'
});
var button = Titanium.UI.createButton({
title: 'Hello',
top: 10,
width: 100,
height: 50
});

var picker = Ti.UI.createPicker({
top:50
});
picker.selectionIndicator = true;

var style = [ 'American', 'Chinese', 'French', 'Indian', 'International','Italian','Japanese & Korean','Southest Asian','Spanish & Latin','Vietnamese'];
var district = [ 'Every where', 'District 1', 'District 2', 'District 3','District 4', 'District 5', 'District 6', 'District 7', 'District 8','District 9','District 10','District 11','District 12' ];

var column1 = Ti.UI.createPickerColumn();

for(var i=0, ilen=style.length; i<ilen; i++){
var row = Ti.UI.createPickerRow({
title: style[i]
});
column1.addRow(row);
}

var column2 = Ti.UI.createPickerColumn();

for(var i=0, ilen=district.length; i<ilen; i++){
var row = Ti.UI.createPickerRow({ title: district[i] });
column2.addRow(row);
}

picker.add([column1,column2]);

/////

win.add(picker);
win.add(button);

win.open();

// must be after picker has been displayed
picker.setSelectedRow(0, 2, false); // select French Restaurant
picker.setSelectedRow(1, 1, false); // in district 1

button.addEventListener('click',function(e)
{

var colone1 = picker.getSelectedRow(0,0);
var colone2 = picker.getSelectedRow(1,0);

Titanium.API.info("la valeur de la colone 1 est",colone1);
win.close();
Titanium.API.info("You clicked the button");
});`

    Try this

    var row = '';
var col = '';
picker.addEventListener('change', function(e) {
    row = e.selectedValue[0];
    col = e.selectedValue[1];
    Ti.API.info("You selected row: " + row + ", column: " + col);
});
// must be after picker has been displayed picker.setSelectedRow(0, 2, false); // select French Restaurant picker.setSelectedRow(1, 1, false); // in district 1
button.addEventListener('click', function(e) {
    alert("You selected row: " + row + ", column: " + col);
});
