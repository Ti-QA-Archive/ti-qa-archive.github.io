  • Debug needs to be a functional object in order to do prototypes:

    var Debug = function()
{
    this._prefix = "something"
};

Debug.prototype.writeLine = function(msg)
{
    Ti.API.log( this._prefix + msg );
};

    Also need to use this in order to set and get _prefix.
