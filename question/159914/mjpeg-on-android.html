  • External URL
    ie: (http://0.0.0.0/img/video.mjpeg?mac=000E8F7F5DFG)

    var mjpeg = 'http://0.0.0.0/img/video.mjpeg?mac=000E8F7F5DFG';
var videoHTML='<html><body style="background:#000;padding:0px;margin:0px;"><img width="320" height="240" src="'+ mjpeg +'"></body></html>';
$.webview.html=videoHTML;
