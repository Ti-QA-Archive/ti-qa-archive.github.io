  • Use the Titanium.Network.networkType and Titanium.Network.NETWORK_NONE : 

    if (Titanium.Network.networkType === Titanium.Network.NETWORK_NONE) {
        console.log("Not online");
    } else {
        console.log("Online);
    }
}
