  • 
var request = Titanium.Network.createHTTPClient();

var localFilename = 'myfile.mp3';    

var localFile = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, localFilename);


request.onload = function() {
    if (request.status >= 200 && request.status < 300) {
        localFile.write(request.responseData);
        Ti.API.info("file downloaded")
    }
}

var url = 'http://www.mywebsite.com/myfile.mp3';
request.open('GET', url);
request.send();
