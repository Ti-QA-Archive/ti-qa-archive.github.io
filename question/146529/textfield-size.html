  • (Titanium.UI.SIZE)[http://docs.appcelerator.com/titanium/latest/#!/api/Titanium.UI-property-SIZE]

     var textField = Ti.UI.createTextField({
     top : 10,
     hintText : 'This is hint text',
     height:  Ti.UI.SIZE ,    /*The SIZE behavior means the view will constrain its size fit its contents.*/
     width: Ti.UI.SIZE ,

 });
