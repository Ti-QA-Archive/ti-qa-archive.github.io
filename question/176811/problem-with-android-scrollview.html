  • Hi Ben,

    Have you set top position of text and try to set top of scrollview too.

    var descriptionScrollView = Ti.UI.createScrollView({
    height:250,
    width:300,
    contentHeight:'auto',
    scrollType: 'vertical',
    top : 0,
    layout : 'vertical'
});

var descriptionLabel = Ti.UI.createLabel({
    height:Ti.UI.SIZE,
    width:"90%",
    top : 0,
    text : 'hi this is text asf sdfs  s sd sf sdf sdf sf sf s fsdf sd'
});
descriptionScrollView.add(descriptionLabel);
    — commented August 7th 2014 by Nitin Chavda
