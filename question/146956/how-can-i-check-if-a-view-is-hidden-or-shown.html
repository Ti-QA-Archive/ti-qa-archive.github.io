How can I check if a view is hidden or shown?

Situation aside I have the need to check if a view is hidden but cannot find anything in the documentation - is there an undocumented property or method I can use to check this?


// Create a view
var view = Ti.UI.createView({
    backgroundColor:    '#ABCDEF',
    layout:                'vertical',
    width:                Ti.UI.FILL,
    height:                100
});
win.add(view);
// Randomly hide it
if(Math.random() > 0.5){
    view.hide();
}
// Is the view hidden or not...?
Ti.API.error('How do I check if the view is hidden??');

    Hi David,

    it looks like there is no hidden property.

    But you can use visible additionally, or add your own property to your view.

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

// Create a view
var view = Ti.UI.createView({
    backgroundColor : '#ABCDEF',
    layout : 'vertical',
    top : 0,
    width : Ti.UI.FILL,
    height : 100
});
win.add(view);

var label = Ti.UI.createLabel({
    top : 150,
    text : 'View is visible'
});
win.add(label);

var button = Ti.UI.createButton({
    top : 200,
    title : 'random view visibility'
});
button.addEventListener('click', function(e) {
    // Randomly hide/show it
    if (Math.random() > 0.5) {
        view.hide();
        view.setVisible(false);
    } else {
        view.show();
        view.setVisible(true);
    }

    // Is the view hidden or not...?
    if (view.visible) {
        label.text = 'View is visible';
    } else {
        label.text = 'View is hidden';
    }

});
win.add(button);

win.open();
