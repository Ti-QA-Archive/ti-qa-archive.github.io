  • Hi Kiran,
    Try this way…..

    var imgView = Ti.UI.createImageView({
                image : "yourimage",
                width : "width",
                height : "height",
                customId : 'yourId' //== set your id here and get on click event like this way
        });
imgView.addEventListener('click', function(e) {
                alert(e.customId);
});
