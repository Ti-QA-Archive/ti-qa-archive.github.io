    Hi

    You are using the search bar directly with the table so it filters the table results and does not start a fresh search, that is fine.

    You have told the table to not use title but instead use name. This is also fine.

    However you have not referenced the attribute name anywhere in your tableViewRow create. You do have it as a label but that is not what the filter is looking for.

    So add the following property to your createTableViewRow code and it should work as you first thought it should.

    name: rows.fieldByName('fname')+' '+rows.fieldByName('lname'),

    • Hi

      This is some standalone code to show you how to use the search filterAttribute, simply create a blank new project and swap the app.js code with the code below - it is a fully functionally app demonstrating the table with search.

      Ti.UI.setBackgroundColor('#000');

var win = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Tab 1',
});
var tab1 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 1',
    window: win
});

var data = [
    { firstname: 'Neva', lastname: 'Asmussen' },
    { firstname: 'Julianne', lastname: 'Bottorff' },
    { firstname: 'Ted', lastname: 'Michelin' },
    { firstname: 'Hugh', lastname: 'Duchene' },
    { firstname: 'Edwina', lastname: 'Englehardt' },
    { firstname: 'Dollie', lastname: 'Mccright' },
    { firstname: 'Edwina', lastname: 'Whipps' },
    { firstname: 'Margery', lastname: 'Klock' },
    { firstname: 'Eve', lastname: 'Reetz' },
    { firstname: 'Kelly', lastname: 'Gaier' }
];

var intRow = 0, intRows = data.length, rows = [], row, strFullName, lblName;
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    strFullName = data[intRow].firstname + ' ' + data[intRow].lastname;
    row = Ti.UI.createTableViewRow({
        height: Ti.UI.SIZE,
        name: strFullName, // this points to name below
        width: Ti.UI.FILL
    });
    view = Ti.UI.createView({
        backgroundColor: '#fff',
        bottom: 10,
        height: Ti.UI.SIZE,
        left: 10,
        right: 10,
        top: 10,
        width: Ti.UI.FILL
    });
    row.add(view);
    lblName = Ti.UI.createLabel({
        height: Ti.UI.SIZE,
        left: 0,
        text: strFullName,
        textAlign: 'left',
        width: Ti.UI.SIZE    
    });
    view.add(lblName);
    rows.push(row);
}
var search = Ti.UI.createSearchBar({
});
var tbl = Ti.UI.createTableView({
    data: rows,
    filterAttribute: 'name', // this points to name above
    height: Ti.UI.FILL,
    search: search,
    width: Ti.UI.FILL
});
win.add(tbl);

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);  
tabGroup.open();
