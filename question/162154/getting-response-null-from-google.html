    var Google = require('googleservices').GoogleService;

var google = new Google({

    clientId : 'xxxxxxxxxxxxxxxxxxx',
    clientSecret : 'xxxxxxxxxxxxx',

    devKey : 'xxxxxxxxxxxxxxxxxxxxxxx'

$.signin.addEventListener('click', function(e) {

    google.login(function(e) {
        Ti.API.info("inside login...............");
        Ti.API.info('Token:................... ' + google.accessToken());
        google.refreshToken(function(e) {
            Ti.API.info('New Token: ' + e.token);

            var params = {
                params : [],
                call : 'userinfo',
                method : 'GET'

            };
            google.callMethod(params, function(e) {
                Ti.API.info(e);
                if (e.data) {
                    var data = JSON.parse(e.data);
                    Ti.API.info(data + "dataaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa");
                    Ti.API.info(data.email);
                    Ti.API.info(data.family_name);
                    Ti.API.info(data.given_name);
                    Ti.API.info(data.picture);
                    Ti.API.info(data.gender);
                    Ti.API.info(data.id);
                }
            });
        });
    });
});
