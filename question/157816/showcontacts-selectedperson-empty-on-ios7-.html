showContacts selectedPerson empty on iOS7 ?

Hi all,

I'm trying to do a simple workflow : show the adress book native view, pick a contact, display it's name.

Here is the code on my Controller : 

var contactSelected = function( e ) {
    alert(JSON.stringify(e));
};
var performAddressBookFunction = function() {
    Ti.Contacts.showContacts({
        selectedPerson : contactSelected,
        animated : true
    });
};
var addressBookDisallowed = function() {
    alert("SyncYou need access to your address book in order to work.");
};
if (Ti.Contacts.contactsAuthorization == Ti.Contacts.AUTHORIZATION_AUTHORIZED) {
    performAddressBookFunction();
} else if (Ti.Contacts.contactsAuthorization == Ti.Contacts.AUTHORIZATION_UNKNOWN) {
    Ti.Contacts.requestAuthorization(function(e) {
        if (e.success) {
            performAddressBookFunction();
        } else {
            addressBookDisallowed();
        }
    });
} else {
    addressBookDisallowed();
}

It'a an Alloy project with SDK 3.1.3GA, latest Titanium Studio.
I'm testing on iOS 7 (Simulator / Device get the same result)

    This works fine for me. The only thing to note is that the JSON.stringify will not enumerate the e.person properties, you have to explicitly mention them

    eg

    alert(JSON.stringify(e.person.firstName));
