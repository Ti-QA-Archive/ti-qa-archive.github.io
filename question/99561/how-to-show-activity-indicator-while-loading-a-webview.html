  • // activity indicator for entertainment
var actInd = Titanium.UI.createActivityIndicator();
actInd.message = 'waiting for godot...';
actInd.show();
...
var webview = Titanium.UI.createWebView({...});
webiew.addEventListener("load", function(e) {
    actInd.hide();
});

  • Hi Mohammad,

    Try the following..

    var wvwChart = Ti.UI.createWebView({id:'spcWvwChart',
                url:'Logic/share_price_chart.html?period=30'
                });
                       var actInd;
            wvwChart.addEventListener('beforeload',function(e){
                actInd = Titanium.UI.createActivityIndicator({
                style:Titanium.UI.iPhone.ActivityIndicatorStyle.DARK,
                height:60,
                width:60
                });
                wvwChart.add(actInd);
                actInd.show();
            });
            wvwChart.addEventListener('load',function(e){ actInd.hide(); });
