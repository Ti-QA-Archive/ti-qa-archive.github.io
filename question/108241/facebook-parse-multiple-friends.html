    Here is one method:

    var fb = {
   "cocacola": {
      "id": "40796308305",
      "name": "Coca-Cola",
      "link": "http://www.facebook.com/coca-cola",
      "category": "Company",
      "website": "http://www.coca-cola.com",
      "username": "coca-cola",
      "products": "Coca-Cola is the most...",
      "likes": 22177256
   },
   "btaylor": {
      "id": "220439",
      "name": "Bret Taylor",
      "first_name": "Bret",
      "last_name": "Taylor",
      "link": "http://www.facebook.com/btaylor",
      "gender": "male",
      "locale": "en_US"
   }
};

for ( var id in fb ) {
    for ( var prop in fb[id]) {
        Ti.API.info('ID ' + id + ' property: ' +
                       prop + ' = ' + fb[id][prop]);
    }
}
