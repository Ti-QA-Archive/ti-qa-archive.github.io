    Hi Peter

    I think this question has gone unanswered for a few days because it looks like you are asking others to create your entire app.

    But I am going to point you in the right direction - to use your own words.

    Create a new default project and replace the contents of app.js with the code below.

    Ti.UI.setBackgroundColor('#000');

var win1 = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Tab 1'
});
var tab1 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 1',
    window: win1
});

var data = [
    {
        hex: '#EFDECD', 
        title: 'Almond',
        leftImage: 'KS_nav_ui.png',
        hasChild: true
    }, 
    {
        hex: '#CD9575', 
        title: 'Antique Brass', 
        leftImage: 'KS_nav_views.png',
        hasChild: true
    }, 
    {
        hex: '#FDD9B5', 
        title: 'Apricot', 
        leftImage: 'KS_nav_ui.png',
        hasChild: true
    }, 
    {
        hex: '#78DBE2', 
        title: 'Aquamarine', 
        leftImage: 'KS_nav_views.png',
        hasChild: true
    }, 
    {
        hex: '#87A96B', 
        title: 'Asparagus', 
        leftImage: 'KS_nav_ui.png',
        hasChild: true
    }
];

var tbl = Ti.UI.createTableView({
    data: data,
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
tbl.addEventListener('click', function (e) {
    var newWin = Ti.UI.createWindow({  
        backgroundColor: e.rowData.hex,
        title: e.rowData.title
    });
    var btn = Ti.UI.createButton({
        title: 'Press Me'
    });
    btn.addEventListener('click', function (e) {
        Ti.Platform.openURL('http://appcelerator.com/');
        // could also have loaded a webView and shown the url in that
    });
    newWin.add(btn);
    tab1.open(newWin, { animated: true });
});
win1.add(tbl);

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);  
tabGroup.open();

    This will create;

    • a simple single tab app
    • with a main window containing a simple table
    • this table contains some sample data
    • click any row and a new window is opened (slide)
    • this window uses two of the values from the sample data used in the table
    • a test button is placed in the centre of the window
    • click the button to open safari in iOS (or chrome in Android) on the appcelerator.com website

    You can extend this by creating custom tableViewRows, adding a webView to the second window to directly open the URL in iOS & Android and the PDF is iOS, but use intents for Android PDF viewing.

    This should give you an idea of just how simple this can be to connect various simple interface elements together.

    I hope this provides you with a sufficient answer.

    • This is the revised code to show you how, based on the original version.

      Ti.UI.setBackgroundColor('#000');

var win1 = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Tab 1'
});
var tab1 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 1',
    window: win1
});
function level3(obj) {
    var win = Ti.UI.createWindow({  
        backgroundColor: '#fff',
        title: obj.title
    });
    var web = Ti.UI.createWebView({
        height: Ti.UI.FILL,
        url: obj.url,
        width: Ti.UI.FILL
    });
    win.add(web);
    tab1.open(win, { animated: true });
}
function level2(obj) {
    var win = Ti.UI.createWindow({  
        backgroundColor: obj.hex,
        title: obj.title
    });
    var data = [
        {
            title: 'Apple',
            url: 'http://www.apple.com',
            hasChild: true
        }, 
        {
            title: 'Microsoft',
            url: 'http://www.microsoft.com',
            hasChild: true
        }, 
        {
            title: 'Appcelerator',
            url: 'http://www.appcelerator.com',
            hasChild: true
        }
    ];

    var tbl = Ti.UI.createTableView({
        data: data,
        height: Ti.UI.FILL,
        width: Ti.UI.FILL
    });
    tbl.addEventListener('click', function (e) {
        level3(e.rowData);
    });
    win.add(tbl);
    tab1.open(win, { animated: true });
}

function level1() {
    var data = [
        {
            hex: '#EFDECD',
