Adding more data to tableView

I am attempting to create a table that gets 3 pieces of data from a database and display them in a row like this:

  • Name
  • Address
  • Phone
  • Name
  • Address
  • Phone

I am not having a problem getting a single item (like the Name) to show up in the tables rows, but not sure how to add more items to each row. Here is what I have done so far that doesn't work:

var win = Ti.UI.currentWindow;
    win.barColor = '#7b4336';
var tableview = Ti.UI.createTableView({
    top:90,
    left:80,
    right:80,
    bottom:120,
    backgroundColor:'transparent'
});
    var db = Ti.Database.install('bcapool3.sqlite','distributor');
    var rows = db.execute('SELECT * FROM distributor WHERE classification="f"');
    var data = [];
    for (var i=rows.length-1; i>=0;i--){
        var row=Ti.UI.createTableViewRow();
        var title=Ti.UI.createLabel({
            text:'' + rows.fieldByName('distributor_name') + '',
            font:{fontSize:10},
            left:90,
            top:2,
            height:14,
            hasChild:false
        });
        var address=Ti.UI.createLbel({
            text:'' + rows.fieldByName('distributor_address') + '', 
            font:{fontSize:10},
            left:90,
            top:16,
            height:14,
            hasChild:false
        });
        var phone=Ti.UI.createLabel({
            text:'' + rows.fieldByName('distributor_phone') + '', 
            font:{fontSize:10},
            left:90,
            bottom:0,
            height:14,
            hasChild:false
        });
    row.add(title);
    row.add(address);
    row.add(phone);
    data.push(row);
    };            
tableview.setData(data);
win.add(tableview);

Would some be so kind to show me how I need to write this to achieve my goal?

  • It looks like you've placed all 3 labels 90 points from the left of the table row. That means all 3 labels are stacked on top of each other. Try setting the layout property of the row to horizontal and change the left value to something like 10.

    • Oh, you may need to add a wrapper view to the row and set the layout on the wrapper view instead. So you would add the labels to the wrapper view, then add the view to the row.
