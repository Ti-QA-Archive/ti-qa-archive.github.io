Android: dynamic inserting of map

In a multi-map-application I try to insert map dynamicly in window. But if I click an a annotation then the app crashes. Here my code:

tabGroup.addEventListener('focus', function(e) {
        if(e.tab.window == win2) {
        var mapoptions = {
            mapType : Ti.Map.HYBRID_TYPE,
            region : {
                latitude : 53.55,
                longitude : 10,
                latitudeDelta : .1,
                longitudeDelta : .1
            }
        };
        var map = Titanium.Map.createView(mapoptions);
        win2.add(map);
        Ti.include('/components/map/pois.json');
        var annotations = [];
        for(var i = 0; i < testpois.length; i++) {
            annotations.push(Ti.Map.createAnnotation({
                latitude : testpois[i].lat,
                longitude : testpois[i].lng,
                title : testpois[i].name
            }));
        }
        map.addAnnotations(annotations);
    };
});

On iPhone it works fine. It also crashes without title in annotation.
Here are testpois.json: http://pastie.org/3815598

Any idea to solve?

Rainer
