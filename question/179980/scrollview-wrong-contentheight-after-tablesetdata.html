ScrollView wrong contentHeight after table.setData

Hi, I want a tableview which does not scroll itself but is integrated in a scrollview. I am filling a tableView after the View is shown when the user clicks a button. But the scrollview does not adjust to the new height, I can not scroll down. What do I need to do?

<Alloy>
    <Window id="window"  title="Statistics">    
        <ScrollView id="scroll" contentHeight=" Ti.UI.SIZE">
            <View id="mainView" layout="vertical">
                <View id="tableParent" width="Ti.UI.SIZE" height="Ti.UI.SIZE" layout="vertical">
                    <Button title="Test" onClick="onClick"/>
                    <TableView id="table" width="200" height="Ti.UI.SIZE" scrollable="false">

                    </TableView>
                </View>
            </View>
        </ScrollView>
    </Window>
</Alloy>

var args = arguments[0] || {};

var tableData = new Array();

for (var i=0;i<20;i++) {
        var row=Ti.UI.createTableViewRow();
        row.title="Test";
        tableData.push(row);
}

var onClick = function(e) {
    $.table.setData(tableData);

};

             <View id="mainView" height="Ti.UI.SIZE" layout="vertical">
