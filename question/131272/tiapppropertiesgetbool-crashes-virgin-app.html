  • You shouldn't have to try/catch. If the element isn't in there then it'll just return null. I haven't tested this but in theory you should be able to just do:

    var value = Ti.App.Properties.getString(elem) || hint;

    If that doesn't work then worst case this:

    var value = Ti.App.Properties.getString(elem) != null ? Ti.App.Properties.getString(elem) : hint;

  • Now it works without crash:

    api.getElem = function(elem, hint, label) {
    var container = Ti.UI.createView({
        layout : 'vertical',
        height : 'auto',
        top : 10
    });
    var label = Titanium.UI.createLabel({
        text : label,
        height : 'auto'
    });
    container.add(label);
    var value;
    try {
        value = Ti.App.Properties.getString(elem)
    } catch (E) {
        value = hint;
    }
    var field = Ti.UI.createTextField({
        height : 36,
        left : 0,
        top : 2,
        width : '100%',
        borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
        value : value,
        hintText : hint
    });
    container.add(field);
    field.addEventListener('change', function() {
        Ti.App.Properties.setString(elem, field.value);
    });
// from geolocation:
    Ti.App.addEventListener(elem + '_ready', function(e) {
        field.value = e.value
        Ti.App.Properties.setString(elem, field.value);
    });
    return container;
};
