    Hi Tsvetan,

    you can try in this way,

    $.mapview.addEventListener('click', function(e) {

 if (e.clicksource == 'title' || e.clicksource == 'subtitle' ) {
         Ti.API.info(e.annotation.title);
         Ti.API.info(e.annotation.subtitles);
    }
});
