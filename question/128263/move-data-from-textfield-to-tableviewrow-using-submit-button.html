  • I got the features functioning the way I wanted except when a new row is appended it's added underneath the previous row. Is there any way to have it append above the previous row?

    Here's the code:

    var win = Ti.UI.currentWindow;
win.title = 'Notes';
win.backgroundImage = '../images/VGbackground.png';
win.barColor = '#800000';
win.tabBarHidden = 'true';

var notesField = Ti.UI.createTextField({
    top:15,
    width:250,
    height:35,
    hintText:'Enter Notes',
    clearOnEdit:'true',
    borderStyle:Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    clearButtonMode:Ti.UI.INPUT_BUTTONMODE_ONFOCUS
});
win.add(notesField);

var submit = Ti.UI.createButton({
    top:60,
    left:210,
    width:75,
    height:30,
    title:'Submit',
    font:{fontFamily:'Trebuchet MS', fontWeight:'bold'},
    color:'#800000'
});
win.add(submit);

    submit.addEventListener('click', function(){
        var row = Ti.UI.createTableViewRow({height:'auto'});
        var label = Ti.UI.createLabel({text:notesField.value, color:'#111', width:'auto', height:'auto',left:7});
        row.add(label);
        tableView.appendRow(row);

        notesField.blur();
    });

var tableView = Ti.UI.createTableView({
    top:100,
    width:295,
    height:305,
    backgroundColor:'white',
    borderRadius:5
});

win.add(tableView);
