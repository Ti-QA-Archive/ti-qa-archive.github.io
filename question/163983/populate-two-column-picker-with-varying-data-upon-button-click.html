function dataLoad(data,column)
{
    for (var x = 0; x < data.length; x++)
    {
        column.addRow(Ti.UI.createPickerRow({
            title:data[x].title, 
            val:data[x].val
        }));
    };
    var actualRow=column.rowCount-data.length;
    for (var i=0; i < actualRow; i++) {
      column.removeRow(column.rows[i]);
    };

}
