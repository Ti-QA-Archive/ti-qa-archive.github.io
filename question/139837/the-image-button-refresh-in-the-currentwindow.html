function setImages(imgArr) {

    });

    var y = 10;
    for(var i = 0; i < imgArr.length; i++) {

        if(i != 0) {
            if(i % 3 == 0) {
                x = 20;
                y = y + 100;
            }
        }

        var imgView = Ti.UI.createImageView({
            top : y,
            left : x,
            width : 80,

//---------------------------------------------------------------------GET FEED
function getImages(url) {
    Ti.App.fireEvent('show_indicator');
    var xhr = Ti.Network.createHTTPClient();
    // example RSS feed

    xhr.setTimeout(120000);

    xhr.onerror = function(e) {
        alert('Internet connection error,\n The Internet connection appears to be offline');
    }

    xhr.onload = function() {
        try {
            var doc = this.responseXML.documentElement;
            var items = doc.getElementsByTagName("item");

            var imgArr = [];

            for(var c = 0; c < items.length; c++) {
                var item = items.item(c);
                var thumbnails = item.getElementsByTagName("thumbnail");

                if(thumbnails && thumbnails.length > 0) {
                    var thumbnail = thumbnails.item(0).getAttribute("url");

                    var larges = item.getElementsByTagName("image");
                    var large = larges.item(0).getAttribute("url");

                    var desc = item.getElementsByTagName("description").item(0).text;
                    var title = '';
                    if(item.getElementsByTagName("title").item(0)) {
                        title = item.getElementsByTagName("title").item(0).text;
                    }
                    var obj = {
                        thumb : thumbnail,
                        large : large,
                        desc : desc,
                        title : title
                    }

                    imgArr.push(obj);
                }
            }
            if(imgArr.length > 0) {
                if(view != null) {
                    var v = scrollView.children[0];
                    scrollView.remove(v);
                    view = null;
                }
                setImages(imgArr);
            } else {
                alert('No Images available');
                Ti.App.fireEvent('hide_indicator');
            }

        } catch (E) {
            alert(E);
            Ti.App.fireEvent('hide_indicator');
        }

    };
    xhr.open("GET", url);
    xhr.send();
}//end function getRSS

function checkNetwork(url) {
    if(Ti.Network.online == true) {
        getImages(url);
    } else {
        alert('Internet connection error,\n The Internet connection appears to be offline');
    }
}
