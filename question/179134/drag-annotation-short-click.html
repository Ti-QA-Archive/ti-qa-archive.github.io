  • Hello Eduardo,

    Try the below sample code. It's not that long to move the annotation. Just press on the pin "half of a second" (not on the top) to drag the annotation. This is as fast as it gets. 

    var MapModule = require('ti.map');
var win = Ti.UI.createWindow({backgroundColor: 'white'});
var opera = MapModule.createAnnotation({
    latitude: -33.8569,
    longitude: 151.2153,
    draggable: true,
    centerOffset: {x: 80, y: 25},
    image: 'SydneyOperaHouse.jpg',
    title: 'Sydney Opera House',
    subtitle: 'Sydney, New South Wales, Australia',    
    rightButton: Ti.UI.iPhone.SystemButton.CONTACT_ADD
});
var mapview = MapModule.createView({
    mapType: MapModule.NORMAL_TYPE,
    region: {latitude: -33.87365, longitude: 151.20689, latitudeDelta: 0.1, longitudeDelta: 0.1 },
    annotations: [opera]
});
win.add(mapview);
win.open();

    First, add "ti.map" module to your "tiapp.xml" file.

    Hope this helps. Thanks.
