Vertical text in TableView Header does not layout correctly

Hi,
I want the text in the TableView Header to be written vertically. But I have some strange layout behaviour:

  • If the text gets too long like for label1 'Label1_long' then the layout of the left property is not correct any more
  • If I use the bottom property like for label2 then the label is put too much down.

What is the correct code to have the header with vertical text?

Alex

var tr = Ti.UI.create2DMatrix();
tr = tr.rotate(270);

var testTableData=new Array();

var section = Ti.UI.createTableViewSection();
var headerView=$.UI.create('View',{
    classes: 'headerView',
    width:Ti.UI.SIZE,
    height:150,
});

var label1= $.UI.create('Label',{    
    text : 'Label1_long',
    left:10,
    width:Ti.UI.SIZE,
    height:Ti.UI.SIZE,
    transform : tr,
});
headerView.add(label1);

var label2= $.UI.create('Label',{    
    text : 'Label2',
    left:40,
    bottom:2,
    width:Ti.UI.SIZE,
    height:Ti.UI.SIZE,
    transform : tr,
});
headerView.add(label2);

var label3= $.UI.create('Label',{    
    text : 'Label3',
    left:70,
    width:Ti.UI.SIZE,
    height:Ti.UI.SIZE,
    transform : tr,
});
headerView.add(label3);



section.headerView=headerView;
testTableData.push(section);

var tv=$.UI.create('TableView',{    
        width:Ti.UI.SIZE,
        height:Ti.UI.SIZE,
});

$.table.setData(testTableData);

    I figured out that rotating the complete view solves the problem:

    var tr = Ti.UI.create2DMatrix();
tr = tr.rotate(270);

var testTableData=new Array();

var section = Ti.UI.createTableViewSection();
var headerView=$.UI.create('View',{
    classes: 'headerView',
    width:Ti.UI.SIZE,
    height:150,
    transform : tr,
});

var label1= $.UI.create('Label',{    
    text : 'Label1_long',
    top:10,
    width:Ti.UI.SIZE,
    height:Ti.UI.SIZE,
});
headerView.add(label1);

var label2= $.UI.create('Label',{    
    text : 'Label2',
    top:40,
    left:1,
    width:Ti.UI.SIZE,
    height:Ti.UI.SIZE,
});
headerView.add(label2);

var label3= $.UI.create('Label',{    
    text : 'Label3',
    top:70,
    width:Ti.UI.SIZE,
    height:Ti.UI.SIZE,
});
headerView.add(label3);



section.headerView=headerView;
testTableData.push(section);

var tv=$.UI.create('TableView',{    
        width:Ti.UI.SIZE,
        height:Ti.UI.SIZE,
});

$.table.setData(testTableData);
