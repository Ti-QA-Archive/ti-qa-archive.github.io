  • Here's an example, change limit to whatever you need:

    var win = Ti.UI.createWindow({
    backgroundColor:'#ededed'
});

var limit = 10;  //set to 240 or whatever you like

var characters = Ti.UI.createLabel({
    text:'0',
    height:40,
    left:20,
    width:100,
    top:10
});
win.add(characters);

var text = Ti.UI.createTextArea({
    value:'',
    left:20,
    right:20,
    height:60,
    top:50,
    borderWidth:1,
    borderRadius:4
});

win.add(text);

text.addEventListener('change', function(){
    if(text.value.length > limit){
        text.value = text.value.substr(0, limit); //trim text to character limit
        alert('limit reached: ' + limit);    
    }

    characters.text = text.value.length;
});

win.open();
