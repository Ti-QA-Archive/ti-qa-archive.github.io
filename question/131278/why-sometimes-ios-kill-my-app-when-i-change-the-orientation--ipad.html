Why sometimes ios kill my app when i change the orientation ? (ipad)

Hello,

I trying to make an app that needs to recreate it's user interface on orientation change. It works but sometimes the ipad is killing my app while the ios simulator don't do it …

var test = {};

test.isChangingOrientation = true;

Ti.include('ui/orientation.js');

test.orientationMode = test.getOrientationMode(Ti.Gesture.getOrientation());

if(test.orientationMode == 'portrait') {
    Ti.include('/ui/portrait/ui.js');
    test.ui.createApplicationWindow();
    test.isChangingOrientation = false;
} else if(test.orientationMode == 'landscape') {
    Ti.include('/ui/landscape/ui.js');
    test.ui.createApplicationWindow();
    test.isChangingOrientation = false;
}

Ti.Gesture.addEventListener('orientationchange', function(e) {
    if(test.isChangingOrientation === false) {
        var o = test.getOrientationMode(e.orientation);

        if(o != 'unknown') {
            if(o != test.orientationMode) {
                test.isChangingOrientation = true;
                test.orientationMode = o;
                if(test.orientationMode == 'portrait') {
test.ui.mainWindow.close();
Ti.include('/ui/portrait/ui.js');
test.ui.createApplicationWindow();
                }
                if(test.orientationMode == 'landscape') {
test.ui.mainWindow.close();
Ti.include('/ui/landscape/ui.js');
test.ui.createApplicationWindow();
                }
                test.isChangingOrientation = false;
            }
        }
    }
});

Any help is appreciated :)

  • Once you post the log then we can help you. Also, are you sure that you need to reconstruct the layout on orientation change? This is a messy route to go down. If you don't hard-code widths and use left: 0, right: 0, then you can generally handle orientation changes automatically. Also, the splitview control handles orientation changes nicely, if it fits in your requirements to use it.
