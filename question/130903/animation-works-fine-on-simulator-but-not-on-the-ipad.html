animation works fine on simulator but not on the iPad

Hi all , i am using the below code to show an image moving left and right when it is clicked . It worked fine in the simulator but doesn't move in the iPad upon clicking . Any idea?

var left_logo = Titanium.UI.createImageView({
   image : 'left_logo.png',
   left : 25,
   height : 84,
   bottom : 20,        
   width : 130

});


left_logo.addEventListener('click', function()
{
   var count = 0;
   stopped = false;
   lnextAnimation = null;

   var a1 = Titanium.UI.createAnimation();
   a1.left = 38;
   a1.duration = 200;

   var a2 = Titanium.UI.createAnimation();
   a2.left = 15;
   a2.duration = 200;

   var a3 = Titanium.UI.createAnimation();
   a3.left = 25;
   a3.duration = 200;

   if (nextAnimation == null) {
       lnextAnimation = a1;
   }
   left_logo.animate(lnextAnimation);


   a1.addEventListener('complete', function()
   {
       if (!stopped)
       {

           left_logo.animate(lnextAnimation);
           lnextAnimation = a2;

           if (count == 7) {
               stopped = true;    
               left_logo.animate(a3);
           }
       count++;            
       }

   });

   a2.addEventListener('complete', function()
   {
       if (!stopped)
       {
           left_logo.animate(lnextAnimation);
           lnextAnimation = a1;

       }
   });


});
