constructor, destroy, off, clean up doubt (alloy)

Some questions/doubts above.

The views that I use in my Alloy project are made as follows:

var args = arguments || [0];

var data = {
    active:true,
    selected:false,
    click:function() {
    }
};

$.title.text = args.title;

var properties = {};

$.box.addEventListener('click',data.click);

$.box.destructor = function() {

    $.destroy();

    $.off();

    $.box.removeEventListener('click',data.click);

    data = properties = args = null;
};

When I want to clear that view I do the follow:

//create the view
var view = Alloy.createController('box',{
    title:'exemple'
}).getView();

$.index.add(view);

//destroy the view
view.destructor();

$.index.remove(view);
