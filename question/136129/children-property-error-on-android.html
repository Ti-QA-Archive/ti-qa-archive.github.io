      function StreamView() {
    var _ = require('/lib/underscore'), theme = require('/ui/theme'), ui = require('/ui/components');
    var url = 'http://ayaraglobal.com/dbs/api/terbaru';
    var self = new ui.View();

    var rowData = [];
    // Create our HTTP Client and name it "loader"
    var loader = Titanium.Network.createHTTPClient();
    // Sets the HTTP request method, and the URL to get data from
    loader.open("GET", url);
    // Runs the function when the data is ready for us to process
    loader.onload = function() {
        var tweets = eval('(' + this.responseText + ')');
        for(var i = 0; i < tweets.count; i++) {
            var tweet = tweets.posts[i].fulltext;
            // The tweet message
            var user = tweets.posts[i].title;
            // The screen name of the user
            var avatar = tweets.posts[i].filePath;
            // The profile image
            // Create a row and set its height to auto
            var row = Titanium.UI.createTableViewRow({
                height : 'auto',
                backgroundColor : '#fff',
                backgroundSelectedColor : '#ccc',
                hasChild : true,
            });

            // Create the view that will contain the text and avatar
            var post_view = Titanium.UI.createView({
                height : 'auto',
                layout : 'vertical',
                top : 5,
                right : 5,
                bottom : 5,
                left : 5
            });
            // Create image view to hold profile pic
            var av_image = Titanium.UI.createImageView({
                image : avatar, // the image for the image view
                top : 0,
                left : 0,
                height : 60,
                width : 60
            });
            post_view.add(av_image);
            var width_gambar = 80;
            var width_lat = Ti.Platform.displayCaps.platformWidth;
            var yyyyy = width_lat - width_gambar;
            var user_lbl = Titanium.UI.createLabel({
                text : user,
                left : 65,
                width : yyyyy,
                top : -60,
                bottom : 2,
                height : 16,
                textAlign : 'left',
                color : '#000',
                font : {
                    fontFamily : 'Trebuchet MS',
                    fontSize : 14,
                    fontWeight : 'bold'

                }
            });
            post_view.add(user_lbl);
            // Create the label to hold the tweet message
            var tweet_lbl = Titanium.UI.createLabel({
                text : tweet,
                left : 65,
                top : 0,
                bottom : 2,
                height : 50,
                width : yyyyy,
                textAlign : 'left',
                color : '#000',
                font : {
                    fontSize : 14
                }
            });
            post_view.add(tweet_lbl);
            var xWindow = Ti.UI.createWindow({
                modal : true,
                backgroundColor : '#fff',

            });

            post_view.addEventListener('click', function(e) {

                //alert(Ti.Platform.displayCaps.platformWidth);
                //var view = row.children[0].children[1].text;
                var view = e.row.tweet_lbl;
                alert(view);
                //var view2 = row.children[0].children[2].text;
                //var innerHTML = '<html><body style="width:"' + width_lat + ';"><h1>' + view + '</h1>' + view2 + '</body></html>';
                //var y = Ti.UI.createWebView({
                //html : innerHTML,
                //});

                xWindow.add(y);
                xWindow.open();
            });
            // Add the post view to the row
            row.add(post_view);
            // Give each row a class name
            row.className = "item" + i;
            // Add row to the rowData array
            rowData[i] = row;
        }
        // Create the table view and set its data source to "rowData" array
        var tableView = Titanium.UI.createTableView({
            data : rowData
        });
        //Add the table view to the window
        self.add(tableView);
    };
    // Send the HTTP request
    loader.send();
    return self;
}

module.exports = StreamView;
