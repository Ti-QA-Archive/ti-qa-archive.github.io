  • You'd be better off creating a showIndicator and hideIndicator function in app.js and then create two listeners for a fireEvent to show and hide them.

    Then you call showIndicator before an activity and hide it as the new window opens.

    You could also send different 'messages' to the event to display different text.

    I do this in most of my apps to show activity until something has completed.

  • Try this (as explained above) :

    
///////////////////
// app.js

var indWin;
var actInd;
var indicatorOn = false;

function hideIndicator(){
    if(indicatorOn==true){
        actInd.hide();
        // fade it off.. sweet!
        indWin.close({opacity:0,duration:200});
        indicatorOn = false;
    }
}

function showIndicator(title_msg,sub_msg){
    if(indicatorOn == false){
        // window container

        // ### make a 'loading_bg.png' file which should be a semi-transparent png file which is the same size as the whole screen
        indWin = Titanium.UI.createWindow({
            top:0,
            height:480,
            width:320,
            backgroundImage:'loading_bg.png'
        }); 

        // make a bar image (you dont have to do this bit, but I do, to place message on)
        var indView = Titanium.UI.createImageView({
            image: 'loading_bar.png',
            top: 190,
            left: 0,
            width:320,
            height:136
        });
        indWin.add(indView);
        // loading indicator
        actInd = Titanium.UI.createActivityIndicator({
            style:Titanium.UI.iPhone.ActivityIndicatorStyle.BIG,
            top:30,
            left:0,
            height:60,
            width:60
        });
        indView.add(actInd);        
        // message
        var message = Titanium.UI.createLabel({
            text:title_msg,
            color:'#fff',
            font:{fontSize:32,fontWeight:'bold'},
            textAlign:'left',
            top:36,
            height:38,
            left:55,
            width:220
        });
        var message2 = Titanium.UI.createLabel({
            text:sub_msg,
            color:'#000',
            font:{fontSize:14,fontWeight:'bold'},
            textAlign:'left',
            top:70,
            height:16,
            left:56,
            width:220
        });
        indView.add(message);
        indView.add(message2);
        indWin.open();
        actInd.show();
        indicatorOn = true;

    }
}

// set up some custom events to show and hide the indicator

Titanium.App.addEventListener('show_indicator', function(e){
    Ti.API.info("SHOWING INDICATOR :"+e.title_msg);
    showIndicator(e.title_msg,e.sub_msg);
});
Titanium.App.addEventListener('hide_indicator', function(e){
    Ti.API.info("HIDING INDICATOR");
    hideIndicator();
});

// ######## Usage:

// To show it
showIndicator('Loading...','hang on a moment');
// To hide it
hideIndicator();

// ## or from elsewhere in your app:

// To show it
Ti.App.fireEvent('show_indicator',{title_msg:'Loading...',sub_msg:'hang on a moment'});
// To hide it
Ti.App.fireEvent('hide_indicator');
