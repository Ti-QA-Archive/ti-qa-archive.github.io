  • problem solved: i add sql command to create that table if dont exis. the app cant find the path to sql.sqlite file i thnk.

    function open_db_gizmo(){
    var db = Titanium.Database.open('twitter_gizmo');
    if(Ti.UI.platform = 'android'){
        db.execute('CREATE TABLE IF NOT EXISTS "tweets" ("id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL UNIQUE , "name" TEXT NOT NULL , "screenname" TEXT, "image" , "tweet" )');
    }
    return db;
}
