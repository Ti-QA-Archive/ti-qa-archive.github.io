    Here is my 2c. The DOC help about it is here: Titanium.Process-module.
    My code is here:

    /**
     * Run an external program and pass the response to the callback function
     * @param {array} aCommand Command to run
     * @param {function} fCalback Callback function
     * @param {bool} bSync True if the command must run sync
     */
    runCommand: function(aCommand, fCallback, bSync){
        var proc = Ti.Process.createProcess(aCommand);

        if (proc) {
            if (bSync) {
                if (typeof fCallback == 'function')
                    fCallback(proc().toString());
            } else {
                if (typeof fCallback == 'function')
                    proc.setOnRead(function(r){
                        fCallback(r.data.toString());
                    });
                proc.launch();
            }
        }
    }
