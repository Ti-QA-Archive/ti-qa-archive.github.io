    I can't reproduce this. You may be doing something different, but you didn't provide any of that information. Here's a test case meeting your description that works just fine.

    index.xml

    <Alloy>
    <Window>
        <ListView>
            <Templates>
                <ItemTemplate name="template2" height="50">
                    <Label bindId="label" class="labels"/>
                </ItemTemplate>
            </Templates>
            <ListSection id="section"/>
        </ListView>
    </Window>
</Alloy>

    index.js

    var dataItem = [];
for (var i = 0; i < 2; i++) {
    dataItem.push({
        template: 'template2',
        label: {
            text: 'row ' + (i+1),
            color: '#' + Math.floor(Math.random()*16777215).toString(16),
            textAlign: 'center',
            width: Ti.UI.FILL
        }
    });
}

$.section.items = dataItem;

$.index.open();

    index.tss

    'Label': {
    width: Ti.UI.SIZE,
    height: Ti.UI.SIZE,
    color: '#000'
}

'.labels': {
    color: '#a00'
}

'#index': {
    backgroundColor: 'white'
}
