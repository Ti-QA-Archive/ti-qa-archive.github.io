  • Try this :

    //index.js

    var UI = require("ui");

//Add Loding indicator
var loading = UI.createActivityIndicator();
$.index.add(loading);

//Remove
$.index.remove(loading);

    //ui.js

    exports.createActivityIndicator = function(message, visible) {
    var activityView = Ti.UI.createView({
        top: 0,
        height: "100%",
        zIndex: "10",
        backgroundImage: '/transparent.png'
    });

    var Indicator = Ti.UI.createActivityIndicator({
        message: 'Loading...',
        color: 'white'
    });

    activityView.add(Indicator);
    Indicator.show();
    return activityView;
};
