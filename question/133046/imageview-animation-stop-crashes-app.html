  • For my View

    var btnRefresh = Ti.UI.createButton({
    backgroundImage:'images/navbarButton.png',
    backgroundSelectedImage:'images/navbarButton_over.png',
    width:40,height:30,
    right: 6,
    top: 7,
    backgroundLeftCap: 2,
});

var btnRefreshImage = Ti.UI.createImageView({
    images: [
        'images/refresh.png',
        'images/refresh2.png',
        'images/refresh3.png',
        'images/refresh4.png',
        'images/refresh5.png',
        'images/refresh6.png',
        'images/refresh7.png',
        'images/refresh8.png'
    ],
    width: 14, height: 15,
    top: 5,
});

btnRefresh.add(btnRefreshImage);

win.setRightNavButton(btnRefresh);

    The Event Listener 

    btnRefresh.addEventListener('click', function() {
    btnRefreshImage.start();

    setTimeout(btnRefreshImage.stop(), 3000);
});
