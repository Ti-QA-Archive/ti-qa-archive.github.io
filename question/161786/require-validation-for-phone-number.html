    Hello Davidson,

    Try this one:

    function checkNumber(no) {
    return !no.match(/[^0-9+]/g);
}

if(checkNumber(handphoneField.value.trim()) == false  && handphoneField.value.trim() !=8)
{
    alert("Error : the handphone number is not vadil");
}
else
{
    alert("the handphone number is 8 digit");
}
