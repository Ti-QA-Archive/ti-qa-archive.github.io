  • What happens if you add the navigationGroup into a splitView?

    var win = Ti.UI.createWindow();

var nav = Ti.UI.iPhone.createNavigationGroup({
   window: win
});

var splitwin = Ti.UI.iPad.createSplitWindow({
    detailView: nav,
    masterView: masterView
});

splitwin.addEventListener('visible',function(e){
    if (e.view == 'detail'){
        e.button.title = "Master";
        win.leftNavButton = e.button;
    } else if (e.view == 'master'){
        win.leftNavButton = null;
    }
});

splitwin.open();

  • Try …

    var wrapper = Ti.UI.createWindow();

var firstWindow = Ti.UI.createWindow();

var secondWindow = Ti.UI.createWindow();

var openSecondWindowButton = Ti.UI.createButton({
  title: 'Open Second Window',
  width: 300,
  height: 50
});
openSecondWindowButton.addEventListener('click', function() {
  nav.open(secondWindow);
});

firstWindow.add(openSecondWindowButton);

var nav = Ti.UI.iPhone.createNavigationGroup({
   window: firstWindow
});

wrapper.add(nav);
wrapper.open();
