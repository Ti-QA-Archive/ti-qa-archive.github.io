    Hello Douglas,

    I tested your code and got 10 errors like this in logcat:

    06-25 17:47:24.816: E/TableViewProxy(3758): (KrollRuntimeThread) [871,1787] Unable to insert section. Index out of range. Non-existent row at 0

    The cause is this part in your code:

    tableView.insertSectionAfter( 0, sectionJaneiro )
tableView.insertSectionAfter( 1, sectionFevereiro )
tableView.insertSectionAfter( 2, sectionMarco )
tableView.insertSectionAfter( 3, sectionAbril )
tableView.insertSectionAfter( 4, sectionJunho )
tableView.insertSectionAfter( 5, sectionJulho )
tableView.insertSectionAfter( 6, sectionAgosto )
tableView.insertSectionAfter( 7, sectionSetembro )
tableView.insertSectionAfter( 8, sectionOutubro )
tableView.insertSectionAfter( 9, sectionNovembro )
tableView.insertSectionAfter( 10, sectionDezembro )

    You are trying to insert a section after a non-existing section, ten times. A quick fix to this would setting the TableView data as an array of sections.

    var sections = [
    sectionJaneiro, 
    sectionFevereiro, 
    sectionMarco, 
    sectionAbril, 
    sectionJunho, 
    sectionJulho, 
    sectionAgosto, 
    sectionSetembro, 
    sectionOutubro, 
    sectionNovembro, 
    sectionDezembro
];

tableView.setData(sections);

    In general, I must say, that you write a lot of redundant code, like the creation of your sections. You could do this in one single function.

    Also, please use semicolons! :-)

        var dialog = Ti.UI.createAlertDialog({
            message: 'Aconteceu algum erro ao tentar conectar. Por favor verifique sua conexão ou tente novamente mais tarde.',
            ok: 'Ok',
            title: 'Ooops'
          }).show()

        loading.hide()
        win.remove( viewLoading )
        win.remove( rectPreto ) 
    },

    onload:function(){
        var result = JSON.parse( this.responseText )


        populaTableView( result.eventos )

        loading.hide()
        win.remove( viewLoading )
        win.remove( rectPreto )
    }
})
