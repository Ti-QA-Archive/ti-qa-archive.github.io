  • Here you go make a navgroup for adding the back button in titanium

    var win1 = Titanium.UI.createWindow();

var win2 = Titanium.UI.createWindow({
    backgroundColor: 'red',
    title: 'Red Window'
});

var win3 = Titanium.UI.createWindow({
    backgroundColor: 'blue',
    title: 'Blue Window'
});

var button = Titanium.UI.createButton({
    title: 'Open Blue Window'
});
button.addEventListener('click', function(){
    nav.open(win3, {animated:true});
});

var nav = Titanium.UI.iPhone.createNavigationGroup({
   window: win2
});

win2.add(button);
win1.add(nav);
win1.open();

  • Is your Window set up as a navigationGroup or modalWindow? This has to be set for you to get the default navBar at the top. Can you try the below snippet and see if it works?

    var win = Ti.UI.createWindow({modal:true});

var back = Ti.UI.createButton({
    title:'Back'
});

back.addEventListener("click", function() {
    win.close();
});

win.leftNavButton = back;

win.open();
