Labels change color when touching them without telling them to do so (bastards)

Hello,

I am making a list of items. The collection of items are all put in a scrollView, and each item consists of a view which is filled with 3 labels. All works fine and as expected, but for one thing. When i touch (click) a label in the list of items, it highlights (color changes from white to grey), as if the touch/click event would be implemented. I have not implemented any event, so why the color changes from white to grey remains a mistery. I would like to know what the easiest way to fix this is, and if it is common knowledge, why labels have a default behaviour for onClick events.

I am testing this on a Samsung Galaxy Ace, and my code is the following:

var win = Ti.UI.createWindow
(
    {
        exitOnClose: true,
        navBarHidden: true
    }
);
win.open();

var body = Ti.UI.createScrollView
(
    {
        width:'100%',
        height:'100%',
        layout: 'vertical'
    }
)

win.add(body);

for(var i = 0; i < 10; i++)
{
    AddItem("name" + i, "Description blablablablabla lskjf asldfkj asdlkjf askdlfj as s fsdj lsdk ja s d" + i, i + "-10-2012")
}

function AddItem(name, description, date)
{
    var itemView = Ti.UI.createView
    (
        {
            width:'100%',
            height: Ti.UI.SIZE,
            top: 5,
            backgroundColor: "#222222"        
        }
    )

    var nameLabel = Ti.UI.createLabel
    (
        {
            text:name,
            font:{fontSize:16},
            top: 2,
            left: 10
        }
    )

    itemView.add(nameLabel);

    var descriptionLabel = Ti.UI.createLabel
    (
        {
            text: description,
            font:{fontSize:14},
            top: 22,
            left: 10,
            right: 10,
            bottom: 2
        }
    )

    itemView.add(descriptionLabel);

    var dateLabel = Ti.UI.createLabel
    (
        {
            text: date,
            font:{fontSize:14},
            top: 2,
            right: 10
        }
    )

    itemView.add(dateLabel);

    body.add(itemView);
}
