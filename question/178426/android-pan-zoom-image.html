android pan zoom image

Hi, I'd like to pinch and zoom an image in android, I tried in this way and it works fine, but i have also to pan left/right/up/down the image when its zoomed. Anyone know if is it possibile?
`
var view = Ti.UI.createImageView({
image: helper.pathImage("tariffe.png"),
enableZoomControls: false,
touchEnabled: true,
autorotate: true,
top: "48dp"
});

var baseHeight = h-60;
var baseWidth = w;
var minSize = 250;

view.height = baseHeight;
view.width = baseWidth;

view.addEventListener('pinch', function(e) {
    if (baseHeight * e.scale >= minSize){
        view.height = baseHeight * e.scale;
        view.width = baseWidth * e.scale;
    }
});

view.addEventListener('touchstart', function(e) {
    baseHeight = view.height;
    baseWidth = view.width;
});

`
