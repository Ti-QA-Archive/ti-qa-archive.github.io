How to return the root window of a navigationGroup ?

I have a navigationGroup with a tableView. When I click on a tableViewRow, I open a new childWindow, so that I don't know exact the number of childWindows. When I reopen this navigationGroup, I want it show root window. How do I remove all chilWindows of navigation and goback the root window ?

var first = Ti.UI.createWindow({
  backgroundColor:"#fff",
  title:"Root Window"
});

var navGroup = Ti.UI.iPhone.createNavigationGroup({
  window:first
});

var bindingData = function (_data, _winParent) {
    var i;
    var _tableView = Titanium.UI.createTableView();
    _winParent.add(_tableView);
    for (i = 0; i < _data.length; i++) {
        var _row = Titanium.UI.createTableViewRow({ title: _data[i].title });
        _tableView.appendRow(_row);

        _row.addEventListener('click', function (e) {
            var childWin = Ti.UI.createWindow({
                backgroundColor: "#fff",
                title: _data[i].title + "Window"
            });
            bindingData(_data[i].children, childWin);
            navGroup.open(childWin);
        });
    }
};

bindingData(data,first);
navGroup.open();
