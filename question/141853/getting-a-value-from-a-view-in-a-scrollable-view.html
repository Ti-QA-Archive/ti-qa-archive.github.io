Getting a value from a view in a scrollable view

From the code below I'm trying to get the offer_view.infoText value from each view so that when the user uses the options menu on android they get to see information.

I know that offersView.currentPage will give me the current page but can't workout how to drill down further in the actual view.

var createOfferView = function(item) {

    var offer_view = Ti.UI.createView();

    var offer_image = Ti.UI.createImageView({
        image : item.image
    });

    offer_view.add(offer_image);

    offer_view.infoText = item.info;

    return offer_view;
}


function offersWindow() {

    //create object instance
    var self = Ti.UI.createWindow({
        backgroundColor : '#C23A25',
        navBarHidden : true,
        title : 'Offers',
        exitOnClose : true,
    });

    var viewData;
    var offersView = Ti.UI.createScrollableView({
        showPagingControl : true
    });

    self.add(offersView);

    var activity = self.activity;
    activity.onCreateOptionsMenu = function(e){
      var menu = e.menu;
      var menuItem = menu.add({ title: "More Info" });
      menuItem.addEventListener("click", function(e) {
        alert(offersView.currentPage.offer_view.infoText);
      });
    };

    return self;
};

module.exports = offersWindow;

    The way I go to find this its small steps at a time. Do 

    alert(offersView.currentPage.children);

    then for each children you do .infoText. example

    alert(offersView.currentPage.children[0].infoText);

    you might have to drill down a lil more b4 you get to the infoText property, but the first alert should give you good look of the structure
