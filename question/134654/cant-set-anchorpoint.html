Can't set anchorPoint

I want center the anchorPoint from an imageView so I can use this for other calculations. But when I set the anchorPoint and build the app, it seems that the anchorPoint is still in the left upper corner of the image.

var dPad = Titanium.UI.createImage
(
{
image:'images/hud/dpad.png',
height: 82.5,
width: 82.5,
left: 240,
top: 160,
anchorPoint: {x:0.5,y:0.5}
}
)

You can see the image have to be add exactly in the middle of the screen (iPhone) but the anchorPoint is now in the middle of the screen. Who can help me with this problem?

    Hi, Koen,

    Try this.

    var win = Titanium.UI.currentWindow;
var t = Ti.UI.create2DMatrix();
t = t.rotate(90);

var a = Titanium.UI.createAnimation();
a.transform = t;
a.duration = 1000;
a.autoreverse = true;

var view1 = Titanium.UI.createView({
    backgroundColor:'#336699',
    top:10,
    left:220,
    height:50,
    width:50,
    anchorPoint:{x:0,y:0}
});
win.add(view1);

var topLeft = Titanium.UI.createButton({
    title:'Top Left',
    height:40,
    width:200,
    top:10,
    left:10
});

topLeft.addEventListener('click', function()
{
    view1.animate(a);
});

win.add(topLeft);
win.open();
