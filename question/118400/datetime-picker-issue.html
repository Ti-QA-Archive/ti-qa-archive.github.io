Date/Time Picker ISSUE

i have this simple TextFeild that pop's up a DateTime Picker on focus ,(below is the code)

Titanium.UI.setBackgroundColor('#000');

// create tab group
var tabGroup = Titanium.UI.createTabGroup();


//
// create base UI tab and root window
//
var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});

var timeTxt = Titanium.UI.createTextField({
      color:'#336699',
      left:10,
       width:250,
       height:40,
       borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
       hintText:'Select date.',
       clearButtonMode:Titanium.UI.INPUT_BUTTONMODE_ONFOCUS
});
timeTxt.addEventListener('focus',function(e){
    Titanium.API.info('Triggered focus on date txtFeild');
    timeTxt.blur();

    //create a pop over
    var ddPO = Titanium.UI.iPad.createPopover({
        width: '320',
        height: 145,
        navBarHidden: true,
        arrowDirection: Titanium.UI.iPad.POPOVER_ARROW_DIRECTION_UP
    });

    var minDate = new Date();

    var maxDate = new Date();
    maxDate.setFullYear(2050);
    maxDate.setMonth(11);
    maxDate.setDate(31);

    var datePicker = Ti.UI.createPicker({
        type:Ti.UI.PICKER_TYPE_DATE_AND_TIME,
        minDate:minDate,
        maxDate:maxDate,
        value:minDate
    });
    datePicker.addEventListener('change',function(e){
        Titanium.API.info('E value = '+e.value);
        var d = e.value;
        Titanium.API.info('converted value = '+d);
        timeTxt.setValue(d);
    });
    ddPO.add(datePicker);

    // turn on the selection indicator (off by default)
    datePicker.selectionIndicator = true;

    ddPO.show({ view: timeTxt,animated: true});                
});    
win1.add(timeTxt);


//
//  add tabs
//
tabGroup.addTab(tab1);  


// open tab group
tabGroup.open();

the issue is that the DateTime selected on the picker shows the current locale DateTime but the TextFeild gets Updated with the GMT time.

But the logs show that e.value represents the Locale time … Am confused

can somebody help me on this issue …

  • I had a similar problem but solved it by using

    textField.setValue(picker.value.toLocaleDateString())

    this will show the date in the phone locale
