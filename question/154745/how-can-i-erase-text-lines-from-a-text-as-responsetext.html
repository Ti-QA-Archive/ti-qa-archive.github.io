             function cleanContent (text) {

            var cort = [];
            var str = '';

            cort = text.split ('\n');    

            if (cort.length > 2){

                 for(i = 2; i < cort.length-1 ; i++){

                    str = str +"\n"+ cort[i] ;
                }
            }

            return str;
        }
