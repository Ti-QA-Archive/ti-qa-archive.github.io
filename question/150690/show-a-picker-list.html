  • This works fine for me:

    
    function showPickerDialog ( _data, _callback) {

        var x = [];
        for (var n in _data) {
            x.push(Ti.UI.createTableViewRow({
                top : 10,
                bottom : 10,
                font :{ fontSize : '18dp' },
                title : _data[n],
                left : 20
            }));
        };

        var pickerDialog = Ti.UI.createAlertDialog({
            androidView : Ti.UI.createTableView({
                data : x
            })
        });
        x=null;

        pickerDialog.androidView.addEventListener('click', function(e) {
            _callback(e.rowData);
            pickerDialog.hide();
            pickerDialog = null;
        });

        pickerDialog.show();

    }
