        var json = JSON.parse('{"Message":"The request is invalid.","ModelState":{"":["Name chris is already taken."]}}');
    for (var i in json.ModelState) {
        Ti.API.info('key= ' + i + ' value= ' + json.ModelState[i]);
    }
