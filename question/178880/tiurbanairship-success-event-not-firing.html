Ti.Urbanairship SUCCESS event not firing

Application type: mobile
Titanium SDK: 3.2.3 GA
Platform & version: Android *

I am using the ti.urbanairship module with the code below. Seems to be registering the device fine because I can send pushes and they work (eventCallback method is called fine). but eventSuccess and eventError never get called. It is my understandard that eventSuccess should get called when the device is successfully registered with urbanairship? not sure why this isn't happening? Also, worth noting that showAppOnClick doesn't seem to work either.

function eventCallback(e) {
Ti.API.info(' Message: ' + e.message);
Ti.API.info(' Payload: ' + e.payload);
}

function eventSuccess(e) {
Ti.API.info('Received device token: ' + e.deviceToken);
}

function eventError(e) {
Ti.API.info('Error:' + e.error);
}

var Urbanairship = require("ti.urbanairship");

Urbanairship.addEventListener(Urbanairship.EVENT_URBAN_AIRSHIP_CALLBACK, eventCallback);
Urbanairship.addEventListener(Urbanairship.EVENT_URBAN_AIRSHIP_SUCCESS, eventSuccess);
Urbanairship.addEventListener(Urbanairship.EVENT_URBAN_AIRSHIP_ERROR, eventError);

Urbanairship.showAppOnClick = true;
Urbanairship.vibrateEnabled = true;
Urbanairship.soundEnabled = true;
Urbanairship.pushEnabled = true;

  • I worked out my issue.

    EVENT_URBAN_AIRSHIP_SUCCESS only occurs when the device registers for the first time.

    Also, showAppOnClick should be showOnAppClick (typo in the documentation).
