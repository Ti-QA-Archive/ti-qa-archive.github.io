Custom leftNavButton (Back-button)

Trying to make the "back" buttom get a custom color or bacground image.Can't get it to work. Any tips? Thanks in advance.

// create table view event listener
tableview.addEventListener('click', function(e)
{
    if (e.rowData.test)
    {
        var btnBack = new Ti.UI.createButton({
        backgroundImage:'images/sbg.png',
        backgroundSelectedImage:'images/sbg.png',
        width:80,height:25
});
        var win = Titanium.UI.createWindow({
            url:e.rowData.test,
            barImage:'../images/nav.png',
            leftNavButton:btnBack,
            title:e.rowData.title
        });
        Titanium.UI.currentTab.open(win,{animated:true});
    }
});


// add table view to the window
Titanium.UI.currentWindow.add(tableview);
