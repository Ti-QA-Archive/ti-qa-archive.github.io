    Your code has no problems, when I pasted it I found the click event is commented so I uncommented it.

    Each tab should have exactly one root window, it cannot be closed. You can close a window opened with Tab.open

    check the following code to understand more.

    
var tabGroup = Ti.UI.createTabGroup();
var myOnlyWin = Ti.UI.createWindow({
    backgroundColor : '#000',
    title : 'Page Title',
    tabBarHidden : true
});
var backBtn = Ti.UI.createButton({
    title : "Back",
    style : Titanium.UI.iPhone.SystemButtonStyle.BORDERED
});
myOnlyWin.leftNavButton = backBtn;
// BACK BUTTON LISTENER

var tab = Ti.UI.createTab({
    title : "NULL",
    window : myOnlyWin
});


var win2 = Ti.UI.createWindow({
    backgroundColor:'white'
});


tabGroup.addTab(tab);
tabGroup.open();

tab.open(win2);

backBtn.addEventListener('click', function() {
    tab.close(win2);
});
