var loginReq = Titanium.Network.createHTTPClient();

{
    var json = this.responseText;

    if (response.logged == true)
    {
        alert("Welcome " + response.name + ". Your email is: " + response.email);
    }
    else
    {
        alert(response.message);
    }
};

loginReq.onerror = function()
