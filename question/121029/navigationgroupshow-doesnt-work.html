NavigationGroup.show doesn't work

I'm using a NavigationGroup on the iPad to navigate between different screens.

Switching works only when I either remove the old window or add the new one. Therefore I can't use animated transitions like with 'show'.

What am I doing wrong?

Here's the source-code:

var rootWin = Titanium.UI.createWindow({
    backgroundColor: "#00FF00",
    modal: true
});

var mainWin = Titanium.UI.createWindow({
    height: 960,
    top: 1,
    backgroundColor: "#FFF",
        modal: true
});

var navGroup = Titanium.UI.iPhone.createNavigationGroup({
    window: mainWin
});


var win2 = Titanium.UI.createWindow({
    backgroundColor:"blue",
    title:"Blue Window",
    modal: true,
    navBarHidden: true
});

navGroup.add(win2);
navGroup.add(mainWin);


navGroup.addEventListener('cw', function(e)
{
    navGroup.show(win2, {animated:true}); //doesnt work
}
)

rootWin.add(navGroup);
rootWin.open();
