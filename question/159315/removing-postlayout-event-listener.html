    • var index ={ width:0, height:0}; // create a new object to hold the coordinates instead
index.addEventListener('postlayout', setWidthAndHeight);

function setWidthAndHeight(){
    index.removeEventListener('postLayout', setWidthAndHeight); // remove the listener to prevent repeats
    alert("width: " + index.rect.width);
    var height = index.rect.height;
    var width = index.rect.width;
    index.height = height;
    index.width = width;
}
