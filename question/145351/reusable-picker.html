  • hi,

    try this code

    var picker = Ti.UI.createPicker();


var titlePickerData = [Ti.UI.createPickerRow({
    title : 'Mr.'
}), Ti.UI.createPickerRow({
    title : 'Ms.'
}), Ti.UI.createPickerRow({
    title : 'Mdm.'
})];
//Picker data for Gender field
var genderPickerData = [Ti.UI.createPickerRow({
    title : 'Male'
}), Ti.UI.createPickerRow({
    title : 'Female'
})];

var c = 1;
var column5 = Ti.UI.createPickerColumn();
for (var i = 0; i < titlePickerData.length; i++) {
    column5.addRow(titlePickerData[i]);
}

picker.add([column5]);

var button = Ti.UI.createButton({
    title : 'Set to Rows',
    top : 34,
    width : 120,
    height : 30
});
win.add(button);

button.addEventListener('click', function() {

    var column = picker.getColumns();

    var rows = column[0].getRows();

    for (var i = 0; i < rows.length; i++) {

        column[0].removeRow(rows[i]);
    }

    if (c == 0) {        

        for (var i = 0; i < titlePickerData.length; i++) {
            column[0].addRow(titlePickerData[i]);
        }
        c = 1;
    } else {        

        for (var i = 0; i < genderPickerData.length; i++) {
            column[0].addRow(genderPickerData[i]);
        }
        c = 0;
    }

    picker.reloadColumn(column[0]);
});
win.add(picker);
