ScrollableView Swipe animation with button event

I have en ScrollableView as shown in the code bello, and I change though the different views using button event. The thing is I don't want to use face in/out , but a swipe animation. how can this be posible. Thanks

//Master View Component Constructor
function MasterView() {

//declare module dependencies
FavoriteView = require('ui/common/Favorite');
NearMeView = require('ui/common/NearMe');

//create object instance, parasitic subclass of Observable
var self = Ti.UI.createView({
    backgroundColor:'white'
});

//Views for scrollableView
var view1 = new NearMeView();
var view2 = new FavoriteView();
var view3 = Ti.UI.createView({ backgroundColor:'#48b' });

var scrollableView = Ti.UI.createScrollableView({
  views:[view1,view2,view3],
  top:45,
  showPagingControl:false
});
self.add(scrollableView);

var nearme = Ti.UI.createButton({
    top:5,
    left:5,
    height:38,
    width:35,
    backgroundImage:'/images/icon-gps.png'
});
self.add(nearme);

var favorites = Ti.UI.createButton({
    top:5,
    left:40,
    height:38,
    width:35,
    backgroundImage:'/images/favorites.png'
});
self.add(favorites);


nearme.addEventListener('click', function(e) {
    view1.animate({opacity: 1, duration: 500});
    scrollableView.currentPage = 0;

});

favorites.addEventListener('click', function(e) {
    view2.animate({opacity: 1, duration: 500});
     scrollableView.currentPage = 1;

});

return self;

};

module.exports = MasterView;

    Hi Rigoberto Miranda

    You can set specified view using scrollToView() method.. it will scroll to specified view.. you need to just pass it's index.

    e.g

    nearme.addEventListener('click', function(e) {
   // view1.animate({opacity: 1, duration: 500});
    scrollableView.scrollToView(0);

});

favorites.addEventListener('click', function(e) {
    //view2.animate({opacity: 1, duration: 500});
     scrollableView.scrollToView(1);

});

    And there is no need to set opacity 0 to every view initially because other then current view never appear at a time.. and if you want animation by using opacity then you can do it.. and for that first set opacity 0 to other then current view and animate it after set current view.

    if you have any doubt then let me know fell free… :)
