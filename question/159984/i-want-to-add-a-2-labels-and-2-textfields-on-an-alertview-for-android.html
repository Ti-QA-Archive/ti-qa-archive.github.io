I want to add a 2 labels and 2 textfields on an alertView for android

I want to add a 2 labels and 2 textfields on an alertView for android. Will it be possible since android does not support Popover feature as in iOS?

    var w = Ti.UI.createWindow({
    backgroundColor : 'black'
});

w.addEventListener('open', function() {

    var l1 = Ti.UI.createLabel({
        top : 0,
        left : 0,
        width : Ti.UI.SIZE,
        height : 40,
        text : 'Label 1'
    });

    var l2 = Ti.UI.createLabel({
        top : 50,
        left : 0,
        width : Ti.UI.SIZE,
        height : 40,
        text : 'Label 2'
    });

    var t1 = Ti.UI.createTextField({
        top : 0,
        right : 5,
        width : 100,
        height : 40
    });
    var t2 = Ti.UI.createTextField({
        top : 50,
        right : 5,
        width : 100,
        height : 40
    });
    var v = Ti.UI.createView({
        width : Ti.UI.SIZE,
        height : Ti.UI.SIZE
    });

    v.add(l1);
    v.add(l2);
    v.add(t1);
    v.add(t2);
    var a = Ti.UI.createAlertDialog({
        androidView : v
    });

    a.show();
});
w.open();
