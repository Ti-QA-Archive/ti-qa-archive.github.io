  • This code works well on both emulator and iPhone 4 in one of my projects:

    var topRightButton = Ti.UI.createButtonBar({
    labels: ['Close'],
    backgroundColor: '#a58d1e'
});

Ti.UI.currentWindow.rightNavButton = topRightButton;

topRightButton.addEventListener('click', function(){
    Ti.UI.currentWindow.close();
});
