HTTPClient not loading data

I am using a HTTPClient to gather the source code from a web site. It will fire the onload function, but does not load any of the source code.


exports.loadCharacter = function(_name){

        var url ="http://maplestory.nexon.net/Rankings/OverallRanking.aspx?type=overall&s=l&pageIndex=1&key="+_name+"&search=true&searchtype=name";
        //url = "http://www.google.com";
        var overallRank = Ti.Network.createHTTPClient({
            onload: function(e) {


                var text = this.responceText;

                Ti.API.log(url);
                Ti.API.log(text);

            },
            onerror: function(e) {
                mainWindow.appAlert("Unable to contact Nexon ranking site.");
            },
            timeout:5000
        });

        overallRank.open("GET", url);
        overallRank.send();

}

the log for the url is correct, but the log for the text is <null>.

does anyone see any problems with the code?
This is the first time i have tried to use CommonJS. Is there something special i have to do for using CommonJS and HTTPClient?
