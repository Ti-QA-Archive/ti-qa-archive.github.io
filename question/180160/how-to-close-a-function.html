    Hello benedict, Try the code below.

    var win2 = Ti.UI.createWindow({
    backgroundColor : 'gray',
    navBarHidden : true
});

var Button = Ti.UI.createButton({
    title : 'click',
    color : 'black',
});

win2.add(Button);

Button.addEventListener('click', function (e){

question1();

});


function question1() {
    Button.hide();

    var header1 = Titanium.UI.createLabel({
        text : 'HOME1',
        color : 'black',
        font : {
            fontSize : 25
        },
        top : '100dp',
        height : '40dp'
    });

    win2.add(header1);

    var skipButton1 = Ti.UI.createButton({
        title : 'Skip1',
        color : 'black',
        top : '80%',
        left : '40%',
        width : '100',
        height : '50'
    });

    win2.add(skipButton1);


    skipButton1.addEventListener('click', function(e) {

        header1.hide();

        question2();

    });

}

function question2() {

    var header2 = Titanium.UI.createLabel({
        text : 'HOME2',
        color : 'black',
        font : {
            fontSize : 25
        },
        top : '100dp',
        height : '40dp'
    });

    win2.add(header2);

    var skipButton2 = Ti.UI.createButton({
        title : 'Skip2',
        color : 'black',
        top : '80%',
        left : '40%',
        width : '100',
        height : '50'
    });

    win2.add(skipButton2);
}

win2.open();
