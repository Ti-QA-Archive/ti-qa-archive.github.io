  • Hi

    The dots are only natively supported on iOS, Android uses arrows left and right to show the paging ability.

    You can however make this yourself by creating a view, placing some image dots in the view and on an event listener update the dots to reflect the page position.

    • var welcome = Ti.UI.createView({ width: '100%', height: '100%', top: 0, backgroundColor:'#333', opacity: 0.5, zIndex: 0, exitOnClose:true });
var overlay = Ti.UI.createView({ width: '90%', height: '90%', top: 40, borderRadius: 4, opacity: 1.0, backgroundColor:'#fff', zIndex:1, exitOnClose:true });
var dots = Ti.UI.createView({ width: '80%', height: '20%', top: '70%', borderRadius: 4,opacity: 1.0, zIndex:2, exitOnClose:true });



    //for scrollable view

    var scrollView = Titanium.UI.createScrollableView({
        views : [ view1, view2, view3]
    });

    var dots1 = [  
                    {backgroundImage:'/images/slider_btn_on.png',views:'view1',},  
                    {backgroundImage:'/images/slider_btn_off.png',views:'view2'},  
                    {backgroundImage:'/images/slider_btn_off.png',views:'view3'}  
                ]; 

    var dots2 = [  
                    {backgroundImage:'/images/slider_btn_off.png',views:'view1',},  
                    {backgroundImage:'/images/slider_btn_on.png',views:'view2'},  
                    {backgroundImage:'/images/slider_btn_off.png',views:'view3'}     
                ];

    var dots3 = [  
                    {backgroundImage:'/images/slider_btn_off.png',views:'view1',},  
                    {backgroundImage:'/images/slider_btn_off.png',views:'view2'},  
                    {backgroundImage:'/images/slider_btn_on.png',views:'view3'}  
                ];

    scrollView.addEventListener('scroll',function(){
        dot1.backgroundImage = dots1[scrollView.currentPage].backgroundImage;  
        dot2.backgroundImage = dots2[scrollView.currentPage].backgroundImage;  
        dot3.backgroundImage = dots3[scrollView.currentPage].backgroundImage;  
    });

    overlay.add(scrollView);
    self.add(dots);
    self.add(overlay);
    self.add(welcome);
