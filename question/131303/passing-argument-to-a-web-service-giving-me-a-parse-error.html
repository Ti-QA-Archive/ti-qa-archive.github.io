Passing argument to a web service giving me a parse error

Hi all ,

My web service expects a parameter i.e. session ID.

I am using the following code but its not working. Its giving me parse error at the xhr1.send() line.


var url = "http://cc2.mycatalogcenter.com/api/api.asmx?op=GetJsonCatalogList";
    Ti.API.info('URL: '+url);
    var xhr1 = Titanium.Network.createHTTPClient();
    xhr1.open('POST',url);
    xhr1.send(c150fb95-2cc5-46f3-99a7-9446f6e0d07c);// giving parse error here . 

     xhr1.onload = function() {
        var obj = this.responseText;
    alert(obj);
    };

If i am passing the parameter wrong then,
How to send a parameter to a web service then?

Thanks

  • Your code should be rewritten like below:

    var url = "http://cc2.mycatalogcenter.com/api/api.asmx?op=GetJsonCatalogList";
var xhr1 = Titanium.Network.createHTTPClient();

// declare onload() before send()
xhr1.onload = function() {
    var obj = this.responseText;
    alert(obj);
};
xhr1.open('POST', url);

// For POST, sessionGuid is required
xhr1.send({sessionGuid: 'c150fb95-2cc5-46f3-99a7-9446f6e0d07c'});
