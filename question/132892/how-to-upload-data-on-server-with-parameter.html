how to upload data on server with parameter?

hi,
i make a application. In which i want to upload data on server. How i can do this?
i send you my code < it is also include with parameter>

`but4.addEventListener('click',function(){

var xhr = Titanium.Network.createHTTPClient({
timeout : '15000',
});

xhr.onload = function (){
Titanium.API.info('onload');
Ti.API.info('Sucess '+ this.status + ' '+ this.readyState + ', res '+ this.responseText);
};

xhr.onerror = function(){
Ti.API.info('error');
};

xhr.ondatastream=function(){
Ti.API.info('ondatastream');
};

var game_name = 'Game Name';

var abcd = 'iPhone_request=<category><name>game_name</name></category>';

    Specific for your case:

    xhr.send({iPhone_request: '<category><name>game_name</name></category>'});

// Then in your server side - for example for php:
// $_POST['iPhone_request'];
