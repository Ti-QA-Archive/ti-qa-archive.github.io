    • try this code , this is working very fine:

      //put these images in your resources folder
var images = ['image1.png','image2.png','image3.png'];

var imageView = Titanium.UI.createImageView({
 images:images,// images is the array of images
 duration:100, // in milliseconds, the time before next frame is shown
 repeatCount:0,  // 0 means animation repeats indefinitely, use > 1 to control repeat count

});

      // this supports png as well as gif too.

      if you want to reverse effect use imageView.reverse = true;
