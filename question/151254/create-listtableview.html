  • Accepted Answer

    var self = Ti.UI.createWindow();

    var headerLabel = Ti.UI.createLabel({
            text    : 'A Yellow Car!'
        });    

    var header = Ti.UI.createView({
            width            : Ti.UI.FILL,
            height            : 100,
            backgroundColor    : '#CCC'
        });
        header.add(headerLabel);

    var data = [];

    for (i=0;i<=3;i++) {
        var rowLabel = Ti.UI.createLabel({
                text        : 'This is a very funny story!, line number # ' + i,
                textAlign    : 'left' 
            });
        var row = Ti.UI.createTableViewRow({
                height            : 35,
                backgroundImage    : '../list-item-bg.png'
               });
               row.add(rowLabel);
        data.push(row)
    };

    var tableView = Ti.UI.createTableView({
            headerView    : header,
            data        : data
        });

    self.add(tableView);

    return self;

    • That's fine, all you need to do is set the tableView height. Try the below updated code.

          var self = Ti.UI.createWindow();

    var headerLabel = Ti.UI.createLabel({
            text    : 'A Yellow Car!'
        });    

    var header = Ti.UI.createView({
            width            : Ti.UI.FILL,
            height            : 100,
            backgroundColor    : '#CCC'
        });
        header.add(headerLabel);

    var data = [];

    var rowHeight = 35;

    for (i=0;i<=3;i++) {
        var rowLabel = Ti.UI.createLabel({
                text        : 'This is a very funny story!, line number # ' + i,
                textAlign    : 'left' 
            });
        var row = Ti.UI.createTableViewRow({
                height            : rowHeight,
                backgroundImage    : '../list-item-bg.png'
               });
               row.add(rowLabel);
        data.push(row)
    };

    var tableView = Ti.UI.createTableView({
            headerView    : header,
            data        : data
        });

    self.add(tableView);

    tableView.height = (data.length * rowHeight) + header.height;

    return self;
