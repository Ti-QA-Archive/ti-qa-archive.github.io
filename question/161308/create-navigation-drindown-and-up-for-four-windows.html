    Hi omkar suryawanshi!!!

    Please check the below code. I hope, it will reduce your problem a little bit.

    The below code is so basic that I believe , it will helpful to you

    app.js

    var awin = Titanium.UI.createWindow({
    backgroundColor:'#fff',
    title:'First Window',
}); 

var aButton = Ti.UI.createButton({
    title : 'GO to B',
    height : 'auto',
    width : 'auto',
    top : 10,
});

aButton.addEventListener('click', function() {
    var ok = Titanium.UI.createWindow({
    url:'bwin.js',
    title:'Second Window',
    backgroundColor:'green',
    _parent:Ti.UI.currentWindow,
    modal:true,
    });
    ok.open();
});

awin.add(aButton);
awin.open();

    bwin.js

    var bwin = Titanium.UI.currentWindow; 

var aButton = Ti.UI.createButton({
    title : 'GO to C',
    height : 'auto',
    width : 'auto',
    top : 10,
});

aButton.addEventListener('click', function() {
    var ok = Titanium.UI.createWindow({
    url:'cwin.js',
    title:'Third Window',
    backgroundColor:'blue',
    _parent:Ti.UI.currentWindow,
    modal:true,
    });
    ok.open();
});

bwin.add(aButton);

    cwin.js

    var cwin = Titanium.UI.currentWindow; 

var aButton = Ti.UI.createButton({
    title : 'GO to D',
    height : 'auto',
    width : 'auto',
    top : 10,
});

aButton.addEventListener('click', function() {
    var ok = Titanium.UI.createWindow({
    url:'dwin.js',
    title:'Four Window',
    backgroundColor:'red',
    _parent:Ti.UI.currentWindow,
    modal:true,
    });
    ok.open();
});

cwin.add(aButton);

    dwin.js

    var dwin = Titanium.UI.currentWindow; 

var aButton = Ti.UI.createButton({
    title : 'GO to A',
    height : 'auto',
    width : 'auto',
    top : 10,
});

aButton.addEventListener('click', function() {
    var ok = Titanium.UI.createWindow({
    url:'app.js',
    title:'First Window',
    backgroundColor:'#fff',
    _parent:Ti.UI.currentWindow,
    modal:true,
    });
    ok.open();

});

dwin.add(aButton);
