    yes,try like this

    var Data[];
for (var Row=0;Row<RowCount;Row++){
//Create table rows       
       var row = Ti.UI.createTableViewRow({
       height:30,
       title:"write what ever U want",
       color:'#000'
       });
        Data.push(row);
       }
var TableView = Titanium.UI.createTableView({
             data:Data,
                       top:90,
                            RowSize:Data.length   //custom property
                    });
TableView.addEventListener('click', function(e){
       Titanium.API.info("height of row"+this.RowSize);

      });

    by using custom proprty we can get any value when we bind required value to our object(labels,tableview..etc)
