  • This hasn't been tested, but should do what you are looking for.

    var yearInput = 2013, monthInput = 3, dayInput = 19, hourInput = 19, minuteInput = 28;
var d = new Date(yearInput, monthInput, dayInput, hourInput, minuteInput, 0, 0);
if (parseInt(d.getFullYear()) == parseInt(yearInput)) {
  Ti.API.info('We have a valid year!')
}
if (parseInt(d.getMonth()) == parseInt(monthInput)) {
  Ti.API.info('We have a valid month!')
}
if (parseInt(d.getDate()) == parseInt(dayInput)) {
  Ti.API.info('We have a valid day!')
}
