    Based on your brief description, here's untested and unstyled sample code:

    index.xml

    <Alloy>
    <Window>
        <View>
            <TextField id="text1"/>
            <TextField id="text2"/>
            <Button id="button" onClick="handleButton"/>
        </View>
    </Window>
</Alloy>

    index.js

    function handleButton(e) {
    // create instance of "newWindow" controller
    var controller = Alloy.createController('newWindow'); 

    // save textfield values to new controller
    controller.showValues($.text1.value, $.text2.value);

    // open the controller's window
    controller.getView().open();
}

$.index.open();

    newWindow.xml

    <Alloy>
    <Window/>
</Alloy>

    newWindow.js

    exports.showValues = function(text1, text2) {
    alert(text1 + ' and ' + text2);
}
