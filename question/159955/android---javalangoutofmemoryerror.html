    function createAboutUsButton() {

            top : 0,

            backgroundColor : 'white',
            layout : "vertical",
            url : 'http://www.cnn.com'
        });

        var buttonClose = Titanium.UI.createButton({
            title : '\u2190 Back',

            bottom: 0,
            width:'100%',
            height:'44dp',
            background:'white',
        });

        if(platformUtil.isiOS7Plus)
        {
            twitterLabel.hide();
        }
        win.add(webView);
        win.add(buttonClose);

        buttonClose.addEventListener('click', function(e) {


        if(platformUtil.isiOS7Plus)

            twitterLabel.show();

            win.close();
            win.remove(webView);
            win.remove(buttonClose);

            webView = null;

        });
        win.open();
    });

    return aboutUsBtn;
};
