Separate style from code

Is there a way to separate the style from the code? Like CSS files for example.
What I've seen in the examples from the documentation, the colors and size of an element are passed as parameters like:

var label1 = Ti.UI.createLabel({
  color: '#900',
  font: { fontSize:48 },
  shadowColor: '#aaa',
  shadowOffset: {x:5, y:5},
  text: 'A simple label',
  textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER,
  top: 30,
  width: 'auto', height: 'auto'
});

Is there a way to put everything in one file so it can be forwarded to a designer?

    Check out the Alloy project that we're working on. It will be our solution going forward, once it's released

    http://projects.appcelerator.com/alloy/docs/Alloy-bootstrap/index.html

    In the meantime, or if you're not up for a total MVC-like framework, you can create a commonJS module for your styles

    // styles.js
var Styles = function() {
    var styles = {};
    styles.colors = {
        font: '#900',
        shadowColor: '#aaa'
    }
    return styles;
}
module.exports = Styles;

// app.js
var Styles = require('styles'),
    styles = new Styles();

var label1 = Ti.UI.createLabel({
   color: styles.colors.font,
    ...
});
