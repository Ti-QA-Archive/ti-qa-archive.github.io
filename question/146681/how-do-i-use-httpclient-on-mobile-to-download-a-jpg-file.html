How do I use HTTPClient on mobile to download a jpg file?

I am trying to use HTTPClient to download a jpg file from our website onto the mobile device. It returns a 501 error. I can browse to the same url in my browser and see the image. How do I get this to work?

Here is my source code:


    var url = "http://www.taleblazer.org/img/uploads/step_zen_logo0.jpg";
    Ti.API.info("Downloading file: " + url);
    var httpFileDownload = Titanium.Network.createHTTPClient();
    httpFileDownload.open("Get", url);
    httpFileDownload.onerror = function(e){
        Ti.API.error("...attempt to download file returned error e: " + JSON.stringify(e));
        Ti.API.error("...status: " + httpFileDownload.status);
        tb.game.ctr.gameSetup.downloadError();
    }
    httpFileDownload.onload = function(){
        Ti.API.error("...downloaded file succesfully");
        tb.files.saveAssetFile(tb.game.storeId, url, this.responseData);
    }
    httpFileDownload.send();

    Hi Lisa, i think you are specifying wrong REST verb to HTTP client,

    you are using httpFileDownload.open("Get", url); //Get

    where i think it should be httpFileDownload.open("GET", url); //GET

    even your log message says the same, if you look. make these changes and then give it a try. it should work without any other modification.

var get_remote_file = function(filename, url,activityIndicator) {
    //alert("inc width : "+indicatorIncCount);
    var file_obj = {
        file : filename,
        url : url,
        path : null
    };

    var file = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, filename);
    if(file.exists()) {
        //alert("alresy exists"+file.nativePath);
        //indicator.width = indicator.width+indicatorIncCount;

        file_obj.path = Titanium.Filesystem.applicationDataDirectory + Titanium.Filesystem.separator;
        //alert(file_obj.path);
        //return file.nativePath;
    } else {

        //alert("not exists");
        if(Titanium.Network.online) {
            //activityIndicator.show();
            var c = Titanium.Network.createHTTPClient();

            c.setTimeout(10000);
            c.onload = function() {

                if(c.status == 200) {

                    //alert("done");
                    var f = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, filename);
                    f.write(this.responseData);
                    file_obj.path = Titanium.Filesystem.applicationDataDirectory + Titanium.Filesystem.separator;
                    //indicator.width = indicator.width+indicatorIncCount;
                } else {
                    file_obj.error = 'file not found';
                    // to set some errors codes
                }
                //fn_end(file_obj);

            };
            c.ondatastream = function(e) {


            };
            c.onerror = function(e) {
            //activityIndicator.hide();
                //activityIndicator.hide();
                alert("No Internet Connection");
                file_obj.error = e.error;
                //fn_end(file_obj);
            };
            c.open('GET', url, false);
            c.send();
        } 

    }
};
