      var d1 = new date(datetoday.getyear(),datetoday.getmonth(),datetoday.getday(),datetoday.gethour(),datetoday.getminute(),datetoday.getsecond());

      var d1 = new date(datetoday.getYear(),datetoday.getMonth(),datetoday.getDay(),datetoday.getHour(),datetoday.getMinute(),datetoday.getSecond());
