TimerWindow.prototype.addDie = function(file,curr_exercise) {
    var die_window = Ti.UI.createWindow({
            title:'Video Player',
            backgroundColor:'red'
        });
    var die_video = Titanium.Media.createVideoPlayer({
        autoplay : true,
        url : dir_path.R + 'video/die/' + file + ".mov",
        scalingMode : Titanium.Media.VIDEO_SCALING_ASPECT_FIT,
        mediaControlStyle : the_screen.getVideoControls(),
        fullscreen : false
    });
    die_video.addEventListener('complete', function(e) {
        // core.pause(1000);
        // alert("closing window");
        if (e.reason == 0) {
            die_window.close();
        }
        // alert("window closed");
        var name = core.exerciseName(curr_exercise);
        exercise_lbl.text = name;
        var url_queue = new Array();
        url_queue.push(core.getAudioPath("lets_go"));
        url_queue.push(core.getExerciseAudioPath(name));
        core.playVoiceOver(url_queue, function(e) {
            object_ref.addVideo(name);
            if (Ti.App.music == 1)
                Ti.App.fireEvent('play_music');
        });
    });
    die_window.add(die_video);
    die_window.open();
    // die_window.addEventListener('open',function(e){
        // die_video.play();
    // });
};
