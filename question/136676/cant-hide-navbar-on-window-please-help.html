  • In main window, if you have a button to open a secondary window, use as example this code:

    var btn01 = Titanium.UI.createButton({
    backgroundImage: 'images/btn01.png',
    backgroundFocusedImage: 'images/btn01_over.png',
    backgroundSelectedImage: 'images/btn01_over.png',
});
btn01.addEventListener('click',function(){
    var w1 = Titanium.UI.createWindow({
        url:'windows/agenda.js',
        navBarHidden: true,
    });
    w1.open();
});

    In secondary window, use only this code:

    var win = Ti.UI.currentWindow;
win.backgroundImage = '../images/background.png';
win.orientationModes = [Titanium.UI.PORTRAIT];

    Works 100% with Android
