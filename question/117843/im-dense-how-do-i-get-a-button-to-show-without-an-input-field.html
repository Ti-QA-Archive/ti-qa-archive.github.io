  • – I modified the vanilla tabs demo.. win2 below will only display the ip address field, until I type in it.. then I see the save button.

    // this sets the background color of the master UIView (when there are no windows/tab groups on it)
    Titanium.UI.setBackgroundColor('#000');

    // create tab group
    var tabGroup = Titanium.UI.createTabGroup();

    //
    // create base UI tab and root window
    //
    var win1 = Titanium.UI.createWindow({
    title:'CounterPoint Wifi',
    // backgroundColor:'#fff'
    url: 'main_menu.js'
    });
    var tab1 = Titanium.UI.createTab({
    icon:'KS_nav_views.png',
    title:'CounterPoint Wifi',
    window:win1
    });
    /*
    var label1 = Titanium.UI.createLabel({
    color:'#999',
    text:'function picker window',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
    //url:'main_menu.js';
    });

    win1.add(label1);
    */
    //
    // create controls tab and root window
    //

    var win2 = Titanium.UI.createWindow({
    title:'Settings',
    backgroundColor:'#fff'
    });
    var tab2 = Titanium.UI.createTab({
    icon:'KS_nav_ui.png',
    title:'Settings',
    window:win2
    });
    /*
    var label2 = Titanium.UI.createLabel({
    color:'#999',
    text:'setting window',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
    });

    win2.add(label2);
    */

    var iplabel= Titanium.UI.createLabel({
    color:'#999',
    text:'Server IP Address:',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'left',
    width:'auto',
    top: 20
    });
    var ipaddress = Titanium.UI.createTextField({
    hintText:"IP of CP server",
    height:40,
    width:300,
    top:60,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
    });

    ipaddress.value= Titanium.App.Properties.getString("server_ip");

    var save_settings = Titanium.UI.createButton({
    width: "98%",
    color: "#555",
    font:{fontSize: 22, fontWeight: "bold"},
    height: 50,
    top:180,
    title: "Save"
    });

    function saveit(e){
    Titanium.App.Properties.setString("server_ip",ipaddress.value);
    var alrt_Success = Titanium.UI.createAlertDialog({
    title: 'Success!',
    message: 'Setting saved',
    buttonNames: ['OK']
    });
    alrt_Success.show();

    }
    save_settings.addEventListener('click',saveit);
    win2.add(iplabel);
    win2.add(ipaddress);
    win2.add(save_settings);

    var win3=Titanium.UI.createWindow({
    title:'Data Import',
    backgroundColor:'#fff'
    });
    var tab3 = Titanium.UI.createTab({
    icon:'KS_nav_ui.png',
    title:'Data Import',
    window:win3
    });

    var import_items = Titanium.UI.createButton({
    width: "98%",
    color: "#555",
    font:{fontSize: 22, fontWeight: "bold"},
    height: 50,
    top: 80,
    visible: true,
    enabled: true,
    style: 1,
    title: "Import"
    });

    var imlabel= Titanium.UI.createLabel({
    color:'#999',
    text:'Server IP Address:',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'left',
    width:'auto',
    top: 20
    });
    function importitems(e) {
    // open/create local sqlit database
    var conn=Titanium.Database.open('cpsql');
    conn.execute('create table if not exists im_item (item_no varchar(30) index, descr varcher(50))');
    var xhr = Ti.Network.createHTTPClient();
    xhr.onload = function(e) {
    //this fires when your request returns successfully
    //this.responseText and this.responseXML
    //will be used often
    var myData = JSON.parse(this.responseText);
    var alrt_echo=Titanium.UI.createAlertDialog({
    title: 'Success!',
    message: 'Setting saved',
    buttonNames: ['OK']

        });
    //alrt_echo.message=myData.ITEM_NO;
    alrt_echo.show();
};

xhr.onerror = function(e) {
  //this fires if Titanium/the native SDK cannot successfully retrieve a resource
};

xhr.open('GET','http://127.0.0.1/jsontest.php');
xhr.send();

    }

    import_items.addEventListener('click',importitems);

    //win3.add(ipaddress);
    win3.add(import_items);

    //
    // add tabs
    //
    tabGroup.addTab(tab1);
    tabGroup.addTab(tab2);
    tabGroup.addTab(tab3);

    // open tab group
    tabGroup.open();
