Button on TableView Row

I have one labels and a button on a tableView row. On clicking the button it opens an option dialog. The selection that I make on the option dialog, the title of the button becomes corresponding to that selection. However i also want the value of the label on the row corresponding to the button that I clicked so that updation in database can be made according to the label value and the option that i had selected.
The situation is easy if I add an event listener on the tableview row but I want it specifically when the button is clicked. Specifically for android

  • homeview.getInspectionforHomeScreen=function(config)
    {
    var result=model.getInspectionforHomeScreen(config);
    var row = baseUI.tableViewRow();
    var homeArrays = [];
    var totalFields = model.getFieldCount(result);
    var status;
    var inspID;
    var btnStatus;

    for (var iRowCount = 0; iRowCount < result.rowCount; iRowCount++) 
{
    var row = baseUI.tableViewRow();
    row.height = '44dp';
    for (var iFieldCount = 0; iFieldCount < totalFields; iFieldCount++) 
    {

        if (iFieldCount == 2) {
            var lblScheduleDate = baseUI.label('10', 0, Ti.UI.SIZE, Ti.UI.SIZE, result.field(iFieldCount));
            lblScheduleDate.font = {
                fontSize : 13,
                fontWeight : 'bold'
            };
            lblScheduleDate.color = 'gray';
            row.add(lblScheduleDate);

        } else if (iFieldCount == 0) {
            var lblInspID = baseUI.label('150', 10, Ti.UI.SIZE, Ti.UI.SIZE, result.field(iFieldCount));
            lblInspID.color = 'gray';
            lblInspID.font = {
                fontSize : 14
            };
            row.add(lblInspID);

        }
        else if (iFieldCount == 3) {
            var lblArea = baseUI.label('200', 10, Ti.UI.SIZE, Ti.UI.SIZE, result.field(iFieldCount));
            lblArea.color = 'gray';
            lblArea.font = {
                fontSize : 14
            };
            row.add(lblArea);
        } else if (iFieldCount == 4) {

            var statusText = model.getStatusTextBasedOnValue(result.field(iFieldCount));
            var params = UIParamLayer.getStatusButtonForHome(config, statusText);
            btnStatus=baseUI.button(params);
            btnStatus.addEventListener('click', function(e)
            {
                //inspID=e.row.children[0].text;
                statusValue=e.source.title;
                if(statusValue=='Close')
                {
                    alert('Inspection is closed');
                }
                else
                {
                    var optionsDialogOpts = 
                    {
                        options:['Open', 'In Process', 'Follow Up', 'Close'],
                    buttonNames:['Cancel'],
                    title:'Please choose status'
                    };

                    dialog=Ti.UI.createOptionDialog(optionsDialogOpts);
                    dialog.show();
                    dialog.addEventListener('click',function(e)
                    {
                        status=optionsDialogOpts.options[e.index];
                        status=model.getStatusValue(config, status);
                        //model.updateInspectionStatusFromHome(inspID, status);

                    });
                }
            });    
            row.add(btnStatus);

        }

    }
            //section.add(row);
    homeArrays.push(row);
    result.next();
}


var tblData = homeArrays;
var tblView = baseUI.tableView(tblData,Titanium.UI.iPhone.TableViewStyle.PLAIN);

return tblView;

    };
