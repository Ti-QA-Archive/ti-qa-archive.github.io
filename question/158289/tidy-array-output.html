    Instead of creating another array why not just do this:

    if (errors.length > 0) {
    var mes = 'The following errors have occurred:';
    for (var i = 0; i < errors.length; i++) {
        Ti.API.info(errors[i].message);
        mes += '\n' + errors[i].message;
     }
    alert(mes);
    return;
}
alert('Hooray! Your form is valid!');
openit();
