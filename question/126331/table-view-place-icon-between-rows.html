Table view place icon between rows

Hi All,
I have a table view and I need to place an image over the line that separate two rows. I succesfully positioned the image with the following code, but the image appers under the preceding row and over its own row. This is my code:

var data=[];
for(i=0;i<10;i++){
var row = Ti.UI.createTableViewRow({
    height : 'auto',
    className : "row"
});

var textView = Ti.UI.createView({
    height : 'auto',
    layout : 'vertical',
    left : 3,
    top : 3,
    bottom : 3,
    right : 3
});
var translateImage = Titanium.UI.createImageView({
    image : 'image.png',
    top : -25,
    zIndex : 200
});
textView.add(translateImage);

var nameLabel = Titanium.UI.createLabel({
    id : 'name',
    text : 'name',
    textAlign : 'center',
    height : 'auto',
    top : 1
});
textView.add(nameLabel);

var moleculeLabel = Titanium.UI.createLabel({
    id : 'molecule',
    text : 'text',
    textAlign : 'center',
    height : 'auto',
    top : 5
});
textView.add(moleculeLabel);

row.add(textView);
data.push[row];
}        
var tableView = Titanium.UI.createTableView({
    backgroundColor:'#85b2c2'
});

tableView.data = data;

  • As far as I know, what you're trying to so with tableview rows is not possible. I think the elements within a row will be 'contained' in the row, so will not be able to get pushed out of it.

    Depending on the size of the set of data/list you're trying to represent, you might want to try doing it using a scrollView with a vertical layout, appending the various views and labels you need like that. Almost like a 'pseudo' tableview.
