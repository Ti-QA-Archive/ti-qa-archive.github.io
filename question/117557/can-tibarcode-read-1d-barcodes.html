  • Ti.Barcode does not support 1D scanning according to Appcelerator.

    That said, to try out QR code scanning, here's sample code which I'm using to implement a simple barcode scanning button:

    var win = Ti.UI.createWindow({
    backgroundColor: 'white'
});

var barcode = require("ti.barcode");
var scan = Ti.UI.createButton({
    width: 200,
    height: 44,
    title: 'Scan!'
});

scan.addEventListener('click', function() {
    barcode.capture({
        success: function(data) {
            Ti.API.error(JSON.stringify(data));
            alert('Scanned!');
        },
        cancel: function(event) {
            alert("Cancel!");
        },
        error: function(event) {
            alert("Error: " + event.message);
        }
    });
});

win.add(scan);
win.open();

    Enjoy!
