  • You can set this in the click listener of tableView like below:
    first specify the id in row. and then check that if that is clicked

    var tablerow = Ti.UI.createTableViewRow({
    height: 70,
    backgroundGradient: {
            type: 'linear',
            startPoint: { x: '0', y: '50%' },
            endPoint: { x: '200dp', y: '50%' },
            colors: [ 
                { color: '#dadada', offset: 0 },
                { color: '#ffffff', offset: 1 }
            ]
        },
    id: "row"
});

tableview.addEventListener("click", function(){
    if(e.source.id == "row"){
        e.source.backgroudGradient = // set whatever gradient you want. 
    }

})
