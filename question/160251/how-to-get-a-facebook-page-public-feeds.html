  • Hello ebrahim sayed, I hope that it will helpful for you

    var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});

var fbmod=require('facebook');
fbmod.appid = 'your app id';
fbmod.permissions = ['publish_stream','read_stream'];
fbmod.addEventListener('login', function(e) {
    if (e.success) {

    alert('Logged in');
    fbmod.requestWithGraphPath('me/home', {}, 'GET', function(e) {
    if (e.success) 
    {
        alert(e.result);
        var fbstob=JSON.parse(e.result);
        Ti.API.info(fbstob);

    } else if (e.error) {
        alert(e.error);
    } else {
        alert('Unknown response');
    }
});

   }
});
fbmod.addEventListener('logout', function(e) {
    alert('Logged out');
});

win1.add(fbmod.createLoginButton({
    top : 50,
    style :fbmod.BUTTON_STYLE_NORMAL
}));


win1.open();
