  • Thanks, I did a work around this weekend before seeing you answer. I stripped off the string data and rebuilt the file so the parser could read it. Seems to be working OK so far.

    function getaccounts() // get accountsEST data

    {

    try{

    var xhr1 = Titanium.Network.createHTTPClient();
    xhr1.onload = function()
    {

    //l2.value = this.responseText;
//Titanium.API.info(this.responseText);    
var xmltext =this.responseText;

var startstring =xmltext.indexOf('<XMLaccounts&gt');
var endstring =xmltext.indexOf('</VFPDataSet&gt');

var newxmldata = '<VFPDataSet> \n' + xmltext.substring(startstring,endstring) + '</VFPDataSet>';
var cleanstring1 = newxmldata.replace(/</gi,'<');
var cleanstring2 = cleanstring1.replace(/>/gi,'>');
//l4.value = cleanstring2;

var xmlObject = Ti.XML.parseString(cleanstring2);
var acctno = xmlObject.documentElement.getElementsByTagName("acct_no");
var acctname = xmlObject.documentElement.getElementsByTagName("acct_name");

if (acctno && acctno.length>0) {
     var x = 0;
    for (var c=0;c<acctno.length;c++)
    {
        var itemacctno = acctno.item(c);
        var itemacctname = acctname.item(c);
        //alert(itemacctname.text);
        db.execute('INSERT INTO ACCOUNTS (ID, ACCT_NO, ACCT_NAME) VALUES(?,?,?)',c,itemacctno.text,itemacctname.text);  
    }

    var rows = db.execute('SELECT * FROM ACCOUNTS');
    Ti.API.info('Rows ' + rows.rowCount);
    l2.text='Accounts : '+rows.rowCount+" downloaded." ; 
} 
else {
      l2.text = 'No Accounts Data Received From Server. Try again.';
}

rows.close();
db.close();

    };
