    • I did figure out solution, but it doesn't behave by closing the entire app, I have to explicitly close all the windows then I'm left with the splash screen. 

      win.addEventListener('android:back',function(e){
        if(APP.WindowStack.length > 1) {
            // Load the View from the WindowStack
    navWin.setDroidView(APP.WindowStack[APP.WindowStack.length - 1]);
            APP.WindowStack.pop();        
         } else if((APP.WindowStack.length == 1) || (APP.WindowStack.length == 0)){
             win.close(); // close main window
             menuWin.close(); // close left navigation window
         } 
    });
