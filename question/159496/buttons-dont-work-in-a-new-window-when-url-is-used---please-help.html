Buttons don't work in a new window, when url is used - Please help!

When I use a url to reference a file in a createWindow, the buttons in the new window doesn't work. The answer is probably very simple but I couldn't figure this out. Please help!

Titanium Studio, build: 3.1.3.201309132423
Platform: iOS

In the example below, button123 doesn't work. It doesn't click, it doesn't do anything.

//In app.js
var win1 = Ti.UI.createWindow({
backgroundColor: 'blue'
});

var button = Titanium.UI.createButton({
title:'New Window'
});

win1.add(button);

button.addEventListener('click', function(e){
var newWin = Ti.UI.createWindow({
url:'new_win.js'
});
newWin.open();
});

win1.open();

//In new_win.js
var win2 = Ti.UI.createWindow({
backgroundColor: 'red'
})

var button123 = Titanium.UI.createButton({
title:'Do Something!'
});

win2.add(button123);

button123.addEventListener('click', function(e){
//Does something cool!
});
});

win2.open();

  • Accepted Answer

    win.url was deprecated many releases ago. I guess I wouldn't be surprised if there were problems with it. You should be using CommonJS modules (or Alloy) instead. I'd also recommend adding event listeners before adding UI components to windows or views.

    Untested, but you could try this:

    //In app.js 
var win1 = Ti.UI.createWindow({ 
    backgroundColor: 'blue'
});

var button = Titanium.UI.createButton({
    title:'New Window' 
});

win1.add(button);

button.addEventListener('click', function(e){ 
  var MakeWin = require('win2');
  var newWin = new MakeWin();
  newWin.open();
});

win1.open();

    // win2.js
var MakeWindow = function() {
  var win2 = Ti.UI.createWindow({
    backgroundColor: 'red'
  });

  var button123 = Titanium.UI.createButton({ 
    title:'Do Something!' 
  }); 

  button123.addEventListener('click', function(e){ 
    alert('does something cool');
  });

  win2.add(button123);
};

module.exports = MakeWindow;
