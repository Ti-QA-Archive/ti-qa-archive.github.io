  • You can use the load event from an ImageView to check on the CoverFlow load completion.

    Try something like this :

    var win = Titanium.UI.currentWindow;
var imageViews = [], coverflow, numLoaded = -1;    // variables
var update, createImage;    // methods

update = function ( ) {
  var images = [];
  for ( var i = 0; i < imageViews.length; i++ ) images.push(imageViews[i].toImage());
  coverflow.images = images;

  numLoaded++;
  if ( numLoaded == imageViews.length ) alert('All images have been loaded');
}

createImage = function ( url ) {
  var imageView = Titanium.UI.createImageView({
    width: 290, height: 195,
    image: url,
    hires: true
  });
  imageView.addEventListener('load', update);

  return imageView;
}

var imageUrls = ['http://www.lorempixel.com/290/195/abstract', 'http://www.lorempixel.com/290/195/food', 'http://www.lorempixel.com/290/195/people', 'http://www.lorempixel.com/290/195/city', 'http://www.lorempixel.com/290/195/fashion'];

for ( var i = 0; i < imageUrls.length; i++ ) imageViews.push(createImage(imageUrls[i]));

coverflow = Titanium.UI.createCoverFlowView({
  images: [],
  backgroundColor: '#fff'
});
win.add(coverflow);
update();

    As you can see, you can now detect when all images have been loaded in the update method, by comparing numLoaded and imageViews.length.
