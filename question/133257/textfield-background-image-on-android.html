    win= Ti.UI.createWindow({
    backgroundColor:'black'
});
var tf1 = Titanium.UI.createTextField({
    color:'white',
    height:35,
    top:10,
    left:10,
    width:250,
    backgroundImage: 'KS_nav_ui.png'
});
win.add(tf1);
win.open();
