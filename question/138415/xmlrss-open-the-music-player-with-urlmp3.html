  • I manage to retrieve the elements in this interface

    var win = Ti.UI.currentWindow;

var obj = win.thisObj;

var scrollView = Titanium.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    top : 0,
    showVerticalScrollIndicator : true,
    showHorizontalScrollIndicator : true
});

var img = Ti.UI.createImageView({
    top : 10,
    left : 10,
    width : 300,
    height : 175,
    image : obj.img
});

scrollView.add(img);

var lbl = Ti.UI.createLabel({
    top : 190,
    left : 10,
    height : 'auto',
    width : 'auto',
    text : obj.title,
    backgroundColor : 'white',
    color : 'black'
});
scrollView.add(lbl);

var mp3 = Ti.UI.createWebView({
    top : 220,
    left : 10,
    height : 'auto',
    width : 300,
    bottom : 20,
    html : '<audio src="' + obj.mp3 + '" autoplay="" controls="" onended="this.play()"></audio>',
    color : 'white'
});
scrollView.add(mp3);
win.add(scrollView);
