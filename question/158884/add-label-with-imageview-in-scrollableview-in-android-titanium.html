    <View id="view">

  • ok, then you can set view's array as a views property and in that view's array, you can add imageview and lable in each view…

    try this code…

    
var arrViews = [];

for(var i =0; i < 3; i++){

var view1 = Ti.UI.createView({
    width : 320,
    height : 250,
    layout : 'vertical'
});

var imgView1 = Ti.UI.createImageView({
    width : 320,
    height : 220,
    backgroundColor : 'red'
});

view1.add(imgView1);

var lbl1 = Ti.UI.createLabel({
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
    text : 'LABEL >> ' + (i + 1)
});

view1.add(lbl1);

arrViews.push(view1);
}

var scrollView = Ti.UI.createScrollableView({
    views : arrViews,
    width : 320,
    height : 250
});
