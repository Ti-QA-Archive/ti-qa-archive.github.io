Table view click listener is not working with sections

I am using table view with sections. Click listener attached to table view works fine in android but it does not get invoked in iPhone. Though if I don't use sections then it gets invoked. Are table sections not supported in iPhone or am I missing something.

Following is the complete code.

//Constructor
function TableAlertsActivityIndicatorActionSheet(){
var TABLE_TITLE = "This is table with sections header";
var TABLE_FOOTER = "This is table with sections footer";
var SECTION_TITLE_FOR_ALERTS = "Following are various types of alerts.";
var SECTION_TITLE_FOR_ACTIVITY_INDICATOR = "Following are various types of activity indicators.";
var SECTION_TITLE_FOR_OPTION_DIALOG_OR_ACTION_SHEET = "Following are various types of option dialog or action sheets.";

var ALERTS_NO_BUTTON = "Show alert without button.";
var ALERTS_SINGLE_BUTTON = "Show alert with one button.";
var ALERTS_TWO_BUTTONS = "Show alert with two buttons.";
var ALERTS_THREE_BUTTONS = "Show alert with three buttons.";

var DIALOG_TITLE = "Alert"
var DIALOG_MSG = "Table row got clicked = ";
var DIALOG_OK_BUTTON = "Ok";
var DIALOG_CANCEL_BUTTON = "Cancel";
var DIALOG_HELP_BUTTON = "Help";

var ACTIVITY_INDICATOR_WITHOUT_MESSAGE = "Show activity indicator without message.";
var ACTIVITY_INDICATOR_WITH_MESSAGE = "Show activity indicator with message.";
var ACTIVITY_INDICATOR_MESSAGE = "Processing...";
var ACTIVITY_INDICATOR_TIMEOUT = 5000; // Milliseconds

var OPTION_DIALOG_OR_ACTION_SHEET = "Show option dialog or action sheet.";
var OPTION_DIALOG_OR_ACTION_SHEET_TITLE = "Sample option dialog or action sheet";

var TABLE_ROW_ICON = "/images/row_icon.png";

var row = null;
var rowTitle = null;
//------------------------------- PRIVATE START -------------------------------------
function tableClickListener(event){
    Ti.API.info("tableClickListener() START");
    Ti.API.info("event index = "+event.index);
    var dialog = null;
    var activityIndicator = null;
    var optionsDialgoOrActionSheet = null;
    rowTitle = event.row.title;
    row = event.row;
    switch(event.index){
        case 0: // Alert without button
            dialog = Ti.UI.createAlertDialog({title:DIALOG_TITLE, message:DIALOG_MSG + rowTitle});
        break;
        case 1: // Alert with one button
            dialog = Ti.UI.createAlertDialog({title:DIALOG_TITLE, message:DIALOG_MSG + rowTitle, buttonNames:[DIALOG_OK_BUTTON]});
        break;
        case 2: // Alert with two buttons
            dialog = Ti.UI.createAlertDialog({title:DIALOG_TITLE, message:DIALOG_MSG + rowTitle, cancel : 1, buttonNames:[DIALOG_OK_BUTTON, DIALOG_CANCEL_BUTTON]});
        break;
        case 3: // Alert with three buttons
            dialog = Ti.UI.createAlertDialog({title:DIALOG_TITLE, message:DIALOG_MSG + rowTitle, cancel : 1, buttonNames:[DIALOG_OK_BUTTON, DIALOG_CANCEL_BUTTON, DIALOG_HELP_BUTTON]});
        break;
        case 4: // Activity indicator without message
            activityIndicator = Ti.UI.createActivityIndicator({height:Ti.UI.SIZE, width:Ti.UI.SIZE});
            event.row.add(activityIndicator);
        break;
        case 5: // Activity indicator with message
            activityIndicator = Ti.UI.createActivityIndicator({height:Ti.UI.SIZE, width:Ti.UI.SIZE, message:ACTIVITY_INDICATOR_MESSAGE, color:"green"});
            event.row.add(activityIndicator);
        break;
        case 6: // Options dialog or action sheet
            optionsDialgoOrActionSheet = Ti.UI.createOptionDialog({cancel:1, options:[DIALOG_OK_BUTTON, DIALOG_CANCEL_BUTTON, DIALOG_HELP_BUTTON],
                destructive:1,selectedIndex:0,title:OPTION_DIALOG_OR_ACTION_SHEET_TITLE});
        break;

    }
    if(dialog != null){
        dialog.addEventListener("click", dialogClickListener);
        dialog.show();
    }
    if(activityIndicator != null){
        activityIndicator.show();
        // Hide the activity indicator after few seconds.
        setTimeout(function(){
            //row.remove(activityIndicator);                
            activityIndicator.hide();
            //row.setTitle(rowTitle);
        },ACTIVITY_INDICATOR_TIMEOUT);
    }
    if(optionsDialgoOrActionSheet != null){
        optionsDialgoOrActionSheet.addEventListener("click", dialogClickListener);
        optionsDialgoOrActionSheet.show();
    }
    Ti.API.info("tableClickListener() END");
}

function dialogClickListener(event){
    Ti.API.info("dialogClickListener() START");
    Ti.API.info("event index = "+event.index);
    var toast = null;
    switch(event.index){
        case 0: // OK button got clicked
            toast = Ti.UI.createNotification({duration:Ti.UI.NOTIFICATION_DURATION_LONG, message:"Ok button got clicked"});
        break;
        case 1: // Cancel button got clicked
            toast = Ti.UI.createNotification({duration:Ti.UI.NOTIFICATION_DURATION_LONG, message:"Cancel button got clicked"});
        break;
        case 2: // Help button got clicked
            toast = Ti.UI.createNotification({duration:Ti.UI.NOTIFICATION_DURATION_LONG, message:"Help button got clicked"});
        break;
    }
    if(toast != null)toast.show();
    Ti.API.info("dialogClickListener() END");
}
//------------------------------- PRIVATE END ----------------------------------------------
/*
 * Return the table
 */
this.getPoc = function(){
    Ti.API.info("getTableWithSections() START");
    var searchBar = Ti.UI.createSearchBar({barColor:"pink", showCancel:true});
    var table = Ti.UI.createTableView({headerTitle:TABLE_TITLE, footerTitle:TABLE_FOOTER, search:searchBar});
    table.addEventListener("click", tableClickListener);
    // Section for alerts.
    var sectionAlerts = Ti.UI.createTableViewSection({headerTitle:SECTION_TITLE_FOR_ALERTS});
    sectionAlerts.add(Ti.UI.createTableViewRow({title:ALERTS_NO_BUTTON, leftImage:TABLE_ROW_ICON, rightImage:TABLE_ROW_ICON}));
    sectionAlerts.add(Ti.UI.createTableViewRow({title:ALERTS_SINGLE_BUTTON, leftImage:TABLE_ROW_ICON, rightImage:TABLE_ROW_ICON}));
    sectionAlerts.add(Ti.UI.createTableViewRow({title:ALERTS_TWO_BUTTONS, leftImage:TABLE_ROW_ICON, rightImage:TABLE_ROW_ICON}));
    sectionAlerts.add(Ti.UI.createTableViewRow({title:ALERTS_THREE_BUTTONS, leftImage:TABLE_ROW_ICON, rightImage:TABLE_ROW_ICON}));
    table.appendSection(sectionAlerts);
    // Section for activity indicator.
    var sectionActivityIndicator = Ti.UI.createTableViewSection({headerTitle:SECTION_TITLE_FOR_ACTIVITY_INDICATOR});
    sectionActivityIndicator.add(Ti.UI.createTableViewRow({title:ACTIVITY_INDICATOR_WITHOUT_MESSAGE, leftImage:TABLE_ROW_ICON, rightImage:TABLE_ROW_ICON}));
    sectionActivityIndicator.add(Ti.UI.createTableViewRow({title:ACTIVITY_INDICATOR_WITH_MESSAGE, leftImage:TABLE_ROW_ICON, rightImage:TABLE_ROW_ICON}));
    table.appendSection(sectionActivityIndicator);
    // Section for option dialog/action sheet
    var sectionOptionDialogOrActionSheet = Ti.UI.createTableViewSection({headerTitle:SECTION_TITLE_FOR_OPTION_DIALOG_OR_ACTION_SHEET});
    sectionOptionDialogOrActionSheet.add(Ti.UI.createTableViewRow({title:OPTION_DIALOG_OR_ACTION_SHEET, leftImage:TABLE_ROW_ICON, rightImage:TABLE_ROW_ICON}));
    table.appendSection(sectionOptionDialogOrActionSheet);
    Ti.API.info("getTableWithSections() END");
    return table;
};

}
module.exports = TableAlertsActivityIndicatorActionSheet;
