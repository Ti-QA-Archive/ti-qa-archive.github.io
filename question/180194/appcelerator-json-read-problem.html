    Hi Erdem

    i made some change in you code. replace this code in your code and try

    var view = Ti.UI.createView();
var tabledata = [];
var tableView = Ti.UI.createTableView();
var xhr = Ti.Network.createHTTPClient({
    onload : function(e) {
        tabledata = JSON.parse(this.responseText);
        Ti.API.info('TABLE DATA = ' + JSON.stringify(tabledata));
        var arrData = [];

        for (var i = 0; i < tabledata.length; i++) {
            Ti.API.info('DATA - ' + tabledata[i].UrunAdi);
            arrData.push({
                title : tabledata[i].UrunAdi
            });
        }

        tableView.setData(arrData);

    },
    onerror : function(e) {
        Ti.API.debug(e.error);
        alert('error');
    },
    timeout : 5000
});
xhr.open("GET", "http://file.bimser.com.tr/DOSYALAR/G2/veriler.js");
xhr.send();
view.add(tableView);
 win.add(view);
win.open();

  • here you go tried with SiraNo you could do the rest

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});
var tabledata = [];
var tableView = Ti.UI.createTableView();
var xhr = Ti.Network.createHTTPClient({
    onload : function(e) {
        var results = JSON.parse(this.responseText);
        for (var i = 0; i < results.length; i++) {
            var row = Ti.UI.createTableViewRow({
                className : 'forumEvent', // used to improve table performance
                selectedBackgroundColor : 'white',
                rowIndex : i, // custom property, useful for determining the row during events
                height : 110
            });

            var labelUserName = Ti.UI.createLabel({
                color : '#576996',
                font : {
                    fontFamily : 'Arial',
                    fontSize : 12,
                    fontWeight : 'bold'
                },
                text : results[i].SiraNo,
                left : 70,
                top : 6,
                width : 200,
                height : 30
            });
            row.add(labelUserName);

            var labelDetails = Ti.UI.createLabel({
                color : '#222',
                font : {
                    fontFamily : 'Arial',
                    fontSize : 10,
                    fontWeight : 'normal'
                },
                text : 'Replied to post with id ',
                left : 70,
                top : 44,
                width : 360
            });
            row.add(labelDetails);

            var labelDate = Ti.UI.createLabel({
                color : '#999',
                font : {
                    fontFamily : 'Arial',
                    fontSize : 8,
                    fontWeight : 'normal'
                },
                text : "Results",
                left : 105,
                bottom : 10,
                width : 200,
                height : 20
            });
            row.add(labelDate);

            tabledata.push(row);
        }
        tableView.setData(tabledata);

    },
    onerror : function(e) {
        Ti.API.debug(e.error);
        alert('error');
    },
    timeout : 5000
});
xhr.open("GET", "http://file.bimser.com.tr/DOSYALAR/G2/veriler.js");
xhr.send();
win.add(tableView);
win.open();
