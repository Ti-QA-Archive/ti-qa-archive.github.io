    • Ok, but that link is over two years old and your code is not working which means that its now out of date.

      The current Facebook documentation states that you can publish links using the /{user-id}/feed so why are you debating it with me?

      I wrote some code a year ago posting to the feed with the following:

      var data = {
    link: '',
    name: '',
    message: '',
    caption: '',
    picture: '',
    description: ''
};

fb.requestWithGraphPath(
      'me/feed'
    ,  data
    , 'POST'
    , function(_e) {
        if (_e.success) {
            Ti.UI.createAlertDialog({
                  title : 'Facebook'
                , message : 'Your post has been successful.'
                , buttons: ['Ok']
            }).show();
        } else if (_e.error) {
            Ti.UI.createAlertDialog({
                  title : 'Facebook Error'
                , message : 'An error occured trying to post - please try again later.'
                , buttons: ['Ok']
            }).show();
    }
});

      which works and is live in the Chef Jimi App
