  • u can resize the photos and images directly from camera and gallery during u fetch them : just add some codes to ur success function of openPhotoGallery and showCamera method. THe code is here:

    var image = Ti.Media.openPhotoGallery({
success : function(e) {
var cropRect = e.cropRect;
var image = e.media;

var imageView = Titanium.UI.createImageView({
image : image
});
if(image.height >= 480) {
imageView.size.height = 480;
}
if(image.width >= 320) {
imageView.size.width = 320;
}
image = imageView.toBlob();
imageView.image = image;
},
cancel : function() {
},
error : function(e) {
alert('Some error has occured, please try again.');
}
});

    i hope this code will help u , coz it helped me alot.
