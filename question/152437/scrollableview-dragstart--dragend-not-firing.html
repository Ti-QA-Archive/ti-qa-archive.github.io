ScrollableView dragstart & dragend not firing?

Ti SDK 3.1.0 / Mac OSX 10.8.2

For some reason the dragstart and dragend don't fire?

Here's some code:

var self = Ti.UI.createWindow({
        backgroundColor:'#ffffff'
    });

    var backgroundMainImage = Ti.UI.createImageView({
        image : 'images//backgroundImage.png', width: '2020dp', height:'100%', top : 0, left : 0
    });
    self.add(backgroundMainImage);

    var view1 = Ti.UI.createView({ backgroundColor:'transparent' });
    var viewLbl1 = Ti.UI.createLabel({
        text : 'View 1', color : '#FFFFFF', font : {fontSize:15, fontWeight:'bold'}, textAlign : 'center'
    });
    view1.add(viewLbl1);

    var view2 = Ti.UI.createView({ backgroundColor:'transparent' });
    var viewLbl2 = Ti.UI.createLabel({
        text : 'View 2', color : '#FFFFFF', font : {fontSize:15, fontWeight:'bold'}, textAlign : 'center'
    });
    view2.add(viewLbl2);

    var view3 = Ti.UI.createView({ backgroundColor:'transparent' });
    var viewLbl3 = Ti.UI.createLabel({
        text : 'View 3', color : '#FFFFFF', font : {fontSize:15, fontWeight:'bold'}, textAlign : 'center'
    });
    view3.add(viewLbl3);

    var scrollableView = Ti.UI.createScrollableView({
      views:[view1,view2,view3], showPagingControl:false, width:'100%', height:'100%',
    });

    scrollableView.addEventListener('dragstart', function(e){
        Ti.API.info('Start evt ' + JSON.stringify(e)); // nottin' happens
    });
    scrollableView.addEventListener('dragend', function(e){
        Ti.API.info('End evt ' + JSON.stringify(e)); // nottin' happens
    });
    scrollableView.addEventListener('touchstart', function(e){
        Ti.API.info('Touch Start evt ' + JSON.stringify(e));
    });

    self.add(scrollableView);


    return self;
