Slideshow not working

Hi, I've been trying to get this piece of code to work but it doesn't show anything other than a blank white canvas.
I got it off this forum and I did not change anything else but only the relevant stuff but i still could not get it to work. could anyone explain why? a beginner here

 var win = Ti.UI.currentWindow;
win.backgroundColor = '#FFF';
var view = Titanium.UI.createImageView({
    top:0,
    width: 320,
    height:400,
    zindex: 30

});
win.add(view);

var slides = [];
var slide1 = {img: 'images/1.jpg', displayTime: 3000};
var slide2 = {img: 'images/1.jpg', displayTime: 3000};
var slide3 = {img: 'images/1.jpg', displayTime: 3000};
var slide4 = {img: 'images/1.jpg', displayTime: 3000}; // create in a Loop
slides.push(slide1);
slides.push(slide2);
slides.push(slide3);
slides.push(slide4);

var idx = 0;
var countSlides = slides.length;
var getNextSlide = function() {
idx = idx >= countSlides ? 0 : idx;
var slide = slides[idx++];
return slide;
};

var showSlide = function(slide) {
view.backgroundImage = slide.img;
setTimeout(function() {
showSlide(getNextSlide());
}, slide.displayTime);
};



showSlide(getNextSlide());
