here is my code.


function refreshChat () {
    var chats= Alloy.Collections.chatsModel;    
    var chatRefresh = Alloy.createModel('refreshChatModel');    
    chatRefresh.save( params );    
    var details = chatRefresh.fetch({
        success: function(collection, response, options) {             
            var msgs = response.data;
            for(var i=0;i<msgs.length;i++){
                message = msgs[i].message;
                var msgID = msgs[i].user_id;
                image = msgs[i].avatar;
                var payload ={
                    avatar : image,
                    message: message,
                    msgID:msgID
                }
            }
            if(msgs.length!=0){                    
                   chats.add(Alloy.createModel("chatsModel",payload));
            }        
           },
           error: function(collection, loginReq, options) {
               alert("error");
           }
   });         
}
