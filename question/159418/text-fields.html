  • Hi Reinel,

    Try this one :

    var cWin = Ti.UI.currentWindow;

var textField = Ti.UI.createTextField({
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    color : '#336699',
    top : 50,
    left : 10,
    width : 250,
    height : 60
});
var textField2 = Ti.UI.createTextField({
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    color : '#336699',
    top : (textField.top + textField.height) + 10,
    left : 10,
    width : 250,
    height : 60
});
var textField3 = Ti.UI.createTextField({
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    color : '#336699',
    top : (textField2.top + textField2.height) + 10,
    left : 10,
    width : 250,
    height : 60
});

cWin.add(textField);
cWin.add(textField2);
cWin.add(textField3);

var button = Ti.UI.createButton({
    title : 'PLAY',
    bottom : 10
});

cWin.add(button);

button.addEventListener('click', function() {
    alert("Your entered: " + 


    var text = " Your favourite games are " + textField.value + ", " + textField2.value + ", " + textField3.value);   //  your entered value in textFiled

    var googleUrl = 'http://translate.google.com/translate_tts?tl=en&q=' + text;
    var xhr = Titanium.Network.createHTTPClient();
    xhr.open("GET", googleUrl);
    xhr.setRequestHeader("Content-Type", "application/json");
    xhr.send();

    xhr.onerror = function(e) {
        alert('On Error');
    };

    xhr.onload = function() {
        var player = Ti.Media.createSound({
            url : googleUrl
        });
        player.play();
    };
});

    I hope this is helpful to you. thanks. :)
