  • Hi Maheish,

    Try this sample code and convert array into JSON string and while getting back convert string into json object that help you.Best luck 

    var dataArray = [];
for (var k = 0; k < 5; k++) {
    var dic = {
        myNote : 'xxx',
        firendnote : 'yyy'
    }
    dataArray.push(dic);
}
var JsonString = JSON.stringify(dataArray);
Ti.API.info(JSON.stringify(dataArray) + "  length=" + dataArray.length);
var objDict = JSON.parse(JsonString);
Ti.API.info(JSON.stringify(objDict) + "  length=" + objDict.length);
