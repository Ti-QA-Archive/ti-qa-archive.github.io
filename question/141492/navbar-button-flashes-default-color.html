    • I fix it by putting the bar color in the event listener. 

      win.searchNavButton.addEventListener('click', function(e)
{
    var winSearch = Titanium.UI.createWindow({
        url:'../search/icd9.js',
        title:'Search',
    });
    winSearch.backgroundImage = '../images/bg11.png';
        winSearch.barImage='../images/navbarbg.png';
        winSearch.barColor='#282828';
    Titanium.UI.currentTab.open(winSearch,{animated:true});
});
