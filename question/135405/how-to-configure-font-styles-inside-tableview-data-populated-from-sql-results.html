How to configure font styles inside tableview, data populated from sql results

Titanium Mobile sdk: 1.8.2

iPhone sdk 4.3

Hi,

I find it hard to apply font styles inside the tableview. Here is my code:

var query = db.execute('SELECT DISTINCT dateadded, dateclosed, qid FROM flags');
    var flagArray = [];

    while (query.isValidRow())
    {

        //add items to arrays
        flagArray.push({title: ''+query.fieldByName('qid')+' - '+query.fieldByName('dateadded')});
        query.next();
    };


    var resultTable = Ti.UI.createTableView({
        backgroundColor: 'transparent',
        borderRadius: 10,
        width: 225,
        height: 330,
        data: flagArray
    });

How can I achieve this? Thanks for your help

    You can set it in your table view row property.
    try to add label rather than set title in your talbe view row.

    flagArray= [];
while (query.isValidRow())
{
    var tRow = Ti.UI.createTableViewRow();
    var lblRowTitle = Ti.UI.createLabel({
        text : query.fieldByName('qid');
        left : 5 // your choice here.
        with : 50,
        font : { size : 14, fontWeight : 'bold' },
        color : "blue"
    });
    var lblRowDateAdded = Ti.UI.createLabel({
        text : query.fieldByName('dateadded');
        left : 60 // your choice here.
        with : 150,
        font : { size : 14, fontWeight : 'bold' },
        color : "black"
    });

    tRow.add(lblRowTitle);
    tRow.add(lblRowDateAdded);

    flagArray.push(tRow );

    query.next();
}
