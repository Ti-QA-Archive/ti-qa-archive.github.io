TableView Automatically refresh every 60 minutes

i have a TableView with a few rows, but that rows are updated in the "server" side. So i need a process in background or something like that to refresh the tableView every 60minutes. Any help ?

    You can use setInterval() to retrieve data from server side periodically, then setData() to render table with new data.

    var tv = Ti.UI.createTableView();
var win = Ti.UI.createWindow({ title:'TableView - Auto Refresh', backgroundColor:'#fff' });
    win.add(tv);
    win.open();

function fetchData() {
    var data = [];
    var l = Math.floor(Math.random() * 10);

    for ( var i = 0; i < l; i++ ) {
        var row = Ti.UI.createTableViewRow({title: 'Row ' + i});
        data.push(row);
    }
    return data;
}

function renderData() {
    tv.setData( fetchData() );
}

setInterval( renderData, 60000 );
