var result = db.execute('select ex, ey from GroundStrokeData');
if (result.isValidRow()){
    BallSpot1.YCor =  result.fieldByName('ey');         
    BallSpot1.top =  result.fieldByName('ey');  
    BallSpot1.XCor = result.fieldByName('ex');  
    BallSpot1.left = result.fieldByName('ex');  
}
