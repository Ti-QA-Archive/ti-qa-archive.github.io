  • You need change the atribute layou to
    layout:'vertical'

    var miWin = Ti.UI.createWindow({layout:'vertical'});
var header = Ti.UI.createLabel ({});
var viewBody = Ti.UI.createView({bottom:50,backgroundColor:'white'});
var footer = Ti.UI.createLabel({text:'footer',height:50});
 var url = "http://dominio.com";
 var client = Ti.Network.createHTTPClient({
     onload : function(e) {
         header.text = " " + this.responseText;
         miWin.add(header);
         miWin.add(viewBody);
         miWin.add(footer);
        miWin.open();
     },
     onerror : function(e) { Ti.API.debug(e.error); },
     timeout : 5000 
 });
 client.open("GET", url);
 client.send();
