  • Layer 1 ScrollableView (back)
  • Layer 2 ScrollView (front)

Example code

var win = Titanium.UI.createWindow();

// >> --------------------------- ScrollableView ------------------------------

var view4 = Titanium.UI.createView({ top:0, height:302, backgroundColor:'#fbf' });
view1.add(Titanium.UI.createLabel({text:'View1', top:0, color:'white'}));

view4.add(Titanium.UI.createLabel({text:'View4', top:0, color:'white'}));

var scrollableView = Ti.UI.createScrollableView({
    top:0,
    weidth:320,
    height:302,
    views:[view1, view2, view3, view4],
    showPagingControl:true,
    pagingControlColor:'transparent',
    pagingControlHeight:50
});
// << --------------------------- ScrollableView ------------------------------

// << --------------------------- ScrollView ----------------------------------
// Layer 2
var menuScrollView = Titanium.UI.createScrollView({

    top:300,

  • Hello Sej,

    See this code with click event for back layer…

    
var win = Titanium.UI.createWindow();

// >> --------------------------- ScrollableView ------------------------------
// Layer 1
var view1 = Titanium.UI.createView({ top:0, height:302, backgroundColor:'#123' });
var view2 = Titanium.UI.createView({ top:0, height:302, backgroundColor:'#246' });
var view3 = Titanium.UI.createView({ top:0, height:302, backgroundColor:'#48b' });
var view4 = Titanium.UI.createView({ top:0, height:302, backgroundColor:'#fbf' });
view1.add(Titanium.UI.createLabel({text:'View1', top:0, color:'white'}));
view2.add(Titanium.UI.createLabel({text:'View2', top:0, color:'white'}));
view3.add(Titanium.UI.createLabel({text:'View3', top:0, color:'white'}));
view4.add(Titanium.UI.createLabel({text:'View4', top:0, color:'white'}));

var scrollableView = Ti.UI.createScrollableView({
    top:0,
    weidth:320,
    height:302,
    views:[view1, view2, view3, view4],
    showPagingControl:true,
    pagingControlColor:'transparent',
    pagingControlHeight:50
});

scrollableView.addEventListener('click',function(e){
    alert('click');
});

// << --------------------------- ScrollableView ------------------------------

// << --------------------------- ScrollView ----------------------------------
// Layer 2
var menuScrollView = Titanium.UI.createScrollView({
    top:scrollableView.top + scrollableView.height + 1,
    contentWidth:'auto',
    contentHeight:'auto',
    borderWidth:1,
    borderColor:'red'
});

var menuView = Titanium.UI.createView({
    top:0,
    height:218,
    backgroundColor:'#ffe928'
});
// << --------------------------- ScrollView ----------------------------------

win.add(scrollableView);
menuScrollView.add(menuView);
win.add(menuScrollView);
win.open();
