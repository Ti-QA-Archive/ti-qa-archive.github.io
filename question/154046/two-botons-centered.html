    Maybe something like this:

    var win = Ti.UI.createWindow({
    backgroundColor: 'white'
});

var optionDialog = Ti.UI.createView({
    width: Ti.UI.SIZE,
    height: Ti.UI.SIZE,
    layout: 'vertical'
}); 

var label = Ti.UI.createLabel({
    text : 'Continue?',
    width : Ti.UI.SIZE
});
optionDialog.add(label);

var options = Ti.UI.createView({
    layout: 'horizontal',
    width: Ti.UI.SIZE,
    height: Ti.UI.SIZE,
    top: '16dp'
});
var btnYes = Ti.UI.createButton({
    title : 'Yes',
    width : Ti.UI.SIZE
});
options.add(btnYes);

var btnNo = Ti.UI.createButton({
    title : 'No',
    left: '16dp',
    width : Ti.UI.SIZE
});
options.add(btnNo);

optionDialog.add(options);
win.add(optionDialog);
win.open();
