    hi,

    you should create instance of row object and add it to tableview like

    
while(row.isValidRow())
{
    var row = Titanium.UI.createTableViewRow({
        title:'' +row.fieldByName('name')+'',
        hasDetail:true
    });

    data.push(row);
    row.next();
};

    so you want check mark in the same row but you are actually creating new row and adding that row in the tableview so just try below code

    tableView.addEventListener('click', function(e){

    var row = e.row;

    var iconImage = Titanium.UI.createImageView({
        image: 'images/check-mark.png',
        width: 50,
        height: 50,
        left: 10,
        top: 10 
    });
    row.add(iconImage);
});

    • so, you can add and hide this image at the time of creation then when check the row switch to show

      while(row.isValidRow())
{
    var row = Titanium.UI.createTableViewRow({        
        hasDetail:true
    });


row.iconImage = Titanium.UI.createImageView({
        image: 'images/check-mark.png',
        width: 50,
        height: 50,
        left: 10,
        top: 10,
    visible : false
    });
    row.add(row.iconImage);

    var lbl = Ti.UI.createLabel({
        text:'' +row.fieldByName('name')+'',
        left : 70,
        width : Ti.UI.SIZE,
        height : 40
    });
    row.add(lbl);

    data.push(row);
    row.next();
};

      and on row click….

      tableView.addEventListener('click', function(e){

    var row = e.row.iconImage;

    if(iconImage.visible){
        iconImage.visible = false;
    }else{
        iconImage.visible = true;        
    }
});
