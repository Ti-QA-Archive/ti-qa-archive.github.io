    Method native notification from function Tools

    this.nativeNotification = function(text,timeOut){
        /*
             Show a toast notification native from android 
             Parameters:
             *text -> message alert
             *timeOut->time duration to show notification, when time ends notification will fade out.
         * */
        var text      = text || "No message found";
        var timeOut   = timeOut || 1000;
        notificationWin = Ti.UI.createNotification({
            duration: timeOut,
            visible:false,
            bottom:'0dp',
            backgroundColor :'#0000',
            message:text,
            width:Ti.Platform.displayCaps.platformWidth,
            softKeyboardOnFocus:Titanium.UI.Android.SOFT_KEYBOARD_HIDE_ON_FOCUS
        });
        notificationWin.show();   
        // Play a device vibration.
        Ti.Media.vibrate(1000);        
    };
