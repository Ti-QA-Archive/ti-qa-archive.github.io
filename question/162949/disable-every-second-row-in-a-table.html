    please check the below code. It may one kind of solution for your problem if you like

    var aTableView = Ti.UI.createTableView();
var data=[];

for(var i=0;i<10;i++)
{
    var j=i+1;
     var row=Titanium.UI.createTableViewRow({
        height: Ti.UI.SIZE,
        title:'row'+j,
        name:i,
        backgroundColor:'white'
    });

    if(j%2==0)
    {
        row.focusable=false;
        row.touchEnabled=false;
        row.backgroundColor='#204581';
        data.push(row);
    }
    else
    {
        data.push(row);
    }

}

aTableView.addEventListener('click', function(e) {
    if(e.index%2==0)
    {
        alert('title: \'' + e.row.title + '\', section: \'' + e.section.headerTitle + '\', index: ' + e.index);
    }

});
aTableView.setData(data);
