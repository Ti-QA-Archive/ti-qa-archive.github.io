    I've re-written your whole code, tested and it works:

    var options_pageView = Ti.UI.createView({
    backgroundColor:'#fff'
});

var scrollView = Ti.UI.createScrollableView({
    views:[options_pageView],
    showPagingControl:true,
    pagingControlColor:'#999',
    pagingControlHeight:15,
    currentPage:0
});

var options_data = [];

var OptionData = [
    {title:'PW'}
];

var options_tableView = Ti.UI.createTableView({
    style:Titanium.UI.iPhone.TableViewStyle.GROUPED
});

for (var i=0;i<OptionData.length;i++) {
    var options_row = Ti.UI.createTableViewRow();

    var option_title = Ti.UI.createLabel({
        text:OptionData[i].title,
        left:6,
        top:'auto',
        height:'auto',
        width:'auto',
        font:{fontSize:16,fontWeight:'bold'}
    });

    var option_button = Ti.UI.createSwitch({
        right:10,
        value:false,
        enabled:true,
        data:OptionData[i]
    });

    option_button.addEventListener('change', function(e)
    {
        if (e.value) {
            Ti.App.Properties.setString('option_'+e.source.data.title, 'true');
            alert("True");
        } else if (!e.value) {
            Ti.App.Properties.setString('option_'+e.source.data.title, 'false');
            alert("False");
        } else {
            alert("Unknown");
        }
    });

    options_row.add(option_title);
    options_row.add(option_button);

    options_data.push(options_row);
};

options_tableView.setData(options_data);

options_pageView.add(options_tableView);

// BOTTOM OF PAGE
win.add(scrollView);
