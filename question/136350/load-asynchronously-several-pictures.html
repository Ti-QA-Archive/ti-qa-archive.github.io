win.add(scrollView);

for (var i=0; i<nb; i++)

  • Hi !

    I found the solution months ago but i forgot to post it, if someone is interested.

    I used a local webview :

    var f1 = Titanium.Filesystem.getFile(Titanium.Filesystem.resourcesDirectory, 'images', 'ActivityIndicator.gif');
var f2 = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory,'ActivityIndicator.gif');
f2.write(f1);

var webviewLoad = Titanium.UI.createWebView({
        top: '35%',
        left: '30%'
})
webviewLoad.html = "<html><head><meta name='viewport' content='width=device-width, minimum-scale=1.0, maximum-scale=2.0, user-scalable=yes' /></head><body><img src="+f2.nativePath+" /></body></html>";

view.add(webviewLoad);

    Hope this will help, it works great and it doesn't lag thanks to local gif.
