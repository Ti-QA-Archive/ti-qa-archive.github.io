    • This is all you have to do to check for the url and decide what to do:

      var tableview = Titanium.UI.createTableView({
    data:tableData
});

tableview.addEventListener('click', function(e) {
    if (e.rowData.url) {
        alert('Open WebView');
    } else {
        alert('Navigate to '+e.row.title);
    }
});
