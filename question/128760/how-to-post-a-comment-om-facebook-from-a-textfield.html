    the same way you post to a feed, the only difference is that you need to get the page's id

    function showRequestResult(e) {
    var s = '';
    if (e.success) {
        s = "SUCCESS";
        if (e.result) {
            s += "; " + e.result;
        }
        if (e.data) {
            s += "; " + e.data;
        }
        if (!e.result && !e.data) {
            s = '"success", but no data from FB.  I am guessing you cancelled the dialog.';
        }
    } else if (e.cancelled) {
        s = "CANCELLED";
    } else {
        s = "FAIL";
        if (e.error) {
            s += "; " + e.error;
        }
    }
    alert(s);
}


var page_id = "INSERT PAGE ID HERE";
var post_text = "TEST POST TO PAGE";
Titanium.Facebook.requestWithGraphPath(page_id + '/feed', {message: post_text}, "POST", showRequestResult);
