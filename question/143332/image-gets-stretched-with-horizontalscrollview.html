  • <HorizontalScrollView
    p1:layout_width="600dp"
    p1:layout_height="wrap_content"
    p1:fillViewport="false">
    <LinearLayout
    p1:orientation="vertical"
    p1:minWidth="25px"
    p1:minHeight="25px"
    p1:id="@+id/linearLayout2"
    p1:background ="@drawable/Map"
    p1:layout_width="500dp"
    p1:layout_height="fill_parent"
    p1:layout_marginTop="0dp"
    p1:layout_marginLeft="0dp"
    p1:layout_margin="0dp"
    p1:scrollbars="horizontal"
    p1:scrollbarStyle="insideOverlay"
    p1:visibility="visible">
