  • Hello Daniele,

    Thanks for raising this issue. You can obtain a table view with a left image (thumbnail) and when clicked on the row, image will be viewed full screen or any size as you need. If you click on fullscreen image, you will be back to tableview.
    Can you please try my sample code? Hope it helps.

    Check the attachment

    Test Case:

    app.js

    var win = Ti.UI.createWindow({
    backGroundColor:'pink'

});


// Create a TableView.
var aTableView = Ti.UI.createTableView();


// Add to the parent view.
win.add(aTableView);


var row = Ti.UI.createTableViewRow({
   height:'auto',

});


var imageView = Ti.UI.createImageView({
    image:'KS_nav_ui.png',
    top:0,
    left:0,

    preventDefaultImage:true,
    //transform:t2
});



 aTableView.addEventListener('click', function(e) {

     var win2 = Ti.UI.createWindow({});
     win2.open();

       var imageView2 = Ti.UI.createImageView({
           image:'KS_nav_ui.png',
            width:Ti.Platform.displayCaps.platformWidth,
            height:Ti.Platform.displayCaps.platformHeight


   });
    win2.add(imageView2);
    win2.addEventListener('click', function(e) {
          win2.close();



    });
});



row.add(imageView);
aTableView.appendRow(row);
win.open();
