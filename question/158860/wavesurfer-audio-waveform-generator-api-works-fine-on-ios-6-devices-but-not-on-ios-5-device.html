        var map = {
            32: 'play',       // space
            38: 'green-mark', // up
            40: 'red-mark',   // down
            37: 'back',       // left
            39: 'forth'       // right
        };
        if (e.keyCode in map) {
            var handler = eventHandlers[map[e.keyCode]];
            e.preventDefault();
            handler && handler(e);
        }
    });

    document.addEventListener('click', function (e) { 
        var action = e.target.dataset && e.target.dataset.action;
        if (action && action in eventHandlers) {
                 eventHandlers[action](e);
        }
    });
});
