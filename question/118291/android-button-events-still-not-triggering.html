    Mik

    The issue descibed in ticket #2844 has been resolved, and merged into both branches 1.6.X and 1.7.X.

    Please would you run the following code in your Android app in your environment, and report here whether you see the issue exhibited for either of the above branches?

    var win = Ti.UI.createWindow({ backgroundColor: '#fff' });

var label = Ti.UI.createLabel({
    text: 'Touch the Button!', textAlign: 'center',
    top: 0, right: 0, bottom: '50%', left: 0
});
win.add(label);

var button = Ti.UI.createButton({
    title: 'Update Label',
    top: '50%', right: 0, bottom: 0, left: 0
});
win.add(button);

var launch = new Date().getTime();
button.addEventListener('click', function() {
    label.text = parseInt((new Date().getTime() - launch) / 100) / 10 + ' seconds since app launched.';
});

win.open();
