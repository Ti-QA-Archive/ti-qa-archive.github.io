  • For my app I just close the tabgroup and basically reopen it…

    var win = Titanium.UI.currentWindow;
win.tabGroup.close();
var home = Titanium.UI.createWindow({  
    url:Titanium.Filesystem.resourcesDirectory + '/windows/home.js',
    title:"Home"
});
var homeTabs = Titanium.UI.createTabGroup(
{
    barColor:appBackgroundColor
});
var homeTab = Titanium.UI.createTab({title:'Home', window:home});
homeTabs.addTab(homeTab);
homeTabs.setActiveTab(0); 
homeTabs.open();

  • In app.js you can create your windows/tabs like so:

    createWindows: function () {

        winHome = new Window({ title: 'Home', url: "home.js" });
        winScores = new Window({ title: 'Scores', url: "scores.js" });
        winTeams = new Window({ title: 'Teams', url: "teams.js" });
        winSettings = new Window({ title: 'Settings', url: "settings.js" });

    },

    createTabs: function () {

        var tabGroup = new TabGroup();
        var tabHome = new Tab({ icon: path + 'images/KS_nav_views.png', title: 'Home', window: winHome });
        var tabScores = new Tab({ icon: path + 'images/KS_nav_ui.png', title: 'Scores', window: winScores });
        var tabTeams = new Tab({ icon: path + 'images/KS_nav_ui.png', title: 'Teams', window: winTeams });
        var tabSettings = new Tab({ icon: path + 'images/KS_nav_phone.png', title: 'Settings', window: winSettings });
        tabGroup.addTab(tabHome);
        tabGroup.addTab(tabScores);
        tabGroup.addTab(tabTeams);
        tabGroup.addTab(tabSettings);
        tabGroup.open();

        Ti.App.addEventListener('switchTab', function (e) {
            switch (e.tab) {
                case 'home': tabGroup.setActiveTab(tabHome); break;
                case 'scores': tabGroup.setActiveTab(tabScores); break;
                case 'teams': tabGroup.setActiveTab(tabTeams); break;
                case 'settings': tabGroup.setActiveTab(tabSettings); break;
            }
        });

    },

    And then within a certain window you can programmatically switch to another window like so:

    Ti.App.fireEvent('switchTab', { tab: 'teams' });
