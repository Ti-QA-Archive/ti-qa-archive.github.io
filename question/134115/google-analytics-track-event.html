Google Analytics track event

I have implemented Google Analytics(trackPageview and trackEvent) in my iPhone Application. It is working fine and I can able to see the pageviews in Google Analytics Dashboard. But I don't know where(how) to check track event in Google Analytics Dashboard.

Here is my sample code 

Titanium.include('analytics.js');
var analytics = new Analytics('UA-xxxxx-y');
Titanium.App.addEventListener('analytics_trackPageview', function(e){
    analytics.trackPageview( e.pageUrl);
}); 

Titanium.App.addEventListener('analytics_trackEvent', function(e){
    analytics.trackEvent(e.category, e.action, e.label, e.value);
});

Titanium.App.Analytics = {
    trackPageview:function(pageUrl){
        Titanium.App.fireEvent('analytics_trackPageview', {pageUrl:pageUrl});
    },
    trackEvent:function(category, action, label, value){
Titanium.App.fireEvent('analytics_trackEvent', {category:category, action:action, label:label, value:value});
    }
}

analytics.start(10);

Titanium.App.addEventListener('close', function(e){
    analytics.stop();
});

to track Event

win.addEventListener('click', function(e){
    Titanium.App.Analytics.trackPageview('win-name');
});

to track Pageview

win.addEventListener('focus', function(e){
      Titanium.App.Analytics.trackEvent('category','action','label', 'value');
});

Am I missing anything?
