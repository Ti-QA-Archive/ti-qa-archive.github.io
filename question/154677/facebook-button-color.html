  • var Btn = Titanium.UI.createButton({
            title:'I am button',
backgroundColor:'red'
        });
Btn.addEventListener('click',function(e){
var counter=0;
if(counter%2==0)
{
Btn.backgroundColor='black'
}
else
{
Btn.backgroundColor='red'
}
counter++;
});
