  • I don't know if this is the best way, but I was able to accomplish this by using a 3DMatrix. If anyone is interested, here is the code below.

    var tabBg = tabGroup.toImage();

var t1 = Ti.UI.create3DMatrix();
t1 = t1.rotate(180,0,1,0);

var bgImage = Ti.UI.createImageView({
    image:tabBg,
    top:0,bottom:0,left:0,right:0,
    transform:t1
});
