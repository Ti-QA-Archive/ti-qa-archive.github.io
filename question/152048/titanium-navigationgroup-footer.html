even when i navigate to other windows/views?

navGroup.open(win);

  • You can create a window with a toolbar and navbar hidden. Add a nav group inside window. You will have access to your toolbar always. The following code (Alloy View) will make you the way

    
<Alloy>
    <Window backgroundColor="#fff" navBarHidden="true">
        <NavigationGroup top="0" height="80%">
            <Window title="My Window On Nav Bar">
            </Window>
        </NavigationGroup>
        <Toolbar bottom="0" height="20%">
            <Items>
                <Button style="Ti.UI.iPhone.SystemButtonStyle.BORDERED">Toolbar Button</Button>
            </Items>
        </Toolbar>
    </Window>
</Alloy>

  • hi,
    try this demo

    var backWin = Ti.UI.createWindow({
    backgroundColor : 'red'
});
var lbl = Ti.UI.createLabel({
    bottom : 0,
    height : 50,
    text : 'Common View',
    width : Ti.UI.SIZE
});
backWin.add(lbl);

backWin.open();

var win = Ti.UI.createWindow({
    top : 0,
    height : 400
});

var self = Ti.UI.createWindow({
    backgroundColor : 'white',
    title : 'First'
});

var btn = Ti.UI.createButton({
    title : 'Next',
    width : 60,
    height : 40
});

btn.addEventListener('click', function(E) {
    var win2 = Ti.UI.createWindow({
        title : 'Second',
        backgroundColor : 'yellow'
    });
    nav.open(win2);
});

self.add(btn);

var nav = Ti.UI.iPhone.createNavigationGroup({
    window : self
});

win.add(nav);
win.open();
