    Hi ravi sharma,

    If you want to show any data in the listView. This is the process to add data in a listView, hear some sample code you can try this

    var win = Ti.UI.createWindow({backgroundColor: 'white'});
var listView = Ti.UI.createListView();
var sections = [];

var nameSection = Ti.UI.createListSection({ headerTitle: 'Name'});
var nameDataSet = [
    {properties: { title: 'baveen'}},
    {properties: { title: 'parveen'}},
     {properties: { title: 'rajat'}},
];
nameSection.setItems(nameDataSet);
sections.push(nameSection);

listView.sections = sections;
win.add(listView);
win.open();
