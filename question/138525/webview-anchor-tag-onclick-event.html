webview anchor tag onClick event

I can't able to add onClick event for anchor in webview

Here is my sample code

in win1.js

var extwebview = Ti.UI.createWebView({url:"external.html"});
win.add(extwebview);

in external.html

<html>
<head>
    <script type="text/javascript">  
    function externalLinks()
    {
      if (!document.getElementsByTagName) return;
      var anchors = document.getElementsByTagName("a");
      for (var i=0; i<anchors.length; i++)
      {

          var anchor = anchors[i];
          if(anchor.getAttribute("href")){
            anchor.target = "_blank";
            alert("index = "+i);
            anchor.onclick = function(){
                alert(i);
            };
       }

      }

    window.onload = externalLinks;
    </script>
</head>
<body>

  • Accepted Answer

    In your Titanium app, you'll need to listen to an event on the WebView:

    myWebView.addEventListener("linkClicked", function(url) {
    Ti.Platform.openURL(url);
});

    Then in the webview, add this snippet to a script tag:

    window.onload = function() {
    var links = document.querySelectorAll("a"),
        i = 0,
        len = links.length;
    while (i < len) {
        links[i++].addEventListener("click", function(evt) {
            evt.preventDefault();
            this.href && Ti.App.fireEvent("linkClicked", this.href);
        }, false);
    }
};

    This code is NOT tested, so you may need to make a few adjustments. Good luck! :)

  • You need to wrap the inside of the for loop in a closure like this:

    function externalLinks() {
    var links = document.querySelectorAll("a"),
        i = 0,
        len = links.length;
    for (; i < len; i++) {
        (function(link, j) {
            if (link.href) {
                link.target = "_blank";
                alert("index = " + j);
                link.addEventListener("click", function() {
                    alert(j);
                }, false);
            }
        }(links[i], i));
    }
}

    I prefer to use document.querySelectorAll() which is available in all modern browsers.

    Ivan's solution would also work, but you would need to wrap each async access to "i" in a closure where my example wraps everything.

  • More DOM friendly version:

    function externalLinks() {
    var anchors = document.links;
    for (var i=0; i < anchors.length; i++) {
        var anchor = anchors[i];
        if(anchor.href) {
            anchor.target = "_blank";
            alert("index = "+i);
            var listener = (function(i) {
                return function(){
                            alert(i);
                        };
            )(i);
            anchor.addEventListener('click', listener, false);
       }

      }
    }
