  • Not sure if this is what you're trying to do, but what I do is I declare two tabGroups and when one closes I open the other one. I do this in app.js as well as in other windows and has been working ok.

    
var tabGroup1 = Titanium.UI.createTabGroup({id: 'tabGroup1' });
...
add tabs as you normally would to tabGroup1

var tabGroup2 = Titanium.UI.createTabGroup({id:'tabGroup2'}):
add tabs as you normally would to tabGroup2

tabGroup1.addEventListener('close', function(e) {
   tabGroup2.open();
});
tabGroup2.addEventListener('close', function(e) {
   tabGroup1.open();
});
tabGroup1.open();
