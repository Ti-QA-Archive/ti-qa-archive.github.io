Scroll view not scrolled when setting the height of the child view to Ti.UI.SIZE Titanium

I have a scroll view which has a content height to auto
and inside of it i have a view which contain all the UI element i'm using
but my problem is when i set the height of the view to Ti.UI.SIZE

the scroll view not scrolled it's just scrolled in case of i add a fixed number e.g. 2000 to the child view height

Here is the code

var scrollView = Ti.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    backgroundColor : '#fff',
    showVerticalScrollIndicator : false,
    showHorizontalScrollIndicator : false,
    height : "100%",
    width : "100%",
    top : 36.6
});

var view = Ti.UI.createView({
    backgroundColor : '#fff',
    borderRadius : 0,
    height :Ti.UI.SIZE,
    width : Ti.UI.FILL
});

i don't any extra space at the bottom of the view so how i can accomplish this ,and thanks in advance

  • Hi Tony,

    Your code is working fine with me. but i change some property for scrollView because you tried to set height:100% and also you set it's top:36.6 so it slide cut at bottom side means 36.6 because scrollview height is '100%' so i put it into comment and add bottom property. and rest of thing working fine. if you have facing same issue again then check your child views means how's you add? what is the height and top position of it… 

    var scrollView = Ti.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    backgroundColor : '#fff',
    showVerticalScrollIndicator : false,
    showHorizontalScrollIndicator : false,
    // height : "100%",
    width : "100%",
    top : 36.6,
    bottom : 0
});
$.winIndex.add(scrollView);

var view = Ti.UI.createView({
    backgroundColor : '#fff',
    borderRadius : 0,
    height :Ti.UI.SIZE,
    width : Ti.UI.FILL
});
var button = Titanium.UI.createButton({
    top : 50,
    height : 300,
    width : 300,
    backgroundColor : "red",
});
view.add(button);

var button = Titanium.UI.createButton({
    top : 400,
    bottom : 20,
    height : 300,
    width : 300,
    backgroundColor : "green",
});
view.add(button);
scrollView.add(view);
