    simple you can do this 

    var pic=Ti.Filesystem.getFile('/Users/username/Library/Application Support/iPhone Simulator/6.1/Applications/C4827464-DCE5-455A-A339-FE16A72AB890/Documents/userpics/803dda3b-454e-ce0b-340d-b5ade3c8d409.jpg').read();//This will return you a blob 
$.userpic.image = pic;//set the blob as the image of the image view
