  • here is a little bit more code as asked : 

    exports.get_window = function(pictures) {

    var picStack = [];
    var len = -42;
    var image_file;
    var new_image = null;
    var new_view = null;
    var viewStack = [];

    var win = Ti.UI.createWindow({
        title:'main window',
        backgroundColor:'#0c4475',
        //layout:'vertical',
    });

    var win2 = Ti.UI.createWindow({
        title:'display pictures',
        backgroundColor:'#0c4475',
        //layout:'vertical',
    });


    var button = Titanium.UI.createButton({
        title:"",
        focusable:true,
        backgroundImage:"appareil_photo.png",
        backgroundSelectedImage:"appareil_photo2.png",
        width:'420dp',
        height:'200dp',
        top:'205dp',
        zIndex:2,
    });

    win.add(button);


    button.addEventListener("click", function(e){
        Titanium.Media.showCamera({
            success:function(e){
                if (e.mediaType === Titanium.Media.MEDIA_TYPE_PHOTO) {
                    image_file = e.media;
                    picStack.push(image_file);
                }
            },
            error:function(e){
                alert('ERROR ERROR ERROR');
            },
            cancel:function(e){
                alert('The event was cancelled :\'(');    
            },
            allowEditing:true,
            saveToPhotoGallery:false,
            mediaTypes:Titanium.Media.MEDIA_TYPE_PHOTO,
        });
    });


    var display_pictures = Titanium.UI.createButton({
        title:"",
        focusable:true,
        width:'50dp',
        height:'50dp',
        top:'205dp',
        left:'5dp',
        zIndex:4,
    });

    win.add(display_pictures);


    display_pictures.addEventListener("click", function(e){

        if (picStack.length !== len) {

            viewStack = null;
            viewStack = [];
            for (i = 0; i < picStack.length; i++) {
                new_image = Ti.UI.createImageView({ image:picStack[i] });
                new_view = Ti.UI.createView({ backgroundColor: '#0c4475'});
                new_view.add(new_image);
                viewStack.push(new_view);
            }

            var scrollableView = Ti.UI.createScrollableView({  
                 views:viewStack,
                  showPagingControl:true
            });
            win2.add(scrollableView);

            len = picStack.length;
        }
        win2.open();
    });


    var exit = Titanium.UI.createButton({
        title:"exit",
        focusable:true,
        width:'200dp',
        height:'100dp',
        bottom:'0dp',
        zIndex:4,
    });

    win2.add(exit);

    exit.addEventListener("click", function(e){
        win2.close();
    });

    win.open();

    return win;
};
