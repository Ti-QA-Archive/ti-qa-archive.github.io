        var myLongitude = response.longitude;
        var myLatitude = response.latitude;
        Ti.API.info(myLongitude + " " + myLatitude + " Plotted");
        //
        // CREATE ANNOTATIONS
        //
        // This is where we are going.
        var myLocation = Titanium.Map.createAnnotation({
            latitude : myLatitude,
            longitude : myLongitude,
            //title:myPoint[c].name,
            //subtitle:myPoint[c].address+" "+myPoint[c].city,
            pincolor : Titanium.Map.ANNOTATION_RED,

        mapview.addAnnotation(myLocation);

        // will set region to last item in list
        mapview.setRegion({
            latitude : myLatitude,
            longitude : myLongitude,
            latitudeDelta : 0.5,
            longitudeDelta : 0.5
        })

    });
}

  • Hi Muhammad,

    You can do that using this code : 

    var annotationList = "here your annotation list"

var anno = [], j = 0;
for (var i = 0; i < annotationList.length; i++) {

    var xhr = Titanium.Network.createHTTPClient();

    xhr.open('GET', 'http://maps.googleapis.com/maps/geo?output=json&q=' + annotationList[i]);
    xhr.onload = function() {
        var json = JSON.parse(this.responseText);
        Ti.API.info(json.Placemark[0].Point.coordinates[1]);
        Ti.API.info(json.Placemark[0].Point.coordinates[0]);
        anno.push(Ti.Map.createAnnotation({
            latitude : json.Placemark[0].Point.coordinates[1],
            longitude : json.Placemark[0].Point.coordinates[0],
            image : 'pics/pin.png',
            animate : true
        }));
        j++;
        if (i == j) {
            mapview.annotations = anno;
        }
    };
    xhr.send();
}
