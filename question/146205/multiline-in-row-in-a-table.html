  • hi,

    this is not a multi line but there are two labels added in a row so you can do that by adding any controls in a row . just create a row and set layout property to vertical or else you can use top property to set object's alignment and then add that in row. simple :)

    var tableData = [];

var win = Ti.UI.createWindow({ backgroundColor: 'white' });

var table = Ti.UI.createTableView({ objName: 'table' });

for (var i = 0; i <= 20; i++){
  var row = Ti.UI.createTableViewRow({
    className: 'row',
    objName: 'row',
    touchEnabled: true,
    height: 100
  });

  var enabledWrapperView = Ti.UI.createView({
    backgroundColor:'#008FD5',
    objName: 'enabledWrapperView',
    rowID: i,
    width: Ti.UI.FILL,
    height:Ti.UI.SIZE,
layout : 'vertical'

  });

  var disabledWrapperView = Ti.UI.createView({
    backgroundColor:'#A2E0FF',
    objName: 'disabledWarpperView',
    touchEnabled: false,
    width: 300, height: '80%'
  });
  enabledWrapperView.add(disabledWrapperView);

  var labe1 = Ti.UI.createLabel({
    top :0,
    backgroundColor:'#313F48',
    color: 'white',
    objName: 'label',
    text: i + '>>',
    touchEnabled: false,
    left: 0,
    width: 200
  });
  disabledWrapperView.add(label1);

  var labe2 = Ti.UI.createLabel({
    top :0,
    backgroundColor:'#313F48',
    color: 'gray',
    objName: 'label',
    text: 'Detail : ' + i,
    touchEnabled: false,
    left: 0,
    width: 200
  });
  disabledWrapperView.add(label2);

  row.add(enabledWrapperView);
  tableData.push(row);
}

table.setData(tableData);

win.add(table);
win.open();
