        tabGroup.addTab(search_tab);
        tabGroup.addTab(more_tab);

tabGroup.addEventListener('click', function(e) {
                Ti.API.info("focus tab index:" + e.index);
                if(e.previousTab) {
                    // Ti.API.info("Previous");
                    e.previousTab.window.remove(e.previousTab.window.children[0]);
                }
                if(e.tab) {
