remove function doesn't remove the view from UI in Android

Hi,

I am developing a Android app with Titanium SDK 3.2.X, Where we can add and remove certain feature from the list, I am making use of view's to display the feature to the user, When the user doesn't want to see the particular feature he can remove it by tapping on the 'X' mark (image) within that view, but after tapping on X mark, the view is not getting removed in Android, but it does remove the view in iOS app. here is the code for this.

        var viewClose = Ti.UI.createView({
            top : 5,
            left : 5,
            width : 130,
            height : 160
        });
        var X = Ti.UI.createImageView({
            image : "/media/wrong.png",
            bubbleParent : false,
            right : 0,
            top : 0,
            height : 18,
            width : 18,
            zIndex : 100
        });
        X.addEventListener("click", function(e) {
            Ti.API.info(e.source.parent);  // Trace points
            viewRow.remove(e.source.parent);
            Ti.API.info(e.source.parent);  // Trace points
            tableView.height = Ti.UI.SIZE;
        });
        viewClose.add(X);

The trace says that,

    Hello Maruti,

    Have you try custom Property like below :

    Try this :

    var viewClose = Ti.UI.createView({
            top : 5,
            left : 5,
            width : 130,
            height : 160
        });
        var X = Ti.UI.createImageView({
            image : "/media/wrong.png",
            bubbleParent : false,
            right : 0,
            top : 0,
            height : 18,
            width : 18,
            zIndex : 100,
            parentView : viewClose   //  Custom Property
        });
        X.addEventListener("click", function(e) {
            Ti.API.info(e.source.parentView);  // 
            viewRow.remove(e.source.parentView);
            Ti.API.info(e.source.parentView);  //  
            tableView.height = Ti.UI.SIZE;
        });
        viewClose.add(X);

    i hope this helps. thanks.

      var win = Ti.UI.createWindow({

        right : 0,

    X.addEventListener("click", function(e) {
        Ti.API.info(e.source.parentView);
        //
        e.source.rowView.remove(e.source.parentView);

    viewClose.add(X);

    row.add(viewClose);
    arrData.push(row);

}
