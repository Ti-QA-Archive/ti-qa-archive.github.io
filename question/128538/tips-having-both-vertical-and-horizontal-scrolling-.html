[Tips] Having BOTH vertical AND horizontal scrolling !

The documentation for the scrollView says that it's not possible on Android to have both vertical and horizontal scrollType. There is a workaround ;)

//
// create base UI tab and root window
//
var win = Titanium.UI.createWindow();

var viewImageTram = Ti.UI.createImageView({
    image : 'images/tram.png'
});

var scrollViewTramH = Titanium.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    top : 0,
    scrollType : 'horizontal',
    showHorizontalScrollIndicator : true,
    zoomScale : 2,
    maxZoomScale : 30,
    minZoomScale : 1
});
scrollViewTramH.add(viewImageTram);

var scrollViewTramV = Titanium.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    top : 0,
    scrollType : 'vertical',
    showVerticalScrollIndicator : true,
    zoomScale : 2,
    maxZoomScale : 30,
    minZoomScale : 1
});
scrollViewTramV.add(scrollViewTramH);

win.add(scrollViewTramV);


win.open();

It works on android ! I don't know for iPhone.
