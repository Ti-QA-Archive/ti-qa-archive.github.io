  • Hi Jez,

    Try this as per your requirement set your height, width as well as top, left.

    var win = Ti.UI.createWindow({
    title : 'Screen Resolution'
});

var screenWidth =  Ti.Platform.displayCaps.platformWidth;
var screenHeight = Ti.Platform.displayCaps.platformHeight;

var Tempview = Ti.UI.createLabel({
    text : 'Your Text',
    height : GetHeight(100), 
    width : GetWidth(100),
});


function GetHeight(value) {
    var temp = (value * 100) / 480;
    return parseInt((screenHeight * temp) / 100);
}

function GetWidth(value) {
    var temp = (value * 100) / 320;
    return parseInt((screenWidth * temp) / 100);
}

win.add(Tempview);


Ti.API.info("Screen Size :" + screenWidth + "x" + screenHeight);
Ti.API.info("Values of top :" + Tempview.top + " " + Tempview.left);

win.open();
