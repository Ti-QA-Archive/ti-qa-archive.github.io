[BUG] Toolbar changes height in landscape mode

Below is a simple example of a toolbar bug. When you have a toolbar on a window and you go from portrait to landscape, the toolbar gets shorter. This presents a UI problem if you're using exact positioning and also makes buttons look dumb as they could get cut off.

// this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#fff');

//
// create base UI tab and root window
//
var win1 = Titanium.UI.createWindow({
    top : 0,
    left : 0,
    right : 0,
    bottom : 0
});

win1.orientationModes = [Titanium.UI.PORTRAIT, Titanium.UI.UPSIDE_PORTRAIT, Titanium.UI.LANDSCAPE_LEFT, Titanium.UI.LANDSCAPE_RIGHT];

var testBtn1 = Titanium.UI.createButton({
    title: 'Test 1',
    style: Titanium.UI.iPhone.SystemButtonStyle.DONE
});

var testBtn2 = Titanium.UI.createButton({
    title: 'Test 2',
    style: Titanium.UI.iPhone.SystemButtonStyle.DONE
});

var flexSpaceButton = Titanium.UI.createButton({
    systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

var toolbar = Titanium.UI.iOS.createToolbar({
    height : 'auto',
    bottom : 0,
    items : [testBtn1, flexSpaceButton, testBtn2],
    borderTop : false,
    borderBottom : false,
    barColor : '#000',
    translucent : false,
    borderWidth : 2,
    borderColor : 'red'
});

win1.add(toolbar);
win1.open();
