Whats wrong with "for" loop?

Below code I can get xml response. When I try to get a single element I can get it. But in a for loop iphone and android says:"Null pointer exception". What is wrong with this code?

suds.invoke('QueryService', callparams, function(xmlDoc) {
            var data = [];
            var rows = xmlDoc.documentElement.getElementsByTagName("row");
                for(var i=0;rows.length;i++){
                    var row = rows.item(i);
                    data.push({
                            city:row.getElementsByTagName("City").item(0).text,
                            store_name:row.getElementsByTagName("Store_name").item(0).text,
                            sales_revenue:row.getElementsByTagName("Sales_revenue").item(0).text,
                            quantity_sold:row.getElementsByTagName("Quantity_sold").item(0).text,
                            lines:row.getElementsByTagName("Lines").item(0).text
                    });
                }
        });
