    Hi
    you trying to add the button like as in iphone navigation buttons? Well then its not possible for android For that you have to make a view like that see here how you can do that for android.

    // main view for window
var mainView = Ti.UI.createView({
    width: Titanium.Platform.displayCaps.platformWidth,
    height: Titanium.Platform.displayCaps.platformHeight,
    backgroundColor:'#ffffff',
    top:0,
    left:0
});
win.add(mainView);

//topview for back button and done button
var topview = Ti.UI.createView({
    width: Titanium.Platform.displayCaps.platformWidth,
    height:40,
    backgroundImage:'../images/TopHeader.png',
    top:0    
});
mainView.add(topview);

// back button
var btnBack = Ti.UI.createButton({
    title: L("Back"),
    backgroundImage:'../images/navButton.png',
    color:'#ffffff',    
    top:7,
    left:5,
    height:25,
    width:60
});
topview.add(btnBack);

    Well if i have misunderstand the question then let me know.
