iAd wont display a second time

Hi, I have an iAd that loads fine, if an error occurs it hides itself fine. But in the case where it loads, then an error occurs and then loads again the screen repositions itself but the iAd never displays.

Here's the code, anyone see anything wrong:

        var iAd = Ti.UI.iOS.createAdView({
         width: 'auto',
         height: 50,
         top:0
        });

        iAd.addEventListener('error',function(e){
            Ti.API.info(e.type);
            Ti.API.info(e.message);
            if (respacedForAd) {
                Ti.API.info("returning nav bar");
                win.showNavBar();
                headerView.top = headerView.top - 50;
                wrapperView.top = wrapperView.top - 50;
                tableview.bottom = tableview.bottom -6;
                respacedForAd =false;
            }
        });

        iAd.addEventListener('load',function(e){
            Ti.API.info(e.type);
            if (!respacedForAd) {
                Ti.API.info("removing nav bar");
                win.hideNavBar();
                headerView.top = headerView.top + 50;
                wrapperView.top = wrapperView.top + 50;
                tableview.bottom = tableview.bottom +6;
                iAd.top = 0;
                iAd.height=50;
                respacedForAd =true;
            }
        });

        win.add(iAd);
        iAd.show();
