    I think you want to list some items on a table and a button at the bottom. If I am right then you have two options to acheive this:

    1. Place both table and button in a vertical layout then add data to table from web service like following:

      var win = Ti.UI.createWindow({
      title:'Append Row',
      layout: 'vertical'
      });

    var table = Ti.UI.createTableView({
    height: Ti.UI.SIZE
    });
    win.add(table);

        var pageNum =1;
    var tableRow=[];

    var button = Ti.UI.createButton({
    title:'Next Page'
    });
    win.add(button);

    function getData(){

        var client=Ti.Network.createHTTPClient({

        onload:function(e){

        var jsonString =JSON.parse(this.responseText);


        for(var i=0;i < jsonString.files.length;i++){
            var row = Ti.UI.createTableViewRow();

            var label=Ti.UI.createLabel({
                text:jsonString.files[i].filename,left:5
            });

                image:jsonString.files[i].thumb,right:5, width:50, height:50
            });

            row.add(img);
            row.add(label);
            tableRow.push(row);
            }
            table.setData(tableRow);
    },

client.open('GET','http://gallery.mobile9.com/c/ringtones_4157/1/?view=json&pg='+pageNum);

2. Add the button in the onload callback after the table rows are displayed.
Actually the HTTPClient works asynchronously i.e it sends your request on network then do not wait for the response and next statements are executed so the code that should execute later executes first and the one which you suppose to be executed first, actually executes later that is why you always see the button on top because table is populated in onload response while the button is added to table outside of the HTTPClient response. You control this behavior by passing callback function to the data fetching request which (the callback function) is executed when network response is received. Try the following:

    var win = Ti.UI.createWindow({
    title:'Append Row'
    });

        var pageNum =1;
    var tableRow=[];

    var addButton = function(){
    var row2 = Ti.UI.createTableViewRow();

    var button = Ti.UI.createButton({
    title:'Next Page'
    });

    button.addEventListener('click',function(e){
    pageNum++;
    getData();
    });

    row2.add(button);
    tableRow.push(row2);
    }

    function getData(callback){

        var client=Ti.Network.createHTTPClient({

        onload:function(e){

        var jsonString =JSON.parse(this.responseText);


        for(var i=0;i < jsonString.files.length;i++){
            var row = Ti.UI.createTableViewRow();

            var label=Ti.UI.createLabel({
                text:jsonString.files[i].filename,left:5
            });

            var img = Ti.UI.createImageView({
                image:jsonString.files[i].thumb,right:5, width:50, height:50
            });

            row.add(img);
            row.add(label);
            tableRow.push(row);
            }
    callback && callback(/*json data*/);    // You can also pass the json data back to the callback as parameter.
            table.setData(tableRow);
    },
    onerror:function(e){
        alert('Connection Fail');
    },
    timeout:20000
});

client.open('GET','http://gallery.mobile9.com/c/ringtones_4157/1/?view=json&pg='+pageNum);
client.send();

    };

    var table = Ti.UI.createTableView();

    win.add(table);

    win.addEventListener('open',function(e){
    getData(addButton); // Pass the addButton function as callback to getData. It will be executed when the server response is received and rows are created.

    });
    win.open();
    ~~~~
