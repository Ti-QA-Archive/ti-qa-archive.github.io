  • hate to gravedig, but I was looking too, couldn't find anything so I wrote one. It formats inputs to American 10 digit phone numbers and blurs the field when they get to 10 digits automagically.

                phone.addEventListener('change', function(e) {
                var p = e.source.value;
                p = p.replace('(','').replace(')','').replace(' ','').replace('-','');
                var pl = p.length;
                if(pl >= 10) {
                    e.source.value = '('+p.substr(0,3)+') '+p.substr(3,3)+'-'+p.substr(6,4);
                    e.source.blur();
                } else if(pl >= 6) {
                    e.source.value = '('+p.substr(0,3)+') '+p.substr(3,3)+'-'+p.substr(6,4);
                } else if(pl >= 3) {
                    e.source.value = '('+p.substr(0,3)+') '+p.substr(3,3);
                } else {
                    e.source.value = p;
                }
            });

    Hope it helps someone else.

    • Thanks for sharing, this helped. I made a minor change to allow the user to back out the phone number through the formatting characters if they make a mistake.

      fields.phoneField.addEventListener('change', function(e) {
        // Get current value
        var p = e.source.value;
        // Convert it to the pure number
        p = p.replace('(','').replace(')','').replace(' ','').replace('-','');
        var pl = p.length;
        if(pl >= 10) {
            // This is too long for a US phone number, blur the field
            e.source.value = '('+p.substr(0,3)+') '+p.substr(3,3)+'-'+p.substr(6,4);
            e.source.blur();
        } else if(pl >= 7) {
            // Needs a dash
            e.source.value = '('+p.substr(0,3)+') '+p.substr(3,3)+'-'+p.substr(6,4);
        } else if(pl >= 4) {
            // Doesn't need dash yet
            e.source.value = '('+p.substr(0,3)+') '+p.substr(3,3);
        } else {
            // Not enough characters for formatting yet
            e.source.value = p;
        }
    });

  • Number format of (XXX) XXX-XXXX including support for backspace.

    var prevLength = 0;

textFieldPhoneNo.addEventListener('change', function(e) {
    var p = e.source.value;
    var pl = p.length;
    if (prevLength <= pl) {
        if (pl == 1) {
            e.source.value = '(' + p;
        } else if (pl == 4) {
            e.source.value = p + ') ';
        } else if (pl == 9) {
            e.source.value = p + '-';
        }
    } else {
        if ((pl == 1) || (pl == 9)) {
            e.source.value = e.source.value.substr(0, e.source.value.length - 1);
        } else if (pl == 5) {
            e.source.value = e.source.value.substr(0, e.source.value.length - 2);
        }
    }
    prevLength = p.length;
});
