    Hello krupa,

    Try this something like this:

    <Alloy>
    <Window id="win" layout="vertical">
        <View id="firstView" onClick="changeFirstView" top="0">
            <Image ...../>
            <Label ...../>
        </View>
        <Label top="0" id="lblFirst" height="0" objVisible="false" text="LOrum ipsum LOrum ipsumLOrum ipsumLOrum ipsumLOrum ipsumLOrum ipsum"/>
    </Window>
</Alloy>

    JS file

    function changeFirstView(e){
    if($.lblFirst.objVisible == true)
    {
        $.lblFirst.height = 0;
        $.lblFirst.objVisible = false;
    }
    else
    {
        $.lblFirst.height = Ti.UI.SIZE;
        $.lblFirst.objVisible = true;
    }
}
