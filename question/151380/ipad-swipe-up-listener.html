    var touchStartPosition;
var self = Ti.UI.createWindow({backgroundColor:'#FFF'});
self.addEventListener('touchstart', function(e) {
    touchStartPosition = e.x;
});
self.addEventListener('touchend', function(e) {
    if (touchStartPosition < e.x) {
        alert('User has swiped down!')
    } 
    if (touchStartPosition > e.x) {
        alert('User has swiped up!');
    }
});
return self;
