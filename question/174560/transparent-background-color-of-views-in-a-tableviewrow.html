Transparent background Color of views in a tableViewRow

How to set transparent background color for different views in a table view row ? I have 12 green view in a row. if i click on a green view then all the green view's (12 views/parent view) background color change to white. i want to make sure if i click on a view then only its background color will be change, other sibling view's background color will be transparent. how can i do that ? I tried like below but that does not work. 

var table = Ti.UI.createTableView({
        height : tableViewHeight,
        left : '5dp',
        right : '5dp',
        top : tableTop,
        borderRadius : 5
    });

    function color_view() {
        var _color = Ti.UI.createView({
            backgroundColor : 'green',
            selectedBackgroundColor : 'transparent',// this line not working
            height : Ti.UI.FILL,
            width : '20dp',
            borderWidth : 1,
            borderColor : 'silver'
        });

        return _color;
    }

    function customViewRow() {
        var row = Ti.UI.createTableViewRow({
            height : '40dp'
        });

        var colorHolder = Ti.UI.createView({
            layout : 'horizontal',
            left : '5dp',
            height : '35dp',
            width : Ti.UI.FILL,
            borderWidth : 1,
            borderColor : 'silver'
        });

        for (var i = 0; i < 12; i++) {
            colorHolder.add(color_view());
        }

        row.selectedBackgroundColor = 'transparent';
        row.add(colorHolder);
        row.className = 'school';
        table.appendRow(row);
    }

    customViewRow();

    Hi,

    I have just copy and paste your code and also added some line to complete the task.So please check the following code. I hope, it will help you. I have tested the code in Android emulator. 

    var win=Titanium.UI.createWindow({ 
    backgroundColor:'#000',
    title:'Testing', 
    layout:'vertical' 
}); 

var table = Ti.UI.createTableView({
        //height : tableViewHeight,
        left : '5dp',
        right : '5dp',
        top : 10,
        borderRadius : 5
    });

    function color_view(id) {
        var _color = Ti.UI.createView({
            backgroundColor : 'green',
            backgroundSelectedColor : 'transparent',
            height : Ti.UI.FILL,
            width : '20dp',
            borderWidth : 1,
            borderColor : 'silver',
            click:'view',
            id:id
        });
        return _color;
    }

    function customViewRow() {
        var row = Ti.UI.createTableViewRow({
            height : '40dp'
        });

        var colorHolder = Ti.UI.createView({
            layout : 'horizontal',
            left : '5dp',
            height : '35dp',
            width : Ti.UI.FILL,
            borderWidth : 1,
            borderColor : 'silver'
        });

        for (var i = 0; i < 12; i++) {
            colorHolder.add(color_view(i));
        }

        row.selectedBackgroundColor = 'transparent';
        row.add(colorHolder);
        row.className = 'school';
        table.appendRow(row);
    }

    customViewRow();
    table.addEventListener('click',function(e){
        if(e.source.click=='view')
        {
            var id=e.source.id;
            for (var i=0; i < e.row.children[0].children.length; i++) {
                if(e.row.children[0].children[i].id==id)
                {
                    e.row.children[0].children[i].backgroundColor='#fff';
                }
                else
                    e.row.children[0].children[i].backgroundColor='transparent';
            };
        }

    });
win.add(table);
win.open();
