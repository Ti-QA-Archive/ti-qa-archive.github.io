    Hi richards i found your solution i am giving you the code for feeding data in one tableView from different feeds, here is the basic code and you can modify it by adding more fields or values you want

    var dataContext = {};
var win = Ti.UI.createWindow({
    height : 480 - 15,
    width : 320,
    backgroundColor : 'white'
});

var TableView = Ti.UI.createTableView({
    height : win.height,
    width : win.width,
    left : 0,
    top : 0
});
win.add(TableView);
var tableData = [];
function TableRows1() {
    var xhr = Ti.Network.createHTTPClient();
    xhr.onload = function() {

        var doc = this.responseXML.documentElement;
        var items = doc.getElementsByTagName("item");
        var x = 0;
        var doctitle = doc.evaluate("//channel/title/text()").item(0).nodeValue;

        dataContext.function1Length = items.length;
        Ti.API.info('Data Context:==================' + dataContext.function1Length);
        for (var i = 0; i < items.length; i++) {
            var item = items.item(i);

            var title = doc.evaluate("//channel/title/text()").item(0).nodeValue;

            var row = Ti.UI.createTableViewRow({
                height : 40,
                width : 320,
                backgroundColor : 'white'
            });
            var label = Ti.UI.createLabel({
                text : title + ' ' + i,
                font : {
                    fontSize : 13,
                    fontWeight : 'bold'
                },
                color : 'black',
                left : 70,
                top : 10,
                height : 'auto',

            });
            row.add(label);

            var thumb = Ti.UI.createImageView({
                image : item.getElementsByTagName("thumbnail").item(0).getAttribute('url'),
                width : 50,
                height : 30,
                left : 10,
                top : 10,
                borderRadius : 7,
                borderWidth : 2,
                borderColor : 'transparent',
            });
            row.add(thumb);
            tableData.push(row);
        }

        ///////////////////////////////////////////////Second Function Feed For the table////////////////////////////////////
        TableRows2();
        //////////////////////////////////////////////////////////////////////////////////////////////////////////////////////////

    }, xhr.onerror = function() {
        alert('Server Error');
    };
    xhr.open('GET', 'https://gdata.youtube.com/feeds/api/users/BBombsCrew/uploads?v=2&alt=rss&prettyprint=true');
    xhr.send();
};
TableRows1();

function TableRows2() {
    var xhr2 = Ti.Network.createHTTPClient();
    xhr2.onload = function() {

        var doc2 = this.responseXML.documentElement;
        var items2 = doc2.getElementsByTagName("item");
        var x2 = 0;
        var doctitle2 = doc2.evaluate("//channel/title/text()").item(0).nodeValue;

        var limit = dataContext.function1Length + 1;
        var upperLimit = items2.length + dataContext.function1Length;
        var k = 0;
        Ti.API.info('Function 2:==========' + limit + '\n' + upperLimit);
        for (limit; limit <= upperLimit; limit++) {

            var item2 = items2.item(k);

            var title2 = doc2.evaluate("//channel/title/text()").item(0).nodeValue;
            Ti.API.info('Function title2:==========' + title2);
            var row2 = Ti.UI.createTableViewRow({
                height : 40,
                width : 320,
                backgroundColor : 'white'
            });
            var label2 = Ti.UI.createLabel({
                text : title2 + ' ' + limit,
                font : {
                    fontSize : 13,
                    fontWeight : 'bold'
                },
                color : 'black',
                left : 70,
                top : 10,
                height : 'auto',

            });
            row2.add(label2);

            var thumb2 = Ti.UI.createImageView({
                image : item2.getElementsByTagName("thumbnail").item(0).getAttribute('url'),
                width : 50,
                height : 30,
                left : 10,
                top : 10,
                borderRadius : 7,
                borderWidth : 2,
                borderColor : 'transparent',
            });
            row2.add(thumb2);
            tableData.push(row2);
            k++;

        }
        TableView.setData(tableData);

    }, xhr2.onerror = function() {
        alert('Server Error');
    };
    xhr2.open('GET', 'https://gdata.youtube.com/feeds/api/users/jonnieduck/uploads?v=2&alt=rss&prettyprint=true');
    xhr2.send();
};

win.open();
