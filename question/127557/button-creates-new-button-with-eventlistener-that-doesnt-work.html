Button creates new button with eventlistener that doesn't work

In my code i created a button (b1) that when clicked creates a new button (b2);
I now want to add an eventlistener to that second button but i can't seem to get this working.
Any ideas on what i'm doing wrong here?

var win = Ti.UI.currentWindow;
var b1 = Titanium.UI.createButton({
  title:'onzichtbare knop',
  color:'#fff',
  height:40,
  width:600,
  top:0,
  left: 0,
  zIndex:100,
});

b1.addEventListener('click', function() {


  alert("b1geklikt");  
  var b2 = Titanium.UI.createButton({
  title:'onzichtbare knop',
  color:'#fff',
  height:40,
  width:600,
  top:200,
  left: 0,
  zIndex:100,
});

b2.addEventListener('click', function() {


  alert("b1geklikt");  
});

b1.add(b2);
win.add(b1);

    this should work…

    var win = Ti.UI.currentWindow;
var b1 = Titanium.UI.createButton({
  title:'onzichtbare knop',
  color:'#fff',
  height:40,
  width:600,
  top:0,
  left: 0,
  zIndex:100,
});

b1.addEventListener('click', function() {
    alert("b1geklikt");  

    var b2 = Titanium.UI.createButton({
      title:'onzichtbare knop',
      color:'#fff',
      height:40,
      width:600,
      top:200,
      left: 0,
      zIndex:100,
    });

    b2.addEventListener('click', function() {
      alert("b1geklikt");  
    });

    //add the second button to win, not to b1 and do it inside eventListener of b1
    win.add(b2);

//finally, close the b1.addEventListener    
});

win.add(b1);
