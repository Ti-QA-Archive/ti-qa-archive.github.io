    Assuming below your Window is called "win1" then the example below should help you out.

    //create your email
function postToEmail(_title, _message) {
    var emailDialog = Titanium.UI.createEmailDialog();
    emailDialog.subject = _title;
    emailDialog.toRecipients = ['info@youremailaddress.com'];
    emailDialog.messageBody = _message;
    emailDialog.open();
}

var buttonEmail = Titanium.UI.createButton({
    width:  280,
    height:  35,
    top: 280,
    left: 20,
    title: 'Send Via Email'
});
buttonEmail.addEventListener('click', function(e){
  postToEmail('Email subject', 'Email body text');
});

win1.add(buttonEmail);
