/*
 * Need to create default tab windows.
 * Using Blue Background
 */

function tabWindow(title, colour) {
    var the_win = Titanium.UI.createWindow({
        title : title,
        navBarHidden : true,
        backgroundColor : colour
    });

    return the_win;
}

/*
 * Create Tab
 */

function createTabUI(tab_title, tab_window) {
    //Create Tab
    var the_tab = Titanium.UI.createTab({
        // icon : dir_path.R + 'images/tabs/home-tab-off.png',
        title : tab_title,
        window : tab_window
    });
    tab_window.containingTab = the_tab;

    return the_tab;
}

ApplicationTabGroup.prototype.addContent = function() {
    //Create Tab Group
    var tabGroup = Titanium.UI.createTabGroup({
    });

    //Load Modules
    var GarageWindow = require('ui/base/GarageWindow');
    var RadiusWindow = require('ui/base/RadiusWindow');

    //Load Windows
    var garage_win = new GarageWindow("Garage Sales");
    var radius_win = new RadiusWindow("Settings");

    //Create Tabs
    var garage_tab = createTabUI("Garage Sale", garage_win.getWindow());
    var settings_tab = createTabUI("Settings", radius_win.getWindow());

    //Add the Tabs
    tabGroup.addTab(garage_tab);
    tabGroup.addTab(settings_tab);

    //Display Tabs
    tabGroup.open();
};

module.exports = ApplicationTabGroup;
