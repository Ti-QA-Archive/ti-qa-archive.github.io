crateNavigationGroup and overlapping navigation bar

Hello, All

I am having some trouble with how I've created the navigation group and I wanted to run it by you to see what I might me doing wrong.

here is the code

index.js

var init = function(){
    var manager = require("WindowManager").getWindowManager();
    var actionsView = Alloy.createController('Actions');
    $.mainWindow.open();
    manager.openFirstWindow(actionsView, $.mainWindow);
}();


        manager.openFirstWindow = function(controller, mainWindow){
            var viewWrapperWindow = Titanium.UI.createWindow();
            viewWrapperWindow.add(controller.getView());

            var nav = Titanium.UI.iPhone.createNavigationGroup({
                window: viewWrapperWindow
            });

            // apply theming
            theme.applyStyles(viewWrapperWindow, theme.Window);
            manager.setNavGroup(nav);
            mainWindow.add(nav);
        };

When the nav bar is rendered, it is overlapping my view. Any ideas?

    You need to move to the NavigationWindow as NavigationGroup has been deprecated.

    Take a look at the release notes for the full details. Specifically check out the iOS 7 transition guide that explains fully how the navigation bar interacts now with child elements.
