adding multiple text fields

I'm only able to get 1 textfield to show up in iOS with this code, and I'm not sure why..

function textfield_rest(_args) {
    var win = Ti.UI.createWindow({
        title:('Settings')
    });


        var adminurl = Titanium.UI.createButton({
        style:Titanium.UI.iPhone.SystemButton.DISCLOSURE
    });

        adminurl.addEventListener('click',function()
    {
        Titanium.UI.createAlertDialog({
            title:'Admin URL',
            message:'Example: http://yourdomain.com'
        }).show();
    });

    var tf1 = Titanium.UI.createTextField({
        color:'#336699',
        top:10,
        left:10,
        width:250,
        height:40,
        leftButton:adminurl,
        hintText:'Admin URL',
        keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
        returnKeyType:Titanium.UI.RETURNKEY_DEFAULT,
        borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
    });
    tf1.addEventListener('return', function()
    {
        tf1.blur();
    });

    var adminusername = Titanium.UI.createButton({
        style:Titanium.UI.iPhone.SystemButton.DISCLOSURE
    });

        adminusername.addEventListener('click',function()
    {
        Titanium.UI.createAlertDialog({
            title:'Admin Email',
            message:'Example: you@yourcompany.com'
        }).show();
    });

    var tf2 = Titanium.UI.createTextField({
        color:'#336699',
        top:10,
        left:10,
        width:250,
        height:40,
        leftButton:adminusername,
        hintText:'Admin Email',
        keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
        returnKeyType:Titanium.UI.RETURNKEY_DEFAULT,
        borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
    });
    tf2.addEventListener('return', function()
    {
        tf2.blur();
    });


    var tf3 = Titanium.UI.createTextField({
        color:'#336699',
        top:10,
        left:10,
        width:250,
        height:40,
        passwordMask:true,
        hintText:'Admin Password',
        keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
        returnKeyType:Titanium.UI.RETURNKEY_DEFAULT,
        borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
    });
    tf3.addEventListener('return', function()
    {
        tf3.blur();
    });


    win.add(tf1,tf2,tf3);
    return win;
}

module.exports = textfield_rest;

    Hello Lincoln :

    In your code problem is you set all TextFields top : 10. if set all TextFields top different OR set Window's layout : &quot;vertical&quot; like : 

    var win = Ti.UI.createWindow({
        title:('Settings'),
        layout : 'vertical'
    });
