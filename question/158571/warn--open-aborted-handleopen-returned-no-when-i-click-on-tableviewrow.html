    APP.JS

    // this sets the background color of the master UIView (when there are no windows/tab groups on it)

    className : "tableRow"

var tableView = Titanium.UI.createTableView({
    data : data
});

tableView.addEventListener("click", function(e) {
    if (e.source.hasChild) {
        var w;
        if (e.rowData.js) { 
            win1 = require(e.rowData.js);
            var w = new win1(e.rowData.title);
        } else {
            w = Titanium.UI.createWindow({
                title : e.source.title,
                backgroundColor : "#FFFFFF",
            });
            var label = Titanium.UI.createLabel({
                text : "A newly opened window from the " + e.source.title + " row",
                width : "auto",
                height : "auto",
                textAlign : "center"
            });

            w.add(label);
        }

        winNav.openWindow(w, {
            animated : true
        });
    } else {
        alert("No window to open :(");
    }
});

win.add(tableView);
// open navigation Window
winNav.open();

    external.js

    function external(title) {

    var win = Titanium.UI.createWindow({
        title : title, //Take the title from the row
        backgroundColor : "#FFFFFF", });

    var inputData = [
    {title:'row 1', header:'CHEF DE PROJECT...'},
    {title:'row 2'},
    {title:'row 3'},
    {title:'row 4', header:'Sample 1', hasChild: true},
    {title:'row 5', header:'Sample 2', hasChild: true},
        ];

    var tableView = Titanium.UI.createTableView({
    data:inputData,
    style:Titanium.UI.iPhone.TableViewStyle.GROUPED
});



    win.add(tableView);

    return win;
};
module.exports = external;

    external1.js

    //Create a pointer to the current window context
function external1(title) {

    var win = Titanium.UI.createWindow({
        title : title, //Take the title from the row
        backgroundColor : "#FFFFFF",
        //dataToPass:e.source.dataToPass,

    });

    var label = Titanium.UI.createLabel({
        text : 'win.dataToPass', //This text is loaded from variables passed into the window through the dataToPass property
        width : "auto",
        height : "auto",
        textAlign : "center"
    });

    win.add(label);

    return win;
};
module.exports = external1;
