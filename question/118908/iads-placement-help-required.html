  • The simulator will not show ads but it will show a fake one.
    Also Apple is not very good at displaying ads, so this is why you need to catch errors.

    /**
 * the ads
 */
var iads = Titanium.UI.iOS.createAdView({
    width: 'auto',
    height: 50,
    top: 0,
    borderColor: '#000000',
    backgroundColor: '#000000',
    zIndex: 9,  // set the zIndex to the upper level
    opacity:1
    });

// I like to show the ads a little after everything is loaded, so I set a timeout
    iads.addEventListener('load', function(e){

       var ad_timeout = setTimeout(function(){
           iads.show();

       },10000)
    });
    iads.addEventListener('error', function(e){
      iads.hide();
      Titanium.API.info('ad error')
    });
    iads.addEventListener('click', function(e){
      Titanium.API.info('ad click')
    });
win.add(iads);
