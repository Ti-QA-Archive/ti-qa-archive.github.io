  • here is service code, the problem is that iBeacon callbacks are still called even if service is stopped 

    Ti.API.info("service");
var iBeacon = require("miga.tibeacon");

var initOps;

iBeacon.stopScanning();
iBeacon.initBeacon({
    success : onSuccess, error:onError, interval: 30, region: onRegion, found:onFound
});

function onSuccess(e){
  Ti.API.info("onSuccess");
  Ti.API.info(JSON.stringify(e));
}  

function onRegion(e){
    Ti.API.info("onRegion");
    Ti.API.info(JSON.stringify(e));
}  

function onFound(e){
    Ti.API.info("onFound");
    Ti.API.info(JSON.stringify(e));
}  

function onError(e){
    Ti.API.info("onError");
  Ti.API.info(JSON.stringify(e));
}  

if (iBeacon.isEnabled()){
  iBeacon.startScanning();
  //iBeacon.stopScanning();
}
