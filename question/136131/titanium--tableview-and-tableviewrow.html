  • Hello eddahbani,
    Try this…

    First.js

    
var win = Ti.UI.currentWindow;

var d = [];

var counter = ['First','Second','Third','Fourth','Fifth'];

var tblView = Ti.UI.createTableView({});

for( i = 0; i < counter.length; i++) {

    var row = Ti.UI.createTableViewRow({
        height : 60,
        hasChild : true,
    });

    var labtitle = Ti.UI.createLabel({
        text : counter[i],
        height : 20,
        left : 10,
        top : 20,
        width : 250,
        color : 'black',    
    });
    row.add(labtitle);
    d.push(row);
}
tblView.setData(d);

win.add(tblView);

tblView.addEventListener('click', function(e) {
    var win = Ti.UI.createWindow({
        title : 'Window 2',
        url : 'second.js',
    });
    Ti.UI.currentTab.open(win, {
        animated : true
    });

});

    second.js

    var win2 = Ti.UI.currentWindow;
win2.backgroundColor = 'white';

var btn = Ti.UI.createButton({
    title : 'Back'
});
win2.leftNavButton = btn;

btn.addEventListener('click',function(e){
    win2.close();
})

    Good Luck….

  • This should get you started;

    var win = Ti.UI.createWindow();

var tbv = Ti.UI.createTableView();

var rows = [{title:'Row 1'},{title:'Row 1'}];

tbv.setData(rows);

tbv.addEventListener('click', function(e)){

    var win2= Ti.UI.createWindow();
    var btn = Ti.UI.createButton({title:'Close', width:100, height:20});
    win2.add(btn);
    btn.addEventListener('click', function(e){
        win2.close();
    });
    win2.open();

}

win.add(tbv);

win.open();
