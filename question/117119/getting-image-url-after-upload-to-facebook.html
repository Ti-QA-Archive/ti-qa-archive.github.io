Getting Image URL After Upload To Facebook

How do I get the image URL after I upload it to facebook. When I use the facebook GraphApi and hae the id returned upon successful upload I can not get any object properties. Any help is appreciated. Here is the code:

Titanium.Media.openPhotoGallery({
    success:function(event)
    {
    var data = {picture: event.media};
    Titanium.Facebook.requestWithGraphPath('me/photos', data, 'POST', 
        function(e) {
            if (e.success) {
                var response = JSON.parse(e.result);
                Ti.API.info(response.id);
                Titanium.Facebook.requestWithGraphPath(response.id, {}, 'GET', function(e) {
                    if (e.success) {
                        Ti.API.info("INTERNAL:"+e);
                    } else if (e.error) {
                        alert(e.error);
                    } else {
                        alert('Unknown response');
                    }
                });
                alert(e);
            } else if (e.error) {
                alert(e.error);
            } else {
                alert('Unknown response');
            }
        });
    },
    cancel:function()
    {
    },
    error:function(error)
    {
    },
    allowEditing:true
});
