  • atob() and btoa() are not standard functions of ECMA 5.1 or ECMA 6 specs.

    If I remember correctly, these are mozilla specific DOM functions later extended to webkit browsers. That is why you can't find it in Titanium.

  • you can convert blob to string using toString() and then use the function given toBin(str) for conversion into binary 

    var f = Ti.Filesystem.getFile('KS_nav_ui.png');
var blob = f.read();
var encoded = Ti.Utils.base64encode(blob);
var strn= encoded.toString();

function toBin(str){
var st,i,j,d;
var arr = [];
var len = str.length;
for (i = 1; i<=len; i++){
               //reverse so its like a stack
 d = str.charCodeAt(len-i);
 for (j = 0; j < 8; j++) {
  st = d%2 == '0' ? "class='zero'" : "" 
  arr.push(d%2);
  d = Math.floor(d/2);
 }
}
       //reverse all bits again.
return arr.reverse().join("");
}


Ti.API.info("binary is"+ toBin(strn));
