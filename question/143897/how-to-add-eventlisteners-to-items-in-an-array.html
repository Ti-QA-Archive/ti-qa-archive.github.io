  • The way I do it in my app is as follows:

    
// Setup the array contents

var b = [];

for (var i = 0; i < x; i++) {
    var button = Titanium.UI.createButton({
        height: bs,
        width: bs,
        index: i,
        id: "btn"+i,
        text: "test"+i
    });

    button.addEventListener('click', function(e){
        alert(e.source.index);
    });

    b.push(button);
}

// Then, somewhere else in my code, iterate through the array and add each object individually

for (var i = 0; i < x; i++) {
    mV.add(b[i]);
}

    Instead of explicitly assigning the button object to a specific array key, I simply use the array object's push function to add it to the array.

    Then when adding the button to my view, I use a for loop to iterate through the array and add each object individually, instead of passing the entire array to the view's add function.

    Good luck with your app, I hope this helps :)
