Android table row value

Hi All,

I'm working with Titanium SDK 2.1.3 GA and i'm trying to get the value of a row label for android.
My code:

            //id
                var labelId = Titanium.UI.createLabel({
                    name:'labelId',
                text:response[i].id,
                font:{fontSize:12,fontWeight:'bold'},
                width:'auto',
                textAlign:'left',
                top:0,
                left:0,
                height:12,
                color:'#000',
                visible:false,
            });
            row.add(labelId);
            //add behavior
            table.addEventListener('click', function(e) {
                //alert(Ti.Platform.osname);
                var myId;
                if (Ti.Platform.osname ==='android') {
                    myId = e.source.labelId;
                    alert('This is myid: ' + myId);
                }
            });

and i always get the error, undefined.
I tried several things such as
myId = e.row.labelId;
myId = e.rowData.labelId;
but without luck, can you help me please?

Yours truly,

Stijn
