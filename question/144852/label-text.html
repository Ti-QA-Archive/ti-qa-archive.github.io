  • thanks alot Rainer i found the solution ..
    the problem is when i add text:something+pointText.item(0).text, text apear in 2 lines that i need it in one line ..
    so i use this to solve it and its work perfect .. 

    var pointText = articleXmlDoc.getElementsByTagName(point);
    var whiteSpace = pointText.item(0).text;
    whiteSpace = whiteSpace.replace('\n', '');
    var viewsText = Ti.UI.createLabel({
        font : {
            fontWeight : 'bold',
            fontSize : 15
        },
        color : 'black',
        textAlign : 'right',
        text : whiteSpace + '\n',
        layout : 'absolute'
    });
