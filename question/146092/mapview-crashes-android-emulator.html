      app.js

      Ti.UI.setBackgroundColor('#000');

var win = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Tab'
});
var tab = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab',
    window: win
});

var btn = Ti.UI.createButton({
    title: 'Map'
});
btn.addEventListener('click', function (e) {
    var Hop = require('/hop');
    var win = Hop.create();
    win.open();
});
win.add(btn);

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab);  
tabGroup.open();

      hop.js (place in root of resources)

      function create() {
    var win = Ti.UI.createWindow({
        title: 'House of Praise, VA',
        backgroundColor: '#13386c',
        barColor: '#9b05ad'     
    });

    var hopData = Ti.UI.createView({ 
        backgroundColor: '#eee',
        borderColor: '#999',
        borderRadius: 10,
        borderWidth: 2,
        bottom: 70,
        height: Ti.UI.SIZE,
        layout: 'vertical',
        left: 20,
        right: 20,
        width: Ti.UI.FILL
    });

    var info = Ti.UI.createLabel({
        bottom: 20,
        color: '#000',
        left: 20,
        right: 20,
        top: 20,
        text: 'Pastor T. S. Akran\nAddress: 941 West Road, Chesapeake VA\n\nPhone: (757) 639-5080\nWebsite: www.houseofpraiseva.org'
    });
    hopData.add(info);

    var annotations = [
        Ti.Map.createAnnotation({
            latitude: 36.712843,
            longitude: -76.322149,
            title: 'House of Praise',
            subtitle: '941 W. Road, Chesapeake, VA.',
            animate: true,
            pincolor: Ti.Map.ANNOTATION_GREEN,
            leftButton: '/images/icons/sermonRSS.png'
        }),
        Ti.Map.createAnnotation({
            latitude: 36.721035,
            longitude: -76.308459,
            title: 'WalMart Store',
            subtitle: 'Chesapeake, VA',
            animate: true,
            pincolor: Ti.Map.ANNOTATION_RED
        }),
        Ti.Map.createAnnotation({
            latitude: 36.713906,
            longitude: -76.318267,
            title: 'Grassfield High School',
            subtitle: 'Chesapeake, VA',
            animate: true,
            pincolor: Ti.Map.ANNOTATION_PURPLE
        })
    ];

    var map = Ti.Map.createView({
        animate: true,
        annotations: annotations,
        mapType: Ti.Map.STANDARD_TYPE,
        region: {
            latitude: 36.712843,
            longitude: -76.322149,
            latitudeDelta: 0.03,
            longitudeDelta: 0.03
        },
        regionFit: true,
        userLocation: true
    });

    win.add(map);
    win.add(hopData);
    return win;
}
exports.create = create;
