  • The setTimeout has to be at least 500ms. 250 doesn't work, and btw, sliding from the side looks pretty cool :)

    // insert this code somewhere in the default app.js
var W = {};
W.DetailWin = function(){
    var win = Ti.UI.createWindow({
        title:'some title'
    });

    var tf = Ti.UI.createTextField({top:10,left:10,right:10,height:40,style:1,backgroundColor:'white',value:'1'});

    win.add(tf);
    win.addEventListener('focus', function(){
        setTimeout(function(){
            tf.focus();
        },500);
    });

    return win;
};
label1.addEventListener('click', function(){
    tab1.open(W.DetailWin());
});
