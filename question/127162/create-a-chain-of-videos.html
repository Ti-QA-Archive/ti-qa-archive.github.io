Create a chain of videos

What I am trying to do is create a chain of video's. So if one video is finished automatically another one is started.

I have the following code:


var video2 = Titanium.Media.createVideoPlayer({
            contentURL : row_data.external_id,
            backgroundColor : '#111',
            movieControlMode : Titanium.Media.VIDEO_CONTROL_DEFAULT,
            autoplay : false,
            zIndex : 3,
            height : 200,
            top : 0,
});
video2.hide();


var video1 = Titanium.Media.createVideoPlayer({
                contentURL : row_data.preroll,
                backgroundColor : '#111',
                movieControlMode : Titanium.Media.VIDEO_CONTROL_DEFAULT,
                autoplay : true,
                zIndex : 3,
                height : 200,
                top : 0,
});



preroll.addEventListener('complete', function() {
                video1.release();
                video1.hide();
                video2.show();
                video2.play();

});

But is seems that the complete event is triggered not only when the video is completed but also in other events. The result is that the video is not finished.

Does anybody know how to do this?
