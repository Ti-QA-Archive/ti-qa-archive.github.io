Window does not open on app.js

Hi Dudes,
I have folowing problem: The Window "LoginWindow" on app.js does not open on the iphone 5 device only the "Default@2x.png" will stay and nothing happens!

On the simulator 6.1 it works fine! :(:(:(

Any Idea?

app.js

var LoginWindow = require('main_windows/LoginWindow');
var winLogin = new LoginWindow();
winLogin.open();

loginWindow.js

function LoginWindow() {

    var win = Titanium.UI.createWindow();

    var scrollView = Ti.UI.createScrollView({
        contentWidth : 'auto',
        contentHeight : 'auto',
        layout : 'vertical',
        backgroundColor : 'white',
        showVerticalScrollIndicator : false,
        showHorizontalScrollIndicator : false,
    });

    win.add(scrollView);
    return win;
}

module.exports = LoginWindow;

  • SDK: 3.1.2.GA
