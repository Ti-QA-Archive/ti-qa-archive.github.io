  • Accepted Answer

    This is really a bug with SDK 3.0.2.GA

    If you add a textfield with passwordMask directly to a tableviewrow, you can only type 1 character. Strange, but true.

    As a workaround you have a use a view, add the textfield to this view and then add the view to the tableviewrow.

    var row = Ti.UI.createTableViewRow({
    backgroundColor: '#fff',
    height: Ti.UI.FILL
});

var requiredViewBecauseOfDamnBug = Ti.UI.createView({
    height: Ti.UI.FILL
});    

var title = Ti.UI.createLabel({
    left: 10,
    width: 90,
    font: {fontSize:17,fontWeight:'bold'},
    text: 'Password'
});

var pwd = Ti.UI.createTextField({
    width: Ti.UI.FILL,
    right: 10,        
    left: 110,
    hintText: 'at least 5 characters', // translate
    borderStyle: Ti.UI.INPUT_BORDERSTYLE_NONE,
    keyboardType: Ti.UI.KEYBOARD_DEFAULT,
    clearButtonMode: Ti.UI.INPUT_BUTTONMODE_ONFOCUS,
    returnKeyType: Ti.UI.RETURNKEY_DONE,
    passwordMask: true,
    autocorrect: false
});

requiredViewBecauseOfDamnBug.add(title);    
requiredViewBecauseOfDamnBug.add(pwd);

row.add(requiredViewBecauseOfDamnBug);
