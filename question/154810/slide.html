  • Mi codigo es:

    var SliderCantidad = Titanium.UI.createSlider({
    top: '150dp',
    min: 1,
    max: 11,
    width: Ti.Platform.displayCaps.platformWidth-20,
    value: 2,        
    leftTrackImage: '/img/sliderleft.png',
    rightTrackImage: '/img/sliderright.png',
    borderRadius: 10           
    });

    var LabelCantidad = Ti.UI.createLabel({
        color: '#ffffff',
        text: 'Mostrar '+SliderCantidad.value+' preguntas',
        top: '130dp', 
        left: '10dp',
        font:{
            fontSize:'16dp',
            fontFamily: 'ArchivoNarrow-Regular',            
        },        
    });

    SliderCantidad.addEventListener('change', function(e) {
        Ti.API.info(e.value);
        var nAge = (Math.round(e.value)-1)*10;
        if(nAge==0){nAge=10;}
        CantidadPreguntas = nAge;
        LabelCantidad.text = 'Mostrar '+nAge+' preguntas';        
    });
