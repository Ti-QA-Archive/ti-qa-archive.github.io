    Hi

    If you are trying to change the specific row you are pressing then you can tweak your code as follows;

    tableView.addeventListener('click', function (e) {
    e.source.setHeight(200px');
});

    • I have re-worked your code to show you how you can achieve what you want to do. This does what you want, you can extend the idea to do much more.

      var TheTable = Titanium.UI.createTableView({
    top: '15%',
    backgroundColor: '#333',
    separatorColor: '#FFF',
    touchEnabled: true
});
winHome.add(TheTable);
var addRow = function (obj) {
    var row = Ti.UI.createTableViewRow({
        height: Ti.UI.SIZE,
        width: Ti.UI.FILL
    });
    var titulos = Ti.UI.createLabel({
        color: '#FFF',
        font: {
            fontFamily: 'ubuntu',
            fontSize: 20
        },
        height: '50dp',
        left: '3%',
        text: obj.title || '',
        textAlign: 'left',
        top: '0',
        //touchEnabled: false
    });

    var barra = Ti.UI.createView({
        backgroundColor: '#555',
        height: '2dp',
        top: '50dp',
        width: '100%'
    });

    var container = Ti.UI.createView({
        backgroundGradient: {
            backFillStart: false,
            colors: ['#343434', '#2C2C2C'],
            endPoint: {
                x: 0,
                y: '90%'
            },
            startPoint: {
                x: 0,
                y: 0
            },
            type: 'linear'
        },
        height: '50dp',
        top: '52dp',
        width: '100%'
    });
    //Elementos que van dentro de container: comentario, checkbox, botones para imagenes.
    var groupCheckbox = Ti.UI.createView({
        backgroundColor: 'red',
        height: '100dp',
        left: '88%'
    });
    //creacion de checkbox
    var si = Ti.UI.createSwitch({
        style: Ti.UI.Android.SWITCH_STYLE_CHECKBOX
    });
    groupCheckbox.add(si);

    row.add(titulos);
    row.add(barra);
    row.add(container);
    row.add(groupCheckbox);

    row.adjust = function (obj) {
        var bajar = Ti.UI.createAnimation({
            height: obj.height
        });
        container.animate(bajar);
    };

    return row;
    //table.hasChild=true;
}
var row, tableData = []
for (var i = 0; i < 20; i++) {
    row = addRow({
        title: 'Pregunta N° ' + i
    });
    tableData.push(row);
}
TheTable.setData(tableData);
//*************************
//creamos un listener para las actiones de cada fila


TheTable.addEventListener('click', function (e) {
    e.row.adjust({
        height: '250dp'
    });
});
