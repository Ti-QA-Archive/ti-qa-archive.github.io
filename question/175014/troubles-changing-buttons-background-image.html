    For solving your issue, I have used the below code. Please try the following code and I hope, it will help you.

    var win = Ti.UI.createWindow({
    backgroundColor: '#fff'
});

var Open_Gallery = Ti.UI.createButton({
    title : 'Open Gallery',
    backgroundImage:'/KS_nav_ui.png',
    width:150,
    height:150,
    value:false

});

Open_Gallery.addEventListener('click', function(e) {
    if(e.source.value)
    {
        this.value=false;
        this.backgroundImage='/KS_nav_ui.png';
        Ti.API.info('true');
    }
    else
    {
        this.value=true;
        this.backgroundImage='/KS_nav_views.png';
        Ti.API.info('false');
    }
});

win.add(Open_Gallery);
win.open();
