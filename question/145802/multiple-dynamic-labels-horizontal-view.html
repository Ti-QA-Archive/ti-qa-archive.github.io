  • Hi

    There is a layout property you can add to provide a horizontal layout. However it will not manage the overflow of the individual label or multiple labels next to each other.

    To try this code simply create a blank new simple project and replace the app.js with the contents of the file below.

    Ti.UI.setBackgroundColor('#000');
var win = Ti.UI.createWindow({  
    backgroundColor: '#ffffff',
    height: Ti.UI.FILL,
    title: 'Test',
    width: Ti.UI.FILL
});

var tab = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Test',
    window: win
});

function addRow(obj) {
    var row = Ti.UI.createTableViewRow({
        backgroundColor: '#fff',
        height: Ti.UI.SIZE,
        width: Ti.UI.FILL
    });

    var view = Ti.UI.createView({
        backgroundColor: '#fff',
        height: Ti.UI.SIZE,
        layout: 'horizontal',
        width: Ti.UI.FILL
    });
    row.add(view);

    var lbl1 = Ti.UI.createLabel({
        color: 'red',
        font: {
            fontSize: 14
        },
        height: Ti.UI.SIZE,
        text: obj.label1,
        width: Ti.UI.SIZE
    });
    view.add(lbl1);

    var lbl2 = Ti.UI.createLabel({
        color: 'green',
        font: {
            fontSize: 14
        },
        height: Ti.UI.SIZE,
        text: obj.label2,
        width: Ti.UI.SIZE
    });
    view.add(lbl2);

    var lbl3 = Ti.UI.createLabel({
        color: 'blue',
        font: {
            fontSize: 14
        },
        height: Ti.UI.SIZE,
        text: obj.label3,
        width: Ti.UI.SIZE
    });
    view.add(lbl3);

    return row;
}

var data = [
    { label1: 'aaa111', label2: 'bbb111', label3: 'ccc111' },
    { label1: 'aaa222', label2: 'bbb222', label3: 'ccc222' },
    { label1: 'aaa333', label2: 'bbb333', label3: 'ccc333' },
    { label1: 'This is very long and should cause a problem', label2: 'bbb444', label3: 'ccc444' }
];

var intRow = 0, intRows = data.length, rows = [], row;
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    row = addRow(data[intRow]);
    rows.push(row);
}
var tbl = Ti.UI.createTableView({
    data: rows,
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
win.add(tbl);

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab);
tabGroup.open();

    If you really need the layout you are talking about you may need to to use the styledLabel module or similiar.
