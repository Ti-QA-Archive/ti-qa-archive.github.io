Alloy Navigation Group

Hi,

i am using custom navigation group in my app..

index.js

$.index.addEventListener('open', function(){
    Ti.API.info('We are in the Index');
    var homeController = Alloy.createController('home');
    homeController.getView().open();
});
$.index.open();

home.js

Alloy.CFG.navgroup = $.nav;
function viewCollectionClick() {
    var collectionController = Alloy.createController('collection');
    $.nav.open(collectionController.getView(), {
        animated : true
    });
}

function checkCatalogueClick() {
    var cataloguesController = Alloy.createController('catalogues');
    cataloguesController.getView().open();
}

collection.js

function goBack() {
    Alloy.CFG.navgroup.close($.collectionWindow, {
        animated : true
    });
}

collection.xml

<Alloy>
    <Window navBarHidden="true" >
        <NavigationGroup id="nav" platform="ios">
            <Window class="container" id="collectionWindow" navBarHidden="true">
                <View id="topNav">
                    <Button id="backBtn" onClick="goBack" />
                    <Label id="winTitle" />
                </View>
                <ScrollView id="collectionScroll" />
            </Window>
        </NavigationGroup>
    </Window>
</Alloy>

when clicking on the backBtn in the collection Section, it take me back to index and i am not able to click on any thing in the app..

how can i make the backBtn work?

  • You appear to be creating a navgroup that has only one window in it, then you are closing that root window, which isn't quite the right way to be using the navgroup. Are you trying to remove the navgroup entirely and go back to the home controller? If so, close the top-level window in the collection controller, not the window that is inside the navgroup.

    function goBack() {
    $.collection.close();
}
