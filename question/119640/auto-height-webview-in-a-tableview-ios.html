  • Hi guys, this one worked for me.

    var webView = Titanium.UI.createWebView({
    html: htmlContent,
    left:0,
    right:0,
    top:0,
    bottom:0,
    height:'auto'
});
webView.addEventListener('load', function(e){
    tblRow.height = webView.evalJS("document.height;");
});

var tblRow = Titanium.UI.createTableViewRow({
    height:'auto'
});
tblRow.add(webView);
