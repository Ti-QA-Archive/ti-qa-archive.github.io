    Hello mirian,

    You can use the Concept of Children.
    Like,

    $.routeListTable.addEventListener("click", function(e) {

    var sections = table.data;
    var section = sections[0];
    var theRow = section.rows[e.index];

    var rowContainer = theRow.getChildren()[0]; //This is "rowContainer"
    var routeListIcon = rowContainer.getChildren()[0]; //This is "routeListIcon"

    routeListIcon.backgroundImage = '/yes.png';

    var objectList = new Alloy.createController("objectList").getView();
    objectList.open();

});

    Let me know if it works..

    Not tested yet..
