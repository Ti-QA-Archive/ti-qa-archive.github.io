touchend not firing on ios

I have an eventlistener on a view to change the backgroundColor when it's touched and to change back on release, but the touchend does not seem to fire, touchstart is fine, and it works ok on android. Not sure what I'm missing.

            eticketsContainer.addEventListener('touchstart', function(e) {
                Ti.API.info('touchstart fired');
                eticketsContainer.backgroundColor = '#C23A25';
                etickets.color = '#FFFFFF';
                arrow.image = '/images/arrow_white.png';
            });

            eticketsContainer.addEventListener('touchend', function(e) {
                Ti.API.info('touchend fired');
                eticketsContainer.backgroundColor = '#FFFFFF';
                etickets.color = '#444';
                arrow.image = '/images/arrow.png';
            }); 

            eticketsContainer.addEventListener('click', function(e) {
                eticketsWin = new EticketsListWindow({
                    containingTab : self.containingTab,
                    tabGroup : self.tabGroup,
                    title : 'Etickets',
                    barColor : '#C23A25',
                    backgroundColor : '#f1f1f1'
                });
                self.containingTab.open(eticketsWin, {
                    animated : true
                });
            });
