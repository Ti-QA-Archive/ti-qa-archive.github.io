Problem uploading photos with ACS

When I try to upload photos with ACS It gives me some weird error:

{"success":false,"error":true,"code":401,"message":"You need to sign in or to sign up before continuing"}

It may look normal but I use the Cloud.Photos.create AFTER the login operation returned success:

myapp.Cloud.Users.login({
            login: 'default',
            password: 'default',
        }, function(e){
            if(e.success)
            {
                alert("Successful login");
                var tempBlob = myapp.funciones.escalarImagen(latestMedia);
                myapp.Cloud.Photos.create({
                    photo: tempBlob,//Ti.Blob representing a picture
                }, function(evt){
                    if(evt.success)
                    {
                        alert("Successful upload");
                        var id = evt.photos[0].id;
                        sendData(id);
                    }
                    if(evt.error)
                    {
                        that.actUploadingData.hide();
                        alert(JSON.stringify(evt));
                    }
                });
            }
            if(e.error)
            {
                that.actSubiendoDatos.hide();
                alert(JSON.stringify(e.error));
            }
        });

Thanks for your time and code strong.
