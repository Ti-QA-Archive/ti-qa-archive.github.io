    Hello Kirthika,

    Try this code :

    var win = Ti.UI.createWindow({
    backgroundColor : "white"
});

var sclView = Ti.UI.createScrollView({
    top : 0,
    width : "100%",
    layout : "horizontal",
    horizontalWrap : false,
    contentWidth : Ti.UI.SIZE,
    height : "75%",
    scrollType : "horizontal",
    backgroundColor:"white"
});

for ( i = 0; i < 10; i++) {
    var view = Ti.UI.createView({
        height : "75%",
        width : "100",
        left : "20dp",
        backgroundColor : "red"
    });

    sclView.add(view);
}

win.add(sclView);
win.open();

    set your image in view's backgroudImage and check.

    i hope this helps. thanks.
