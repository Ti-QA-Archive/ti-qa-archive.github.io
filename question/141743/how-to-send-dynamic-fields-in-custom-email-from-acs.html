how to send dynamic fields in custom email from ACS?

Hi,

I am new to ACS. I'd like to send email with my custom email. so I created email template as follow:

Name    Subject    Body    
first     judul    isi nya {{isi}}

the i use the js sdk as follow:

    var data = {
      recipients: 'maul_firdaus@yahoo.com',
      template: 'first',
      dynamic_fields: {'isi':'asik'}
    };
    sdk.sendRequest('custom_mailer/email_from_template.json', 'POST', data, callback);

And the logs shows:

  • This code works fine for me 100%.

    function doClick(e) {

    var Cloud = require('ti.cloud');

    Cloud.Emails.send({
    template: 'mytemplate', //create template on site https://my.appcelerator.com/app in CLOUD - MANAGE ACS
    recipients: 'email@email.com',
    from: 'email@emai.com', //must be defined for SMTP
    name_in_template: 'matus', //  {{name_in_template}}
    team_in_template: 'my team' // {{team_in_template}}
}, function (e) {
    if (e.success) {
        alert('OK');
    } else {
        alert('Error:\n' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});
}
