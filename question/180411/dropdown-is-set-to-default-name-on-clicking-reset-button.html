    Hello Supriya,

    Thanks for your inquiry, First you needpicker.setSelectedRow(0, 0, false); to set the first row as default selected row in your picker. Then, you have to add an event listener and inside it you need to execute that again to reset your picker. Try the below code.

    Ti.UI.backgroundColor = 'white';
var win = Ti.UI.createWindow({
  exitOnClose: true,
  layout: 'vertical'
});

var picker = Ti.UI.createPicker({
  top:50
});

var data = [];
data[0]=Ti.UI.createPickerRow({title:'Bananas'});
data[1]=Ti.UI.createPickerRow({title:'Strawberries'});
data[2]=Ti.UI.createPickerRow({title:'Mangos'});
data[3]=Ti.UI.createPickerRow({title:'Grapes'});

picker.add(data);

picker.selectionIndicator = true;

win.add(picker);

win.open();

picker.setSelectedRow(0, 0, false);

var  resetbutton = Titanium.UI.createButton({ title: 'Login', right: 50, bottom: 30, width: 80, height: 60 });

resetbutton.addEventListener('click',function(e)
{
   picker.setSelectedRow(0, 0, false);
});
win.add(resetbutton);

    Hope this helps you. If it does, make as best answer. Thanks.
