  • I use simple timeout (I didn't find a way how to cancel activity indicator using back button). Something like this:

    var actInd = Ti.UI.createActivityIndicator({
    height: 20,
    width: 20
});

var t;

var activityTimeout = function() {
    ready();
};

var busy = function() {
    if (!t) {
        //if activity is not finishes in 25 seconds remove activity indicator
        t = setTimeout(activityTimeout, 25000);
    }
    actInd.show();
};

var ready = function() {
    actInd.hide();
    if (t) {
        clearTimeout(t);
    }
};
