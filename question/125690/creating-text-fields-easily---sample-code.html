Creating text fields easily - sample code

One of the things I've labored with using Titanium is the manual layout of components. Beside making my code appear bloated, I've often found myself going back and doing lots of tuning to get things just the way I wanted. So, I created a class I called MyUIElements to help me out a bit.

If you've ever had to layout, say, a big form with lots of entry fields, this might be useful to you. It's clearly a work in progress and, similar to my other code examples, may not represent the best or most efficient mechanisms for performing its intended task - but it has worked for me. So, I invite you to give it a try.

Using the class is easy. First create a file with whatever name you like and copy the source code (shown below) into that file. I named my file my_ui_elements.js and included it in my project with the statement Ti.include(&#39;&#x2F;my_ui_elements.js&#39;);

Here's some code to create three fields in the middle of the screen. Each one has a title and some hint text. And, the fields understand how to interpret the NEXT button.

var f_email = MyUIElements.createEditFieldWithLabel (    null,                {top:130, left: 10, width:300, height:60}, {top:0, text:'Email Address'}, {value:'', hintText:'Enter your email address here', autocapitalization: Titanium.UI.TEXT_AUTOCAPITALIZATION_NONE});
var f_last = MyUIElements.createEditFieldWithLabel (    f_email.edit_field,    {top:70, left: 10, width:300, height:60}, {top:0, text:'Last Name'}, {value:'', hintText:'Enter your last name here'});
var f_first = MyUIElements.createEditFieldWithLabel (    f_last.edit_field,    {top:10, left: 10, width:300, height:60}, {top:0, text:'First Name'}, {value:'', hintText:'Enter your first name here'});

win1.add (f_first);
win1.add (f_last);
win1.add (f_email);

I should note that the reason the fields are done in seemingly reverse order is so that some of the objects are created a priori before they are used to setup support for pressing NEXT.

Here's the code. And comments or suggestions you have for me or other developers would, of course, be welcome. And, if you have suggestions for other layout models or tools I'd like to learn about them.

MyUIElements = {};

/**
 * Create an edit field with a label
 */
MyUIElements.createEditFieldWithLabel = function (next_field, container_args, label_args, field_args)
{
    if (container_args == null) {    container_args = {};    }
    if (label_args == null)     {    label_args = {};        }
    if (field_args == null)     {    field_args = {};        }

    var label_offset = 6;
    var offset = 2;
    var padding = 4;
    var label_height = 22;
    var text_field_height = 32;
    var default_border_radius = 10;
    var default_border_width = 2;

    // ------------------------------------------
    // containing window
    // ------------------------------------------
    if (container_args.show_border) {
        container_args.borderWidth = 1;
        container_args.borderColor = '#FF0000';
    }
    var w_container = Ti.UI.createWindow (container_args);

    // ------------------------------------------
    // label
    // ------------------------------------------
    label_args.top = 0;
    label_args.left = offset + label_offset;
    label_args.width = (container_args.width - (2*(offset + label_offset)));
    label_args.height = text_field_height;
    if (label_args.font == null) { label_args.font = {fontSize:12,fontWeight:'normal', fontFamily:'Arial'}; }
    if (label_args.show_border) {
        label_args.borderWidth = 1;
        label_args.borderColor = '#00FF00';
    }
    var label = Titanium.UI.createLabel(label_args);
    w_container.add (label);

    // ------------------------------------------
    // Field
    // ------------------------------------------
    if (next_field != null) { field_args.returnKeyType = Ti.UI.RETURNKEY_NEXT; }
    if (field_args.font == null) { field_args.font = {fontSize:14,fontWeight:'normal', fontFamily:'Arial'}; }
    if (field_args.color == null) { field_args.color = '#336699'; }
    if (field_args.textAlign == null) { field_args.textAlign = 'left'; }
    if (field_args.clearOnEdit == null) { field_args.clearOnEdit = false; }
    if (field_args.borderRadius == null) { field_args.borderRadius = default_border_radius; }
    if (field_args.borderWidth == null)  { field_args.borderWidth = default_border_width; }
    if (field_args.borderColor == null)  { field_args.borderColor = '#BEC0C7'; }
    if (field_args.borderStyle == null)  { field_args.borderStyle = Ti.UI.INPUT_BORDERSTYLE_ROUNDED; }
    if (field_args.returnKeyType == null)  { field_args.returnKeyType = Ti.UI.RETURNKEY_NEXTINPUT_BORDERSTYLE_ROUNDED; }
    if (field_args.keyboardType == null)  { field_args.keyboardType = Ti.UI.KEYBOARD_ASCII; }

    // calculated, positional settings
    field_args.height = text_field_height;
    field_args.top = label_height + padding;
    field_args.left = offset;
    field_args.width = (container_args.width - (2*offset));
    field_args.clearOnEdit = false;
    field_args.textAlign = 'left';

    var field = Titanium.UI.createTextField(field_args);
    w_container.add (field);
    w_container.edit_field = field;

    // -----------------------------------------------
    // Add event listener for when the return button is pressed
    // -----------------------------------------------
    if (next_field != null) {
        w_container.edit_field.addEventListener('return', function(e)
        {
            next_field.focus();
        });
    }

    return w_container;
};
