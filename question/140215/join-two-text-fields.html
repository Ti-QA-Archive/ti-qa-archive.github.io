    Here you are :)

    var tableViewWithTwoRows = Ti.UI.createTableView({backgroundColor:'transparent',separatorStyle:'none',width:310});
    tableViewWithTwoRows.topRow = Ti.UI.createTableViewRow({
        width:310,
        height:50,
        backgroundImage:'TableViewRoundedTop.png'
    });
    tableViewWithTwoRows.bottomRow = Ti.UI.createTableViewRow({
        width:310,
        height:50,
        backgroundImage:'TableViewRoundedBottom.png'
    });

    tableViewWithTwoRows.appendRow(tableViewWithTwoRows.topRow);
    tableViewWithTwoRows.appendRow(tableViewWithTwoRows.bottomRow);

    var win = Ti.UI.createWindow({backgroundColor:'#336699'});
        win.add(tableViewWithTwoRows);
        win.open();

    You can download the images from my site via the following urls:
    http://iampnelson.com/tmp/TableViewRoundedTop.png
    http://iampnelson.com/tmp/TableViewRoundedBottom.png
    http://iampnelson.com/tmp/TableViewMiddle.png

    I also provided the middle image, so you can play around with having more than 2 rows. Let me know if you have any questions.
