  • Is this for Android or iOS? On Android you can set the view of an option dialog:

    var inputView = Titanium.UI.createView({
        backgroundColor:'#111',
    });

    var locField = Titanium.UI.createTextField({
        hintText:'Enter Location',
        height:35,
        width:250,
        borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
    });

    inputView.add(locField);
    if (utils.isAndroid()) {

        var dialog = Titanium.UI.createOptionDialog({
            androidView: inputView,
            buttonNames: ['OK'],
            title: 'Set Location'
        });

        dialog.show();
        dialog.addEventListener('click', function(e){
            Titanium.API.info("Dialoge Click : " + JSON.stringify(e));
            if (e.button == true && e.index == 0) {
                var val = locField.value;
                if (!val || val.length == 0) 
                    return;
                handleSetLocation(val);

            }
        });
    }
