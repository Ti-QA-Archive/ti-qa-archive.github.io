    Hi aparna,

    the window property softInputMode don't exist for SDK 3.3.0, even if it is still referenced in the Documentation for Supporting Multiple Platforms in a Single Codebase - Branching.

    So you have to use windowSoftInputMode.

    I slightly modified your code, to get easier control over object properties.
    Then i used the windowSoftInputMode property and tested the code on Nexus 5 device, Genymotion emulator, Android Emulator and iOS Simulator.

    These are my results, tested with Studio 3.3.0.201407100905 and Titanium SDK 3.3.0.GA

    • Device: Nexus 5 Android Version 4.4.4 working!
    • Genymotion Emulator: Nexus 5 Android Version 4.4.2 not working
    • Android Emulator: Galaxy Nexus Android Version 4.3 working!
    • iOS Simulator: iPhone Retina (4-inch 64-bit) iOS 7.1 working!

    modified code

    var data = [];

// create and add new row
function addRow(labelText, textHintText, textKeyboardType) {
    var row = Ti.UI.createTableViewRow({
        className : 'personInfo'
    });
    var label = Ti.UI.createLabel({
        text : labelText,
        left : Ti.Platform.displayCaps.platformWidth * 0.1,
        font : {
            fontWeight : 'bold'
        },
        top : 5,
        color : 'white'
    });
    row.add(label);
    var text = Ti.UI.createTextField({
        hintText : textHintText,
        left : Ti.Platform.displayCaps.platformWidth * 0.1,
        top : 40,
        softKeyboardOnFocus : (Ti.UI.Android) ? Titanium.UI.Android.SOFT_KEYBOARD_SHOW_ON_FOCUS : '',
        keyboardType : textKeyboardType
    });
    text.addEventListener('focus', function(e) {
        // check if TextField got focused
        Ti.API.info('===> TextField focus for: ' + text.hintText);
    });
    row.add(text);

    data.push(row);
}

var searchWin = function(title) {

    //create a window
    var win = Ti.UI.createWindow({
        title : title,
        navBarHidden : false,
        modal : true,
        backgroundImage : '/images/bg.png',
        //softInputMode:Ti.UI.Android.SOFT_INPUT_ADJUST_PAN,
        windowSoftInputMode : (Ti.UI.Android) ? Ti.UI.Android.SOFT_INPUT_ADJUST_PAN : ''
    });

    //search table
    var tableView = Ti.UI.createTableView({
        separatorColor : 'white',
        top : '7%'
    });

    // add 3 rows with different keyboard types
    addRow('Name', 'Enter Name', Ti.UI.KEYBOARD_DEFAULT);
    addRow('Mobile Number', 'Enter Mobile Number', Ti.UI.KEYBOARD_NUMBERS_PUNCTUATION);
    addRow('Profession', 'Enter Profession', Ti.UI.KEYBOARD_DEFAULT);

    tableView.setData(data);
    win.add(tableView);

    return win;

};

exports.searchWin = searchWin;
