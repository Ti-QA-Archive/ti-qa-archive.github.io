    var containerView = Ti.UI.createView({layout:'vertical'});

var labelContainerView = Ti.UI.createView();
var label1 = Ti.UI.createLabel();

labelContainerView.add(label1);

containerView.add(labelContainerView);

var label3 = Ti.UI.createLabel();

containerView.add(labelContainerView);

containerView.add(label3);
