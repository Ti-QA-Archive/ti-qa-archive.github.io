Popover crash with toggle

I have a problem with a popover that I try to hide or show when someone taps the left navigation button, it simply crashes the whole application if someone taps the button very fast.

This is the code:

[...]

var popover = Ti.UI.iPad.createPopover({ 
    width:300, 
    height:250,
    navBarHidden: true,
    isVisible:false
}); 

win.setLeftNavButton(bChoose);

bChoose.addEventListener("click", function(e) {
    if (!popover.isVisible) {
        popover.show({ view:e.source, animated:false});
        popover.isVisible = true;
    } else {
        popover.hide();
        popover.isVisible = false;
    }
});
