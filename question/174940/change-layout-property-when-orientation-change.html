  • Hello Harrison,

    Try like this…

    Titanium.Gesture.addEventListener('orientationchange', function(e) {
    switch (Titanium.Gesture.orientation) {
        // For LandScap.......
        case Titanium.UI.LANDSCAPE_LEFT:
        case Titanium.UI.LANDSCAPE_RIGHT:

            mainView.children[0].backgroundColor = '#3170F0';
            mainView.children[0].setLayout('horizontal');

            break;

        // For Portrait.......

        case Titanium.UI.PORTRAIT:
        case Titanium.UI.UPSIDE_PORTRAIT:

            mainView.children[0].backgroundColor = '#227026';
            mainView.children[0].setLayout('vertical');

            break;
    }
});

Titanium.Gesture.fireEvent('orientationchange');

    I hope it will help you, Best Luck.

  • What I did in my app:

    var containerOfFields = Ti.UI.createView({
    ...
    layout: 'horizontal',
    ...
});
var label1 = Ti.UI.createTextField({
    ...
    width : // something bigger than 50% of the container's width
    ...
});

var label2 = Ti.UI.createTextField({
    ...
    width : // something bigger than 50% of the container's width
    ...
});
containerOfFields.add(label1);
containerOfFields.add(label2);

Ti.Gesture.addEventListener("orientationchange", function(e) {
    if(Titanium.Gesture.orientation == Ti.UI.LANDSCAPE_LEFT ||
       Titanium.Gesture.orientation == Ti.UI.LANDSCAPE_RIGHT) {
        label1.width = label2.width = // something lower than 50% of the container's width
    } else {
        label1.width = label2.width = // something bigger than 50% of the container's width
    }
});

    The above solution causes to set the labels under each other on portrait, and horizontal on landscape, even if the layout property is always 'horizontal'.
