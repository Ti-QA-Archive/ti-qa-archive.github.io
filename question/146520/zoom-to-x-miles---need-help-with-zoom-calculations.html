  • I ended up going with the following: Hopefully this helps someone out.

    function zoomToMiles(miles) {

    mapview.userLocation = true;

    Titanium.Geolocation.preferredProvider = "gps";
    Titanium.Geolocation.purpose = "user coordinates";
    Titanium.Geolocation.distanceFilter = 10;

    var CurrentLat;
    var CurrentLon;

    Titanium.Geolocation.getCurrentPosition(function(e) {
        if (!e.success || e.error) {

            alert('error ' + JSON.stringify(e.error));
            return;
        }
        CurrentLat = e.coords.latitude;
        CurrentLon = e.coords.longitude;

        if (CurrentLat != null) {

            var scalingFactor = Math.abs((Math.cos(2 * 3.14 * CurrentLat / 360.0) ));

            var latDelta = miles / 69.0;
            var longDelta = miles / (scalingFactor * 69.0);

            var region11 = {
                latitude : CurrentLat,
                longitude : CurrentLon,
                latitudeDelta : latDelta,
                longitudeDelta : longDelta
            };

            mapview.setRegion(region11);
        } else {
            Ti.API.info('Location Not Found');
        }

    });

}// end zoomToMiles

    latitudeDelta : km * 18/2000,
    longitudeDelta : km * 18/2000,
