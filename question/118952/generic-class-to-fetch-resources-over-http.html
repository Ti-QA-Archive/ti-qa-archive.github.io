Generic class to fetch resources over HTTP

Hi All,

i'm trying to create a general class which i can instantiate, fetch a resource and let my other code know the resource has been fetched, as it can continue script execution (as synchronous mode does not function properly). The thing i keep struggling with is the fact that the onload handler for the xhr component seems to run in a different context then my main script. Has anyone ever written such a class? Please find my trying below.

var HTTP = function(){

    var url = "";

    var fetchMethod = "GET";

    this.fetchedUrl = false;

    /*
     * not supported in titanium! This means a callback to a method following 
     * the request is a must
     */
    var fetchMode = true;

    this.loopDelay = 200;


    this.fetchURL = function(method, url, callBackMethod, failMsg){

        xhr = Titanium.Network.createHTTPClient();

        xhr.onload = function(){

            Ti.API.debug('URL succesfully fetched: ' + url);
            // Now parse the feed XML 
            var xml = this.responseXML;
            this.doCallback(xml, callBackMethod);
            return true;
        };

        xhr.onerror = function(e){
            //this.networkError(xhr.status, url, failMsg);
            return false;
        };

        xhr.open("GET", url);
        xhr.send();
    };

    this.doCallBack = function(data, nextCallback){
        Ti.API.debug('setting fetchURL to true, was: ' + this.fetchUrl);
        this.fetchedUrl = true;
        nextCallback(data);
        return true;
    };
};
