  • I must apologise people the JSON I pasted it in was Ti.API.log output. I havee sinc sorted the issue with this function 

    function sortData(data) {
    var sorted = [];
    Object.keys(data).sort(function(a,b){
        return data[a].name < data[b].name ? -1 : 1
    }).forEach(function(key){
        sorted.push(data[key]);
    });
    return sorted;
}
