Change label text to variable string problem

Hi guys,

I can't seem to find the problem in this piece of code which should work 100% fine, but it does not.

So I want to change the text of a label (label.setText()), but it does not seem to be able to read the string value of the variable.

PS:

  1. Yes, the tempString does contain a string of text, when I alert it
  2. tempString.toString() doesn't work either

The code:

var labelDirections = Titanium.UI.createLabel({
        text : 'ha',
        color : '#ffffff',
        font : {
            fontSize : 14
        },
        top : imageViewLogo.top + imageViewLogo.height + 15,
        left : 10,
        width : 75,
        height : 20
    });
    self.add(labelDirections);

    var departureTime = 1362241730;

    var url = "http://maps.googleapis.com/maps/api/directions/json?origin=" + Ti.App.Properties.getString('street') + Ti.App.Properties.getString('houseNumber') + Ti.App.Properties.getString('city') + "&destination=Herestraat 49 Leuven&sensor=false&mode=transit&departure_time=" + departureTime;
    var client = Ti.Network.createHTTPClient({
        onload : function(e) {
            var json = JSON.parse(this.responseText);
            var options = json.routes[0].legs[0].steps;

            var tempString = '';
            for (var i = 0; i < options.length; i++) {
                tempString += options[i].html_instructions + '\n';
            };
            labelDirections.setText(tempString);
        },
        onerror : function(e) {
            alert('Error: ' + e.error);
        },
        timeout : 5000
    });
    client.open("GET", url);
    client.send();

So the problem is situated on this line:

labelDirections.setText(tempString);

Please help me out.
