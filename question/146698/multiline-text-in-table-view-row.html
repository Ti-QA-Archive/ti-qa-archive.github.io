Multiline text in table view row

Same question link in Stack Overflow

I'm working on a tableView based application in Titanium.
I need to show long text contents in my table view rows.

This is the code I'm using to create my table view row.

var row = Ti.UI.createTableViewRow({
        backgroundColor : '#F0F0F0',
        color           : 'black',
        font            : { fontSize:10 ,fontWeight:'bold',fontFamily:'Arial'},
        height          :'80',
        title            : firstRowContent
       });

My issue is I need to wrap the text in my table view row and need to display it in multiline.

In iOS there is lineBreakMode and numberOfLines properties. And I can easily do it.

Is there any way to do it in titanium ? I searched a lot, but everyone says that put a '\n' in between the text or add multiple labels.

    Hi

    Yes it is VERY possible and is VERY common, you just need to create a custom tableViewRow.

    To explain this I have created a simple example that shows several lines of lines in each row, some shorter and some longer, I have also included optional images and subtitles within the row.

    You have complete control over the look of these rows, take a look and see what I mean.

    function addRow(obj) {
    var row = Ti.UI.createTableViewRow({
        backgroundColor: '#eee',
        color: 'black',
        font: {
            fontSize: 10,
            fontWeight: 'bold',
            fontFamily: 'Arial'
        },
        height: Ti.UI.SIZE,
        layout: 'horizontal',
        width: Ti.UI.FILL
    });
    if (obj.image) {
        var img = Ti.UI.createImageView({
            height: Ti.UI.SIZE,
            image: obj.image,
            left: 10,
            width: Ti.UI.SIZE
        });
        row.add(img);
    }
    var view = Ti.UI.createView({
        bottom: 10,
        height: Ti.UI.SIZE,
        layout: 'vertical',
        top: 10,
        width: Ti.UI.FILL
    });
    row.add(view);
    var lblTitle = Ti.UI.createLabel({
        color: '#000',
        height: Ti.UI.SIZE,
        left: 10,
        right: 10,
        text: obj.title || '',
        textAlign: 'left'
    });
    view.add(lblTitle);
    if (obj.subtitle) {
        var lblSubTitle = Ti.UI.createLabel({
            color: '#444',
            height: Ti.UI.SIZE,
            left: 10,
            right: 10,
            text: obj.subtitle || '',
            textAlign: 'left',
            top: 10
        });
        view.add(lblSubTitle);
    }
    return row;
}
var data = [
    { title: 'This is really long text and will wrap over more than one line - change the font size to see it change.', subtitle: 'You can even add more information if you want.' },
    { title: 'This is a short line.', image: 'KS_nav_views.png' },
    { title: 'This is another really long line and should be even bigger than the first one but who knows maybe it is shorter - no it is longer I was just playing.' }
];
var rows = [], intRow = 0, intRows = data.length;
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    rows.push(addRow({
        title: data[intRow].title,
        subtitle: data[intRow].subtitle,
        image: data[intRow].image
    }));
}
var tbl = Ti.UI.createTableView({
    data: rows,
    height: Ti.UI.FILL,
    minRowHeight: 40,
    width: Ti.UI.FILL
});
win.add(tbl);

    This is tested on iOS and Android and works perfectly - being stylish is another question :)
