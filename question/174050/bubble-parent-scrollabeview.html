  • Hi

    I "think" you want to be able to have a scrollableView that contains several scrollViews, when the user touches any scrollView - that event is passed to the scrollableView. That event listener will then handle all those events. This way you only have one event handler to worry about?

    var page1 = Ti.UI.createScrollView({ backgroundColor: 'red', action: 'one' });
var page2 = Ti.UI.createScrollView({ backgroundColor: 'green', action: 'two' });
var page3 = Ti.UI.createScrollView({ backgroundColor: 'blue', action: 'three' });
var page4 = Ti.UI.createScrollView({ backgroundColor: 'yellow', action: 'four' });

var test = Ti.UI.createView({
    backgroundColor: 'orange',
    height: 200,
    touchEnabled: false,
    width: 200
});
page1.add(test);

var scrollView = Ti.UI.createScrollableView({
    showVerticalScrollIndicator: false,
    verticalBounce: false,
    views: [
        page1,
        page2,
        page3,
        page4
    ]
});
scrollView.addEventListener('singletap',function(e){
    Ti.API.log("Touching");
    Ti.API.info('action: ' + e.source.action);
    switch (e.source.actino) {
    case 'one':
        alert('Hi');
        break;
    case 'two':
        // do somthing else
        break;
    case 'three':
        // do somthing else
        break;
    case 'four':
        // do somthing else
        break;
    }
});
// add "scrollView" to a parent

    This is a working example showing you how that can work.
