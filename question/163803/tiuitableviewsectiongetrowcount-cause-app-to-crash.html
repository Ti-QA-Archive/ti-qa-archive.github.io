Ti.UI.TableViewSection.getRowCount() cause app to crash

My env is : Titanium SDK 3.2.2.GA, OS X Mavericks 10.9.2, iOS7, iOS simulator, Xcode 5.0.2

Is this known bug or there is something wrong with my local configuration? There is sample code:

(function() {
    var self = Ti.UI.createWindow();

    var table = Ti.UI.createTableView({
        style : Ti.UI.iPhone.TableViewStyle.GROUPED,
        top : 0,
        separatorInsets : {
            left : 0
        },
        backgroundColor : '#F8F8F8'
    });
    table.headerView = Ti.UI.createView({
        height : 80
    });

    var tableViewSection = Ti.UI.createTableViewSection();

    var highLabel = Ti.UI.createLabel({
        left : 6,
        font : {
            4 : 4
        },
        height : 44,
        text : 'High'
    });

    var tableRow = Ti.UI.createTableViewRow({
        height : 44,
        hasChild : false,
        backgroundColor : '#fff'
    });

    tableRow.add(highLabel);

    tableViewSection.add(tableRow);
    table.setData([tableViewSection]);

    self.add(table);

    tableRow.addEventListener('click', function(e) {
        Ti.API.info('Row count : ' + tableViewSection.getRowCount());
    });

    self.open();

})();
