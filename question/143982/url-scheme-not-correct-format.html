      <ios>
        <plist>
            <dict>
                <key>CFBundleURLTypes</key>
                <array>
                    <dict>
                        <key>CFBundleURLName</key>
                        <string>com.FrankDase.54House</string>
                        <key>CFBundleURLSchemes</key>
                        <array>
                            <string>Radio54House</string>
                        </array>
                    </dict>
                </array>
            </dict>
        </plist>
    </ios>
