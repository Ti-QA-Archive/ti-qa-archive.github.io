App level event listener with arguments

How would I pass arguments to the sayHello function below?

function SayHello(data) {
    alert(data);
}

//setup event listener
Ti.App.addEventListener ('test', sayHello);

//fire the event, send data
Ti.App.fireEvent('test',data);

    function SayHello(data) {
    alert(data.value);
}

//setup event listener
Ti.App.addEventListener ('test', sayHello);

//fire the event, send data
Ti.App.fireEvent('test',{value:' hello there'});
