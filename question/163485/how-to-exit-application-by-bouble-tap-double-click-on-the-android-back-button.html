  • Hi Deepan, use the below code to achieve the required effect.

    var firstTap, secondTap;
window.addEventListener("androidback",function(e){
    if(firstTap == null){
        firstTap = new Date();
    }else{
        secondTap = new Date();
        var diff = (secondTap - firstTap)/1000;
        firstTap = new Date();
        Ti.API.info('diff is '+diff);
        if(diff < 1){ // checking for difference of 1 second between each back click to exit the app
            Ti.Android.currentActivity.finish();
        }
    }
});

    Do note that exiting the app by double clicking the back button is not good according to android design guidelines.
