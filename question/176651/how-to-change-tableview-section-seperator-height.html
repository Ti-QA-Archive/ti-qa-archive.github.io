    Hi

    You are using the simple section header provided as a convenience as many layouts work very well with that exact style. However when your needs exceed that simple style you have to create your own.

    This is a really quick example but should clearly show where you make the change and how you can use it to suit your needs;

    Ti.UI.backgroundColor = 'white';
var win = Ti.UI.createWindow();

var addSection = function (obj) {
    var view = Ti.UI.createView({
        backgroundColor: 'orange',
        height: 100,
        width: Ti.UI.FILL
    });
    view.add(Ti.UI.createLabel({
        backgroundColor: '#000',
        font: {
            fontWeight: 'bold'
        },
        height: TI.UI.SIZE,
        left: 10,
        text: obj.left,
        width: 100
    }));
    view.add(Ti.UI.createLabel({
        backgroundColor: '#000',
        height: TI.UI.SIZE,
        right: 10,
        text: obj.right,
        textAlign: 'right',
        width: 100
    }));
    var section = Ti.UI.createTableViewSection({
        headerView: view
    });
    return section;
}

var sectionFruit = addSection({
    left: 'Fruit',
    right: 'Hello'
});
sectionFruit.add(Ti.UI.createTableViewRow({ title: 'Apples' }));
sectionFruit.add(Ti.UI.createTableViewRow({ title: 'Bananas' }));

var sectionVeg = addSection({
    left: 'Vegetables',
    right: 'World'
});
sectionVeg.add(Ti.UI.createTableViewRow({ title: 'Carrots' }));
sectionVeg.add(Ti.UI.createTableViewRow({ title: 'Potatoes' }));

var table = Ti.UI.createTableView({
    data: [
        sectionFruit,
        sectionVeg
    ]
});
win.add(table);

win.open();
