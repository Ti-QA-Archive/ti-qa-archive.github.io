  • Short answer: you can,

    at the time I'm working at this module.

    Usage:

    var dbmodule = new (require('sqlite.adapter'))('http://URLtoSQLiteDB');
dbmodule.startmirror();
dbmodule.addEventListener('onload',function(_e) {
    if (_e.succes ==true) Ti.Database.open(_e.dbname);
});
dbmodule.addEventlistener('onprogress',function(_e) {
};
