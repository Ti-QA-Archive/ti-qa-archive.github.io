  • Hi,

    Try this, Hope it solves your question

    var tbData = [];

//Start Loop with i as counter

var row = Ti.UI.createTableViewRow({
    hasChild:true
});

var lab_left = Ti.UI.createLabel({
    title:'Settings '+i,
    height:'auto',
    width:'auto',
    textAlign:'left',
    left:0
});
row.add(lab_left);


var lab_right = Ti.UI.createLabel({
    title:'value '+i,
    height:'auto',
    width:'auto',
    textAlign:'right',
    right:0
});
row.add(lab_right);

tbData.push(row);

//End of Loop

var tb = Ti.UI.createTableView({
    data:tbData,
    style:Ti.UI.iPhone.TableViewStyle.GROUPED
});
