Swipe effect not working properly...

Hi,
This is a pretty simple scenario wherin I have two views viz., view0 & view1 and I have used touchend & touchstart eventlisteners to give the swipe effect. Sometimes I experience a change in views when i should not on just the click of the mouse.I am running this code for Ipad simulator SDK 1.8.0.1. Can anyone please put light on what exactly it is that i am doing wrong here.
Thanks!!! 

var currentViewIndex = 0;
var view0 = Ti.UI.createView({
    backgroundColor : '(149,201,236)',
    borderRadius : 10,
    width : 1030,
    height : 600,
    top : 0,
    transition : Ti.UI.iPhone.AnimationStyle.CURL_UP
});
var view1 = Ti.UI.createView({
    backgroundImage : 'view1_background.png',
    borderRadius : 10,
    width : 1030,
    height : 600,
    top : 0,
    transition : Ti.UI.iPhone.AnimationStyle.CURL_UP
});

view_array = [view0, view1]; 

for(var view in view_array) {
    win.add(view_array[view]);
}

var showView = function(viewIndex) {
    for(var view in view_array) {
        view_array[view].hide();
    }
    view_array[viewIndex].show();
}

view0.addEventListener('touchstart',function(e){
            x = e.x;
    });

view0.addEventListener('touchend',function(e){
           x=x - e.x;
                if(x<-320)
        { 
                    if(currentViewIndex!=0){
                    currentViewIndex=currentViewIndex-1;
                    showView(currentViewIndex);
                }
                 else if(x>320 && currentViewIndex<1)
            {
                            currentViewIndex=currentViewIndex+1;
                            showView(currentViewIndex);
             }
});
