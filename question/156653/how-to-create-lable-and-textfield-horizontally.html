  • Hello ravi sharma,

    I have post your desired code for your help. I hope it will help you. But I would like to suggest for reading Titanium SDK docs and play around Kitchen Sink Project for more helps. Kitchen Sink will guide to basic implementation part. 

    //FirstView Component Constructor
function FirstView() {
    //create object instance, a parasitic subclass of Observable
    var self = Ti.UI.createView({
        layout : 'vertical',
    });

    //label using localization-ready strings from <app dir>/i18n/en/strings.xml
    nameView = Ti.UI.createView({
        layout : 'horizontal',
        width : Ti.UI.FILL,
        height : Ti.UI.SIZE,
        top : 10,
    });

    self.add(nameView);
    var nameLabel = Ti.UI.createLabel({
        color : "#00000",
        text : "Name",
        top : "40px",
        left : "40px",
        width : 122,
        height : 40,
    });

    nameView.add(nameLabel);

    var nameTextField = Ti.UI.createTextField({
        top : "40px",
        left : "10px",
        width : 200,
        height : 40,
        borderColor : '#000',
        borderWidth : 4,
        borderRadius : 4,
        hintText : 'nameTextField',

    });
    nameView.add(nameTextField);

    passwordView = Ti.UI.createView({
        layout : 'horizontal',
        width : Ti.UI.FILL,
        height : Ti.UI.SIZE,
        top : 10,
    });

    self.add(passwordView);
    var label = Ti.UI.createLabel({
        color : '#000000',
        text : String.format(L('welcome'), 'Titanium'),
        height : Ti.UI.SIZE,
        width : Ti.UI.SIZE,
        left : "40px",

    });
    passwordView.add(label);

    var btn = Ti.UI.createButton({
        title : "click Me",
        height : 'auto',
        width : 'auto',
        top : '20px',
        left : '40px'

    });

    btn.addEventListener('click', function(e) {
        alert("click Me");
    });
    self.add(btn);
    //Add behavior for UI
    label.addEventListener('click', function(e) {
        alert(e.source.text);
    });

    return self;
}

module.exports = FirstView;

  • test example:

    var win = Ti.UI.createWindow({
    backgroundColor : 'green'
});

var self = Ti.UI.createView({
    layout : "vertical",

});

var self1 = Ti.UI.createView({
    layout : "horizontal",
    top : 20,
    backgroundColor:'red',
    height:50
});

var self2 = Ti.UI.createView({
    layout : "horizontal",
    top : 150,
    backgroundColor:'yellow'
});

var nameLabel = Ti.UI.createLabel({

    text : "Name",

    left : 15,
    width : 50,
    height : 35
});

var nameTextField = Ti.UI.createTextField({

    height : 35,
    width : 140,
});

self1.add(nameLabel);
self1.add(nameTextField);
self.add(self1);

var passwordLabel = Ti.UI.createLabel({

    text : "Password",
    top : 150,
    left : 15,
    width : 50,
    height : 35
});

var passwordTextField = Ti.UI.createTextField({
    top : 150,
    height : 35,
    width : 140,
});
self2.add(passwordLabel);
self2.add(passwordTextField);
self.add(self2);

win.add(self);
win.open();
