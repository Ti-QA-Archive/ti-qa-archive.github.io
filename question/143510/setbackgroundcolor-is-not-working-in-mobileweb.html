  • Not sure what you're doing, but I just did a test and it works as expected in Titanium Mobile Web. I basically do this:

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

var b = Ti.UI.createButton({
    title: 'click me',
    left: '400px',
    top: '50px'
});

b.addEventListener('click', function () {
    if (tb.value) {
        tb.backgroundColor = "green";
    } else {
        tb.backgroundColor = "red";
    }
});
win.add(b);

var tb = Ti.UI.createTextField({ width: 100, value: "hi", backgroundColor: "#ff0" });
win.add(tb);

win.open();
