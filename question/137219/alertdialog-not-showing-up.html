myBtn.addEventListener('click',function(e){

    message: 'sure you want to delete it',
   buttonNames: ['Yes','No']
  }).show();
});

    Try this code

    var dlg = Titanium.UI.createAlertDialog({
    message:'Do you want delete this?', 
    buttonNames: ['Yes','No']
  });
// if you want to perform action on user click so add a eventListener to the dialog 
  dlg.addEventListener('click', function(ev) {
    if (ev.index == 0) {
         //for your yes click
      }
    else if(ev.index ==1)
    {
        //for your no click
    }

  });
  dlg.show();
