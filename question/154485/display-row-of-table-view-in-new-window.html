                top : '5dp',
                color : 'red',

            });
            flyerLabel = Ti.UI.createLabel({
                text : 'Marque' + prod.marque,

                    fontSize : '16dp'

    • Do not add event listener for each row. set only one for the whole tableview
      and never instantiate a window by using url property, this is bad practice even for a sample.

      var win = Ti.UI.createWindow({
                    url : "sample1.js"
                });

      instead create a commonJS module like in this example :
      http://developer.appcelerator.com/question/154018/moving-to-common-js-modules
