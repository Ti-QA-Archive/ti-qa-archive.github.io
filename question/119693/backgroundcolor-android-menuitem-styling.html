backgroundColor Android menuItem styling

Hi i'm using below code to add a menu witch works fine.
But how do i style the menu or the menuItems?

var activity = Ti.Android.currentActivity;
var tab = Ti.UI.currentTab;

var LOGIN = 1, LOGOUT = 2;
var loggedIn = false;
activity.onCreateOptionsMenu = function(e) {
    var menu = e.menu;

    var update = menu.add({ title: "Uppdater", itemId: LOGIN });
    update.setIcon("/KS_nav_ui.png");
    update.addEventListener("click", function(e) {
        loggedIn = true;
    });
    var changeDay = menu.add({ title: "Ändra dag", itemId: LOGOUT });
    changeDay.setIcon("/KS_nav_views.png");
    changeDay.addEventListener("click", function(e) {

    });
};
