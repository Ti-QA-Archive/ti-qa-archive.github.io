  • hi saurabh gupta

    please see the below sample code I think it will help out.

    Ti.UI.setBackgroundColor('#000');

    var win = Ti.UI.createWindow({

    backgroundColor: 'black',

    exitOnClose: true,

    fullscreen: false,

    title: 'TableView Demo'
    });

    var tableData = [];

    for (var i=1; i<=20; i++){

    var row = Ti.UI.createTableViewRow({

    className:'forumEvent', // used to improve table performance

selectedBackgroundColor:'white',

rowIndex:i, // custom property, useful for determining the row during events

height:110

    });

    var labelUrl = Ti.UI.createLabel({

    color:'#576996',

font:{fontFamily:'Arial', fontSize:defaultFontSize+6, fontWeight:'bold'},

text:'url' + i,// put here specific url obj[i].url

left:70, top: 6,

width:200, height: 30

    });
    row.add(labelUrl);

    row.url = obj[i].url;// here we are setting the url in a row

    tableData.push(row);

    }

    var tableView = Ti.UI.createTableView({

    backgroundColor:'white',

    data:tableData

    });

    win.add(tableView);

    // tableview eventclicklistener

    tableView.addEventListener('click', function(e){

    alert(e.row.url); // this will provide the url of specific row which is clicked in tableview.

    });

    win.open();
