  • Hi,

    Using that change event you can find related result.

    var customerNumInput = Ti.UI.createTextField({
        hintText: 'Enter Value'
    });

customerNumInput.addEventListener('change', function(e) {
        Ti.API.info(e.value);
         // Check condition here for that textField value
});
