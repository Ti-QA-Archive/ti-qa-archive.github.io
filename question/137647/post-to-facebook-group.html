    Hi Michael,

    Try this code just set your GroupId .Best luck

    if(Ti.Facebook.loggedIn && !(Titanium.Network.networkType == Titanium.Network.NETWORK_NONE)) {
        var data = {
            link : "http://www.google.com/",
            name : "Hello",
            message : "Test",
            caption : "This is a test message",
            picture : "http://www.youngmoneymagazine.co.uk/wordpress/wp-content/uploads/2011/12/test-image.jpg",
            description : "Testing my app"
        };
        Titanium.Facebook.requestWithGraphPath('/GroupId/feed', data, 'POST', function(e) {
            if(e.success) {
                alert('Success');
            }
            if(e.error) {
                alert(e.error);
            }
        });
    }
