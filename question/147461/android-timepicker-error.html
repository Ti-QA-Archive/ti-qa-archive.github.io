    Hi Miguel Ángel Castaño,

    I dont see the sense of vista view and why you open the picker in a window. Here my code:

    exports.create = function(options) {

    var picker = Ti.UI.createPicker({
        minDate : options.min,
        maxDate : options.max,
        useSpinner : true,
        value : options.value,
        type : Ti.UI.PICKER_TYPE_DATE
        //    transform : Ti.UI.create2DMatrix().scale(0.9)
    });
    self.picker.addEventListener('change', function(_e) {
        self.value = _e.value;
    })
    if (Ti.Platform.name === 'android') {
        //  only for A.
        // iPhone regelt das selber
        self.picker.showDatePickerDialog({
            value : options.value,
            callback : function(e) {
                var res = {
                    data : e.value
                };
                res.cancel = (e.cancel) ? true : false;
                self.fireEvent('picker:close', res);
            }
        });
    }
    });
    self.cancel.addEventListener('click', function() {
        var hider = Ti.UI.createAnimation({
        //    transform : Ti.UI.create2DMatrix().scale(3),
            duration : 1300,
            bottom : -800,
            opacity : 0
        });
        hider.addEventListener('complete', function() {
            self.fireEvent('picker:close', {
                data : null,
                cancel : true
            });

        });
        self.animate(hider);

    });
    return picker;
}
