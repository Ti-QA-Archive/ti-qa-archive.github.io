    • Raj, below code is working fine for me so just check it…

      
var win = Ti.UI.currentWindow;

var firstNameText = Ti.UI.createTextField({
    width : 250,
    height : 30,
    top : 75,
    left : 25,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    returnKeyType : Ti.UI.RETURNKEY_DONE,
});

win.add(firstNameText);

var call = Ti.UI.createButton({
    title : "Call",
    style : Ti.UI.iPhone.SystemButtonStyle.PLAIN
});

win.setRightNavButton(call);

call.addEventListener('click', function() {

    var dialog = Ti.UI.createAlertDialog({
        buttonNames : ["Call", "Cancel"],
        title : 'Call',
        cancel : 1,
        message : 'Are you sure you want to call '+ firstNameText.value+' ?' 

    });
    dialog.addEventListener('click', function(e) {
        if(e.index == 0) {
            Ti.Platform.openURL('tel:' + firstNameText.value);
        } else {
        }
    });
    dialog.show();
});
