  • set backgroundColor of textField ,like backgroundColor:'white' and cal fieldFocused and fieldBlurred methods in
    textfield's focus and blur event handlers

    function fieldFocused(textField) {
   textField.borderRadius = 5;
   textField.borderColor = '#FE642E';
}

function fieldBlurred(textField) {
   textField.borderColor = 'transparent';
}
