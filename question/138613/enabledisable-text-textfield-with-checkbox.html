enable/disable text TextField with checkbox

So, I have a textfield with enabled : false, and a checkbox that, when turned value = true, turns the property enabled of the textfield to true. No problem until here, but if I uncheck the checkbox, I want the textefield to become enabled : false,, but I cant do it, it is always enabled…

Any idea ?

    Without seeing your code hard to truly help, but you could do something like this:

    var self = Ti.UI.createWindow({
    title: 'TextEdit',
    backgroundColor: '#fff'
});

var text = Ti.UI.createTextField({
    top: '10dp',
    left: '10dp',
    right: '10dp',
    paddingLeft: 5,
    paddingRight: 5,
    width: Ti.UI.FILL,
    height: '38dp',
    borderWidth: 1,
    borderColor: '#aaa',
    borderRadius: 5,
    editable: false
});

self.add(text);

var toggle = Ti.UI.createSwitch({
    value: false
});

toggle.addEventListener('change', function(e) {
    text.setEnabled(e.value);
});

self.add(toggle);
self.open();
