I can't put a tab group inside a navgroup (double top bar, other issues), so I'm wondering if there's a way of building a slide out menu differently so I can have my tabs? Anyone done this?

  • a navigation group within a navWindow is a private case of a tab in a tab group, if you downloaded the code from Tutsplus which is here just change navgroup to tab and navWindow to tabgroup.
    here are the code changes you need to do:

    var navWindow = Ti.UI.createWindow({ width:320 &#x2F;&#x2F; Set the width of the sliding window to avoid cut out from animation });

    to

    var navWindow = Ti.UI.createTabGroup({ width:320 &#x2F;&#x2F; Set the width of the sliding window to avoid cut out from animation });

    and

    var navGroup = Ti.UI.iPhone.createNavigationGroup({ window:win });

    `to

    var navGroup = Ti.UI.iPhone.createTab({ window:win });

    and finally change navWindow.add(navGroup); to navWindow.addTab(navGroup);

    hope this helps :)

  • I just had the same problem and managed to crack it using Alloy. I've uploaded a mini-project onto GitHub with an example: https://github.com/app17/Titanium.SliderMenu

    To solve it I used composite layouts (and zIndex) - I've include the view xml and the controller js below. (only tested on iOS so use at your peril on Android)

    I think it should be possible to create the same "window within navgroup within window within tab etc" combination using js code too if you're still doing it old school.

    index.xml

    <Alloy>
    <TabGroup>
        <Tab title="Tab 1" icon="KS_nav_ui.png" layout="composite">
            <Window id="menuWindow" top="0" left="0" zIndex="1" layout="composite">
                <TableView width="150" top="0" left="0" onClick="changeView"
                    height="Ti.UI.SIZE" zIndex="1">
                    <TableViewRow title="red" colour="red" />
                    <TableViewRow title="orange" colour="orange" />
                    <TableViewRow title="yellow" colour="yellow" />
                    <TableViewRow title="blue" colour="blue" />
                    <TableViewRow title="green" colour="green" />
                </TableView>
                <Window id="mainWindow" width="320"
                    left="0" zIndex="100">
                    <NavigationGroup id="colourfulNav">
                        <Window id="topWindow" title="Colours" layout="composite">
                            <LeftNavButton>
                                <Button id="filterButton" onClick="displayMenu" title="Pick a Colour!" />
                            </LeftNavButton>
                            <View id="redView" backgroundColor="red" top="0" left="0" height="Ti.UI.FILL" width="Ti.UI.FILL" visible="false"/>
                            <View id="orangeView" backgroundColor="orange" top="0" left="0" height="Ti.UI.FILL" width="Ti.UI.FILL" visible="false"/>
                            <View id="yellowView" backgroundColor="yellow" top="0" left="0" height="Ti.UI.FILL" width="Ti.UI.FILL" visible="false"/>
                            <View id="greenView" backgroundColor="green" top="0" left="0" height="Ti.UI.FILL" width="Ti.UI.FILL" visible="false"/>
                            <View id="blueView" backgroundColor="blue" top="0" left="0" height="Ti.UI.FILL" width="Ti.UI.FILL" visible="false"/>
                        </Window>
                    </NavigationGroup>
                </Window>
            </Window>
        </Tab>
        <Tab title="Tab 2" icon="KS_nav_views.png">
            <Window title="Tab 2">
                <Label>I am Window 2</Label>
            </Window>
        </Tab>
    </TabGroup>
</Alloy>

    index.js

    $.index.open();

// set initial "toggle" value for button
// P.S. I did this in the controller rather than the model because
// true/false values are interpreted as strings and I always 
// forget to convert them!
$.filterButton.toggle = false;

var rainbow = [ $.redView, $.orangeView, $.yellowView, $.greenView, $.blueView ];


function changeView( e ){

    rainbow.forEach( function( colour ) {
        colour.visible = false;
    });

    switch( e.source.colour ) {
    case "red" :
        $.redView.visible = true;
        break;
    case "orange" :
        $.orangeView.visible = true;
        break;
    case "yellow" :
        $.yellowView.visible = true;
        break;        

    case "green" :
        $.greenView.visible = true;
        break;
    case "blue" :
        $.blueView.visible = true;
        break;
    }
}

function displayMenu( e ){

    Ti.API.debug( "e: " + JSON.stringify( e));

    // If the menu is opened
    if( $.filterButton.toggle ){
        $.mainWindow.animate({
            left:0,
            duration:400,
            curve:Ti.UI.ANIMATION_CURVE_EASE_IN_OUT
        });

        $.filterButton.toggle = false;
    }
    // If the menu isn't opened
    else{
        $.mainWindow.animate({
            left:150,
            duration:400,
            curve:Ti.UI.ANIMATION_CURVE_EASE_IN_OUT
        });

        $.filterButton.toggle = true;
    }
}

  • Important notes before you read the solution:

    The behavior of the slide menu is based on the theme you work with.

    On Default Titanium Theme (i.e. did not touch the tiapp.xml):

    • When navBarHidden == false then entire tabgroup slides to the left.

    • Else Tabgroup navbar stays and the all the tabs slide to the left.

    On holo them:

    • when you set you android section in tiapp.xml to be something like:
      <android xmlns:android="http://schemas.android.com/apk/res/android">
   <!--Add the following tool and manifest entries. 
       Also add your theme name: android:theme="@style/Theme.MyThemeName"
   -->
     <supports-screens android:anyDensity="false"/>
   <tool-api-level>14</tool-api-level>
   <manifest>
       <uses-sdk android:minSdkVersion="14" android:targetSdkVersion="16"/>
   </manifest>
</android>
      then:

    • When navBarHidden == false then entire tabgroup slides to the left (same as in default titanium theme).

    • Else only current tab window slides to the left.

    Finally here is the solution :)

    (function() {
    // The menu toggle state
    var toggled = false;

    var tabGroup = Ti.UI.createTabGroup({
        navBarHidden : true, // feel free to change this, and see the different behaviors as I wrote above
        zIndex : 2,
        backgroundColor : 'white', // This is important default is not good cause it's translucent and you'll see the slide menu below.
        exitOnClose : true
    });

    var slideMenu = Ti.UI.createWindow({
        zIndex : -1,
    });
    slideMenu.add(Ti.UI.createTableView({
        data : [{
            title : 'item 1'
        }, {
            title : 'item 2'
        }],
        top : 0,
        left : 0,
    }));

    tabGroup.addEventListener('open', function(e) {
        slideMenu.open();
    });

    //Open/close slide menu
    tabGroup.addEventListener('swipe', function(e) {
        toggle();
    });

    var tab1 = Ti.UI.createTab({
        title : 'tab1',
        window : Ti.UI.createWindow({
        })
    });

    var tab2 = Ti.UI.createTab({
        title : 'tab2',
        window : Ti.UI.createWindow({
        })
    });

    tabGroup.addTab(tab1);
    tabGroup.addTab(tab2);

    function toggle() {
        var left, right;
        if (toggled) {
            left = 0;
            right = 0;
        } else {
            left = 200;
            right = -200;
        }

        tabGroup.animate({
            left : left,
            right : right,
            duration : 400,
            curve : toggled ? Ti.UI.ANIMATION_CURVE_EASE_OUT : Ti.UI.ANIMATION_CURVE_EASE_IN
        }, function(o) {
            toggled = !toggled;
            tabGroup.applyProperties({
                left : left,
                right : -right,
            });
        });
    }


    tabGroup.open();

})();

  • Nav group is depreciated. Hence I had to opt some other way to do this job. Here's the code. It might help you guys too.

    
var curWin = Ti.UI.currentWindow;

var menuBtn = Ti.UI.createButton({
    title : "Menu",
    top : 20,
    left : 5,
    zIndex : 999
});

var windowView = Ti.UI.createView({
    top : 60,
    left : 0,
    right : 0,
    width : '100%',
    height : '100%',
    backgroundColor : "#252525",
    zIndex : 1
});
curWin.add(windowView);

curWin.add(menuBtn);
var menuWindow = Ti.UI.createView({
    top : 20,
    left : 0,
    width : 150
});

curWin.add(menuWindow);
//// ---- Menu Table
// Menu Titles
var menuTitles = [{
    title : 'Menu 1'
}, {
    title : 'Menu 2'
}, {
    title : 'Menu 3'
}, {
    title : 'Menu 4'
}, {
    title : 'Menu 5'
}, {
    title : 'Menu 6'
}];
// Tableview
var tableView = Ti.UI.createTableView({
    data : menuTitles
});
menuWindow.add(tableView);
var toggle = false;
menuWindow.hide();
menuBtn.addEventListener('click', function(e) {
    if (toggle) {
        menuWindow.hide();
        toggle = false;
        windowView.left = 0;
        menuBtn.left = 5;
    } else {
        menuWindow.show();
        toggle = true;
        windowView.left = 150;
        menuBtn.left = 160;
    }
});

windowView.addEventListener('click', function(e) {
    if (toggle) {
        menuWindow.hide();
        toggle = false;
        windowView.left = 0;
        menuBtn.left = 5;
    }
});
