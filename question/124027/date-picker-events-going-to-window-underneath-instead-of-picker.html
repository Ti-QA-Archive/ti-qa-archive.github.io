date picker: events going to window underneath instead of picker

I'm attempting to position a date picker over my window. The picker shows up fine, but it does not correspond to change / click events - the events go to the items underneath the picker, as if the picker wasn't there at all. ie. if I click on the date to attempt to change it, the click registers with whatever UI element is under that part of the picker.

Interestingly the Save and Cancel buttons work fine.

What am I doing wrong?

var datePickerview = Titanium.UI.createView({
    visible: false, backgroundColor: "black",
    bottom: 0, height: 'auto'
});

var datePicker = Titanium.UI.createPicker({
    type: Titanium.UI.PICKER_TYPE_DATE,
    // selectionIndicator: true,
    bottom: 40
});
datePicker.addEventListener('change',function(e){
    Ti.API.info("=== CHANGE ====");
        datePicker.value = e.value;
});

var dateSaveButton = Ti.UI.createLabel({
    text: 'Save',
    right: 10, width: 80, height: 40, bottom: 0,
    textAlign: 'center', color: 'white', backgroundColor: 'black', borderWidth: 1
});
var dateCancelButton = Ti.UI.createLabel({
    text: 'Cancel',
    left: 10, width: 80, height: 40, bottom: 0,
    textAlign: 'center', color: 'white', backgroundColor: 'black', borderWidth: 1
});
dateSaveButton.addEventListener("click", function(e){
    Ti.API.info("=== SAVE ===: " + datePicker.value);
    if (datePicker.value) {
        dateValueLabel.text = datePicker.value;
        dateValueLabel.color = 'black';
    }
    datePickerview.visible = false;
    // datePickerWin.close();
});
dateCancelButton.addEventListener("click", function(e){
    datePickerview.visible = false;
    // datePickerWin.close();
});

datePickerview.add(dateSaveButton);
datePickerview.add(dateCancelButton);
datePickerview.add(datePicker);
