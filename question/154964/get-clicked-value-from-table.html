Get clicked value from table

Hi.

How do i get my value "custom_item" from my table?

I will have different custom_item values so i need to get the one from the row i clicked.

Here is my code:

var tableData = [ 
{title: 'Apples',
  font:{fontSize:'17dp',
        fontWeight:'bold'},
        custom_item : '1261'
}, 
{title: 'Bananas',
 font:{fontSize:'17dp',
        fontWeight:'bold'},
        custom_item : '1262'
}, 
{title: 'Carrots',
 font:{fontSize:'17dp',
        fontWeight:'bold'},
        custom_item : '1263'}, 
{title: 'Potatoes',
 font:{fontSize:'17dp',
        fontWeight:'bold'},
        custom_item : '1264'} ];
var table = Ti.UI.createTableView({
  data: tableData
});
win.add(table);

table.addEventListener('click', function(e) {
    Ti.API.info(tableData.e.custom_item);
});

(tableData.e.custom_item) is wrong…..

Thanx

  • Accepted Answer

    try

    table.addEventListener('click', function(e) {
    Ti.API.info(e.rowData.custom_item);
});

    or

    table.addEventListener('click', function(e) {
    Ti.API.info(e.row.custom_item);
});
