function showRequestResult(e) {
    var s = '';
    if (e.success) {
        s = "SUCCESS";
        if (e.result) {
            s += "; " + e.result;
        }
        if (e.data) {
            s += "; " + e.data;
        }
        if (!e.result && !e.data) {
            s = '"success", but no data from FB.  I am guessing you cancelled the dialog.';
        }
    } else if (e.cancelled) {
        s = "CANCELLED";
    } else {
        s = "FAIL";
        if (e.error) {
            s += "; " + e.error;
        }
    }
    alert(s);
}
