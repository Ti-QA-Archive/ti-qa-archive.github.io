     xhr.onload = function () {

    var json = json.vestigingen;
    var pos;
    var annotationArray = [];

    var mapview = Titanium.Map.createView({
            animate: true,
            region: {
                latitude: 51.68278,

    win1.add(mapview);

    for (pos = 0; pos < json.length; pos++) {
        var annotation = Titanium.Map.createAnnotation({
            latitude: obj.map_latitude,
