textfield hintLabel text styling

Anyone know if you can change the color other style attributes of the hint text in a label? I can't seem to find any mention of it.

    Dan already pointed you to the right direction. The following example is an implementation for a textarea (witch does not have the hintText option). You can use any color you want.

    Cheers

    // This is the 'Hint Text'
var LabelCommentHintText = Ti.UI.createLabel({
            text: 'Type your comment here (optional)',
// Here I disable the touch on this particular label
            touchEnabled: false,
            height: 100,
            top: 260,
            left: 17,
            width: 330,
// here you can change the color since this is a simple label
            color: 'blue',
            visible: true,
            font: {
                fontSize: 15,
                fontFamily: 'TrebuchetMS-Bold'
            }
        });
// This is the text area  
        var comment = Titanium.UI.createTextArea({
            value: '',
            height: 100,
            top: 260,
            left: 10,
            width: 330,
            autocorrect: false,
            autocapitalization: false,
            borderRadius: 10,
            font: {
                fontSize: 15,
                fontFamily: 'TrebuchetMS-Bold'
            }
        });
// Hide  HintText on TextArea focus
        comment.addEventListener('focus', function (e) {
            LabelCommentHintText.visible = false;
        });
