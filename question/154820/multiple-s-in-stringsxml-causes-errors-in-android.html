  • I ended up creating my own foratting function. You enter code like: {0} is order number {1} at address {2}

    
this.format = function(s,arguments) {

        var args = arguments;

        var value= s.replace(/{(\d+)}/g, function(match, number) {
            alert(match+' > '+number+'> '+args[number]);
            return typeof args[number] != 'undefined' ? args[number] : match;
        });
        return value;

    }
