Copy and sort list

Hi,

in my geoproject I have a list of panoramio photos with coordinates.
For flying on latitude and longitudes I need two list: one list with photos sorted by lat and an other on lngs.

Here my code:

function getLatLngList() {
    var panos = datas.panoramio; // result of xhr req.
    var len = panos.length;
    var i;
    var lats = []; // temp list for ordered points
    for (i=0;i<len;i++) {
        lats.push(panos[i]); 
    }  
    // now we have an identical copy
    // sortby lat:
    lats.sort(function(a,b) {
         if (a.lat < b.lat) {return -1;}
          if (a.lat > b.lat) {return 1;}
          return 0;
    }); 
    var latlist = [];
    // copy in list
    for(i=0;i<len;i++) {
        latlist.push(lats[i].i); // collect as copy
    }
    lats = null; // remnove from memory
    var lngs = [];
    for (i=0;i<len;i++) {
        lngs.push(panos[i]);
    }
    lngs.sort(function(a,b) {
        if (a.lat < b.lat) {return -1;}
          if (a.lat > b.lat) {return 1;}
          return 0;
    });
    var lnglist = [];
    for(i=0;i<len;i++) {
        lnglist.push(lngs[i].i);
    }
    lngs=null;
    return {lats:latlist,lngs:lnglist};
}

The problem: in both lists is the same order. ???
Altough I think I have made a copy …

Any ideas?
