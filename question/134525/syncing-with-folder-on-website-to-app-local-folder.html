    Hi Samuel,

    You can request and save a file like this:

    
var xhr = Ti.Network.createHTTPClient();
xhr.open('GET', 'http://yourserver/pdfs/file.pdf');
xhr.send();

xhr.onload = function(){
var data = this.responseData;

//get or create directory
var dir = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, 'pdf');
if(!dir.exists()) 
{
    dir.createDirectory();
}
//create new file
var file = Ti.Filesystem.getFile(dir, 'file.pdf');
file.write(data);
}
