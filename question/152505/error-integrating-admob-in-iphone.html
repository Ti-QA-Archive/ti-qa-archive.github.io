  • It Works fine for me. You can try this code

    var Admob = require('ti.admob');  
     var View = Ti.UI.createView({
        height : 60,
        width : 320,
        bottom : 0 ,
        left : 0 ,
        backgroundColor : 'black',
    })
    bootWindow.add(View);

    var adView = Admob.createView({
        height : View.height,
        width : Ti.Platform.displayCaps.platformWidth,
        bottom : 0,
        publisherId : 'Your_key',
        testing : false,
    })
    View.add(adView);
    adView.addEventListener(Admob.AD_RECEIVED,function() {
     Ti.API.info('ad receiced')
    });  
    adView.addEventListener(Admob.AD_NOT_RECEIVED,function() {
      Ti.API.info('ad not received')
    })
