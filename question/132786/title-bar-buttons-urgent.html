    • Pasting this in app.js should get you started down this path. ButtonBar is what you want.

      var tg = Ti.UI.createTabGroup();

var win = Ti.UI.createWindow({backgroundColor:'white'});
var t1 = Ti.UI.createTab({
    title:'recents',
    window:win
});
tg.addTab(t1);

var buttonBar = Ti.UI.createButtonBar({
    labels:['All', 'Missed'],
    index:0,
    width:120,
    style:Ti.UI.iPhone.SystemButtonStyle.BAR
});
win.titleControl = buttonBar;

var t2 = Ti.UI.createTab({
    title:'other',
    window:Ti.UI.createWindow({title:'other',backgroundColor:'white'})
});
tg.addTab(t2);

tg.open();
