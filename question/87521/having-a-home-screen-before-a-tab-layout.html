Having a home screen before a tab layout?

This is a very basic question, but something that I still haven't found a proper solution for.

We have an application that we want to show the splashscreen, then go to a sort of "home" screen which looks like the splashscreen without tabs but has a bunch of buttons on it - When you then click one of the buttons we want to load that view which is placed in a tabgroup and then you have regular tabs to click around.

Any ideas how to do this basic thing on Android?

On iPhone this is easy with the following piece of code:

var tabData = [
    {title:lang('Home'),icon:'tab_home.png',url:'views/splash.js',navBarHidden:true,tabBarHidden:true},
    {title:lang('Search'),icon:'tab_search.png',url:'views/search_view.js'},
    {title:lang('Map'),icon:'tab_map.png',url:'views/map_view.js'},
    {title:lang('Functions'),icon:'tab_settings.png',url:'views/functions_view.js',badge:unreadNum},
    {title:lang('Contact'),icon:'tab_contact.png',url:'views/contact_view.js'}
];

var tabGroup = Titanium.UI.createTabGroup();
for(var i in tabData){
    var win = Titanium.UI.createWindow({
        url: tabData[i].url,
        navBarHidden:(tabData[i].navBarHidden?true:false),
        tabBarHidden:(tabData[i].tabBarHidden?true:false),
        lang:lang,
        db:db,
        config:config,
        debug:debug
    });

    tabGroup.addTab(Titanium.UI.createTab({
        title:tabData[i].title,
        icon:tabData[i].icon,
        badge:(tabData[i].badge?tabData[i].badge:null),
        window:win
    }));
}

tabGroup.open();

    • I have found an example which works only on iOS:

      http://developer.appcelerator.com/question/79491/login-and-tabgroup

      I modified it a bit and it seems to work now on Android, but I don't have iOS device to test, so maybe it doesn't work 100%-ly on those.

      The idea is to not close the tabGroup when user wants to log out, as it gives just a blank screen on Android (which you have to 'escape' manually). Instead, open 'win' again with navBarHidden:true and it will cover the tabs. I hope this behavior will not break in the next Titanium mobile release. :)

      The modified code for Android for those who also struggled (thanks for Aaron Saunders for the iOS version!):

      // this sets the background color of the master
// UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#fff');

// create tab group
var tabGroup;

var win = Titanium.UI.createWindow({
    title:'TabViewLogin',
    tabBarHidden:true,
    backgroundColor:'#fff'
});

var username = Titanium.UI.createTextField({
    color:'#336699',
    top:10,
    left:10,
    width:300,
    height:40,
    hintText:'Username',
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});
win.add(username);

var password = Titanium.UI.createTextField({
    color:'#336699',
    top:60,
    left:10,
    width:300,
    height:40,
    hintText:'Password',
    passwordMask:true,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});
win.add(password);

var loginBtn = Titanium.UI.createButton({
    title:'Login',
    top:110,
    width:90,
    height:35,
    borderRadius:1,
    font:{fontWeight:'bold',fontSize:14}
});
win.add(loginBtn);
win.open();

Titanium.App.addEventListener('logout', function(e) {
    win.open({navBarHidden:true});
});

loginBtn.addEventListener('click',function(e)
{
    if ( tabGroup == undefined ) {
        tabGroup = Titanium.UI.createTabGroup();
        //
        // create base UI tab and root window
        //
        var win1 = Titanium.UI.createWindow({  
            title:'Tab 1',
            backgroundColor:'#fff'
        });
        var tab1 = Titanium.UI.createTab({  
            icon:'KS_nav_views.png',
            title:'Tab 1',
            window:win1
        });
        var label1 = Titanium.UI.createLabel({
            color:'#999',
            text:'I am Window 1',
            font:{fontSize:20,fontFamily:'Helvetica Neue'},
            textAlign:'center',
            width:'auto'
        });
        win1.add(label1);

        //
        // create controls tab and root window
        //
        var win2 = Titanium.UI.createWindow({  
            title:'Tab 2',
            backgroundColor:'#eee'
        });
        var tab2 = Titanium.UI.createTab({  
            icon:'KS_nav_ui.png',
            title:'Tab 2',
            window:win2
        });
        var label2 = Titanium.UI.createLabel({
            color:'#999',
            text:'I am Window 2',
            font:{fontSize:20,fontFamily:'Helvetica Neue'},
            textAlign:'center',
            width:'auto'
        });
        win2.add(label2);

        var button = Ti.UI.createButton({
            title: "Logout"
        });
        button.addEventListener('click',function(e) {
            Ti.App.fireEvent('logout');
        });
        win2.add(button);

        //
        //  add tabs
        //
        tabGroup.addTab(tab1);
        tabGroup.addTab(tab2);
        tabGroup.open();
    }
    // open tab group
    win.close();
});

  • The best solution I've found for doing splash screens is to maintain two full sized views in a window. Then you'd just switch between them. We have a splash screen that goes to a login screen that then becomes the home page. We use a variable to track which one is supposed to be showing and use a function to determine which view/function should be run for each state of the variable.

    This also solves the issue of them hitting the tab and having them not see the login screen or splash screen again.

    The other option is to just have the splash screen opened when the tab was originally drawn as an independent window not apart of the tab group navigation.

    So to summarize, you can either use modal windows or full screen views to do splash screens.
