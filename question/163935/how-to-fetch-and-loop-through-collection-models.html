How to fetch and loop through collection models

Hello,

How can I loop through the models from a collection?
I need to plot annotations to a map (using alloy).

    I do something like this:

    var markers = Alloy.Collections.markers;

markers.fetch({
  success: displayMarkers,
  error: Ti.API.error
});

function displayMarkers(){
  markers.each( function(marker){

    var markerAnnotation = map.createAnnotation({
      latitude: marker.get('latitude'),
      longitude: marker.get('longitude'),
      title: marker.get('title'),
      subtitle: marker.get('subtitle')
    });

    $.mapview.addAnnotation(markerAnnotation);
  });
}
