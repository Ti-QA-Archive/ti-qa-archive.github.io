  • For the benefit of others

    var Cloud = require('ti.cloud');
var form = Ti.UI.currentWindow;

var addbtn = Ti.UI.createButton({
    top : 25,
    height : 70,
    left : 370,
    width : 100,
    title : 'Add'
});

var label = Ti.UI.createLabel({
    top : 30,
    height : 70,
    left : 20,
    text : 'Enter Your Fieldname',
    color : 'black',
    font : {
        fontSize : '15dp',
        fontFamily : 'Calibri',
    }
});

var data = Ti.UI.createTextField({
    top : '50',
    height : '70',
    width : '150'
});

var tableview = Ti.UI.createTableView({
    top : '100',
    height : '800',
    width : '600',
});

var tabledata = [];

for (var i = 0; i <= 2; i++) {
    var row = Ti.UI.createTableViewRow({
        rowIndex : i,
        height : 110,
        backgroundColor : 'white'
    });
    var tf1 = Ti.UI.createTextField({
        top : 20,
        height : 70,
        width : 370,
    });
    row.add(tf1);
    tabledata.push(row);
}
tableview.setData(tabledata);

addbtn.addEventListener('click', function (e) {
    var tabledata = [];
    var row = Ti.UI.createTableViewRow({
        rowIndex : 12,
        height : 110,
        backgroundColor : 'white'
    });

    var tf1 = Ti.UI.createTextField({
        top : 20,
        height : 70,
        width : 370,
    });

    row.add(tf1);
    tableview.appendRow(row);
});
var submitbtn = Ti.UI.createButton({
    top : 25,
    height : 70,
    left : 270,
    width : 100,
    title : 'Submit'
});

submitbtn.addEventListener('click', function (e) {
    Cloud.Objects.create({
        classname : 'cars',
        fields : {
            make : tf1.value,
            // color: tf1.value,
            // year: tf1.value
        }
    }, function (e) {
        if (e.success) {
            var car = e.cars[0];
            alert('Success:\n' + 'id: ' + car.id + '\n' + 'make: ' + car.make + '\n' + 'color: ' + car.color + '\n' + 'year: ' + car.year + '\n' + 'created_at: ' + car.created_at);
        } else {
            alert('Error:\n' + ((e.error && e.message) || JSON.stringify(e)));
        }
    });
});

form.add(submitbtn);
form.add(addbtn);
form.add(label);
form.add(tableview);

  • Hi

    I am confused by your code because you appear to want to be able to store the answer to three values in the cloud but allow the ability to add extra rows - thus fields.

    That said this will have to be a generic answer to show you one way to gain the reference you are after. The key here is scope - you need to access some means of determining the value of something that in your code no longer has scope from your button.

    The easiest way for you is to use an array and then add your text fields to that array - later on - reference the array. There are more re-usable approaches but this is the best fit for your current coding style.

    var fields = [];
for (var i = 0; i <= 2; i++) {
    var row = Ti.UI.createTableViewRow({
        rowIndex : i,
        height : 110,
        backgroundColor : 'white'
    });
    field.push(Ti.UI.createTextField({
        top : 20,
        height : 70,
        width : 370,
    }));
    row.add(fields[fields.length-1]);
    tabledata.push(row);
}
tableview.setData(tabledata);

addbtn.addEventListener('click', function (e) {
    var tabledata = [];
    var row = Ti.UI.createTableViewRow({
        rowIndex : 12,
        height : 110,
        backgroundColor : 'white'
    });

    field.push(Ti.UI.createTextField({
        top : 20,
        height : 70,
        width : 370,
    }));
    row.add(fields[fields.length-1]);
    tableview.appendRow(row);
});

    And then in the button;

    submitbtn.addEventListener('click', function (e) {
    Cloud.Objects.create({
        classname : 'cars',
        fields : {
            make : fields[0].value, // example reference
            // color: fields[1].value, // example reference
            // year: fields[2].value // example reference
        }
    }, function (e) {
        if (e.success) {
            var car = e.cars[0];
            alert('Success:\n' + 'id: ' + car.id + '\n' + 'make: ' + car.make + '\n' + 'color: ' + car.color + '\n' + 'year: ' + car.year + '\n' + 'created_at: ' + car.created_at);
        } else {
            alert('Error:\n' + ((e.error && e.message) || JSON.stringify(e)));
        }
    });
});
