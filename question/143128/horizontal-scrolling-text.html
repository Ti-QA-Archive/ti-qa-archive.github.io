    Hi Armindo,

    You can use the following code as a sample for animation:

    var win = Titanium.UI.createWindow({ 
    title:'Marquee',
    backgroundColor:'#fff'
});

var view = Ti.UI.createView({
    width:'100%',
});
var label = Titanium.UI.createLabel({
    left: 100,
    color:'black',
    text:'horizontal auto scrolling text'
});

var animation = Titanium.UI.createAnimation({
    left:-150,
    duration:5000
});

animation.addEventListener('complete',function() {
    animation.left = 100;
    label.animate(animation); 
});

label.animate(animation);
view.add(label);
win.add(view);

win.open();

    If you feel this addresses your need, then please accept it as an answer

    • Hi Armindo,

      i think this is what you need (text never stops scrolling):

      var win = Titanium.UI.createWindow({ 
    title:'Marquee',
    backgroundColor:'#fff'
});

var view = Ti.UI.createView({
    width:'100%',
});
var label = Titanium.UI.createLabel({
    left: 100,
    color:'black',
    text:'horizontal auto scrolling text'
});
var left = true;
var animation = Titanium.UI.createAnimation();
animation.duration = 5000;
animation.left=0;
var animationHandler = function() {
  if(left){
      animation.left = 100;
      left = false;
  }else{
      animation.left = 0;
      left = true;
  }

  label.animate(animation);
};
animation.addEventListener('complete',animationHandler);
label.animate(animation);

view.add(label);
win.add(view);

win.open();
