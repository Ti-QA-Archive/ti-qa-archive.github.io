function tablerows(){
    var db = Ti.Database.open('db_app');
    var rows = db.execute('SELECT * FROM aaa ');

    var tbl_data = [];
    while (rows.isValidRow()){

        var row = Ti.UI.createTableViewRow({
            id: rows.fieldByName('id'),
            height: 40,
            selectionStyle: 'none'
        });
        var row_view = Ti.UI.createView({
            backgroundColor: '#fff',
            borderRadius: 10,
            right: 3,
            top: 3,
            bottom: 3,
            left: 2,
            opacity: 0.5
        });
        var label = Ti.UI.createLabel({
            text: rows.fieldByName('name'),
            left: 8
        });

         var add = Ti.UI.createImageView({
            image: '../images/add.png',
            right: 0,
            width: 29,
            height: 29
        });    

        var rem = Ti.UI.createImageView({
            image: '../images/remove.png',
            right: 0,
            width: 29,
            height: 29
        });    

        row.filter = label.text;

         row.add(row_view);
         row.add(label);         
        row.add(add);
        row.add(rem);

         rem.hide();

        tbl_data.push(row);
        rows.next();
    }
    tableview.setData(tbl_data);

    rows.close();
    db.close();
}
