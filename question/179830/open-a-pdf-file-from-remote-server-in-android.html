    Hello Bunty Rohra,

    Normally pdf file does not support on android. So you can use PDF reader module from marketplace or Google Apps has a fantastic little service for viewing PDF files. Here is the sample code

    Ti.UI.setBackgroundColor('#222');

var win = Ti.UI.createWindow({
    navBarHidden : true,
    exitOnClose : true
});

// Create a Button.
var open = Ti.UI.createButton({
    title : 'open',
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,
    top : 30,

});

// Listen for click events.
open.addEventListener('click', function() {
    var pdfUrl = "http://www.irs.gov/pub/irs-pdf/fw4.pdf";

    Ti.Platform.openURL("https://docs.google.com/gview?embedded=true&url=" + Ti.Network.encodeURIComponent(pdfUrl));

});

// Add to the parent view.
win.add(open);

win.open();

    If you have any concern, then let me know.

    Thanks.

  • Dear Bunty,
    There are many way to open pdf file in android,please check below options and use as per your requirement.

    1.Use marketplace module for it but it's a paid module.
https://marketplace.appcelerator.com/apps/6897?restoreSearch=true#!overview

2.You can open pdf in webview in android using this code just place your remote PDF Url in url variable.
var win = Titanium.UI.createWindow({
    title:'My First Window',
    backgroundColor:'#fff',
});
//put your pdf file remote url in url variable.
var pdfUrl = "http://docs.google.com/viewer?embedded=true&url="+ url ;
        var webView = Ti.UI.createWebView({
            left : 0,
            url : pdfUrl,
            top : 10,
            width : 320,
            height : 350,
        });
win.add(webView);
win.open();

3.you can open pdf url as an external url but that will be outside from your app.
var pdfUrl = "http://www.xxx.com/test.pdf";
Ti.Platform.openURL("https://docs.google.com/gview?embedded=true&url=" + Ti.Network.encodeURIComponent(pdfUrl));

    Thanks,
    Jitendra
