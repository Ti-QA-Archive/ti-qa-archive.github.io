How to style buttons?

I have a button, like this:

var button = Ti.UI.createButton({
    height:28,
    width:64,
    title:"Search",
    top:140,
    left:12,
    color: 'gray',
    font:{fontSize:14, fontWeight:'normal', fontFamily:'Arial'}
});
win.add(button);

It is a white button, with gray font, and when you press it, the background color turns blue, and the font turns white..

Question:

Is it possible to change the initial color of the background to blue and the font color to white, and when you press it, the background turns red and the font color turns green??

I have tried using the "backgroundColor" property, but it doesnt seem to work.

Thanks again guys..

  • Hope this is what you are looking for

    
var button = Ti.UI.createButton({
    title:"I can blink",
    backgroundColor : 'blue',
    backgroundImage : 'none',
    color: 'white',
    selectedColor: 'green'
});
win.add(button)
button.addEventListener('touchstart', function(){
    this.setBackgroundColor('red');
});
button.addEventListener('touchend', function(){
    this.setBackgroundColor('blue');
});

  • Hello Donal,

    Try this…

    
var button = Ti.UI.createButton({
    height:28,
    width:64,
    title:"Search",
    top:140,
    left:12,
    backgroundColor : 'blue',
    color: 'white',
    font:{fontSize:14, fontWeight:'normal', fontFamily:'Arial'},
    style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN
});
win.add(button);

button.addEventListener('click',function(e){
    button.backgroundColor = 'red'
    button.color = 'green';
});
