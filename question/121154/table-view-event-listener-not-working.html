table view event listener not working

I have the requirement of showing the first label in bold of the table row and rest in normal. So i am able to do so but the event listener on them is not working. ON clicking the table row nothing happens, I am using titanium sdk as 1.6.2 for Iphone deployment.

var win = Titanium.UI.currentWindow;

var categories = [
        {title: "BMW",hasChild:true,category_name:"BMW",test:'BMW.js'},
        {title:'HONDA',hasChild:true,category_name:"HONDA",test:'HONDA.js'},
        {title:'NISSAN',hasChild:true,category_name:"NISSAN",test:'NISSAN.js'},
        {title:'CHEVY',hasChild:true,category_name:"CHEVY",test:'CHEVY.js'},
        {title:'DUCATI',hasChild:true,category_name:"DUCATI",test:'DUCATI.js'},
        {title:'MVR',hasChild:true,category_name:"MVR",test:'MVR.js'}
    ];




var rows = [];

for(var v in categories){
  var row = Titanium.UI.createTableViewRow({

      hasChild:true,
      category_name:categories[v].category_name
  });


  var label = Ti.UI.createLabel({
       text:categories[v].title,
       font:{fontSize:20},

  });
  if(Number(v) === 0){
    label.font = {fontWeight:'bold',fontSize:22};
}

  row.add(label);
  rows.push(row); 
}

var table = Titanium.UI.createTableView({
    data:rows   
});
table.addEventListener('click', function(e)
{


    if (e.rowData.test)
    {

        var win1 = Titanium.UI.createWindow({
            url:e.rowData.test,
            title:e.rowData.title
        });
        win1.open();
    }
});    


win.add(table);

thanks

    Try changing your table eventlistener to this:

    table.addEventListener('click', function(e)
{
var i = e.index

    if (categories[i].test)
    {

        var win1 = Titanium.UI.createWindow({
            url:categories[i].test,
            title:categories[i].title
        });
        win1.open();
    }
});

    • to be honest you dont even need the 'test' part at all, you could just use the title + '.js'

      Example

      table.addEventListener('click', function(e)
{
        var i = e.index
        var win1 = Titanium.UI.createWindow({
            url:categories[i].title + '.js',
            title:categories[i].title
        });
        win1.open();
});

      Every row will have a title, you just need to make sure your .js files are the same as your row titles.
