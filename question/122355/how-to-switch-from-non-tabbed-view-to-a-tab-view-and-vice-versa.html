I can't seem to understand how the navigation is supposed to work. Can anyone supply a short example? I either lose my back button or I have to create my own which now is missing the animation. Any clue? What am I doing wrong?

table.js

var tableView = Titanium.UI.createTableView({
    data:rowData
});

tableView.addEventListener('click', function(e){
    var i = e.index;
    if (rowData[i].test){
        // set properties on the window object, then open.  we will print them out in the new window
        var groupDetails = Titanium.UI.createWindow({
            url:'tabs.js'
        });
        groupDetails.objProp1 = {groupid: rowData[i].test, groupname: titleData[i]};

        groupsTabGroup.addTab(groupsTab);
        groupsTabGroup.open();
    }
});

//Add the table view to the window
win.add(tableView);
