Cannot place same button object on mutiple views in a ScrollableView object

We want to place a common help button on each of the subviews that were loaded into a horizontal scrollableView object. When we do that, the button only shows up on one of the views. The workaround was to create multiple instances of the same button and place the unique button instances on each subview.

Is this a bug with scrollableViews? (or a feature?)

Here's code that illustrates the problem


var frontWindow = Titanium.UI.createWindow();


var theview1 = Ti.UI.createView({
    backgroundColor: 'blue',
});
var theview2 = Ti.UI.createView({
    backgroundColor: 'red',
});
var theview3 = Ti.UI.createView({
    backgroundColor: 'white',
});

var button1 = Ti.UI.createButton({
    color:'red',
    title:'Titanium 1',
    height:100,
    width:200
});

theview1.add(button1);
theview2.add(button1);
theview3.add(button1);

var viewsarray = [theview1, theview2, theview3];

var scrollview =  Ti.UI.createScrollableView({
    backgroundColor: '#F6D565',
    views: viewsarray,
    showPagingControl:true,
    pagingControlColor:'#336699',
    pagingControlHeight:20,
});
frontWindow.add(scrollview);

frontWindow.open();

Work-around code


var frontWindow = Titanium.UI.createWindow();


var theview1 = Ti.UI.createView({
    backgroundColor: 'blue',
});
var theview2 = Ti.UI.createView({
    backgroundColor: 'red',
});
var theview3 = Ti.UI.createView({
    backgroundColor: 'white',
});

var button1 = Ti.UI.createButton({
    color:'red',
    title:'Titanium 1',
    height:100,
    width:200
});
var button2 = Ti.UI.createButton({
    color:'blue',
    title:'Titanium 2',
    height:100,
    width:200
});
var button3 = Ti.UI.createButton({
    color:'black',
    title:'Titanium 3',
    height:100,
    width:200
});

theview1.add(button1);
theview2.add(button2);
theview3.add(button3);

var viewsarray = [theview1, theview2, theview3];

var scrollview =  Ti.UI.createScrollableView({
    backgroundColor: '#F6D565',
    views: viewsarray,
    showPagingControl:true,
    pagingControlColor:'#336699',
    pagingControlHeight:20,
});
frontWindow.add(scrollview);

frontWindow.open();
