optionDialog - buttons & cancel don't work.

Using the examples shown here:
http://docs.appcelerator.com/titanium/latest/#!/api/Titanium.UI.OptionDialog

I am unable to get any of the buttons (including cancel) to work. The dialog opens, but I ca't dismiss it.

I'm running it in Mobile Web Preview in Browser.

var win3 = Titanium.UI.createWindow({  
    title:'test',
    backgroundColor:'#fff',
    navBarHidden : false
});

var optionsBtn = Titanium.UI.createButton({
    title: '?',
    top: 0,
    right: 0,
    width: "64dp",
    height: "64dp"
});

var settings = Ti.UI.createOptionDialog({
  cancel: 2,
  options: ['Disconnect', 'Exit', 'Cancel'],
  selectedIndex: 2,
  title: 'Options'
});

optionsBtn.addEventListener('click',function(e){
   var dialog = settings.show();
});
settings.addEventListener('click',function(e){
    Titanium.API.info("Clicked"); // never called
});


win3.add(optionsBtn);
win3.open();

Any ideas?
