Creating Gradient Toolbar on iOS

I'm trying to create a gradient background toolbar that ranges from dark gray to medium gray. I'd like the title in a centered label and a login button at the right. So far, everything I try puts the title on the left and the button not quite to the right. The background is always a system blue gradient. Any hints?

  homeWindow = Ti.UI.createWindow({
    backgroundColor: '#000',
    color: '#fff'
  });

  flexButton = Ti.UI.createButton({
    style: Titanium.UI.iPhone.SystemButtonStyle.FLEXIBLE_SPACE
  });

  loginButton = Ti.UI.createButton({
    title: 'login',
    style: Titanium.UI.iPhone.SystemButtonStyle.BAR,
    right: 0
  });

  mainTitle = Ti.UI.createLabel({
    text: 'iPet: Home Health Care',
    textAlign: 'center',
    color: '#fff',
    width: '100%'
  });

  titleBar = Ti.UI.createToolbar({
    items: [mainTitle, flexButton, loginButton],
    backgroundGradient: {
      type: 'linear',
      startPoint: {
        x: '50%',
        y: '0px'
      },
      endPoint: {
        x: '50%',
        y: '40px'
      },
      colors: [
        {
          color: '#333',
          offset: 0.0
        }, {
          color: '#777',
          offset: 1.0
        }
      ]
    },
    top: 0,
    height: '40px',
    width: '100%'
  });

  homeWindow.add(titleBar);

  homeWindow.open();
