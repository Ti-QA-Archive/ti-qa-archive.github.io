// getting device token
function getDeviceToken() {
    if (OS_ANDROID) {
        var CloudPush = require('ti.cloudpush');
        CloudPush.enabled = true;
        CloudPush.debug = true;
        CloudPush.showTrayNotificationsWhenFocused = true;
        CloudPush.focusAppOnPush = false;
        CloudPush.retrieveDeviceToken({
            success : function deviceTokenSuccess(e) {
                deviceToken = e.deviceToken;
                alert(deviceToken);
                registerPush();                        
            },
            error : function deviceTokenError(e) {
                alert('Failed to register for push! ' + e.error);
            }
        });                                 
    } else if (OS_IOS) {                                                        
        Ti.Network.registerForPushNotifications({
            // Specifies which notifications                             to receive
            types : [Ti.Network.NOTIFICATION_TYPE_BADGE, Ti.Network.NOTIFICATION_TYPE_ALERT, Ti.Network.NOTIFICATION_TYPE_SOUND],
            success : function(e) {
                alert(JSON.stringify(e));
                deviceToken = e.deviceToken;
                registerPush();
                alert("deviceToken = " + deviceToken);
            },
            error : function(e) {                                                                        
                alert("573 Error: " + e.message);
            },
            callback : function(e) {
                alert("push notification received" + JSON.stringify(e.data));
            }
        });
    }
}

// register for push notification on cloud server
function registerPush(){
    if (OS_ANDROID) {
        Cloud.PushNotifications.subscribe({
            channel: 'customers',
            type:'gcm',                                                                        
            device_token: deviceToken
        }, function (e) {
            if (e.success) {
                alert('Success Channel:'+((e.error && e.message) || JSON.stringify(e)));
            } else {
                alert('Error Channel:' + ((e.error && e.message) || JSON.stringify(e)));
            }
        });
     } else if(OS_IOS) {
         Cloud.PushNotifications.subscribe({
            channel: 'customers',
            type:'ios',                                                                        
            device_token: deviceToken
        }, function (e) {
            if (e.success) {
                alert('Success Channel:'+((e.error && e.message) || JSON.stringify(e)));
            } else {
                alert('Error Channel:' + ((e.error && e.message) || JSON.stringify(e)));
            }
        });

     }
}
