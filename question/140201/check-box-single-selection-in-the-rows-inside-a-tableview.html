var checkcount = 0;
var ids = [];
tableView.addEventListener('click', function (e) {

  var index = e.index;
  var section = e.section;

   if(e.row.hasCheck == true){
        section.rows[index].hasCheck = false;
        section.rows[index].children[0].color = '#5CA651';
        checkcount--;
        ids[checkcount] = '';
    }

   else{

     setTimeout(function()
     {
                if(section.rows[index].hasCheck == true){

           section.rows[index].hasCheck = false;
           section.rows[index].children[0].color = '#5CA651';

        }


        ids[checkcount] = e.rowData.chatUserID;
        section.rows[index].hasCheck = true;
        section.rows[index].children[0].color = '#5CA651';
        checkcount++;
     },250);
   }

});
