    Hi Dharshana,

    Try this code that may solve your problem.You need to use Ti.App.fireEvent instead of Ti.API.fireEvent.

    Checkout this documentation

    label2.addEventListener('click', function(){
    //label2.text='Some text'; //********** <= This is Work
    Ti.App.fireEvent('fevent',{msg:'dsfsdf'}); // THis is now working
});

Ti.App.addEventListener('fevnet', function(e){
    label2.text = 'form fired event';
})
