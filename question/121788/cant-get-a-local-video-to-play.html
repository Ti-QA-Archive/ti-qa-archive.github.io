Can't get a local video to play

I've followed the method used in KitchenSink but when I click the 'imageView' the video keeps loading and never starts playing. here's my code.

(function() {
    cuf.ui.createScrollableListVideos = function(properties) {
        var mainView = Titanium.UI.createView({
            top: properties.top,
            width: 590,
            height: 115,
            zIndex: 1500,
        });

        var imageView = Titanium.UI.createView({
            top: 0,
            left: 0,
            width: 170,
            height: 115,
            backgroundImage: properties.thum
        });

        var videoViewer = Titanium.UI.createView({
            top: 0,
            left: 0,
            width: 590,
            height: 115,
            backgroundColor: 'black',
            title: 'Video Viewer',
            zIndex: 1600
        });

        var videoOptions = {
            url: Titanium.Filesystem.resourcesDirectory + Titanium.Filesystem.separator + 'movie4.mp4',
            backgroundColor: '#111',
        };

        if (parseFloat(Titanium.Platform.version) >= 3.2) {
            videoOptions.mediaControlStyle = Titanium.Media.VIDEO_CONTROL_NONE;
        } else {
            videoOptions.movieControlMode = Titanium.Media.VIDEO_CONTROL_NONE;
        }

        var currentVideo = Titanium.Media.createVideoPlayer(videoOptions);

        videoViewer.add(currentVideo);

        imageView.addEventListener('click', function(e) {

            mainView.add(videoViewer);
            currentVideo.play();

            videoViewer.addEventListener('click', function(e) {
                currentVideo.stop();
                mainView.remove(videoViewer);
            });
        });
        mainView.add(imageView);

        var titleLabel = Titanium.UI.createLabel({
            top: 10,
            left: 180,
            width: 400,
            height: 'auto',
            color: '#FFF',
            font: {
                fontWeight: 'Bold'
            },
            text: properties.title
        });

        var descriptionLabel = Titanium.UI.createLabel({
            top: 40,
            left: 180,
            width: 400,
            color: '#FFF',
            height: 'auto',
            text: properties.description
        });

        mainView.add(titleLabel);
        mainView.add(descriptionLabel);

        return mainView;
    };

    It looks like you're trying to use the url property, but what you actually have in there is a full filesystem path. Try changing that to a relative path (ie:'../movie.mp4') or using the media property. See these docs for more info: http://developer.appcelerator.com/apidoc/mobile/latest/Titanium.Media.VideoPlayer-object.html
