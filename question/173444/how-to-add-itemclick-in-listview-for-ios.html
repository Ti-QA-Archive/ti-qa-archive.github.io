    Thanks for raising this issue. I have tested your issue with my sample code. In my code I have set a list view using a custom template that contains a button and call a function when that button is clicked. Please test my sample code in your environments.

    index.js

      function listViewClicked() {
    alert('ListView vclicked');
}

function buttonClick() {
    alert('button clicked');
}
var data = [];
for (var i = 0; i < 10; i++) {
    data.push({

     button: {title: 'click'}

    });
}

    index.xml

      <Alloy>
    <Window class="container">
        <ListView id="lv" top="0dp" defaultItemTemplate="temp1" onItemclick="listViewClicked">
            <Templates>
                <ItemTemplate name="temp1">
                        <Button backgroundColor="red"  bindId ="button" onClick="buttonClick" >

                        </Button>
                </ItemTemplate>
            </Templates>          
            <ListSection id="ls"/>
        </ListView>       
    </Window>
