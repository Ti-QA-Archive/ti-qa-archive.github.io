  • Solution: you remove the old content and add new.

    exports.create = function() {
    var win = Ti.UI.create Window();
    var fooview = require('foo').create();
    var barview = require('bar').create();
    win.add(fooview);
    fooview.addEventlistener('click',function(){
        win.add(barview);
        win.remove(fooview);
    });
}
