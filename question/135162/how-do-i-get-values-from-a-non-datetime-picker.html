    • like this

      var langdata = [];
langdata[0] = Ti.UI.createPickerRow({
    title : 'D'
});
langdata[1] = Ti.UI.createPickerRow({
    title : 'A'
});
langdata[2] = Ti.UI.createPickerRow({
    title : 'RUS'
});

var country_picker = Ti.UI.createPicker({
    type : Ti.UI.PICKER_TYPE_PLAIN,
    font : {
        fontFamily : 'Arial',
        fontWeight : 'bold',
        fontSize : 16
    },
    top : '35%',
    left : '0%',
    bottom : '40%',
    height : 50,
    width : 90
});

country_picker.add(langdata);

var country = country_picker.value;

country_picker.addEventListener('change', function(e) {
    alert(e.row.title);
    /// <--- wanna get the selected Title here
});
