    • i used this code : 

                  if(Ti.Platform.osname == 'android'){
                var opts = { title: 'Choisir le numéro' };
                var numbers = [];

                Ti.Contacts.showContacts({
                    selectedPerson: function(e){
                        for(var key in e.person.phone){
                            for(var i=0; i<e.person.phone[key].length; i++){
                                numbers.push(e.person.phone[key][i]);
                            }
                        }
                        opts.options = numbers;
                        opts.buttonNames = ['Annuler'];
                        var dialog = Ti.UI.createOptionDialog(opts);
                        dialog.show();
                        dialog.addEventListener('click',function(e){
                            if(e.button){}
                              else
                                saisieLabel.text = opts.options[e.index];
                        });
                    }
                });
            }
            else{
                Ti.Contacts.showContacts({
                    selectedProperty: function(e){
                        saisieLabel.text = e.value;
                    }
                });
            }

  • Hi Djamel,

    I used the following code and it fetched me the number:

    Titanium.UI.setBackgroundColor('#000');

var win = Titanium.UI.createWindow({
    title : 'win',
    backgroundColor : '#fff',
    layout: 'vertical'
});

var button = Ti.UI.createButton({
    title : 'Show Contacts',
    zindex:0
});

button.addEventListener('click', function(e) {
    //button contact
    Ti.Contacts.showContacts({
        cancel: function(c){
            Ti.API.info('*** Section cancelled: '+JSON.stringify(c)+' ***');
        },
        selectedPerson : function(e) {
            Ti.API.info('*** Obtained Value: '+JSON.stringify(e)+' ***');
            label.text = e.person.phone.home[0];
        }
    });
});
win.add(button);

var label = Ti.UI.createLabel({
    text : 'Selected Contact',
    zindex:1
});
win.add(label);

win.open();

    Please note that the number was present in the Home section of contact.

    I hope this helps and addresses the need.
