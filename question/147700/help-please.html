    var win = Titanium.UI.createWindow({
    title:"Web View Controls",
    backgroundColor:"#000000",
    showScrollbars:false,
    backgroundImage:"Default.png",
    fullscreen:true,
    statusBarHidden:true,
    //Our first screen will not have a navBar
    //but if you prefer you can keep the navBar visible.
    navBarHidden:true,
    tabBarHidden:true,

    });

    var webView = Titanium.UI.createWebView({
    url:"http://mobile.mysite.com",
    showScrollbars:false,
    enableZoomControls:false,
    backgroundColor:"#000000",
    disableBounce:true,
    backgroundImage:"Default.png"

    });

    var backButton = Titanium.UI.createButton({
    title: "",
    bottom:12,
    left:12,
    width:0,
    height:0
    });

    var forwardButton = Titanium.UI.createButton({
    title: "",
    bottom:12,
    right:12,
    width:0,
    height:0
    });

    backButton.hide();
    forwardButton.hide();

    webView.addEventListener("load", function(e){
    if(webView.canGoBack()){
    backButton.show();
    }else{
    backButton.hide();
    }

    if(webView.canGoForward()){
    forwardButton.show();
}else{
    forwardButton.hide();
}

    });

    backButton.addEventListener("click", function(e){
    webView.goBack();
    })

    forwardButton.addEventListener("click", function(e){
    webView.goForward();
    })

    win.add(webView);
    win.add(backButton);
    win.add(forwardButton);

    win.open();
