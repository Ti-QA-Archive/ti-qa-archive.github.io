Get file from phone

function selectPhoto(){
    Titanium.Media.openPhotoGallery({
        success : function(event){
            var imgPath = event.media.nativePath;
            var f = Titanium.Filesystem.getFile(Titanium.Filesystem.externalStorageDirectory, imgPath);
            var preview = Ti.UI.createImageView({
                url : f,
                width: "5%"
            });
            $.preview.add(preview);
            mediaTypes : [Titanium.Media.MEDIA_TYPE_PHOTO];
        }
    });
};
