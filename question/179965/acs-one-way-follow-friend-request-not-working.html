Cloud.Friends.add({
        user_ids : userInfo.user.id,
        approval_required : false
        }, function(e) {
            if (e.success) {
                alert('Friend(s) added' );
                Ti.API.info("Freinds returned" + JSON.stringify(e));
            } else {
                alert('Error:\n' + ((e.error && e.message) || JSON.stringify(e)));
            }
        });
