  • oks so you can do it:

    app.js

    (function() {

    var win = Ti.UI.createWindow({
        backgroundColor:'gray'
    });

webView = Ti.UI.createWebView({
    url:'pag.html'
});

Ti.App.addEventListener('webPageReady', function(e){

    alert('request from html: '+e.dato);
});

win.add(webView);

win.open();
})();

    pag.html

    
<!DOCTYPE html>
<html lang="es">
  <meta http-equiv="Content-Type" content="text/html; charset=utf-8"/>

<script class="code" type="text/javascript">
function displayDate(e)
{
Ti.App.fireEvent('webPageReady',{dato:'my data'});
}
</script>


<button type="button" onclick="displayDate()">send alert to Titanium</button>  
</html>

    So now you can use the module of database on app.js
