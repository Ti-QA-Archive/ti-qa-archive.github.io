Tabbed Application Basics in Titanium

I'm trying to create a very basic Android app using Javascript in Titanium with no prior knowledge of app development. I want to achieve a basic tabbed application, with three tabs (Home, Events & News). At first I mistakenly put all the commands in the ApplicationTabGroup.js file and got a nice looking application, but obviously when I did more advanced stuff I came into difficulties and realised I needed separate js files in the ui/common file. I've tried to copy the Kitchen Sink application and have put the following commands in the relevant js files, but I can't get the application to run when I try to preview it in a web browser or on my phone. I was wondering if someone could tell me what I need to include/remove in the relevant js files to make my application work! Sorry for asking such an easy, stupid question - but any help would be much appreciated!!!

ApplicationTabGroup.js:

function ApplicationTabGroup() {
    //create module instance
    var self = Ti.UI.createTabGroup(),
    HomeWindow = require('ui/common/HomeWindow'),
    EventsWindow = require('ui/common/EventsWindow'),
    NewsWindow = require('ui/common/NewsWindow');

    //Creating App Tabs
    var homeWin = new HomeWindow(L('home_title')),
        eventsWin = new EventsWindow(L('events_title')),
        newsWin = new NewsWindow(L('news_title'));

    var homeTab = Ti.UI.createTab({
        title: L('home_title'),
        icon: '/images/Blank.jpg',
        window: homeWin
    });
    homeWin.containingTab = homeTab;

return self;
};

module.exports = ApplicationTabGroup;

HomeWindow.js:

function HomeWindow(title) {
    var self = Ti.UI.createWindow({
        title:title,
        backgroundColor: 'green'
    });
}

return self;

module.exports = HomeWindow;
