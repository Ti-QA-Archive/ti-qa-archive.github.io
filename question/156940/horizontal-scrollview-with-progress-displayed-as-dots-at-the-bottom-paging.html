  • Hi

    this works for both android and IPhone just make the view according to your desire

    var self = Ti.UI.createWindow({
        backgroundColor : 'white',
        navBarHidden : true,
        exitOnClose : true
    });
    var view1 = Ti.UI.createImageView({
        backgroundImage : '/images/tour/tour1.png',
        width : Ti.UI.FILL,
        height : Ti.UI.FILL
    });
    var view2 = Ti.UI.createImageView({
        backgroundImage : '/images/tour/tour2.png',
        width : Ti.UI.FILL,
        height : Ti.UI.FILL
    });
    var view3 = Ti.UI.createImageView({
        backgroundImage : '/images/tour/tour3.png',
        width : Ti.UI.FILL,
        height : Ti.UI.FILL
    });

    var scrollableView = Ti.UI.createScrollableView({
        top : '0dp',
        views : [view1, view2, view3],
        showPagingControl : true,
        pagingControlColor : 'transparent',
        pagingControlHeight : '0dp',
        zIndex : '1'
    });
    self.add(scrollableView);

    var btnSkipTour = Titanium.UI.createButton({
        title : 'Start Tour',
        backgroundImage : '/images/start_now.png',
        height : Ti.UI.SIZE,
        width : '80%',
        bottom : '0dp',
        zIndex : 5
    });
    view3.add(btnSkipTour);
    btnSkipTour.addEventListener('click', function() {
        new LoginScreen().open();
    });
    self.addEventListener('android:back', function(e) {
        self.close();
    });
    self.open();

    Thanks
