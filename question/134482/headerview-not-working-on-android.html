HeaderView not working on Android

On Android the following code produces the blue header bar but no text(label), yet it works on iPhone. Any ideas?

var resultHeader = Ti.UI.createView({
        width:'100%',
        top:0,
        height:'44dp',
        backgroundColor:'#008EBF'
    });
    var resultsLbl = Ti.UI.createLabel({
        color:'white',
        font:{fontSize:'16dp',fontFamily:util.getCustomFont()},
        height:'22dp',
        textAlign:'center',
        top:'4dp',
        width:'100%',
        text:'28 Restaurants Found'
    });


    resultHeader.add(resultsLbl);

    var results = Ti.UI.createTableView({
        bottom:0,
        data:data,
        height:'100%',
        headerView:resultHeader,
        backgroundColor:'#008EBF'
    });

    hey Frank,

    got the solution, it's because you set view's width to '100%',,
    try this code..

    var resultHeader = Ti.UI.createView({
    width : '320',
    top : 0,
    height : '44',
    backgroundColor : 'black'
});
var resultsLbl = Ti.UI.createLabel({
    color : 'white',
    font : {
        fontSize : '16',
    },
    height : '22dp',
    textAlign : 'center',
    top : '4dp',
    width : '100%',
    text : '28 Restaurants Found'
});

resultHeader.add(resultsLbl);

var results = Ti.UI.createTableView({
    bottom : 0,
    height : '100%',
    headerView : resultHeader,
    backgroundColor : '#008EBF'
});
win.add(results);
function createRow(title) {
    var tvr = Ti.UI.createTableViewRow({

    });
    //comment next two lines for it to work correctly.
    var vw = Ti.UI.createView();
    tvr.add(vw);
    var lbl = Ti.UI.createLabel({
        text : title
    });
    tvr.add(lbl);

    return tvr;
}

var data = [];
for(var i = 0; i <= 20; i++) {
    data.push(createRow('Row ' + i));
}
results.data = data;
