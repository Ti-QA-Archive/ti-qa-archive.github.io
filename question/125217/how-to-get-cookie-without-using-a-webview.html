  • This is working just fine for me with the following code:

    var XHR = Ti.Network.createHTTPClient();

XHR.open("POST", "http://ebay.com");

// on success, show cookie contents
XHR.onload = function() {
     var cookie = this.getResponseHeader("Set-Cookie");
     alert("cookie: "+cookie);     
};
// on error, show message
XHR.onerror = function(e) {
    alert("XHR.onerror " + e);
}
