  • hi Bojaraj,

    try this one

    var txt = Ti.UI.createTextField({
    top : 10,
    left : 10,
    width : 100,
    height : 30,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});
var l = '';

txt.addEventListener('return', function(e) {
    l = txt.value;
    var ln = l.length;    
    var c = l.substr(ln - 1);
    alert(c);
});

win.add(txt);
