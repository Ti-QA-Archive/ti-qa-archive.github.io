Switch on tableview rows. Exclusively for android

I have 5 rows in a tableview and each one has a switch at the end. At a time only one switch can be activated. The user can toggle between the rows but once one switch is 'YES' all others have to be 'NO'. I need this specifically for android.. Thanks

    var w = Ti.UI.createWindow({
    backgroundColor : 'white'
});

var rc = 10;

var d = [];
var sid = -1;

var sa = [];

function addChangeEventListener(obj) {

    obj.addEventListener('change', function(e) {

        // switch is being turned on
        if (e.source.value == true) {

            // loop thru our switches and change any other than this one OFF if its currently ON
            for (var x = 0, y = sa.length; x < y; x++) {
                if (sa[x].id !== e.source.id && sa[x].value == true) {
                    sa[x].value = false;
                }
            }

        }

    });
}

for (var x = 0; x < rc; x++) {

    var r = Ti.UI.createTableViewRow();

    var s = Ti.UI.createSwitch({
        right : 5,
        id : x
    });

    addChangeEventListener(s);

    r.add(s);
    sa.push(s);
    d.push(r);
}

var t = Ti.UI.createTableView({
    data : d
});

w.add(t);
w.open();
