    Hi

    You have TOO much logic when BOTH statements you provide are the same.

    Simplify;

    var cart_arr = [];

function doTableClick(e) {
    if (e.source.id === 'productAddToCart') {
        cart_arr.push({
            product_id: args.productId,
            quantity: quantity
        });
    }
    alert(JSON.stringify(cart_arr));
}
