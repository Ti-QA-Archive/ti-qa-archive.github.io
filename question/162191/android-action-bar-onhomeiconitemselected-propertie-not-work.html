Android Action Bar: onHomeIconItemSelected propertie not work

I'm using this code on Titanium CLI 3.1.1, and tested on Galaxy Nexus with android 4.3

var win = $.mywin;
var actionBar; 

win.addEventListener("open", function() {
    Ti.API.info('Window "mywin" open');
    if (Ti.Platform.osname === "android") {
        if (! win.activity) {
            Ti.API.error("Can't access action bar on a lightweight window.");
        } else {
            actionBar = win.activity.actionBar;
            if (actionBar) {
                actionBar.icon = "/images/icon.png";
                actionBar.title = "Degrado Ambientale";
                actionBar.navigationMode = Ti.Android.NAVIGATION_MODE_STANDARD;
                actionBar.displayHomeAsUp = true;
                actionBar.onHomeIconItemSelected = function() {
                    Ti.API.info("Home icon clicked!");
                    win.close();
                 };
            }
        }
    }
});

The window "mywin" is opened correctly and action bar is styled correctly, but if I touch on home icon the window isn't closed (and the console does not show the message "Home icon clicked").
I try to set onHomeIconItemSelected with the method setOnHomeIconItemSeleted, but the result is the same.

What I'm wrong?
