Cant get label to be at the top...

I have some simple code, and no matter what I do, I cant get my label to be on the top (trying to align it to the image).

Here is my simple code:

var win = Titanium.UI.currentWindow;


var scrollView = Titanium.UI.createScrollView({
    contentWidth:'auto',
    contentHeight:'auto',

    //top:2,
    left:2,
    right:2
});

var containerView = Ti.UI.createView({
    backgroundColor:'#fff',
    borderRadius:10,
    width:310,
    height:600,
    top:10
});


var image=Titanium.UI.createImageView({
    image:win.image,
    top:10,
    borderColor:'#000',
    left:10,
    width:100,
    height:100
});

var dealTitle=Titanium.UI.createLabel({
    top:10,
    //left:150,
    width:100,
    right:10,
    text:'win.deal_title',
    font:{fontWeight:'bold'}

});


containerView.add(image);
containerView.add(dealTitle);
scrollView.add(containerView);
win.add(scrollView);

Instead, its making the dealTitle look like its 10 below the middle of the screen, instead of 10 from the top…the image is in the top-left corner like I want it.
