    Hi E PL,

    Thanks for your concern.

    Please check my test case below for display images in 2x2 layout inside a scrollview.

    Ti.UI.setBackgroundColor('white');

var win = Ti.UI.createWindow({
  backgroundColor: 'white',
  exitOnClose: true,
});

var scrollView = Ti.UI.createScrollView({
  contentWidth: 'auto',
  contentHeight: 'auto',
  showVerticalScrollIndicator: true,
  showHorizontalScrollIndicator: true,
  height: '90%',
  width: Ti.UI.FILL,
  layout:'vertical'
});

var view = Ti.UI.createView({
  borderRadius: 10,
  top: 10,
  height: 420,
  width: Ti.UI.FILL,
  layout:'horizontal'
});
scrollView.add(view);

win.add(scrollView);

//var deviceWidth = Titanium.Platform.displayCaps.platformWidth;

var anImageView1 = Ti.UI.createImageView({
    top:10,
    image : 'image.png',
    borderColor:'green',
    borderWidth:3,
    width : "45%",
    height : 200,
    left : '3%'
});
view.add(anImageView1);

var anImageView2 = Ti.UI.createImageView({
    top:10,
    image : 'image.png',
    borderColor:'green',
    borderWidth:3,
    width : "45%",
    height : 200,
    right : '3%',
});
view.add(anImageView2);

var anImageView3 = Ti.UI.createImageView({
    top:10,
    image : 'image.png',
    borderColor:'green',
    borderWidth:3,
    width : "45%",
    height : 200,
    left : '3%'
});
view.add(anImageView3);

var anImageView4 = Ti.UI.createImageView({
    top:10,
    image : 'image.png',
    borderColor:'green',
    borderWidth:3,
    width : "45%",
    height : 200,
    right : '3%'
});
view.add(anImageView4);

win.open();

    Hope it'll help you to solve your problem. If you have any query about my answer then feel free to ask.
