MapView inside TableView.headerView disabled up/down scrolling

I have a MapView inside TableView.headerView. While the left/right scrolling of the map works as it should, up/down scrolling gets fired on the tableview but not the mapview (so essentially the table scrolls when the map really should).

code below:

        var container = Ti.UI.createView({
            top: 4,
            width: 304,
            height: 194,
            borderRadius: 5,
            borderColor: '#fff',
            borderWidth: 2,
            backgroundColor: '#fff'
        }),
        mapview = Titanium.Map.createView({
            width: 295,
            height: 185,
            borderRadius: 5,
            borderColor: '#777',
            borderWidth: 1,
            mapType: Titanium.Map.STANDARD_TYPE,
            region:{
                latitude: lat, 
                longitude: lon, 
                latitudeDelta:0.002, 
                longitudeDelta:0.002
            },
            animate: true,
            regionFit: true,
            annotations: [venueMarker],
            userLocation: false
        }),
        addrLabel = Ti.UI.createLabel({
            text: ' '+title,
            backgroundColor: '#000',
            opacity: 0.7,
            color: '#fff',
            bottom: 4,
            right: 4,
            width: 295,
            height: 22,
            textAlign: 'left',
            shadowColor: '#000',
            shadowOffset: {x:0,y:1},
            font: {
                fontSize:14,
                fontWeight:'bold'    
            }
        });

        container.add(mapview);
        container.add(addrLabel);


        var tableView = Titanium.UI.createTableView({
            data: createFeedRows()
        });        
        tableView.headerView = container;
