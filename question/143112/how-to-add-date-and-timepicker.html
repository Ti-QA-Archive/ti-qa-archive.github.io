  • Hii .. Shanuka … Use this code it will work on both Android n iPhone:
    inline code
    createDateTimePicker : function(currentWin, type, currentDate, textElement) {

            var valueSelected=currentDate.toJSON();
        var minDate = new Date();
        minDate.setFullYear(1990);
        minDate.setMonth(0);
        minDate.setDate(1);
        var maxDate = new Date();

        //var value = new Date();
        var value = currentDate;
        /*value.setMinutes(10);
         value.setHours(13);
         value.setSeconds(48);
         */
        var picker = {};

        var popupView = Titanium.UI.createView({
            backgroundImage : CommonLayout.picker.backgroundImage,
            height : CommonLayout.picker.height,
            width : CommonLayout.picker.width,
            zIndex : 90
        });

        var btnOk = Titanium.UI.createButton({

        bottom : 85,
            width : CommonLayout.Buttons.btnb.width,
            height : CommonLayout.Buttons.btnb.height,
            title : 'OK',
            color : 'white',
            backgroundImage : CommonLayout.Buttons.btnb.backgroundImage,
            backgroundSelectedImage : CommonLayout.Buttons.btnb.backgroundSelectedImage,
            font : CommonLayout.fonts.buttonFont
        });



        switch(type) {
            case 1: {
                picker = Ti.UI.createPicker({
                    type : Ti.UI.PICKER_TYPE_TIME,
                    value : value,
                    top:10,
                });
                break;
            }
            case 2:
            case 3: {
                picker = Ti.UI.createPicker({
                    type : Titanium.UI.PICKER_TYPE_DATE,
                    minDate : minDate,
                    value : maxDate,

                });
                break;
            }


        }

        popupView.add(picker);
        popupView.add(btnOk);
        //win1.add(popupWin);
        currentWin.add(popupView);

        picker.addEventListener('change', function(e) {
            //dobPicker.text = e.value;
            //alert("selected date "+e.value);

            valueSelected = e.value.toJSON();

        });

        btnOk.addEventListener('click', function(e) {
            //popupWin.hide();
            // TODO: save selected date.
            //  currentWin.datePickerValueSelected(valueSelected);
            Ti.API.info("value selected "+valueSelected);
            if(type!=3)
            {
                textElement.value = valueSelected;
            }
            else
            {
                textElement.value = valueSelected.split("T")[0];
            }
            currentWin.remove(popupView);
        });
    }
