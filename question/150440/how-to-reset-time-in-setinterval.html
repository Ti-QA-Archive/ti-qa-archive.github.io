  • I would recommend using setTimeout instead of setInterval. Then for free you don't have to worry about clearing it out.

    
var scrollView = Titanium.UI.createScrollableView({
    views : [ label1, label2,label3] 
});

var t = 0; 

function Timer() {
    if (t >= scrollView.views.length) { 
        t = 0; 
    } 
    scrollView.scrollToView(t); 
    t++; 
    setTimeout(function() {
        new Timer()    
    1000);
}

new Timer();
