    var clk = [0, 0, 0, 0];

    whenever you want to change the propery values like in this example you wanted to change the backgrounColor just change it with view's reference like this 

    view01.backgroundColor = '#fff';       // view01 is the name of your view and backgroundColor is the property of view

    , don't create the view again. It will also gives memory leaks and wouldn't behave properly. Hope it will help you.

  • win.add(require('toggleview').create());


//toggleview.js:
exports.create() {
    var clk = [0, 0, 0, 0];
    var self = Ti.UI.createView({
        backgroundColor : '#fff',
        width : '30 dp',
        height : '30 dp',
        top : '25%',
        left : '5%'
    });
    self.addEventListener('click', function() {
        if (clk[0] === 0) {
            clk[0] = 1;
            self.backgroundColor = '#0f0';
        } else {
            clk[0] = 0;
            self.backgroundColor = '#fff';
        }
    });
    return self;
}

    Your mistake: you have created every time a new view, but you have to update.
