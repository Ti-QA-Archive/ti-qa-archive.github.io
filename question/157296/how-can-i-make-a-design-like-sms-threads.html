    exports.createBalloon = function(_options) {
    var color = _options.color || 'white';
    var text   = _options.message || 'nothing to say';
    var self = Ti.UI.createView({
        backgroundImage: ###, // strip
        backgroundRepeat : true,
        height: Ti.Ui.SIZE
    });
    self.add(Ti.UI.createLabel{top:10,text:text,left:10,right:10})
    self.add(Ti.UI.createImageView{top:0,image: __TOPIMAGEDEPENDINGOFCOLOR});
    self.add(Ti.UI.createImageView{bottom:0,image: __BOTTOMIMAGEDEPENDINGOFCOLOR});
    return self;
}
