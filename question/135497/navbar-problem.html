    Titanium.UI.iPhone.NavigationGroup.

    — answered April 17th 2012 by Minh Nguyen
    permalink
    10 Comments

    • Try this sample code:

      var rootWindow = Titanium.UI.createWindow({title : "Main Window"});
var img = Ti.UI.createImageView({image: 'http://pierre.chachatelier.fr/programmation/images/mozodojo-original-image.jpg'});
rootWindow.add(img);

var nav = Titanium.UI.iPhone.createNavigationGroup({
    window : rootWindow
});

var win = Titanium.UI.createWindow();
win.add(nav);
win.open();

img.addEventListener('click', function(e){
    var detailWin = Titanium.UI.createWindow({backgroundColor : "white", title : "Video Window"});
    nav.open(detailWin, {animated : true});
});

        // backgroundImage : 'img/1.jpg'

    tab1.open(win);

tabGroup.addTab(tab1);

tabGroup.open();
