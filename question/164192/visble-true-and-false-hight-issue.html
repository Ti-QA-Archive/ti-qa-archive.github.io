    • Ti.UI.SIZE is a constant not a String, height takes numbers, string with units like '0dp' but not '0'

      $.CashPayment.visible = true;
$.CashPayment.height = Ti.UI.SIZE;    
$.BankTransfers.visible = false;
$.BankTransfers.height = 0;           
$.ZymCardPayments.visible = false;
$.ZymCardPayments.height = 0;
