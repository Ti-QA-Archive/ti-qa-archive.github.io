How to insert a view into a horizontal or vertical layout?

view.add(newView) puts the new view at the end of the children.

What I need to do is to be able to insert a view into the children. I can't use a tableView because I need to be able to toggle scrolling on android. (Turning off touch events is not an option).

So, instead, I am forced to use a ScrollView while setting scrollingEnabled true/false and having the ScrollView with a vertical layout. But if I have 5 items in my table view, I'm looking to be able to insert a new item between, say, items 1 and 2.

  • Hello Brian,

    You can follow this sample code. Here 20 views have been added vertically into the scrollView.

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

if (Ti.UI.Android) {
    win.windowSoftInputMode = Ti.UI.Android.SOFT_INPUT_ADJUST_PAN;
}

function createRow(i) {
    var row = Ti.UI.createView({
        backgroundColor : 'white',
        borderColor : '#bbb',
        borderWidth : 1,
        width : '100%',
        height : 70,
        top : 0,
        left : 0
    });
    var inputTextField = Ti.UI.createTextField({
        hintText : 'Enter value ' + i,
        keyboardType : Ti.UI.KEYBOARD_NUMBERS_PUNCTUATION,
        top : 10,
        left : '10%',
        width : '80%',
        height : 60
    });
    row.add(inputTextField);
    return row;
}

var scrollView = Ti.UI.createScrollView({
    bottom : 20,
    contentHeight : 'auto',
    layout : 'vertical'
});

for (var i = 0; i <= 20; i++) {
    var row = createRow(i);
    scrollView.add(row);
}
win.add(scrollView);
win.open();

    If you have any concern, then feel free to ask me.

    Thanks.
