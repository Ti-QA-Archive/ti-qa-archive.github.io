TabGroup on Android Causing black screen

Hello,

It seems that when I attempt to open my tabGroup on the Android, I get nothing but a black screen. This is not one of the features I was looking for. I know that it has nothing to do with the window that it is attempting to open because I can write home.open(); and that works fine. If anyone has any insights or ideas please, I need your help!

// this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');
// set the background color of all windows
Ti.UI.backgroundColor = "#e3e3e3";

/*
*  create the windows and create the tabs
* add the windows to the tabs
* create the tabgroup and then add the tabs to it
* open the tab group
* 
* Tabs are : Home, About, Twitter, Feedback
* */

var home = Ti.UI.createWindow({
    title : 'Home',
    url : 'home.js',
    navBarHidden : true
});

var about = Ti.UI.createWindow({
    title : 'About',
    url : 'about.js',
    barColor : '#E69B26'
});
var twitter = Ti.UI.createWindow({
    title : 'Twitter',
    url : 'twitter.js',
    barColor : '#E69B26'
});
var feedback = Ti.UI.createWindow({
    title : 'Feedback',
    url : 'feedback.js',
    barColor : '#E69B26'
});

var home_tab = Ti.UI.createTab({
    title : 'Home',
icon : 'images/home.png',
    window : home
});

var about_tab = Ti.UI.createTab({
    title : 'About',
    icon : 'images/info.png',
    window : about
});

var twitter_tab = Ti.UI.createTab({
    title : 'Twitter',
    window : twitter,
    icon : 'images/twitter.png'
});

var feedback_tab = Ti.UI.createTab({
    title : 'feedback',
    icon : 'images/icon_feedback.png',
    window : feedback
});

var tg = Ti.UI.createTabGroup({
    tabs : [home_tab, about_tab, twitter_tab, feedback_tab]
});
home.tg = tg;
setTimeout(function(){
    tg.open();
}, 1000);

    Hi,

    Try using addTab() instead of setting the tabs property in the createTab constructor.

    var tg = Ti.UI.createTabGroup({
});

tg.addTab(home_tab);
tg.addTab(about_tab);
tg.addTab(twitter_tab);
tg.addTab(feedback_tab);

    Hope that helps.

    • Awesome! You just saved me! 

      var tg = Ti.UI.createTabGroup({});
var tabs = [home_tab, about_tab, twitter_tab, feedback_tab];
for(var i in tabs){
    tg.addTab(tabs[i]);
}
tg.open();

      Worked for me!
