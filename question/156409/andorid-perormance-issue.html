  • Hi Jens Fischer,

    You should change some code in your code to run it smoothly in android. You can follow this sample code. You just copy and paste this code in app.js file and run it.

    var win = Ti.UI.createWindow({
    layout : 'vertical',
    height : Titanium.UI.FILL,
    width : Titanium.UI.FILL,
    backgroundColor : '#fff'
});
function createRow(i) {
    var mainView = Ti.UI.createView({
     width:'100%', height: 150,
    layout: 'vertical',
});

var view1 = Ti.UI.createView({
    height : 40,
    width : Titanium.UI.FILL,
    backgroundColor : '#123',
    top : 10,

});

mainView.add(view1);

var view2 = Ti.UI.createView({
    height : 40,
    width : Titanium.UI.FILL,
    backgroundColor : '#246',
    top : 10
});

mainView.add(view2);
var view3 = Ti.UI.createView({
    height : 40,
    width : Titanium.UI.FILL,
    backgroundColor : '#48b',
    top : 10
});
mainView.add(view3);

return mainView;
}

var scrollView = Ti.UI.createScrollView({
  bottom:0,
  contentHeight: 'auto',
  layout: 'vertical'
});

for(var i = 0; i <= 20; i++){
var row = createRow(i);
  scrollView.add(row);
}
win.add(scrollView);

win.open();

    You should more study Titanium API documentation to become a good apps developer. And you can follow this coding style. And when you run any portion of kitchenSink you can copy and paste the code in app.js file and remove the module.exports portion and every fields add to the window and after all open the window as like that code. I think you understand.
