    Hi

    This will fix your tabbedBar problem, I had to fix a few sections of your code as there were many lines missing from your example.

    tab1.js

    var win = Ti.UI.currentWindow;

var tabbedBar = Ti.UI.iOS.createTabbedBar({
    height: 38,
    index: 0,
    labels: ['tableView1', 'tableView2'],
    style: Ti.UI.iPhone.SystemButtonStyle.BAR,
    top: 0,
    width: Ti.UI.FILL
});
win.add(tabbedBar);

//define our table view
var data1 = [
    { title: 'One' },
    { title: 'Two' },
    { title: 'Three' }
];
var parentView = Ti.UI.createView({
    height: Ti.UI.FILL,
    top: 38,
    width: Ti.UI.FILL
});
win.add(parentView);
var tableView1 = Titanium.UI.createTableView({
    data: data1,
    height: Ti.UI.FILL,
    left: 0,
    rowHeight: 70,
    top: Ti.UI.FILL,
    width: 320
});
parentView.add(tableView1);
var data2 = [
    { title: 'Red' },
    { title: 'Green' },
    { title: 'Blue' }
];
var tableView2 = Titanium.UI.createTableView({
    data: data2,
    height: Ti.UI.FILL,
    left: 0,
    rowHeight: 70,
    top: Ti.UI.FILL,
    width: 320
});
//parentView.add(tableView2);

/*
 var artists = Ti.UI.createTableView({

});
*/
tabbedBar.addEventListener('click', function(e){
    switch(e.index){
        case 0: {
            parentView.animate({
                view: tableView1,
                transition: Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
            });
            break;
        }
        case 1: {
            parentView.animate({
                view: tableView2,
                transition: Ti.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT
            });
            //win.add(aritsts);
            //win.remove(tableView);
            break;
        }
    }
});

    I am not sure what your other code problem is as you have not provided any code to explain that but I suspect this will get you most of the way.

    The key to the swap is to place the first tableView inside a parent view, you do not add the second table, the flip handles that for you. Then you flip the parent view and not the window.
