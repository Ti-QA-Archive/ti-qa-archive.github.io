Issues with TableViewSections.headerview

I have created a series of tableViewSections and put headerview in them made of a view and a label. I am using 4 sections. I have noticed that the first two render just fine. But the third and fourth headerview match the 2nd one.
I declare them as such:

var viewKS = Ti.UI.createView({height: 70, backgroundColor:'black'});
var lblKS = Ti.UI.createLabel({text:'Kindle', color:'white', fontSize: 100});
viewKS.add(lblKS);
var KindleSection = Ti.UI.createTableViewSection({
    headerView: viewKS
});

var viewNYT = Ti.UI.createView({height: 50, backgroundColor: 'black'});
var lblNYT = Ti.UI.createLabel({text:'NYC', color:'white', fontSize:100});
viewNYT.add(lblNYT);
var NYTSection = Ti.UI.createTableViewSection({
    headerView: viewNYT
});
var viewAMZ = Ti.UI.createView({height: 50, backgroundColor: 'black'});
var lblAMZ = Ti.UI.createLabel({text:'Amazon', color:'white', fontSize:100});
viewAMZ.add(lblAMZ);
var AmazonSection = Ti.UI.createTableViewSection({
    headerView: viewAMZ
});

var viewAud = Ti.UI.createView({height: 50, backgroundColor: 'black'});
var lblAud = Ti.UI.createLabel({text:'Aud', color:'white', fontSize:100});
viewAud.add(lblAud);
var AudibleSection = Ti.UI.createTableViewSection({
    headerView: viewAud
});

var tv = Ti.UI.createTableView();
var data = [];
data.push(NYTSection);
data.push(AmazonSection);
data.push(KindleSection);
data.push(AudibleSection);
tv.setData(data);

the only difference between the four are the variable names and the text of the label.
I end up with 1 unique header and 3 that are the same. I have tried declaring them in a different order. It is always the 2nd one that is repeated twice.

I also have noticed that the fontsize does not seem to matter. I would like to make the text bigger.
