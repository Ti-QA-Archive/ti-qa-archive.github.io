    Hi

    I am assuming you want to centre as in; left, centre & right rather than; top, centre & bottom.

    If so create an outer view that you add myview to instead of the scrollView and add the outer to the scrollView.

    var outer = Ti.UI.createView({
  height: Ti.UI.SIZE,
  width: Ti.UI.SIZE
});
scrollView.add(outer);
...
outer.add(myView);
// scrollView.add(myView);

    This should allow the children in the myView to sit in a parent view that will sit in the centre because outer has no left, right, top or bottom.

    If you tweak outer to have a fixed width you can control how wide the child views are.

    I ended up with this solution:

    
var winThumbs = Ti.UI.currentWindow;

var scrollView = Titanium.UI.createScrollView({

    backgroundColor: '#123'
});

var myView = Titanium.UI.createView({
    layout: 'horizontal',
    width: Ti.UI.SIZE
    });

var outer = Ti.UI.createView({
    backgroundColor: '#EEE',
     height: Ti.UI.SIZE,
  //width: Ti.UI.SIZE
});

// add thumbs

// Images array
var images = [
    {filename: '../../photos/thumbnails/zwei100x100.jpg', caption: 'Kitten placeholder'},
    {filename: '../../photos/thumbnails/eins200x200.jpg', caption: 'Kitten placeholder'},
    {filename: '../../photos/thumbnails/zwei100x100.jpg', caption: 'Kitten placeholder'},


];

for (var i=0;i<images.length;i++) { 
    //create the image view
  var currentImg = images[i].filename;
  var thumb = Titanium.UI.createImageView({ image: images[i].filename ,width: 100,borderRadius:10,  borderWidth: 5 }); 
  myView.add(thumb);
}


scrollView.add(outer);
outer.add(myView);
winThumbs.add(scrollView);
