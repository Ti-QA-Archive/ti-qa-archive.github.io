selectedBackgroundGradient on tableview row

Does anyone know how to get the selectedBackgroundGradient on tableview row to work properly. So when each row is clicked it changes to a different gradient. Heres my sample code below. It works if you use backgroundColor but not backgroundGradient.

// create table view data object
var data = [
    {title:'Row 1', hasChild:true, color:'black', selectedColor:'#fff',selectedBackgroundGradient:{type:'linear',colors:[{color:'#222222',position:0.0},{color:'#000000',position:1.0}]},backgroundGradient:{type:'linear',colors:[{color:'#f1f1f1',position:0.0},{color:'#d3d3d3',position:1.0}]}},
    {title:'Row 2', hasChild:true, color:'black', selectedColor:'#fff',selectedBackgroundGradient:{type:'linear',colors:[{color:'#222222',position:0.0},{color:'#000000',position:1.0}]},backgroundGradient:{type:'linear',colors:[{color:'#f1f1f1',position:0.0},{color:'#d3d3d3',position:1.0}]}},
    {title:'Row 3', hasChild:true, color:'black', selectedColor:'#fff',selectedBackgroundGradient:{type:'linear',colors:[{color:'#222222',position:0.0},{color:'#000000',position:1.0}]},backgroundGradient:{type:'linear',colors:[{color:'#f1f1f1',position:0.0},{color:'#d3d3d3',position:1.0}]}},
];

// create table view
var tableview = Titanium.UI.createTableView({
    data:data,
    allowsSelection:true
});

// add table view to the window
Titanium.UI.currentWindow.add(tableview);

  • It's ok solved it. If anyone has the same problem here's my sample code:

    var lastRowClicked;
var slectedBg; 
var bgGradient = ({type:'linear',colors:[{color:'#f1f1f1',position:0.0},{color:'#d3d3d3',position:1.0}]});
var bgGradientSelected = ({type:'linear',colors:[{color:'#222222',position:0.0},{color:'#000000',position:1.0}]});

tableView.addEventListener('click', function(e) {

if (lastRowClicked && lastRowClicked !== e.row) {        
    slectedBg = false;         
    lastRowClicked.backgroundGradient = bgGradient;            
    }
    slectedBg = true;
    if (slectedBg == true) {
        lastRowClicked = e.row;
        lastRowClicked.backgroundGradient = bgGradientSelected; 
    }
    else {
        lastRowClicked = undefined;
}

});
