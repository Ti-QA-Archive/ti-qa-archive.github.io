  • Hello,

    Try this code:

    var textarea = Ti.UI.createTextArea({
    top:10,
    left:10,
    right:10,
    bottom:10,
    editable:false,
    value:'some data whith link http://www.google.com',
    font:{fontSize:15},
    autoLink:Ti.UI.AUTODETECT_ALL
});

//Ti.UI.AUTODETECT_ALL
//Ti.UI.AUTODETECT_LINK
//Ti.UI.AUTODETECT_PHONE
//Ti.UI.AUTODETECT_PHONE
//Ti.UI.AUTODETECT_ADDRESS

      var lblLink = Titanium.UI.createLabel({
    text:'some data whith link http://www.google.com',
     font: {
            fontSize: 16,
            fontStyle: 'bold',
        },
     top:1,
     color:    '#FFFFFF',
     autoLink:Titanium.UI.Android.LINKIFY_ALL,
     left:2
});

  • I resolve with:

    var text = Ti.UI.createView({
    width: 'auto',
    height: 'auto',
    layout: 'horizontal'
});
var link = Ti.UI.createLabel({
    text: 'click here',
    left: 0,
    color: 'blue',
    width: 'auto'
});
link.addEventListener('click', function() {
    // open url
});

text.add(link);
text.add(Ti.UI.createLabel({
    text: 'to open window',
    left: 2,
    width: 'auto'
}));

    Result: [view][link]click here[/link] to open window[/view]
