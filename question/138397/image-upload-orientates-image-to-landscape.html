      function resizeImage($image, $new_name, $path)
    {

        $error = false;
        $sizes = getimagesize($image);


            // Set a maximum height and width
            $width = 300;
            $height =300;
            $width_orig = $sizes[0];
            $height_orig = $sizes[1];

            if ($width && ($width_orig < $height_orig)) {
               $width = ($height / $height_orig) * $width_orig;
            } else {
               $height = ($width / $width_orig) * $height_orig;
            }    

            // Resample
            $image_p = imagecreatetruecolor($width, $height);

            //echo "<br>".$sizes['mime'];
            if ($sizes['mime'] == "image/png")  {
                $ximage = imagecreatefrompng($image);
                //echo "png";
            } 

            if (($sizes['mime'] == "image/jpeg") || ($sizes['mime'] == "image/pjpeg")) {
                $ximage = imagecreatefromjpeg($image);
                //echo "jpg";
            } 



        if (!$error) {
            imagecopyresampled($image_p, $ximage, 0, 0, 0, 0, $width, $height, $width_orig, $height_orig);


            imagejpeg($image_p,$_SERVER['DOCUMENT_ROOT'].$path.'/'.$new_name.'.png', 100);

            // Set a maximum height and width
                $width = 75;
                $height = 75;
                $width_orig = $sizes[0];
                $height_orig = $sizes[1];

                if ($width && ($width_orig < $height_orig)) {
                   $width = ($height / $height_orig) * $width_orig;
                } else {
                   $height = ($width / $width_orig) * $height_orig;
                }    

                // Resample
                $small = imagecreatetruecolor($width, $height);
                imagecopyresampled($small, $ximage, 0, 0, 0, 0, $width, $height, $width_orig, $height_orig);
                imagejpeg($small,$_SERVER['DOCUMENT_ROOT'].$path.'/small_'.$new_name.'.png', 100);
            }    
        return true;

    }
