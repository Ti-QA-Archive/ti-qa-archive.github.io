Setting views width/height with percentage is not working.

Based on this tutorial I should be able to set width/height of my views with string &#39;value%&#39;

var instance = Ti.UI.createView({
        top : 0,
        height : e.height,
        width : '100%',
        backgroundColor : 'blue',
        layout : 'vertical'
    });

    this.container = Ti.UI.createView({
        top : 0,
        width : '100%',
        layout : 'horizontal',
        //height : e.height * 0.8,
        height : '80%',
        backgroundColor : 'yellow'
    });    
    instance.add(this.container);

    this.leftContainer = Ti.UI.createView({
        top : 0,
        left : 0,
        width : '50%',
        layout : 'vertical',
        //height : e.height * 0.8,
        backgroundColor : 'green'
    });
    this.container.add(this.leftContainer);

    this.rightContainer = Ti.UI.createView({
        top : 0,
        left : 0,
        width : '50%',
        layout : 'vertical',
        //height : e.height * 0.8,
        backgroundColor : 'red'
    });
    this.container.add(this.rightContainer);

But this is not working, only my 'main' view is shown (blue). If I change width values to some number it appears but I would like to do it with percentages.

I am trying this on iOS, 1.8.2
