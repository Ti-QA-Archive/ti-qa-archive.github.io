    Hi Bunty,

    There is no any sharing intent action like android Intent.ACTION_SEND. so for ios you can sharing your content using creating option dialog.

    e.g

    var arrOptions = ["WhatsApp", "Facebook", "Twitter", "Email"];
if (OS_IOS) {
    arrOptions.push("Cancel");
}
var shareOptionDialog = Titanium.UI.createOptionDialog({
    options : arrOptions,
    title : "Share",
    cancel : 4,
});

//YOU CAN SHOW DIALOG USING

shareOptionDialog.show();


shareOptionDialog.addEventListener('click', function(e) {
    var ind = e.index;
    if (ind == 1) {
        //SHARE ON FB
    } else if (ind == 2) {
        //SHARE ON TWITTER
    } else if (ind == 3) {
        //SHARE VIA MAIL
    } else if (ind == 0) {
        //SHARE ON WHATSAPP
    }

});

    if you have any doubt then let me know.. :)

      Alloy.Globals.shareOnFacebook = function(alertTitle, obj,callBack) {

    Alloy.Globals.facebook.appid = 'xxxxxxxxxxxxxx'; // HERE YOUR APP ID
    Alloy.Globals.facebook.permissions = ['publish_stream', 'read_stream', 'publish_actions', 'manage_notifications'];

    Alloy.Globals.facebook.forceDialogAuth = true;

    if (Alloy.Globals.facebook.loggedIn) {
            Alloy.Globals.facebook.permissions = ['publish_actions'];
            Alloy.Globals.facebook.dialog('feed', {
                link : "",//WHATEVER YOU WANT
                description : obj.messageBody,
                picture : "",
                name : obj.caption
            }, function(e) {
                var isSuccess = false;
                if (e.success && e.result) {
                    Alloy.Globals.ShowAlert(alertTitle, Alloy.Globals.selectedLanguage.shareFBSuccess);
                    isSuccess = true;
                } else if((e.success && !e.result) || e.cancelled){
                    Alloy.Globals.ShowAlert(alertTitle, Alloy.Globals.selectedLanguage.facebookCancel);
                    isSuccess = false; 
                }   else {
                    Ti.API.info("face book post error :" + e.result +'\n error details :' + e.error);
                    Alloy.Globals.ShowAlert(alertTitle, Alloy.Globals.selectedLanguage.shareFBFail);
                    isSuccess = false;
                }
                Alloy.Globals.hideLoading();
                callBack(isSuccess);
            });

            // Alloy.Globals.facebook.authorize();

    } else {

        Alloy.Globals.facebook.authorize();
        Alloy.Globals.facebook.addEventListener('login', function(e) {
            Ti.API.info('Logged in successfully');
            Ti.API.info('FACEBOOK LOGIN = ' + JSON.stringify(e));
            Alloy.Globals.facebook.permissions = ['publish_actions'];

      Alloy.Globals.shareOnTwitter = function(alertTitle, messageBody,callBack) {
    var len = (messageBody.length > 140) ? 140 : messageBody.length;
    msgBody = messageBody.substr(0, len);

    Alloy.Globals.showLoading(Alloy.Globals.selectedLanguage.loading, false);

    var RUN_TESTS = false;

    if (RUN_TESTS) {
        Ti.include('twitter/tests.js');
    } else {
        var accessTokenKey = Ti.App.Properties.getString('twitterAccessTokenKey'), accessTokenSecret = Ti.App.Properties.getString('twitterAccessTokenSecret');

        var Twitter = Alloy.createController('twitter').Twitter;

        var client = Twitter({
            consumerKey : "",//YOUR APP CONSUMER KEY
            consumerSecret : "",//YOUR SECRET KEY
            accessTokenKey : accessTokenKey,
            accessTokenSecret : accessTokenSecret
        });

        client.addEventListener('login', function(e) {
            if (e.success) {
                Ti.App.Properties.setString('twitterAccessTokenKey', e.accessTokenKey);
                Ti.App.Properties.setString('twitterAccessTokenSecret', e.accessTokenSecret);

                client.request("1.1/statuses/update.json", {
                    status : msgBody
                }, "POST", function(e) {
                    var isSuccess = false;
                    if (e.success) {    
                        Alloy.Globals.ShowAlert(alertTitle, Alloy.Globals.selectedLanguage.shareTwitterSuccess);
                        isSuccess = true;
                    } else if (e.result.text.indexOf("duplicate") > 0)  {
                        Alloy.Globals.ShowAlert(alertTitle, Alloy.Globals.selectedLanguage.duplicateMessage);
                        isSuccess = false;
                    } else {
                        Alloy.Globals.ShowAlert(alertTitle, Alloy.Globals.selectedLanguage.shareTwitterFail);
                        isSuccess = false;
                    }
                    Alloy.Globals.hideLoading();
                    callBack(isSuccess);
                });

            } else {
                Alloy.Globals.hideLoading();
                callBack(false);
            }
        });

        client.authorize();

    }

};

      Alloy.Globals.sendMail = function(message,callBack) {
    var emailDialog = Ti.UI.createEmailDialog({
    });
    if (emailDialog.isSupported()) {
        //emailDialog.subject = "info";
        emailDialog.toRecipients = "";
        emailDialog.messageBody = message;
        emailDialog.html = true;
        callBack(true);
        emailDialog.open();

    } else {
        callBack(false);
        Alloy.Globals.ShowAlert(Alloy.Globals.selectedLanguage.sendMail, Alloy.Globals.selectedLanguage.mailAccountMsg);
    }
};
