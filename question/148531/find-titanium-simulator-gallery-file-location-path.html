Find Titanium simulator gallery file location path.

I am trying cook book code on Titanium studio, and simulator showing "No Photos"

Is there any path to set photos and let simulator can get photos.

Platform: OS X 10.8.2

//obtain an image from the gallery
Titanium.Media.openPhotoGallery({

    success:function(event)
    {
        var image = event.media;

        // set image view
        Ti.API.debug('Our type was: '+event.mediaType);
        if(event.mediaType == Ti.Media.MEDIA_TYPE_PHOTO)
        {
           // set image view
            var imgView = Titanium.UI.createImageView({
                   top: 0,
                   left: 0,
                   width: 286,
                   height: 337,
                   image: image
            });

            Ti.API.debug('adding imageview to scrollableview');
            scrollingView.addView(imgView);
        }        
    },
    cancel:function()
    {
        //user cancelled the action fron within
        //the photo gallery
    }
});
