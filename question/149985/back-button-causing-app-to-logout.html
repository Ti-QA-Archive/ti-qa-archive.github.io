  • Hi Sean,

    Try like this : 

    $.index.addEventListener('android:back', function() {
    var od = Ti.UI.createOptionDialog({
        title:"Logout",
        options:[L('ok'), L('cancel')],
        cancel:1
    });    
    od.addEventListener('click', function(e) {
        if(e.index==0){
            //Logout
        } else {
            //Nothing or as you want
        }
    });        
    od.show();
});
