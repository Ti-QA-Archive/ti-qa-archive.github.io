});

menuButton.addEventListener('click', function(e) {
    menuButton.backgroundImage = '/images/btnHomeOver.png';
    if(Ti.App.Properties.getString('inMenu') == 'false'){
        Ti.App.Properties.setString('inMenu', true);
        var t = setTimeout(function(e){
               menuButton.backgroundImage = '/images/btnHome.png';
        },300);
        var slide_it_left = Titanium.UI.createAnimation();
        slide_it_left.left = Ti.Platform.displayCaps.platformWidth - 60;
        slide_it_left.duration = 300;
        legalView.animate(slide_it_left);    
    }
    else{
        Ti.App.Properties.setString('inMenu', false);
        var t = setTimeout(function(e){
               menuButton.backgroundImage = '/images/btnHome.png';
        },300);
        var slide_it_left = Titanium.UI.createAnimation();
        slide_it_left.left = 0;
        slide_it_left.duration = 300;
        legalView.animate(slide_it_left);

});
