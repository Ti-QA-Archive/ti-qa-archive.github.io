content of the tableviewrow

I'm doing a program that organizes the tuples of a tableviewrow through up and down buttons, and I want to know the order of the tuples to make a validation if the select order was correct.

Any idea how do ?

That is my code

var activity = Titanium.Android.currentActivity; 
var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    exitOnClose:true
});


var data = [];

for(var i = 1; i <= 5; i++) {
    var row = Ti.UI.createTableViewRow({

    });
    row.add(Ti.UI.createLabel({
        text: 'Sentence ' + i, textAlign: 'left',
        width: Ti.UI.FILL, color: '#000'
    }));
    row.add(Ti.UI.createButton({
        title: 'Up', action: 'moveUp',
        width: 80, right: 70, borderRadius:30
    }));
    row.add(Ti.UI.createButton({
        title: 'Down', action: 'moveDown',
        width: 80, right: 10,borderRadius:30
    }));
    data.push(row);
}

var table = Ti.UI.createTableView({
    data: data
});
var btnExit = Titanium.UI.createButton({
title:'Exit',width: 100,
height:30,top: 300,
left:0,backgroundColor:'black',
color:'white',borderRadius:10
});

btnExit.addEventListener('click',function(){
activity.finish();    
});

table.addEventListener('click', function(evt) {
    var action = evt.source.action,
        index = evt.index,
        isFirstRow = index === 0,
        isLastRow = index + 1 === data.length;
    if(action === 'moveUp' && !isFirstRow) {
        swapRows(index, index - 1);
    }
    else if(action === 'moveDown' && !isLastRow) {
        swapRows(index, index + 1);
    }
});

win.add(table);
win.add(btnExit);
win.open();


function swapRows(indexOne, indexTwo) {
    var temp = data[indexOne];
    data[indexOne] = data[indexTwo];
    data[indexTwo] = temp;
    table.data = data;
}

    Hi,
    You can set the id, title or any other property of tableviewrow and can iterate over it to check the order at a particular time. In this way you can add your validation based on order :
    You can iterate as mentioned below :

    btnExit.addEventListener('click',function(){
//activity.finish();
  Ti.API.info('CURRENT ORDER :');
  for(i=0;i<5;i++){
      Ti.API.info(JSON.stringify(table.data[0].rows[i].title));
  }
});
