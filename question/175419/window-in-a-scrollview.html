  • Hi Yazmin Ortiz,

    Thanks for your concern.

    If you are using the example, then just try the below code.

    menuTable.addEventListener('click', function(e){

    if(e.row.title=="menu A")
    {
        Ti.UI.createWindow({
            url:'win2.js',
            backgroundColor:'white'
        }).open();
    }
});

    If you click menu A then it will open a new window. You can customise in your own way. I've also used example code for testing.

    • Hi Ortiz,

      Thanks for your concern.

      You can't open/add a window inside a scroll view. Please see the link for details.

      You can add drawers to every new window, but it's not the best way to do what you want. That's how you can gain your desired output.

      Please check my test code below:

      app.js

      var TiDrawerLayout = require('com.tripvi.drawerlayout');

var win = Ti.UI.createWindow({
    backgroundColor: 'white',
    navBarHidden: true,
});

var menuTable = Ti.UI.createTableView({
    data: [
        { title: "menu A" },
        { title: "menu B" },
        { title: "menu C" },
        { title: "menu D" },
        { title: "menu E" },
    ],
    backgroundColor: "#ddd",
});

var contentView = Ti.UI.createScrollView({
    backgroundColor: "#fff",
    layout: "vertical",
});

var drawer = TiDrawerLayout.createDrawer({
    leftView: menuTable,
    centerView: contentView,
    leftDrawerWidth: "240dp",
    top: "50dp",
    width: Ti.UI.FILL,
    height: Ti.UI.FILL,
});

var actionBar = Ti.UI.createView({
    top: 0,
    height: "50dp",
    backgroundColor: "#333",
});

var menuButton = Ti.UI.createButton({
    left: "15dp",
    title: "Drawer",
});
actionBar.add(menuButton);

menuButton.addEventListener("click", function(e) {
    drawer.toggleLeftWindow();
});

menuTable.addEventListener('click', function(e){

    if(e.row.title=="menu A")
    {
        var newWin=Ti.UI.createWindow({
            url:'/newWin.js',
            backgroundColor:'green',
        });
        newWin.open();
        //win.close();
    }
});

win.add(actionBar);
win.add(drawer);
win.open();

      And in newWin.js

      var TiDrawer = require('com.tripvi.drawerlayout');
var newWin = Ti.UI.currentWindow;

var menuTable = Ti.UI.createTableView({
    data: [
        { title: "menu A" },
        { title: "menu B" },
        { title: "menu C" },
        { title: "menu D" },
        { title: "menu E" },
    ],
    backgroundColor: "#ddd",
});

var contentView = Ti.UI.createScrollView({
    backgroundColor: "#fff",
    layout: "vertical",
});

var drawer = TiDrawer.createDrawer({
    leftView: menuTable,
    centerView: contentView,
    leftDrawerWidth: "240dp",
    top: "50dp",
    width: Ti.UI.FILL,
    height: Ti.UI.FILL,
});

var actionBar = Ti.UI.createView({
    top: 0,
    height: "50dp",
    backgroundColor: "#333",
});

var menuButton = Ti.UI.createButton({
    left: "15dp",
    title: "Drawer",
});
actionBar.add(menuButton);

menuButton.addEventListener("click", function(e) {
    drawer.toggleLeftWindow();
});

var button = Ti.UI.createButton({
    title:'Appcelerator Titanium Rockz !!!',
    top:100,
    width:'auto',
    height:'auto'
});
contentView.add(button);

newWin.add(actionBar);
newWin.add(drawer);
newWin.open();

      Hope it will help you to solve your problem.

      Thanks
