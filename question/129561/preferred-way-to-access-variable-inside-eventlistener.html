Preferred way to access variable inside eventlistener

What's the preferred way to access variable inside an eventlistener? I found there are many ways to achieve this and curious which method could be used to avoid memory leak

var win = Ti.UI.createWindow({
    objImg: null
});
var img = Ti.UI.createImageView();
win.objImg = img;

// Situation A - access event emitter

// method 1
win.addEventListener('open', function() {
    win.add(img);
});

// method 2
win.addEventListener('open', function() {
    this.add(img);
});

// method 3
win.addEventListener('open', function(e) {
    e.source.add(img);
});

// Situation B - access external variable inside event listener

// method 1
win.addEventListener('open', function() {
    win.add(img);
});

// method 2
win.addEventListener('open', function() {
    win.add(this.objImg);
});
