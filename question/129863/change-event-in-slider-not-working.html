    There are two bugs walking through your code. This should work:

    var win = Ti.UI.createWindow({backgroundColor:'#fff'});

var slider_label=Titanium.UI.createLabel({  //must be 'slider_label' and not 'silder_label'
    text:'',
    top:300,
    width:100,
    height:200,
    font:{fontSize:17},
    color:'orange'
});


var basicSlider = Titanium.UI.createSlider({
    min:0,
    max:100,
    value:5,
    width:300, 
    height:'auto',
    top:300, //yep, out of my screen
});

win.add(basicSlider);
win.add(slider_label); //slider_label and not 'label'

basicSlider.addEventListener('change',function(e) {
   slider_label.text = e.value + '' ;
});

win.open();
