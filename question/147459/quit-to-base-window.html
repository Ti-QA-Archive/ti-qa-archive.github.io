    • Sorry for incomplete answer when you want to close all window except base window fire a event and pass your array and close all window.

      Ti.App.addEventListener("closwAllWindow", function() {
    if (Ti.App.openWinArray.length > 1) {
        var arr = Ti.App.openWinArray;
        for (var i = arr.length; i > startPoint; i--) {
            var k = arr[i - 1];
            Ti.App.openWinArray.pop(k);
            k.close();
        }
    }
});

      Ti.App.openWinArray will be your global Array.
