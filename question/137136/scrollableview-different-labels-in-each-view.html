ScrollableView different labels in each view

Hi all,

How to change the title bar or descriptionLabel by the name of the current view?

I try with this, but doesn't work.


var view0 = Ti.UI.createView({ backgroundImage:'images/0.jpg' , title:'Example' });
var view1 = Ti.UI.createImageView({ image:'images/1.jpg' ,title:'Example Car'});
var view2 = Ti.UI.createImageView({ image:'images/2.jpg', title:'Example Beach'});

var scrollableView = Ti.UI.createScrollableView({ 
top:0,
left:0,
right:0,
bottom:40,
  views:[view0,view1,view2],
  showPagingControl:false,
  currentPage:0 , 
 });

win.add(scrollableView);
win.open();

var cTitle = 'proti';
scrollableView.addEventListener('scroll',function(e){
    cTitle = e.currentTitle;
});

var descriptionLabel = Ti.UI.createLabel({
    text : ''+cTitle+'',
    width : 320,
    bottom : 45,
    height : 'auto',
    backgroundColor : '#000',
    font : {fontSize : 12, fontWeight : 'bold'    },
    color : '#FFF'
});
win.add(descriptionLabel);

    hi,

    you can do that with

    scrollableView.addEventListener('scroll',function(e){
    var view = e.currentView;
    cTitle = view.title;
});
