  • Accepted Answer

    Here is a quick snippet.

    Titanium.Media.showCamera({
                success:function(e) {
                    if(e.mediaType == Ti.Media.MEDIA_TYPE_PHOTO) {

                        var imageAsTaken = Ti.UI.createImageView({
                                         image: e.media
                                    });
                        Thumbnail = imageAsTaken.toImage();
                      Thumbnail = Thumbnail.imageAsThumbnail(100); /*Resizing to 100*100 */

                    /*--------------------------------
                        YOUR ACS CODE HERE.  "Thumbnail" is now your image object.
                    --------------------------------*/    

                    } else {
                        alert("got the wrong type back ="+event.mediaType);
                    }
                },
                cancel:function() {
                    alert("cancel");
                },
                error:function(error) {
                    alert(error);
                },
                saveToPhotoGallery:true,
                allowEditing:true,
                mediaTypes:[Ti.Media.MEDIA_TYPE_VIDEO,Ti.Media.MEDIA_TYPE_PHOTO]
            });

    You can use the built in functions available to any image blob - Titanium.Blob including;

    • Titanium.Blob-method-imageAsResized
    • Titanium.Blob-method-imageAsThumbnail
