Changing font from eventListener doesnt see to work.

I have a picker for font size that updates a preview label. The font of the label, however, does not update. To demonstrate, I showed that things like text do work in this listener. I am new to JavaScript to I have no idea when things are pass by reference or pass by value.

Thank you for your time.

for (var x = 0; x < 5; x++) {    
    var label = Titanium.UI.createLabel({
        text: (x*5+10).toString(),
        color: '#ffffff',
        backgroundColor: '#000000',
        font: {
            fontWeight:'bold',
            fontSize: 20
        }
    });

    row[x] = Titanium.UI.createPickerRow({
        hasChild:true,
        title: (x*5+10).toString()
    });
    row[x].add(label);
}

var picker = Titanium.UI.createPicker();
picker.add(row);

var preview = Titanium.UI.createLabel({
    text: 'Preview',
    font: {
        fontWeight:'bold',
        fontSize: 20
    },
    color: '#ffffff',
    backgroundColor: '#000000',
    bottom: 20
});

picker.addEventListener('change', function (e) {
    preview.font.fontSize = (e.rowIndex*5+10);
    preview.text = preview.font.fontSize;
});
