    Hi

    Try something more like this;

    var addRow = function (dish) {
    if (dish.category == category) {
        var row = Ti.UI.createView({
                height : '25%',
                top : 0
            });

        var dishPic = Ti.UI.createImageView({
                image : dish.photo.urls.square_75,
                top : '10%',
                left : 0
            });

        var dishName = Ti.UI.createLabel({
                text : dish.name,
                font : {
                    fontSize : 20
                },
                height : "50%",
                textAlign : 'left',
                width : '43%',
                top : '10%',
                left : '15%',
                color : 'black'
            });

        var dishPrice = Ti.UI.createLabel({
                text : '(S$) ' + dish.price,
                font : {
                    fontSize : 20
                },
                height : "50%",
                textAlign : 'left',
                width : '43%',
                top : '10%',
                left : '60%',
                color : 'black'
            });

        var orderBtn = Ti.UI.createButton({
                title : 'Order',
                textAlign : 'left',
                top : '10%',
                color : 'black',
                left : '80%'
            });

        orderBtn.addEventListener('click', function (e) {
            var order = Titanium.UI.createWindow({
                    url : "/ui/tablet/OrderList.js",
                    // backgroundImage : '/images/Animalinfobg.jpg',
                    modal : true,
                    fullscreen : true,
                    navBarHidden : true
                });
            order.dishname = dishName.getText();
            order.open();
        });

        /*
         * Dynamic rating bar
         */
        var ratingBar1 = ratingbar.createRatingBar({
                top : '50%',
                left : '14%',
                rating : 2,
                stars : 5,
                isIndicator : false
            });

        row.add(ratingBar1);
        row.add(dishPic);
        row.add(dishName);
        row.add(dishPrice);
        row.add(orderBtn);
    }

    return row;
};

for (var i = 0; i < e.dishes.length; i++) {
    rightcontainer.add(addRow(e.dishes[i]));
}

    This creates a function to handle a single instance of the "this" you need - it then returns that instance. YOu simply call it many times - in this case in your loop.

    This principle should do what you want - I have not tried your code - but have reformatted it into the new structure.

    • Another thing to remember is that events should "bubble up" so clicking a button sends the event to the button and then to the view the button is inside and then to the window the view is inside.

      So you could try simply adding a single event listener to the parent view rightcontainer and listen for all click events - mark your button so you can determine what to do with it and respond accordingly.

      // comment out existing event listener
// change button as follows;
var orderBtn = Ti.UI.createButton({
    name: 'button', // this is new
    dishname: dish.name, // this is new
    title : 'Order',
    textAlign : 'left',
    top : '10%',
    color : 'black',
    left : '80%'
});

// then underneath the loop and all views being added
// the generic event listener

rightcontainer.addEventListener('click', function (e) {
    // only processes if it was a button pressed
    if (e.source.name === 'button') {
        var order = Titanium.UI.createWindow({
                url : "/ui/tablet/OrderList.js",
                // backgroundImage : '/images/Animalinfobg.jpg',
                modal : true,
                fullscreen : true,
                navBarHidden : true
            });
        order.dishname = e.source.dishname; // reads from the new property
        order.open();
    }
});
