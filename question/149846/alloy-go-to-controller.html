Script Error = 'undefined' is not a function (evaluating '$.job.open()') at job.js (line 12).

My button calls this function

function job(){
    var controller = Alloy.createController('job');
    controller.getView().open({modal:true});
}

And the content of job.js is only $.job.open();

SDK-version is 3.0.2.GA, iOS version 6.1 for iPad.

    • Hi Rakesh,

      Here is the sample code for that:

      var TOP_ANIMATE_START = -320;

var win = Titanium.UI.createWindow({
    layout: 'vertical',
    backgroundColor:'#003300',
    height : 480,
    width : 320,
});

var btn = Ti.UI.createButton({
    top: 100,
    left: 30,
    title: 'Open child window with animation'
});
btn.addEventListener('click', function() {
        var childWin = Titanium.UI.createWindow({
            layout: 'vertical',
            backgroundColor:'blue',
            height : 480,
               width : 320,
        });

       childWin.left = TOP_ANIMATE_START;

        var btn1 = Ti.UI.createButton({
            top: 50,
            left: 100,
            title: 'close child window'
        });
        btn1.addEventListener('click', function() {
            childWin.close();
        });

        childWin.add(btn1);
        childWin.open();

        childWin.animate({left:0, duration:1000}, function(){
        });
});

win.add(btn);
win.open();
