More efficient array

var images = [];
images['home'] = 'images/icons/home.png';
images['page'] = 'images/icons/about.png';
images['blog'] = 'images/icons/design.png';

    var images = [{
    home : 'images/icons/home.png'
}, {
    page : 'images/icons/about.png'
}, {
    blog : 'images/icons/design.png'
}];
