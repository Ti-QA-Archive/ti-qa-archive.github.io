    Hi Simone, you have to use events, here is an example:

    In your code:

    var webView = Ti.UI.createWebView({
        url : "esempio.html"
    });
    webView.setScalesPageToFit(true);

    webView.addEventListener("load", function(e) {
        //alert("Caricato");

        var myValues = [['Year', 'Sales', 'Expenses'], ['2004', 1000, 400], ['2005', 1170, 460], ['2006', 660, 1120], ['2007', 1030, 540]];

        var myOptions = {
            title : 'Company Performance',
            hAxis : {
                title : 'Year',
                titleTextStyle : {
                    color : 'red'
                }
            }
        };

        Ti.App.fireEvent('app:drawChart', {
            values : myValues,
            options : myOptions,
        });
    });

    scrollView.add(webView);

    and your HTML have to be:

    <html>
    <head>
        <script type="text/javascript" src="https://www.google.com/jsapi"></script>
        <script type="text/javascript">
            var currentValues = null;
            var currentOptions = null;

            Ti.App.addEventListener("app:drawChart", function(e) {
                currentValues = e.values;
                currentOptions = e.options;

                var data = google.visualization.arrayToDataTable(currentValues);

                var chart = new google.visualization.ColumnChart(document.getElementById('chart_div'));

                chart.draw(data, currentOptions);
            });

            google.load("visualization", "1", {
                packages : ["corechart"]
            });
        </script>
    </head>
    <body>
        <div id="chart_div" style="width: 900px; height: 500px;"></div>
    </body>
</html>
