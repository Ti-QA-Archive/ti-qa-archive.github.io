Behavior about button click inside table row

My code as below:

this.win = Ti.UI.createWindow({
    backgroundColor : '#fff'
});

var label = Ti.UI.createLabel({
    text : 'Selected 0';
});

var tableData = [];

for(var i = 0; i < 10; i++) {
    var row = Ti.UI.createTableViewRow();
    var button = Ti.UI.createButton({
    title : 'click me',
        width : 100,
        height : 40,
    });
    row.add(button);
    tableData.push(row);

    button.addEventListener('click', function(e) {
        label.text = 'selected!!!';
    });
}

var tableView = Ti.UI.createTableView({
    data : tableData
});

this.win.add(tableView);

this.win.open();

If I change the label to 


this.label = Ti.UI.createLabel({
    text : 'Selected 0';
});

and inside the button listener changed to 


    button.addEventListener('click', function(e) {
        this.label.text = 'selected!!!';
    });

the app crashed with the error undefined object. Can someone explain this behavior please?
