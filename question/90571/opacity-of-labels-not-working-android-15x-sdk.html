  • The opacity is only working for the background color not for the color of the text (see here).

    var win1 = Titanium.UI.createWindow({  
    title:'Win 1',
    exitOnClose: true,
    backgroundColor:'#fff'
});

var label1 = Ti.UI.createLabel({
    backgroundColor: 'red',
    opacity: 0.5,
    top: 100,
    left: 100,
    width: 200,
    height: 200,
    text: 'Opacity for background is working'
});
win1.add(label1);

var label2 = Ti.UI.createLabel({
    color: 'black',
    opacity: 0.3,
    font: {fontSize: 24, fontWeight: 'bold'},
    top: 230,
    left: 40,
    text: 'Opacity for the text is not working'
});
win1.add(label2);
win1.open({fullscreen: true});
