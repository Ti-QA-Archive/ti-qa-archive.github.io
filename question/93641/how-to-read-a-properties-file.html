  • set:

    Titanium.App.Properties.setString('stored_thingy','you are lazy!')

    get:

    var stored_thingy = Titanium.App.Properties.getString('stored_thingy');

Ti.API.info(stored_thingy);  // will tell you how much effort you put in to finding a solution ;p
