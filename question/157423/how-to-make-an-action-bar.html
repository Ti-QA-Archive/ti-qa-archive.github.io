How to make an action bar?

Hi guys I am trying to make an action bar…I've been using the example given in the api doc but it doesn't work…any code snippets would help.

http://docs.appcelerator.com/titanium/3.0/#!/api/Titanium.Android.ActionBar

var win = Ti.UI.createWindow({
    title: "Old Title",
    navBarHidden: false
});
var actionBar;

win.addEventListener("open", function() {
    if (Ti.Platform.osname === "android") {
        if (! win.activity) {
            Ti.API.error("Can't access action bar on a lightweight window.");
        } else {
            actionBar = win.activity.actionBar;
            if (actionBar) {
                actionBar.backgroundImage = "/bg.png";
                actionBar.title = "New Title";
                actionBar.onHomeIconItemSelected = function() {
                    Ti.API.info("Home icon clicked!");
                };
            }
        }
    }
});

win.open();

    replace your android section in tiapp.xml with this snippet 

    <android xmlns:android="http://schemas.android.com/apk/res/android">
    <tool-api-level>15</tool-api-level>
    <manifest>
        <uses-sdk android:targetSdkVersion="11"/>
        <!-- other manifest entries -->
    </manifest>
</android>

    for more info about ( tool-api-level ) and ( uses-sdk ) you can read more here and here
