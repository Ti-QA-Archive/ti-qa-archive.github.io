  • Maybe this is not the best way to do this, but it is a solution that worked for me!
    I hope it helped!

    // Alloy.js
Alloy.Globals.focusedWin = false;


// win in focus
var focusWin = function(e) {
    if(!Alloy.Globals.focusedWin){
        // ... your code
        Alloy.Globals.focusedWin = true;
    }
};
win.addEventListener('focus', focusWin);



// ... another window
win2.addEventListener('open',function(e){
    Alloy.Globals.focused = false;
});
