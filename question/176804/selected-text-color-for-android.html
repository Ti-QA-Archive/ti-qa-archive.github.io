selected text color for android

I wanna set selected text color of button for android.

But, 'selectedColor' property can't be used in android.

Do you have any idea?

    Hi SeungChan,

    even if this is no native Android behavior, you can do this with event listeners.

    var win = Titanium.UI.createWindow({
    layout : 'vertical',
    backgroundColor : 'white'
});

var buttonColorDefault = 'white';
var buttonColorSelected = 'green';

var button1 = Ti.UI.createButton({
    title : 'my Button1',
    backgroundColor : 'red',
    color : buttonColorDefault,
    selectedColor : buttonColorSelected, // iOS only
    top : 100
});
button1.addEventListener('touchstart', function(e) {
    if (Ti.Platform.name === 'android') {
        Ti.API.info('Android button1 touchstart');
        button1.color = buttonColorSelected;
    }
});
button1.addEventListener('touchend', function(e) {
    if (Ti.Platform.name === 'android') {
        Ti.API.info('Android button1 touchend');
        button1.color = buttonColorDefault;
    }
});

win.add(button1);
win.open();
