    Hello Orhan,

    why don't you use tabbedBar at the top of the window instead of the tabs…

    see this ….

    
var win = Titanium.UI.createWindow({
    backgroundColor : 'white'
});

var tb1 = Titanium.UI.createTabbedBar({
    labels : ['Books', 'Info', 'Map', 'Photos'],
    backgroundColor:'#336699',
    top : 0,
    style : Titanium.UI.iPhone.SystemButtonStyle.BAR,
    height : 40,
    width : '100%',
    index : 1,
});
win.add(tb1);

tb1.addEventListener('click', function(e) {
    if(e.index == 0) {
        alert('Books');
    }
    if(e.index == 1) {
        alert('Info');
    }
    if(e.index == 2) {
        alert('Maps');
    }
    if(e.index == 3) {
        alert('Photos');
    }
})

win.open();

var win = Titanium.UI.createWindow({
    backgroundColor : 'white'
});

var tb1 = Titanium.UI.createTabbedBar({
    labels : ['Books', 'Info', 'Map', 'Photos'],
    backgroundColor : '#336699',
    top : 0,
    style : Titanium.UI.iPhone.SystemButtonStyle.BAR,
    height : 40,
    width : '100%',
    index : 1,
});
win.add(tb1);

var newWin = Ti.UI.createWindow({
    url : 'books.js'
});

tb1.addEventListener('click', function(e) {
    if(e.index == 0) {
        var newWin = Ti.UI.createWindow({
            url : 'books.js',
            top : 41
        });
        newWin.open();
    }
    if(e.index == 1) {
        var newWin = Ti.UI.createWindow({
            url : 'info.js',
            top : 41
        });
        newWin.open();
    }
    if(e.index == 2) {
        var newWin = Ti.UI.createWindow({
            url : 'map.js',
            top : 41
        });
        newWin.open();
    }
    if(e.index == 3) {
        var newWin = Ti.UI.createWindow({
            url : 'galary.js',
            top : 41
        });
        newWin.open();
    }
})

win.open();
