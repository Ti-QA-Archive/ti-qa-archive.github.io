Now we add buttons to the left pane:

HomeWindow.prototype.addButton = function(txt, left_pos, top_pos, img_width, img_height, img) {
    var btn = Ti.UI.createButton({
        // title : txt,
        top : top_pos,
        left : left_pos,
        height : img_height,
        width : img_width,
        backgroundImage : img,
        zIndex : 11
    });
    this.buttons[this.button_count] = btn;
    this.button_count += 1;
    // this.win.add(btn);
    return btn;
};
