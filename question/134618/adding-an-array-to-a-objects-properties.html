Adding an array to a objects properties

I'm tinkering for the first time with Titanium. I created a couple of text fields and a button. In the button I added a couple of items to the property list (objName and neededObjs) and in neededObjs I am adding an array that contains my text fields.

Here's my code:

var btnLogIn = Titanium.UI.createButton({
objName: "btnLogIn",
neededObjs: {

    obj1: txtUsername,
    obj2: txtPassword
},
title: "Submit",
width: 120,
height: 35,
top: 200,
left: mainWin.width/2-60

});

I then have an eventListener and that points to a function where I am going to validate the text fields. Thing is I can alert and see the array, which makes me think it is no longer an array. Am I going about this wrong?

Thanks

  • Hi Steve,

    What you're doing is creating an object called neededObjs not an array. To access it, you can do the following.

    btnLogin.addEventListener('click', function(e) {
     Ti.API.debug('txtUsername'+btnLogin.neededObjs.obj1);
     Ti.API.debug('txtPassword'+btnLogin.neededObjs.obj2);
});

    Hope this helps.

    Cheers,
    Paul

  • Here's a snippet from one of my projects.

    
var submitErrors = [];
        submitButton.addEventListener('click', function() {
            firstNameTextField.blur();
            LastNameTextField.blur();
            companyNameTextField.blur();
            emailTextField.blur();

            if (firstNameTextField.value === '') {
                firstNameTextField.borderColor = app.errorred;
                firstNameTextField.backgroundColor = app.errorlightred;
                submitErrors[0] = 'firstname';
            } else {
                firstNameTextField.borderColor = app.reflexblue;
                firstNameTextField.backgroundColor = '#E5E5FF';
                submitErrors[0] = '';
            }

            if (LastNameTextField.value === '') {
                LastNameTextField.borderColor = app.errorred;
                LastNameTextField.backgroundColor = app.errorlightred;
                submitErrors[1] = 'lastname';
            } else {
                LastNameTextField.borderColor = app.reflexblue;
                LastNameTextField.backgroundColor = '#E5E5FF';
                submitErrors[1] = '';
            }

            if (companyNameTextField.value === '') {
                companyNameTextField.borderColor = app.errorred;
                companyNameTextField.backgroundColor = app.errorlightred;
                submitErrors[2] = 'companyname';
            } else {
                companyNameTextField.borderColor = app.reflexblue;
                companyNameTextField.backgroundColor = '#E5E5FF';
                submitErrors[2] = '';
            }

            if (emailTextField.value === '') {
                emailTextField.borderColor = app.errorred;
                emailTextField.backgroundColor = app.errorlightred;
                submitErrors[3] = 'email';
            } else {
                var reg = /^([A-Za-z0-9_\-\.])+\@([A-Za-z0-9_\-\.])+\.([A-Za-z]{2,4})$/;
                var address = emailTextField.value;
                if(reg.test(address) === false) {
                    submitErrors[3] = 'email';
                    emailTextField.borderColor = app.errorred;
                    emailTextField.backgroundColor = app.errorlightred;
                 } else {
                    emailTextField.borderColor = app.reflexblue;
                    emailTextField.backgroundColor = '#E5E5FF';
                    submitErrors[3] = '';
                }
            }
            var i = 0;
            for(var s=0;s<submitErrors.length;s++) {
                if (submitErrors[s] !== '') {
                    i++;
                }
            }
            if (i > 0) {
                // do something with your error
            }
        });
