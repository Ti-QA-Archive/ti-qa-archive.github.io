  • Hi, you can use this code :

    var win = Ti.UI.createWindow({
    backgroundColor:'#ffffff'
});
var scrollView = Ti.UI.createScrollView({
    contentHeight:'auto',
    contentWidth:'auto',
    top:0,
    showHorizontalScrollIndicator:false,
    showVerticalScrollIndicator:true
});

var maxWidth = Ti.Platform.displayCaps.platformWidth;
var settings = {
    number:100,
    height:20,
    width:20,
    padding:3   
}
settings.perLine = Math.floor(maxWidth / (settings.width + settings.padding));
settings.lines = Math.ceil(settings.number / settings.perLine);

var counter = 0;
for (var i = 0; i < settings.lines; i++) {
    for (var j = 0; j < settings.perLine; j++) {
        var image = Ti.UI.createImageView({
            url:'cool.gif',
            height:settings.height,
            width:settings.width,
            left:(settings.padding*(j+1)) + (settings.width*j),
            top:(settings.padding*(i+1)) + (settings.height*i)
        });     
        scrollView.add(image);
        counter++;
        if (counter >= settings.number) break;
    }
}

win.add(scrollView);
win.open();

    You may also refer http://developer.appcelerator.com/question/891/grid-layout this link.

    You can set your images in desire manner in rows and column using both ways.

  • Hi

    The answer Moiz gave will solve your problem, but here is another way of achieving your goal.

    The difference here is that you can use the layout property of a view to handle all of the positioning of the child views, this means you do not need to work anything out including which line this may appear on, it also means you could have different sized images IF you chose to.

    To test this create a new default project and replace the contents of the app.js file with the code below. You can also pick and choose from the code if you prefer.

    var win = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Answers'
});
var tab = Ti.UI.createTab({
    icon: 'KS_nav_views.png',
    title: 'Answers',
    window: win
});

//example data
var json = [
    { image: 'http://placehold.it/96x96/ff0000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/ee0000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/dd0000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/cc0000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/bb0000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/aa0000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/990000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/880000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/770000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/660000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/550000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/440000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/330000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/220000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/110000/ffffff&text=Hi' },
    { image: 'http://placehold.it/96x96/000000/ffffff&text=Hi' }
];
var scroll = Ti.UI.createScrollView({
    contentHeight: Ti.UI.SIZE,
    contentWidth: Ti.UI.SIZE,
    height: Ti.UI.FILL,
    showHorizontalScrollIndicator: false,
    showVerticalScrollIndicator: true,
    width: Ti.UI.FILL
});
win.add(scroll);
// provides the ability to offset positions
// also the layout property
var view = Ti.UI.createView({
    bottom: 8,
    height: Ti.UI.SIZE,
    layout: 'horizontal',
    left: 0,
    right: 8,
    top: 0,
    width: Ti.UI.FILL
});
scroll.add(view);
var intImage = 0, intImages = json.length, img;
for (intImage = 0; intImage < intImages; intImage = intImage + 1) {
    img = Ti.UI.createImageView({
        height: 96,
        image: json[intImage].image,
        left: 8,
        top: 8,
        width: 96
    });
    view.add(img);
}

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab);
tabGroup.open();
