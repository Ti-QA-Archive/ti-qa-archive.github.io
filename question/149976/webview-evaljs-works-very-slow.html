  • following I use:

    newest titanium sdk

    function count_selections(p) {
    var count = 0;

    for (var i = 0; i < questions[p].answers.length; i++) {
        if (questions[p].answers[i].type == 'text') {
            var text_value = webview.evalJS('document.getElementById(' + questions[p].answers[i].id + ').value');
            if (text_value != '') {
                count++;
            }
        } else {
            var selected = webview.evalJS('document.getElementById(' + questions[p].answers[i].id + ').checked');
            if (selected == 'true') {
                count++;
            }
        };

    };
    return count;
};

    AND

    function select_warning(p) {
    for (var i = 0; i < questions[p].answers.length; i++) {
        if (questions[p].answers[i].type == 'text') {
            var text_value = webview.evalJS('document.getElementById(' + questions[p].answers[i].id + ').value');
            if (text_value != '' && questions[p].answers[i].selected_warning != '' && count_selections(p) > 1) {
                return questions[p].answers[i].selected_warning;
            }
        } else {
            var selected = webview.evalJS('document.getElementById(' + questions[p].answers[i].id + ').checked');
            if (selected == 'true' && questions[p].answers[i].selected_warning != '' && count_selections(p) > 1) {
                return questions[p].answers[i].selected_warning;
            }

        }

    }
    return '';
};

    • evalJS is always bound to be slow. (Think about it, you are evaluating dynamic JavaScript at runtime, thus preventing any optimization from the browser.)

      Combined to the fact that you are running on a mobile device that has a lot less resources than a standard desktop computer.

      I'm afraid the only way around this is to find a different approach to your code.
