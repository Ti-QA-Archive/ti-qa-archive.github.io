  • on your code you have some tags spell wrong.
    this should give you the lay out that you are looking for

    
var gridUIControl = Ti.UI.createView({
    layout:'vertical',
    //style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN,
    backgroundColor:'#FEEEFE',
    borderWidth:3,
    height:300,
    width:250,
    borderColor:'#DDFFFF'
});
var imageHolder = Ti.UI.createView({
    backgroundColor:'#0000FF',
    borderWidth:5,
    borderColor:'red',
    right : '3dp',
    left:'3dp',
    top:0,
    height:gridUIControl.height-18
});
gridUIControl.add(imageHolder);
var favorite = Ti.UI.createImageView({
    // image : Ti.App.loadImage('icon_Favorite_12px'),
    visible : true,
    right : '3dp',
    bottom : '3dp',
    backgroundColor:'yellow',
    top:0,
    height:18
});
gridUIControl.add(favorite);

    • you can fix that by doing this

      var gridUIControl = Ti.UI.createView({
    layout:'vertical',
    //style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN,
    backgroundColor:'#FEEEFE',
    borderWidth:3,
    height:300,
    width:250,
    borderColor:'#DDFFFF'
});

var imageHolder = Ti.UI.createView({
    backgroundColor:'#0000FF',
    borderWidth:5,
    borderColor:'red',
    top:0,
    height:gridUIControl.height-18
});
gridUIControl.add(imageHolder);

var imageTop = Ti.UI.createImageView({
    image : Ti.App.loadImage('icon_Favorite_12px'),
    left : 3 // <- delete this if you just want to align center on IO. For android you have to tell it where to go or is just going to push it left
});

favorite.add(imageTop);

var favorite = Ti.UI.createView({
    backgroundColor:'yellow',
    top:0,
    height:18
});
gridUIControl.add(favorite);

var imageBot = Ti.UI.createImageView({
    image : Ti.App.loadImage('icon_Favorite_12px'),
    right : 3
});

favorite.add(imageBot);
