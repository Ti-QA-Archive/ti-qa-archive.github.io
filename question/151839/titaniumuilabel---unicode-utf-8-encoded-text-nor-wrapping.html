  • Here what i use

    
var db = Alloy.Globals.openDB();
    var records = db.execute("SELECT * FROM verses where book_id=? and chapter_id=? ORDER BY id", bookId, chaptersId);
    while (records.isValidRow()) {
        var verse = records.fieldByName('verse_id') + '. ' + records.fieldByName('verse');
        Ti.API.info(verse);
        var lbl = Ti.UI.createLabel({
            html: verse,
            wordWrap: true,
            ellipsize: true,
            font : {
                fontSize : 30,
                fontWeight : "bold"
            },
            color: "#000",
            top: 10,
            width: "98%",
            height: Ti.UI.SIZE
        }); 
        $.scrollView.add(lbl);
        records.next();
    }
    records.close();
    db.close();
