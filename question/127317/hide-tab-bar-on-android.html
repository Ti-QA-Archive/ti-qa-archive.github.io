Hide tab bar on Android

Hi, I'm trying to hide the tab bar in tabGroup on android - but it doesn't seem to work. It works on iPhone but not Android:

var tabGroup = Titanium.UI.createTabGroup({});

var win1 = Titanium.UI.createWindow({
    backgroundColor: 'green',
    tabBarHidden: true
});

var tab1 = Titanium.UI.createTab({
    window: win1
});

tabGroup.addTab(tab1);
tabGroup.open();

anyone knows what's wrong?

  • There solution.

    <?xml version="1.0" encoding="utf-8"?>
<TabHost xmlns:android="http://schemas.android.com/apk/res/android"
    android:id="@android:id/tabhost"
    android:layout_width="fill_parent"
    android:layout_height="fill_parent">

    <LinearLayout
        android:orientation="vertical"
        android:layout_width="fill_parent"
        android:layout_height="fill_parent"
        android:padding="0dp">

        <FrameLayout
            android:id="@android:id/tabcontent"
            android:layout_width="fill_parent"
            android:layout_height="wrap_content"
            android:padding="0dp"
            android:layout_weight="1"/>

        <TabWidget
            android:id="@android:id/tabs"
            android:layout_width="fill_parent"
            android:layout_height="0dp"
            android:layout_weight="0"/>

    </LinearLayout>

</TabHost>

    Put these lines in '<app-project-dir>/platform/android/res/layout/titanium_tabgroup.xml' file and make clean bulid.

  • Vittorio's solution is nice, but I didn't want to actually show the tab bar leaving the screen.

    My solution is based on his code. On Android, project starts with an empty black screen, showing the TabGroup right afterwards, already without a tab bar.

    If you/client don't mind the black screen, here is an Alloy TabGroup + TableView example, already set up for iOS and Android (solution on index.js - doWindowFocus method):

    index.xml

    <Alloy>
    <TabGroup id="tabGroup">
        <Tab id="tab">
            <Window id="win" title="Main Window" onFocus="doWindowFocus">
                <TableView onClick="doTableViewClick">
                    <TableViewRow id="tableViewRow1" title="Row 1" />
                    <TableViewRow id="tableViewRow2" title="Row 2" />
                    <TableViewRow id="tableViewRow3" title="Row 3" />    
                </TableView>
            </Window>
        </Tab>
    </TabGroup>
</Alloy>

    index.tss

    "TabGroup": {
    backgroundColor: "white"
},

"TabGroup[platform=android]": {
    title: "Main Window"
},

"#win": {
    backgroundColor: "white",
    tabBarHidden: true  // hides the tab bar on iOS
},

"TableViewRow": {
    hasChild: true
},

"#tableViewRow1": {
    className: "leftImageAndTextRowStructure",
    leftImage: "/shared/images/row1.png"
},

"#tableViewRow2": {
    className: "leftImageAndTextRowStructure",
    leftImage: "/shared/images/row2.png"
},

"#tableViewRow3": {
    className: "leftImageAndTextRowStructure",
    leftImage: "/shared/images/row3.png"
}

    index.js

    function doWindowFocus(e)
{
    // Hides the tab bar on Android

    if (OS_ANDROID)
    {
        function hideTabBar()
        {
            $.tabGroup.animate({
                opacity: 1,
                top: -94,
                duration: 1
            });
        }

        $.tabGroup.visible = false;
        $.tabGroup.opacity = 0;

        setTimeout(hideTabBar, 2);
    }
}

function doTableViewClick(e)
{
    var detailWindowProperties =
    {
        selectedItem: e.rowData.title
    }

    var detailWindow = Alloy.createController("detailwin", detailWindowProperties).getView();

    $.tab.open(detailWindow, {animated: true});
}

$.tabGroup.open();

    detailwin.xml

    <Alloy>
    <Window id="detailWindow">
        <Label id="label" />
    </Window>
</Alloy>

    detailwin.tss

    "Window": {
    backgroundColor: "white"
}

"Label": {
    width: "auto",
    height: "auto"
}

    detailwin.js

    var args = arguments[0] || {};

$.detailWindow.title = args.selectedItem;
$.label.text = args.selectedItem + " was selected";
