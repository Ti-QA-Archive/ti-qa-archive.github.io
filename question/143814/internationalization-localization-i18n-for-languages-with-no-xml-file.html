    Hi Leonardo,
    I had the same problem and i fixed it by using Ti.Locale.getCurrentLanguage()
    in the app.js i check the phone language and if its one of the language you have you use it as default if not you use English here is my code :
    app.js part

    if(Ti.App.Properties.getString('lang') == null)
    {
        if(Ti.Locale.getCurrentLanguage() == 'fr')
        {
            Ti.App.Properties.setString('lang', 'fr');
        }
        else
        {
            Ti.App.Properties.setString('lang', 'en');
        }
    }

    Ti.App.Properties.getString('lang') i m using this to allow the user to change a language

    this is the L function code:

    function L(text) {
    var lang = Ti.App.Properties.getString("lang");
    var file = Ti.Filesystem.getFile('i18n/' + lang + '/strings.xml');
    var xmltext = file.read().text;
    var xmldata=Titanium.XML.parseString(xmltext);
    var elements = xmldata.documentElement.getElementsByTagName('string');
    for(var i=0; i<elements.length; i++){
        if(elements.item(i).getAttribute('name') == text)
            return elements.item(i).text;
    }
}

module.exports = L;

    Hope it will help you :)
