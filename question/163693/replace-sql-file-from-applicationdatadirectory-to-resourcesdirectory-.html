      I have solve my issue using this way in both android as well as iOS

      if (OS_IOS) {
        var f = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory, '_alloy_.sql');
        if (f.exists()) {
            Ti.API.info('IOS---> Resource file Find');
        }
        //// Private ////
        var testFile = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory);
        var privateDir = testFile.nativePath.replace('Documents/', '');
        privateDir += 'Library/Private%20Documents/';

        //// Get that File from Private ////
        var DBfile = Ti.Filesystem.getFile(privateDir, "_alloy_.sql");
        if (DBfile.exists()) {
            Ti.API.info('DBfile Available');
        }

        //// In iOS we need to delete that File and than create ////
        DBfile.deleteFile();
        if (DBfile.exists()) {
            Ti.API.info('DBfile Available');
        } else {
            Ti.API.info('DBfile Deleted');
        }
        var file = Ti.Filesystem.getFile(privateDir, "_alloy_.sql");
        file.createFile();

        //// USE of Resource File /////
        var source = f.read();
        Ti.API.info('sor len' + source.length);
        file.append(source);
    } else {

        var f = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory, '_alloy_');
        if (f.exists()) {
            Ti.API.info('ANDROID---> Resource file Find');
        }

        //// Private ////
        var privateDir = 'file:///data/data/' + Ti.App.getID() + '/databases/';

        //// Get that File from Private ////
        var DBfile = Ti.Filesystem.getFile(privateDir, "_alloy_");
        if (DBfile.exists()) {
            Ti.API.info('DBfile Available');
        }

        //DBfile.deleteFile();

        if (DBfile.exists()) {
            Ti.API.info('DBfile Available');
        } else {
            Ti.API.info('DBfile Deleted');
        }

        //// USE of Resource File /////
        DBfile.write(''); // Set Null
        var source = f.read();
        Ti.API.info('sor len' + source.length);
        DBfile.write(source);

    }

    Firstly i want to inform you that the method in the filesystem that is used to check whether the file exists or not is called exists() not exist() as in your code, I don't know why you didn't got the error,
    Secondly please copy paste this code and run your app, hopefully your problem will be resolved.

    var testFile = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory);
var privateDir = testFile.nativePath.replace('Documents/', '');
privateDir += 'Library/Private%20Documents';

var file = Ti.Filesystem.getFile(privateDir, "_alloy_.sql");

var sourcefile = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory, '_alloy_.sql');

if (sourcefile.exists() && file.exists()) {
    file.deleteFile();
    var createDestinationFile = Ti.Filesystem.getFile(privateDir, sourcefile.name);
    createDestinationFile.write(sourcefile.read());
}

  • The alloy.sql file is the automatically-generated database file created when you instantiate a model in your application. You should not be manually manipulating that file. If you need to modify the structure of that database or a table that it contains, use a migration file. If you want to supply your own database, which will be installed along with your app, see the sample app in the Alloy repo.
