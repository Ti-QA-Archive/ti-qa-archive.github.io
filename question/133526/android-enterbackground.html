  • I found a way to make this work on Android:

    var resumed = false;

Ti.Android.currentActivity.addEventListener('pause', function(e) {
    resumed = true;

});
Ti.Android.currentActivity.addEventListener('resume', function(e) {
    if(resumed == true) {
        alert('let\'s update');
        resumed = false;
    }
});
