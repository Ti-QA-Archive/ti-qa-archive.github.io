    Hi,

    try this one..

    var like = Titanium.UI.createImageView({
    image : 'images/liked.png',
    left : 270,
    top : 5,
    height : 30,
    width : 30
});

like.addEventListener('click', function(e) {
    e.source.image= 'images/unliked.png';
    alert('like pressed');
});
