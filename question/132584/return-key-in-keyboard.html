        keyboardType: Titanium.UI.KEYBOARD_DECIMAL_PAD
});

    you can do that by this way

    var flexSpace = Titanium.UI.createButton({
    systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});
var btnDone = Titanium.UI.createButton({    
    title : 'Done',
    width : 67,
    height : 32
});
btnDone.addEventListener('click',function(e){
    txt.blur();
});

var btnCancle = Titanium.UI.createButton({
    title : 'Cancel',
    width : 67,
    height : 32
});
var txt = Titanium.UI.createTextField({
    width : 140,
    textAlign : 'left',
    font : {
        fontSize : 14,
        fontWeight : 'bold'
    },
    top : 10,
    left : 155,
    height : 25,
    color : '#535353',
    keyboardToolbar : [flexSpace, btnCancle, flexSpace,flexSpace, btnDone, flexSpace],
    keyboardToolbarColor : '#999',
    keyboardToolbarHeight : 40,
    keyboardType : Titanium.UI.KEYBOARD_DECIMAL_PAD
});
