    • ok here you go make a window and hide the navbar

      var win=Ti.UI.createWindow({
backgroundColor:'white'
navBarHidden:true
})
var navBarView = Ti.UI.createView({
    top : 0,
    left : 0,
    height : 45,
    width : '100%',
    backgroundColor : 'black'
});
win.add(navBarView);

var lblTitle = Ti.UI.createLabel({
    text : win.title,
    top : 0,
    left : '18.75%',
    height : navBarView.height,
    width : '62.5%',
    textAlign : 'right',
    color : '#FFFFFF',
    font : {
        fontWeight : 'bold',
        fontSize : 22
    }
});
win.add(lblTitle);
