So just to clarify, imagine you are typing and someone sitting opposite you is reading what you are typing. The text needs to be appearing left to right for him and the letters will be the right way up for him, but you will be typing backwards for you and right to left….

    Hi

    I think I get what you mean - so I have put a simple test app together for you to see how easy it can be.

    You do not mention if this is Android, iOS (iPhone or iPad), which SDK version, target platform OS. It should not matter for this as it will work across the board, but please try to add this information in future questions, helps you get more answers.

    Create a blank simple project and replace the contents of the app.js with the code below.

    Ti.UI.setBackgroundColor('#000');
var tab9;
(function () {
    var win = Ti.UI.createWindow({  
        backgroundColor: '#fff',
        height: Ti.UI.FILL,
        title: 'Text',
        width: Ti.UI.FILL
    });
    var txt = Ti.UI.createTextField({
        borderColor: '#666',
        height: 50,
        left: 20,
        paddingLeft: 10,
        paddingRight: 10,
        right: 20,
        top: 50,
        width: Ti.UI.FILL
    });
    win.add(txt);
    txt.transform = Ti.UI.create2DMatrix().rotate(-180);
    tab9 = Ti.UI.createTab({  
        icon: 'KS_nav_views.png',
        title: 'Text',
        window: win
    });
})();
var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab9);
tabGroup.open();
