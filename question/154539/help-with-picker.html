    Hi

    This is how you do it, to test this code create a new default (non-alloy) app and replace the contents of app.js with the code below.

    Ti.UI.setBackgroundColor('#000');
var win1 = Ti.UI.createWindow({  
    backgroundColor: '#fff',
    title: 'Tab 1'
});
var tab1 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 1',
    window: win1
});
var options = [
    { id: 1, title: 'manzana' },
    { id: 2, title: 'naranja' },
    { id: 2, title: 'uva' }
];
var data = [], intItem = 0, intItems = options.length;
for (intItem = 0; intItem < intItems; intItem = intItem + 1) {
    data.push(Ti.UI.createPickerRow(options[intItem]));
}
var picker = Ti.UI.createPicker({
    selectionIndicator : true,
    top: 50
});
picker.add(data);
picker.addEventListener('change', function (e) {
    Ti.API.info('rowIndex: ' + e.rowIndex);
    Ti.API.info('row node: ' + JSON.stringify(options[e.rowIndex]));
    Ti.API.info('method 1 - row id: ' + options[e.rowIndex].id);
    Ti.API.info('method 2 - row id: ' + e.row.id);
});
win1.add(picker);
var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);  
tabGroup.open();

    This is the important bit.

    Ti.API.info('rowIndex: ' + e.rowIndex);
Ti.API.info('row node: ' + JSON.stringify(options[e.rowIndex]));
Ti.API.info('method 1 - row id: ' + options[e.rowIndex].id);
Ti.API.info('method 2 - row id: ' + e.row.id);
