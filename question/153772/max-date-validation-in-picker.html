max date validation in picker

how to do max date validation??

xml view:

<Window class="container" id="datepicker_win">
    <Picker id="date_picker" left="10" right="10" top="50" selectionIndicator="true" useSpinner="true">
</Window>

controller:

$.date_picker.showDatePickerDialog({
    value: new Date(),
    callback: function(e) {
        if (e.cancel) {
            $.datepicker_win.close();
        } else {
            var thisdate = e.value;
            var d = ("0" + thisdate.getDate()).slice(-2);
            var m = month[thisdate.getMonth()];
            var y = thisdate.getFullYear();
            newdate = m+"/"+d+"/"+y ;
            $.datepicker_win.close();
            $.dvr_date.value = newdate;        
        }
    }
});

    Hi,

    try this one

    $.date_picker.showDatePickerDialog({
    value: new Date(),
    callback: function(e) {
        if (e.cancel) {
            $.datepicker_win.close();
        } else {
            var thisdate = e.value;
        var now = Date.parse(new Date());
        var selectedDate = Date.parse(new Date(thisdate));

        if(selectedDate > now){
        alert('Future Date');
        return;
        }

            var d = ("0" + thisdate.getDate()).slice(-2);
            var m = month[thisdate.getMonth()];
            var y = thisdate.getFullYear();
            newdate = m+"/"+d+"/"+y ;
            $.datepicker_win.close();
            $.dvr_date.value = newdate;     
        }
    }
});
