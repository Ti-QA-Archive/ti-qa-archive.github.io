    • you can create a window and add your custom image View as described in the above link..

      create a seperate .js file and paste this code in that file with your imageView…

      then add this .js file in app.js only once after that you can call fireEvent anywhere in your application…like..

      Ti.App.fireEvent('show_indicator');

      Ti.App.fireEvent('hide_indicator');

      // ==============================================================================
// Create custom loading indicator
// ==============================================================================
var indWin = null;
var actWin = null;
var actInd;
var indicatorShowing = false;
function showIndicator(title) {
    indicatorShowing = true;

    // window container
    indWin = Titanium.UI.createWindow({
        backgroundColor:'transparent',
        height:480,
        width:320,
        zIndex: 0
    });

    //add your imageView with the arrays of images

    indWin.open();

}

function hideIndicator() {

    if(indicatorShowing == true) {

        if(indWin != null) {

            indWin.close();
            indWin = null
        }
        indicatorShowing = false;
    }

}

// ==============================================================================
// Add global event handlers to hide/show custom indicator
// ==============================================================================
Titanium.App.addEventListener('show_indicator', function(e) {
    if(e.title == null) {
        e.title = 'Loading... ';
    }
    if(indicatorShowing == true) {
        hideIndicator();
    }
    showIndicator(e.title);
});
Titanium.App.addEventListener('hide_indicator', function(e) {
    hideIndicator();
});
