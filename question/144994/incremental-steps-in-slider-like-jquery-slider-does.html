    Untested:

    var slider = Titanium.UI.createSlider({
    top: 50,
    min: 0,
    max: 10,
    width: '100%',
    value: 5
    });


slider.addEventListener('change', function(e) {
    this.value = Math.round(e.value);
});

      Ti.UI.setBackgroundColor('#222');

var win = Ti.UI.createWindow(); 
var increment = 50;

var label = Ti.UI.createLabel({color: 'white', top:50, text: '50'});

var slider = Titanium.UI.createSlider({
    top: 150,
    min: 0,
    max: 10,
    width: '100%',
    value: 1
    });


slider.addEventListener('touchend', function(e){
    this.value = Math.round(e.value);
    label.text = this.value * increment;
});


win.add(label);
win.add(slider);
win.open();
