TableView was not displayed on the window.

Hi,
I created one table view, and adding one label and switch box every row of the table view, by following the site: http://cssgallery.info/custom-row-for-tableview-in-appcelerator-titanium/
But,when I am running this application, I didn't get any table view on the window. window displayed with empty space.
Here is my code:

var tableview = Titanium.UI.createTableView({});
var CustomData = [
{ name:'Nokia', checkb:"NokiA"},
{ name:'Samsung', checkb:"samsu"},
{ name:'Motrola', checkb:"moto" },
{name:'BlackBerry', checkb:"Black" }
];
var data1=[];
for (var i = CustomData.length - 1; i <= 0; i--) {
var row = Titanium.UI.createTableViewRow();

var name = Titanium.UI.createLabel({
text:CustomData[i].name,
font:{fontSize:10,fontWeight:'bold'},
width:'auto',
textAlign:'left',
top:1,
left:10,
height:6
});

var checkb = Titanium.UI.createSwitch({
    style:Titanium.UI.Android.SWITCH_STYLE_CHECKBOX,
    text:CustomData[i].checkb,
    value:false,left:270
});

row.add(name);
row.add(checkb);
//row.className = 'coutry_row';
data1.push(row);
};
tableview.setData(data1);
win1.add(tableview);

Could you please tell me any one, what is the problem in this code.
solution also..
