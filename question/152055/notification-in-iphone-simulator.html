    • Paste this code into app.js and test it.

      var win = Titanium.UI.createWindow({
    backgroundColor : 'black',
    modal : Ti.UI.iPhone.MODAL_PRESENTATION_FORMSHEET
});

// Notification Lable
var notify = Ti.UI.createLabel({
    top : 10,
    left : 10,
    right : 10,
    height : 40,
    borderRadius : 3,
    borderWidth : 1,
    zIndex : 6000,
    opacity : 0,
    textAlign : 'center'
});
win.add(notify);

//show notification
function NotifyMe(bgcolor, title, titlecolor) {

    notify.text = title;
    notify.color = titlecolor;
    notify.backgroundColor = bgcolor;
    notify.borderColor = titlecolor;
    notify.animate({
        opacity : 1,
        duration : 1000
    });
    setTimeout(function(e) {
        notify.animate({
            opacity : 0,
            duration : 1000
        });
    }, 3000);
}


// Create a Button.
var ClickMe = Ti.UI.createButton({
    title : 'ClickMe',
    height : 30,
    width : 200,
});

// Listen for click events.
ClickMe.addEventListener('click', function() {
    NotifyMe('#f1dfdf', 'Toast-style notification for iphone', '#be4945');
});

// Add to the parent view.
win.add(ClickMe);


win.open();

      you can change it's layout of graphics as you want.
