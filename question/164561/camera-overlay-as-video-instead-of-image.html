    • Hey Osian,

      Yes I did!! After researching about video transparencies on ios I found that it wont allow alpha channels, the only way to do it would be to use ffmpeg. I then found a work around where I exported the video as a png sequence with transparency and looped through the sequence. My video was only a few seconds long as I would assume long videos may cause performance issues. Below is a sample of the code I used

      // Set frames per second
var FPS = 24;
var TriangleImage = Ti.UI.createImageView({
        width:193,
        backgroundColor:'transparent',
        height:159,
        top:(Ti.Platform.displayCaps.platformHeight - Grid.height) / 2 + 158,
        zIndex: 11
});

overlay.add(TriangleImage);

// Triangle Animation
// this is the amount of frames in my png sequence
var TriangleArrayLength=20;
var TriangleIndex=0;
function TriangleAnimation(){
        TriangleImage.image = "assets/images/triangle/triangle" + TriangleIndex + ".png";
        TriangleIndex++;
    // Here I can add sound or other elements based on frame index
     switch (TriangleIndex){
                case 1:
                     triangle.play();
                break;
                case 14:
                     triangle.stop();
                break;
                case 21:
                    TriangleIndex=0;
            // Make sure we clear the loop as it will run infinatley
                     clearInterval(TriangleAnimate);
                break;
            }
}
// Set interval fire up
var TriangleAnimate = setInterval(TriangleAnimation,1000/FPS);

      I hope this helps.
