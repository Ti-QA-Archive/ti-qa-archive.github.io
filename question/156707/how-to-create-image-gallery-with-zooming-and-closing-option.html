    Take new Alloy project and Look below code ..

    if (Titanium.Platform.osname != 'android') {

    var ScrollableImageView = Titanium.UI.createScrollableView({
    showPagingControl : false,
    bottom : 0,
    top : 0,
    left : 0,
    right : 0
});

var image_arr = [];
for (var i = 0; i  <  5; i++) {

    var ScrollView = Ti.UI.createScrollView({
        height : '100%',
        width : '100%',
        contentWidth : 'auto',
        contentHeight : 'auto',
        showVerticalScrollIndicator : true,
        showHorizontalScrollIndicator : true,
        maxZoomScale : 3,
        minZoomScale : 1,
    });
    imageheader = Ti.UI.createImageView({
        left : 0,
        right : 0,
        image : '/appicon.png'
    });
    ScrollView.add(imageheader);
    image_arr.push(ScrollView);
}

ScrollableImageView.views = image_arr;

    }

    $.index.add(ScrollableImageView);

    $.index.open();
