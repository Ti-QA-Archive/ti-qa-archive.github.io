Open URL in Safari from Text Label

I have a url in the text lable "l2" that I would like to make active - is there a simple way to do this?

var win = Titanium.UI.currentWindow;

var l = Titanium.UI.createLabel({
    text:'How To Write a Book.',
    color:'#000000',
    height:'auto',
    top:15,
    width:300,
    textAlign:'left',
    font:{fontWeight:'bold',fontSize:20}
});

var l2 = Titanium.UI.createLabel({
    text:'If you like this tip be sure to pick up a copy of the book 
"How to Write a Book", available at Amazon. http://www.example.com',
    color:'#000000',
    height:'auto',
    top:255,
    width:300,
    textAlign:'left',
    font:{fontFamily:'Arial-ItalicMT',fontSize:14}
});



win.add(l, l2);
