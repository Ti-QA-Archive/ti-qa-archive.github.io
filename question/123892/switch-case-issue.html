  • Hi balu,
    try this

    var winHome=Ti.UI.createWindow({
    title:'Checked Demo',
    backgroundColor:'#fff'
    });
winHome.open();

var viewMain = Titanium.UI.createView({
   backgroundColor:'blue',
   top:10,
   width:200,
   height:250
});
winHome.add(viewMain);

var data = [];
var switchData=[];
var lastRow = 5;
for (var c=0;c<lastRow;c++)
{
    switchData[c]=false;
    var TableRow = Ti.UI.createTableViewRow({height:50,title:'Row '+(c+1)});
    var BasicSwitch = Titanium.UI.createSwitch({
    value:false,
    title:"Student"+c+":" + false,
    style:Titanium.UI.Android.SWITCH_STYLE_CHECKBOX,
    row:c
        });
    BasicSwitch.addEventListener('change', function(e) {
        switchData[this.row]= e.value;
        this.title = "Student"+this.row+":" + e.value;
    });
    TableRow.add(BasicSwitch);
    data.push(TableRow);
}

var tableview = Titanium.UI.createTableView({
    data:data,
    top:0
});

viewMain.add(tableview);

var ButtonOk = Titanium.UI.createButton({
    title:'OK',
    width:100,
    height:40,
    top:260
});
ButtonOk.addEventListener('click',function(e){

    alert('RowsData:'+'Student1:'+switchData[0]+'Student2:'+switchData[1]+'Student3:'+switchData[2]+'Student4:'+switchData[3]+'Student5:'+switchData[4]);
})
winHome.add(ButtonOk);
