tableView addEventListener

Hello

How I can submit an event to each row in a tableView?, To be more specific a "addEventListener", this is my code.

var table = Ti.UI.createTableView({
      top:0,
    });

//seleccion de la base de datos

var db = Ti.Database.install('/concesionarios.db','concesionarios');
var sql = db.execute("SELECT * FROM ciudades");


var datos = [];
var nombre = [];
while(sql.rowCount > 0 && sql.isValidRow() ) {
    nombre = sql.fieldByName('nombre');
    id = sql.fieldByName('id');

    datos.push({
        title:nombre,
        color:'#000',
        height:30,
        hasChild:false,
        id:id
    });
    sql.next();
}

table.setData(datos);
sql.close();
db.close();
