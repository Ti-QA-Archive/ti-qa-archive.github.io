Show users location with a bullet

Hello

I have created a mapview, and successfully reset it to the users location after GPS callback is finished. However, the users location does not appear on the map as a bullet, as per standard google maps. userLocation is set to true, which I thought would create the bullet. Here are the two code snippets in play:

var map = Titanium.Map.createView({
    mapType: Titanium.Map.STANDARD_TYPE,
    animate:true,
    regionFit:true,
        userLocation: true
});
app.geolocation.setMapRegion(map);

Titanium.Geolocation.purpose = "Recieve User Location";
Titanium.Geolocation.accuracy = Titanium.Geolocation.ACCURACY_BEST;
Titanium.Geolocation.distanceFilter = 10;

app.geolocation = {
    setMapRegion : function(map) {
        Titanium.Geolocation.getCurrentPosition(function(e) {
               if (e.error) {
                alert('HFL cannot get your current location');
                return;
            };

            var region = {
                latitude: e.coords.latitude,
                longitude: e.coords.longitude,
                animate:true,
                latitudeDelta:0.007,
                longitudeDelta:0.007
            };
            map.setLocation(region);
            });    
    }
};

I have tried setting userLocation parameter after the callback, but that doesn't seem to help either. Help or code samples would be much appreciated! Thank you :)
