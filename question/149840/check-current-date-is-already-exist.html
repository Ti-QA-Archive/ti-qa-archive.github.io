      var date = new Date();
    var d  = date.getDate();
    var day = (d < 10) ? '0' + d : d;
    var m = date.getMonth() + 1;
    var month = (m < 10) ? '0' + m : m;
    var yy = date.getYear();
    var year = (yy < 1000) ? yy + 1900 : yy;

    var curdate = (day + "/" + month + "/" + year);

    var db = Ti.Database.open("test7"); 
    var rs = db.execute('SELECT * FROM home_data WHERE date = ?', curdate);

    if(rs.getRowCount() == 0){
        Ti.API.info('no data');
    }
    else
    {
        Ti.API.info('data ok');
    }
