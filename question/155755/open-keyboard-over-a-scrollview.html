  • Hi

    Assuming this is for Android then set the property softKeyboardOnFocus to Ti.UI.Android.SOFT_INPUT_ADJUST_PAN on your scrollView.

    This should allow the keyboard to work with the scrollView as you wanted.

    • window :

      var win4 = Titanium.UI.createWindow({  
    backgroundColor:'white',
    softKeyboardOnFocus : Ti.UI.Android.SOFT_INPUT_ADJUST_PAN
});

      Georges :

      viewResultat[0] = Ti.UI.createScrollView({
            backgroundColor : "white",
            contentHeight : 'auto',
            scrollType:'vertical',
            height : longueurDevice - (longueurDevice/6), ///longueurDevice is the device's height
            top : 0,
            softKeyboardOnFocus : Ti.UI.Android.SOFT_INPUT_ADJUST_PAN,
            showHorizontalScrollIndicator : false,
            horizontalBounce: false,
            showVerticalScrollIndicator:true,
            verticalBounce:true
});

      Simon :

      viewResultat[numeroRecherche] = Ti.UI.createScrollView({
            backgroundColor : "white",
            contentHeight : 'auto',
            height : longueurDevice - (longueurDevice/3),
            top : longueurDevice/5,
            softKeyboardOnFocus : Ti.UI.Android.SOFT_INPUT_ADJUST_PAN,
            scrollType:'vertical',
            showHorizontalScrollIndicator : false,
            horizontalBounce: false,
            showVerticalScrollIndicator:true,
            verticalBounce:true
});

      Then :

      win4.add(viewResultat[0]);
