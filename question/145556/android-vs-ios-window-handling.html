function CallBack_onSuccess(arg) {

    var data = JSON.parse(arg);

    if(data.articles!=undefined)
    {

        if (Ti.Platform.osname != 'android' ) {
            var newsView = require('/src/views/iPhone/tab3');
        }
        else if (Ti.Platform.osname == 'android')
        {
            var newsView = require('/src/views/Android/tab3');
        }
        var mainView = newsView.createTab3View({
            data:arg,
        })
        if (Ti.Platform.osname != 'android' ) {
            loadMessage.hide();
            win.add(mainView);
            win.open();
        }
        else if (Ti.Platform.osname == 'android')
        {
            winEvents.add(mainView);
            winEvents.open();
        }

    }
    else
    {
        win.backgroundImage='/images/noInternet.png';
        if (Ti.Platform.osname != 'android' ) {
            loadMessage.hide();
        }
    }

}

function CallBack_onError(arg) {
    win.backgroundImage='/images/noInternet.png';

        loadMessage.hide();
