    title : 'A 100'  
}, {
    title : 'B 200'
}];
var tableView = Titanium.UI.createTableView({
    search : search,
    filterCaseInsensitive : false,
    filterAttribute : "title",
    data : data
});

win.add(tableView);

    • @sabari. below example is not proper solution for searching with custom row but it is temporary solution and it's work.

      var win = Ti.UI.createWindow();
win.open();

var search = Titanium.UI.createSearchBar({
    showCancel : false,
    hintText : 'Search Customer'

    e.value ;// search string as user types

var Customdata = [{
    title : 'A',
    call : '01'
}, {
    title : 'B',
    call : '02'
}, {
    title : 'a',
    call : '03'
}, {
    title : 'b',
    call : '04'

var data  = [];


for (var i =0; i <  Customdata.length ; i++) {

    var row = Ti.UI.createTableViewRow({
        height : 45,
        backgroundColor : '#FFF',
        id : i,
        title : Customdata[i].title,
        color :"transparent",
        selectionStyle : 0   
    });

    var lbl = Ti.UI.createLabel({
        text : Customdata[i].title,
        font : {
            fontSize : 16,
            fontWeight : 'bold'
        },

        width : 270,
        touchEnabled : false,
        color : '#5a5a5a',
    });
    row.add(lbl);

    data.push(row);

tableView.data = data;
tableView.addEventListener('click', selectRow);

function selectRow(e) {
    var data = Customdata[e.index].call;
    Ti.API.info('>>> Selected Row Data :: ' + data);

}
win.add(tableView);

  • Hi

    You have not provided any test data to review that your filter matches, so you need to make sure the table data rows each have a property title otherwise the option you set will fail.

    If you add a SearchBar directly to the TableView or ListView then the searchbar will filter the items/rows automatically. If you have 100 items then entering a a value into the SearchBar will reduce those 100 items to those that match the search value to the items in the TableView or ListView.

    Filters are used when your list already has all the items it will ever have for this session and the user needs to filter that list down quickly. Removing the value from the SearchBar shows all previous results.

    To confirm your use of the searchBar is in filter mode - it will only reduce the amount of rows based on what you type. No event listeners are required.
