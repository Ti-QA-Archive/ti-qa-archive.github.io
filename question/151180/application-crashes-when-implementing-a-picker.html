var picker = Ti.UI.createPicker();

var data = [];

// this should work ...
data[0] = Ti.UI.createPickerRow({
    title : 'Bananas',
    my_item : 'b'
});

//data[0]=Ti.UI.createPickerRow({title:1,my_item:'b'});
