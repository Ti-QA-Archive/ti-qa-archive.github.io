Multiple Filters on TableView

Hello,

I have a custom tableview. I am using a filter on my tableViewRow like this:

var row = Ti.UI.createTableViewRow({
height:"auto",
hasChild:true,
searchFilter:data.searchResult[i].name
})

Obviously this will filter the results based on their name. However, I want to be able to filter it based on more criteria as returned from my server (i.e. name, address, email, telephone, etc).

How can I do this?

  • Try this sample code. It does exactly what you need

    // Function to dump data
function dumpData() {
    var data = [
        {
            title: 'fb',
            name:  'facebook',
            email: 'abc@fb.com'
        },
        {
            title: 'google',
            name:  'google plus',
            email: 'xyz@google.com'
        },
        {
            title: 'twitter',
            name:  'twitter',
            email: 'foo@tw.com'
        }
    ];
    return data;
}

// You can customize to whatever search algorithm you want
function mysearch(value, data) {
    var searchResults = [];
    for(var i = 0, len = data.length; i < len; i ++) {
        if (data[i].title.indexOf(value) != -1 ||
            data[i].name.indexOf(value) != -1 ||
            data[i].email.indexOf(value) != -1) 
        {
            searchResults.push(data[i]);    
        }
    }
    return searchResults;
}

var win = Ti.UI.createWindow({backgroundColor: 'white'});
// Search bar attached to window
var searchbar = Ti.UI.createSearchBar({ top: 0, height: 43, showCancel: false });
win.add(searchbar);

var data = dumpData();
var tbl = Ti.UI.createTableView({
    data: data,
    top: searchbar.height
});

win.add(tbl);
win.open();

// This will do the trick :)
searchbar.addEventListener('change', function(e){
    tbl.setData([]);

    var searchResults = mysearch(e.value, data);

    if (searchResults.length == 0) {// "No Result"
        tbl.setData([
            {title: ''},
            {title: ''},
            {title: 'No Result'},
        ]);    
    } else {
        tbl.setData( searchResults );
    }
});

searchbar.addEventListener('return', function(e){
    searchbar.blur();    
});

tbl.addEventListener('click', function(e) {
    searchbar.blur();
});

    The idea is building custom searchbar attached to window instead of tableview. All you need do is modifying mysearch() to archive your need.
