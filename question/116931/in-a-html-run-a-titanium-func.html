    i had to remove all of the images since I dont have them, but the code works for me?

    Titanium.UI.setBackgroundColor('#000000');

var window = Titanium.UI.createWindow({
    fullscreen: true,
    backgroundColor: "#000",
    navBarHidden: true
});
var webview = Titanium.UI.createWebView({url:'/html/index.html'});
window.add(webview);
window.open();


Ti.App.addEventListener('openURL', function(e){
    Ti.Platform.openURL(e.url);
});

    the html file

    <!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Transitional//EN" "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd">
<html xmlns="http://www.w3.org/1999/xhtml">
<head>
<meta http-equiv="Content-Type" content="text/html; charset=utf-8" />

<body>
<a onclick="Ti.App.fireEvent('openURL', { url: 'http://www.google.com'});" >Matriculate Ya</a>
</body>
</html>
