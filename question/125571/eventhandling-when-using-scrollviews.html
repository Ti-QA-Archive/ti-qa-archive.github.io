EventHandling when using scrollViews

hi,

I am trying to receive a doubletab event on a modal-window.

var win = Ti.UI.createWindow({
            height: "100%",
            layout: 'vertical',
            title: _data.headline,
            width: "100%"
        });

        var sV = Ti.UI.createScrollView({
            bottom: 0,
            canCancelEvents: false,
            contentHeight: 'auto',
            contentWidth: '100%',
            height: 618,
//            height: 'auto',
            layout: 'vertical',
            top: 150,
            showVerticalScrollIndicator: true,
            width: "100%"
        })

        var entryHeadline = Ti.UI.createLabel({
            height: 'auto',
            left: 5,
            text: _data.headline,
            top: 0,
//            touchEnabled: false,
            width: 'auto'
        });
        var entryTeaser = Ti.UI.createLabel({
            height: 'auto',
            left: 5,
            text: _data.teaser,
            top: 5,
//            touchEnabled: false,
            width: 'auto'
        });
        var entryImg = Ti.UI.createImageView({
            height: 'auto',
            left: 5,
            image: _data.image,
            top: 5,
//            touchEnabled: false,
            width: 'auto'
        });
        var entryFullText = Ti.UI.createLabel({
            height: 'auto',
            left: 5,
            text: _data.content,
            top: 5,
//            touchEnabled: false,
            width: 'auto'
        });
        sV.add(entryHeadline);
        sV.add(entryImg);
        sV.add(entryTeaser);
        sV.add(entryFullText);
        win.add(sV);
        win.addEventListener('doubletap',function(e){
            win.close({animated: true});
        });

        return win;
    };

But when i tab into a empty area (only scrollView) my eventListener is not triggered. Even using canCancelEvents: false makes no difference.

Any ideas how i can forward my to the window?

  • If I'm understanding correctly, you could pass the double tap events through your scrollview to the window by setting

    sV.toucheEnabled = false;

    But this will likely disable scrolling. You could always capture the double tap events at the scrollview and then fire another event to your window.

    sV.addEventListener('doubletap', function(e) {
    win.fireEvent('doubletap', e);
});
