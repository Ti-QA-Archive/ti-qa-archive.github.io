    AH, I don't think I fully understood your question til right now. Her eis a working example of long Label, in a ScrollView, which is in turn in a ScrollableView. The ScrollView is vertically scrollable and the ScrollableView is still able to be swiped horizontally to change views.

    Ti.UI.backgroundColor = '#000';

var win = Ti.UI.createWindow();
var view1 = Ti.UI.createScrollView({
    backgroundColor:'#ff0000',
    contentHeight:'auto',
    contentWidth:'auto'
});
var label = Ti.UI.createLabel({
    text:'Here is some really big text to show that the containing ScrollView will play nice with the ScrollableView that contains it.',
    height:'auto',
    width:'100%',
    font: {
        fontSize:64    
    }
});

var view2 = Ti.UI.createView({
    backgroundColor:'#00ff00'
});
var view3 = Ti.UI.createView({
    backgroundColor:'#0000ff'
});
var scrollable = Ti.UI.createScrollableView({
    views:[view1, view2, view3]
});

view1.add(label);
win.add(scrollable);

win.open();

    var win = Ti.UI.createWindow();
var label = Ti.UI.createLabel({
    color:'#ffffff',
    height:'auto',
    width:'auto',
    text:'this is some really long text. this is some really long text. this is some really long text. this is some really long text. '
});
var scroll = Ti.UI.createScrollView({
    contentWidth:'auto',
    contentHeight:'auto',
    showVerticalScrollIndicator:false,
    showHorizontalScrollIndicator:true
});

scroll.add(label);
win.add(scroll);
win.open();

    • This is a limitation of the underlying platforms. Android best practices mention that nesting scrollable components is a bad idea. You're not even really supposed to put a scrollable tableview in a scrollview. There's no real way for the device to know which scrollable view should be receiving the events, as you are experiencing.
