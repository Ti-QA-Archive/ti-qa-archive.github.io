    Hi Marvin,

    I don't know you like this code or not but try it.

    var win = Ti.UI.currentWindow;
var doc = '<item>[129236] Description of the product</item>';
var it = Titanium.XML.parseString(doc);
var deals = it.getElementsByTagName('item');
var t = deals.item(0).text;
var fI = t.indexOf("[");
var sI = t.indexOf("]");

var n = t.substr(fI+1, sI-1);
alert(n);
