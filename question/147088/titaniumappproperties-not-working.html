Titanium.App.Properties not working

i have some data (objects) which i want to store in Titanium.App.Properties

var ds= require("/ui/common/ds.common.utitlities");
    var data=ds.database()
    var dbconn=data.open('devdb') //database connection
    var rows=data.search(dbconn,"select is_module from meta where module='events'")

i want to do store database connection in Titanium.App.Properties like

Ti.App.Properties.setObject('dbconn', data.open('devdb'))
var dbconn = Ti.App.Properties.getObject('dbconn')
var rows=data.search(dbconn,"select is_module from meta where module='events'")

but it is not working. what may be wrong here ?

###ds.common.utitlities



exports.database=function(){
    return{
        alert:function(string){
            //alert(string)
        },
        install:function (file,name){

                     try{
                         var dbObj=Ti.Database.install(file,name);
                         Ti.API.log("Database: "+name+" is created") 
                         return dbObj;                         
                     }
                     catch(err){
                         alert("db connection error: "+err)
                     }

                     },
        open:function(name){
            var db = Ti.Database.open(name);

            return db

        },
        search:function(dbObj,statement){
            //alert('search')
            try{
                var row=dbObj.execute(statement)
                return row

            }catch(e){
                alert("search error:"+e)

            }
        }

  • the setObject method is used to save complex javascript objects that can be serialised as JSON. The open method returns an instance of the database proxy. This is not an object so cannot be saved in properties
