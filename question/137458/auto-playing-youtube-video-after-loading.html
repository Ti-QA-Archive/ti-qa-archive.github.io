  • This is the code Im using…

    // Get youtube url
    var youtubeURL = win.dataToPass;

    if (osName == 'android') //Check running in android os
    {
    Ti.Platform.openURL(youtubeURL);
    }
    else
    {
    var webview = Titanium.UI.createWebView({url:youtubeURL, fullscreen:true});
    win.add(webview);
    }
