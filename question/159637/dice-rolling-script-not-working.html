  • function rollDice() {
    var die1 = Math.floor(Math.random() * 6) + 1;
    var die2 = Math.floor(Math.random() * 6) + 1;
    var diceTotal = die1 + die2;
    d1.value = die1;
    d2.value = die2;
};

var w = Ti.UI.createWindow();

var b = Ti.UI.createButton({
    top : 5,
    width : 90,
    height : 40,
    title : 'Roll!'
});

b.addEventListener('click', function() {
    rollDice();
});

var d1 = Titanium.UI.createTextField({
    color : '#336699',
    height : 35,
    top : 50,
    right : 50,
    width : 250,
    borderStyle : Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
    value : '0',
    fontSize : 20,
});

var d2 = Titanium.UI.createTextField({
    color : '#336699',
    height : 35,
    top : 90,
    right : 50,
    width : 250,
    borderStyle : Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
    value : '0',
    fontSize : 20,
});

w.add(b);
w.add(d1);
w.add(d2);

w.open();
