Modal Window open another Modal Window, how?

Hi,

I'm trying to open a modal window by another modal window. But the iOS really can't do this. Someone how know to do this?

I need to open a modal window by another modal window. See what I'm trying here:

var win = Ti.UI.createWindow({
    backgroundColor:"green"
});
win.open();

var winBase1 = Ti.UI.createWindow({
    zIndex:10,
    navBarHidden:true,
    backgroundColor:"yellow",
    modal:true
});
var nav1 = Ti.UI.iPhone.createNavigationGroup({
    window:Ti.UI.createWindow({
        backgroundColor:"yellow"
    })
});
winBase1.add(nav1);
winBase1.open();

setTimeout(function() {
    var winBase2 = Ti.UI.createWindow({
        backgroundColor:"blue",
        zIndex:100,
        modal:true
    });
    var nav2 = Ti.UI.iPhone.createNavigationGroup({
        window:Ti.UI.createWindow({
            backgroundColor:"blue"
        })
    });
    winBase2.add(nav2);
    winBase2.open();
}, 1000);

When I try to do this, I have this error message:

2011-12-23 14:58:02.336 Modals[30420:17903] * Terminating app due to uncaught exception 'NSInvalidArgumentException', reason: 'Application tried to present modally an active controller <UIViewController: 0xa3ab760>.'**
