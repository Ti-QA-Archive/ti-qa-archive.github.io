  • Here is the method:

    var searchbyjob = Titanium.UI.createTextField({

    font : {
        fontSize : 20,
        fontFamily : 'Helvetica Neue'
    },
    textAlign : 'center',
    top : '60dp',
    width : '80%',
    left : '10%',
    right : '10%',
    hight : 'auto',
    hintText : 'Search By Job',
    keyboardType : Titanium.UI.KEYBOARD_DEFAULT

});

    And this how you would get value

    var labvalue=searchbyjob.value
