checkboxes - how to select only one and not both

I have the following code for a yes and no button. When yes is selected it turns on. When I select no, yes stays on. Is it possible to only select one?

var checkyes = Titanium.UI.createButton({
    image: '../images/yes-off.png',
    checked: false,
    width: 81,
    height: 41,
    left: 10,
    bottom: 160
});
checkyes.addEventListener('click', function(){
      if (checkyes.checked){
           checkyes.checked = false;
           checkyes.image = '../images/yes-off.png';
      } else {
           checkyes.checked = true;
           checkyes.image = '../images/yes-on.png';
      };
});
win.add(checkyes);


var checkno = Titanium.UI.createButton({
    image: '../images/no-off.png',
    checked: false,
    width: 80,
    height: 41,
    left: 92,
    bottom: 160
});
checkno.addEventListener('click', function(){
      if (checkno.checked){
           checkno.checked = false;
           checkno.image = '../images/no-off.png';
      } else {
           checkno.checked = true;
           checkno.image = '../images/no-on.png';
      };
});
win.add(checkno);

    Try this. You weren't setting the state of the other button. 

    
var checkyes = Titanium.UI.createButton({
    image: '../images/yes-off.png',
    checked: false,
    width: 81,
    height: 41,
    left: 10,
    bottom: 160
});

var checkno = Titanium.UI.createButton({
    image: '../images/no-off.png',
    checked: false,
    width: 80,
    height: 41,
    left: 92,
    bottom: 160
});

win.add(checkyes);
win.add(checkno);

checkyes.addEventListener('click', function(){
      if (checkyes.checked){
           checkyes.checked = false;
           checkyes.image = '../images/yes-off.png';

           checkno.checked = true; // you need to change the no button
           checkno.image = '../images/no-on.png';

      } else {
           checkyes.checked = true;
           checkyes.image = '../images/yes-on.png';

           checkno.checked = false; // you need to change the no button
           checkno.image = '../images/no-off.png';


      };
});

checkno.addEventListener('click', function(){
      if (checkno.checked){
           checkno.checked = false;
           checkno.image = '../images/no-off.png';

           checkyes.checked = true; // you need to change the yes button
           checkyes.image = '../images/yes-on.png';

      } else {
           checkno.checked = true;
           checkno.image = '../images/no-on.png';

           checkyes.checked = false; // you need to change the yes button
           checkyes.image = '../images/yes-off.png';

      };
});

    You might want to begin with one of the buttons on, for it to be a true 'toggle' / radio switch though.

    There no doubt are probably much 'cleaner' ways of doing this with some neat logic code, but this is how to make it work using the method you are using ;)

    hope this helps
