    hi Ali,

    var row = Ti.UI.createTableViewRow({
    newsId : <your db id>
});

tableView.addEventListener('click',function(e){
    var winB = Ti.UI.createWindow({
        url:<url>,
        newsId : e.row.newsId
    });
});
