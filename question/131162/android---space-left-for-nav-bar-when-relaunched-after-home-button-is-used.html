  • Hi,

    I've set up this ugly workaround : opening a window and closing it right after, which has the effect to reset the main window to the top : 

        if (Ti.Platform.name === "android") {
        mainWindow.addEventListener("focus", function() {
            Titanium.API.info("focus called");            
            var win = Titanium.UI.createWindow({
                title: '',
                navBarHidden: true,
                fullscreen: true,
                exitOnClose: false,
                backgroundColor : 'transparent'
            });
            win.orientationModes = [Ti.UI.PORTRAIT];
            win.addEventListener('open', function () {
                win.close({animated:false});
            });
            win.open({animated:false});
        });
    }

    It results in a black flash of moving window when resuming.

    If anyone has another idea or clue, please let me know.
