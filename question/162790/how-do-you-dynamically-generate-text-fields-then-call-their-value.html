    Hi,

    please check the below code. I have modified your code for testing purpose. I hope, it will help you

    app.js

    var win = Titanium.UI.createWindow({
title:'TEST',
backgroundColor:'#fff' ,
navBarHidden:true,
layout:'vertical'
});

Ti.App.Properties.setInt('numberTeams',3);
var aButton = Ti.UI.createButton({
    title : 'Click',
    height : 'auto',
    width : 'auto',
});

aButton.addEventListener('click', function() {
    Titanium.UI.createWindow({
                    url :'/new.js',
                    backgroundColor:'white',
                    modal:true,
                    layout:'vertical'
                }).open();
});

win.add(aButton);

win.open();

    new.js

    var windows=Ti.UI.currentWindow;
var text = [];
var view=Ti.UI.createView({
    top:0,
    left:0,
    height:Ti.UI.SIZE,
    width:Ti.UI.FILL,
    //layout:'vertical'
});

windows.add(view);

var TEST = Ti.UI.createButton({
    title : 'TEST',
    top:20,
    height : 'auto',
    width : 'auto'
});

TEST.addEventListener('click', function() {
    alert(text[0].value);
});


windows.addEventListener('open', function(e){
   //alert('humm');
   var top=10;
   var x = Ti.App.Properties.getInt('numberTeams');
   for(var i = 0; i < x ; i++){
       var t = i+1;
       var label = 'label_name' + t;
       label = Ti.UI.createLabel({
           text: 'Team '+t+' :', 
           top:10+top,
           left:10,
           width: '30%',
           height: 70,
        });
        text[i] = Ti.UI.createTextField({
           width:'50%',
           top:20+top,
           right:10,
           height: 70,
           hintText:'Team '+t,
        });
        view.add(label);
        view.add(text[i]);  
        top=top+70;
   }

windows.add(TEST);
});
