    You should create row in for loop

    // row = Ti.UI.createTableViewRow({
    // height : '70'
// });

for( pos = 0; pos < json.length; pos++) {
    var row = Ti.UI.createTableViewRow({
        height : '70'
    });

    row.title = json[pos].navn;
    row.hasChild = true;

    dataArray.push(row);
};
table = Titanium.UI.createTableView({
    data : dataArray
});
// table.appendRow(row);
win.add(table);
