    Hi Paul,

    As per your code you are change the value of array not the table view row.. so you must need to set title of row on click event of dialog e.g i set custom property to dialog and set clicked row reference to it.

    table.addEventListener('click', function(e) {
        if (e.rowData.id == 1) {
         var textfield = Ti.UI.createTextField({
                keyboardType:Titanium.UI.KEYBOARD_NUMBER_PAD,
                value : data[1].title
            });

            var dialog = Ti.UI.createAlertDialog({
                title: 'No. of Passengers',
                androidView: textfield,
                buttonNames: ['OK', 'Cancel'],
        tblRow : e.source
            });

        dialog.addEventListener('click', function(e){
                    if (e.index == 0) {                         
                            //data[1].title = textfield.value;
                e.source.tblRow.title = textfield.value; 

                            Ti.API.info('Textfield Value: '+textfield.value);
                            Ti.API.info('Row Title: '+data[1].title);
                        }               
                }); 
                dialog.show();
    }
    ...

});

    hope it's works for you.. if you have any doubt the let me know…
