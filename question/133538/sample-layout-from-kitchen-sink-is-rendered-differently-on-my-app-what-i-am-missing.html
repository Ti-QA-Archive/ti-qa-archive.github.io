  • This is the code of my app:

    var win = Ti.UI.createWindow({
    title: 'Test',
    backgroundColor: '#13386c',
    exitOnClose: true
});

var firstNameLabel = Ti.UI.createLabel({
    text:'First Name',
    top:10,
    left:30,
    width:100,
    height:'auto'
});
win.add(firstNameLabel);

var firstNameField = Ti.UI.createTextField({
    hintText:'enter first name',
    top:35,
    left:30,
    width:250,
    height:35,
    borderStyle:Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});
win.add(firstNameField);

var lastNameLabel = Ti.UI.createLabel({
    text:'Last Name',
    top:75,
    left:30,
    width:100,
    height:'auto'
});
win.add(lastNameLabel);

var lastNameField = Ti.UI.createTextField({
    hintText:'enter last name',
    top:100,
    left:30,
    width:250,
    height:35,
    borderStyle:Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});
win.add(lastNameField);    

var saveButton = Ti.UI.createButton({
    title:'Save',
    top:170,
    left:30,
    width:250,
    height:30
});
win.add(saveButton);            

var messageView = Titanium.UI.createView({
    bottom:10,
    backgroundColor:'#111',
    height:40,
    width:270,
    borderRadius:10
});

var messageLabel = Titanium.UI.createLabel({
    color:'#fff',
    text:'Register for a free toaster!',
    height:'auto',
    width:'auto',
    textAlign:'center'
});
messageView.add(messageLabel);

win.add(messageView);
win.open();
