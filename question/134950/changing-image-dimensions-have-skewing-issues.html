Changing image dimensions have skewing issues

Titanium Mobile SDK 1.8.2
iPhone SDK 4.3

Hi,

I am animating a photo via eventlistener but the result skews the maximized photo. Am not sure if this is a good practice to detect size dimensions but here goes my code:

photo01.addEventListener('click', function() {
    var pHeight = photo01.height;
    if (pHeight == 420) {
        photo01.animate({
            height: 65,
            width: 100,
            top: 6,
            left: 10
        });
        photo01.setHeight(65);
    } else {
        photo01.animate({
            width: 320,
            height: 420,
            top: 0,
            left: 0,
            backgroundColor: 'black'
        })
        photo01.setHeight(420); // if i remove this it doens't skew the photo
    }

});
