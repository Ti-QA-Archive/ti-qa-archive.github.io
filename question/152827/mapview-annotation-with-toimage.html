Code of Blog

var addImageWithLocation = function(mapView, imgUrl, lat, lng) {
  var imgBlob = Titanium.UI.createImageView({
    image: imgUrl,
    borderColor:'#fff',
    borderWidth:12,
    hires: true
  }).toImage();

  var thumbnailImageView = Ti.UI.createImageView({
    image: imgBlob,
    width:64,
    height:64,
    borderColor:'#999',
    borderWidth:1,
    hires: true
  });

  var annotation = Titanium.Map.createAnnotation({
    latitude: lat,
    longitude: lng,
    title:'',
    animate:true,
    image: thumbnailImageView.toImage()
  });
  mapView.addAnnotation(annotation);
};

  • Hi

    Try removing…

    pincolor: Ti.Map.ANNOTATION_RED,

    …and changing this.

    backgroundImage:'image/lenolima.jpg' ,
// to
image:'image/lenolima.jpg' ,

    Also if you are using TiSDK 3.1 and Android you MUST start using the Modules.Map version 2 of maps that replaces the now deprecated version 1 of maps.

    • Android Platform Notes
      This module provides native map support using Google Maps API v1. As of March 3, 2013, Google is no longer issuing Google Maps API v1 keys and this module can no longer be used to support native maps on Android unless you have these keys.
