Will not fire HTML in window.js

I keep running into this issue of not being able to fire a local HTML file in my app from a window.js file. The HTML file is nothing special just a basic contact us type of code. My question is this…
Do I need to call the HTML file something special? Do I need to have url before the sports.html? I'm confused at this point….

Here is sample of the full code so you can see the logic and arguments. Thanks for your advice and help in advance!

//new button
var openWebpagemenu = Titanium.UI.createButton({
    title:"Contact", 
    backgroundImage: "images/contactbutton.png",
    width:"100%",    
    height:"15%",
    top:"70%",
    id: "Contact", 

    color: '#000000'                                    
});

openWebpagemenu.addEventListener("click", function(e){

// Create a window and add a webview to it that opens your url:
var webviewWindow = Titanium.UI.createWindow({title: 'Contact', fullscreen: false});             
var myWebView = Ti.UI.createWindow('/sports.html'); 
webviewWindow.add(myWebView); 

// Will need a button to close the new window:
var closeButton = Titanium.UI.createButton({title:'Close', style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN});
webviewWindow.setLeftNavButton(closeButton);

closeButton.addEventListener('click',function(evt){
    webviewWindow.close();
});

webviewWindow.open({modal:true});
webviewWindow.open();  
})

  • Accepted Answer

    Looks like it is not finding the HTML file. Try with the absolute path. Make use of Titanium.Filesystem.resourcesDirectory 

    var myWebView = Ti.UI.createWebView({ url: Titanium.Filesystem.resourcesDirectory +'sports.html' });

    Obviously this assumes that you have placed the HTML in same dir as app.js, so place the file there or adjust the path

    Creating a new window using URL element is not really what you want in this case, try webview as shown

  • This may be a minor miss with frustrating results but you are using createWindow for the webView, so even though your variable is called myWebView its really a window. Here the code.. I would try this with the absolute path of the html file.. 

    
        var webview = Titanium.UI.createWebView({
            url : "/sports.html"  //preferably absolute path of the html - Use Titanium.Filesystem.resourcesDirectory if you have to
        });
        var myWebViewWindow = Titanium.UI.createWindow({
            title : "My Window Title"
        });
        myWebViewWindow.add(webview);

        myWebViewWindow.open({
            modal : true    //or false, based on what you are trying to do
        });

        var closeButton = Titanium.UI.createButton({
            title : 'Close'
        });
        myWebViewWindow.rightNavButton = closeButton;
        closeButton.addEventListener('click', function(e) {
            myWebViewWindow.close();
        });
