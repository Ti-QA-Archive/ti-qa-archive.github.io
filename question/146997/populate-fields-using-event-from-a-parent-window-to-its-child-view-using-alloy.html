Populate fields using event from a parent window to it's child view using Alloy

I am using Alloy and I have a Window - "subCategoryWin" which pulls in a child view using require tag, the child here being "toolbarBottom" In the parent window, I'd like to call a function on an event such as click and populate the TextField inside the toolbarBottom. What is currently happening is it's adding the number of items to the Textfield, however it's not showing that number until I go to a new window. Any suggestions on how to make this work?

subcategory.xml

<Alloy>
        <Window id="subCategoryWin" title="Menu Categories" class="container">
            <Require type="view" src="header" id="headerSection"/>
            <Require type="view" src="toolbarBottom" id="toolbarSection"/> 
        </Window>

</Alloy>

subcategory.js


 Table.addEventListener('click', function(e) {

    exports.updateCount = function() {
          //require dependencies
            updateCount = require('toolbarBottom').updateCount;
            //attempt to fire function located in toolbar
            updateCartCount();

    };

});

toobarBottom.xml

<Alloy>
    <View id="messageToolbar">
        <Toolbar platform="ios" id="toolbar"  class="bottom">
                    <Items>
                        <FlexSpace/>
                        <FlexSpace/>
                        <TextField id="itemCount" value=""></TextField>
                        <Label id="separator"/>
                        <Button id="Btn">Ok</Button>
                    </Items>    
           </Toolbar>
    </View>
</Alloy>

toobarBottom.js

var updateCount = function(){
    var items=Alloy.CFG.totalItems; 
    for (p in items) {
        $.itemCount.value = items.length;
        Alloy.CFG.totalItems =  items.length;
    }   

};
