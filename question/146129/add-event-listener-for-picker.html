    • ok,let me explain first,

      change event is only call when user plays with picker so first time we have to check which row is selected to do that just test below code

      var column = Ti.UI.createPickerColumn();

column.addRow(Ti.UI.createPickerRow({
    title : 'Bananas'
}));
column.addRow(Ti.UI.createPickerRow({
    title : 'Strawberries'
}));
column.addRow(Ti.UI.createPickerRow({
    title : 'Mangos'
}));
column.addRow(Ti.UI.createPickerRow({
    title : 'Grapes'
}));

var picker = Ti.UI.createPicker({
    top : 50,
    columns : [column]
});

picker.selectionIndicator = true;

picker.addEventListener('change', function(e) {
    alert(e.row.title);
})

win.add(picker);
setTimeout(function(e) {

    alert(picker.getSelectedRow(0).title);
}, 200);
