Update tab-title

Hi,

iam trying to update a tab-title. First i make an xhr call, and in its
onload-callback i want a tab-title change from "Login" to "my Profile".

File: login.js:


var loginReq = Ti.Network.createHTTPClient();

loginReq.onload = function(){
    var json = this.responseText;
    var response = JSON.parse(json);

    if(response.logged == true){
        //alert("Welcome "+response.name+".Your Email is "+response.email);
        username.blur();
        password.blur();
        Ti.App.fireEvent('grantEntrance',{
            name: response.name,
            email: response.email
        });


        Titanium.App.Properties.setString("email",response.email);
        Titanium.App.Properties.setString("mitglied_id",response.mitglied_id);


        win.remove(username);
        win.remove(password);
        win.remove(loginBtn);

        var theLabel = Ti.UI.createLabel({
            text: 'Hallo '+response.name
        });
        win.title = 'Hallo '+response.name;
        win.add(theLabel);
    }else{
        alert(response.message);
    }
};

But the tabs are defined in app.js, so i tried it with Ti.App.fireEvent():
in app.js:

File: app.js:

Ti.App.addEventListener('grantEntrance',function(event){

    tabLoginFenster.setTitle('mein Profil');
    alert('Logged in');


});

tabLoginFenster ist the var for the Tab "Login", defined in app.js.
But setTitle does nothing.

Any idea ?
