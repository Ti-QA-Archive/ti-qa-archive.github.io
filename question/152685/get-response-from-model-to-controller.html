Get response from model to controller

Hi..
I'm a newbie here.. In my project I use Rest api and model to remote login. I send login details to Rest api and send the success response to Model.
Here is my model code


exports.definition = {
    config: {
        "URL": "http://somesite.com/api/login/",
        // "debug": 1, 
        "adapter": {
            "type": "restapi",
            "collection_name": "userModel"
        },
    },

     extendModel : function(Model) {

         _.extend(Model.prototype, {
             parse : function(_resp, loginReq) { 
                 var response = _resp;
                 return  response;

    extendCollection: function(Collection) {        
        _.extend(Collection.prototype, {});
