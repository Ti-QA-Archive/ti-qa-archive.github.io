ScrollView - No Scroll

this.detailContainer = Ti.UI.createScrollView({
        top : 55,
        backgroundColor : 'white',
        contentWidth : 'auto',
        contentHeight : 'auto',
        showVerticalScrollIndicator : true,
        showHorizontalScrollIndicator : true
    });

    var town_image = Ti.UI.createLabel({
        top : 20,
        left : 5
    });

    var town_desc = Ti.UI.createLabel({
        top : 100,
        left : 0
    });

    this.detailContainer.addEventListener('itemSelected', function(e) {
        //Town Data
        town_name.text = e.name;
        town_image.backgroundImage = dir_path.R + 'images/town/places/' + e.img;
        town_image.height = 37;
        town_image.width = 45;
        town_desc.text = e.desc;
    });
    //Add the Top Tray
    this.top_tray = Ti.UI.createLabel({
        top : 0,
        left : 0,
        height : 33,
        width : 320,
        backgroundImage : dir_path.R + "images/town/detail/steepbanks-window-up-top.png"
    });

    //Add Listener for top tray
    var top_tray = this.top_tray;
    var bottom_tray = this.bottom_tray;
    top_tray.addEventListener('click', function(e) {
        top_tray.setVisible(false);
        bottom_tray.setVisible(true);
    });
    //Add Listener for bottom tray
    var top_tray = this.top_tray;
    var bottom_tray = this.bottom_tray;
    bottom_tray.addEventListener('click', function(e) {
        top_tray.setVisible(true);
        bottom_tray.setVisible(false);
    });

    this.detailContainer.add(this.top_tray);
    this.detailContainer.add(welcome_txt);
    this.detailContainer.add(town_name);
    this.detailContainer.add(town_image);
    this.detailContainer.add(town_desc);
    this.detailContainer.add(this.bottom_tray);
    this.win.add(this.detailContainer);

SO in this code I have added a scrollview, and the event listener fills the town_desc with text longer than the scroll view, and I see no scroll in this view.

What am I doing wrong?

  • hi Steven,
    this is happened because you do not set the conetentHeight of the scrollView after setting some content in the controls

    so try this one…

    this.detailContainer.addEventListener('itemSelected', function(e) {
        //Town Data
        town_name.text = e.name;
        town_image.backgroundImage = dir_path.R + 'images/town/places/' + e.img;
        town_image.height = 37;
        town_image.width = 45;
        town_desc.text = e.desc;


this.detailContainer.contentHeight = this.top_tray.toImage().height + welcome_txt.toImage().height + town_name.toImage().height + town_image.toImage().height + town_desc.toImage().height + this.bottom_tray.toImage().height;
});

    good luck
