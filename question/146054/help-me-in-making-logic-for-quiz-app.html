    Hi

    Using your own code structure I have moved things around a bit and created a function to update the labels each time the next button is pressed.

    var quiz_counter = 0; // assume you had this but just in case

var btnnext = Titanium.UI.createButton({
    title : 'Next',
    bottom : '20%',
    width : '50%',
    height : '10%'
});
QuizWin.add(btnnext);
var labquiz = Titanium.UI.createLabel({
    text : '', // leave this blank 
    top : '20%',
    left : '10%',
    color : 'black'
});
QuizWin.add(labquiz);

var laboption1 = Titanium.UI.createLabel({
    text : '', // leave this blank 
    top : '30%',
    left : '10%',
    color : 'black'
});

QuizWin.add(laboption1);

var laboption2 = Titanium.UI.createLabel({
    text : '', // leave this blank 
    top : '40%',
    left : '10%',
    color : 'black'
});
QuizWin.add(laboption2);

var laboption3 = Titanium.UI.createLabel({
    text : '', // leave this blank 
    top : '50%',
    left : '10%',
    color : 'black'
});
QuizWin.add(laboption3);

var laboption4 = Titanium.UI.createLabel({
    text : '', // leave this blank 
    top : '60%',
    left : '10%',
    color : 'black'

});
QuizWin.add(laboption4);

function showQuestionSet() {
    labquiz.setText(quizes[quiz_counter]);
    laboption1.setText(option1[quiz_counter]);
    laboption2.setText(option2[quiz_counter]);
    laboption3.setText(option3[quiz_counter]);
    laboption4.setText(option4[quiz_counter]);
    if (quiz_counter !== (quizes.length - 1)) {
        quiz_counter++;
    }
}

btnnext.addEventListener('click', function (e) {
    showQuestionSet(); // show the next set of questions
});
showQuestionSet(); // show the first set of questions on start-up
