Why can't I restart audio after it has been paused?

I have audio which starts when the Android app starts. The thing is, once the audio is paused, it can not be played again. Whats going on here?
Android API 2.1, HVGA and Titanium (1.2.2) SDK version 1.7.0.

Here are my codes:

var window1 = Titanium.UI.createWindow({  

});

var sound = Titanium.Media.createSound({
    url:'soundtrack.mp3',
    preload:true

});

window1.addEventListener('open',function()  //eventlistener which starts audio
{
    sound.looping = true;
    sound.allowBackground = true;
    sound.play();
});

var tab1 = Titanium.UI.createTab({  
    id: 'tab1',

    window:window1
});

var window2 = Titanium.UI.createWindow({  

});

var button1 = Ti.UI.createButton({
    title:"Play"
});

window2.add(button1);

button1.addEventListener('open',function()
{
    sound.looping = true;
    sound.allowBackground = true;
    sound.play();

});

var button2 = Ti.UI.createButton({
    title:"Pause"
});

window2.add(button2);


button2.addEventListener('click',function()
{
    sound.pause();

});
