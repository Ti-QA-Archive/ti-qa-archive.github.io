Splashscreen and transition

Hi,
when my app go to resume mode (leave the background mode), I have to display a splashscreen.
I use something like this :

function showSplashScreen () {
    var welcomeWindow = Titanium.UI.createWindow({
        backgroundColor:'black'
    });

    var imageView = Titanium.UI.createImageView({
        image:'/iphone/Default-Landscape.png',
        canScale: 'false',
        touchEnabled:false
    });

    welcomeWindow.add(imageView);

    welcomeWindow.orientationModes = [
        Titanium.UI.LANDSCAPE_LEFT,
        Titanium.UI.LANDSCAPE_RIGHT
    ];


    //welcomeWindow.open({fullscreen:true});
    welcomeWindow.open({fullscreen:true,transition:Ti.UI.iPhone.AnimationStyle.CURL_DOWN});

    var timeOut = function(){
       welcomeWindow.close();
    }
    setTimeout (timeOut, 3000);

};

the problem is that with the transition when the splashscreen close, it show a black screen instead of the window.

removing the transition like this work :

welcomeWindow.open({fullscreen:true});

any idea of how to be able to keep the transition ?

Thanks
