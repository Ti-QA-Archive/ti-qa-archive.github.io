    Try this code is working in android also.

        var os = Titanium.Platform.osname;
    var model = Titanium.Platform.model;

    if(model.indexOf('iPod') >= 0) {
        alert("Your device does not support call facility.");
        return;
    } else if(model.indexOf('iPhone') >= 0) {
        var os = Titanium.Platform.osname;
        var model = Titanium.Platform.model;

        if(model.indexOf('iPod') >= 0) {
            alert("Your device does not support call facility.");
            return;
        } else if((model.indexOf('iPhone') >= 0) || os == 'android') {
            var dialog = Ti.UI.createAlertDialog({
                buttonNames : ["Call", "Cancel"],
                title : 'Call',
                cancel : 1,
                message : 'Are you sure you want\n to call ?.'
            });

            dialog.addEventListener('click', function(e) {
                if(e.index == 0) {
                    Ti.Platform.openURL('tel:Yournumber');
                } else {
                }
            });
            dialog.show();

        } else {
            alert('Not supported for this device');
        }

    } else {
        alert('Not supported for this device');
    }
