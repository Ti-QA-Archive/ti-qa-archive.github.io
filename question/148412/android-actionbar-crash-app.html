Android actionBar crash app

Hi, I have a problem with Android and actionBar, as i read in the doc i tried to add the actionBar in my app. Here the code

function ApplicationWindow(navController) {
  var self = Ti.UI.createWindow({
        backgroundColor:K.colors.MAIN_BACKGROUND,
        backgroundImage: K.mainWin.BGIMAGE,
        barImage: Utils.gi(K.mainWin.NAVBARRAW),
        barColor: K.mainWin.BARCOLOR
    });

    //...set some buttons and images...

    var actionBar;
    self.addEventListener('open', function(){
        if(K.isandroid){
            actionBar = self.getActivity().actionBar;
            if (actionBar) {
                actionBar.backgroundImage = K.mainWin.NAVBAR;
                actionBar.title = "New Title";
                actionBar.onHomeIconItemSelected = function() {
                    Ti.API.info("Home icon clicked!");
                };
            }
        }
    });

    return self;
}

but when i try to run the app crash without any message (Unfortunately, app has stopped), looking in the log i get this

  • I ran into same problem, so, 3 things to check :

    1. Make sure you have set API level to 11 or higher in your tiapp.xml.
    2. set navBarHidden="false" (mandatory!, if you have set it to true in your tiapp.xml to make the splash screen look pretty), in your 'self' window instantiation call.
    3. As Tyler said, check the path of image as well once, just to make sure.
