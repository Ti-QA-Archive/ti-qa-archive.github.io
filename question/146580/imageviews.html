    width is real width of image
height is real height of image
scaleWidth is width used to resize image in aspect ratio to the height
function resizeImage(width,height,scaleWidth)
{
    var widthforimage = scaleWidth;
    var getImgwidth = width;
    var getImgHeight = height;

    var newImgWidth = widthforimage;
    var newImgHeight = 0;
    var per=0;
    if (getImgwidth < widthforimage)
    {
        per = (100 * (widthforimage - getImgwidth)) / getImgwidth;
        newImgHeight = ((getImgHeight * per) / 100) + getImgHeight;
    }
    else
    {
        per = (100 * (getImgwidth - widthforimage)) / getImgwidth;
        newImgHeight = getImgHeight - ((getImgHeight * per) / 100);
    }
if(newImgHeight>'your criteria for max height')
newImgHeight='max image height';
}
