    Hi

    This test app works great for me, tested as working on Android without issue and correctly swaps out the tick mark.

    To try it create a brand new default project and replace the contents of app.js with the code below;

var tbl = Ti.UI.createTableView({
    //data: rows,
    width: Ti.UI.FILL,
    height: Ti.UI.FILL,
    top: 0,
    left: 0,
    style: Ti.UI.iPhone.TableViewStyle.GROUPED,
    backgroundColor: 'transparent',
    separatorColor: '#c4c7cc'
});
tbl.addEventListener('click',function (e) {
    Ti.API.info("Is::::"+e.row.hasCheck);
    if (e.row.hasCheck === true) {
        e.row.hasCheck = false;
    } else {
        e.row.hasCheck = true;
    }
    Ti.API.info("Pass to::::" + e.row.hasCheck);
});
win.add(tbl);

var intRow = 0, intRows = data.length, rows = [];
function addRowFromData(data) {
    var row = Ti.UI.createTableViewRow({
        backgroundColor: data.hex,
        color: '#000',
        hasChild: data.hasChild,
        hasCheck: data.hasCheck,
        leftImage: data.leftImage,
        font: {
            fontSize: '14dp'
        },
        height: '50dp',
        className: 'dummyRow',
        data: data,
        title: data.title
    });

    if (data.index === 0) {
        row.header = 'Active';
    }
    rows.push(row);
};
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    data[intRow].index = intRow;
    addRowFromData(data[intRow]);
}
tbl.setData(rows);
var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab);
tabGroup.open();
