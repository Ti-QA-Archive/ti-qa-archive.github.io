  • Here is the simplest useCase :

    app.js : 

    

var tabGroup = Titanium.UI.createTabGroup();

var win1 = Titanium.UI.createWindow({  
});

var tab1 = Titanium.UI.createTab({  
    title:'tab1',
    window:win1,
});

var label1 = Titanium.UI.createLabel({
    text:'tab1'
});

win1.add(label1);




var win4 = Titanium.UI.createWindow({
         title:'tab4',
         layout:'vertical',
         url:'groupe.js'
});

var tab4 = Titanium.UI.createTab({  
    title:'Groupe',
    window:win4
});

var label4 = Titanium.UI.createLabel({
    width:'auto'
});
tabGroup.addTab(tab1);  
tabGroup.addTab(tab4);  


tabGroup.open();

    groupe.js

    var groupeWin = Titanium.UI.currentWindow;

var data = [{title:'first row'}];


var tableView = Titanium.UI.createTableView({
    data:data,
    width:'90%',
    backgroundColor:'green'
});

groupeWin.add(tableView);

    Try to rotate from portrait to landscape and then back to portrait. Setting 300 instead of 90% fixes the problem
