  • Hi Grace,

    Use below code to retrieve the user name from FB, you need to create facebook graph api request with "me"

    Add this code inside Facebook login eventListener
    For more info refer this link

    http://docs.appcelerator.com/titanium/2.1/#!/api/Titanium.Facebook 

    Titanium.Facebook.addEventListener('login', function(e) {    
    if (e.success) {
                Titanium.Facebook.requestWithGraphPath('me', {}, 'GET', function(e) {
                    if (e.success) {
                        var response = JSON.parse(e.result);
                        fbUserNameService.text = response.name;
                    } else if (e.error) {
                        alert("Error = "+e.error);
                    } else {
                        alert('Unknown response');
                    }
                });
    }else if (e.error) {
        alert("Error = "+e.error);
    } else if (e.cancelled) {
      alert("canceld");
    }
});

    Hope this helps you
