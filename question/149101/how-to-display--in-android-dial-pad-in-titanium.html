  • Hi Ali Akram,
    If you use the %23 it will display the # symbol in Dial pad
    Using URL Encoding we can display the # symbol

    var win = Ti.UI.createWindow();
win.backgroundColor = 'white';
// Create a Button.
var aButton = Ti.UI.createButton({
    title : 'Charge',
    height : 40,
    width : 150,
    top : 300,
    left : 80
});

var value = '123456789%23';

win.add(aButton);
aButton.addEventListener('click', function() {
    Titanium.Platform.openURL('tel:' + value);
});
win.open();
