    Hello Rama,

    It simple see below.

    ###index.xml

    <Alloy>
    <Window backgroundColor="white">
        <Button id="btn" onClick="openWindow"/>
    </Window>
</Alloy>

    ###index.js

    $.index.open();

function openWindow(){
    var win = Alloy.createController("test").getView();
    win.open();
}

    ###test.xml

    <Alloy>
    <Window id="winTest" backgroundColor="white">
        <Button id="btn" onClick="closeWindow"/>
    </Window>
</Alloy>

    ###test.js

    function closeWindow(){
    $.winTest.close();  
}

    for more information see this Titanium.UI.Window.
