Get data from server when app loads a window/tabGroup

I have a button that gets data from a server but I want it to happen automatically when I get in my app (like an onload function or something).
This is the button code I have:

//funcao refresh
buttonrefresh.addEventListener('click', function(e) {

    Ti.UI.Android.hideSoftKeyboard();
    feeds.visible = true;
    labelofertas.visible = true;
    labelresultados.visible = false;
    respostas.visible = false;

    //acesso local
    var url = "http://10.0.2.2/jobfinder/feeds.php";

    //acesso server daniel
    //var url = "http://danielfdsilva.com/luismelo/wipmobile/feeds.php";

    var params = "?params=" + inputpesquisa.value;
    var encodedURI = encodeURI(url + params);

    var xhr = Titanium.Network.createHTTPClient();
    xhr.open("GET", encodedURI);
    xhr.send();

    xhr.onload = function() {
        var addfeeds = [];
        var response = this.responseText;
        var array_resp = response.split("|");

        var answers = [];
        for(var i = 0; i <= array_resp.length - 2; i++) {

            addfeeds.push(Titanium.UI.createTableViewRow({
                title : array_resp[i],
                className : 'className',
                height : '70 px',
                backgroundImage : 'tabelrow_bkg.png',
            }));
        }
        feeds.setData(addfeeds);
    };
    xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
    xhr.onerror = function(e) {
        alert('Transmission error: ' + e.error);
    };
});
