//AppId
Titanium.Facebook.appid = 'XXX';
Ti.Facebook.forceDialogAuth = false;

//Extended Permissions der App

//Login-Prozedur mit Success- und Fehlerbehandlung

if (e.success) {

    var alertFBLogin = Ti.UI.createAlertDialog({

    alertFBLogin.show();

});
