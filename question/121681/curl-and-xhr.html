cUrl and Xhr

this there a way to use xhr to send cUrl commands? I am trying to convert the following. I searched around but the answers were scattered.

curl -u username:password http:&#x2F;&#x2F;myweb.com&#x2F;api&#x2F;list&#x2F;update.xml -d status=&#39;update worked&#39;

  • var upload = Titanium.Network.createHTTPClient();
upload.open("POST","http://myweb.com/api/list/update.xml");
var authstr = 'Basic ' +Titanium.Utils.base64encode(userName+':'+password);
upload.setRequestHeader('Authorization', authstr);

upload.onload = function (xhr_data) {}
upload.onerror = function (xhr_data) {}

upload.send({"status":"update worked"});
