    • yes im using commonjs like this :

      exports.createWindow = function(){

    aproposWin = true;
    var winAbout = Ti.UI.createWindow({
        backgroundColor: '#212121',
        tabBarHidden: true,
        navBarHidden: true
    });

    var navBarContainer = Ti.UI.createView({
        height: 44,
        backgroundImage: 'images/barImage.png',
        top: 0
    });

    var backButton = Ti.UI.createButton({
        backgroundImage: 'images/back.png',
        font: {fontWeight: 'bold', fontSize: 12},
        color: 'white',
        width: require('BusinessLogic/util').getSizeTag(L('other'))+25,
        title: '  ' + L('other'),
        height: 34,
        left: 10
    });

    var navTitle = Ti.UI.createLabel({
        color: 'white',
        font: {fontSize: 15, fontWeight: 'bold'},
        text: L('about')
    });

    navBarContainer.add(backButton);
    navBarContainer.add(navTitle);

    winAbout.add(navBarContainer);

    backButton.addEventListener('click', function(e){
        aproposWin = false;
        winAbout.close();
    });

    var webView = Ti.UI.createWebView({
        top: 44,
        url: 'https://seetch.com/about/mobile/' + Ti.App.Properties.getString('lang')
    });

    winAbout.add(webView);

    return winAbout;
};
