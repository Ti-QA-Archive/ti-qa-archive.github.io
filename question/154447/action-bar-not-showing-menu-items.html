  • HI to access the window activity, you should access the activity after window is opened.

    try like that

    win.addEventListener("open", function(evt) {
    if (Ti.Platform.osname === "android") {
        if (! win.activity) {
            Ti.API.error("Can't access action bar on a lightweight window.");
        } else {
            actionBar = win.activity.actionBar;
            if (actionBar) {
                actionBar.backgroundImage = "/images/Components/bg_top.jpg";
                actionBar.title = "";
                actionBar.logo="/images/Components/left_icon.png";
                actionBar.onHomeIconItemSelected = function() {
                    Ti.API.info("Home icon clicked!");
                };
            }
        }
    }

    win.activity.onCreateOptionsMenu = function(e) { 
        var menu = e.menu; 
        var menuItem = menu.add({ 
        title : "Compose", 
        //icon : "/images/Components/map.png", 
        showAsAction : Ti.Android.SHOW_AS_ACTION_ALWAYS 
    }); 
        menuItem.addEventListener("click", function(e) { 
        Ti.API.info("Action Item Clicked!"); 
    }); 
};


});
