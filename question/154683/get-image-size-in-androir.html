Get image size in androir

Hi, to scale my images i use imageFactory after this question but now i have a problem with android again. When i scale the image i need to know the size and in IOS it works fine but in android no. My code is:

var newImage        = Ti.UI.createImageView(stylesheet.contenidoImg);
newImage.image     = '/images/apart.cortizo.jpg';
newImage.image     = common.resizeImg('images', 'apart.cortizo.jpg', newImage.toImage().width, newImage.toImage().height, 500, 0);

function resizeImg(folder, img, currWidth, currHeight, newWidth, newHeight){
    var f = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory, folder, img);
    var blob = f.read();

    var ratio = currWidth / currHeight;

    if (newWidth == 0){
        var w = newHeight * ratio;
        var h = newHeight;
    } else if (newHeight == 0){
        var w = newWidth;
        var h = newWidth / ratio;
    } else {
        var w = newWidth;
        var h = newHeight;
    }
    return ImageFactory.imageAsResized(blob, { width:w, height:h});
}

How can i do it in android?

thanks a lot

    This'll do it for ya :) Without adding to view or window, and without opening a window.

    var file = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory+"background.png");
var blob = file.read();
alert("w: "+blob.width+", h: "+blob.height);
