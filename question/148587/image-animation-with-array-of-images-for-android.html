    • Try this

      var win = Ti.UI.createWindow();
var data = [];
var label, img1, img2, img3, view;
var color = ['#F6CECE', '#CEE3F6', '#4B8A08', '#81F7D8', '#04B486', '#CEE3F6', '#58FA82', '#E3CEF6', '#3B240B'];
var color1 = ['#F6CE5E', '#C9E3F6', '#1B8A08', '#61F7D8', '#84B486', '#7EE3F6', '#98FA82', '#13CEF6', '#8B240B'];
var index = 8;
for (var i = 0; i < 8; i++) {
    view = Ti.UI.createView({
        backgroundColor : 'transparent',
        width : 'auto',
        height : 'auto',
        layout : 'horizontal'
    });
    //  Ti.API.info('view id :' + view.id);

    img1 = Ti.UI.createImageView({
        backgroundColor : color[i],
        top : '12%',
        width : '70dp',
        height : '80dp',
        left : 5,
        image : '/image1.png'

    });
    view.add(img1);

    img2 = Ti.UI.createImageView({
        backgroundColor : color1[i],
        top : '9%',
        width : '150dp',
        height : '120dp',
        image : '/image1.png',
        left : 5

    });
    view.add(img2);

    img3 = Ti.UI.createImageView({
        backgroundColor : color[index],
        top : '12%',
        width : '70dp',
        height : '80dp',
        image : '/image1.png',
        left : 5

    });
    view.add(img3);

    label = Ti.UI.createLabel({
        text : 'Place Name',
        font : {
            fontSize : '15dp'
        },
        color : 'black',
        top : '40%',
    });
    view.add(label);

    data.push(view);
    index--;

}
var scrollabel = Ti.UI.createScrollableView({
    top : '10%',
    views : data,
    height : Ti.UI.FILL,
    width : Ti.UI.FILL,
    focusable : true,
});
win.add(scrollabel);

win.open();
