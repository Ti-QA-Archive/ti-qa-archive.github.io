Navigatore not showing up

I have following code in my app.js

// create welcome window
var win1 = Titanium.UI.createWindow({
    title: 'Expenses Tracker',
    barColor: '#912CEE',
    backgroundColor: '#666',
    backgroundImage: '/images/default.png',
    navBarHidden: false,
    url: '/controller/notrip.js',
    titleid: 'notrip_win_title'
});

// create about window
var win2 = Titanium.UI.createWindow({
    title: 'About App',
    barColor: '#912CEE',
    backgroundColor: '#666',
    navBarHidden: false,
    url: '/controller/about.js',
    titleid: 'about_win_title'
});

// add info button on navigation bar on win1
var infodark = Titanium.UI.createButton({
    width: 30,
    systemButton: Titanium.UI.iPhone.SystemButton.INFO_DARK
});
infodark.addEventListener('click', function(e) {
    win2.open({
        modal: true,
        modalTransitionStyle: Titanium.UI.iPhone.MODAL_TRANSITION_STYLE_FLIP_HORIZONTAL
    });
});
win1.leftNavButton = infodark;

win1.open();

On application launch, I am not able to see navigation bar in win1 screen.

But when I do this, I can see navigation bar.

win1.open({
    modal: true,
    modalTransitionStyle: Ti.UI.iPhone.MODAL_TRANSITION_STYLE_FLIP_HORIZONTAL
});

Any thoughts why it's failing for simple window.open()?

    you have to create at least one tabBar or a navigation group to get the titlebar to show. Here is a solution using the navigationGroup

    var baseWindow = Ti.UI.createWindow({
    backgroundColor : 'white'
});

// create welcome window
var win1 = Titanium.UI.createWindow({
    title: 'Expenses Tracker',
    barColor: '#912CEE',
    backgroundColor: '#666',
    backgroundImage: '/images/default.png',
    url: '/controller/notrip.js',
    titleid: 'notrip_win_title'
});

// create about window
var win2 = Titanium.UI.createWindow({
    title: 'About App',
    barColor: '#912CEE',
    backgroundColor: '#666',
    navBarHidden: false,
    url: '/controller/about.js',
    titleid: 'about_win_title'
});

// add info button on navigation bar on win1
var infodark = Titanium.UI.createButton({
    width: 30,
    systemButton: Titanium.UI.iPhone.SystemButton.INFO_DARK
});
infodark.addEventListener('click', function(e) {
    win2.open({
        modal: true,
        modalTransitionStyle: Titanium.UI.iPhone.MODAL_TRANSITION_STYLE_FLIP_HORIZONTAL
    });
});
win1.leftNavButton = infodark;

var navGroup = Ti.UI.iPhone.createNavigationGroup({
    window : win1
});

baseWindow.add(navGroup);
baseWindow.open();
