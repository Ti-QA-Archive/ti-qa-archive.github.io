  • I exceeded this isue editing TiUITableView.m file which path is "Users/<YourUserName>/Library/Application Support/Titanium/mobilesdk/osx/3.1.1.v20130613185544/iphone/Classes"

    Orginal TiUITableView.m as below;

    ..............
#pragma mark Search Display Controller Delegates

- (void) searchDisplayControllerDidEndSearch:(UISearchDisplayController *)controller
{
    animateHide = YES;
    [self performSelector:@selector(hideSearchScreen:) withObject:nil afterDelay:0.2];
}
@end

    I changed it as below;

    ..............
#pragma mark Search Display Controller Delegates

- (BOOL) searchDisplayController:(UISearchDisplayController *)controller shouldReloadTableForSearchString:(NSString *)searchString
{
    [controller.searchResultsTableView setBackgroundColor:RGBACOLOR(0, 0, 0, 1)];
    [controller.searchResultsTableView setSeparatorColor:RGBACOLOR(34, 34, 34, 1)];
    return YES;
}

- (void) searchDisplayControllerDidEndSearch:(UISearchDisplayController *)controller
{
    animateHide = YES;
    [self performSelector:@selector(hideSearchScreen:) withObject:nil afterDelay:0.2];
}
@end
