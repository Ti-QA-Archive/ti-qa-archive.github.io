Button on tableview row

Hi guys, I have been battling with this problem for hours and can't get it to work :/ I'm trying to add a button to my dynamically created tableview rows. When this button is clicked I want it to add the row's data to the users favourites. The problem I'm faced with is the following…When you click on a tableview row it takes you to a details view. The buttons are added to the rows, but when I try and click the button it opens the details view. How can I get the button click without moving to the details page? Any solution would be much appreciated. Here is my code:

var data = [];

var mainTable = Ti.UI.createTableView();

for (var i=0; i<rdata.length; i++) {

    var row = Ti.UI.createTableViewRow({
        height: 50,
        hasChild: true
    });

    var lblTitle = Ti.UI.createLabel({
        text: rdata[i].title
        top: 5,
        left: 10,
        height: 20
    });
    row.add(lblTitle);

    var lblTime = Ti.UI.createLabel({
        text: rdata[i].evTime,
        top: 30,
        left: 10,
        height: 20
    });
    row.add(lblTime);

    var btnFav = Ti.UI.createButton({
        title: 'Add to Fav',
        top: 40,
        left: 10,
        height: 20,
        width: 40
    });
    row.add(btnFav);

    data.push(row)
}

mainTable.data = data;
