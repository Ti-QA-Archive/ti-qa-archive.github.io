Cancel upload

Hi Guys,

I have a bit of a dilemma to do with Facebook uploading of images. I had a look through Q & A as well as the API docs regarding cancelling an upload.

The problem I'm having is in my app there is a Facebook upload button that takes a screenshot and posts it via the REST API.

Now when the user is on a slow 3g connection sometimes it could take over 2 minutes.

So i would like to implement a timer that after a certain amount of time if the e.success or e.error for Ti.Ui.Facebook.Request isn't fired it would cancel the upload and notify the user that the connection currently is too slow. Or perhaps implement a cancel button for facebook upload so that the user could cancel the upload.

I am running TI Studio 1.0.7 and TI 1.8.0.1v20111222130907 continuos build. Running the app on an iphone 4 with iOS 5.0.1

Any input would be appreciated

Thanks

  • No problems solution posted below

    var xhr

Titanium.App.addEventListener('cancel_upload', function(e) // if the cancel button is pressed fires appevent
{
    Ti.API.info("cancel upload");
    xhr.abort();
});

var caption
var link_url = 'http://itunes.apple.com/us/app/YOUR_APP_NAME/id11111111?ls=1&mt=8';

function sendPhoto()  // once user is authorized call function
{
    Titanium.App.fireEvent('show_indicator_load'); // show the loading indicator which also contains the cancel button
    caption = 'Check out this app '+ link_url;
    xhr = Titanium.Network.createHTTPClient();
    xhr.timeout = 120000; //2 min timeout
    xhr.onerror = function(e){
            Titanium.App.fireEvent('hide_indicator_load'); // hide the loading indicator stored in app.js
            alert('error uploading please try again later');
    };
    xhr.onload = function(){    
        Ti.API.info("done is" + this.status);
        Titanium.App.fireEvent('hide_indicator_load'); // hide the loading indicator stored in app.js
        Ti.API.info(this.responseText);        
        // check for a reply.id here....            
        var reply = JSON.parse(this.responseText);
        if(reply.id){
            //Ti.UI.createAlertDialog({title:'Facebook', message:'Your photo to Facebook was posted'}).show();
        } else {
            //Ti.UI.createAlertDialog({title:'Facebook Error', message:'Your photo was not posted for some reason!'}).show();
        }
    };

    var endPoint = 'https://graph.facebook.com/me/photos'; //link to upload photo
    xhr.open('POST',endPoint);
    xhr.send({
        access_token:Ti.Facebook.accessToken,
        picture:f.read(), // photo to upload screen captured earlier and stored to var f
        caption:caption,
    });
Ti.API.info("access token is   " + Ti.Facebook.accessToken);
}
