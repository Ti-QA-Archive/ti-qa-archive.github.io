ACS Social Integration throws error : "Couldn`t find module 'ti.cloud' "

The very basic code i use:

var Cloud = require('ti.cloud');
var fb = require('facebook');
fb.appid = "429949710450931";
fb.permissions = ['publish_stream', 'read_stream'];

var win = Ti.UI.createWindow({
    title : "ACS Social Integrations"
});

var fbSignupBtn = Ti.UI.createButton({
    title : "Login with Facebook",
    width : 160,
    top : 50
});
win.add(fbSignupBtn);

fbSignupBtn.addEventListener('click', function() {
    if (!fb.loggedIn) {
        fb.authorize();
    }
});

//add SocialIntegrations in Facebook login event listener
fb.addEventListener('login', function(e) {
    if (e.success) {
        alert("login Success ");
        //code for SocialIntegrations
    } else if (e.error) {
        alert("Error = " + e.error);
    } else if (e.cancelled) {
        alert("Canceld");
    }
});

win.open();

It does check for facebook permission in web browser and when it returns to app the it throws this error : "Couldn`t find module 'ti.cloud' "
