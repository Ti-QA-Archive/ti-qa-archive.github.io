    • Don't know exactly what code, but here are some parts:

      index.js

      var server = require('server');


// Called from index.xml: <Window id="win" onOpen="doOpen">

function doOpen(e) {
    if (server.isRunning()) {
        $.run.setValue(true);
    } else if (Ti.App.Properties.getBool('boot_enabled') && isBootTime()) {
        server.start();
        return $.win.close();
    }

    try {
        var actionBar = $.win.activity.actionBar;
        actionBar.title = '';
        actionBar.onHomeIconItemSelected = openAbout;
        $.win.activity.invalidateOptionsMenu();
    } catch(er) {
    }
}

function openPreferences() {
    try {
        Ti.UI.Android.openPreferences();
    } catch(er) {
        //
    }
}

      server.js

      var    URL = 'organizer.js',
    milliseconds = parseInt(Ti.App.Properties.getString('interval')||900000);


exports.intent = Ti.Android.createServiceIntent({
    url: URL
});

exports.intent.putExtra('interval', milliseconds);

exports.start = function(){
    var msg = L('start_service');

    if (!exports.isRunning()) {
        Ti.Android.startService(exports.intent);
        Ti.API.debug(msg);
        Ti.App.fireEvent('organizer', {text: msg} );
        Alloy.Globals.toast(msg);
    }
}

exports.stop = function(){
    var msg = L('stop_service');

    if (exports.isRunning()) {
        Ti.Android.stopService(exports.intent);
        Ti.API.debug(msg);
        Ti.App.fireEvent('organizer', {text: msg} );
        Alloy.Globals.toast(msg);
    }
}

exports.isRunning = function(){
    return Ti.Android.isServiceRunning(exports.intent);
}
