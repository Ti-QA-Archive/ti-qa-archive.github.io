    • Hi Nazir,

      did you tried Titanium.UI.iPhone.ActivityIndicatorStyle.BIG ?

      Here is some code for you:

      Ti.UI.backgroundColor = 'white';

var win1 = Ti.UI.createWindow({
  backgroundColor: 'gray'
});

var win2 = Ti.UI.createWindow({
  backgroundColor: 'blue',
  fullscreen: true
});

var bigActivityIndicator = Ti.UI.createActivityIndicator({
  color: 'green',
  font: {fontFamily:'Helvetica Neue', fontSize:20, fontWeight:'bold'},  //26
  message: 'Loading BIG...',
  style:Ti.UI.iPhone.ActivityIndicatorStyle.BIG,
  top:100,
  left:50,
  height:Ti.UI.SIZE,
  width:Ti.UI.SIZE
});
var darkActivityIndicator = Ti.UI.createActivityIndicator({
  color: 'green',
  font: {fontFamily:'Helvetica Neue', fontSize:20, fontWeight:'bold'},  //26
  message: 'Loading DARK...',
  style:Ti.UI.iPhone.ActivityIndicatorStyle.DARK,
  top:200,
  left:50,
  height:Ti.UI.SIZE,
  width:Ti.UI.SIZE
});
var plainActivityIndicator = Ti.UI.createActivityIndicator({
  color: 'green',
  font: {fontFamily:'Helvetica Neue', fontSize:20, fontWeight:'bold'},  //26
  message: 'Loading PLAIN...',
  style:Ti.UI.iPhone.ActivityIndicatorStyle.PLAIN,
  top:300,
  left:50,
  height:Ti.UI.SIZE,
  width:Ti.UI.SIZE
});

// The activity indicator must be added to a window or view for it to appear
win2.add(bigActivityIndicator);
win2.add(darkActivityIndicator);
win2.add(plainActivityIndicator);

// eventListeners must always be loaded before the event is likely to fire
// hence, the open() method must be positioned before the window is opened
win2.addEventListener('open', function (e) {
  bigActivityIndicator.show();
  darkActivityIndicator.show();
  plainActivityIndicator.show();
  // do some work that takes 6 seconds
  // ie. replace the following setTimeout block with your code
  setTimeout(function(){
    e.source.close();
    bigActivityIndicator.hide();
    darkActivityIndicator.hide();
    plainActivityIndicator.hide();
  }, 6000);
});

win1.open();
win2.open();

      If BIG is still not big enough for you, build your own activity indicator using Titanium.UI.ImageView and a series of PNG images.

      Here is some untested code for that, which assumes 10 PNG images from indicator1.png to indicator10.png

      var win1 = Ti.UI.createWindow({
  backgroundColor: 'blue'
});

var indicatorImages = [];
for (var i=1; i<=10; i++)
{
    indicatorImages.push('images/indicator' + i +'.png');
}

var indicatorView = Titanium.UI.createImageView({
    images:indicatorImages,
    duration:200,  
    repeatCount:0, 
    top:200,
    height:100,
    width:100,
    backgroundColor:'transparent'
});

indicatorView.addEventListener('load', function(e)
{
    // animate indicatorView
    indicatorView.start();
});

win1.add(indicatorView);

win1.open();

// stop indicatorView animation with indicatorView.stop();

      Hope that helps.

      Matthias
