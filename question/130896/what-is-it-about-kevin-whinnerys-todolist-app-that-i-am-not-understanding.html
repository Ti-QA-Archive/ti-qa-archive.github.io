What is it about Kevin Whinnery's ToDoList app that I am not understanding?

I'm trying to create a very simple project based on Kevin's ToDoList app but rather than open a window, I want to open a tabGroup. If I import Kevin's project and edit his code it works. But if I create a new Project and enter the code from scratch, it crashes on open. What am I doing wrong?

Here's my app.js:

var ApplicationTabGroup = require('ui/ApplicationTabGroup');
new ApplicationTabGroup().open();

and here's my ApplicationTabGroup.js (which sits in the ui folder)


function ApplicationTabGroup() {

    var self = Titanium.UI.createTabGroup();

    var win1 = Titanium.UI.createWindow({
        title : 'Tab 1',
        backgroundColor : '#fff'
    });
    var tab1 = Titanium.UI.createTab({
        title : 'Tab 1',
        window : win1
    });

    var win2 = Titanium.UI.createWindow({
        title : 'Tab 2',
        backgroundColor : '#fff'
    });
    var tab2 = Titanium.UI.createTab({
        title : 'Tab 2',
        window : win2
    });

    self.addTab(tab1);
    self.addTab(tab2);

    return self;
}

module.exports = ApplicationTabGroup;

If I stick this code into Kevin's project it works fine but in a new project that I create (and I have tried twice now) it crashes immediately. Any ideas?
