Audio Player Not Working with Android 4.0.3 and 4.0.4

I am creating an app that will play audio files. I got the audio files playing on most Android devices, but it won't work with 4.0.3. and 4.0.4. I got it working on older versions(oldest tested 2.3.4) and newer versions.

Here is my part of the code that plays the audio:

function playAudioFile(fileName) {
    var audioFile = fileName + ".mp3";
    var fileUrl = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, audioFile);

    if (Alloy.Globals.Sound === null || Alloy.Globals.Sound === undefined) {

        Alloy.Globals.Sound = Ti.Media.createSound({
            url : fileUrl,
            allowBackground : true
        });
        Alloy.Globals.Sound.play();
    } 
    else {
        Alloy.Globals.Sound.stop();
        Alloy.Globals.Sound = null;
        Alloy.Globals.Sound = Ti.Media.createSound({
            url : fileUrl,
            allowBackground : true
        });
        Alloy.Globals.Sound.play();
    }

}
