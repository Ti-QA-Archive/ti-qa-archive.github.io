  • module.exports = function(_needle,_collections,,_resultcallback) {
    // _needle is only one character 
    for (var i=0;i<_collections.length;i++) {
        if (_collections[i][0] === _needle) {
            _resultcallback({found:true,item:_collections[i]})
            break;
        }
    }
    _resultcallback({found:false})

}
