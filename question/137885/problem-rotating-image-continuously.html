Problem rotating image continuously

titanium mobile sdk 2.0
iphone sdk 5.1

Hi,

having problem rotating an image continuously. This code rotates the image 90 degrees clockwise then counterclockwise. I wanted to make a full continuous clockwise rotation but can't figure out how.

    var sunRotate = Ti.UI.create2DMatrix();
        sunRotate = sunRotate.rotate(90);
    var sunAnim = Ti.UI.createAnimation({
        transform: sunRotate,
        duration: 500
    });
    var sunRotateBack = Ti.UI.create2DMatrix();
        sunRotateBack = sunRotate.rotate(-90);
    var sunAnimBack = Ti.UI.createAnimation({
        transform: sunRotateBack,
        duration: 500
    });

    sunAnim.addEventListener('complete', function(){
        fullsun.animate(sunAnimBack);
    })

    setInterval(function(){
        fullsun.animate(sunAnim);
    },500)

  • hi Nash,

    if you want only clockwise rotation then try this

    var tt = 10;
setInterval(function(e) {
    var a = Titanium.UI.createAnimation();
    a.transform = t;
    a.duration = 5;
    var t = Ti.UI.create2DMatrix();
    t = t.rotate(tt);
    a.transform = t;
    tt = tt + 10;
    fullsun.animate(a);
}, 10);
