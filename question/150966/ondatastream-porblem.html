ondatastream Porblem

Hi,
I'm developing for both ios and android using sdk 3.0.

In this app I have to get a big json data with an http request :

// get all the json sent by the server
exports.getData = function(date, success, error) 
{
    var url = http://www.myservice........;

    Ti.API.info('get data : ' + url);

    var xhr = Ti.Network.createHTTPClient({
        onload: function(f) {
            try{
            //Ti.API.debug(this.responseText);
            json = JSON.parse(this.responseText);
            success(json);
           }
           catch(e){
                   alert('error');
           }
        },
        onerror: function(Err) {
            error();
        },
        ondatastream: function(e){
            Ti.API.info('Progress >> ' + e.progress);
            Ti.App.fireEvent('json', {progress: e.progress});
        },
           timeout: 120000
    });

    xhr.open("GET", url);
    xhr.send();
};

The ondatastream function return the value '1' just one time !!

Console :
[INFO] Progress >> 1

so i don't have a progression i have only one value !!

Can someone help me with this ?

  • Hi Kheiro,

    Value of e.progress =1 signifies that all data has been transfered.

    Try to use getOndatastream( ) method of Titanium.Network.HTTPClient to look back frequently for the status of datastream.
