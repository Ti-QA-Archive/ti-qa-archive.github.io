    • The app works on my iOS devices, but I used iOS specific tabbed bar. To make it android friendly, I added android detection and a workaround that pretends to be act like the tabbed bar. After the error, clicking continue actually shows my workaround in a view, which holds state properly.

      If the problem is indeed a view, the only new views are the tb1 and tb2 views added in the same way that I added tb1 in the iOS version.

      var isAndroid = (Ti.Platform.osname=='android') ? true : false;
if(isAndroid) {
    // do Android specific stuff
// TAB 1
    var spacer = Math.round(Ti.Platform.displayCaps.platformWidth*0.1667);
    var width = spacer-1;
    var height = 30;

    // TAB 1
    var tb1 = Ti.UI.createButton({
        title: "All",
        index:1,
        width:width,
        height:height,
        left:(spacer*2),
        top:170,
        backgroundColor:'#333',
        borderRadius:6,
        borderColor:'white',
        color:"#fff"
    });
    win.add(tb1);

    // TAB 2
    var tb2 = Ti.UI.createButton({
        title: "Me",
        width:width,
        height:height,
        left:(spacer*3),
        top:170,
        backgroundColor:'#000',
        borderRadius:6,
        borderColor:'white',
        color:"#fff"
    });
    win.add(tb2);
    var currTab = tb1;

    // ADD EVENT LISTENERS
    tb1.addEventListener('click',function() {
        currTab.backgroundColor = '#000';
        this.backgroundColor = '#333';
        currTab = this;
        this.index=1;
        Ti.API.info('tb1.index:  ' + tb1.index);
    });

    tb2.addEventListener('click',function() {
        currTab.backgroundColor = '#000';
        this.backgroundColor = '#333';
        currTab = this;
        tb1.index=0;
        Ti.API.info('tb1.index:  ' + tb1.index);
    });
    if (tb1.index==1)
        {
            Ti.API.info('android check - tb1.index:  ' + tb1.index);
            win.add(basicSliderLabel3);
            win.add(basicSlider3);    
        } else {
            win.remove(basicSliderLabel3);
            win.remove(basicSlider3);            
        }

} else {
// do iOS stuff
//
// BASIC TABBED BAR
//
    var tb1 = Titanium.UI.iOS.createTabbedBar({
        labels:['All', 'Me'],
        backgroundColor:'#336699',
        top:170,
        style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
        height:25,
        width:50,
        left: 150,
        index:0
    });

    tb1.addEventListener('click', function(e)
    {
        if (tb1.index==1)
        {
            Ti.API.info('tb1.index:  ' + e.index + ' act val ' + tb1.index);
            win.add(basicSliderLabel3);
            win.add(basicSlider3);    
        } else {
            win.remove(basicSliderLabel3);
            win.remove(basicSlider3);            
        }
    });
    win.add(tb1);    
}
