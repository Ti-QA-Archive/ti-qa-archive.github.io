  • Hi Mehul Modi!!!

    Please try the below code. It may helpful to you.

    var fbmod1=require('facebook');
fbmod1.appid = 'your api';
fbmod1.permissions = ['read_stream','user_photos','publish_stream'];

fbmod1.requestWithGraphPath('me', {}, 'GET', function(e) {
    if (e.success) {
        det=JSON.parse(e.result);
        Ti.API.info(det.name);
      alert("Name : "+det.name+"  Username : "+det.username+"  Gender : "+det.gender);
    } else if (e.error) {
        alert(e.error);
    } else {
        alert('Unknown response');
    }
});
