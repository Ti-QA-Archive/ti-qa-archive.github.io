  • Hi

    If the code you provided is in the correct order then I would reorder the code before trying anything else as scroller references home which has not been defined at that moment.

    var win= new ui.Window({
    navBarHidden:true,
    exitOnClose:true,
    backgroundImage:'/images/back.png'
});

var home= Ti.UI.createView();
var picker = Ti.UI.createPicker();

var data = [];
data[0]=Ti.UI.createPickerRow({title:'Bananas',custom_item:'b'});
data[1]=Ti.UI.createPickerRow({title:'Strawberries',custom_item:'s'});
data[2]=Ti.UI.createPickerRow({title:'Mangos',custom_item:'m'});
data[3]=Ti.UI.createPickerRow({title:'Grapes',custom_item:'g'});

// turn on the selection indicator (off by default)
picker.selectionIndicator = true;
picker.add(data);
home.add(picker);

var scroller = Ti.UI.createScrollableView({
    top:56, //100-44
    left:0,
    right:0,
    bottom:0,
    views:[home],  // now home has been defined and can be referenced
    showPagingControl:false 
});
win.add(scroller);
