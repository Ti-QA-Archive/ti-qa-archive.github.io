Keeping cookies persistent in a webview

I am using a webview to do certain actions for the user. The user logs-in using a form in webview to a 3rd party.

However, the session gets killed if the app is relaunched. I am trying to make the cookies persist across sessions.

I have tried storing them in the app's properties, but its not working.

webView.addEventListener("beforeload", function(e) {
    var domain = getDomain(webView.url);
    var cookies = Ti.App.Properties.getString("Cookies." + domain, null);
    if(cookies) {
      webView.evalJS("document.cookie=\"" + cookies + "\";");
      Ti.API.debug("Cookies loaded for domain " + domain + ": " + cookies);
    }
});

webView.addEventListener("load", function(e){
    // Save the cookies
    var cookies = webView.evalJS("document.cookie");
    var domain = getDomain(webView.url);
    Ti.App.Properties.setString("Cookies." + domain, cookies);
    Ti.API.debug("Cookies saved for domain " + domain + ": " + cookies);
});

Any suggestions on how to make a webview persistent?
