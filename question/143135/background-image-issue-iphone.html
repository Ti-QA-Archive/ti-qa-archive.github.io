Background Image Issue (iPhone)

function createImage(img) {
    var left_image = Ti.UI.createImageView({
            top: 5,
            left : 5,
            width : 80,
            height : 40,
            backgroundImage : img
        });
    return left_image;
}

function addRow(rows, counter,data_object) {
    var row = createRow(data_object);
    var row_label = createLabel(data_object.name, 100);
    var img_path = data_object.images[0].substring(0,data_object.images[0].length-4)+"-thumb."+data_object.images[0].substring(data_object.images[0].length-3,data_object.images[0].length);
    var row_img = createImage(img_path);
    row.add(row_img);
    row.add(row_label);
    rows.push(row);
    return rows;
}

This code shows the images on the simulator, the the iPhone does not show any images in the table view.

  • Make sure that there is no difference with the images names (especially with caps)
    The Mac simulator seem a little more forgiving with file names (ex. you call Image.png but it's actually image.png) This can work in the Simulator.

    The iPhone device will not be so permissive though.
