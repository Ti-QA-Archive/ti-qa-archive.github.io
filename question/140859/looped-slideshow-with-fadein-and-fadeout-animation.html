Looped Slideshow with FadeIn and FadeOut Animation

Hello,
This slideshow is working great but i want to make it fadeIn and fadeOut Effect but can't do that:(
This is the working code

var win = Ti.UI.currentWindow;
win.backgroundColor = '#FFF';
var view = Titanium.UI.createImageView({
    top:0,
    width: 320,
    height:400,
    zindex: 30

});
win.add(view);

var slides = [];
var slide1 = {img: 'slides/1.png', displayTime: 3000};
var slide2 = {img: 'slides/2.png', displayTime: 3000};
var slide3 = {img: 'slides/3.png', displayTime: 3000};
var slide4 = {img: 'slides/4.png', displayTime: 3000}; // create in a Loop
slides.push(slide1);
slides.push(slide2);
slides.push(slide3);
slides.push(slide4);

var idx = 0;
var countSlides = slides.length;
var getNextSlide = function() {
idx = idx >= countSlides ? 0 : idx;
var slide = slides[idx++];
return slide;
}
var showSlide = function(slide) {
view.backgroundImage = slide.img;
setTimeout(function() {
showSlide(getNextSlide());
}, slide.displayTime);
}



showSlide(getNextSlide());

    • Thanks for the example code, I have been looking for a solution on this for a while now. The way you could smooth out the transition is to use the autoreverse:true attribute.

      view.animate({
      //curve: Ti.UI.ANIMATION_CURVE_EASE_IN_OUT,
      opacity: 1,
      duration: 2600, // half the displaytime value + some more just to create the extra smoothness.
      autoreverse:true // this will fade back the image
      });
