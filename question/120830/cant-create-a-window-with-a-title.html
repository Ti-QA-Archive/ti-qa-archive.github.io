  • Manu

    I think you need to use the fullscreen:false property. The following has been tested and works on Titanium 1.7.X / Android 2.2:

    var win = Titanium.UI.createWindow({
  title:'My App',
  backgroundColor: 'green'
});

var button= Titanium.UI.createButton({
  color:'black',
  title:'Search',
  top:200,
  width:200,
  height:50
});

win.add(button);
win.open();

button.addEventListener('click',function(){
  var win2 = Titanium.UI.createWindow({
    title:'Search Results',
    backgroundColor:'white',
    fullscreen:false
  });
  win2.open();
});

    Cheers
