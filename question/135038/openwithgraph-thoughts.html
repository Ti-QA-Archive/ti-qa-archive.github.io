  • Hi Jason,

    Try this that may solve your problem.best luck

    Titanium.Facebook.appid = '134793934930';
Titanium.Facebook.permissions = ['email', 'likes', 'status'];
Titanium.Facebook.authorize();

var AccessToken = 'your accesstoken'; //====

Titanium.Facebook.requestWithGraphPath('me/likes?access_token=' + AccessToken, {}, 'GET', function(e) {
    if(e.success) {
        var data = JSON.parse(e.result);
        Ti.API.info("Name:" + data.name);
        Ti.API.info("email:" + data.email);
        Ti.API.info("facebook Id:" + data.id);
        Ti.API.info("likes:" + Titanium.API.debug(JSON.stringify(data.likes)));
        //Ti.API.info("status:"+data.status); //Ti.API.info("locations:"+data.locations);
    } else if(e.error) {
        alert(e.error);
    } else {
        alert('Unknown response.');
    }
});
