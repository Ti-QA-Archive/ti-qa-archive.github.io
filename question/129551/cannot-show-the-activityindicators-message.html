Cannot show the ActivityIndicator's message.

I have create a activityIndicator object in iphone project:

var actInd = Titanium.UI.createActivityIndicator({
    height:50,
    width:210,
    top:100,
    color:'black',
    message:'Loading...',
    zIndex:100,
    font : {fontFamily:'Helvetica Neue', fontSize:15,fontWeight:'bold'},
    style : Titanium.UI.iPhone.ActivityIndicatorStyle.DARK
});

and add the click event for a button:

button1.addEventListener('click', function()
{    
    actInd.show();
});

and then just can show the style icon but cannot see the message, but if I put the actInd.show() after createActivityIndicator and can be show the message,just like:

var actInd = Titanium.UI.createActivityIndicator({
    height:50,
    width:210,
    top:100,
    color:'black',
    message:'Loading...',
    zIndex:100,
    font : {fontFamily:'Helvetica Neue', fontSize:15,fontWeight:'bold'},
    style : Titanium.UI.iPhone.ActivityIndicatorStyle.DARK
});

actInd.show();

I don't know why,please help, thanks!!

BTW,it's fine in android.
