Change image on click of custom tableview row

Development enviroment-titanium studio-3.1.0 , iphone-6.0.

I have created the custom tableview row as following code:

function setData() {  

                     var dataArray = []; 
                     var userid = Titanium.App.Properties.getString("looking_userid");
                     var db = Ti.Database.open('scorm-db');
                     db.execute('CREATE TABLE IF NOT EXISTS assign_courses(left_image TEXT, user_id INTEGER, course_id INTEGER, module_id INTEGER,course_name TEXT, module_name TEXT, module_type TEXT ,zip_file TEXT, launch_path TEXT);');      
                     var course_list = db.execute("SELECT DISTINCT course_id,course_name,left_image FROM assign_courses where user_id = '"+userid+"' ORDER BY course_id");
                     var cnt=0;
                     while (course_list.isValidRow())  
                      { 
                      var downloadicon=course_list.fieldByName('left_image');
                         var row = Ti.UI.createTableViewRow({
                var row = Ti.UI.createTableViewRow({
            touchEnabled : true,
            height : '90dp',
            width : 'auto',
          });
 var rowLogo = Ti.UI.createImageView({
            touchEnabled : false,
            image : downloadicon,
            top : '5dp',
            height : '80dp',
            width : '80dp',
            left : '5dp',
        });
     row.add(rowLogo )
    dataArray.push(row);                
                       course_list.next(); 
                       cnt++;
                       }

                    tableview.setData(dataArray);
                   db.close();  

 };

I want to change the rowLogo on click event of tableview row in following code

tableview.addEventListener('click',
  function(e) {
   // change the rowLogo image on click event 
  }
)

  • change 

    var rowLogo = Ti.UI.createImageView({
....
});

    into 

    
row.rowLogo = Ti.UI.createImageView({
....
});

row.add(row.rowLogo )

    then 

    tableview.addEventListener('click',
  function(e) {
    e.row.rowLogo.image = "/images/my_new_image.png"
  }
)
