  • var ta10 = Titanium.UI.createTextArea({
    value:'Luminal width:\n\n< 0.6 cm (distal to the last confluence before the inferior vena cava)\n\n> 0.6 cm ~ right cardiac insufficiency\n\n',
    height:140,
    width:300,
    top:1570,
    font:{fontSize:20,fontFamily:'Helvetica',},
    color:'#000000',
    textAlign:'left',
    borderWidth:2,
    borderColor:'#555',
    borderRadius:5,
    editable: "true"
});
scrollView.add(ta10);

    This is the code I am using.
