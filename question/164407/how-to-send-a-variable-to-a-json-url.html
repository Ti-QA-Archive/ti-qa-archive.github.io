  • Hi,

    pass the key as a parameter to your send function. It is also important that you encode the value with encodeURIComponent().

    function send(key) {

    var xhr = Titanium.Network.createHTTPClient();

    xhr.onerror = function(e) {
        var error = e.error;
        alert(error);               
    };

    xhr.open('GET', 'http://serv.open.com/api/v01/groups?key='+encodeURIComponent(key));
    ...

    textField.addEventListener('return', function(e) {
    send(e.value);
});
