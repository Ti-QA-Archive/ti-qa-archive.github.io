unrecognized selector sent to instance

So I started with the Master Detail template and was trying to get the basics going before adding my DB connections. I built the project and it worked fine. I answered the door and ran it again when I got back and I get unrecognized selector @ line 32 which is detailContainer.add(detailView); To narrow the problem down I thought there error was in my DetailView.js so I commented out all the code. The error still persists. Below is the entire block of code from my ApplicationWindow.js including the above code.

function ApplicationWindow() {
    //declare module dependencies
    var MasterView = require('ui/common/MasterView'),
        DetailView = require('ui/common/DetailView');

    //create object instance
    var self = Ti.UI.createWindow({
        backgroundColor:'#ffffff'
    });

    //construct UI
    var masterView = new MasterView(),
        detailView = new DetailView();

    //create master view container
    var masterContainer = Ti.UI.createView({
        top:0,
        bottom:0,
        left:0,
        width:240
    });
    masterContainer.add(masterView);
    self.add(masterContainer);

    //create detail view container
    var detailContainer = Ti.UI.createView({
        top:0,
        bottom:0,
        right:0,
        left:240
    });
    detailContainer.add(detailView);
    self.add(detailContainer);        

    //add behavior for master view
    masterView.addEventListener('itemSelected', function(e) {
        detailView.fireEvent('itemSelected',e);
    });

    return self;
};

module.exports = ApplicationWindow;

    Hello Gilbert Blanchette,

    In your DetailView.js code, you are missing return statements return self;. That’s why when you add detailContainer.add(detailView); Its return empty value.

    Please use updated your code with return statement then you problem will solve.
