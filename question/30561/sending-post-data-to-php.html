  • you would normally use the XHR client

    var xhr = Ti.Network.createHttpClient();
xhr.onreadystatechange=function()
{
    if (this.readyState==4) {
        // post completed
    }
};
xhr.open("POST","http://my.com/url.php");
xhr.send("key=value&pair=boo");
