Zoom imageview to fullscreen with unproportional start size

Hi,

I have the following use case for an iPad app:

  • imageview with an 1024x768 image
  • initially it should be 534x400 pixels big
  • on a click event the image should be scaled to 1024x768 and on another click event be scaled back to 534x400

The problem: The image must initially be opened on screen with an unproportional size. I only know how to initialize it with a proportional size changes like this:

var t2 = Titanium.UI.create2DMatrix();
t2 = t2.scale(0.5);

var image = Titanium.UI.createImageView({
        image: image_path,
        top:100,
        left:200,
        width:1024,
        height:768,
        transform:t2
    });

Though I would need something like this:

var t2 = Titanium.UI.create2DMatrix();
t2 = t2.scaleToXSize(534);
t2 = t2.scaleToYSize(400);

var image = Titanium.UI.createImageView({
        image: image_path,
        top:100,
        left:200,
        width:1024,
        height:768,
        transform:t2
    });

Another way could be to "animate the imageview to 534x400 each time is is initialized" but that doesn't seem correct:


var image = Titanium.UI.createImageView({
        image: image_path,
        top:100,
        left:200,
        width:1024,
        height:768,
    });    
image.animate({width:534, height:400});

win.add(image);

Anybody know a solution to this use case?

  • OK sorry, now I understood your hint. And it seems to work! I have to do some further testing, but first impressions are positive! I'm also recognizing a pinch gesture on top of the image in order to scale it, here's the code:

    var image_scale_up_animation = Titanium.UI.createAnimation();
image_scale_up_animation.width=1024;
image_scale_up_animation.height=768;
image_scale_up_animation.zIndex = 100;
image_scale_up_animation.duration = 500;
image_scale_up_animation.center = current_window.center;
image_scale_up_animation.addEventListener("complete", function(e){
    media_object_animation_running = false;
    media_object_scaled = true;  
});

var image_scale_down_animation = Titanium.UI.createAnimation();   
image_scale_down_animation.width=532;
image_scale_down_animation.height=400;
image_scale_down_animation.center = mediabox_position;
image_scale_down_animation.zIndex = 0;
image_scale_down_animation.duration = 500;
image_scale_down_animation.addEventListener("complete", function(e){
    media_object_animation_running = false;
    media_object_scaled = false;
});

var image_view = Titanium.UI.createView({
    backgroundImage:image_path,
    width:532,
    height:400
})

var scroll_view_top = Titanium.UI.createScrollView({
    width:1024,
    height:768,
    left:0,
    top:0,
    showHorizontalScrollIndicator:false,
    showVerticalScrollIndicator:false,
    maxZoomScale:100,
    minZoomScale:0.1,
    backgroundColor:"transparent"
});

image_view.add(scroll_view_top);

scroll_view_top.addEventListener('scale', function(e)
{
    if(!media_object_animation_running & !animation_active)
    {
        media_object_animation_running = true;
        if (!media_object_scaled)
        {    
            image_view.animate(image_scale_up_animation);
        }
        else if(media_object_scaled)
        {    
            image_view.animate(image_scale_down_animation);        
        }
    }
});
