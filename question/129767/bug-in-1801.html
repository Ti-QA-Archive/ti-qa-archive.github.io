Bug in 1.8.0.1

I just find a bug regarding row deletion. No matter what index you provide, using row delete always deletes the first row…..

Here is my code:

var win = Ti.UI.createWindow({
    title : 'xmas'
});

var table = Ti.UI.createTableView({
    title : 'just a test'
});

win.add(table);
win.open();

var data = ['goodday', 'goodtime', 'goodboy', 'goodgirl'];
var arr = [];

for(var i = 0, max = data.length; i < max; i++) {

    arr[i] = Ti.UI.createTableViewRow({
        title : data[i],
        name : data[i]
    });
}

table.setData(arr);
table.allrows = arr;


for(var j = 0, len = table.allrows.length; j < len; j++) {
    table.allrows[j].addEventListener('longclick', function(e) {

        var index = table.getIndexByName(e.rowData.name);
        //alert(table.data[0].rowCount);
        table.deleteRow(index);
    });
}
