Dynamic option dialog

Hi,
I plan an UI to display actions on a event obkect. Here my code snippet:

self.listview.addEventListener('itemclick', function(_e) {
        var event = _e.event, options = [], onclickactions = [];
        if (event.mp4) {
            options.push('VideoPodcast'); // collecting of options
            onclickactions.push(function() { // collecting of actions
                require('ui/videowidget').create(event); 
            });
        }
        /* and other actions ... */
        var dialog = Ti.UI.createOptionDialog({
            options : options,
            title : 'Auswahl'
        });
        dialog.show();
        console.log(onclickactions);
        dialog.addEventListener('click', function(_d) {
            onclickactions[_d.index];
        });
    });

But on click nothing happens.
