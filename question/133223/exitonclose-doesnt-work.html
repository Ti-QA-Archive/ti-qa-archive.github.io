    for exit on close to work, window should also be modal:

    var $mainWindow = Ti.UI.createWindow({
    title: 'Alex Window',
    borderRadius: 5,
    borderWidth: 8,
    backgroundColor: '#9DD8F2',
    modal: true,
    exitOnClose: true
});

  • This is "app.js" file content.

    var $mainWindow = Ti.UI.createWindow({
    title: 'Alex Window',
    borderRadius: 5,
    borderWidth: 8,
    backgroundColor: '#9DD8F2',
    exitOnClose: true
});

var $closeBtn = Ti.UI.createButton({
    title: 'Close Window'
});

$closeBtn.addEventListener('click', function () {
    $mainWindow.close();
});

$mainWindow.add($closeBtn);

$mainWindow.open();

    And I supposed that clicking on a button would close the window that would lead to app closing as well.
