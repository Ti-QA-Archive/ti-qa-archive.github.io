  • Hello Michael,

    Take a look at the below code,

    var win = Ti.UI.createWindow({
    backgroundColor : 'white',
    windowSoftInputMode : Ti.UI.Android.SOFT_INPUT_ADJUST_RESIZE
});

var textField = Ti.UI.createTextField({    
    color : '#336699',
    top: 0,
    softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_DEFAULT_ON_FOCUS,
    keyboardType : Ti.UI.KEYBOARD_DEFAULT,
    returnKeyType : Ti.UI.RETURNKEY_DEFAULT,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    width : Ti.UI.FILL,
    height : 60,
});

var view = Titanium.UI.createView({
   backgroundColor:'gray',
   layout : 'horizontal',
   width:Ti.UI.FILL,
   height:120,
   bottom:0
});

win.add(view);


var aButton = Ti.UI.createButton({
    title : 'aButton',
    height : Ti.UI.FILL,
    width : '50%',
});

// Listen for click events.
aButton.addEventListener('click', function() {
    alert('\'aButton\' was clicked!');
});

// Add to the parent view.
view.add(aButton);

var bButton = Ti.UI.createButton({
    title : 'bButton',
    height : Ti.UI.FILL,
    width : '50%',
});

// Listen for click events.
bButton.addEventListener('click', function() {
    alert('\'bButton\' was clicked!');
});

// Add to the parent view.
view.add(bButton);

win.add(textField);
win.open();

    Here a text field and two buttons in a view is used. By setting softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_DEFAULT_ON_FOCUS, on textfield opens the softkeyboard, and by setting windowSoftInputMode : Ti.UI.Android.SOFT_INPUT_ADJUST_RESIZE in window will position the buttonsview right over the soft keyboard.

    Hope this helps. Thanks.
