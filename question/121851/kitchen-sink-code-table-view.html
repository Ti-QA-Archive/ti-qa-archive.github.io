  • Darren

    This may be what you need. Paste it into a blank app.js file, and it will run without modification:

    var tabGroup = Ti.UI.createTabGroup();

var win1 = Ti.UI.createWindow({
  title:'Tab 1',
  backgroundColor:'white'
});
var tab1 = Ti.UI.createTab({
  title:'Tab 1',
  window:win1
});

var label1 = Ti.UI.createLabel({
  color:'#999',
  text:'I am Window 1',
  textAlign:'center'
});

win1.add(label1);

var win2 = Ti.UI.createWindow({
  title:'Tab 2',
  backgroundColor:'black'
});
var tab2 = Ti.UI.createTab({
  title:'Tab 2',
  window:win2
});

var data = [
  {title:'UWIC Accommodation Service', hasChild:true, test:'../examples/movie.js'},
  {title:'Hall Fees', hasChild:true, test:'../examples/vibrate.js'},
  {title:'Accomodation Video', hasChild:true, test:'../examples/geolocation.js'},
  {title:'Contact Page', hasChild:true, test:'../examples/accelerometer.js'}
];

var tableview = Ti.UI.createTableView({
  data:data
});

win2.add(tableview);

tabGroup.addTab(tab1);
tabGroup.addTab(tab2);

tabGroup.open();

tableview.addEventListener('click', function(e){
  if (e.rowData.test)
  {
    var win = Ti.UI.createWindow({
      url:e.rowData.test,
      title:e.rowData.title
    });
    Ti.UI.currentTab.open(win,{animated:true});
  }
});

    Hope it helps

  • i'm a little confused on this question but here goes. Note that all vars are not fully set cus it takes to long to do so atm.

    var winMain; // main window with yur buttons

var button1,button9; // buttons 1-9

button1.addEventListener('click', function(){
    var modalWin = window // create new window
    modalWin.modal = true; // create modal window

    var tableview; // your tableview
    var tableData; // your data for in the tableview

    tableview.data = tableData;

    var CloseButton; // regular button

    CloseButton.addEventListener('click', function(){
        modalWin.close();
    });

    modalWin.add(CloseButton);
    modalWin.add(tableview);
    modalWin.open();
});

winMain.add(button1); // add all buttons

winMain.open();

    Has what youre asking for, hope it helps
