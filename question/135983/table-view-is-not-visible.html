Table View is Not Visible

Hello,

i'm new to Titanium, and i'm trying to create a table view and add it to a scroll view, using the following code, but it's not working:

var add_win= Ti.UI.currentWindow;

var test_option_view = Titanium.UI.createScrollView({
        contentWidth:'auto',
        contentHeight:'auto',
        showVerticalScrollIndicator:true,
        showHorizontalScrollIndicator:true,
        top: 0,
        layout:'vertical'
    });

var test_all_options = add_win.test_options;
var options_table = Ti.UI.createTableView();
var test_options_data = [];

for (i=0; i<test_all_options.length; i++)
{
    row = Ti.UI.createTableViewRow({
        height:'60dp',
        hasCheck:true
    });
    var lo = Ti.UI.createLabel({
        text:test_all_options[i],
        color:'#000',
        left:5,
        top:5,
        backgroundColor:'#0f0',
        font:{fontSize:20, fontWeight:'bold'},
        width:150,
        height:50
    });
    row.add(lo);
    test_options_data[i]=row;
}

options_table.setData(test_options_data);

test_option_view.add(options_table);
add_win.add(test_option_view);

not that if after creating the lo i put alert(test_all_options[i]); i'm getting an alert that contain a string for each iteration

but i'm wondering what is the problem, when i go to the window i get only a white screen
