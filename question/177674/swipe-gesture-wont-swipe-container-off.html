    Hello, Samia. Use this fixed code. here animation is called on the Container.

    var win1 = Titanium.UI.createWindow({ title : 'Tab 1', backgroundColor : '#fff' });
var viewContainer = Titanium.UI.createImageView({ backgroundColor : 'white', width : '100%', height : '100%', top : 0 });
var view = Titanium.UI.createImageView({ backgroundColor : 'green', width : '100%', height : '80%', top : 0 });
var view1 = Titanium.UI.createView({ backgroundColor : 'red', width : '100%', height : '20%', bottom : 0 });
viewContainer.add(view); 
viewContainer.add(view1);
viewContainer.addEventListener('swipe', function(e) {
if (e.direction == 'right') {

   alert('You swiped to the right'); 

} 
else if (e.direction == 'left') {

var anim = Ti.UI.createAnimation({
left : -300,
duration : 200,
curve : Ti.UI.ANIMATION_CURVE_EASE_OUT
});
anim.addEventListener('start', function(_startanicallback) {
});
anim.addEventListener('complete', function(_anicallback) {
});
viewContainer.animate(anim);
}
});
win1.add(viewContainer); win1.open();
