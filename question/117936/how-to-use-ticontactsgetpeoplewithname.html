  • FIXED IT ——-

    For anyone else that needs a working example…
    It searches for the contact in the address book, if they person is found it does nothing, if they are not found it pops up asking if we wish to add a contact and adds one if we click yes.

    var savedContact = null;
if (Ti.Contacts.getPeopleWithName('Adam Kearsley')[0] == null)
{
//create contact
var alert3 = Titanium.UI.createAlertDialog({
      title:'Contacts', 
      message:'Would you like to add Adam to your phonebook?',
      buttonNames: ['Yes','No'],
      cancel:1
    });
    alert3.show();
    alert3.addEventListener('click', function(et)
    {
    if (et.index == 0) { // clicked "Yes"
//CREATE CONTACT
var contact = Titanium.Contacts.createPerson();
contact.firstName = 'Adam';
contact.lastName = 'Kearsley';
var phone = {"work" : ['01123 333444']};
contact.phone = phone;
contact.note = 'HELLOOO!';
Titanium.Contacts.save();
    }
    });
}
else
{
var savedContact = Ti.Contacts.getPeopleWithName('Plantec Holdings Claims')[0];
}
