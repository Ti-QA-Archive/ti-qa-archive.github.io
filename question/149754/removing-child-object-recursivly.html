Removing child object recursivly

I want to remove all the objects in window before closing it. I tried this method and it seems to working for some extent. If im not mistaken this is not an proper way to do so. 

function removeChildren(parentObj) {
    var child = parentObj.children;
    if (child && child.length > 0) {
        for (var i = 0; i < child.length; i++) {
            var subChildren = child[i].children;
            if (subChildren && subChildren.length > 0) {
                for (var k = 0; k < subChildren.length; k++) {
                    var subSubChildren = subChildren[k].children
                    if (subSubChildren && subSubChildren.length > 0) {
                        for (var j = 0; j < subSubChildren.length; j++) {
                            subChildren[k].remove(subSubChildren[j]);
                            subSubChildren[j] = null;
                        };
                    } else {
                        child[i].remove(subChildren[k]);
                        subChildren[k] = null;
                    };
                };
            } else {
                parentObj.remove(child[i]);
                child[i] =null;
            };

        };
    };
};

I also tried this code which i found from the Titanium Q&A, however i getting an error saying 

undefined  i in Object

in iOS and and warning saying child already removed 

function removeChildrens(object) {
    for (i in object.children) {
        var child=object.children[0];
        removeChildrens(child);
        object.remove(child);
        child=null;
    }
}

is there a better way to do this task?

    alle childs are subobjects of window.In this case nulling of window removes all childs.

    var window = Ti.UI.createWindow();
window.tv = Ti.UI.createTableView();
window.add(window.tv);
window.addEventListener('close',function(){
    window = null;
})
