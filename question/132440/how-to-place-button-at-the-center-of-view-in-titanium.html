  • Right this is down to the fact layout: horizontal cannot be centered as it automatically floats the child objects.
    So what you have to do, if you want a centered horizontal view with lots of children button is to wrap that view in another view 

    var container    = Ti.UI.createView({
    height: 200 // or whatever you want,
    width: Ti.UI.FILL
});

var horz_buttons = Ti.UI.createView({
    layout: horizontal,
    height: Ti.UI.SIZE,
    width: Ti.UI.SIZE
});

button  = Ti.UI.createButton({
    // you get the picture 
});

// add buttons to the horizontal layout
horz_buttons.add(button); 
horz_buttons.add(button;

container.add(horz_buttons);
window.add(container);

    what this will do is the container will float the horzs_buttons container into the center rather than you trying to get horz_button to float the children.

    old issue i know but i thought i give some clarity on what was actually happening with your code, to which you've probably already figured out haha :)

  • Ti.Platform.displayCaps.platformWidth and Ti.Platform.displayCaps.platformHeight would help

    var mainWin = Titanium.UI.createWindow({
    backgroundColor : '#FF00FF',
    layout : 'vertical'
});

var view = Titanium.UI.createView({
    height : '100%',
    width : '100%',
    backgroundColor : '#FFDDEE',
    layout : 'vertical'
});

var btnWidth = 80;
var btnHeight = 40;
var button = Titanium.UI.createButton({
    width : btnWidth,
    height : btnHeight, //'auto',
    left : (Ti.Platform.displayCaps.platformWidth - btnWidth) / 2,
    top : (Ti.Platform.displayCaps.platformHeight - btnHeight) / 2,
    title : 'centered',
    backgroundColor : 'red'
});

view.add(button);
mainWin.add(view);
mainWin.open();
