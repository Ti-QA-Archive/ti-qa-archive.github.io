  • Something like this:

    e.SOMETHINGELSE = [];
for (var i = 0; i < e.REPLACETHIS.length; i++) {
    var return = e.REPLACETHIS[i];
    e.SOMETHINGELSE[i] = return;
}
delete e.REPLACETHIS;

  • I want to change the classname, and all places with class1 should say class2. class2 is saved in a string, so I can replace it after classname: HERE, but not here: e.HERE

    function getData() {

    var data = [];

    Cloud.Objects.query({
        classname : "class1"

    }, function(e) {
        if (e.success) {
            //alert('Success:\n' + JSON.stringify(e.class1));
            for (var i = 0; i < e.class1.length; i++) {

                var stuff = e.class1[i];

                var row = Ti.UI.createTableViewRow({
                    id : stuff.id,
                    hasChild : true,
                    height : '100dp',
                    backgroundColor : '#000'
                });

                data.push(row);
            }

            tableView.data = data;
        } else {
            Titanium.API.log('Error query: ' + ((e.error && e.message) || JSON.stringify(e)));
        }

    });

};

getData();
