I'm sturck when passing data between contexts

Hey champs!

I've been reading http://developer.appcelerator.com/blog/2010/12/titanium-guides-project-js-environment.html and that kicked my to start making some changes to my little app that is based on 4 windows with a tabGroup.

Set up:

app.js

Titanium.UI.setBackgroundColor('#000');

var myApi = {
    ui: {}
};

Ti.include(
    'windows/window.convert.js',
    'core/ui.js'
);

window.convert.js

(function () {
    myApi.ui.createWindowConvert = function () {
        var windowAbout = Titanium.UI.createWindow({
           titleid:'win_title',
            barImage: 'images/ui/barTop.png',
            backgroundImage:'images/ui/contentBg.png'
        });

        return windowAbout;

    };
})();

ui.js

(function() {

    var tabGroup = Ti.UI.createTabGroup();

    var tab1 = Ti.UI.createTab({  
        icon:'images/icons/icon_convert.png',
        titleid:'tab_title_convert',
       backgroundImage:'images/ui/barBottom.png',
       window:myApi.ui.createWindowConvert
    });

    var tab2 = Ti.UI.createTab({  
        icon:'images/icons/icon_convert.png',
        titleid:'tab_title_convert',
        backgroundImage:'images/ui/barBottom.png',
        window:myApi.ui.createWindowConvert
    });

    tabGroup.addTab(tab1);  
    tabGroup.addTab(tab2);  

    tabGroup.open();
})();

Now this is just a snippet code so that you can see my point/idea. I'm using same window for 2 tabs just to get some output.

So the problem is.
When I run the app in simulator I get following error:

[ERROR] Script Error = -[KrollWrapper setParentOrientationController:]: unrecognized selector sent to instance 0xa28e210 at ui.js (line 25).

Line 25 in ui.js is : tabGroup.open(); So I don't get it.. Anyone else have an idea of what that could be ? or is it not posible to pass "win" object into a tabGroup like that.

Thanks!

  • Accepted Answer

    Replace window:myApi.ui.createWindowConvert by window:myApi.ui.createWindowConvert() would help :)

    (function() {

    var tabGroup = Ti.UI.createTabGroup();

    var tab1 = Ti.UI.createTab({
        icon : 'images/icons/icon_convert.png',
        titleid : 'tab_title_convert',
        backgroundImage : 'images/ui/barBottom.png',
        window : myApi.ui.createWindowConvert()
    });

    var tab2 = Ti.UI.createTab({
        icon : 'images/icons/icon_convert.png',
        titleid : 'tab_title_convert',
        backgroundImage : 'images/ui/barBottom.png',
        window : myApi.ui.createWindowConvert()
    });

    tabGroup.addTab(tab1);
    tabGroup.addTab(tab2);

    tabGroup.open();
})();
