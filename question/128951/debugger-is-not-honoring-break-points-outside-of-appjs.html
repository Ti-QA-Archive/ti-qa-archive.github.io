    • I'm using the Monkey Patch code (and AppTabGroup) from Forging Titanium "Episode 6: A Single Context Tab Group Template"…

      exports.monkeypatch = function(object) {
    var scriptRegistry = {},
        old_require = object.require;
    object.require = function(moduleName) {
        if (!scriptRegistry[moduleName]) {
            var mod = old_require(moduleName),
                moduleRoot = moduleName.split(/[\/ ]+/).pop();
            if (typeof(mod[moduleRoot]) === 'function') {
                scriptRegistry[moduleName] = mod[moduleRoot];
            }
            else {
                scriptRegistry[moduleName] = mod;
            }    
        }
        return scriptRegistry[moduleName];
    };
};

      I create a separate JavaScript file for each of my windows. Here is one simple example:

      // ExampleWindow.js

exports.ExampleWindow = function(args) {
    var win = Ti.UI.createWindow(args);

    // Window configuration code here...

    return win;
}

      Then in app.js, and elsewhere, I create and open windows in one of the following two ways:

      // app.js

require('lib/require_patch').monkeypatch(this);

var globals = {};

(function() {
    var AppTabGroup = require('ui/AppTabGroup');

    var ExampleWindow = require('views/ExampleWindow');
    var OtherWindow = require('views/OtherWindow');

    // *****
    // Window open method #1: Assign to a tab in the tab group
    // *****
    globals.tabs = new AppTabGroup(
        {
            title: 'Example',
            window: new ExampleWindow({title:'Example',backgroundColor:'white'})
        },
        {
            title: 'Other',
            window: new OtherWindow({title:'Other',backgroundColor:'white'})
        }
    );

    globals.tabs.open();


    // *****
    // Window open method #2: Open directly (in this case as a modal sheet over the tabs)
    // *****

    // Login window
    var LoginWindow = require('views/LoginWindow');
    var loginWin = new LoginWindow();
    loginWin.open({modal: true});
})();
