  • I ran a test and its the layout in your tableviewrow which is causing you issues.

    This works:

    var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    modal: true
});

var title = [ 'one', 'two', '3' ], table_data = [];

for(var i=0;i<title.length;i++) {

    var direction_label=Titanium.UI.createLabel({
        color:'black',
        font:{ fontWeight:"normal" },
        left:10,
        top:0,
        text:title[i]
        });

    var thisRow = Ti.UI.createTableViewRow({
    className:"directiondata",
    layout:"absolute",
    height:75
    });
    thisRow.add(direction_label);
    table_data.push(thisRow);

}

var table = Titanium.UI.createTableView({data:table_data});
win.add(table);

win.open();

    I have never used layout before and unless you need it I would not use it.

    Hope this helps.

    Jamesy
