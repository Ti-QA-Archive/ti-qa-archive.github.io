    Check this :

    index.xml

    <Alloy>
    <Window id="win">
        <Button id="btn" onClick="doClick">

        </Button>
    </Window>

function doClick(e){

        text : "This is a label"
    }).getView();
    controller.open();
}

    next.xml

    <Alloy>
    <Window backgroundColor="white">
        <Label id="label"></Label>
    </Window>
</Alloy>

    next.js

    var args = arguments[0] || {};


$.label.text = args.text;
