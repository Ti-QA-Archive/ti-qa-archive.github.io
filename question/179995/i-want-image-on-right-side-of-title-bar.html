    Hi Bunty Rohra,

    Thanks for your concern.

    Maybe you do not add the button to the NavigiationWindow.

    Test Code:

    var btnLeft = Ti.UI.createButton({
    title : 'left'
});
var btnRight = Ti.UI.createButton({
    title : 'Email',
    top : '50',
    width : Ti.UI.FILL,
    right : true,
    backgroundColor : 'gray',
});
btnLeft.addEventListener('click', function(e) {
    alert('left');
});
btnRight.addEventListener('click', function(e) {
    alert('right');
});
// main window
var win = Ti.UI.createWindow({
    backgroundColor : 'red',
    leftNavButton : btnLeft, // one way to set a nav button
    title : 'Red Window'
});
win.setRightNavButton(btnRight);
// other way to set the nav button
// nav window
var nav = Ti.UI.iOS.createNavigationWindow({
    window : win
});
nav.open();
