  • i created a showweb.js files and then i send the URL to it.

    showweb.js

    var win = Ti.UI.currentWindow;
win.barColor = '#000';
var webview = Titanium.UI.createWebView({url:win.myurl});
win.add(webview);

    then you can open this using

    var webwin = Titanium.UI.createWindow({
        url: 'showweb.js',
        title:"Google",
        backgroundColor: '#fff',
        myurl:"http://www.google.com"
    });
webwin.open({modal:true});

    This opens the modal window with a title, you can also set nav buttons and bars etc… when calling the window, and showweb.js can be re-used for other links etc…

  • OK this works…

    app.js

    
// Your Window
var win = Ti.UI.createWindow();
win.open(); 



var data = [];

for (var c=0;c<4;c++)
{
    var row = Ti.UI.createTableViewRow();
    row.selectedBackgroundColor = '#fff';
    row.height = 100;
    row.className = 'datarow';
    row.clickName = 'row';

    var Links = Ti.UI.createLabel({
        color:'#ff0000',
        font:{fontSize:16,fontWeight:'bold', fontFamily:'Arial'},
        left:70,
        top:2,
        height:30,
        width:245,
        clickName:'Links',
        text:'Link'
    });

    row.add(Links);

    var comment = Ti.UI.createLabel({
        color:'#222',
        font:{fontSize:16,fontWeight:'normal', fontFamily:'Arial'},
        left:70,
        top:21,
        height:50,
        width:245,
        clickName:'comment',
        text:'http://www.google.co.uk'
    });
    row.add(comment);

    row.link = 'http://www.google.co.uk';

    data.push(row);
};

//
// create table view (
//
tableView = Titanium.UI.createTableView({
    data:data,
    filterAttribute:'filter',
    backgroundColor:'white'
});

win.add(tableView);

    tableView.addEventListener('click', function(e)
{
var webwin = Titanium.UI.createWindow({
        url: 'showweb.js',
        title:e.rowData.link,
        backgroundColor: '#fff',
        myurl:e.rowData.link
    });
var back_button = Titanium.UI.createButton({
    title:'Back',
    style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN
    });
webwin.setLeftNavButton(back_button);
back_button.addEventListener('click', function(e)
{
    webwin.close();
})

        webwin.open({modal:true});
});

    webwin.js

    var win = Ti.UI.currentWindow;
win.barColor = '#000';
var webview = Titanium.UI.createWebView({url:win.myurl});
win.add(webview);

    Shows table with text, shows the website in a modal window with title and back button etc…
