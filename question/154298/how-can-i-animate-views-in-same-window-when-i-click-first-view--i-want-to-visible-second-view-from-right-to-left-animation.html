index.js

function redClick(e) { 

$.green.setVisible(true);
$.red.setVisible(false);

$.green.animate(
{
left:0,

  duration:500});

}
function greenClick(e) {
$.gray.setVisible(true);
$.green.setVisible(false);

}
function grayClick(e) {
$.red.setVisible(true);
$.gray.setVisible(false);

}

$.index.open();
