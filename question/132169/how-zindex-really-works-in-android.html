How zIndex really works in Android

I have put a sample code below stating how I am trying to use the zIndex in my APP.

var windows1 = Titanium.UI.createWindow({
    fullscreen : true,
    backgroundColor : "#fedcba",
    navBarHidden : true
});

windows1.open();

var button1 = Ti.UI.createButton({
    height : 100,
    width : 100,
    top : 90,
    left : 90,
    text : 'GO'
});
var width = Titanium.Platform.displayCaps.platformWidth;
var height = Titanium.Platform.displayCaps.platformHeight;
var button2 = Ti.UI.createButton({
    height : 100,
    width : 100,
    top : (height - 300),
    left : (width - 300),
    text : 'GO'
});
var view1 = Ti.UI.createView({
    width : width,
    opacity : 1.0,
    top : 0,
    left : 0,
    height : height
});
var view2 = Ti.UI.createView({
    width : width,
    opacity : 1.0,
    backgroundImage : 'images/bg_tile.jpg',
    top : -height,
    left : 0,
    height : height,
    zindex:100
});
view2.add(button2);
view1.add(button1);
windows1.add(view2);
windows1.add(view1);

button1.addEventListener('click', function() {
    view2.animate({
        left : 0,
        top : 0,
        duration : 500
    });
});
button2.addEventListener('click', function() {
    view2.animate({
        top : -height,
        left : 0,
        duration : 500
    });
});

On Click of button1 I view2 from top slides in and gives it a nice animation effect .Now, button2 click is not working and button1 is still clickable. I want that view2 should make an overlay over view1 and nothing from view1 should be accessible. Only view2 elements should be accessible like fields buttons etc.
