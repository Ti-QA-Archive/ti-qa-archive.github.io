    You can have a look at picker_android_spinner_date.js in KitchenSink.

    var picker = Ti.UI.createPicker({
    useSpinner: true,
    type:Ti.UI.PICKER_TYPE_DATE,
    minDate:minDate,
    maxDate:maxDate,
    value:value
});

    Hope it helps.

  • Hi Andrew

    I have made one demo try to make changes as per your need in this. 

    
// create main window
var win = Ti.UI.createWindow({
    backgroundColor:'#CCCCCC'
});

var isAndroid = Ti.Platform.osname === 'android';

if (isAndroid){
    var status = Ti.UI.createLabel({
        top: 5, left: 5, right: 5, height: 40, textAlign:'center', color:'#000000'
    });
    win.add(status);

    function showStatus(s) {status.text = s;}

    var names = ['Bananas', 'Strawberries', 'Mangos', 'Grapes'];
    var verbs = ['red', 'green', 'blue', 'orange'];

    var rows1 = [];
    for (var i = 0; i < names.length; i++) {
        rows1.push(Ti.UI.createPickerRow({title: names[i]}));
    }

    var rows2 = [];
    for (i = 0; i < verbs.length; i++) {
        rows2.push(Ti.UI.createPickerRow({title: verbs[i]}));
    }

    var column1 = Ti.UI.createPickerColumn( {
        rows: rows1, font: {fontSize: "12"}
    });
    var column2 = Ti.UI.createPickerColumn( {
        rows: rows2, font: {fontSize: "12"}
    });

    var picker = Ti.UI.createPicker({
        useSpinner: true, visibleItems: 7,
        type : Ti.UI.PICKER_TYPE_PLAIN,
        top: 150, height: 200,
        columns: [ column1, column2]
    });

    picker.addEventListener('change', function(e) {
        showStatus(e.selectedValue[1] + " " + e.selectedValue[0]);
    });

    win.add(picker);

}else{
    var picker = Ti.UI.createPicker();
    var column1 = Ti.UI.createPickerColumn({opacity:0});
    column1.addRow(Ti.UI.createPickerRow({title:'Bananas',custom_item:'b'}));
    column1.addRow(Ti.UI.createPickerRow({title:'Strawberries',custom_item:'s', selected:true}));
    column1.addRow(Ti.UI.createPickerRow({title:'Mangos',custom_item:'m'}));
    column1.addRow(Ti.UI.createPickerRow({title:'Grapes',custom_item:'g'}));

    var column2 = Ti.UI.createPickerColumn();
    column2.addRow(Ti.UI.createPickerRow({title:'red'}));
    column2.addRow(Ti.UI.createPickerRow({title:'green'}));
    column2.addRow(Ti.UI.createPickerRow({title:'blue'}));
    column2.addRow(Ti.UI.createPickerRow({title:'orange'}));


    // 2 columns as an array
    picker.add([column1,column2]);

    // turn on the selection indicator (off by default)
    picker.selectionIndicator = true;

    win.add(picker);

    var label = Ti.UI.createLabel({
        text:'Picker Result',
        top:10,
        width:'auto',
        height:'auto',
        textAlign:'center',
    });
    win.add(label);


    picker.addEventListener('change',function(e)
    {
        label.text = e.selectedValue[1] + " " + e.selectedValue[0]
    });

}
    // open window
    win.open();

    Hope the demo will help to get what exactly you wants to acheive

    Regards

    Nikunj
