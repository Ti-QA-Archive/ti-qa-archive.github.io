animation left not working

I am havin a very odd issue, on Ipad, SDK 4.3 Titanium 1.6.2. I cant get a imageView to animate sliding from one side of the screen. it just fades in. but it doesn't move to the side. any ideas?

var imagePlayer1 = Titanium.UI.createImageView({
    image:'../images/logo.png',
    opacity:0,
    width:254,
    height:375,
    top:137,
    left:0
});

var imagePlayer1Animation = Titanium.UI.createAnimation({
   opacity: 1,
   left:505,
   duration:500,
   delay:500
});


imagePlayer1.animate(imagePlayer1Animation);

win.add(imagePlayer1);

it just stays in the left:0 and fades in but it doesn't slide.
Regards,
Andrew.

  • I just tested this to be working correctly… are you testing this in the simulator or the device? Does this code work if this is all thats in app.js:

    var tabGroup = Ti.UI.createTabGroup();

var win = Titanium.UI.createWindow({
  title:'Window 1',
  backgroundColor:'blue'
});

var tab1 = Ti.UI.createTab({
    title: 'test',
    window: win
});

var imagePlayer1 = Titanium.UI.createImageView({
    image:'logo.jpeg',
    opacity:0,
    width:254,
    height:375,
    top:137,
    left:0
});

var imagePlayer1Animation = Titanium.UI.createAnimation({
   opacity: 1,
   left:505,
   duration:500,
   delay:500
});


imagePlayer1.animate(imagePlayer1Animation);

win.add(imagePlayer1);

tabGroup.addTab(tab1);

tabGroup.open();
