  • Hi Omkar,

    Try this complete code

    $.winIndex.layout="vertical";
for (var i = 0; i < 5; i++) {
    var backView = Ti.UI.createView({
        width : "100%",
        height : Ti.UI.SIZE,
        layout : "horizontal"
    });
    $.winIndex.add(backView);

    var imgCheckBox = Ti.UI.createImageView({
        left : 10,
        width : 20,
        height : 20,
        image : "imgUnCheck.png",
        isSelect : false
    });
    backView.add(imgCheckBox);

    imgCheckBox.addEventListener("click", function(e) {
        if (e.source.isSelect) {
            e.source.image = "imgUnCheck.png";
        } else {
            e.source.image = "imgCheck.png";
        }
        e.source.isSelect = !e.source.isSelect;
    });

    var lbl = Ti.UI.createLabel({
        left : 30,
        right : 10,
        text : "LABEL TEXT"
    });
    backView.add(lbl);

}

    Thanks
