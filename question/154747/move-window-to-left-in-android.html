My example code:

//Menu window in the bottom
var menuWindow = Ti.UI.createWindow({
    top : 0,
    left : 0,
    width : '200dp'
});

var menuTitles = [
    {title: 'Menu 1'},
    {title: 'Menu 2'},
    {title: 'Menu 3'},
    {title: 'Menu 4'},
    {title: 'Menu 5'},
    {title: 'Menu 6'}
];
var tableView = Ti.UI.createTableView({
    data:menuTitles
});
menuWindow.add(tableView);
menuWindow.open();

//Content window in the front
var content = Ti.UI.createWindow({backgroundColor: '#E4E4E4',
    navBarHidden:true,
    orientationModes: [Ti.UI.PORTRAIT],
    width : '100%'
});
content.open();

//Animation
content.animate({
    left:stylesheet.headerMenu.width,
    duration:400,
    curve:Ti.UI.ANIMATION_CURVE_EASE_IN_OUT
});
