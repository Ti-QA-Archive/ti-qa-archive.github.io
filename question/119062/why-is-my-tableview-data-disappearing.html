Why is my TableView data disappearing?

~~~
var first = Ti.UI.createWindow({
backgroundColor:"#fff",
title:"My App"
});
var label = Ti.UI.createLabel({ text: "poke me to open the next window" });
first.add(label);

//Here's the nav group that will hold them both…
var navGroup = Ti.UI.iPhone.createNavigationGroup({
window:first
});

//Here's a window we want to push onto the stack…
var second = Ti.UI.createWindow({
backgroundColor:'blue',
title:"Child Window"
});
second.add(Ti.UI.createLabel({text:"Here's the child"}));

var third = Ti.UI.createWindow({
backgroundColor:'green',
title:"Green Child"
});

third.add(Ti.UI.createLabel({text:"Here's the green child"}));
var label2 = Ti.UI.createLabel({ text: "poke me to open the green window" });
second.add(label2);

//When the label on the first window receives a touch, open the second
label.addEventListener("click", function(e) {
navGroup.open(second);
});

var data = [{title:"Row 1"},{title:"Row 2"}];
var table = Titanium.UI.createTableView({data:data});
third.add(table);

label2.addEventListener("click", function(e) {
navGroup.open(third);
});

//This is the main window of the application
var main = Ti.UI.createWindow();
main.add(navGroup);
main.open();

— asked April 28th 2011 by Stephen Flournoy
