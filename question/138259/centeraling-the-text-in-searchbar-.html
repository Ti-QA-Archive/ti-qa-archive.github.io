  • var searchField = Titanium.UI.createTextField({
    top:0,
    left:"5%",
    width:"90%",
    textAlign:"center",
    returnKeyType:Titanium.UI.RETURNKEY_SEARCH
});
win.add(searchField);

searchField.addEventListener('return',function(e)
{
    searchField.blur();

    Ti.Api.info("you entered:" + e.value); // Do something with the value ..
});
