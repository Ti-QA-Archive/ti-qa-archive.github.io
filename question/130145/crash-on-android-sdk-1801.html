  • Thanks Ben for your help, i've tried to clean the project but it doesn't work.
    Here is the code for the tabgroup 

    /**
 * Navigation tabs group
 *
 * @file NavigationTabGroup.js
 */
(function(){
    rbw.ui.createTabGroup = function() {
        var navigationtabGroup = Titanium.UI.createTabGroup();

        if(Ti.Platform.osname === "iphone") {
            navigationtabGroup.opacity = 0;
        }

        var tabs = appCfg.getTabs();
        for(var i = 0; i < tabs.length; i++) {
            var tab_id                     = tabs.item(i).getElementsByTagName("id").item(0).text;
            var tab_xmlfile             = tabs.item(i).getElementsByTagName("id").item(0).getAttribute("xml");
            var tab_title                 = tabs.item(i).getElementsByTagName("title").item(0).text;
            var tab_icon                 = tabs.item(i).getElementsByTagName("icon").item(0).text;
            var tab_icon_active         = tabs.item(i).getElementsByTagName("activeicon").item(0).text;
            var tab_window                 = tabs.item(i).getElementsByTagName("window").item(0).text;
            var tab_window_className     = tabs.item(i).getElementsByTagName("window").item(0).getAttribute("className");
            var tab_window_id             = tabs.item(i).getElementsByTagName("window").item(0).getAttribute("id");
            var tab_window_xmlkey         = tabs.item(i).getElementsByTagName("xmlkey").item(0).text;

            var tab = Titanium.UI.createTab({
                title                 : tab_title,
                icon                 : tab_icon,
                tab_xmlfile         : tab_xmlfile,
                tab_window_xmlkey     : tab_window_xmlkey,
                tab_window             : tab_window
            });

            navigationtabGroup.addTab(tab);
        };

        navigationtabGroup.addEventListener('open', function(e) {
            Ti.API.info('Event:open navigationtabGroup');
            if(Ti.Platform.osname === "iphone") {
                navigationtabGroup.tabs[0].myCustomTabBar.show();
                navigationtabGroup.tabs[1].window.open();
                navigationtabGroup.tabs[2].window.open();
                navigationtabGroup.tabs[4].window.open();
                navigationtabGroup.tabs[3].window.open();
            }

            navigationtabGroup.setActiveTab(0);

            startpage.close({
                opacity : 0,
                duration : 350
            });
            navigationtabGroup.animate({
                opacity : 1,
                duration : 500
            });    
        });

        return navigationtabGroup;
    }
})();

    It works on iOS.
