  • Here's my login code which works perfectly in all cases except when the app is deleted in the users Facebook account.

    fb.appid = 'xxxxxxxxxxxxxxxxxx';
fb.permissions = ['read_stream'];
fb.forceDialogAuth = false;
fb.addEventListener('login', function(e) {
    if (e.success) {
        fb.reauthorize(['email'], 'me', function(res) {
            if (res && res.success) {
                //Login successful
            } else {
                alert(res.error);
            }
        });
    } else if (e.error) {
        alert(e.error);
    } else if (e.cancelled) {
        alert("Canceled");
    }
});

    Titanium SDK 3.1.3
    Studio 3.1.3.201309132423
    ios7 on device
