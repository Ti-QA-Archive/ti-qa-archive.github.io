    Hello Parijat,

    See, this code is working great for me…

    var self = Ti.UI.currentWindow;
var dataArray = [];

self.addEventListener('load', function() {
    selfView.visible = false;
    secnView.visible = true;
});

var selfView = Ti.UI.createView({
    backgroundColor : 'red'
});

var nameField = Ti.UI.createTextField({
    height : 'auto',
    top : 80,
    left : 130,
    width : 100,
    hintText : 'Enter'

});

var btn = Ti.UI.createButton({
    title : 'button',
    name : 'hello'
});

btn.addEventListener('click', function(e) {

    var table1 = Ti.UI.createTableView();

    dataArray.push(Ti.UI.createTableViewRow({
        title : nameField.value
    }));

    table1.setData(dataArray);

    secnView.add(table1);
    secnView.visible = true;
    selfView.visible = false;

});

selfView.add(btn);
selfView.add(nameField);

self.add(selfView);

var secnView = Ti.UI.createView({
    backgroundColor : 'white',
    visible : false
});

var b1 = Titanium.UI.createLabel({
    text : 'back',
    bottom : 20,
    width : 50,
    height : 50,
    backgroundColor : 'green'

});

b1.addEventListener('click', function() {
    secnView.visible = false;
    selfView.visible = true;
});

secnView.add(b1);
self.add(secnView);

    • Parijat, also try this code, back button also working great….

      
var self = Ti.UI.currentWindow;
var dataArray = [];
var dataArray2 = [];

var selfView = Ti.UI.createView({
    backgroundColor : 'red'
});

var nameField = Ti.UI.createTextField({
    height : 'auto',
    top : 80,
    left : 130,
    width : 100,
    hintText : 'Enter'

});

var btn = Ti.UI.createButton({
    title : 'button',
    name : 'hello'
});

btn.addEventListener('click', function(e) {


    dataArray2.push(nameField.value);

    var table1 = Ti.UI.createTableView({top : 0,bottom : 80});

    for ( i = 0; i < dataArray2.length; i++) {
        dataArray.push(Ti.UI.createTableViewRow({
            title : dataArray2[i]
        }));
    }
    table1.setData(dataArray);

    secnView.add(table1);
    secnView.visible = true;
    selfView.visible = false;
    nameField.blur();

});

selfView.add(btn);
selfView.add(nameField);

self.add(selfView);

var secnView = Ti.UI.createView({
    backgroundColor : 'white',
    visible : false
});

var b1 = Titanium.UI.createLabel({
    text : 'back',
    bottom : 20,
    width : 50,
    height : 50,
    backgroundColor : 'green'

});

b1.addEventListener('click', function() {
    dataArray = [];
    secnView.visible = false;
    selfView.visible = true;
    nameField.value = '';
    nameField.focus();    
});

secnView.add(b1);
self.add(secnView);

self.addEventListener('focus', function() {
    nameField.focus();
    selfView.visible = true;
    secnView.visible = false;
});
