  • Do you handle window property of the tabs? If not, the problem could come from here.
    By the way, I found a workaround (which is quite ugly) : 

    //
// Using items as a function returning an array
// make it not transformed into a LJavaLang.Object
// so that it can be used to create a tab object
//
var items = function() {
    return [{
        title: 'First window',
         window: "/window1.js"
    },{
        title: 'Second window',
        window: "/window2.js"
    }];
}

var group = Titanium.UI.createTabGroup();

var tab;
for (var i = 0, iLength = items.length ; i < iLength ; i++) {
    // Calling the items function returning an array here
    tab = Titanium.UI.createTab(items()[i]);
    group.addTab(tab);
}
tab = null;

group.open();

    In addition, I would suggest you (quite as Christian did) to store your tabs into separate variables, because I'm not sure that Titanium will keep a reference on the tab if you destroy or override the variable.
    So finally, the code I would suggest you to have is :

    var items = function() {
    return [{
        title: 'First window',
         window: "/window1.js"
    },{
        title: 'Second window',
        window: "/window2.js"
    }];
}

var group = Titanium.UI.createTabGroup();

var tabs = [];
for (var i = 0, iLength = items.length ; i < iLength ; i++) {
    var tab = Titanium.UI.createTab(items()[i]);
    group.addTab(tab);
    tabs.push(tab); // or tabs[i] = tab;
}

group.open();

    I hope this will work for you.
