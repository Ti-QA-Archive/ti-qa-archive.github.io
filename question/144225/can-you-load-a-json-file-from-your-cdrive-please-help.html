var json , Inbox , Task , i , row , TaskName;

var table = Ti.UI.createTableView({ top:'12%'});
var tableData = [];

var xhr = Ti.Network.createHTTPClient({});
    xhr.open("GET", url);
        xhr.onload =  function() {
