how do i get IDs from ACS call

hi,

function getUserID(){
var id;
    Cloud.Users.query({
            page: 1,
            per_page: 10,
            where: {
                age: { '$gt': 28 },
            }
        }, function (e) {
            if (e.success) {
            id=e.users[1].id;    
            } else {
                alert('Error:\\n' +
                    ((e.error && e.message) || JSON.stringify(e)));
            }
        });
    return id;
}

i understand i cannot do that. is there another way to do it?

  • function getUserID(callback){

    Cloud.Users.query({
            page: 1,
            per_page: 10,
            where: {
                age: { '$gt': 28 },
            }
        }, function (e) {
            if (e.success) {
            callback(e.users[1].id);   
            } else {
                alert('Error:\\n' +
                    ((e.error && e.message) || JSON.stringify(e)));
            }
        });

}

    then

    getUserId(function(id){
    alert(id);
});
