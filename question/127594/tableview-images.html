  • The simplified version looks like the following:

    function setTableRow(text, checked){
    var checkBox = Ti.UI.createImageView({
    image: 'active.jpg'
    });

    var textLabel = Ti.UI.createLabel({
text: text});

var row = Ti.UI.createTableRow();

row.add(checkBox);
row.add(textLabel);

return row;

    }

    function updateTableView() {

    data = [];


for (var i=0; i<tableData.size; i++) {
    var tableRow = setTableRow(tableData[i].text, tableData[i].checked), 

    data.push(tableRow);
}

tableView.data = data;

currentWindow.add(tableView);

    }

    Ti.App.addEventListener('customEvent', updateTableView);

    -event triggered when something modifies the tableData vector
