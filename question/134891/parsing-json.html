  • Your onload should be modified like this:

    xhr.onload = function() {
    var images = [];
    for( i = 0; i < tn.length; i++) {
        // view.images = tn[i].url.thumb;
        images.push(tn[i].url.thumb);
    }

    view.images = images;
}
