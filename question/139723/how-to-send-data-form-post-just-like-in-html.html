How To send data (form) POST, just like in HTML

Hello all !

it's not a question but an answer !

you can see the code below, let a comment if this helps :)

TitaniumJS File

    //XHR OBJECT
    var xhrURL = "http://www.mydomain.fr/scripts/contactFrom.php";
    var xhr_object = Ti.Network.createHTTPClient();
    xhr_object.open("POST", xhrURL, true);

    //XHR HEADERS
    xhr_object.setRequestHeader( "Content-type" , "application/x-www-form-urlencoded" );
    xhr_object.setRequestHeader( "Content-length" , postData.length);
    xhr_object.setRequestHeader( "Connection" , "close" );

    //YOUR DATA
    var postData = "var1="+var1Value;
    postData+= "&var2="+var2Value;
    postData+= "&var3="+var3Value;

    //SEND YOUR DATA TO XHR OPENED OBJECT
    xhr_object.send(postData);

    //IF LOAD OK
    xhr_object.onload = function(){ 
        alert(this.responseText) //Here you should see all your posted vars
    }

    //IF LOAD ERROR
    xhr_object.onerror = function(){ 
        alert("Houla ! Big Problem !")
    }
