ti.canvas and fillText, prints the text upside-down

Hi together,

I try to print text with the ti.canvas-module for iphone. In my example the line and the text are visible, but the text is printed upside-down. If I change the size by .font("Arial", -12) the upside-down is changing into mirror-written text. Has anbody a hint how to print the text correct?

  var canvas = require('ti.canvas');
  canvasView = canvas.createView();
  canvasView.Width  = 1200;
  canvasView.Height = 900;

  canvasView.begin();

  canvasView.lineWidth(3);
  canvasView.strokeStyle("#880088");
  canvasView.moveTo(  50,  600);   
  canvasView.lineTo(1150,  600);

  canvasView.font("Arial", 12);
  canvasView.fillStyle("#008800"); 
  canvasView.lineWidth(1);
  canvasView.fillText("Test-Text", 100, 100);

  canvasView.stroke();

  canvasView.commit();

  Titanium.Media.saveToPhotoGallery(canvasView.toImage());
