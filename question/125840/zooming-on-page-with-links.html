Zooming on page with links

I have a page that is able to do zoom and pan. But the problem is that the link on the page does not change position when i zoom or pan the page.

any idea how do i make the button or link change position during the zooming or panning stage?

  • Accepted Answer

    In your code, add pp_link to mainView or to scrollView, not to window.

    var mainView = Titanium.UI.createView({backgroundImage:'../images/contact/contact01-thedeveloper-overview.jpg'});



var scrollView = Titanium.UI.createScrollView({
    views:[mainView],
    showPagingControl:true,
    pagingControlColor:"transparent",
    clipViews:false,
    maxZoomScale:3,
    currentPage:0,
});


var win = Titanium.UI.currentWindow;
scrollView.add(mainView);
win.add(scrollView);



var pp_link = Titanium.UI.createButton({
    //image:'../images/right.png',
    height:30,
    width:100,
    top:705,
    left:570,
    style:6
});

pp_link.addEventListener('click', function(e)
{
    var win = Titanium.UI.createWindow({
            url:'contact_developer_p.js',
} );
    win.open({transition:Ti.UI.iPhone.AnimationStyle.CURL_DOWN});
    win.backgroundColor='e3e3e5';
});

mainView.add(pp_link);
