    Hi Ahmad Hosseini,

    Thanks for your question.

    I've created a demo test case for your issue. Please check the below code and see if it helps you.

    var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    layout: 'vertical'
});

var count=0;
var flag=0;
//1
var aButton1 = Ti.UI.createButton({
    title : 'Button 1',
    height : 'auto',
    width : 'auto',
    top : 10,
});
aButton1.addEventListener('click', function() {
    if(flag!=0)
    {
        flag+= count;
        count=0;
        ++flag;
        if(flag==5)
        {
            alert("Oh no, incorrect order !!!");
            flag=0;
            count=0;
        }    
    }
    else if(count==0)
    {
        count++;
    }

});
win.add(aButton1);
//2
var aButton2 = Ti.UI.createButton({
    title : 'Button 2',
    height : 'auto',
    width : 'auto',
    top : 10,
});
aButton2.addEventListener('click', function() {
    if(count==1)
    {
        count++;
    }
    else
    {
        flag+= count;
        count=0;
        ++flag;
        if(flag==5)
        {
            alert("Oh no, inorrect order !!!");
            flag=0;
            count=0;
        }    
    }
});
win.add(aButton2);
//3
var aButton3 = Ti.UI.createButton({
    title : 'Button 3',
    height : 'auto',
    width : 'auto',
    top : 10,
});
aButton3.addEventListener('click', function() {
    if(count==2)
    {
        count++;
    }    
    else
    {
        flag+= count;
        count=0;
        ++flag;
        if(flag==5)
        {
            alert("Oh no, inorrect order !!!");
            flag=0;
            count=0;
        }    
    }
});
win.add(aButton3);
//4
var aButton4 = Ti.UI.createButton({
    title : 'Button 4',
    height : 'auto',
    width : 'auto',
    top : 10,
});
aButton4.addEventListener('click', function() {
    if(count==3)
    {
        count++;
    }
    else
    {
        flag+= count;
        count=0;
        ++flag;
        if(flag==5)
        {
            alert("Oh no, inorrect order !!!");
            flag=0;
            count=0;
        }    
    }
});
win.add(aButton4);
//5
var aButton5 = Ti.UI.createButton({
    title : 'Button 5',
    height : 'auto',
    width : 'auto',
    top : 10,
});
aButton5.addEventListener('click', function() {
    if(count==4)
    {
        count++;
        alert("Yeah, Correct order !!!");
        count=0;
        flag=0;
    }
    else
    {
        flag+= count;
        count=0;
        ++flag;
        if(flag==5)
        {
            alert("Oh no, inorrect order !!!");
            flag=0;
            count=0;
        }    
    }    
});
win.add(aButton5);

win.open();

    It's a simple logic and you can modify it according to your requirement. Hope it''ll help you to solve your problem. If you have any concern about my answer, then feel free to let us know. And if it resolves your problem, then mark this answer as accepted.
