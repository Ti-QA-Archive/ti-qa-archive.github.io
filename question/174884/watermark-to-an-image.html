  • Hi, I share this code where an image is superimposed on the other.

    var win = Ti.UI.createWindow({
    layout:"vertical",        
});
var image1 = Ti.UI.createImageView({
    image: "img/titans.png",
    width:"80%",  
    });


var image2 = Ti.UI.createImageView({
    image:"img/travel.png",
    width:128,
    top:-100
});


var tempImage = Ti.UI.createView({layout:"vertical", width:'100%' });

tempImage.add(image1);
tempImage.add(image2);

win.add(tempImage);
win.open();
