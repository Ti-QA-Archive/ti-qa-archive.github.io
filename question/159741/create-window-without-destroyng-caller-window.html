  • Yes Francesco Take a look at this example:

    I made a demo that opens new window by clicking on open new window button and when you select any item from window 2 it will updated to window1.

    index.xml

    <Alloy>
    <Window id="newWindow" title="New Window" backgroundColor="white">
        <Picker id="newpicker" top="50" selectionIndicator="true" onChange="changeValue" useSpinner="true" >
            <PickerColumn id="column1">
                <PickerRow title="new Bananas"/>
                <PickerRow title="new Strawberries"/>
                <PickerRow title="new Mangos"/>
                <PickerRow title="new Grapes"/>
            </PickerColumn>
        </Picker>

        <Button bottom="20dp" onClick="addValue">Add</Button>
    </Window>
</Alloy>

    index.js

    $.win.open();

var openNewWindow = function(){
    var newWindow = Alloy.createController('newWin').getView();
    newWindow.changeValue=function(val){
        var row = Ti.UI.createPickerRow({
        title : val
    });
        $.picker.add(row);
    };
    newWindow.open();
};

    newWin.xml

    <Alloy>
    <Window id="newWindow" title="New Window">
        <Picker id="newpicker" top="50" selectionIndicator="true" onChange="changeValue" useSpinner="true" >
            <PickerColumn id="column1">
                <PickerRow title="new Bananas"/>
                <PickerRow title="new Strawberries"/>
                <PickerRow title="new Mangos"/>
                <PickerRow title="new Grapes"/>
            </PickerColumn>
        </Picker>

        <Button bottom="20dp" onClick="addValue">Add</Button>
    </Window>
</Alloy>

    newWin.js

    var selValue = "";
var addValue = function(){
    $.newWindow.changeValue(selValue);
    $.newWindow.close();
};

var changeValue = function(e){
    Ti.API.info("Text : " +  JSON.stringify(e));
    selValue = e.row.title;
};

    Best of luck, Happy coding :)
