  • Hi Sagar!
    Please Use Array for this purpose.

    var images = ["splashes/splash.png", "splashes/splash2.png", "splashes/splash3.png", "splashes/splash4.png", "splashes/splash5.png", "splashes/splash6.png", "splashes/splash7.png", "splashes/splash8.png", "splashes/splash9.png", "splashes/splash10.png", "splashes/splash11.png", "splashes/splash12.png", "splashes/splash13.png", "splashes/splash14.png", "splashes/splash15.png", "splashes/splash16.png", "splashes/splash17.png", "splashes/splash18.png", "splashes/splash19.png", "splashes/splash20.png", "splashes/splash21.png", "splashes/splash22.png"];
        var image_array_index = Math.round(Math.random() * images.length - 1);
        if(image_array_index == -1) {
            image_array_index = 0;
        }
        var splashScreen = Ti.UI.createImageView({
            image : images[image_array_index],
            height : app_height,
            width : app_width
        });
        win1.add(splashScreen);
