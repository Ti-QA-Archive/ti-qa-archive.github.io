    Alloy.Collections.toilets = Alloy.createCollection('Toilets');

function newToiletRecord(name, address, lat, lng) {
    // alert("loading the data");

    var toilet = Alloy.Collections.toilets;

    // Create a new model for the todo collection
    var new_toilet = Alloy.createModel('Toilets', {
        "name" : name,
        "address" : address,
        "lat" : lat,
        "lng" : lng,
        "features" : "",
        "location" : "",
        "accessability" : "",
        "opening_hours" : "",
        "additional_features" : "",
        "notes" : ""
    });

    // add new model to the global collection
    toilet.add(new_toilet);

    // save the model to persistent storage
    new_toilet.save();
}

function loadData() {
    var toilet = Alloy.Collections.toilets;
    toilet.fetch();
    if (toilet.length == 0) {
        newToiletRecord("Toilet 1", "3 Wallis St", "35.67", "134.67");
        newToiletRecord("Toilet 2", "3 Blue St", "35.67", "134.67");
    }
}
