Android Video- black screen while loading issue

Platform Android, while loading video off of a url I see a black screen that makes the phone appear off.

Is there a way to put a loading screen in there?
I'm also looking for a way to reduce the loading time for videos.

var movieLabel = Titanium.UI.createLabel({
        text:'Loading movie...',
        width:'auto',
        height:50,
        color:'#FFF',
        font:{fontSize:24,fontFamily:'Helvetica Neue'}
    });    


// Add code to the event listener that plays your movie to display the label above and 
// then poll the activeMovie.playbackState every 1 second.
//
//The code below will listen for a click in a tableview and run the function

tableview.addEventListener('click', function(e){
    Titanium.UI.currentWindow.add(activeMovie);
    movieLabel.show();
        activeMovie.play();
    activeMovie.add(movieLabel);

        var interval_id = setInterval(function(){    
       if(activeMovie.playbackState == Titanium.Media.VIDEO_LOAD_STATE_PLAYABLE)
        {        
            movieLabel.hide();
            clearInterval(interval_id);
            activeMovie.play();    
        }
       }, 1000);

    });
