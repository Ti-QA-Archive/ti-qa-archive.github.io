function dB () {

    dB.getCustomCard = function () {
        var db = Ti.Database.open('whatcards');
        var cardDb = db.execute('SELECT * FROM customcards ORDER BY id');

        var cardArray = [];
        while (cardDb.isValidRow()) {
            cardArray.push({id:cardDb.fieldByName('id'),text:cardDb.fieldByName('card')}); 
            cardDb.next();
        }
        cardDb.close(); 

        return cardArray;
    }


    dB.getAllCards = function () {
        var db = Titanium.Database.install('/db/whatcards', 'whatcards');
        var cardDb = db.execute('SELECT * FROM (SELECT * FROM whatcards UNION SELECT * FROM customcards) ORDER BY RANDOM()');

        var cardArray = [];
        while (cardDb.isValidRow()) {
            cardArray.push({id:cardDb.fieldByName('id'),text:cardDb.fieldByName('card')}); 
            cardDb.next();
        }
        cardDb.close(); 
        return cardArray;
    }
}



module.exports = dB;

  • Thanks to @Malcolm Hollingsworth for the incredible help of debugging this issue. The width in Titanium has a bug if you use a shadow offset on the iPhone. If you want a shadow effect of any sort with multiple lines, make sure to use labels, a background and foreground so you dont have to deal with this issue.
