  • Accepted Answer

    //check if city already set.
var city = Ti.App.Properties.getString("City");
if(!city || city.length == 0) {
  //city not set yet, create buttons to let them choose.
  var nyButton = Ti.UI.createButton({ title: "NY" });
  nyButton.addEventListener("click", function() { Ti.App.Properties.setString("City", "NY"); });
  Ti.UI.currentWindow.add(nyButton);
  var caButton = Ti.UI.createButton({ title: "CA" });
  caButton.addEventListener("click", function() { Ti.App.Properties.setString("City", "CA"); });
  Ti.UI.currentWindow.add(caButton);
}
else {
  //city set, do something else.
}
