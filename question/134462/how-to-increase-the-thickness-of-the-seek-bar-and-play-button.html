    (function() {

     sound.UI = {};

    sound.UI.create = function(){
// /



            Titanium.Media.audioSessionMode = Titanium.Media.AUDIO_SESSION_MODE_AMBIENT;


            var dir = Titanium.Filesystem.getFile(Titanium.Filesystem.resourcesDirectory,'sound/');
            var music=dir.getDirectoryListing();
            var pos=-1;
            var sound1 = null;

            function forw(){    
                if (sound1!=null){

                    sound1.stop();
                }

                pos=pos+1;
                if (pos == music.length){
                    pos=0;
                }

                sound1 = Titanium.Media.createSound();

                sound1.url='sound/'+music[pos];
                sound1.preload = true;
                sound1.looping = false;
                sound1.play();
            //    Ti.API.info(sound1.url);

                    sound1.addEventListener('complete', function()
                    {
                        Ti.App.fireEvent('play');

                    });



            }
        Ti.App.addEventListener('play',function(){
        //    Ti.API.info("complete");
        //    Ti.API.info(sound1.url);
            //var sound1=null;
            forw();
        });

        var soundView = Titanium.UI.createView({
            left: 0,
            top: 378,
            width: 194,
           width: 130,
                height: 34,
                backgroundColor: '#000000',
                zIndex:10
            });


        //
        // PLAY
        //

        var play = Titanium.UI.createImageView({
            left:0,
            top:0,
            width: 32,
            height: 32,
            image:'images/sound/play.png'
        });
        soundView.add(play);

        play.addEventListener('click', function()
        {
            sound1.play();
            //pb.max = sound.duration;
        });
        soundView.add(play);

        //
        // PAUSE
        //
        var pause = Titanium.UI.createImageView({
            left: 29,
            top: 0,
            width: 32,
            height: 32,
             image:'images/sound/pause.png'
        });
        soundView.add(pause);

        pause.addEventListener('click', function()
        {
            sound1.pause();
        });
        var ff = Titanium.UI.createImageView({
            left: 58,
            top: 0,
            width: 32,
            height: 32,
            image:'images/sound/ff.png'

        });
        soundView.add(ff);

        ff.addEventListener('click', function()
        {
            forw();
        });



        var sound_ico= Titanium.UI.createImageView({
            left: 98,
            top: 0,
            width: 32,
            height: 32,
            image:'images/sound/sound.png'
        });
        soundView.add(sound_ico);

        soundView.animate({right:0, left:-380, duration:2000});

        var anim=true;
            sound_ico.addEventListener('click', function()
            {
                if(!anim){
                soundView.animate({right:0, left:-380, duration:300});
                anim=true;
              } else {
                  soundView.animate({right:0, left:-194, duration:300});
                  anim=false;
              }
            });


        forw();
        return soundView;
    };


})();
