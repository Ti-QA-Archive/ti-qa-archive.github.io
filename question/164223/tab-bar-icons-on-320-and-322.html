  • Hello user 4683,

    Thanks for your sample code. Here I have added sample code for your issues. In my sample code, I add a navBarHidden property in the Tabgroup and also I add a label, dark picker and a text field with colors. The icon is present in the Tab Group. Please test my code in your environments. I hope it solves your problem.

    Check attachment

    Test Code:

     var tabGroup = Titanium.UI.createTabGroup({
 navBarHidden:true
});

var win1 = Titanium.UI.createWindow({
 title : 'Search',
 backgroundColor : '#fff',
 layout:'vertical'

});
SearchTradesman_Tab = Titanium.UI.createTab({
 icon : 'images/20.png',
 title : 'Tab 1',
 window : win1,

});

var picker = Ti.UI.createPicker({
 top : 50,
 useSpinner : true
});
picker.selectionIndicator = true;

var fruit = ['Bananas', 'Strawberries', 'Mangos', 'Grapes'];
var color = ['red', 'green', 'blue', 'orange'];

var column1 = Ti.UI.createPickerColumn();

for (var i = 0, ilen = fruit.length; i < ilen; i++) {
 var row = Ti.UI.createPickerRow({
  title : fruit[i]
 });
 column1.addRow(row);
}

var column2 = Ti.UI.createPickerColumn();

for (var i = 0, ilen = color.length; i < ilen; i++) {
 var row = Ti.UI.createPickerRow({
  title : color[i]
 });
 column2.addRow(row);
}

picker.add([column1, column2]);

picker.setSelectedRow(0, 2, false);
// select Mangos
picker.setSelectedRow(1, 3, false);
// select Orange// select Mangos
win1.add(picker);


// Create a Label.
var aLabel = Ti.UI.createLabel({
    text : 'welcome',
    color : 'red',
    top : 10,
    textAlign : 'center'
});

// Add to the parent view.
win1.add(aLabel);


// Create a TextField.
var aTextField = Ti.UI.createTextField({
    height : 35,
    top : 10,
    left : 40,
    width : 240,
    hintText : 'This is hint text',
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    backgroundColor : 'pink',
});


// Add to the parent view.
win1.add(aTextField);


tabGroup.addTab(SearchTradesman_Tab);
// open tab group
tabGroup.open();
