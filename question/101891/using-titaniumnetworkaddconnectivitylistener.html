    if(Ti.Network.online) { // true if online }

  • Ok so I found this with my search of google. Does this look like something I would need to do?

    function Uonline()
{   
    //listen for any network changes
    Titanium.Network.addConnectivityListener('change', function(e) 
    {     
        if (e.online)
        {
            $("#img_download").html('<a href="#"><img src="images/update/download.png" class="rollover" /></a>');
            $("#img_upload").html('<a href="#"><img src="images/update/upload.png" class="rollover" /></a>');    
        }   
        else
        {
            $("#img_download").html('<img src="images/update/download-disabled.png" />"');
            $("#img_upload").html('<img src="images/update/upload-disabled.png" />"');    
        }    
    });
}
