                var lineView = Ti.UI.createView({
                        height:'1',
                        width:'100%',
                        bottom:0,
                        left:0,
                        backgroundColor:'#666'
                });
                row.add(lineView);

                var trophyImage = Ti.UI.createImageView({
                    image:'images/timerpic.png',
                    width:15,
                    height:15,
                    left:5
                });
                row.add(trophyImage);

                rows.next();

        rows.close();
        db.close();
        tablelogg.setData(tableData);
