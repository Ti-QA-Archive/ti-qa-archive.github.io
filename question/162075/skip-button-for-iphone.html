Skip Button for iPhone

I have a few statements in an array and I'd like to cycle through them with a skip button. I've added an event listener to the button but it only seems to effect the button and not my statements. Here's the code example:

//Label for Statement
var statementLabel = Ti.UI.createLabel({
    text: 'Statements!',
    font: {fontSize:24},
    backgroundColor: '#F66',
    textAlign: "center",
    borderRadius: 4,
    top: 100,
    left: 10,
    width: 300,
    height: "auto"
});


//Array for Different Statements
var statements = [
    {s:"Hello"},
    {s:"Goodbye"},
    {s:"Left"},
    {s:"Right"},
    {s:"Up"}
];

var count = 0;

var answer;

//The Button

skipS = Ti.UI.createButton({
    title: ' Skip ',
    font:{fontSize:15,fontFamily:'Helvectica Neue'},
    textAlign: 'center',
    top: 119,
    left: 100,
    width: 'auto',
    borderColor: '#007FFF',
    borderRadius: 7    
});

skipS.addEventListener("click", function(){
    count++;

    if (count % 1 ==0){
        //code to show statements
        var data = question[Math.floor(Math.random() * question.length)];
        this.text = data.q;
        this.backgroundColor = '#666';
    }

view3.add(statementLabel);
view3.add(skipS);

How can I connect the button to cycle through the statements when clicked?
