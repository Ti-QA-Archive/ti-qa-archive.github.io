var view = Titanium.UI.createScrollView({
        contentWidth:'auto',
        contentHeight:'auto',
        //height:300,
        showVerticalScrollIndicator:true,
        showHorizontalScrollIndicator:true,
        top: 0,

    });

    backgroundColor:'#fff',
    layout:'horizontal'
});


var quantitySliderLabel = Titanium.UI.createLabel({
    text:'Quantity' ,
    color:'#000',
    font:{
        fontFamily:'Helvetica Neue',
        fontSize:15
    },
    textAlign:'center',
    top:10,
    width:70,
    height:'auto'
});



var quantitySliderValue = Titanium.UI.createLabel({
    text:'0' ,
    color:'#000',
    font:{
        fontFamily:'Helvetica Neue',
        fontSize:15
    },
    textAlign:'center',
    top:10,
    width:40,
    height:'auto'
});

var quantitySlider = Titanium.UI.createSlider({
    min:0,
    max:100,

{
