  • First you need to assign the javascript file to each image. I highly recommend that you use commonJS module for each view like this:

    firstWindow.js

    function firstWindow(){
    var self = {};
    var ww = Ti.UI.createWindow();
    function getView(){
        return ww;
    }
    return self;

}
module.exports = firstWindow;

    then you link to the appropriate image:

    var view1 = Ti.UI.createView({
    backgroundImage:'../images/0.jpg',
    //borderRadius:20,borderWidth:1,borderColor:'#336699',
    width:320,
    height:240,
    left:10,
    jsFile: 'firstWindow'
});

    then you add an event listener to the main WINDOW :

    win.addEventListener('click', function(e){
    if(e.source.jsFile){
        var NewWin = require(e.source.jsFile);
        var win1 = new NewWin();
        win1.getView().open();
    }
}
