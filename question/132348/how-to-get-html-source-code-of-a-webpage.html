  • var xhr = Titanium.Network.createHTTPClient();
xhr.onload = function() {
    Titanium.API.info(xhr.responseText);
};
xhr.onerror = function() {
    Titanium.API.info('error');
};
xhr.open("GET","http://www.cnn.com");
xhr.send();
