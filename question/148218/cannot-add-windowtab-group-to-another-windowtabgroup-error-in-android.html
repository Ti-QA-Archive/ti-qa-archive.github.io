cannot add window/tab group to another window/tabgroup error in Android

Hi,
I am building a simple android application which have an actionBar in it.
Here is my code.

In app.js

var mywin = Ti.UI.createWindow({
    title:'main Window',
    backgroundColor:'red',
    exitOnClose:true
});

var myBtn = Ti.UI.createWindow({
   title:'hello' 
});

myBtn.addEventListener('click',function(e){
    var newWin = require('/newWin');
    new newWin().open();
});

mywin.add(myBtn);
mywin.open();

In newWin.js

function newWin()
{
    var self = Ti.UI.createWindow({
        title:'hello',
        backgroundColor: 'green'
    });
    var actionBar;
    self.addEventListener('open',function(e){
      actionBar = self.activity.actionBar;
      actionBar.icon = "/KS_nav_ui.png";
      actionBar.displayHomeAsUp = true;
      actionBar.setOnHomeIconItemSelected = function() {
          self.close();
      }
    });
    return self;
}
module.exports = newWin;

while running this I get the following error

cannot add window&#x2F;tab group to another window&#x2F;tabgroup. I dont know what it is.

And am I doing the actionBar implementation correctly?
