HeaderView of TableViewSection doesn't look right in landscape mode

Hello everyone,

I'm trying to create a tableView with 2 sections in it. Both have a custom headerView in it. The first section also has an image in the headerView, the second headerView is just a label. When I'm holding my phone in portrait mode it looks ok. But when I switch to landscape mode the image in the first headerView also shows up in the second headerView, so I see the image in both headerViews while it only needs to be in the first headerView. I really don't understand why titanium does this. Anyone got an idea?

This is the code I use to create the tableView:

function createTableHeader(headerTitle)
{
    var viewHeader = Ti.UI.createView({
        height: 40,
        width: Titanium.UI.FILL,
    });
    var labelHeader = Ti.UI.createLabel({
        text:headerTitle,
        font:{fontSize:26,fontWeight:'bold'},
        color:'#ffffff',
        textAlign:'left',
        left:0,
        height:40
    });

    viewHeader.add(labelHeader);

    return viewHeader;
}

function createTable()
{
    var header1 = createTableHeader("Taken");
    var header2 = createTableHeader("Not yet taken");

    //Here I'm adding the image to the first header. In landscape mode this picture gets added to the second header aswell somehow, it shouldn't do that..
    var iconsHeader = Ti.UI.createImageView({
        image: '/quizicons.png',
        top: 5,
        right: 10,
        height: 30,
        width: 77
    });
    header1.add(iconsHeader);

    var sec1 = Ti.UI.createTableViewSection({headerView:header1});
    sec1.add(Ti.UI.createTableViewRow({title:'row',backgroundColor:'#fff', color: '#000'}));
    sec1.add(Ti.UI.createTableViewRow({title:'blaat', backgroundColor:'#fff', color: '#000'}));
    sec1.add(Ti.UI.createTableViewRow({title:'test', backgroundColor:'#fff', color: '#000'}));
    sec1.add(Ti.UI.createTableViewRow({title:'nogwat', backgroundColor:'#fff', color: '#000'}));

    var sec2 = Ti.UI.createTableViewSection({headerView:header2});
    sec2.add(Ti.UI.createTableViewRow({title:'nog een sectie', backgroundColor:'#fff', color: '#000'}));
    sec2.add(Ti.UI.createTableViewRow({title:'En weer een', backgroundColor:'#fff', color: '#000'}));
    sec2.add(Ti.UI.createTableViewRow({title:'De laatste', backgroundColor:'#fff', color: '#000'}));

    var tableview = Ti.UI.createTableView({
      data: [sec1, sec2],
      seperatorColor: '#ff0000'
    });

    return tableview;
}
