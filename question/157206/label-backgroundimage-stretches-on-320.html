    Try it with backgroundTopCap and backgroundLeftCap, e.g.

    var imageText2 = Titanium.UI.createLabel({
    bottom:0,
    left:50,
    width:100,
    height:40,
    backgroundTopCap: 15, 
    backgroundLeftCap: 10,
    backgroundImage:'images/BalloonInputField.png'
});

    • Did you try this? You can change Ti.UI.SIZE with the real width and height of the button. Also width of container view can be set to fix width. 

      var navBar = Ti.UI.createView({
    height: 44,
    backgroundColor:'#000'
});

var navButton = Ti.UI.createButton({
    width: Ti.UI.SIZE,
    height: Ti.UI.SIZE,
    backgroundImage: theme.icons.logoImage
});

navBar.add(navButton)

win.titleControl = navBar;
