left button click doesn't work on Android...?

Hi All.
I would really love some help with this little strange issue.
I have a mapView with 3 annotations which each should return a different value, close the view and open another view when the user clicks the leftButton. It works fine on iPhone but on Android nothing happens when I click the leftButton. Anyone who knows how to fix this? Help is much appreciated ;) I am providing the code for the mapView, annotations and the eventListener below.
Thanks so much for help and time!
Cheers,
Louise


var hubs = [
    {latitude:55.683543, longitude:12.504839, title:'myTitle1', subtitle:'mySubTitle1', customProperty:"no1"}, 
    {latitude:55.659916, longitude:12.591834, title:'myTitle2', subtitle:'mySubTitle2', customProperty:"no2"},
    {latitude:55.677772, longitude:12.617063, title:'myTitle3', subtitle:'mySubTitle3', customProperty:"no3"}
    ];

    var annotationObject = [];

    for (var u = 0; u < hubs.length; u++){
    annotationObject[u] = Titanium.Map.createAnnotation({
        latitude:hubs[u].latitude,
        longitude:hubs[u].longitude,
        title:hubs[u].title,
        subtitle:hubs[u].subtitle,
        image:'img1.png',
        leftButton:'img2.png',
        leftButtonText:'You have clicked ' + hubs[u].title,
        animate:true,
        customProperty:hubs[u].customProperty
    });
};

    var myOtherView = Ti.UI.createView({
    fullscreen:true,
    backgroundColor:'red'
    });

    var myMapView = Ti.Map.createView({
        top:0,
        left:0,
        width:windowWidth,
        height:windowHeight,
        mapType:Titanium.Map.STANDARD_TYPE,
        region:{
            latitude:myPosLat,
            longitude:myPosLong,
            latitudeDelta:1,
            longitudeDelta:1
        },
        animation:true,
        regionFit:false,
        userLocation:true,
        annotations:annotationObject    
    });

myMapView.addEventListener('click', function(e){
    if (e.clicksource === 'leftButton') {
    myLabel.show();
    myMapView.hide();
    myOtherView.show();
    comLabel.setText(e.annotation.leftButtonText);
    }
});
