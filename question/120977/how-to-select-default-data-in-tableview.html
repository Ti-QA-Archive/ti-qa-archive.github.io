How to Select Default Data in TableView

I have a splitview table with a master and detail view. By default, the detail side is blank when the page is opened until a master item is selected. How would one go about setting a default in the master side that would populate the detail when the page is opened. Here is my code for that section (not shown is the tabledata I already have defined):

    function tableClick(evt) {
        var evtData = {
            "row" : evt.index,
            "title": evt.row.title,
            "url": evt.row.url
        }
        Ti.App.fireEvent('app:rowClicked', evtData);

    }

    MASTER_WINDOW.init = function(){
        var tableview = Titanium.UI.createTableView({
        data:tableData,
        top:109,
        left:15,
        right:15,
        backgroundColor:'transparent',
        selectedBackgroundColor:'#991f34'
        //separatorStyle: Titanium.UI.iPhone.TableViewSeparatorStyle.NONE
        });

        var wrapView = Titanium.UI.createView({
    backgroundImage:'../images/side.jpg'
    });
        wrapView.add(tableview);
        win.add(wrapView);        

        tableview.addEventListener('click',tableClick);                        
    }

})();
MASTER_WINDOW.init();

  • Hi,

    As you have wired pretty much every thing when a tableview row is clicked, all you have to do is to call

    masterTableview.selectRow(0/*by default selects the first row*/);

    at the end which will in turn will fire the table click event and your detail view construction logic will take care of the rest.
