    Hi

    Titanium does not have a problem parse either of those examples. I have created a very simple test to show this;

    var test1 = '{"paths":{"head":"images/ss/heads/","neck":"images/ss/necks/","body":"images/ss/bodies/","shoulder":"images/ss/shoulders/"}}';
var test2 = '[{"paths":{"head":"images/ss/heads/","neck":"images/ss/necks/","body":"images/ss/bodies/","shoulder":"images/ss/shoulders/"}}]';
var json1 = JSON.parse(test1);
var json2 = JSON.parse(test2);
alert('test1: ' + json1.paths.head + '\ntest2: ' + json2[0].paths.head);

    It takes both of your example objects as TEXT and then parses them, display the same result in both ways.

    • Parsing from a file also works without issue;

      A simple reusable function to return the JSON data object from the contents of a file.

      var getData = function (strFileName) {
    var file1 = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory, strFileName);
    var test1 = file1.read().text;
    file1 = null;
    return JSON.parse(test1);
};

      Usage example (and test);

      var json1 = getData('test1.txt');
var json2 = getData('test2.txt');
var result = 'test1: ' + json1.paths.head + '\ntest2: ' + json2[0].paths.head;
alert(result);

      "body":"images/ss/bodies/",
      "shoulder":"images/ss/shoulders/"

      "body":"images/ss/bodies/",
      "shoulder":"images/ss/shoulders/"
   }

alert(d[0].paths.head);
