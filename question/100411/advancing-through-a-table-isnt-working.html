Advancing through a table isn't working.

Hi,

I'm making an iPhone app with a TableView, and for one of the rows in my table, hasChild is true, and "test" has a link to the new view I want to open when the row is clicked.

I then have an event listener that listens for someone to click that row in the table. Once it registers a click, the new view is linked to the "test" URL, and they should go to the new view.

Unfortunately, it's not working. I keep getting the following error in my output panel:

[WARN] Exception in event callback. {
expressionBeginOffset = 412;
expressionCaretOffset = 434;
expressionEndOffset = 439;
line = 22;
message = "Result of expression 'Titanium.UI.currentTab' [undefined] is not an object.";
name = TypeError;
sourceId = 240455872;
sourceURL = "file://localhost/timer/newtimer.js";
}

Here's what my code looks like:

var curWin = Titanium.UI.currentWindow;

var data = [
    {title:"Label", hasChild:true, test:"label.js"}//,
    {title:"Sound"},
    {title:"Timers"}
]

var table2 = Ti.UI.createTableView({
    style:Ti.UI.iPhone.TableViewStyle.GROUPED,
    data:data
});

table2.addEventListener('click', function(e)
{
    if (e.row.hasChild)
    {
        var win = Titanium.UI.createWindow({
            url:e.rowData.test,
            title:e.rowData.title
        });
        Titanium.UI.currentTab.open(win,{animated:true});
    }
});

curWin.add(table2);

Any ideas on how to fix this? Thanks.

    I see now what your problem is (since seeing your app.js): Your tableview in newtimer.js is in a modal window, meaning there is no tab at all.

    Modal windows can't open other windows. They just open and close.

    You might want to change the code structure you have, to make things easier:

    app.js :

    
// this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');

//Titanium.UI.iPhone.StatusBar.TRANSLUCENT_BLACK;

// create tab group
var tabGroup = Titanium.UI.createTabGroup();

// put the code for the initial win1 window in a separate js file in th timers folder:

var win1 = Titanium.UI.createWindow({  
    //id: 'win1',
    backgroundColor:'#d2d2d2',
    title:'Timers',
    url:'/timer/timers.js'
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Timers',
    window:win1
});

//
// create controls tab and root window
//
var win2 = Titanium.UI.createWindow({  
    title:'Settings',
    backgroundColor:'#fff'
});
var tab2 = Titanium.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Settings',
    window:win2
});

var label2 = Titanium.UI.createLabel({
    color:'#999',
    text:'I am your dead dog, haunting you for not feeding me.',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
});

win2.add(label2);

//
//  add tabs
//
tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);  


// open tab group
tabGroup.open();

    Then in your timer folder, add:

    /timer/timers.js :

    
var curWin = Titanium.UI.currentWindow;

var b1 = Titanium.UI.createButton({
   title: 'Edit'
});

var b2 = Titanium.UI.createButton({
    systemButton:Titanium.UI.iPhone.SystemButton.ADD
});

curWin.leftNavButton = b1;
curWin.rightNavButton = b2;

var data = [{title:"Row 1"},{title:"Row 2"}];
var table = Titanium.UI.createTableView({data:data});
curWin.add(table);


b2.addEventListener('click', function()
{
    var w = Titanium.UI.createWindow({
        backgroundColor:'#336699',
        title:'Modal Window',
        barColor:'black',
        tabBarHidden:true,
        url:'newtimer.js'
    });
    var saveBtn = Titanium.UI.createButton({
        title:'Save',
        style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN
    });
    w.setLeftNavButton(saveBtn);
    saveBtn.addEventListener('click',function()
    {
        w.close();
    });
    Titanium.UI.currentTab.open(w,{animated:true});
});

    and then,

    /timer/newtimer.js

    
var curWin = Titanium.UI.currentWindow;

var data = [
    {title:"Label", hasChild:true, test:"label.js"},
    {title:"Sound"},

];

        });
        Titanium.UI.currentTab.open(win,{animated:true});
    }
});

curWin.add(table2);

    And also,

    /timer/label.js :

var a_label = Ti.UI.createLabel({

    text:'I am a label',
    color:'#fff',
    top:10,
    left:10,
    right:10,
    height:20

curWin.add(a_label);

    etc…

    hope this helps!
