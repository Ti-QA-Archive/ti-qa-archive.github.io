    if you create a plain window and get the height, this should be height you are looking for. 

    
var win = Ti.UI.createWindow();
win.open();
Ti.API.info(win.height);

  • it's not simple at all… here's what i have done for android:

    var densityBar = 38; //default top , doesn't use tabs in this example
switch(Titanium.Platform.displayCaps.density){
case "low":
densityBar = 19;
case "medium":
densityBar = 25;
case "high":
densityBar = 38;
}
var Androidheight = Ti.Platform.displayCaps.height-densityBar;
var verticalAlign = Ti.UI.createView({
height:100,
top:(Androidheight-100)/2,
background:'#000',
width:100000,
left:0
});
win1.add(verticalAlign);
