Put a new line inside a label

Hi all ,

var label3= Titanium.UI.createLabel({
    text:'speaks and \nsolution world oneself has ',
    color:'black',
    height:'auto',
    width:'auto',
    left:232,
    font:{fontSize:14},
    //textAlign:'center',
    top:480
});
view.add(label3);
