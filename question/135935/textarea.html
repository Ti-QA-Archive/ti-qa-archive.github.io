Textarea

Hi all, I want to increase and decrease the font size of a textArea using two buttons (+ and -). So the font size is a variable that I increase and decrease on click on the buttons, here is my code

var sizeFont = 18;

var enr = Ti.UI.createTextArea({
    left:'68%',
    top:'15%',
    editable:false,
    enabled:false,
    backgroundColor:'transparent',
    borderRadius:'0',
    value:'hello',
    font:{fontSize:sizeFont}
});

plusBtn.addEventListener('click',function(e) {
 sizeFont++;    
 });

minusBtn.addEventListener('click',function(e) {
 sizeFont--;
 });

But how to let the textArea reload to take the new value?

    Hi Maroun,

    Create one funciton and on button click event set your font size.

    plusBtn.addEventListener('click',function(e) {
 sizeFont++;    
GrowSize(sizeFont);
 });

minusBtn.addEventListener('click',function(e) {
 sizeFont--;
GrowSize(sizeFont);
 });
function GrowSize(newSize)
{
enr.font = {
    fontSize:newSize
};
}
