I've got an scrollableView with some scrollViews. Each scrollView contains 1 picker.
When I try to select the picker, nothing pops up. But when I use the back button, I can see that the pickeRows are created behind the scrollableView.

I tried already with zIndex but didn't work.

Maybe it's not possible, but if I read this it looks like at least someone did it.

  • there is no issue in creating pickers in scrollableviews 

    
Titanium.UI.setBackgroundColor('#000');

// create tab group
var tabGroup = Titanium.UI.createTabGroup();


//
// create base UI tab and root window
//
var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'a
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});



var picker= Ti.UI.createPicker();

var data = [];
data[0]=Ti.UI.createPickerRow({title:'Bananas',custom_item:'b'});
data[1]=Ti.UI.createPickerRow({title:'Strawberries',custom_item:'s'});
data[2]=Ti.UI.createPickerRow({title:'Mangos',custom_item:'m'});
data[3]=Ti.UI.createPickerRow({title:'Grapes',custom_item:'g'});

// turn on the selection indicator (off by default)
picker.selectionIndicator = true;

picker.add(data);

var scrview= Ti.UI.createScrollableView({
    height:1000,
    width:600,
    backgroundColor:'silver'
})

scrview.add( picker);

win1.add(scrview);

//
// create controls tab and root window
//
var win2 = Titanium.UI.createWindow({  
    title:'Tab 2',
    backgroundColor:'#fff'
});
var tab2 = Titanium.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Tab 2',
    window:win2
});



//
//  add tabs
//
tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);  


// open tab group
tabGroup.open();
