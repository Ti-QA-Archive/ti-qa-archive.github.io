    if (categoryArray.length == 0) {
    var db = Titanium.Database.open('bus');

    var dbrows = db.execute('select "' + searchDate + '" as selectedDate,line_no FROM all1 WHERE line_no="' + win.line_no +'" ' );
    while (dbrows.isValidRow()) {
        categoryArray.push({
            datelist:dbrows.fieldByName('selectedDate'),
            line_no: dbrows.field(1), 
        }); 

        dbrows.next();
    }
    dbrows.close();
    db.close();

}

    • i have not yet tried using variables in SELECT statements in Titanium, but you normally don't want to explicitly build query strings like that to prevent SQL Injection. You would do something like:

      var dbrows = db.execute('select ? as selectedDate, line_no FROM all1 WHERE line_no = ?', searchDate, win.line_no);
