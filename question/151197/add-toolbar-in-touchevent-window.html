Add toolbar in touchevent window

I currently have a window that has opened from a touch event in the app.js which is a scrollable gallery but for some reason the toolbar that I want is not showing up on the bottom of the windows. I am coding for ios and using the latest sdks and updates.

box4.addEventListener('click', function(e)
{
    var pictureGallery = Titanium.UI.createWindow({
});
    var portview1 = Ti.UI.createView({
    backgroundColor:'#FFF'
});

var intropic = Ti.UI.createImageView({
    image:'../images/tour1.jpg',
    left:0,
    height:250,
    width:320,
    top:0,
})
portview1.add(intropic);

var intro1 = Ti.UI.createLabel({
    color:'#000',
    font:{fontSize:24, fontWeight:'bold'},
    top:265,
//    textAlign:'center',
    text:'VIEW 1',
    height:'auto',
    width:'auto'
});
portview1.add(intro1);

var portview2 = Ti.UI.createView({
    backgroundColor:'#FFF'
});

var view2pic = Ti.UI.createImageView({
    image:'../images/tour2.jpg',
    left:0,
    height:260,
    width:320,
    top:0,
})
portview2.add(view2pic);

var view21 = Ti.UI.createLabel({
    color:'#000',
    font:{fontSize:24, fontWeight:'bold'},
    top:265,
    text:'VIEW 2',
    height:'auto',
    width:'auto'
});
portview2.add(view21);

var views = [portview1, portview2];

var scrollView = Titanium.UI.createScrollableView({
    views:views,
    showPagingControl:false,
//    pagingControlHeight:25,
    maxZoomScale:2.0,
    currentPage:0
});

pictureGallery.add(scrollView);

var i=1;
var activeView = portview1;

scrollView.addEventListener('scroll', function(e)
{
    activeView = e.view;  // the object handle to the view that is about to become visible
    i = e.currentPage;
    Titanium.API.info("scroll called - current index " + i + ' active view ' + activeView);
});
scrollView.addEventListener('click', function(e)
{
    Ti.API.info('ScrollView received click event, source = ' + e.source);
});
scrollView.addEventListener('touchend', function(e)
{
    Ti.API.info('ScrollView received touchend event, source = ' + e.source);
});

var galleft = Titanium.UI.createButton({
    image:'../images/icon_arrow_left.png'
});
galleft.addEventListener('click', function (e) {
    var intPage = scrollView.getCurrentPage();
    if (intPage > 0) {
        scrollView.scrollToView(intPage - 1);
    }
});

var galright = Titanium.UI.createButton({
    image:'../images/icon_arrow_right.png'
});
galright.addEventListener('click', function (e) {
    var intPage = scrollView.getCurrentPage();
    if (intPage < (views.length - 1)) {
        scrollView.scrollToView(intPage + 1);
    }
});

var galflexSpace = Titanium.UI.createButton({
    systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

    var piclabel = Titanium.UI.createButton({
        title:'Gallery',
        color:'#fff',
        style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN
    });

    var picflexSpace = Titanium.UI.createButton({
        systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
    });
    var picclose = Titanium.UI.createButton({
        title:'Close',
        style:Titanium.UI.iPhone.SystemButtonStyle.DONE
    });

    picclose.addEventListener('click', function()
    {
        pictureGallery.close();
    });

    var pictoolbar = Titanium.UI.iOS.createToolbar({
        items:[picflexSpace, piclabel, picflexSpace, picclose],
        top:0,
        borderTop:false,
        borderBottom:true
    });
    pictureGallery.setToolbar([galflexSpace,galleft,galflexSpace,galright,galflexSpace]);
    pictureGallery.add(pictoolbar);
    pictureGallery.open();
});
