    I have read all of you and Sanat Mamtora conversion and I hope the below code will helpful for you.

    app.js

    var win = Titanium.UI.createWindow({
        backgroundColor:'red'
    }); 

var ok=require('cwin');

var aButton = Ti.UI.createButton({
    title : 'GO to C',
    height : 'auto',
    width : 'auto',
    top : 10,
});

aButton.addEventListener('click', function() {

    var wins=ok.getWin();
    wins.open();
});

win.add(aButton);
win.open();

    cwin.js

    exports.getWin=function()
{
    var cwin = Titanium.UI.createWindow({
        backgroundColor:'red'
    });

    title : 'Back',
    height : 'auto',
    width : 'auto',
    top : 10,
});

aButton.addEventListener('click', function() {
    //alert("ok!!!!!!!!!!!");
    cwin.close();
});
cwin.add(aButton);
return cwin;
