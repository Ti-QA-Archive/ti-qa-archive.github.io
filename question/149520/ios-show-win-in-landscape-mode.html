[ios] show win in landscape mode

Hi, my app has been made to use in portrait mode but just a win has to be shown in landscape mode, so I create that win in this way

function WebappWindow(navController) {    
    Titanium.UI.orientation = Titanium.UI.LANDSCAPE_RIGHT;
    Ti.Gesture.orientation = Titanium.UI.LANDSCAPE_RIGHT;

    var self = Ti.UI.createWindow({
        orientationModes: [Ti.UI.LANDSCAPE_RIGHT],
        orientation: Ti.UI.LANDSCAPE_RIGHT
    });

    self.orientationModes = [ Titanium.UI.LANDSCAPE_RIGHT ];

    var wv = Ti.UI.createWebView({
        url: "http://google.com"
    });

    self.add(wv);

    return self;
}

module.exports = WebappWindow;

it works fine, so the win is show in landscape_right mode but when i turn the phone in portrait mode the win is set in portrait. How can i avoid this?

In my tiapp.xml i have

<orientations device="iphone">
            <orientation>Ti.UI.PORTRAIT</orientation>
</orientations>

Thank you

  • hi,

    if you want to give support for both portrait and landscape mode in iphone as you want then you should set both orientations in tiapp.xml

    <orientations device="iphone">
    <orientation>Ti.UI.PORTRAIT</orientation>
    <orientation>Ti.UI.LANDSCAPE_RIGHT</orientation>
</orientations>

    and remember when you change any content in tiapp.xml file then you should clean your project first to see the new changes.
