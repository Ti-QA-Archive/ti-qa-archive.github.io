    Hello Sabari,

    Try this :

    var btn = Ti.UI.createButton({
    height : 100,
    width : 200,
    title : "contacts"
});

btn.addEventListener("click", function(e) {
    Ti.Contacts.showContacts({
        selectedPerson : function(e) {
            var person = e.person;
            Ti.API.info("Selected Person :=>" + JSON.stringify(person));
            Ti.API.info('Home : :  ' + JSON.stringify(e.person.phone.home));
            Ti.API.info('Work : :  ' + JSON.stringify(e.person.phone.work));
            Ti.API.info('mobile  : :  ' + JSON.stringify(e.person.phone.mobile));
            Ti.API.info('Other : :  ' + JSON.stringify(e.person.phone.other));
        }
    });
});

    i hope this helps. thanks.
