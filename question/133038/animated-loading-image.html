  • Try something like this in SDK 2.x:

    var isLoading = true;
var runAnim=function(){
  // stop rotating the image if not loading
  if (!isLoading) return;
  // start rotating the image, 'recursively' call runAnim when rotation ends
  imgView.animate({
    duration : 1000,
    anchorPoint : {x : 0.5, y : 0.5}, // NOTE: anchorPoint is buggy in SDK 1.8 !!!
    transform : Ti.UI.create2DMatrix().rotate(360)
  }, runAnim); 
};
runAnim();
xhr.onload=function(){ isLoading = false; };
