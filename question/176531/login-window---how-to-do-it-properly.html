  • You will have to do perform logic on app launch to determine if the user is logged in or not. For simplicity, I will use a App Property to store this. Place this in your app.js file.

    var loggedIn = Ti.App.Properties.getBool('isLoggedIn');

if (loggedIn){
    openApp();
} else {
    openLogin();
}

function openApp(){
    //put your window creation code here for the app
}

function openLogin(){
    //put your window creation here for the login process

    //once the user has successfully logged in fire:
    //Ti.App.fireEvent('loginSuccessful');
}

function loginSuccessful(){
    Ti.App.Properties.setBool('isLoggedIn', true);
    //code here to close your login windows

    openApp();
}

Ti.App.addEventListener('loginSuccessful', loginSuccessful);
