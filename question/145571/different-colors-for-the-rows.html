    Hi

    Guessing you have not played with custom tableViewRows.

    Try this

    function addRow(obj) {
  var row = Ti.UI.createTableViewRow({
    backgroundColor: obj.color,
    height: 40,
    width: Ti.UI.FILL
  });
  var lbl = Ti.UI.createLabel({
    height: Ti.UI.SIZE,
    left: 0,
    text: obj.title,
    textAlign: 'left',
    width: Ti.UI.FILL
  });
  row.add(lbl);
  return row;
}
var data = [
  { title: "One", color: "#ff0000" },
  { title: "Two", color: "#00ff00" },
  { title: "Three", color: "#0000ff" },
  { title: "Four", color: "#ffff00" }
];
var intRow = 0, intRows = data.length, rows = [];
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    rows.push(addRow(data[intRow]));
}
var table = Ti.UI.createTableView({
  data: rows,
  height: Ti.UI.FILL,
  width: Ti.UI.FILL
});

    By passing in your own parameters you can create any combination of table rows you can think of.

    Btw this was written on an iPad which harder than you think to write code in these comment boxes - so apologies for any typos.
