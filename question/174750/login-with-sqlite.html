    This should solve the problem - as long as there are not other issues elsewhere.

    var db = Ti.Database.open('users');
var rows = db.execute('SELECT username FROM users where username = ?', username.value);

if (rows.getRowCount() > 0) {
  alert('You logged in'); 
}
rows.close();
db.close();
