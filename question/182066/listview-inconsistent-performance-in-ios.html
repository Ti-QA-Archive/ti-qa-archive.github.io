  • Hello VARUN,

    ListView is data-oriented for ios so view does not show it's actual behavior. So you can use absolute layout for that. And I just change something in you code and it works for me android and ios as well.

    ##index

    <Alloy>
    <Window class="container" title="Jobs" navTintColor="black" barColor="#70B5DA">

        <ListView id="listView" defaultItemTemplate="userTemplate" top="5">
            <Templates>
                <ItemTemplate id="userTemplate" name="userTemplate">


                            <Label id="fontsizeBigBold" bindId="sid" top ="1"  text="S4715dafd" class="left"/>
                            <Label  id="fontsizeBig" bindId="time" top ="1" text="30ST" class="right"/>

                            <Label id="fontsizeBig" bindId="jobid" top ="25" text="JOB:S4715dafd_01" class="left" />
                            <Label id="fontsizeBig" bindId="number" top =25"  text="ABCRE$%SBA" class="right"/>


                            <Label id="fontsizeSmall" bindId="Pickup"  top = 50 text="Pick up: PORT ALBERT ENSTINE TERMINAL" class="left"/>
                            <Label id="fontsizeSmall" bindId="LiveLoad"  top = 65 text="Live Load: Brockland blue CO master bru" class="left"/>
                            <Label id="fontsizeSmall" bindId="LiveUnload" top = 80 text="Live Unload: PORT ALEZA BETH KA DHABA" class="left"/>


                </ItemTemplate>
            </Templates>

            <!-- <SearchBar id="searchBar" /> -->
            <ListSection id="listItem">
                <ListItem/>
            </ListSection>

        </ListView>
        <!-- </View> -->
    </Window>
</Alloy>
