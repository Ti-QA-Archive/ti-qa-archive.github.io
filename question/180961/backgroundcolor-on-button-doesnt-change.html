backgroundColor on button doesn't change

Hi,

I have 4 buttons, each one display a different view. When the user click on a button i want it to be unfocusable and change the backgroundColor.

    var b1 = Titanium.UI.createButton({
        title:'b1',
        color:'white',
        top:'80px',
        left:'29%',
        height:'100px',
        width:'49%',
        backgroundColor:'blue',
        borderColor:'black',
        borderWidth:'1px',
        enabled:true,
        focusable:true,
    });

The 4 buttons have the same body.

When a button is focus i want to add this properties :
b1.focusable = false;
b1.enabled = false;
b1.backgroundColor = &#39;purple&#39;;
And it's work.

But when i want to undo this (when the user click on an other button) i use the same code, the backgroundColor doesn't change but the button is focusable and enabled.
b1.focusable = true;
b1.enabled = true;
b1.backgroundColor = &#39;blue;&#39;

I saw that set the backgroundImage to 'none' correct this issue but when i do it, not only the backgroundColor stay purple but the focus and enable stay on false.

I tryed to make 2 functions :

function setTrue(){
    this.enabled = true;
    this.focusable = true;
    this.setBackgroudColor = 'blue';
    };

function setFalse(){
    this.enabled = false;
    this.focusable = false;
    this.setBackgroudColor = 'purple';
    };

But when i call them with into an event listener : b1.setTrue();
An error occur.

So i made 2 functions for each button : 

    function b1setTrue(){
        b1.enabled = true;
        b1.focusable = true;
        b1.backgroudColor = 'blue';
    };

    function b1setFalse(){
        b1.enabled = false;
        b1.focusable = false;
        b1.backgroundColor = 'purple';
    };

And i call them in the event listener : b1setFalse(); this work, but with b1setTrue(); the backgroundColor stay purple.

Do someone knows how to fix this ?
And how use this functions ?

    Hi Marius,

    you can do it by using storing previous button reference.. and also you can use common function for all button..

    here i share some code try it and modified if you needed…

    var win = Ti.UI.createWindow({
    layout : "vertical"
});
function setTrue(btn){
    btn.enabled = true;
    btn.focusable = true;
    btn.backgroundColor = 'blue';
}

function setFalse(btn){
    btn.enabled = false;
    btn.focusable = false;
    btn.backgroundColor = 'red';
}

var preBtn = undefined;
for (var i = 0; i < 4; i++) {
    var btn = Titanium.UI.createButton({
        title : 'b' + i,
        color : 'white',
        top : '20px',
        left : '29%',
        height : '100px',
        width : '49%',
        backgroundColor : 'blue',
        borderColor : 'black',
        borderWidth : '1px',
        enabled : true,
        focusable : true,
    });

    btn.addEventListener("click",function(e){
        if(preBtn != undefined){
            setTrue(preBtn);
        }
        preBtn = e.source;
        setFalse(e.source);
    });

    win.add(btn);

}

    Hope this is helps you..
