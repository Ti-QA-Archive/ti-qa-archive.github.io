    function giveMeDate() {
    var date = new Date();
        var formattedDate = date.getFullYear() + '-' + date.getMonth()+1) + '-' + date.getDate() + '  ' + date.getHours() + ':' + date.getMinutes() + ':' + date.getSeconds();
    return formattedDate;
}
