Solution ---- Add tabs to tabgroup dynamically

———> My app.js


var tabGroup =  Titanium.UI.createTabGroup();

var homeWin = Titanium.UI.createWindow
({
    url:'home.js',
    title:'Home',
    navBarHidden:true
});

var homeWin1 = Titanium.UI.createWindow
({
    title:'Home1',
    navBarHidden:true
});

var homeWin2 = Titanium.UI.createWindow
({
    title:'Home3',
    navBarHidden:true
});

var tab = Titanium.UI.createTab
({
    window:homeWin,
    title:'Home',
    icon:'KS_nav_ui.png'
});
tabGroup.addTab(tab);

var tab1 = Titanium.UI.createTab
({
    window:homeWin1,
    title:'Home',
    icon:'KS_nav_ui.png'
});
tabGroup.addTab(tab1);

var tab2 = Titanium.UI.createTab
({
    window:homeWin2,
    title:'Home',
    icon:'KS_nav_ui.png'
});
tabGroup.addTab(tab2);

tabGroup.open({animated:true});

———> My home.js



var win = Titanium.UI.currentWindow;

var but = Titanium.UI.createButton
({
    title:'add tab'
})
win.add(but);

but.addEventListener('click',function()
{
    var homeWin2 = Titanium.UI.createWindow
    ({
        //url:'home.js',
        title:'Home3',
        navBarHidden:true
    });

    var tab = Titanium.UI.createTab
    ({
        window:homeWin2,
        title:'Home3',
        //icon:'default_app_logo.png'
    });

    Titanium.UI.currentTabGroup.addTab(tab);

});
