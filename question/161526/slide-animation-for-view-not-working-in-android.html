var open = true;
function animateOpen(e){
    open = !open;

    slide_in = Ti.UI.createAnimation({
        curve : Ti.UI.ANIMATION_CURVE_EASE_IN_OUT,
        duration : 500,
        right : 0
    });
    slide_out  = Ti.UI.createAnimation({
        curve : Ti.UI.ANIMATION_CURVE_EASE_IN_OUT,
        duration : 500,
        right : -195
    });

    if(open){
        $.slideView.animate(slide_in);        
    }else{
        $.slideView.animate(slide_out);
    }
}
