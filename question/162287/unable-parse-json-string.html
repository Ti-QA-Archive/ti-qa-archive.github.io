Unable Parse Json String

Hi there,

i am getting "unable to parse json string" , what should i do ?

js file

var send_comment = function() {
    var createReq = Titanium.Network.createHTTPClient();
    var params = {

        username : "dataUsername",
    };

    createReq.onload = function() {
        var response = JSON.parse(this.responseText);
        alert(response.status); // This gives "Unable to parse JSON String" error
    };
    createReq.open("POST", adress);
    createReq.send(params);
}

json file

{"status":"Ok"}

    Try it this way:

    var send_comment = function() {
    var createReq = Titanium.Network.createHTTPClient({
        onload: function() {
            Ti.API.info("Raw text: " + this.responseText);
            var response = JSON.parse(this.responseText);
            alert(response.status);
        }
    });
    var params = {
        username : "dataUsername",
    };

    createReq.open("POST", adress);
    createReq.send(params);
}

  • Hi

    Test your JSON output using a web browser to call the same URL and include the parameter.

    • In your browser - select view source
    • copy the contents
    • Paste the clipboard into a web page jsonlint.com
    • Validate the JSON - any errors will show up for you to see.

    If the JSON does validate then it is a script issue, if it fails to validate it is how you are creating your JSON.
