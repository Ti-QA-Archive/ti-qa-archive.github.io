    This is the code for you.. and its working fine what you asked above.

    
var win1 = Titanium.UI.createWindow({
    title : 'Window 1',
    backgroundColor : 'red'
});

var label = Ti.UI.createLabel({
    text:'Right now window 1',
    top:50,
    left:100,
    height:60,
    width:150
});

win1.add(label);

var win2 = Titanium.UI.createWindow({
    title : 'Classic RPG Finder',
    backgroundColor : 'green'
});

var labelWin2 = Ti.UI.createLabel({
    text:'Now window 2',
    top:50,
    left:100,
    height:60,
    width:150
});

win2.add(labelWin2);

var button = Ti.UI.createButton({
    title:'open another window',
    top:150,
    left:100,
    height:60,
    width:150
});

button.addEventListener('click',function(){
    win1.close();
    win2.open();
});
win1.add(button);
win1.open();
