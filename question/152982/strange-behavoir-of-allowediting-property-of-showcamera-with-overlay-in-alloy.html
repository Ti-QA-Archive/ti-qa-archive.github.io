index.js

function openCamera() {
    var view = Ti.UI.createView({
        left : 0,
        top : 0,
        width : 320,
        height : 460
    });

    var btnClick = Ti.UI.createButton({
        title : 'Take',
        width : 70,
        height : 40,
        bottom : 10,
        left : 20
    });

    btnClick.addEventListener('click', function(e) {
        Ti.Media.takePicture();
    });

    view.add(btnClick);

    var btnCancel = Ti.UI.createButton({
        title : 'Cancel',
        width : 70,
        height : 40,
        bottom : 10,
        right : 20
    });

    btnCancel.addEventListener('click', function(e) {
        Ti.Media.hideCamera();
    });
    view.add(btnCancel);

    Ti.Media.showCamera({
        success : function(e) {

        },
        cancel : function(e) {

        },
        error : function(e) {

        },
        overlay : view,
        allowEditing : true,
        mediaTypes : [Ti.Media.MEDIA_TYPE_PHOTO],        
        showControls : false
    });
}

$.win.open();
