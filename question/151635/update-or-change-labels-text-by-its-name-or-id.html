  • Hi Kerberos,
    You can do it like this:

    var data = [
    {text: '\n Friday, May 10th \n _____________________ \n \n 9h to 17h'},
    {text: '\n Saturday, May 11th \n _____________________ \n \n 8h to 16h'}
    ];

var data1 = [];
for(var i=0; i< data.length; i++){

var row = Ti.UI.createTableViewRow({
    height: '100dp',
    width: Ti.UI.FILL    
});


var datetime =  Ti.UI.createLabel({
    top: 'auto',
    text: data[i].text,
    font:{fontSize:'14dp',fontWeight : 'normal'},
    height: 'auto', 
    width: 'auto',
    color: 'black',
    id: 'specific_label'
    textAlign: 'left',
    left: '2%'
});
row.add(datetime);

data1.push(row);
}
var table = Ti.UI.createTableView({
    top: '58%',
    data: data1
});
win.add(table);
table.addEventListener('click', function(e){
    if(e.source.id == 'specific label'){
        // do the click related stuff
    }
});

    I hope this will work.
