Cascaded images Zindex position not working

i have some buttons which are cascaded horizontally.if i click any button,that button should come on top between other buttons(zoom effect). can any body please suggest me how to get this effect..here is my code..

var win = Ti.UI.createWindow({
    //fullscreen:true,
    navBarHidden:true,
    exitOnClose:true,
    tabBarHidden:true,
    backgroundColor:"#336699",

});

var chapters=["Data Handling","Factorisation","Linear Equation in one Variable","Mensuration","Cubes and Cube Roots","Playing with Numbers","Practical Geometry","Rational Numbers","Squares and Square Roots"];

var scrollView = Titanium.UI.createScrollView({
    contentWidth:'auto',
    contentHeight:150,
    left:0,
    width:"100%",
    height:400,
    scrollType:'horizontal',
    top:20,
    //showHorizontalScrollIndicator:true,

win.add(scrollView);

for (var i=0; i < chapters.length; i++) {
    var ChpButton = Titanium.UI.createButton({
        title:chapters[i],
        backgroundImage:'backgnd_Img.png',
        left:i*150,
        top:30,
        textAlign:'center',
        height:100,
        width:200,
        //id:"button_"+i,
        font: {

        zIndex:-i,
        _index:i

    ChpButton.addEventListener('click', clickCallback );

var currentBtn = null;

{

        btn. width= width,
        btn.height= height,
        btn.top=top

}

    //var inPos = e.source.zIndex
    if (currentBtn) {
        zoom(currentBtn, 200, 100,30);
        currentBtn.zIndex = 0
    }

         currentBtn = e.source;
         currentBtn.zIndex = 1
        zoom(e.source, 250, 120,25);

win.open();

  • Try this logic this will work:
    suppose u've set default zIndex of all buttons to 5.

    button1.addEventListener('click',function(e){
    button1.zIndex = 6;
    // set all button's zIndex to 5 or less than 6.
});

    on click of other button do the same.
