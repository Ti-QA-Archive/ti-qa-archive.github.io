Managing the detail view in windows with tab groups

Hello,

I have one single tab group with two windows

function AppTabGroup() {
    //declare module dependencies
    var AppWindow = require('ui/AppWindow');
    var AppSettings = require ('ui/AppSetting');

    //create module instance
    var self = Ti.UI.createTabGroup();

    //create app tabs
    var win1 = new AppWindow(L('home')),
        win2 = new AppSettings(L('settings'));

    var tab1 = Ti.UI.createTab({
        title: L('home'),
        icon: '/images/KS_nav_ui.png',
        window: win1
    });
    win1.containingTab = tab1;

    var tab2 = Ti.UI.createTab({
        title: L('settings'),
        icon: '/images/KS_nav_views.png',
        window: win2
    });
    win2.containingTab = tab2;

    self.addTab(tab1);
    self.addTab(tab2);

    return self;
};

module.exports = AppTabGroup;

And,

function AppWindow(title) {
    var self = Ti.UI.createWindow({
