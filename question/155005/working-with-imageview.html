    Hi

    In case you meant position each one next to each other automatically then this is the simplest way.

    var view = Ti.UI.createView({
    height: Ti.UI.SIZE,
    layout: 'horizontal', // important part.
    top: 0,
    width: Ti.UI.FILL
});
for (var i=0; i < 4; i++ ) {
    var img=Ti.UI.createImageView({
        width:100,
        height:100,
        borderRadius:50,
        backgroundColor:'#bdbdbd'
    });
    view.add(img);
}

    • You can also create margins so that each child view of a parent with layout set can be positioned away form the others.

      var img = Ti.UI.createImageView({
    width: 100,
    height: 100,
    borderRadius: 50,
    backgroundColor: '#bdbdbd',
    top: 10, // now acts as margin
    bottom: 10, // now acts as margin
    left: 10, // now acts as margin
    right: 10 // now acts as margin
});

    • Here you go;

      var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
}); 
var view = Ti.UI.createView({
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
var width = 320;
var height = 480;
var size = 50;
var radius = 120;
var intCircles = 7;
var intCircle = 0;
var angle = 0;
var step = (2 * Math.PI) / intCircles;
var point = {};
for (intCircle = 0; intCircle < intCircles; intCircle = intCircle + 1 ) {
    point = {
        x: Math.round(width / 2 + radius * Math.cos(angle) - (size / 2)),
        y: Math.round(height / 2 + radius * Math.sin(angle) - (size / 2))
    }
    var img = Ti.UI.createImageView({
        backgroundColor: '#bdbdbd',
        borderRadius: parseInt(size / 2, 10),
        height: size,
        left: point.x,
        top: point.y,
        width: size
    });
    view.add(img);
    angle = angle + step;
}
win.add(view);
win.open();

      Change these values to suit your needs;

      var size = 50; // circle size
var radius = 120; // distance from centre
var intCircles = 7; // how many circles you want
