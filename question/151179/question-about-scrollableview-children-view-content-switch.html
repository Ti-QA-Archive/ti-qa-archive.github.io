  • OK thanks Mauro i used your idea
    i have an array with 200 objects… when i arrive with curIndex near the end (200) i can fetch more objects :)
    (in every subview i have 9 objects)

    here the code i use

    
var subview = Array(
        Ti.UI.createView({
            width:Ti.UI.FILL,
            height:Ti.UI.FILL,
            layout:'horizontal',
        }),
        Ti.UI.createView({
            width:Ti.UI.FILL,
            height:Ti.UI.FILL,
            layout:'horizontal',
        }),
        Ti.UI.createView({
            width:Ti.UI.FILL,
            height:Ti.UI.FILL,
            layout:'horizontal',
        })
    );

    var scrollView = Ti.UI.createScrollableView({
        top:0,
        left:0,
        width:Ti.UI.FILL,
        height:Ti.UI.FILL,
        showPagingControl:false,
        views:subview
    });

    scrollView.addEventListener('scrollEnd', function(e)
    {
            if(scrollView.getCurrentPage()==2){
                tmpview = scrollView.views[0];
                subview[0] = scrollView.views[1];
                subview[1] = scrollView.views[2];
                scrollView.removeView(tmpview);
                tmpview.removeAllChildren();
                if(parseInt(curIndex)+18 > result.length){
                    materiali.fetch(function(data){
                        result = result.concat(data);
                        offset =  result.length;
                        subview[2] = build(tmpview,parseInt(curIndex));
                    },{limit:numitems,offset:offset,random:0});                    
                }else{
                    subview[2] = build(tmpview,parseInt(curIndex));
                }
                scrollView.views = subview;
                scrollView.setCurrentPage(1);
            }    
            if(scrollView.getCurrentPage()==0){
                if(parseInt(curIndex)-36>=0){
                    tmpview = scrollView.views[2];
                    subview[1] = scrollView.views[0];
                    subview[2] = scrollView.views[1];
                    scrollView.removeView(tmpview);
                    tmpview.removeAllChildren();
                    subview[0] = build(tmpview,parseInt(curIndex)-36);
                    curIndex = curIndex + 18;
                    scrollView.views = subview;
                    scrollView.setCurrentPage(1);
                }

            }    

    });
    mainView.add(scrollView);


    function build(sub,index){
        var start = index || 0;
        if(result[start]){
            sub.add(bigImageMat(result[start++]));
        }
        if(result[start]){
            sub.add(smallImageMat(result[start++]));
        }
        if(result[start]){
            var tmp = smallImageMat(result[start++]);
            tmp.top = 256;
            tmp.left = -256;
            tmp.type ='secondo';
            tmp.zIndex= 100;
            sub.add(tmp);
            tmp = null;
        }

        if(result[start]){
            var tmp = smallImageMat(result[start++]);
            tmp.type ='terzo',
            sub.add(tmp);
            tmp = null;
        }
        if(result[start]){
            var tmp = smallImageMat(result[start++]);
            tmp.type ='quarto',
            sub.add(tmp);
            tmp = null;
        }
        if(result[start]){
            sub.add(smallImageMat(result[start++]));
        }
        if(result[start]){
            sub.add(smallImageMat(result[start++]));
        }
        if(result[start]){
            sub.add(smallImageMat(result[start++]));
        }
        if(result[start]){
            sub.add(smallImageMat(result[start++]));
        }
        curIndex = start;
        return sub;
    }
