    you can do something like this in commonJS approach

    win1.js

    function win1()
{
    //create a window
    //create text field say myText
    //create a button
    //In button event listener
    btn.addEventListener('click',function(){
        //require the win2 here
        var _win2 = require('/win2'); //this should be your win2.js file path
        var win2 = new _win2(myText.value); //passing the data in win2
        win2.open();
    });

}
module.exports = win1;

    win2.js

    function win2(data)
{
    var win = Ti.UI.createWindow({title:data}); // it will show the passed value as title from win1.
    return win;
}
module.exports = win2;

    You can also create a separate commonJS module file to hold your non persistence data, and you can also use Ti.App.Properties, but it is not best practice to always use Ti.App.properties

    Hope that helped!!
