Assigning an ID to a custom Objekt... :(

Hey guys,

i am trying the following to assign id's to my textfields…. But it doesn't work - any suggestions?

for( var i = 0; i<editFields.length; i++)
        {
            editFields[i].setID(i);
            editFields[i].addEventListener('return', onReturn);
            Ti.API.info( editFields[i].getID() );
        }

        function onReturn(e) {
            Ti.API.info( e );
            Ti.API.info( e.source );
            Ti.API.info( e.source.ID );
            Ti.API.info( e.source.getID() );
        }

That's my stuff for assigning the ID. But in the return Listener ID and "getID()" always are null :(

Tf.ui.createEditableTextField = function( startValue, customObject ) {
    var ID;
    var fieldView = Ti.UI.createView( Tf.combine({layout:'horizontal', width:300}, customObject ));    

    var field =  Ti.UI.createTextField({
        value:startValue,
        color:'#666666',
        font:
        {
            fontSize:12
        },
        returnKeyType:Ti.UI.RETURNKEY_NEXT,
        enabled:false
    });
    fieldView.add(field);



    // Set ID
    fieldView.setID = function(_id) {
        ID = _id;
        Ti.API.info( ID );
    }

    // Get ID
    fieldView.getID = function() {
        return ID;
    }



    return fieldView;
}

That's the "component" itself.

I really have no idea, why this shit doesn't work :) Help would be great.

regards,
Nico

  • Titanium is probably stomping on your functions as their names start with 'set' and 'get'. This is interfering with some of the Titanium magic that builds the app. A strange and frustrating issue, I know, but it pretty much went away since CommonJS modules was introduced (check the discussion in the post comments).

    It appears you are using the older namespace pattern, which is fine, except in situations like this. Try renaming your getID/setID functions, or better yet, look into using CommonJS modules instead.
