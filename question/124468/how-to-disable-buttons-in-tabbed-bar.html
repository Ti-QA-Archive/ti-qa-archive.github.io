  • Accepted Answer

    seems a bug in tabbedBar initialization.

    Force to initialize labels after a while, like this:

    var buttonObjects = [
            {title:'Map', enabled:true},
            {title:'Hybrid', enabled:true},
            {title:'Satellite', enabled:true},
        ];

    var bb1 = Titanium.UI.createTabbedBar({
        labels:buttonObjects,
        backgroundColor:'#336699',
        top:100,
        style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
        height:40,
        width:300
    });
    win.add(bb1);

setTimeout(function(){
    buttonObjects[1].enabled = false;
    buttonObjects[2].enabled = false;
    bb1.labels = buttonObjects;
}, 500);

    After initialization you can change labels enabled property this way

        buttonObjects[0].enabled = false;
    buttonObjects[1].enabled = true;
    buttonObjects[2].enabled = false;
    bb1.labels = buttonObjects;
