    • It's exactly as in the documented examples, you just reference your UI element like you typically do in Alloy. Using inline styles for simplicity.

      index.xml

      <Alloy>
    <Window backgroundColor="#fff">
        <View id="mover" backgroundColor="#a00" height="100" width="100" top="0" left="0" onClick="doAnimation"/>
    </Window>
</Alloy>

      index.js

      function doAnimation() {
    $.mover.animate(Ti.UI.createAnimation({
        duration: 1500,
        left: 100,
        top: 150    
    }));    
}

$.index.open();

      var t1 = Ti.UI.create3DMatrix();
t1 = t1.translate(0, 100, 200);
t1.m34 = 1.0/-90;
var a1 = Ti.UI.createAnimation();
a1.transform = t1;
a1.duration = 800;
menuView.animate(a1);
