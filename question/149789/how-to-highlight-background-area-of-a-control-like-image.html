Titanium SDK:3.0 version
Platform & version: Android
Device:Android Emulator
Host Operating System: Windows 8

    @jai pratap : Here is one simple example

    var featuredRow = [] ;

for(var i=0;i<10;i++){    // create 10 rows 

featuredRow[i] = Ti.UI.createTableViewRow({
    height : 75,
    id : i, // each rows have different id
});

featuredRow[i].addEventListener('click',function(e){      // with the use of id we check that on which row user clicked
    alert('Number of Row ID'+ i); // in alert we get that row id.
});
}
