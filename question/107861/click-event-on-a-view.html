    It should work, just tested… what does your code look like

    var view = Ti.UI.createView({backgroundColor:'#f3f'});
view.sayHello = function(){ return "from View";};
detailWindow.add(view);

view.addEventListener('click', function(e){
    alert(e.source.sayHello());
});
