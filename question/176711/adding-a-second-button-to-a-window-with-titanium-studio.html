    Hi

    You got a little confused there and tried nested one thing inside another - a good coder does not play with Russian nesting dolls :-)

    HomeWindow.js

    function create(obj) {
    var win = Ti.UI.createWindow({
        title: obj.title || 'Untitled',
        backgroundColor: '#336699',
        layout:'vertical'
    });

    //Creating the Exec Button
    var execButton = Ti.UI.createButton({
        title:'Exec',
        width:200,
        height:40,
        top:300
    }); 
    win.add(execButton);
    execButton.addEventListener('click', function(e) {
        var execWin = require('ui/common/home/exec_window').create({
            title: 'The Exec' 
        });
        execWin.open();
    });
    //Creating the About Button
    var aboutButton = Ti.UI.createButton({
        title:'About',
        width:200,
        height:40,
        top:200
    }); 
    win.add(aboutButton);
    aboutButton.addEventListener('click', function(e) { 
        var aboutWin = require('ui/common/home/about_window').create({
            title: 'About Matchday Society' 
        });      
        aboutWin.open();
    }); 
    return win;
}
exports.create = create;

    As you are adding two buttons at the same level to the HomeWindow, you add your code at the same level - as you will see above, the execButton and aboutButton are lined up at the same indentation level.

    BTW you will find it much easier if you give your variables more suitable names, you will see here that the names clear up what ever thing does. Whilst in this specific case; neither execWin or aboutWin conflict - they do shout "I am this - not that other thing".
