      index.xml

      <Alloy>
    <Window id="winHome" >
        <View id="leftView">
                <TableView id="tableView" class="tblView"/>
        </View>
            <View id="mainView">
                <View id="navView">
                    <Button id="btnList" onClick="showLeftPanel"/>
                </View>
            </View>
</Window>
</Alloy>

      index.tss

      "#winHome":{
    backgroundColor: 'red',
    zIndex: 0,
}
"#leftView":{
    height: '100%',
    width:'75%',
    top: 0,
    left: 0,
    backgroundColor: 'blue',
    zIndex: 1

    image: ' btnImage.png',  //  put your image path here

    width: Ti.UI.SIZE,
    style: 'PLAIN',
    backgroundColor: 'transparent'
},

      index.js

      var animateToLeft = Ti.UI.createAnimation({
    left : 0,
    curve : Ti.UI.ANIMATION_CURVE_EASE_OUT,
    duration : 500,
});

var animateToRight = Ti.UI.createAnimation({
    left : '75%',
    curve : Ti.UI.ANIMATION_CURVE_EASE_OUT,
    duration : 500,
});
var position = "left";
var showLeftPanel = function(e) {
    if (position == "left") {
        $.mainView.animate(animateToRight);
        position = "right";
    } else {
        $.mainView.animate(animateToLeft);
        position = "left";
    }
};

$.winHome.open();
