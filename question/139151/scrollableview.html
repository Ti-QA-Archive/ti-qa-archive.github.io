    With 2.1.0.GA this works perfectly for me:
    (with 2.0.1.GA indeed has an issue, so why don't you update? :) )

    var win = Titanium.UI.createWindow({  
    backgroundColor:'#fff'
});

var views = [];

for (var i = 0; i<5 ; i++ ) {
    var view = Ti.UI.createView();

    var web_view = Ti.UI.createWebView({
        html:'<p> webview '+i+'</p>'
    });

    view.add(web_view);

    views.push(view);

}

var sv = Ti.UI.createScrollableView({
    views:views,
      showPagingControl:true
});

win.add(sv);
win.open();
