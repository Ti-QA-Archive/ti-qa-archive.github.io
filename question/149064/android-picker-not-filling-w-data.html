Android Picker Not Filling w/ Data

I am trying to create custom datepicker for Android (I know it has a native type for doing this, but that has not been working…). The below code only prints one empty column; the rest of the columns are hidden. Does anyone have any insight on why this could be? Your help is appreciated!

    //Need 4 value lists: month, date, hour, am/pm
    var date = new Date();
    date = date.getFullYear();
    var units = { 
        year: [date, date + 1, date + 2],
        month: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'], 
        day: [], 
        hour: [], 
    };
    for (var i=1;i<=31;i++) { units['day'].push(i); };
    for (var i=1;i<=12;i++) { units['hour'].push(i) };

    //Create 4 pickers, one for each field
    var counter = 0;
    var pickerCol = [];
    var colWidth = globals['width'] * settings['addTaskNameWidth'] * .3;
    for (var f in units) {
        pickerCol[counter] = Ti.UI.createPickerColumn({
            height: '20%',
            width: colWidth,

        //Create a row for each item in this picker
        for (var r in units[f]) {
            pickerCol[counter].addRow(Ti.UI.createPickerRow({
                title: units[f][r]
            }));
        counter++;

    var picker = Ti.UI.createPicker({
        columns: pickerCol,
        useSpinner: true,
        selectionIndicator: true,
        visibleItems: 1,
    });
    taskExpire.add(picker);
