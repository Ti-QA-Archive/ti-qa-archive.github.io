    I haven't tested this, but you could try using the dataTransform function of your data binding and add a "font" property to the list of model attributes that will determine what font is used by $.newstitle.

    controler.xml

    <Alloy>
    <Collection src="message" />
    <Window id="newsmaster" title="News" onFocus="windowGotFocus"> 

        <TableView id="tableViewNews" dataCollection="message" dataTransform="doTransform" onClick="openDetail" editable="true" moveable="false">           
            <TableViewRow class="newsrow" hasDetail="true" id="{MessageId}" messageId="{MessageId}">
                <Label id="newstitle" text="{Title}" font="{Font}">
                <Label id="newstext" text="{Text}">
                <Label id="newstime" text="{CreationTime}">
            </TableViewRow>

        </TableView>
    </Window>
</Alloy>

    controller.js

    var fontSize = 16;

function doTransform(model) {
    var attrs = model.toJSON();
    if (attrs.isUnread) {
        attrs.Font = {
            fontSize: fontSize,
            fontWeight: 'bold'
        };
    } else {
        attrs.Font = {
            fontSize: fontSize,
            fontWeight: 'normal'
        };
    }
    return attrs;
}
