    Hi Sameera Mudalige,

    Apologies for the late response, please check my test sample code below.

    var url = "http://query.yahooapis.com/v1/public/yql?q=select%20*%20from%20feed%20where%20url%3D%22http%3A%2F%2Fafrifame.net%2Fmusic2%2Ftest.xml%22&format=json&diagnostics=false";
var win = Ti.UI.createWindow({
    backgroundColor: '#000'
});

var table = Ti.UI.createTableView();
var tableData = [];
var data = []; //empty data 

var xhr = Ti.Network.createHTTPClient({

    onload: function() {
   //create a json object using the JSON.PARSE function
   var jsonObject = JSON.parse(this.responseText);

   alert(jsonObject.query.count + "-- " + jsonObject.query.created + "-- " + jsonObject.query.lang);

        //create a table row
        var row = Titanium.UI.createTableViewRow({
            backgroundColor: '#fff',
            title: jsonObject.query.count + "-- " + jsonObject.query.created + "-- " + jsonObject.query.lang,
        });
        tableData.push(row);
        table.setData(tableData);
    },
        onerror: function(e) {
            alert('There was an error retrieving the remote data. Try again.');
        },
        timeout:5000
});

xhr.open("GET", url);
xhr.send();

win.add(table);
win.open();

    I'm using it for the test case, here my json data,

    {"query":{"count":0,"created":"2014-10-24T10:26:31Z","lang":"en-US","results":null}}

    And that's why I'm using jsonObject.query, but you don't need the "query" portion. You can use jsonObject.name Or whatever you need.
