function performCalendarWriteFunctions(){
    var defCalendar = Ti.Calendar.defaultCalendar;
    var date1 = new Date(2013,11,19,11,33,0),

        date2 = new Date(2013,11,19,15,33,0);
        Ti.API.info(date1);
    Ti.API.info('Date1 : '+ date1 + 'Date2 : '+ date2);
    var event1 = defCalendar.createEvent({
                        title: 'My Title',
                        notes: 'My Notes',
                        location: 'My Location',
                        begin: date1,
                        end: date2,
                        availability: Ti.Calendar.AVAILABILITY_FREE,
                        allDay: false,

                });
    var alert1 = event1.createAlert({
                        absoluteDate: new Date(2013,11,19,9,33,0)
                });

    var allAlerts = new Array(alert1);
    event1.alerts = allAlerts;
    event1.save(Ti.Calendar.SPAN_THISEVENT);
    printEventDetails(event1.id);
}
var win = Ti.UI.createWindow({
                        backgroundColor: 'white',
                        title: 'Calendar Demo'
            });

var label = Ti.UI.createLabel({
                        text: 'Check console log'+"\r\n",
                        height: Ti.UI.size,
                        width: Ti.UI.size
            });
win.add(label);

if(Ti.Calendar.eventsAuthorization == Ti.Calendar.AUTHORIZATION_AUTHORIZED) {
    performCalendarWriteFunctions();
} else {
    Ti.Calendar.requestEventsAuthorization(function(e){
            if (e.success) {
                performCalendarWriteFunctions();
            } else {
                alert('Access to calendar is not allowed');
            }
        });
}
win.open();
