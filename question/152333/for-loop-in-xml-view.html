  • Here is the simple function for retrieving the data from sqlite

    function option1fun() {
    var db = Ti.Database.open('scimed');
    var dataArray = new Array();
    var counter = 0;
    var option1 = db.execute("SELECT * FROM Answers where Answer_ID=1 limit " + Ti.App.qq + "");
    while (option1.isValidRow()) {
        dataArray[counter] = option1.fieldByName('Answer');

        option1.next();
        counter++;
    };
    db.close();
    return dataArray;
}
