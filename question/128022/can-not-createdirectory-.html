var getRemoteFile = function (url) {
    var rootPath = Titanium.Filesystem.applicationDataDirectory + 'Pictures';

    var fileName = 'p'+ url.substring(url.lastIndexOf('/') + 1, url.lastIndexOf('.') - 1);
    Ti.API.info('fileName = ' + fileName);

    var f = Titanium.Filesystem.getFile(rootPath, fileName);
    if (f.exists() == false) {
        var _ck = f.createDirectory();
        Ti.API.info('createDirectory  = ' + _ck);
