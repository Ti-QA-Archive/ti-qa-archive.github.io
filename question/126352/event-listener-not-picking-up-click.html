Event Listener not picking up click

Hello all,

I'm trying to get an tableview row to activate a telephone call, but I think the event listener is not connecting. Here is the code:

var win = Titanium.UI.currentWindow;
win.title = "Jonker Accident Support";
win.backgroundImage = 'images/roadsidebg.png';

var data = [];

for(var i = 0; i < 3; i++) {

    var row = Ti.UI.createTableViewRow({

        hasChild : false,
        color : '#000',
    });

    var lbl = Ti.UI.createLabel({

        textAlign : 'center',
        font : {fontSize : 16}
    });

    if(i == 0) {
        lbl.text = 'Monday to Friday - 7:30 am to 6:00 pm';
    } else if(i == 1) {
        lbl.text = 'Saturdays - 8:00 am to 5:00 pm';
    } else if(i == 2) {
        lbl.text = 'Press to Connect';
    }
    row.add(lbl);

    data[i] = row;
}

var tableview = Titanium.UI.createTableView({
    style : Titanium.UI.iPhone.TableViewStyle.GROUPED,
    minRowHeight : 45,
    data : data,
    rowBackgroundColor : 'white',
    backgroundColor : 'transparent'
});

win.add(tableview);

tableview.addEventListener('click', function(e) {
    var rowdata = e.target.text

    if(rowdata === "Press to Connect") {
        Ti.Platform.openURL('tel:604-539-0303');
    }
});

Does anyone know why the Event Listener isn't catching the click?

    try this approach

        if(i == 0) {
        lbl.text = 'Monday to Friday - 7:30 am to 6:00 pm';
    } else if(i == 1) {
        lbl.text = 'Saturdays - 8:00 am to 5:00 pm';
    } else if(i == 2) {
        lbl.text = 'Press to Connect';
        row.id = "PHONE_NUMBER"
    }

    and then do this

    tableview.addEventListener('click', function(e) {
    if(e.row.id === "PHONE_NUMBER") {
        Ti.Platform.openURL('tel:604-539-0303');
    }
});
