Alloy - Data Binding TableView with TableViewSection

I am familiar with the docs on how to use TableViewSection with Alloy when using static XML. I'm curious if you can use data binding with a collection and grouping in order to dynamically add TableViewSections. Right now I'm manually creating the sections and rows using Ti.UI.


function renderTable(){
    var groups = Plants.groupBy(function(Plant){ return Plant.get("season"); });
    for(season in groups){
        var models = groups[season];
        var tableSection = Ti.UI.createTableViewSection({headerTitle: season});

        for (var model_idx in models){
            var thisModel = models[model_idx];
            tableSection.add(Ti.UI.createTableViewRow({title: thisModel.get('name')}));
        }
        $.table.appendSection(tableSection);
    }
}
