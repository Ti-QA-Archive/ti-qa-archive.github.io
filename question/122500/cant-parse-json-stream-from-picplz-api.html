  • I hackishly fixed it by preparsing the json string and removing any text before '[' and after ']'.

    var tmpvar = this.responseText;
tmpvar = tmpvar.slice(tmpvar.indexOf("["), tmpvar.lastIndexOf("]")+1);    
var pictures = eval('('+tmpvar+')');

    whole code now works and goes like this:

    var data = [];
    var xhr = Ti.Network.createHTTPClient();
    xhr.timeout = 1000000;
    xhr.open("GET","http://picplz.com/api/v2/feed.json?&type=interesting");

    xhr.onload = function()
    {
        try
        {
            var tmpvar = this.responseText;
            tmpvar = tmpvar.slice(tmpvar.indexOf("["), tmpvar.lastIndexOf("]")+1);    
            var pictures = eval('('+tmpvar+')');

            for (var c=0;c<pictures.length;c++){
                var user = pictures[c].creator.username;
                var pic_url = pictures[c].pic_files["100sh"].img_url;
                var bgcolor = (c % 2) == 0 ? '#fff' : '#eee';
                var row = Ti.UI.createTableViewRow({hasChild:true,height:'auto',backgroundColor:bgcolor});

                // Create a vertical layout view to hold all the info labels and images for each tweet
                var post_view = Ti.UI.createView({
                    height:55,
                    layout:'vertical',
                    left:5,
                    top:5,
                    bottom:5,
                    right:5
                });

                var av = Ti.UI.createImageView({
                        image:pic_url,
                        left:0,

                        width:48
                    });
                // Add the avatar image to the view
                post_view.add(av);

                    bottom:2,
                    height:16,
                    textAlign:'left',
                    color:'#444444',
