Developing a comments system with tableView

Hi All

I am trying to develop a comment system where people can comment and reply i am using titaniums tableView.

people are able to add comments the problem i am having is i would like a adjustable table row height that sizes to the amount of test within the comment, problem is i can only set my row to a specific height so if someone write a huge comment it goes over the row area.

Here is part of my code

var commentBox = Ti.UI.createTextField({
        backgroundColor : '#eee',
        borderRadius: 10,
        color : '#000',
        font : {
            fontSize : 13,
            fontFamily : 'Comfortaa'
        },
        value : 'Please enter a comment',
        paddingLeft : 10,
        top : 20,
        width : '90%',
        height : 60
    });

var commentsTable = Ti.UI.createTableView({
        borderColor: 'transparent',
        backgroundColor:'transparent',
        borderColor: 'transparent',
        separatorColor: 'transparent',
        footerTitle:'',
        top : 0,
        data : []
    });

commentBox.addEventListener('focus', function(evt) {

        commentBox.value = '';

    });

    commentBox.addEventListener('return', function(evt) {

           var row = Ti.UI.createTableViewRow({
                backgroundImage:'transparent',
                backgroundColor: 'transparent',
                height:'auto',
                width: '100%'
            });

            var commentArea = Titanium.UI.createView({
                backgroundImage:'/images/pand/commentbox.png',
                backgroundColor: 'transparent',
                font: {
                    fontSize: 8
                },
                color: '#fff ',
                height:'auto',
                width: 573,
                right: 0
            });

            row.add(commentArea);

            var image = Ti.UI.createImageView({
              image:'/images/pand/avatar.png',
              width: 52,
              height: 56,
              left: 0,
              top:0
            });

            row.add(image);

            var episodeDate = Ti.UI.createLabel({
                height: 18,
                text: 'Now',
                color: '#53759C',
                width: 'auto',
                right: 10,
                bottom: 10,
                wordWrap: true,
                font: {
                    fontSize: 13
                },
            });

            var episodeUsername = Ti.UI.createLabel({
                height : 18,
                text : Ti.App.Properties.getString('username'),
                color : '#53759C',
                width : 'auto',
                left : 30,
                top : 10,
                wordWrap : true,
                font : {
                    fontSize : 13
                },
            });

            var episodeDescription = Ti.UI.createLabel({
                text : evt.value,
                height:'auto',
                wordWrap: true,
                color: '#999',
                left: 30,
                top: 25,
                width: 500,
                font: {
                    fontSize: 13
                },
            });

            commentArea.add(episodeDate);
            commentArea.add(episodeUsername);
            commentArea.add(episodeDescription);

            commentsTable.appendRow(row);

    });
