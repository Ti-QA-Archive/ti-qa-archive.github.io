  • This works for me

    index.xml

    <Alloy>
    <Window id="win">
        <ListView id="listView" defaultItemTemplate="default">
            <Templates>
                <ItemTemplate name="default">
                    <View id="advert" ns="TiAdmob" bindId='adDetail' height="50" width="320" publisherId="your ad/publisher unit id here" />
                </ItemTemplate>
            </Templates>
        </ListView>
    </Window>
</Alloy>

    index.js

    var s = Ti.UI.createListSection();

var d = [];

d.push({
    adDetail : {
        testing : true,
        dateOfBirth : new Date(1985, 10, 1, 12, 1, 1),
        gender : 'male',
        keywords : '',
        adBackgroundColor : 'black'
    }
});

s.setItems(d);

$.listView.sections = [s];

$.win.open();

    I think the key is to ensure the publisherId is set in the <itemTemplate> first, its the only way i could get it to display
