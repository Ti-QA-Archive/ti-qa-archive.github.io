    You absolutely can add two text fields to the android view and show them both simultanously. eg

    var win = Ti.UI.createWindow({
    backgroundColor : 'blue'
});

var b = Ti.UI.createButton({width:200, height:50, title:'Click'});

b.addEventListener('click', function(){

    var av = Ti.UI.createView({width:Ti.UI.FILL, height:Ti.UI.FILL, layout:'vertical'});

    var i1 = Ti.UI.createTextField({top:0, left:0, right:0, height:40});
    var i2 = Ti.UI.createTextField({top:10, left:0, right:0, height:40});

    av.add(i1);
    av.add(i2);

    var a = Ti.UI.createAlertDialog({
        androidView:av,
        message:'Fill in the Text Fields'
    });

    a.show();

});
win.add(b);

win.open();

  • Hi Jiale He,

    I think solved here your height of inputView problem . you just try this code it works perfectly.

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

// Create a Button.
var click = Ti.UI.createButton({
    title : 'click',
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,

});

// Listen for click events.
click.addEventListener('click', function() {

    inputView = Ti.UI.createView({
        width : Ti.UI.FILL,
        height : 200,
        layout : 'vertical'
    });
    var tf = Ti.UI.createTextField({
        right : 10,
        left : 10,
        top : 5,
        returnKeyType : Ti.UI.RETURNKEY_NEXT,
        height : Ti.UI.SIZE,
        hintText : 'Enter your User Name',
    });
    inputView.add(tf);
    var tf2 = Ti.UI.createTextField({
        top : 10,
        right : 10,
        left : 10,
        bottom : 5,
        height : Ti.UI.SIZE,
        passwordMask : true,
        hintText : 'Enter Password',
        returnKeyType : Ti.UI.RETURNKEY_DONE,
    });

    inputView.add(tf2);

    var dialog = Ti.UI.createAlertDialog({
        title : 'Warning',
        buttonNames : ['Confirm', "Cancel"],
        androidView : inputView
    });

    dialog.show();

});

// Add to the parent view.
win.add(click);

win.open();
