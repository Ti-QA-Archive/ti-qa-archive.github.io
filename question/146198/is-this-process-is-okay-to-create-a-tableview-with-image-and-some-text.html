    Hi

    This is the sort of thing you should do;

    function addRow(obj) {
    var row = Ti.UI.createTableViewRow({
        height: Ti.UI.SIZE,
        width: Ti.UI.FILL
    });
    var view = Ti.UI.createView({
        bottom: 5,
        height: Ti.UI.SIZE,
        left: 5,
        right: 5,
        top: 5,
        width: Ti.UI.FILL
    });
    row.add(view);
    var img = Ti.UI.createImageView({
        height: Ti.UI.SIZE,
        image: obj.image,
        width: Ti.UI.SIZE        
    });
    view.add(img);
    var lbl = Ti.UI.createLabel({
        height: Ti.UI.SIZE,
        left: 10,
        text: obj.title,
        width: Ti.UI.SIZE        
    });
    view.add(lbl);
    return row;
}
var data = [
    { title: 'One', image: 'test1.png' },
    { title: 'Two', image: 'test2.png' },
    { title: 'Three', image: 'test3.png' },
    { title: 'Four', image: 'test4.png' }
];
var intRow = 0, intRows = data.length, rows = [], row;
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    row = addRow({
        image: data[intRow].image,
        title: data[intRow].title
    });
    rows.push(row);
}
var tbl = Ti.UI.createTableView({
    data: rows,
    height: TI.UI.FILL,
    width: TI.UI.FILL
});
win.add(tbl);

    This will create a table using an array that adds custom tableViewRows so that you can lay out the interior however you want.
