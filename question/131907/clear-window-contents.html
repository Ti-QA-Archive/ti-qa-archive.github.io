  • In my project, I use one window + one main view called container. Other views will be added to container. And we can clear window content by removing that container, then re-add it. Here's sample code:

    function initContainer() {
    return Ti.UI.createView({
        width: '100%', height: '100%', top: 0, left: 0, backgroundColor: '#cfd0d2', layout: 'vertical'
    });
}

var win = Ti.UI.createWindow({
    backgroundColor: '#cfd0d2', layout: 'vertical'
});

var container = initContainer();

var label1 = Ti.UI.createLabel({
    text: 'Label 1',
    top: 0,
    width: '100%',
    height: 20,
    backgroundColor: 'green'
});

container.add(label1);
win.add(container);
win.open();

setTimeout(function() {
    var _oldContainer = win.children[0];
    var _newContainer = initContainer();

    var label2 = Ti.UI.createLabel({
        text: 'Label 2',
        top: 0,
        width: '100%',
        height: 20,
        backgroundColor: 'red'
    }); 
    _newContainer.add(label2);
    win.add(_newContainer);

    win.remove(_oldContainer);
    _oldContainer = null;

}, 1000);

    Hope this can give you some hints,

    Minh
