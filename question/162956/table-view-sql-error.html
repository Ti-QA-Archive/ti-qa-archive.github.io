  • Sorry!!! hope this is more clear to read

    index.js

    //reference collection
var myDogs = Alloy.Collections.dogs;

//create a new model object (dog) and assign it a breed name and a group
var dog = Alloy.createModel('dogs', {
    breed: 'Affenpinscher',
    group: 'Terrier AKC Toy'
});

//add the dogs to the collection and persist to the database
myDogs.add(dog);
dog.save();

//$.index.open();

    dogs.js

    exports.definition = {
    config: {
        columns: {
            "breed": "text",
            "group": "text"
        },
        adapter: {
            type: "sql",
            collection_name: "dogs"
        }
    },

    index.xml

    <Alloy>
    <Collection src="dogs"/>
    <Window class="container">
        <!--Add TableView and TableViewRow -->
        <TableView dataCollection="dogs">
            <TableViewRow breed="{breed}" group="{group}"></TableViewRow>
            </TableView>
    </Window>
</Alloy>
