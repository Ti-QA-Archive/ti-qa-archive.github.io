    You would obviously want to style this how you want, but this should get you started. You need to use "sections" with their headerView property for custom headers. There is an example in the Kitchensink.

    function create_table(response) {
    var parties = response.parties.standards;
    var data = [];
    var party=[];
    var party_info=[];
    for(var item in parties) {
        party = parties[item];

        var section = Ti.UI.createTableViewSection();

        var custom_header = Ti.UI.createView({
            height: '24dp',
            backgroundColor: 'black'
        });

        var event_date = Titanium.UI.createLabel({
            text : item,
            height : 'auto',
            width : 'auto',
            left : '10dp',
            color : 'red',
            font:{fontWeight:'bold'}
        });

        custom_header.add(event_date);

        section.headerView = custom_header;

        for(var i = 0,len=party.length;i < len; i++) {

            party_info.push(party[i]);

            var custom_row = Ti.UI.createTableViewRow({
                height : 90,
                backgroundColor:'black',
                className: 'party'
            });

            var event_name = Titanium.UI.createLabel({
                text : party[i].name,
                height : 'auto',
                width : 'auto',
                left : 100,
                top : 5,
                color : 'red',
                font:{fontWeight:'bold'}
            });

            custom_row.add(event_name);
            section.add(custom_row);
        }
        data.push(section);
    }
    tableView.setData(data);
}
