    Hi

    This code will replicate the visual you are after using a standard a tableView with standard a tableViewRow model.

    This code contains a fully working app to demonstrate the principle. Create a new basic app and swap the entire app.js with the code below. You will need to use your own row background image of course.

    The principle is simple;

    • create table to gain all the normal benefits that provides
    • stop the table scrolling
    • use a borderRadius on the table itself
    • fix the height to rowHeight * rows
    • add this table to a scrollable view

    To fake sections, simply stack additional tables into the scrollView.

    Ti.UI.setBackgroundColor('#000');

var tab1, tab2, tab3;
(function () {
    var win = Ti.UI.createWindow({  
        backgroundColor: '#000',
        title: 'Tab 1',
    });
    tab1 = Ti.UI.createTab({  
        icon: 'KS_nav_views.png',
        title: 'Tab 1',
        window: win
    });

    var data = [
        { firstname: 'Neva', lastname: 'Asmussen' },
        { firstname: 'Julianne', lastname: 'Bottorff' },
        { firstname: 'Ted', lastname: 'Michelin' },
        { firstname: 'Hugh', lastname: 'Duchene' },
        { firstname: 'Edwina', lastname: 'Englehardt' },
        { firstname: 'Dollie', lastname: 'Mccright' },
        { firstname: 'Edwina', lastname: 'Whipps' },
        { firstname: 'Margery', lastname: 'Klock' },
        { firstname: 'Eve', lastname: 'Reetz' },
        { firstname: 'Kelly', lastname: 'Gaier' }
    ];

    var intRow = 0, intRows = data.length, rows = [], row, strFullName, lblName;
    for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
        strFullName = data[intRow].firstname + ' ' + data[intRow].lastname;
        row = Ti.UI.createTableViewRow({
            backgroundImage: 'bar.white.png',
            hasDetail: true,
            height: 57,
            name: strFullName, // this points to name below
            selectionStyle: Ti.UI.iPhone.TableViewCellSelectionStyle.GRAY,
            width: Ti.UI.FILL
        });
        view = Ti.UI.createView({
            height: Ti.UI.FILL,
            width: Ti.UI.FILL
        });
        row.add(view);
        lblName = Ti.UI.createLabel({
            color: '#000',
            height: Ti.UI.SIZE,
            left: 10,
            right: 10,
            text: strFullName,
            textAlign: 'left',
            width: Ti.UI.SIZE    
        });
        view.add(lblName);
        rows.push(row);
    }
    var scrollView = Ti.UI.createScrollView({
        bottom: 5,
        contentHeight: Ti.UI.SIZE,
        contentWidth: Ti.UI.SIZE,
        height: Ti.UI.FILL,
        showHorizontalScrollIndicator: true,
        showVerticalScrollIndicator: true,
        top: 5,
        width: Ti.UI.FILL
    });
    win.add(scrollView);
    var tbl = Ti.UI.createTableView({
        backgroundColor: 'transparent',
        borderColor: '#666',
        borderRadius: 10,
        borderWidth: 1,
        bottom: 10,
        data: rows,
        height: 57 * intRows,
        left: 10,
        right: 10,
        scrollable: false,
        top: 10,
        width: Ti.UI.FILL
    });
    scrollView.add(tbl);
})();

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);
tabGroup.open();
