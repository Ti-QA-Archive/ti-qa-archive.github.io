    var win = Ti.UI.currentWindow;

    //– Create the sub windows
    var username= Ti.UI.createWindow();
    var agenda = Ti.UI.createWindow();
    var details = Ti.UI.createWindow();

    //– We set the background here since this wont change
    win.backgroundImage='/pics/blank.png';

    //– This method will close the crusts/details window and open the toppings window
    function openAgenda(e)
    {
    if (e.agenda)
    {
    details.close();
    }
    else
    {
    agenda.close();
    }

    agenda.url             = 'agenda.js';
agenda.path            = e.path;
agenda.returnToppings = e.agenda;

agenda.open();

    }

    //– This method will close the toppings window and open the details window
    function openDetails(e)
    {
    agenda.close();

    details.path        = e.path;
details.agenda     = e.agenda;
details.url         = 'details.js';

details.open();

    }

    //– This gets called after an order is submitted. It basically starts the app over.
    function resetApp()
    {
    details.close();
    openAgenda({});
    }

    //– Have our app listen for our custom events
    Ti.App.addEventListener('agenda',openAgenda);

    Ti.App.addEventListener('details',openDetails);
    Ti.App.addEventListener('cancelDetails',openAgenda);
    Ti.App.addEventListener('resetApp',resetApp);

    //– First launch, open the crust window with an empty object
    openAgenda({});

    //– This method toggles a topping item by checking the selected property
    //- It will fade a new topping in and also remove a topping when it gets unchecked
    function agendaListClick(e)
    {
    if (e.source.selected)
    {
    e.source.selected = false;
    e.source.backgroundImage = '/pics/checkbox_no.png';

        numAgenda -= 1;
    if (agenda[e.source.agendaID].container != null)
    {
        agendaHolder.remove(agenda[e.source.agendaID].container);
        agenda[e.source.agendaID].container = null;
    }
}
else
{
    //-- If numToppings is less than maxToppings, add the new topping else alert them
    if (numAgenda < maxAgenda)
    {
        e.source.selected = true;
        e.source.backgroundImage = '/pics/checkbox_yes.png';

        var aAgenda = Ti.UI.createView({
            backgroundImage:agenda[e.source.agendaID].path
        });

        if (Ti.Platform.osname == 'android')
        {
            aAgenda.image = agenda[e.source.agendaID].path;
        }
        else
        {
            aAgenda.opacity = 0;
            aAgenda.animate({
                opacity:1,
                duration:500
            });
        }
        agendaHolder.add(aAgenda);
        agenda[e.source.agendaID].container = aAgenda;
        numAgenda += 1;
    }
    else
    {
        alert("" + numAgenda + " is the max.");
    }
}

    }

    function createAgendaList()
    {
    scrollView.opacity = 0;
    scrollView.top = 155;
    scrollView.height = 120;
    scrollView.contentWidth = Ti.Platform.displayCaps.platformWidth;
    scrollView.contentHeight = 'auto';
    scrollView.showVerticalScrollIndicator = true;
    win.add(scrollView);

    for (i = 0; i < agenda.length; i++)
{
    //-- The label
    var agendaLabel = Ti.UI.createLabel({
        text:agenda[i].title,
        font:{
            fontFamily:'Verdana',
            fontWeight:'bold',
            fontSize:14
        },
        color:'#000',
        shadowColor:'#333',
        shadowOffset:{x:1,y:1},
        textAlign:'left',
        width:Ti.Platform.displayCaps.platformWidth - 10,
        left:10
    });

    //-- We add a custom property 'selected' to our checkbox view
    var checkbox = Ti.UI.createView({
        width:340,
        height:16,
        backgroundImage:'/pics/checkbox_no.png',
        selected:false,
        agendaID:i
    });


    if (win.returnAgenda)
    {
        for (j = 0; j < win.returnAgenda.length; j++)
        {
            if (win.returnAgenda[j] == agenda[i].title)
            {
                var aAgenda = Ti.UI.createView({
                    backgroundImage:agenda[i].path
                });

                if (Ti.Platform.osname == 'android')
                {
                    aAgenda.image = agenda[i].path;
                }
                else
                {
                    aAgenda.opacity = 0;
                    aAgenda.animate({
                        opacity:1,
                        duration:500
                    });
                }
                agendaHolder.add(aAgenda);
                agenda[i].container = aAgenda;
                checkbox.backgroundImage = '/pics/checkbox_yes.png';
                checkbox.selected = true;
                numAgenda += 1;
            }
        }
    }

    var toggler = Ti.UI.createView({
        width:Ti.Platform.displayCaps.platformWidth,
        height:20,
        top: i * 20
    });

    //-- We use the singletap event rather than the click since its in a scroll view
    checkbox.addEventListener('singletap',agendaListClick);
    toggler.add(agendaLabel);
    toggler.add(checkbox);

    scrollView.add(toggler);
}
scrollView.animate({
    opacity:1,
    duration:500
});

    }
    createAgendaList();
