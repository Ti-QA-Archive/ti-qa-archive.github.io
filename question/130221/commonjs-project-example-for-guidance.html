  • I'm running into this problem too since I'm coming from a Titanium Desktop environment to Titanium Mobile.
    What I did was search for the most basic example I can find and recode it the way I like.

    The example I got was that one.

    I'do changed the code to mimic a site I have that grabs a json file of products and display then. The site is this one.

    The Ti code is here (app.js):

    App = {};

App.Main = function(){
    // this sets the background color of the master UIView (when there are no windows/tab groups on it)
    Ti.UI.setBackgroundColor('#000');

    this.init.call(this);
}

App.Main.prototype.init = function(){
    // create tab group
    var tabGroup = Ti.UI.createTabGroup();

    //
    // create base UI tab and root window
    //
    var win1 = Titanium.UI.createWindow({  
        title:'Tab 1',
        backgroundColor:'#fff',
        tabBarHidden: true
    });
    var tab1 = Titanium.UI.createTab({  
        icon:'KS_nav_views.png',
        title:'Tab 1',
        window:win1
    });

    var label1 = Titanium.UI.createLabel({
        color:'#999',
        text:'I am Window 1',
        font:{fontSize:20,fontFamily:'Helvetica Neue'},
        textAlign:'center',
        width:'auto'
    });

    win1.add(label1);

    //
    // create controls tab and root window
    //
    var win2 = Titanium.UI.createWindow({  
        title:'Tab 2',
        backgroundColor:'#fff',
        tabBarHidden: true
    });
    var tab2 = Titanium.UI.createTab({  
        icon:'KS_nav_ui.png',
        title:'Tab 2',
        window:win2
    });

    var label2 = Titanium.UI.createLabel({
        color:'#999',
        text:'I am Window 2',
        font:{fontSize:20,fontFamily:'Helvetica Neue'},
        textAlign:'center',
        width:'auto'
    });

    win2.add(label2);



    //
    //  add tabs
    //
    tabGroup.addTab(tab1);  
    tabGroup.addTab(tab2);  


    // open tab group
    tabGroup.open();

    return tabGroup;
}

var A = new App.Main();

    Note that I put everithing inside the init function and the optimal here would create a function for each step of the proccess and call of than in init.
    If I change the code to reflect it I'll come back to update it.

    Anyways, I hope it helps and give a good start for other beginners too. ;)

    • Ok I pasted the wrong code up there. The code above is the very first code I wrote.

      The real one I discribed is here (app.js):

      Ghs = {};

Ghs.App = function(){
    // this sets the background color of the master UIView (when there are no windows/tab groups on it)
    Ti.UI.setBackgroundColor('#000');

    this.init.call(this);
}

Ghs.App.prototype.init = function(){
    this.layout = this.createLayout({});
    this.getData();
}

Ghs.App.prototype.toast = function(sMsg){
    var toastNotification = Ti.UI.createNotification({
        duration: Ti.UI.NOTIFICATION_DURATION_SHORT,
        message: sMsg
    });

    toastNotification.show();
}

Ghs.App.prototype.getData = function(){
    var scope = this,
        xhr = Ti.Network.createHTTPClient(),
        res = '';

    xhr.onload = function(){
        try {
            if (this.responseText && (res = JSON.parse(this.responseText)) && res.items) {
                for (var c = 0; c < res.items.length; ++c) {
                    if (res.items[c].title && (res.items[c].url != '#')) {
                        scope.layout.tableview.appendRow(Ti.UI.createTableViewRow({
                            title: res.items[c].title,
                            uri: res.items[c].url,
                            leftImage: res.items[c].media.m,
                            backgroundColor:'#000',
                            color: '#FF0'
                        }));
                    }
                }
            }
        } catch(e) {
            alert(e);
        }
    }

    xhr.open('GET', 'http://www.ghsix.com.br/portal/cfg/site.json');
    xhr.send();
}

Ghs.App.prototype.createLayout = function(cfg){
    var scope = this,
        layout = {
            data: cfg.data||[]
        };

    layout.win = Ti.UI.createWindow(cfg.window||{
        backgroundColor:'#fff'
    });

    layout.tableview = Ti.UI.createTableView(cfg.tableview||{
        headerTitle: 'Portal de Serviços',
        backgroundColor: '#000',
        data: layout.data
    });

    layout.tableview.addEventListener('click', function (e){
        if (e.rowData && e.rowData.uri && (e.rowData.uri != '#')) {
            scope.toast(e.rowData.title);

            var intent = Ti.Android.createIntent({
                action: Ti.Android.ACTION_VIEW,
                data: e.rowData.uri,
            });

            intent.addCategory(Ti.Android.CATEGORY_BROWSABLE);
            Ti.Android.currentActivity.startActivity(intent);
        }
    });

    layout.win.add(layout.tableview);

    layout.win.open();

    return layout;
}

var app = new Ghs.App();
