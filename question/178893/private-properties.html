I have this controller, let say called: mainmap.js that among other things, shows a map and some annotations loaded from a REST webservice. Between callbacks and eventlisteners, I have this properties that area defined inside the controller file, and used in some methods.

var _curCoord = {
    lat : null,
    lon : null
};

var _curProp = {

Called inside an eventlistener to store data

if(evt.clicksource == 'pin'){

        // carga la informacion de la propiedad
        _curProp.idprop = evt.annotation.titleid;

        loadPropImages(evt.annotation.titleid);

if(_curProp.idprop == 'undefined')

Ti.API.info('toggle fav ' + _curProp.idprop);
if(!favico.favorite)
