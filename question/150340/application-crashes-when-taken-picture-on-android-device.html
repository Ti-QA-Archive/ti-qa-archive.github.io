Application crashes when taken picture on android device

I am trying to take a picture with the camera and display it in an imageView but the app crashes just after I take the picture. The code works fine on iPhone. I'm using a Sony Experia with SDK 4.0.3

This is my code:

takePhotoBtn.addEventListener('click',function(e) {
    Titanium.Media.showCamera({
    success:function(event) {
        // called when media returned from the camera
        Ti.API.debug('Our type was: '+event.mediaType);
        if(event.mediaType == Ti.Media.MEDIA_TYPE_PHOTO) {

            imageView.image = event.media.imageAsThumbnail(450);

        } else {
            alert("got the wrong type back ="+event.mediaType);
        }
    },
    cancel:function() {
        // called when user cancels taking a picture
    },
    error:function(error) {
        // called when there's an error
        var a = Titanium.UI.createAlertDialog({title:'Camera'});
        if (error.code == Titanium.Media.NO_CAMERA) {
            a.setMessage('Please run this test on device');
        } else {
            a.setMessage('Unexpected error: ' + error.code);
        }
        a.show();
    },
    //saveToPhotoGallery:true,
    //allowEditing:true,
    mediaTypes:[Ti.Media.MEDIA_TYPE_PHOTO]
});
});
