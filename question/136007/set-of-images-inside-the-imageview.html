    hi

    try this

    var win = Ti.UI.createWindow({
});

var scroll = Ti.UI.createScrollView({
    layout : 'horizontal',
    contentHeight : 'auto',
    contentWidth : 'auto',
    top : 0,
    height : 400,
    width : 320
});

var images = ['image1.png', 'image2.png', 'image3.png', 'image4.png'];

for(var i = 0, i < images.length;i++) {
    var imgView = Ti.UI.createImageView({
        left : 5,
        image : images[i],
        width : 'auto',
        height : 'auto'
    });
    scroll.add(imgView);
}
win.add(scroll);
win.open({
    modal : true
});

  • var container = Ti.UI.createWindow({layout:'horizontal'});

var images = ['image1.png','image2.png','image3.png','image4.png'];


for (var x = 0, y = images.length;x<y;x++){
    var imgView = Ti.UI.createImageView({left:5, image:images[x], width:'auto', height:'auto');
    container.add(imgView);
}

container.open();
