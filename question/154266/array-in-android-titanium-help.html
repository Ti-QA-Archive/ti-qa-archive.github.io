    Hi

    You need to do something like this;

    data array

    var questions = [
    {
        question: 'First Question',
        answers: [
            'Mulitple choice a1',
            'Mulitple choice a2',
            'Mulitple choice a3'
        ]
    },
    {
        question: 'Second Question',
        answers: [
            'Mulitple choice b1',
            'Mulitple choice b2',
            'Mulitple choice b3'
        ]
    }
];

    method to handle the specific question

    var asLayout = function (obj) {
    var asAnswer = function (txt) {
        var view = Ti.UI.createView({
            backgroundColor: '#fff',
            height: Ti.UI.SIZE,
            layout: 'horizontal',
            top: 10,
            width: Ti.UI.FILL
        });
        var radio = Ti.UI.createSwitch({
            backgroundImage: 'Images/rdBtn_Not_Selected.png'
        });
        view.add(radio);
        var lbl = Ti.UI.createLabel({
            left: 10,
            text: txt
        });
        view.add(lbl);
        return view;
    };
    var view = Ti.UI.createView({
        backgroundColor: '#fff',
        height: Ti.UI.FILL,
        layout: 'vertical',
        width: Ti.UI.FILL
    });
    var lblQuestion = Ti.UI.createLabel({
        text: obj.question
    });
    view.add(lblQuestion);
    var intAnswer = 0, intAnswers = obj.answers, lblAnswer;
    for (intAnswer = 0; intQuestion < intAnswers; intAnswer = intAnswer + 1) {
        view.add(asAnswer(obj.answer[intAnswer]));
    }
    return view;
};

    Then add as required

    win.add(questions[1]);

    Extend this as you need. The key here is that the questions and answers are in a self contained array that you can access as you need - combined with this you have a method that takes the data one node at a time and returns the containing view.
