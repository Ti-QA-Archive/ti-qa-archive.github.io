    • Hello Kususma,

      Check below sample.

      var win = Ti.UI.createWindow({
    backgroundColor : "pink"
});

var view = Ti.UI.createView({
    top : "50dp",
    height : "50dp",
    backgroundColor : "#a8a8a8",
    width : "90%"
});
win.add(view);

var tf = Ti.UI.createTextField({
    borderColor : "transparent",
    borderWidth : "0",
    left : "5dp",
    right : "70dp",
    height : Ti.UI.FILL
});
view.add(tf);

var icon = Ti.UI.createView({
    backgroundColor : "blue",
    right : "10dp",
    width : "40dp",
    height : "40dp",
    borderRadius : "7"
});
view.add(icon);

win.open();
