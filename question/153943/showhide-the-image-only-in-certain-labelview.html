  • apply property zIndex into your view and imageview both:

    imageview.zIndex = 5 // or whatever you want;
view.zIndex = 6 // must be greater than your imageview's zindex

    • Here, the image should be visible only in testView2. Whenever a part of the image is dragged into testView2; that part of the image should become visible.

      In testView1, the image should not be visible.

      var win = Titanium.UI.createWindow({backgroundColor:'white'});

      var testView = Ti.UI.createView({
backgroundColor : 'white',
borderColor:'black',
width:250,
height:250
});

var imageView = Ti.UI.createImageView({
image:'images.jpeg',
width:100,
height:100,
left:100
});

var testView2 = Ti.UI.createView({
backgroundColor:'none',
borderColor:'red',
width:150,
height:150
});

testView.setZIndex(13); 
imageView.setZIndex(12);
testView2.setZIndex(15);

win.add(testView);
testView.add(imageView);
testView.add(testView2);

imageView.addEventListener('touchmove',function(e){


var t = Ti.UI.create2DMatrix().translate((e.x+0.5),(e.y+0.5));
       imageView.transform = t;
})
