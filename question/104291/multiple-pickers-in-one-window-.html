  • The following code (a slight variation on yours) works as expected - drop into an app.js to see it working:

    var project=Ti.UI.createView({});

project.add(Ti.UI.createLabel({ text:'Project', left:20, top:130, color:'#FFF',font:{fontSize:20}}));

var project_picker=Ti.UI.createPicker({ top:126, left:150, selectionIndicator:true });

var data=[];
data[0]=Ti.UI.createPickerRow({title:'Project 1',id:'p1'});
data[1]=Ti.UI.createPickerRow({title:'Project 2',id:'p2'});
data[2]=Ti.UI.createPickerRow({title:'Project 3',id:'p3'});
data[3]=Ti.UI.createPickerRow({title:'Project 4',id:'p4'});

project_picker.add(data);
project_picker.setSelectedRow(0,1,false);
project.add(project_picker);

var repeat=Ti.UI.createView({top:55});

repeat.add(Ti.UI.createLabel({ text:'Repeat', left:20, top:200, color:'#FFF',font:{fontSize:20}}));

var repeat_picker=Ti.UI.createPicker({ top:196, left:150, selectionIndicator:true });

var data=[];
data[0]=Ti.UI.createPickerRow({title:'Every day',id:'p1'});
data[1]=Ti.UI.createPickerRow({title:'Every week',id:'p2'});
data[2]=Ti.UI.createPickerRow({title:'Every month',id:'p3'});
data[3]=Ti.UI.createPickerRow({title:'Every second month',id:'p4'});

repeat_picker.add(data);
repeat_picker.setSelectedRow(0,1,false);
repeat.add(repeat_picker);

var win = Ti.UI.createWindow();
win.add(project);
win.add(repeat);
win.open();
