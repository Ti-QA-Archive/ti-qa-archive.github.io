  • hi,
    you can create a label on button's click event .

    var c = 0;

button.addEventListener('click',function(e){
    var lbl = Ti.UI.createButton({
        width : 50,
        height : 20,
        text : 'No >>' + c
    });

    win.add(lbl);
    c++;
});
