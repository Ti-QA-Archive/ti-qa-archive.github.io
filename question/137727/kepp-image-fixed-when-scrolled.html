    OK. Here's how you need to change the cover-art image

    See this simple test app.js 

    


Titanium.UI.setBackgroundColor('#000');

var win = Ti.UI.createWindow({
    width:320,
    top:0,
    bottom:50
});


var spotify_url = "http://open.spotify.com/user/kallewangstedt/playlist/22BTIdKyI2gN5xu5GaiIg9";

var web = Ti.UI.createWebView({
    url:spotify_url
});

// Add the webview
win.add(web);
// Open the window.
win.open();


var htmlHack = '';
// Find the img with id = 'cover-art'
htmlHack += 'var cover_art_img = document.getElementById("cover-art");';

// Set a new source image - needs to be size 128px x 157px which is the same size as the musical note.
htmlHack += 'cover_art_img.src = "http://placekitten.com/128/157";';

web.addEventListener('load',function(){

    // If you want to see the contents of the remote html, use this. 
    // This is how I could discover the CSS id of the cover-art image.

    // Ti.API.info('loaded html content is : ');
    // Ti.API.info(web.html);

    // Swap the img#cover-art src reference
    web.evalJS(htmlHack);

});

  • If you mean you want to do this WITHIN the webView, then you need to do it with CSS within your HTML.

    Create a div element with the content you want inside it. Then use the 'position:fixed' CSS style on it along with a high 'z-index' (to place it above the other content) and also a 'top' value in pixels to fix it where you need to it to stay from the top.

    Here are some examples : http://limpid.nl/lab/css/fixed/

    If, however you want to do this in the app UI, you can simply place a view/image/label over the webview with a higher zIndex property.
