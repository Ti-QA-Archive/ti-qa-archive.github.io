  • It is not possible to have a tabGroup inside a navGroup because the tabGroup needs to be the root view of the app.

    Try something like this:

    
var win1 = Ti.UI.createwindow({
    title:'Accounts',

    tabBarHidden:true //this is the trick

});

tab1 = Ti.UI.createTab({
    title:'Tweets',
    window:win1
});


var wiredbob_win = Ti.UI.createwindow({
    title:'wiredbob',

    tabBarHidden:false //this is the trick

});


tab1.open(wiredbob_win);


var michael_win = Ti.UI.createwindow({
    title:'',

    tabBarHidden:true //this is the trick

});


tab1.open(michael_win);


tabGroup.add(tab1);
tabGroup.open();

    The code is far from being complete but u get the idea :)
