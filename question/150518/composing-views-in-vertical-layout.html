  • Try this : 

    var win = Ti.UI.createWindow({
 height : "100%",
 width : "100%"
});

var mainView = Ti.UI.createView({
 height:"80%",
 width:"80%",
 backgroundColor : "white"
});
win.add(mainView);

var view = Ti.UI.createView({
 top:"0dp",
 bottom:"50dp",
 left:"0dp",
 right:"0dp",
 layout:"vertical",
 backgroundColor : "white"
});
mainView.add(view);
var view1 = Ti.UI.createView({
 backgroundColor : "blue",
 left:"5dp",
 right:"5dp",
 height:"45dp"
});
view.add(view1);
var view2 = Ti.UI.createView({
 backgroundColor : "yellow",
 left:"5dp",
 right:"5dp",
 height:"45dp"
});
view.add(view2);
var view3 = Ti.UI.createView({
 backgroundColor : "yellow",
 bottom:"0dp",
 left:"5dp",
 right:"5dp",
 height:"45dp"
});
mainView.add(view3);
win.open();
