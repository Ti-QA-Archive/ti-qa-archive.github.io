delete a row on tableviewrow

Hello everyone.
I create a FireEvent to create TableViewRow as you can see bellow.
However, it has a problem. After I delete a row by "tableView.deleteRow(tableView.getIndexByName("rowname"));" and create a row again, a previous row pop up again.

Can someone tell me why it happens?
Thanks.

Titanium.App.addEventListener('test_123', function(){
    var row2 = Ti.UI.createTableViewRow({
        name: 'rowname',
        height: '70dp'
    });
    tableView.setData(rowData);
    rowData.push(row2);    

    var dialog = Titanium.UI.createOptionDialog();
    dialog.setOptions(["Cancel", "Delete"]);
    dialog.setDestructive(1);
    dialog.setCancel(2);

    dialog.addEventListener('click', function(event){
        if(event.index == 0){
            event.cancel == true:
        }else if(event.index == 1){
            tableView.deleteRow(tableView.getIndexByName("rowname"));
    });

    row2.addEventListener('click', function(){
        dialog.show();
    });
});

btn.addEventListener('click', function(){
    Titanium.App.fireEvent('test_123',{});
});
