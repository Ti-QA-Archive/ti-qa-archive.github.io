  • firstly create any javascript file
    indicator.js

    function indicator_(){
    // For Weight Measurement Day
    var win     = Ti.UI.createWindow({
        height : (Ti.Platform.osname != "android") ? 80 : 80,
        width    : 130,
        //bottom:(Ti.Platform.osname != "android") ? -256 : -125,
        backgroundColor : "#E1E1E1",
        opacity    : 1,
        borderRadius: "9",

    });
    var indView_background = Titanium.UI.createView({
            height:"100%",
            width:"100%",
            backgroundColor:'#000',
            borderRadius:10,
            opacity:0.8,
            touchEnabled:false
        });
    var actInd = Titanium.UI.createActivityIndicator({
        style:Titanium.UI.iPhone.ActivityIndicatorStyle.BIG,
        top    : 5,
    });
    var message = Titanium.UI.createLabel({
            text:"Loading....",
            color:'#ffffff',
            textAlign:'center',
            font:{fontSize:18,fontWeight:'bold'},
            height:'auto',
            width:'auto',
            bottom:5,
    });
    indView_background.add(actInd);
    actInd.show();
    indView_background.add(message);
    win.add(indView_background);

    return win;
}
module.exports =indicator_;

    var obj =  require("indicator");

  • The native iOS picker has given me trouble in the past:

    • it is challenging to build a cross-platform app if your UI is based around that picker; android's picker is very different visually
    • the iOS picker doesn't fit on screen in landscape mode

    You might want to consider my SimplePicker class, part of the TitanUp library.
