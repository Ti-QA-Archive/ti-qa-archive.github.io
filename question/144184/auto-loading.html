auto loading

Hi,

I have a view in which i display more than 100 photos.
what i want to do is to load only 10 photos and when i scroll at the end to the view its scrolls the next 10 photos..the same for the others…

here is my code :

function loadMedias(){
        indicateur.show();
        require('BusinessLogic/getMedias').getMedias(
            idAlbum,
            function update(medias){

                var content = Ti.UI.createView({
                    layout: 'horizontal',
                    height: Ti.UI.SIZE,
                    top: 0
                });

                var leftContent = Ti.UI.createView({
                    top: 0,
                    layout: 'vertical',
                    width: 160,
                    height: Ti.UI.SIZE
                });

                var rightContent = Ti.UI.createView({
                    top: 0,
                    layout: 'vertical',
                    width: 160,
                    height: Ti.UI.SIZE
                });

                for(var i=0; i < medias.length; i++){

                    var mediaView = Ti.UI.createImageView({
                        image : medias[i].thumb,
                        width: 152,
                        top: 8,
                        id: medias[i].id
                    });

                    mediaView.addEventListener('click', function(e){
                        var winMedia = require('ui/winMedia').createWindow(
                            e.source.id,
                            medias, 
                            offline, 
                            function ifMediaDeleted(){
                                loadMedias();
                            }
                        );
                        winMedias.tabGroup.activeTab.open(winMedia);
                    });

                    if(i % 2 == 0)
                        leftContent.add(mediaView);
                    else
                        rightContent.add(mediaView);

                }


                content.add(leftContent);
                content.add(rightContent);

                mediasScroll.add(content);    
                indicateur.hide();
            },

            function error(msg){
                alert(msg);
                indicateur.hide();
            }
        );
    }

  • I achieve this effect like facebook twitter using a tableview because tableviews have the event "scroll" and you can detect when the user scrolls to the end… but the View in the other hand has no event listener like scroll… my sugestion would be to change your layout to tableview OR use the "swipe" event listener maybe to load up more photos.
