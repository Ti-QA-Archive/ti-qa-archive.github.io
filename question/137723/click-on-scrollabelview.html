function loadHeadline() {
    var rows = Ti.UI.createTableViewRow({
    });
    var loader = Titanium.Network.createHTTPClient();
    var views = []
    // Sets the HTTP request method, and the URL to get data from
    loader.open("GET", "http://ayaraglobal.com/dbs/api/highlight/");
    loader.onload = function() {

        var highlights = eval('(' + this.responseText + ')');
        for (var i = 0; i < highlights.count; i++) {
            var imagesV = Ti.UI.createImageView({
                image : highlights.posts[i].filePath,
                left : 0,
                top : 0,
                width : device_width,
                height : scrollable_height
            })
            views.push(imagesV);

        }
        var scrollView = Ti.UI.createScrollableView({
            views : views,
            top : 0,
            right : 0,
            bottom : 0,
            left : 0,
            width : device_width,
            height : scrollable_height,
            showPagingControl : true,
            pagingControlHeight : 30,
        })
        rows.add(scrollView);
        scrollView.addEventListener('click', function(e) {
            alert('ScrollView received click event, source = ' + e.source);
        });
        rowData[0] = rows
    }
    loader.send()

    loadBoard();
    loadnews()
}

    • Like I said, add the event listener to the imageviews themselves. Sometimes you just have to try it and throw a breakpoint into the event handler to figure out what data is accessible to you.

      In this case, when the event handler fires, the event object will contain a reference to the imageview in the "source" property. From that imageview, you can get the image property:

      for (var i = 0; i < highlights.count; i++) {
            var imagesV = Ti.UI.createImageView({
                image : highlights.posts[i].filePath,
                left : 0,
                top : 0,
                width : device_width,
                height : scrollable_height
            })

            imagesV.addEventListener ('click', function (e) {
               alert (e.source.image);
            });
            views.push(imagesV);

        }

      — commented June 1st 2012 by Jason Priebe

    • One other thing – you might want to try listening for 'singletap' events rather than 'click'. I think that 'click' gets fired prematurely when you try to swipe your scrollable view.
