  • Hello Angelo Gelmi,

    Thanks for writing. I have tested this issue with my test code its working good in latest Ti Tools. I use a Android 4.2.2 device for testing. It may be help to you.

    Please check the attachments: https://www.dropbox.com/sh/iq72z7z18hyl2ql/KNS9LqgItI

    Please follow my step for testing:

    • Create a classic Ti project
    • Paste this code in app.js file and tiapp.xml file
    • Enable cloud and add module cloud and cloudpush
    • Go to cloud.google.com, create a new project with the same name of titanium app. Click the project and generate the server key
    • Enable Google Cloud Messaging for Android in API section from cloud.google.com
    • Go to my.appcelerator.com and drag the project and click Manage ACS
    • At Application Dashboard in setting option insert GCM API key and GCM Sender ID(Example :427480293456) from provided by cloud.coogle.com
    • Create user in development option that you will use for login
    • Run this code with testing environment and after running click on Register For Push Notification Button 
    Titanium.UI.setBackgroundColor('#000');

var win = Ti.UI.createWindow({
    backgroundColor : '#ccc',
    title : 'Android Cloud Push Notification'
});

var CloudPush = require('ti.cloudpush');
CloudPush.debug = true;
CloudPush.enabled = true;
CloudPush.showTrayNotificationsWhenFocused = true;
CloudPush.focusAppOnPush = false;

var deviceToken;
var Cloud = require('ti.cloud');
Cloud.debug = true;

var submit = Ti.UI.createButton({
    title : 'Register For Push Notification',
    color : '#000',
    height : 73,
    width : 200,
    top : 100,
});

win.add(submit);

submit.addEventListener('click', function(e) {
    CloudPush.retrieveDeviceToken({
        success : function deviceTokenSuccess(e) {
            alert('Device Token: ' + e.deviceToken);
            deviceToken = e.deviceToken;
            loginDefault();
        },
        error : function deviceTokenError(e) {
            alert('Failed to register for push! ' + e.error);
        }
    });
});

function loginDefault(e) {
    //Create a Default User in Cloud Console, and login
    Cloud.Users.login({
        login : 'rmitro@appcelerator.com',
        password : 'rakhimonimitro'
    }, function(e) {
        if (e.success) {
            alert("login success");
            defaultSubscribe();
        } else {
            alert('Error: ' + ((e.error && e.message) || JSON.stringify(e)));
        }
    });
}

function defaultSubscribe() {
    Cloud.PushNotifications.subscribe({
        channel : 'alert',
        device_token : deviceToken,
        type : 'android'
    }, function(e) {
        if (e.success) {
            alert('Subscribed for Push Notification!');
        } else {
            alert('Error:' + ((e.error && e.message) || JSON.stringify(e)));
        }
    });
}



win.open();
