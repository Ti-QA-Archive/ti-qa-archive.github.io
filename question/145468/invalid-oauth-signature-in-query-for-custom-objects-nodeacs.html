Invalid OAuth signature in Query for custom objects Node.ACS

I am trying to retrieve a set of custom objects created by the current logged in user but I get a 

{
  "success": false,
  "error": true,
  "code": 401,
  "message": "Invalid OAuth signature"
}

Code snippet in Node.ACS

ACS.Users.showMe(function(data) {
        ACS.Objects.query({
            classname: 'Topic',
            page: 1,
            per_page: 10,
            where: {
                "user_id": data.users[0].id
            },
            order: 'starts'
        }, function(data) {
            res.json(data);
        }, req, res);
    }, req, res);
