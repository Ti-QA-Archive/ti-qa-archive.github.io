    Hello Weina,

    just Try this code

    var win = Ti.UI.currentWindow;

var arr = [];

var view = Ti.UI.createView({
    top : 0,
    bottom : 0,
    backgroundColor : 'red'
});
win.add(view);

view.addEventListener('click',function(e){
    for (var i=0; i < arr.length; i++) {
      arr[i].blur();
    };
});

var textfield1 = Ti.UI.createTextField({
    top : 10,
    value : 'text1',
    left : 10,
    width : 100,
    height : 30
});

view.add(textfield1);

var textfield2 = Ti.UI.createTextField({
    top : textfield1.top + textfield1.height,
    value : 'text2',
    left : 10,
    width : 100,
    height : 30
});

view.add(textfield2);

arr.push(textfield1);
arr.push(textfield2);

    hope it will help you…
