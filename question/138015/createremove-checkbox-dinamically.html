    Hi Nelson,

    Try this demo code set your height and widht as per your requirement.

    var scrollView = Ti.UI.createScrollView({
    height : 320,
    width : 320,
    top : 0,
});
var View = Ti.UI.createView({
    height : 'auto',
    width : 320,
    top : 0
});
var top = 0;
for (var i = 0; i < 10; i++) {
    top += 60;
    checkbox = Ti.UI.createSwitch({
        style : Ti.UI.Android.SWITCH_STYLE_CHECKBOX,
        left : '30 px',
        top : top,
        title : i,
        // visible : false,
        zIndex : '100',
    });
    checkbox.visible = true;
    View.add(checkbox);
}
scrollView.add(View);
win.add(scrollView);

var btn = Ti.UI.createButton({
    height : 32,
    width : 32,
    right : 30,
    bottom : 15,
    title : 'Remove'
});
btn.addEventListener("click", function(e) {
    scrollView.remove(scrollView.children[0]);
});
win.add(btn);
