Can't display an image over another image :(

Hi people!
I am making an app and I am new to Titanium. I want to ask a very basic question, I am trying to show an image over another image. I followed tutorial and made this program. It shows correct output when I run it on browser but when I run it on any Android device it only shows the first image i.e headerBar image which I am using in following code. Please help me what am I missing?
I also tried creating a label and display it over the image but same problem.

    layout:'vertical',
    title: "Home",
    backgroundColor: "#FFFFFF",
    exitOnClose: true,
    fullscreen:false
});


var headerBarImageView = Ti.UI.createImageView({
    image: "Images/Header-bar.png",
    height: "10%",
    width: "100%",
    top: 0,
    zIndex: 1
});

var logoImageView = Ti.UI.createImageView({

    zIndex: 2
});

var view1 = Ti.UI.createView();

headerBarImageView.add(logoImageView);
view1.add(headerBarImageView);

win.add(view1);

win.open();

        title:'Close',
        top: 100,
        width: 200,
        height: 100
    });

    win.add(closeButton);

    closeButton.addEventListener('click',function()
    {
        win.close();
    });

    var view1 = Ti.UI.createView({
        height : Ti.UI.SIZE,
        width : Ti.UI.SIZE,
        backgroundColor : 'transparent' 
    });

    view1.add(headerBarImageView);

    view1.add(logoImageView);

    win.add(view1);

    win.open();
