    You are going to need to do something like

    var rows = [
    [ 0, 'red' ],
    [ 1, 'blue' ],
    [ 2, 'green' ]
];

function make_csv(table)
{
    var text = "";
    for(var row = 0; row < table.length; ++row)
    {
        var r = table[row];
        for(var col = 0; col < r.length; ++col)
        {
            text += '"' + r[col] + '"';

            // do not put comma after last column
            if(col < (r.length-1))
            {
                text += ',';
            }
        }

        // next line
        text += '\n';
    }
    return text;
}
