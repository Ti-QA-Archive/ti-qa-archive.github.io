    Hi

    I have tried adding and removing the route from a map and it works without any problems, except for a small visual issue that the route may not visually leave until you click the map. Can you try this code and see what you end up with?

    Create a brand new simple project and replace the contents of the app.js with the code below.

    Ti.UI.setBackgroundColor('#000');
var win = Ti.UI.createWindow({  
    height: Ti.UI.FILL,
    title: 'Map',
    width: Ti.UI.FILL
});
function addPin(obj) {
    var pin = Ti.Map.createAnnotation({
        latitude: obj.lat,
        longitude:  obj.lon,
        title: obj.title,
        subtitle: obj.subtitle,
        pincolor: Ti.Map.ANNOTATION_RED,
        animate: true
    });
    return pin;
}
var data = [];
data.push(addPin({
    lat: 52.97,
    lon: -1.18,
    title: 'Nottingham'
}));
data.push(addPin({
    lat: 53.3822,
    lon: -1.4667,
    title: 'Sheffield'
}));
data.push(addPin({
    lat: 51.5171,
    lon: -0.1062,
    title: 'London'
}));

//var points = ;
var route = {
    color: 'red',
    name: 'test',
    points: [
        { latitude: 52.97, longitude: -1.18 },
        { latitude: 53.3822, longitude: -1.4667 },
        { latitude: 51.5171, longitude: -0.1062 }
    ],
    width: 2
};
var map = Ti.Map.createView({
    annotations: data,
    height: Ti.UI.FILL,
    region: {
        latitude: 52.97,
        longitude: -0.18, 
        latitudeDelta: 2,
        longitudeDelta: 2
    },
    width: Ti.UI.FILL
});
win.add(map);
map.addRoute(route);
var btn = Ti.UI.createButton({
    right: 50,
    title: 'Remove Route',
    top: 50
});
win.add(btn);
btn.addEventListener('click', function (e) {
    map.removeRoute(route);
    map.fireEvent('click');
});
var tab = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Map',
    window: win
});
var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab);
tabGroup.open();
