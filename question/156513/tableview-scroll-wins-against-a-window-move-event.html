tableview scroll wins against a window move event

I have the following code

Titanium.UI.setBackgroundColor('#000');

var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});


var posLabel = Titanium.UI.createLabel({
    color : 'eee',
    backgroundColor : '666',
    text : 'Position =',
    width : '100%',
    height : 30,
    zIndex : 2,
    bottom : 0
});
win1.add(posLabel);

var table = Titanium.UI.createTableView();

var data = [];
for(var i = 0; i < 6; i++) {
    var row = Ti.UI.createTableViewRow();
    var label = Ti.UI.createLabel({
        font : {
            fontSize : 20,
            fontWeight : 'bold'
        },
        left : 10,
        text : 'Row ' + (i)
    });

    row.add(label);
    row.backgroundColor = 'f0f0f0';
    data.push(row);
}

table.setData(data);
win1.add(table);

win1.addEventListener('touchmove', function(e) {
    table.touchEnabled = false;
    posLabel.text = 'Position = X: ' + e.x + ' Y: ' + e.y;
});

win1.open();

I want to be able to drag the tableview around but when I move up or down the scrollevent for the tableview takes over. Is there anyway to accomplish this. In my code the 2nd try to move works fine since then the touchEnabled is set
