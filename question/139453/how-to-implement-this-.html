  • Make view model. For example:

    var viewModel = [
    { title: 'One', input: Ti.UI.createTextField, conf: {} },
    { title: 'Two', input: Ti.UI.createTextField, conf: {} }
];

function createForm(parent) {
    parent.layout = 'horizontal'; //force horizontal layout

    for (var i = 0; i < viewModel.length; ++i) {
        parent.add(Ti.UI.createLabel({ text: viewModel[i].title, width: '50%' });
        parent.add(viewModel[i].input(viewModel[i].conf);
    }
}

    Now, if you want to add one more text field, you can just modify view model.

    That should get you started.
