var data = [];

var storedprefid = Titanium.App.Properties.getString("list");

if (storedprefid == null || storedprefid == '') { 

    Titanium.UI.Android.openPreferences();

} else {

var storedprefidNEW = Titanium.App.Properties.getString("list");    

var xhr = Ti.Network.createHTTPClient();
xhr.open("GET","http://www.freshpointmarketing.com/iphone/objects/XML/AND.php?ID=" + storedprefidNEW);
xhr.onload = function()
{
    try
    {
        var doc = this.responseXML.documentElement;
        var items = doc.getElementsByTagName("item");
        var x = 0;
        var doctitle = doc.evaluate("//channel/title/text()").item(0).nodeValue;
        for (var c=0;c<items.length;c++)
        {
            var item = items.item(c);
            var thumbnails = item.getElementsByTagName("media:thumbnail");
            if (thumbnails && thumbnails.length > 0)
            {
                var media = thumbnails.item(0).getAttribute("url");
                var title = item.getElementsByTagName("title").item(0).text;
                var description = item.getElementsByTagName("description").item(0).text;
                var row = Ti.UI.createTableViewRow({height:80,backgroundColor:'#3192fd' });

                var label = Ti.UI.createLabel({
                    text:title,
                    left:72,
                    top:5,
                    bottom:5,
                    right:5,
                    height:22,            
                    font:{fontSize:16,fontWeight:'bold'},

                });

                var label1 = Ti.UI.createLabel({
                    text:description,
                    left:72,
                    top:50,
                    bottom:5,
                    right:5,
                    height:22,                
                    font:{fontSize:12,fontWeight:'bold'},                


                });                

                row.add(label);
                row.add(label1);

                var img;
                if (Titanium.Platform.name == 'android') 
                {
                    // iphone moved to a single image property - android needs to do the same
                    img = Ti.UI.createImageView({
                        url:media,
                        left:5,
                        height:60,
                        width:60
                    });

                }
                else
                {
                    img = Ti.UI.createImageView({
                        image:media,
                        left:5,
                        height:60,
                        width:60
                    });

                }
                row.add(img);
                data[x++] = row;
                row.url = item.getElementsByTagName("link").item(0).text;
            }
        }
        var tableview = Titanium.UI.createTableView({data:data});
        Titanium.UI.currentWindow.add(tableview);


        tableview.addEventListener("focus",function(e) {

            Ti.API.info('test focus');

        tableview.reload({data:data});

        });

        tableview.addEventListener("blur",function(e) {

            Ti.API.info('test focus');

        tableview.reload({data:data});

        });

        tableview.addEventListener('click',function(e)
        {
            var w = Ti.UI.createWindow({title:doctitle});

             Ti.Platform.openURL(e.rowData.url);

            w.open({modal:true});
        });
    }
    catch(E)
    {
        alert(E);
    }
};
xhr.send();

}

Titanium.Android.currentActivity.onCreateOptionsMenu = function (e) {

    var menuItem = e.menu.add({ title: 'Select Your Location' });
    menuItem.setIcon('KS_nav_ui.png');
    menuItem.addEventListener('click', function (e) {

        Titanium.UI.Android.openPreferences();
    });

    var menuItem = e.menu.add({ title: 'Close' });
    menuItem.setIcon('KS_nav_ui.png');
    menuItem.addEventListener('click', function (e) {

        Titanium.Android.Menu.close;
        //win1.refresh();   
    });

};
