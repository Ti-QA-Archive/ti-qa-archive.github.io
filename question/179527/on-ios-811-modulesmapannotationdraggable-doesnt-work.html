  • Hi Tetsuya Shiraishi,

    Thanks for your concern.

    Yes, we can drag the annotation on Ti.Map. Please check my test code below and see if it works for you.

    var Map = require('ti.map');
var win = Ti.UI.createWindow();

var annotations = [
    Map.createAnnotation({
        latitude: 32.8245525,
        longitude: -117.0951632,
        title: 'Somewhere',
        animate: true,
        pincolor: Map.ANNOTATION_GREEN,
        draggable: true
    }),
];
var mapview = Map.createView({
    mapType: Map.NORMAL_TYPE,
    region: {
        latitude:32.8245525,
        longitude:-117.0951632,
        latitudeDelta:.05,
        longitudeDelta:.05
    },
    animate:true,
    regionFit:true,
    userLocation:false,
    annotations: annotations
});

win.add(mapview);
win.open();

    Thanks

    • workarounds…!!
      Enjoy it….

      var MapModule = require('ti.map');
var win = Ti.UI.createWindow({backgroundColor: 'white'});


var mapview = MapModule.createView({
    mapType: MapModule.STANDARD_TYPE,
    region: {latitude:'51.51098606916964', longitude:'-0.1338958740234375', latitudeDelta:0.04, longitudeDelta:0.04},
    animate:true,
    regionFit:true,
    userLocation:true
});


var addLocation = Titanium.UI.createButton({
    title: 'Colocar pin',
    backgroundColor: '#f5a11c',
    borderRadius: 5,
    borderColor: '#b16003',
    color: 'white',
    width: '100dp',
    top: 50,
       height: '25dp' 
    });


addLocation.addEventListener('click',function(){
    if (annoAdded == true){
        mapview.removeAnnotation(addAnno);
        addNewAnno();
    }else{
        addNewAnno();
    }
});

var annoAdded = false;
var addAnno;
function addNewAnno(){
    addAnno = MapModule.createAnnotation({
        latitude:Titanium.App.Properties.getString('goLat'),
        longitude:Titanium.App.Properties.getString('goLong'),
        title:"Add title",
        subtitle:'add subtitle',
        pincolor: MapModule.ANNOTATION_GREEN,
        animate:false,
        draggable:true
    }); 
    mapview.addAnnotation(addAnno);
    mapview.selectAnnotation(addAnno);
    annoAdded = true;
}

mapview.addEventListener('regionChanged',function(evt){
    Titanium.App.Properties.setString('goLat',evt.latitude.toPrecision(10));
    Titanium.App.Properties.setString('goLong',evt.longitude.toPrecision(10));
    if (annoAdded == true){
        mapview.removeAnnotation(addAnno);
        addNewAnno();
    }
});

win.add(mapview);
win.add(addLocation);
win.open();
