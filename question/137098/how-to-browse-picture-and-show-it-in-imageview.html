  • Hi Arun,
    Call this function and you will show all picture from your device.

    var newImg;
function openLibrary() {
    Titanium.Media.openPhotoGallery({
        success : function(event) {
            newImg = event.media;
            ImgView.image = newImg;

        },
        cancel : function() {
        },
        error : function(error) {
        }
    });
}
