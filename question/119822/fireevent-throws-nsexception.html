fireEvent throws NSException

When I fireEvent with a data object that includes a method I get an NSExcpetion thrown and terminate.

This is the code I added on the top of a fresh app.js file:

var createTest = function(spec) {
    var title = 'Title';
    var that = {};
    that.getTitle = function () {
        return title;
    };
    return that;
};

var newTest = createTest();
var t = function (data) {  
    Titanium.API.debug('testfunc: ' + data);
};
Ti.App.addEventListener('testevent',t);
Ti.App.fireEvent('testevent',{'test':newTest});

If I remove the getTitle function property I can avoid the exception. Why is this not working?
(using Ti studio and mobile SDK 1.6.2)
