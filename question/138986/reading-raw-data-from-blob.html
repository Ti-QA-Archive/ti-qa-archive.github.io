  • Selfsolved:

    exports.get = function(foo) {
    switch (typeof foo) {
        case 'string':
            return Ti.createBuffer({
                value : foo
            });
            break;
        case 'object':
            var instream = Titanium.Stream.createStream({
                mode : Titanium.Stream.MODE_READ,
                source : foo // e.media is a Blob
            });
            var buffer = Ti.createBuffer({
                length : foo.length
            });
            while ((instream.read(buffer)) > 0) {
            }
            instream.close();
            return buffer;
            break;
    }
}
