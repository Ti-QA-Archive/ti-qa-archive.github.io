    • Hi, Make sure you have the 2.1.3GA SDK version. (I think they addressed it in that)

      I have been using something a bit like this and adding '+tallFix+' to my image urls : 

      

var isRetina = false;
var isTall = false;
var tallFix = '';
var extra_height = 0;
if(Ti.Platform.displayCaps.density=="high"){
    isRetina = true;
    // iPhone5
    if(Ti.Platform.displayCaps.platformHeight==568){
        isTall = true;
        tallFix = '-568h@2x';
        extra_height = 88;
        //Ti.API.info('HELLO iPHONE5!!');
    }
}
