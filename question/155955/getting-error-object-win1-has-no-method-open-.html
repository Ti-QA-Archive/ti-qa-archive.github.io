  • Hi,
    you need to add two more function in Application window for open and close

    this.open = function(_args) {
    win.open(_args);
};

this.close = function(_args) {
    win.close(_args);
};

    your ApplicationWindow.js should be like this

    //Application Window Component Constructor
function win1() {



    var wn = Ti.UI.createWindow({
        backgroundColor:'#ffffff',
        navBarHidden:true,
        exitOnClose:true
    });

    var textField = Ti.UI.createTextField({
  borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
  color: '#336699',
  top: 10, 
  left: 10,
  width: 250, 
  height: 60
});

var alertButton2 = Titanium.UI.createButton({

    width:22,
    height:22,
    title:'click',
    right:15,
    top:49
});

    alertButton2.addEventListener('click',function(){
        //require the win2 here
        var _win2 = require('/win2'); //this should be your win2.js file path
        var win2 = new _win2(myText.value); //passing the data in win2
        win2.open();
    });

//This method is added into Window object in app.js (where ever require this js file)
 this.open = function(_args) {
    win.open(_args);
};


//This method is added into Window object in app.js (where ever require this js file)
this.close = function(_args) {
    win.close(_args);
};

}

//make constructor function the public component interface
module.exports = win1;

    I hope this will fix your issue.
