  • I rebuilded the DB in another software (SQLite Browser) and then it worked to load the DB, but now the data var seems to be empty :(
    This is my getFilters method

    function getFilters()
{
    var data = [];
    var db = Ti.Database.install('filters.db','filters');
    var rows = db.execute('SELECT * FROM Filters');
    while (rows.isValidRow())
    {
        data.push({
            title: rows.fieldByName('Manufacture') + ' #' + rows.fieldByName('FilterNR'),
            RGB: rows.fieldByName('RGB'),
            Description: rows.fieldByName('Description')
        });
        rows.next();
    }
    return data;
}

    And i call it here

        var tableViewMain = Titanium.UI.createTableView({
        top: '60dip',
        bottom: '60dip',
        data: getFilters()
    });
