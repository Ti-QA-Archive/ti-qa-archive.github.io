Right, left on imageView does nothing

I'm running Titanium Studio 3.3.0 on OS X and testing on Android 4.4.2 on a Nexus 5.

I'm trying to get a next and back button to align left and right but they keep becoming centered.

The relevant code is

var window = Titanium.UI.currentWindow;
var header = Titanium.UI.createView({
    height: Math.floor(Ti.Platform.displayCaps.platformHeight * .095),
    width: '100%'
});

var backBut = Titanium.UI.createImageView({
    backgroundColor: 'transparent',
    image: 'back',
    left: 0
});
backBut.addEventListener('click', function () {
    win.close();
});
var nextBut = Titanium.UI.createImageView({
    backgroundColor: 'transparent',
    image: 'next.png',
    right: 0,
});
