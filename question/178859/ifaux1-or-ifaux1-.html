  • Hi,

    To add the global variable in alloy.js you should use this way in alloy.js:

    Alloy.Globals.aux = 0;

    In Index.js use it in this way:

    

loginReq.onload = function()
{
    var json = this.responseText;
    var response = JSON.parse(json);
    if (response.logged == true)
    {
        Alloy.Globals.aux=1;
        homes.open();
        $.index.close();
        alert("Welcome " + response.name + ". Your email is: " + response.email);
    }
    else
    {
        alert(response.message);
    }
};

    And in other controller home.js reterive the value and preform functionally in this way:

    
Alloy.Globals.loading.hide();
var picker = Ti.UI.createPicker({
    left:10, 
    top: 10, 
    height:50, 
    opacity:0.5, backgroundColor: "Black", borderRadius:"2"
}); 
var data = [];
data.push(Titanium.UI.createPickerRow({id:'0',title:'Mis cursos'})); 
data.push(Titanium.UI.createPickerRow({id:'1',title:'Todos Cursos'})); 
if(aux===1){
    data.push(Titanium.UI.createPickerRow({id:'2',title:'Crear'})); 
}
picker.add(data);

    This is the correct way to use the global variable in multiple controllers. Hope this will solve your problem.
