  • Here is some code that I've tested and works well. It's based on experimenting with the tutorial by M. Prakash found here: Alloy Framework - Use Preloaded Sqlite Database.

    NOTE: The following assumes a Collection declared as follows:

    var dataCollection = Alloy.Collections.fighters;

    Create:

    //*** Create a model object with the desired data
var model = Alloy.createModel('fighters',  { "name":  "Mark E. P.", "nickname":  "Champion of Breakfasts" });
//*** Insert into the model into the collection so that table is updated when you refresh
dataCollection.add(model);
//*** Insert the new record into the database
dataCollection.forEach(function(model){ model.save(); });

    Read:

    //*** Set the criteria for the record in question
var model = Alloy.createModel('fighters', {"id": "11"});
//*** Read the selected data into a model object
var detailObj = dataCollection.get(model);
//*** Use the model.get() method to retrieve the data
var myName = detailObj.get('name');
var myNickame = detailObj.get('nickname');

    Update:

    //*** Set the criteria for the record in question
var model = Alloy.createModel('fighters', {"id": "11"});
//*** Use the .set method to create a model object with the new data
model.set({ "name": "Betram K. Noobish", "nickname": "Holographic Hero" } );
//** Update the model so that the record refreshes in your table
model.save();
//*** Update the database
dataCollection.forEach(function(model){ model.save(); });

    Delete:

    //*** Set the criteria for the record in question
var model = Alloy.createModel('fighters', {"id": "11"});
//*** Remove model so that it no longer shows up in the table
dataCollection.remove(model);
//*** Delete record from database
dataCollection.sync('delete', model, function (resp) {
    //*** Error handler for sync 
    alert(resp); 
});

    Please let me know if you have any problems with this code (and please give details). I'll do my best to correct it quickly.
