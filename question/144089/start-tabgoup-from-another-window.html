  • Hi you can try like this… for single file code..

    var baseWin = Titanium.UI.createWindow({
    title : 'BASE',
    backgroundColor : '#13386c'
});

var main = Titanium.UI.createWindow({
    title : 'Home',
    ExitOnClose : true,
    navBarHidden : false,
    backgroundColor : 'red'
});

var test = Titanium.UI.createButton({
    title : 'Test',
    style : Titanium.UI.iPhone.SystemButtonStyle.DONE
});

test.addEventListener('click', function(e) {
    alert('ashish');
    var tabGroup = Ti.UI.createTabGroup({
        backgroundColor : 'blue',
        activeTabBackgroundColor : 'yellow',
        color : 'white'
    });

    var win1 = Ti.UI.createWindow({
        backgroundColor : '#13386c',
        navBarHidden : true,
        width : '100%'
        //fullscreen : false
    })

    var tab1 = Ti.UI.createTab({
        window : win1,
        title : 'win1',
        backgroundColor : 'blue',
        activeTabBackgroundColor : 'yellow',
        color : 'white'
    });

    tabGroup.addTab(tab1);
    //tabGroup.open();

    navGroup.open(tabGroup);
});

var controls = Ti.UI.iOS.createToolbar({
    items : [test],
    bottom : 'base',
    borderTop : true,
    borderBottom : true,
    translucent : false
});

main.add(controls);

//main.open();

var navGroup = Ti.UI.iPhone.createNavigationGroup({
    window : main
});

main.navGroup = navGroup;

baseWin.add(navGroup);
baseWin.open()
