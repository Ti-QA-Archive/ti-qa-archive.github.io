TableView eventlistner issues

I've used Dan Tamas' tutorial to create a tableview with multiple labels in each tableviewrow….

http://cssgallery.info/custom-row-for-tableview-in-appcelerator-titanium/

However, I've been having some issues with attaching an eventlistener to each of those rows in the table.

Each tableviewrow is similar to this

var win = Titanium.UI.currentWindow;

var TheTable = Titanium.UI.createTableView();

var CustomData = [
{test_title:"Title1", test_subtitle:"Sub1",hasChild:true, url:'testing1.js'},
{test_title:"Title2"), test_subtitle:"Sub2",hasChild:true, url:'testing2.js'}
];

var data=[];

for (var i = CustomData.length - 1; i >= 0; i--){

    var row = Titanium.UI.createTableViewRow({height:'auto', layout:'vertical'});

    var test_title = Titanium.UI.createLabel({
        text:CustomData[i].test_title,
    font:{fontSize:16, fontWeight:'bold'},
    top:5,
    left:10,
    height:'auto'
    });

    var test_subtitle =  Titanium.UI.createLabel({
    text:CustomData[i].test_subtitle,
    font:{fontSize:14},
    left:10,
    top:3,
    width:270,
    height:'auto'
    });


    row.add(test_title);
    row.add(test_subtitle);
    row.hasChild=CustomData[i].hasChild;
    row.className = 'tour_row';
    data.push(row);
};


TheTable.setData(data);


TheTable.addEventListener('click', function(e){

        var win3 = Titanium.UI.createWindow({
            url:e.rowData.url
                            });

        Titanium.UI.currentTab.open(win3,{animated:true});
});

win.add(TheTable);

When I click on the table row it fires… however the url doesn't load as I want it.

Anybody got any ideas how to solve my problem?

    You only define the url property in the CustomData array, but never add the property to the actual table row. Try something like:

    var row = Titanium.UI.createTableViewRow({
  url: CustomData[i].url,
  height:'auto', 
  layout:'vertical'
});
