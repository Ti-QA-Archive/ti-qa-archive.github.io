    • Ok, here is my own answer …
      Structure your code like this (try with the example, I add som buttons with a listener between each windows)

      app.js

      var main = Ti.UI.createWindow();
var win1 = Ti.UI.createWindow({url:'first.js'});
Ti.App.nav = Ti.UI.iPhone.createNavigationGroup({window:win1});
main.add(Ti.App.nav);
main.open();

      first.js

      var win1 = Titanium.UI.currentWindow;
var nav = Ti.App.nav;
var win2 = Ti.UI.createWindow({url:'second.js'});
nav.open(win2);

      second.js

      var win2 = Titanium.UI.currentWindow;
var nav = Ti.App.nav;
nav.window = win2;
var win3 = Ti.UI.createWindow({url:'third.js'});
nav.open(win3);

      third.js

      var win3 = Titanium.UI.currentWindow;
var nav = Ti.App.nav;
nav.window = win3;
var win4 = Ti.UI.createWindow({url:'fourth.js'});
nav.open(win4);

      fourth.js

      var win3 = Titanium.UI.currentWindow;
var nav = Ti.App.nav;
nav.window = win3;

      SVN : https://sourceforge.net/p/tinavgroup/code-0/2/tree/navGroup/
