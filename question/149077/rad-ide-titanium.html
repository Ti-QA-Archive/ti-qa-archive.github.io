    Hi

    Firstly - welcome.

    A big difference between what you are used to and Titanium is that there is no visual IDE. Some find that limiting but the vast majority do not. It may be strange for a while but you will find the lack of an IDE is a benefit as you concentrate on the quality of the code rather than the position of things.

    Do not get me wrong - I am used to having visual IDEs with other dev environments, and the ability to select something and add code to it can seem easy. But here I have found I focus on structure.

    There are lots of guides, videos, tutorials, example apps and example code to play with. Some things to start with;

    • Titanium SDK & Titanium Studio (Main Documentation hub)
    • API
    • Guides
    • Videos

    Add some of the sample code and you will be off to flying start.

    • Here is something to play with - think a slightly more detailed hello world;

      Create a new default project and then replace the contents of the app.js with the code below.

      Ti.UI.setBackgroundColor('#fff');

var win1 = Ti.UI.createWindow({
    backgroundColor: '#FFF',
    height: Ti.UI.FILL,
    title: 'Window 1',
    width: Ti.UI.FILL
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title: 'Window 1',
    window: win1
});

var data = [
    { title: 'One', hasChild: true },
    { title: 'Two', hasChild: true },
    { title: 'Three', hasChild: true },
    { title: 'Four', hasChild: true },
    { title: 'Five', hasChild: true }
];
var tbl = Ti.UI.createTableView({
    data: data,
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
win1.add(tbl);
tbl.addEventListener('click', function (e) {
    var win = Ti.UI.createWindow({
        backgroundColor: '#FFF',
        height: Ti.UI.FILL,
        title: e.rowData.title,
        width: Ti.UI.FILL
    });
    var lbl = Ti.UI.createLabel({
        height: Ti.UI.SIZE,
        text: e.rowData.title,
        top: 100,
        width: Ti.UI.SIZE
    });
    win.add(lbl);
    var btn = Ti.UI.createButton({
        bottom: 100,
        height: Ti.UI.SIZE,
        title: 'Press',
        width: Ti.UI.SIZE
    });
    win.add(btn);
    btn.addEventListener('click', function (e) {
        alert('Hello');
    });
    win.add(btn);
    if (Ti.Platform.name === 'android') {
        win.open();
    } else {
        tab1.open(win, { animated: true });
    }
});

var win2 = Ti.UI.createWindow({
    backgroundColor: '#FFF',
    height: Ti.UI.FILL,
    title: 'Window 2',
    width: Ti.UI.FILL
});
var tab2 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title: 'Window 2',
    window: win2
});

var tabGroup = Titanium.UI.createTabGroup();
tabGroup.addTab(tab1);
tabGroup.addTab(tab2);
tabGroup.open();
