    finally create model

    exports.definition = {
    config: {
    columns: {
    "name": "string",
    "age": "number",
    "address":"string",
    "sex":"string"
    },
    adapter: {
    type: "sql",
    collection_name: "student",
    db_name: "todaystest",
    dbfile: "/todaystest.sqlite"
    }
    },
    extendModel: function(Model) {     .extend(Model.prototype, {
    // extended functions and properties go he

            insertStudent: function(){
            var self=this;
            var insertQuery='INSERT INTO Student (name,age,address,sex) VALUES ("'+self.attributes.name+'",'+self.attributes.age+','+self.attributes.address+','+self.attributes.sex+')';
            this.save({query : insertQuery});


        },
        deleteStudent: function(){
            var self=this;
            var deleteQuery='DELETE FROM Student' ;
            self.destroy({query : deleteQuery});


        }
    });


    return Model;


},

extendCollection: function(Collection) {
    _.extend(Collection.prototype, {
        // extended functions and properties go here
        getAll : function(){
            var self=this;
            self.fetch({query:"Select Name,Age,address,sex From Student" });
        }
    });

    return Collection;
}

    };
