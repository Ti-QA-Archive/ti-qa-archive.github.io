Android crash downloading big text file

Dears,

 after TITANIUM update i got stucked on this simple function. On IOS works perfectly and also on AND before the update!

when i call my function with

scaricadump ('myFile.txt','http://www.myDomain.com/myFile.php')

i expect that function write myFile.txt on app_data directory.
Problem on Android is memory. The text file contain more than 3000 long lines, about 1.5Mb.
This crash my app on both, simulator and device.
Trying with little file works fine. 

function scaricadump(file, remote) {
    var PD = Titanium.Network.createHTTPClient();
    PD.timeout = 50 * 1000;
    var f = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, file);
    if (f.exists()) {
        f.deleteFile();
        PD.onload = function() {
            Ti.API.info(this.responseText); //DEBUG
            f.write(this.responseText);
        };
        PD.open('POST', remote);
        PD.send();
    } else {
        PD.onload = function() {
            f.write(this.responseText);
        };
        PD.open('POST', remote);
        PD.send();
    }
    return f.nativePath;
}

Do you have any suggestions?
Return back on code that previously worked great…. this drive me crazy!!! :-D

thanks
Steve

  • Hi steven, this is my solution.

    
function download(file, remote) {
    var PD = Titanium.Network.createHTTPClient();
    PD.onload = function() {
        var tempf = Ti.Filesystem.getFile(PD.responseData.nativePath);
        var dest = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, file);
        if (dest.exists)
            dest.deleteFile();
        tempf.move(dest.nativePath);
    };
    PD.timeout = 50 * 1000;
    PD.open('POST', remote);
    PD.send();
}
