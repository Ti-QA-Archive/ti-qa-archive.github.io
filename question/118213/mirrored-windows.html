mirrored windows

Any ideas how to get two windows mirrored to each other?
The idea is t do an iPad game where two users get presented with the same question, each user is at on end of the iPad so the screen needs to be splited in two and the question are mirrored so that each user can read it.
is this possible with titanium?

    Hi,

    Try the code below, I believe that this is what you want.

    
// SPLIT WINDOWS
var screenHeight = 1004;
var windowHeight = screenHeight/2;



// PLAYER 1
var w1 = Ti.UI.createWindow({
    backgroundColor:'#cff',
    height:windowHeight,
    top:windowHeight
});


q1a = Ti.UI.createLabel({
    text:'This is an question?',
    font:{fontSize:26},
    top:50,
    left:10,
    right:10,
    height:50
});
r1a = Ti.UI.createButton({
    title:'Yes',
    width:100,
    height:50,
    top:300,
    left:100
});
r2a = Ti.UI.createButton({
    title:'No',
    width:100,
    height:50,
    top:300,
    right:100
});

w1.add(q1a);
w1.add(r1a);
w1.add(r2a);




// PLAYER 2
var t = Ti.UI.create2DMatrix();
t = t.rotate(180);
var w2 = Ti.UI.createWindow({
    backgroundColor:'#ffc',
    height:windowHeight,
    top:0,
    transform:t
});

q1b = Ti.UI.createLabel({
    text:'This is an question?',
    font:{fontSize:26},
    top:50,
    left:10,
    right:10,
    height:50
});
r1b = Ti.UI.createButton({
    title:'Yes',
    width:100,
    height:50,
    top:300,
    left:100
});
r2b = Ti.UI.createButton({
    title:'No',
    width:100,
    height:50,
    top:300,
    right:100
});

w2.add(q1b);
w2.add(r1b);
w2.add(r2b);




// STATUS
var w3 = Ti.UI.createWindow({
    backgroundColor:'#fcf'
});
var winner = Ti.UI.createLabel({
    text:'',
    height:50,
    font:{fontSize:30},
    color:'#f00',
    top:(windowHeight-25),
    left:0,
    right:0,
    textAlign:'center'
});
var mtx_Sync = Ti.UI.create2DMatrix();
var ani_spin = Titanium.UI.createAnimation({
    curve:Ti.UI.ANIMATION_CURVE_LINEAR
});
mtx_Sync = mtx_Sync.rotate(90);
ani_spin.transform = mtx_Sync;
ani_spin.duration = 400;
ani_spin.addEventListener('complete', function(e) {
    mtx_Sync = mtx_Sync.rotate(90);
    ani_spin.transform = mtx_Sync;
    winner.animate(ani_spin);
});
w3.add(winner);





// EVENTS
r1a.addEventListener('click',function(){
    w1.close();
    w2.close();
    winner.text = 'Player 1 wins!';
    w3.open();
    winner.animate(ani_spin);
});
r1b.addEventListener('click',function(){
    w1.close();
    w2.close();
    winner.text = 'Player 2 wins!';
    w3.open();
    winner.animate(ani_spin);
});


// OPEN WINDOWS
w1.open();
w2.open();
