    Hi Adnan

    This should do it. As I mentioned it replaces the heavyweight webview and the need to manage CSS and HTML in the process.

    This uses a scrollview (same way as the underlying nature of a tableview and then adds rows using the exmaple function. The function accepts simple parameters which you can of course change to meet you needs.

    I was not sure how you refereed to the bars position so I went with percentage and left the right hand number unrelated to that percentage as your example graphic did.

    I hope this at least shows you a different way that you can try. Obviously you would fit this to your own code needs.

    Let me know what you think.

    function addBar(obj) {
    var viewBar = Ti.UI.createView({
        backgroundColor: obj.backgroundColor,
        height: 50,
        width: Ti.UI.FILL
    });

    var viewTitle = Ti.UI.createView({
        backgroundColor: obj.barColor,
        height: Ti.UI.FILL,
        left: 0,
        width: obj.percentage + '%'
    });
    viewBar.add(viewTitle);

    var lblTitle = Ti.UI.createLabel({
        color: '#fff',
        font: {
            fontSize: 14,
            fontWeight: 'bold'
        },
        height: Ti.UI.SIZE,
        left: 10,
        text: obj.title,
        width: Ti.UI.SIZE
    });
    viewTitle.add(lblTitle);

    var lblValue = Ti.UI.createLabel({
        color: '#000',
        font: {
            fontSize: 18,
            fontWeight: 'bold'
        },
        height: Ti.UI.SIZE,
        right: 10,
        text: obj.value,
        width: Ti.UI.SIZE
    });
    viewBar.add(lblValue);
    return viewBar;
}

var scrollView = Ti.UI.createScrollView({
    contentHeight: 'auto',
    contentWidth: 'auto',
    height: Ti.UI.FILL,
    layout: 'vertical',
    showHorizontalScrollIndicator: true,
    showVerticalScrollIndicator: true,
    width: Ti.UI.FILL
});
win.add(scrollView);

    • Just for clarity are you asking if I can tweak it for to include the separator bar and gradient?

      The heavy weight references can be explained like this; using a web view is like including a full web browser inside your app (which is fine if you have to) but it requires lots of memory as it is capable and ready to do so much more, whereas a few views can be considered closer to a handful of divs within a web page from a memory perspective, plus the os has built in cache and redrawing routines for standard views.

      There is nothing wrong with including a web view, I do in my apps when appropriate, but only if you are going to show web page information, otherwise a few good views with layout or tableviews are more efficient and easier to maintain.

      function addBar(obj) {
    var viewBar = Ti.UI.createView({
        backgroundColor: obj.backgroundColor,
        height: 70,
        width: Ti.UI.FILL
    });

    var viewTitle = Ti.UI.createView({
        backgroundColor: obj.barColor,
        height: Ti.UI.FILL,
        left: 0,
        width: obj.percentage + '%'
    });
    viewBar.add(viewTitle);
    var viewGradient = Ti.UI.createView({
        backgroundColor: 'transparent',
        backgroundGradient: {
            colors: [
                { color: obj.barColor, offset: 0.0 },
                { color: obj.backgroundColor, offset: 1.0 }
            ],
            endPoint: { x: '100%', y: '0%' },
            startPoint: { x: '0%', y: '0%' },
            type: 'linear'
        },
        height: Ti.UI.FILL,
        right: 0,
        width: 20
    });
    viewTitle.add(viewGradient);

    var lblPercent = Ti.UI.createLabel({
        color: '#fff',
        font: {
            fontSize: 22,
            fontWeight: 'bold'
        },
        height: Ti.UI.SIZE,
        left: 15,
        text: obj.percentage + '%',
        width: Ti.UI.SIZE
    });
    viewBar.add(lblPercent);

    var lblTitle = Ti.UI.createLabel({
        color: '#000',
        font: {
            fontSize: 18,
            fontWeight: 'bold'
        },
        height: Ti.UI.SIZE,
        text: obj.title,
        textAlign: 'center',
        width: Ti.UI.FILL
    });
    viewBar.add(lblTitle);

    var lblValue = Ti.UI.createLabel({
        color: '#fff',
        font: {
            fontSize: 22,
            fontWeight: 'bold'
        },
        height: Ti.UI.SIZE,
        right: 15,
        text: obj.value,
        width: Ti.UI.SIZE
    });
    viewBar.add(lblValue);

    var viewLine = Ti.UI.createView({
        backgroundColor: '#000',
        bottom: 0,
        height: 1,
        width: Ti.UI.FILL
    });
    viewBar.add(viewLine);

    return viewBar;
}
