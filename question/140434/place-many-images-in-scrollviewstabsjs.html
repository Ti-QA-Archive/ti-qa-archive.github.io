    You need to set scrollview property correctly to get proper scroll ,just try making scrollview in this way:

     //
var scrollView = Titanium.UI.createScrollView({ 
    top:10,
    height:"25%",
    width:"90%",
    borderRadius:10,
   contentWidth: 'auto',
    contentHeight: 'auto',
    showVerticalScrollIndicator:true,
    showHorizontalScrollIndicator: true });
