    I think I didn't fulling understand your question, but if you want to add a html to a table, you can simply add a webview to a row of the table.

    Example : 

    //-- OUR HTML
var htmltekst = '<h3>this is html</h3>';
var webview = Titanium.UI.createWebView({
    html: htmltekst
});

//-- THE ROW WE WANT TO ADD THE WEBVIEW
var myRow = Titanium.UI.createTableViewRow({
    height: 100,
});
myRow.add(webview);

var table = Ti.UI.createTableView({
    style: Ti.UI.iPhone.TableViewStyle.GROUPED,
    data: [{
        title: "Well look at this"
    }, {
        title: "TweetDeck is cool"
    },
    myRow
    ]
});

table.addEventListener('click', function(e){
    alert('clicked');
});
win.add(table);

    Also labels have to ability of parsing html with the attribute html but it's android only.

    Hope it helps.
