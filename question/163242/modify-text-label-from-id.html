Modify text label from id

Hi, I'm new to programming with titanium.
I have a problem with changing the text of some label dynamically created!
i have this code:

for (var c = 0; c < 10; c++) {
    var view = Ti.UI.createView({
        top:10 + (90*c),
        borderRadius:10,
        backgroundColor: 'white',
        width:300,
        height:80,
        id:'view_' + c
    });

    var label = Ti.UI.createLabel({
        top:5,
        left:10,
        text:'Hello Word'
        id:'label_' + c
    });

    var btn = Ti.UI.createButton({            
        top: 30,
        left:210,
        width: 27,
        height: 27,
        id:'btn_' + c

    });

    view.add(label);
    view.add(btn);
    windows.add(view);

}

now how do I change the text of the view with id "view_3" with the "btn_3"?

if I use this code:

    btn.addEventListener('click',function (e){        
        view.text ='OK';
    });

is changed only the last label

    You can use this solution:

    var buttonsData = [];
var LabelData = [];

for (var c = 0; c < 10; c++) {
    var view = Ti.UI.createView({
        top:10 + (90*c),
        borderRadius:10,
        backgroundColor: 'white',
        width:300,
        height:80,
        id:'view_' + c
    });

    var label = Ti.UI.createLabel({
        top:5,
        left:10,
        text:'Hello Word'
        id: c
    });
   LabelData.push(label);
    var btn = Ti.UI.createButton({          
        top: 30,
        left:210,
        width: 27,
        height: 27,
        id: c

    });
   buttonsData.push(btn);
    view.add(label);
    view.add(btn);
    windows.add(view);

}


btn.addEventListener('click',function (e){      
        var id = e.source.id;
    LabelData[id].text = 'OK';
    });
