function setSoundSource(e){
switch (e.source.id){

        case "E":
            player.setUrl('audio/E.wav');
        break;
        case "A":
            player.setUrl('audio/A.wav');
        break;
        case "D":
            player.setUrl('audio/D.wav');
        break;
        case "G":
            player.setUrl('audio/G.wav');
        break;
        case "B":
            player.setUrl('audio/B.wav');
        break;
        case "te":
            player.setUrl('audio/TE.wav');
        break;

    }

}

function ding(e){

    player.volume = 1.0;
    if (player.playing || player.paused){

}
