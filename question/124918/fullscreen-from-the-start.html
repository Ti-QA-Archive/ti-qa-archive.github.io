    navBarHidden:true

    backgroundColor:'#96BD0D',
    fullscreen:true,
    navBarHidden:true

    icon:'images/icon_iputin.png',
    title:'Put __ In __',
    backgroundColor: '#96BD0D',
    window:win1
});

var win2 = Titanium.UI.createWindow({  
    backgroundColor:'#99cc00',
    backgroundImage:'images/bg.png',
    navBarHidden:true
});
var tab2 = Titanium.UI.createTab({  
    icon:'images/icon_settings.png',
    title:'settings',
    backgroundColor: '#99cc00',
    window:win2
});

tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);


// open tab group
tabGroup.open({
    activeTab:tab1
});
;

win1.open({
    navBarHidden:true
