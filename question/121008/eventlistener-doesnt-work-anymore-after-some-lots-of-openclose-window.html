EventListener doesn't work anymore after some lots of open/close window

Hello,

My EventListener stops working in a window after opening/closing it about 10 times.
Here is the main Window called first:

app.js:

Titanium.UI.setBackgroundColor('#000');

var maxWidth=Titanium.Platform.displayCaps.platformWidth;
var maxHeight=Titanium.Platform.displayCaps.platformHeight;
if(maxHeight>maxWidth){
    maxHeight=maxWidth;
    maxWidth=Titanium.Platform.displayCaps.platformHeight;
}

var slideWin=Ti.UI.createWindow({
    url:'slide.js',
    title:'Second',
  backgroundColor:'#fff',
  orientationModes:[Titanium.UI.LANDSCAPE_RIGHT,Titanium.UI.LANDSCAPE_LEFT],
  exitOnClose:true,
  fullscreen:true,
  navBarHidden:true
});

var mainWin=Ti.UI.createWindow({
        title:'First',
      backgroundColor:'#000',
      orientationModes:[Titanium.UI.LANDSCAPE_RIGHT,Titanium.UI.LANDSCAPE_LEFT],
      exitOnClose:true,
      fullscreen:true,
      navBarHidden:true
    });

var btn=Ti.UI.createButton({

    top:100,
    left:100,
    width:200,
    height:20,
    title:'OUUUUUUUUUUUUUUUUUUUUUUUUUUI'

});

btn.addEventListener('click',function(e){

    slideWin.open();

});


mainWin.add(btn);
mainWin.open();

slide.js

// this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');


var btn=Ti.UI.createButton({

    top:100,
    left:100,
    width:200,
    height:20,
    title:'NOOOOOOOOOOOOOOOOOOOOOOOOON'

});

btn.addEventListener('click',function(e){

    Ti.UI.currentWindow.close();

});

Ti.UI.currentWindow.add(btn);

After opening/closing the "slide.js" window, the last eventlistener:
btn.addEventListener('click',function(e){

Ti.UI.currentWindow.close();

});

Doesn't work anymore.

I'm thinking about an invalid way to close the window or call the open ? But adding a "removeEventListener" doesn't seems to avoid the issue.

Thank you for your help

  • Ah, I see (I need more sleep :) ).

    The problem is that "slideWin" has defined "exitOnClose". Comment that out and event will fire and everything will work fine. There could be a problem when back button on your device is clicked to close current window. That's why I usually recreate window every time button is clicked, it seems to solve problems with back button on Android.

    So … copy-paste "slideWin" to "btn" click event, remove "exitOnClose" …

    btn.addEventListener('click',function(e){

    Ti.API.info('Button clicked!');

    var slideWin=Ti.UI.createWindow({
        url:'slide.js',
        title:'Second',
      backgroundColor:'#fff',
      orientationModes: [Titanium.UI.LANDSCAPE_RIGHT,Titanium.UI.LANDSCAPE_LEFT],
      //exitOnClose:true,
      fullscreen:true,
      navBarHidden:true
    });

    slideWin.open();

});

    … and it should work (I use 1.7).
