  • This works for me:

    var currentItent = Titanium.Android.currentActivity.getIntent();

if (currentItent.hasExtra(Ti.Android.EXTRA_STREAM)) {
    var iname = Ti.Android.EXTRA_STREAM;
    var blob = currentItent.getBlobExtra(iname);

    //devicefilename = afbeelding.file.name;

    var tempFile = Titanium.Filesystem.createTempFile();
    tempFile.write(blob);
    imageContents = tempFile.read();
    preview.image = blob;

    var type = currentItent.getType();   // image/png      image/jpeg

    switch (type) {
        case 'image/png':
            devicefilename = 'intentfile.png';
        break;
        case 'image/jpeg':
            devicefilename = 'intentfile.jpg';
        break;
        case 'image/*':
            devicefilename = 'intentfile.jpg';
        break;
    };

}
