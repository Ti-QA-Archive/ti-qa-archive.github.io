TableView data not showing

Hi,

I have some code which is supposed to show 2 rows of data with an image and 2 labels but when I run the code the text and images don't show up. Here's the code:

var tableData = [];

dataArray=[{title:'Watch Dogs', url:'../images/Watch_Dogs_Xbox_case.JPG', genre:'Adventure'},
{title:'Forza', url:'../images/Forza_Xbox_case.JPG', genre:'Racing'}];




for(var i=0;i<dataArray.length;i++){
    var row = Ti.UI.createTableViewRow({
        height:50,
        width:'100%',
        layout:'horizontal'
});

var rowImg = Ti.UI.createImageView({
      image: dataArray[i].url,
      height:140,
      width:140
    });


var txtView = Titanium.UI.createView({
   borderRadius:10,
   layout :'vertical',
   top: 50,
   height: Ti.UI.Fill,
   width: Ti.UI.FILL
});

var title = Titanium.UI.createLabel({
    text: 'Title: '+dataArray[i].title,
    color: 'blue',
    textAlign: Ti.UI.TEXT_ALIGNMENT_LEFT
});

var genre = Titanium.UI.createLabel({
    text: 'Genre: '+dataArray[i].genre,
    color: 'blue',
    textAlign: Ti.UI.TEXT_ALIGNMENT_RIGHT
});

txtView.add(title);
txtView.add(genre);

row.add(rowImg);
row.add(txtView);
tableData.push(row);

}

var newReleasesTable = Titanium.UI.createTableView({ 
data:tableData }); 

newReleases.add(newReleasesTable);
