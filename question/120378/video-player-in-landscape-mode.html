Video Player in Landscape Mode

Hello,

I am trying to add a video player which plays video in landscape mode when a button fires a click, here is the simple code I'm trying with no luck!

button.addEventListener('click', function(e)
{

    win.orientationModes = [Ti.UI.LANDSCAPE_RIGHT];
    var win1 = Ti.UI.createWindow({});

    var activeMovie = Titanium.Media.createVideoPlayer({
        contentURL:'trailer/videolink.mp4',
        backgroundColor:'#000',
        scalingMode:Titanium.Media.VIDEO_SCALING_ASPECT_FIT,
        movieControlMode:Titanium.Media.VIDEO_CONTROL_DEFAULT,
        movieControlStyle:Titanium.Media.VIDEO_CONTROL_DEFAULT
    });

    activeMovie.fullscreen = true;
    activeMovie.play();
    win1.add(activeMovie);

    activeMovie.addEventListener('fullscreen', function(e){
        if (!e.entering){                                     
            activeMovie.stop();
            activeMovie.release();
            //win1.remove(activeMovie);
            win.remove(win1);
            win.orientationModes = [Ti.UI.PORTRAIT];
        }


    });

    win.add(win1);
});

Can someone suggest me any other alternative code to get it working.

Thanks.
