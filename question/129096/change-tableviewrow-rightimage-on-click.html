Change TableViewRow rightImage on Click

I'm trying to change the rightImage of a TableViewRow when the row is clicked. When I click the row, it updates the image source but the image does not refresh. I have to navigate away and back to the page for the image to update.

var familyRow = Ti.UI.createTableViewRow({title: 'Family', rightImage: 'images/green.png'});
var listsTable = Titanium.UI.createTableView();
listsTable.appendRow(familyRow);

listsTable.addEventListener('click', function(e) {
    if (e.source['rightImage'] == 'images/green.png') {
        e.source['rightImage'] = 'images/red.png'
    } else {
            e.source['rightImage'] = 'images/green.png';
    }
});

win.add(listsTable);

Thanks for the help.
