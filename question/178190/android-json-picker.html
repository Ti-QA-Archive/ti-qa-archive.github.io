var url = "http://89.237.149.36/ws/resources/shop/acts";
 var client = Ti.Network.createHTTPClient({
     onload : function(e) {
          var pickerData=this.responseText;
          var datas=pickerData.acts;

     },

 });

 client.send();
}

    • The fillPicker is an asynchronous call so your "picker.add(data)" gets called before your fillPicker has returned any data.

      You need to pass a callback function to fillPicker and call it in the onload function passing the data.

      E.g.

      //...
function cb(data){
picker.add(data);
}
fillPicker(cb);
//...

      then in your onload function:

      function fillPicker(cb) {
        var url = "http://89.237.149.36/ws/resources/shop/acts";
        var client = Ti.Network.createHTTPClient({
            onload : function(e) {
                var pickerData = this.responseText;
                var datas = pickerData.acts;
                alert(datas.length);
                for (var i = 0; i < datas.length; i++) {
                    data[i] = Ti.UI.createPickerRow({
                        title : datas[i].actName,
                        code : datas[i].actCode
                    });
                }
                cb(data)
            },
