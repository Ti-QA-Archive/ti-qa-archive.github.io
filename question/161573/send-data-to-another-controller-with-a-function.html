send data to another controller with a function

Hi everyone.

I send in my controller "data.js" data (as for example) using a function to another controler "projet.js".
The controller "projet.js" correctly receives the data. I tested with an alert.
But my problem is that I can not used the data received in my function to change the text value of the label in the view of controlling "projet.js".

Can you please help me? I do not know what bug.
Thank you very much.

data.js:

function showDetail(infos){
    var myView = Ti.UI.createView({
        width:'100%',
        height:'283dp',
        bottom:0,
        top:'10dp',
        layout:'vertical'

    });

        myView.add(Ti.UI.createLabel({
            width:'100%',
            height:'25dp',
            color: "#003058",
            font: {
                fontSize: '18dp',
                fontFamily: 'futuralt-book',
            },
            textAlign:'center',
            text:infos.titre
        }));
    myView.add(Ti.UI.createImageView({
        width:'100%',
        height:'auto',
        top:0,
        image:'/img/media/'+infos.prefix+'/1.jpg',
    }));

    myView.addEventListener('click',function(e)
    {    var projet = Alloy.createController('projet').getView();
        Alloy.createController('projet').createdPage(infos.titre);
        projet.open();
            projet.animate({left:0});
    });

    return myView;
};
exports.showDetail = showDetail;

projet.js:

exports.createdPage = function(titre){
    $.titre.text = titre; 
};

projet.xml:

<Alloy>
    <Window class="container">
        <Label id="titre"/>
    </Window>
</Alloy>

    var projet = Alloy.createController('projet').getView();
Alloy.createController('projet').createdPage(infos.titre);
projet.open();

    You are creating two controllers here. I think this should be:

    var projet = Alloy.createController('projet');
projet.createdPage(infos.titre);
projet.getView().open();
