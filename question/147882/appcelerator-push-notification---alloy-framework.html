  • Hi,
    Below is the code snipped I've used. Registered the app and push notification configurations along with package Id in my.appcelerator.com under development and the given user has been created.

    Please check.

    Index.xml:
    <Alloy>
    <Window title="Cloud Sample">
    <Button id="connectCloud" title="Connect Cloud" left="5" top="5" onClick="checkCloudFunc" />
    </Window>
    </Alloy>

    index.js
    var CloudPush = require('ti.cloudpush');
    var Cloud = require('ti.cloud');

    function checkCloudFunc(e)
    {
    CloudPush.retrieveDeviceToken({
    success: function deviceTokenSuccess(e) {
    deviceToken = e.deviceToken;
    loginDefault();
    },
    error: function deviceTokenError(e) {
    alert('Failed to register for push! ' + e.error);
    }
    });
    }

    function loginDefault(e)
    {
    //Create a Default User in Cloud Console, and login
    Cloud.Users.login({
    login: 'test123',
    password: 'test123'
    }, function (e) {
    if (e.success) {
    defaultSubscribe();
    } else {
    alert('Error: ' +((e.error && e.message) || JSON.stringify(e)));
    }
    });
    }

    function defaultSubscribe(e)
    {
    Cloud.PushNotifications.subscribe({
    channel: 'alert',
    device_token: deviceToken,
    type: 'android'
    }, function (e){
    if (e.success) {
    alert('Subscribed for Push Notification!');
    }else{
    alert('Error:' +((e.error && e.message) || JSON.stringify(e)));
    }
    });
    }
