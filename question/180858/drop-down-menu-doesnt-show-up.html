drop down menu doesn't show up

this the following code, I receive a variable that based on it I create dropdown menus. So variable args.session could be 1-10.

But when I click on the label the dropdown menu doesn't show up.
In the console I see the following:

WARN] :   New layout set while view [object TiUIView] animating: Will relayout after animation.
[WARN] :   New layout set while view [object TiUIView] animating: Will relayout after animation.
[WARN] :   New layout set while view [object TiUIView] animating: Will relayout after animation.
[WARN] :   New layout set while view [object TiUIView] animating: Will relayout after animation.
[WARN] :   New layout set while view [object TiUIView] animating: Will relayout after animation.
[WARN] :   New layout set while view [object TiUIView] animating: Will relayout after animation.

here is my code

Ti.UI.backgroundColor = '#000';
var win = Ti.UI.createWindow();

var args = arguments[0] || {};
Ti.API.info("args ==> "+JSON.stringify(args));

var array = [];
var pickerViewArray = [];
var tableView = Ti.UI.createTableView({data:array, style:Ti.UI.iPhone.TableViewStyle.GROUPED});

for(var i=0; i< args.sessions; i++){

    // here I create 1..10 labels
    var session = Ti.UI.createTableViewRow({height:46, className:i, title: i+"."});
    var value = Ti.UI.createLabel({color:'#3D4460', text:"", font:{fontSize:17, fontWeight:'normal'}, top:11, left:150, height:20, width:180, textAlign:'right'});        

    session.add(value);
    array.push(session);    
    tableView.setData(array);

    var picker = Ti.UI.createPicker({top:0, selectionIndicator: true});    
    var pickerView = Ti.UI.createView({height:248,bottom:-248});
    var pickerHours = [];

    // setting the picker values
    for(var j=0; j<24; j++){
        pickerHours.push(Ti.UI.createPickerRow({title:"{j}"}));
    }

    picker.add(pickerHours);
    pickerView.add(picker);

    picker.addEventListener('change',function(e){
        value.text = picker.getSelectedRow(0).title;
        tableView.setData(array);
    });    

    pickerViewArray.push(pickerView);    
    win.add(pickerView);
}

// animations
var slideIn =  Ti.UI.createAnimation({bottom:-43});
var slideOut =  Ti.UI.createAnimation({bottom:-251});

tableView.addEventListener('click', function(eventObject){

    //sliding out all labels pickerView        
    for(var i=0; i < pickerViewArray.length; i++){        
        pickerViewArray[i].animate(slideOut);
    }

    //sliding in the selected label's pickerView    
    pickerViewArray[eventObject.rowData.className].animate(slideIn);

});

win.add(tableView);
win.open();

    ok I took the win.add(pickerView);

    out of the longer loop and added this:

    for(var i=0; i &lt; pickerViewArray.length; i++){ win.add(pickerViewArray[i]); }

    before win.open()
