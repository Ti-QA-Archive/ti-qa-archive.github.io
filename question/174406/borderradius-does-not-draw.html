BorderRadius does not draw

Hi Guys,

With this code below, borderRadius: 15, does not have the drawing as it should. Anyone know why ?

function buildTableRow(inParam) {'use strict';
    var row = Ti.UI.createTableViewRow({                // Rows of Items
    borderRadius: 15,
    title:  inParam.title,
    height:  viewheight,
    indentionLevel:  inParam.level,
    backgroundGradient: {
        type:  'linear',
        colors:  [{
            color:  '#414141',
            offset:  0.0
        }, {
            color:  '#B2B2B2',
            offset:  0.12
        }, {
            color:  '#CCCCCC',
            offset:  0.85
        }]
    },
    color:  '#000000',
    font: {
        fontSize:  14,
        fontWeight:  'normal'
    },
    LEVEL:  inParam.level,
    EXPANDED:  (inParam.level < 2)  ?  true  :  false,
    CHILDEXPANDED:  (inParam.level < 1)  ?  true  :  false,
    CHILD:  inParam.child
});
