  • hi,

    try this

    for (var i = 0; i < 10; i++) {
    var star = Ti.UI.createImageView({
        height : 24,
        width : 24,
        left : 5 + (i * 25),
        backgroundColor : 'white',
        isChecked : false
    });

    star.addEventListener('click', function(e) {
        if (e.source.isChecked == false) {
            e.source.backgroundColor = 'red';
            e.source.isChecked = true;
        } else {
            e.source.backgroundColor = 'white';
            e.source.isChecked = false;
        }
    });

    win.add(star);
}

    you can set image instead of backgroundColor
