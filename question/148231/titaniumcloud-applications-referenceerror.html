    Hi selvam, you are making a small mistake.
    You have required the cloud with variable name cloud , but you are using Cloud with capital C to access cloud functions. Please verify that. just change 

    var cloud = require('ti.cloud');

    with 

    var Cloud = require('ti.cloud');

    I think that is your problem.. Check and tell me if it helped.

function CloudView() {
    var Cloud = require('ti.cloud');
    //var cloud = require('ti.cloud');

    var win = Titanium.UI.createWindow({
        title : 'Video Recording from Appcelerator Titanium',

    });

    var submit = Ti.UI.createButton({
        color : '#000000',
        title : 'Click me ',
        top : 30,

            first_name : 'selvam',
            last_name : 'as',
            password : 'selvam',
            password_confirmation : 'selvam'

                Ti.API.log("success");

                alert('error');
                Ti.API.log("error");
                alert('Error:\\n' + ((e.error && e.message) || JSON.stringify(e)));
            }
        });
    });
    return win;
}
