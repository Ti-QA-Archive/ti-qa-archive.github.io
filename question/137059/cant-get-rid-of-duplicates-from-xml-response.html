  • I finally figured it out. Hope someone else can find use in it too:

    var data = [];

var client = Titanium.Network.createHTTPClient();
client.onload = function() { 
//alert(this.responseText);

function checkForVal(arr, val) { 
    for (var i=0;i<arr.length;i++) { 
        if (arr[i] == val) { 
            return true; 
        }
    }
}

var x = 0;
var xml = Ti.XML.parseString(this.getResponseText());

var interiorColor = xml.documentElement.getElementsByTagName('VehicleDescription');
var ic = interiorColor.item(0).getElementsByTagName('interiorColor');
//var interiorName = ic.item(i).getAttribute('colorName');
var intCol = [];

    for (var i=0;i<ic.length;i++) {
        var intName = ic.item(i).getAttribute('colorName');
        if (!checkForVal(intCol,intName)) {
            intCol.push(intName);
        }    
    }

        for (color in intCol) {
            Ti.API.info(intCol[color]);

            var column = Ti.UI.createPickerColumn();

                var interiorRow = Ti.UI.createPickerRow({
                    title:intCol[color]
                });
                column.addRow(interiorRow);

            data[x++] = interiorRow;
        }    

    var picker = Ti.UI.createPicker();

    picker.add(data);

    picker.selectionIndicator = true;

    win.add(picker);

        picker.addEventListener('change', function(e){
            Ti.App.fireEvent('changeInterior',{
                text:e.row.title
            });
        });
};
client.onerror = function()
{ 
alert("insideonerror");
};
