  • When you create a controller with Alloy, you can pass options:

    // index.js
var newWindow = Alloy.createController('newWindow', { data: [ ... ] }; // create with options
newWindow.open();

// newWindow.js
var data = arguments[0].data || []; // expect arguments[0].data to be array
$.someLabel.setText(data.join());
