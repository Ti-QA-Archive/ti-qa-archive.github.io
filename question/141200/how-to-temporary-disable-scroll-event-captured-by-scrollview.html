    • For this code, you will find that if you use your finger slide to left/right on the the scrollable view, you will find this view is very difficult to change the image, you need to slide it exactly horizontal to change the image.

      Titanium.UI.setBackgroundColor('#000');

var scrollHeight = 200;
var index_win = Titanium.UI.createWindow({  
    id:'index_win',
    navBarHidden:true,
    backgroundColor:'#dceffe',
    exitOnClose: true,
    modal:true
});
var index_tab = Titanium.UI.createTab({  
    title:'title1',
    window:index_win
});
//top recommend scrolling bar        
var topScrollContainerGroup=Titanium.UI.createScrollableView({
                        top:'0dp',
                        height : scrollHeight+'dp',
                        backgroundColor:'#FFF',
                        contentHeight: 'auto',
                        contentWidth: 'auto',
                        showVerticalScrollIndicator:false,
                        showHorizontalScrollIndicator:true,
                        zIndex:100
});    

//**************************************
//the scroll bar information
//**************************************
var json = {data:[
            {'event_name' : 'name1'},
            {'event_name' : 'name2'},
            {'event_name' : 'name3'},
            ]};
var data = json.data;

for(var i=0;i<json.data.length;i++){
    var topScrollContainer=Titanium.UI.createView({
                    top:'0dp',
                    height : scrollHeight+'dp',
                    backgroundColor:'#FFF'
    });    

    var spotEvent = Titanium.UI.createImageView({
        left:0,
        top:0,
        zIndex:1,
        backgroundColor:'#000',
        width:'100%',
        height:scrollHeight+'dp',
        defaultImage:'/images/default/laser-event.jpg',
        //url:"http://www.colourbox.com/preview/4364611-606462-tick-tack-toe-on-a-napkin.jpg"

    });

    var spottitleView=Titanium.UI.createView({
                    height : '40dp',
                            zIndex:10,
                    top:'160dp',
                    opacity:0.6,
                    backgroundColor:'#000',
                    touchEnabled : false
    })

    var spotEventTitle = Ti.UI.createLabel({
                    text: json.data[i].event_name,
                    font : {
                        fontSize : '16sp',
                        fontWeight : 'bold'
                    },
                    height : '20dp',
                    zIndex:10,
                    top:'160dp',
                    color : '#FFF',
                            touchEnabled : false
            });
    topScrollContainer.add(spotEvent);
    topScrollContainer.add(spottitleView);    
    topScrollContainer.add(spotEventTitle);        
    topScrollContainerGroup.addView(topScrollContainer);
    }

//*********End of scrollbar info****************************************************
//**********************************************************************************

//bottom table menu    
var indexSubMenu = Titanium.UI.createTableView({
    backgroundColor:'#ffffff',
    top:scrollHeight+'dp',
    bottom:'40dp'
});
indexSubMenu.setData(data);


var globalScrollContainer=Titanium.UI.createScrollView({
    top:'50dp',
    contentHeight: '1000dp',
    contentWidth: 'auto',
    showVerticalScrollIndicator:false,
    showHorizontalScrollIndicator:false,
    backgroundColor:'#dceffe',
    touchEnabled:false,
    scrollType:'vertical',
})
globalScrollContainer.addEventListener("scroll",function(e){
    Ti.API.info("scroll is called... X:" + e.x + ", Y:" +e.y);
});

globalScrollContainer.add(topScrollContainerGroup);
globalScrollContainer.add(indexSubMenu);

index_win.add(globalScrollContainer);
index_win.open();
