  • Hi

    No, you can't. But you can do:

    var myWin = Titanium.UI.createWindow({
    fullscreen :false,
    navBarHidden:true

});

var myNavbar = Ti.UI.createLabel({
    height:30,
    backgroundColor:'#000',
    color:'#fff',
    text:'My custom title',
    top:0
});

myWin.add(myNavbar);
myWin.open();
