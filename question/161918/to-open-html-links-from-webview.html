    var win=Ti.UI.createWindow({
    layout:'vertical',
    backgroundColor:'white'
});

var data = [
    {title:'Facebook',linkUrl:'https://www.facebook.com/' },
    {title:'Linkedin',linkUrl:'https://www.linkedin.com/'},
    {title:'Titanium',linkUrl:'http://www.appcelerator.com/' },
    {title:'Google',linkUrl:'http://www.google.com/'}
];
var aTableView = Ti.UI.createTableView({
    data:data,
    layout:'vertical',
    height:'52%',
    left:'7%',
    width:'85%',
});

aTableView.addEventListener('click', function(e) {
    winWebView.url=e.rowData.linkUrl;
    alert(e.rowData.linkUrl);
});

win.add(aTableView);

var winWebView=Ti.UI.createWebView({
        height:'45%',
        left:'7%',
        width:'85%',
        top:'1%'
    });
    win.add(winWebView);
win.open();

  • Hi

    All emoticons are simply unicode values representing a special area of the character table.

    For arrows you could use this guide - althought not all unicode characters are supported and you must be using a suitable font - the default font is usually fine.

    • Unicode arrows

    You simply place a \u before the hex value shown on the page.so right arrow would be \u2192 (althought I have not tried that specific code to make sure it is supported - you may need one of the other arrows.

    In a string you simply add it as any normal text and it gets picked up automatically.

    var lbl = Ti.UI.createLabel({
    text: 'Turn \u2192'
});
