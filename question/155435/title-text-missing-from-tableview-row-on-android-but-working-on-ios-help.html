Title text missing from tableview row on android but working on iOS, help?

Hey, title says it all really is there anything i'm missing that may prevent title text showing on android.

var left_menu_items = [{
    title:'Home',
        url: site_url,
        image: 'images/icons/home.png'
}];

function getTableViewRow(data){
    var row = Ti.UI.createTableViewRow({
    backgroundColor:'transparent',
    selectedBackgroundColor:'#54b5e9',
    height:47,
    color: '#FFFFFFF',
    font:{
        fontFamily:'Helvetica Neue',fontSize:16
    },
    title:data.title,
    url:data.url,
    leftImage:data.image
});

return row;
}

var rows = [];
for (var x = 0;x<left_menu_items.length;x++){
    rows.push(getTableViewRow(left_menu_items[x]));
}

var tableView = Ti.UI.createTableView({
    backgroundColor: '#1f2226',
    separatorColor: '#15171a',
    data: left_menu_items,
    top: 45,
    zIndex: 14
});

tableView.setData(rows);

    • As an aside, why don't you set the rows directly to the table instead of setting first left_menu_items then again rows?

      There are 2 drawbacks here:

      • you populate the table 2 times, this will slow down the app pretty much
      • setters are slower than direct access so

      table.data = my_rows is faster than table.setData(my_rows)

      just my 2c ;)
