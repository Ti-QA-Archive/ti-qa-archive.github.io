Parse.com setting up a pointer: Installation to _User

hello,
Does anyone have an example of setting up a Pointer between a _User and an Installation (in Parse.com?). Here's my attempt and all i get is error: <!DOCTYPE html>.
(user is field name for the Pointer in Installation class and I masked the class names, although that was probably pointless)

var request = Titanium.Network.createHTTPClient(
{
        onload: function(e) 
        {    
        var res = this.responseText;
        Ti.API.info('result: ' + res);
        },
        onerror: function(e) 
        {
              Ti.API.info("error: " + this.responseText);
        }
      });
      var params = 
      {
            "user": 
            {
                  "__type": "Pointer",
                  "className": "_User",
                  "objectId": "zzzzzz"
            }
         };

      // Register device token with Parse
      request.open('POST', 'https://api.parse.com/1/installations/yyyyyyy');
      request.setRequestHeader('X-Parse-Application-Id', ParseAppID);
      request.setRequestHeader('X-Parse-REST-API-Key', ParseRestAPI);
      request.setRequestHeader('Content-Type', 'application/json');
      request.send(JSON.stringify(params));
