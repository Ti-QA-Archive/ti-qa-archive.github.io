  • How about something like this? (given values for audio.title and audio.url)

    
                var audioHTML = "<!DOCTYPE html><html><head><title>" + audio.title + "</title></head>";
                audioHTML = audioHTML + "<body>";
                audioHTML = audioHTML + "<audio controls=\"controls\"><source src=\"" + audio.url + "\" type=\"audio/mp3\" /></audio>";
                audioHTML = audioHTML + "</body></html>";

                var audioWebView = Titanium.UI.createWebView({
                    html:audioHTML,
                    top:20,
                    left:10,
                    width:200,
                    height:100
                });
                win.add(audioWebView);
