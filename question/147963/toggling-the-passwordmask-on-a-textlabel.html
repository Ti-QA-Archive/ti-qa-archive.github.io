Toggling the passwordmask on a textlabel

I am trying to create a show password/hide password using a switch. 

    var passwordTextField = Ti.UI.createTextField({
        borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
          color: '#336699',
          width: '80%',
          top: 15,
          hintText: 'password',
          passwordMask: true
    });

    var showPWtext = Ti.UI.createLabel({
        text: 'Show Password: ',
        width: '50%',
        textAlign: Ti.UI.TEXT_ALIGNMENT_RIGHT
    })
    var showPWbox = Ti.UI.createSwitch({
        title:"Show Password",
        value:false,
        textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER

    });

    var showPWview = Ti.UI.createView({
        width: '100%',
        height: Ti.UI.SIZE,
        layout: 'horizontal'
    });

    showPWview.add(showPWtext);
    showPWview.add(showPWbox);

    // add it to the window, etc...

Hitting the toggle button the first time exposes the text, but hitting it a second time does not restore the mask. Is my logic flawed or is this just not possible? Has anyone used a different successful method?

    hi,

    your code just works fine but anyway try the below code

    var self = Ti.UI.createWindow({    
    backgroundColor : 'white',    
});

var txtPassword = Ti.UI.createTextField({
    top : 5,
    width : 270,
    height : 40,
    hintText : 'Password',
    passwordMask : true,
    borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
});

self.add(txtPassword);

var showPWbox = Ti.UI.createSwitch({
    top:50,
    title : "Show Password",
    value : false,
    textAlign : Ti.UI.TEXT_ALIGNMENT_CENTER

});
showPWbox.addEventListener('change',function(e){

    if(e.value){
        txtPassword.passwordMask = false;
    }else{
        txtPassword.passwordMask = true;
    }
});

self.add(showPWbox);

self.open();
