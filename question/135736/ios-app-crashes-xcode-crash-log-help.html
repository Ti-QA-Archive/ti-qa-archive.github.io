  • Sure here is the code. Its basically a run tracker application and the bg service just continues to track the user via a timer and gps hits and then inserts data into db. Also on a side note i recently read that I need to have the 'location' key in my plist file for background services even though I never have and yet it still works.. Guessing thats not a requirement?

    Ti.include('/includes/runtrackerlibrary.js', '/includes/datalayer.js');
// --------------------------------------------------------------------------------------------------------------------------------
// Copyright, 2012, Libertas Technologies, LLC. All Rights Reserved.
// Last mods: 2/10/2012 - aroyce@libertastechnologies.com - www.libertastechnologies.com
// --------------------------------------------------------------------------------------------------------------------------------
//coord points are now stored in db, not passed.

try {
    Ti.Geolocation.preferredProvider = Titanium.Geolocation.PROVIDER_GPS;
    Ti.Geolocation.accuracy = Titanium.Geolocation.ACCURACY_NEAREST_TEN_METERS;
    Ti.Geolocation.distanceFilter = 0;

    if(Ti.App.Properties.getInt('TrackStatus', 0) == 2 && Ti.App.Properties.getBool('Paused', false) == false) {
        var intBGMileAVG = Ti.App.Properties.getInt('intBGMileAVG', 1);
        var TimerID = 0;
        var TimerCount = Ti.App.Properties.getInt('TimerCount', 0);
        var oldTimerCount = -1;
        var Points = [];
        var RunID = Ti.App.Properties.getInt('RunID', 0);
        var TotalDistanceRan = Ti.App.Properties.getDouble('TDR', 0);
        var milesArray = [];
        var startDateTime = Ti.App.Properties.getString('startDateTime', new Date());
        var oldDiffTime = Ti.App.Properties.getString('oldDiffTime', new Date());

        //setup timer
        TimerID = setInterval(function() {
            var diffSeconds = (new Date() - new Date(startDateTime)) / 1000;
            diffSeconds = parseInt(diffSeconds.toString());
            TimerCount = diffSeconds;
            Ti.App.Properties.setInt('TimerCount', diffSeconds);

            // Ti.App.Properties.setInt('TimerCount', TimerCount);
            // //if(TimerCount % 300 == 0) {
            // //AnnounceRunStatus(TotalDistanceRan, TimerCount);
            // //}
        }, 1000);
        //close in resume function
        //dbOpen('StERun');

        //GPS Handler
        var i = 0;
        var locationCallback = function(e) {
            try {

                if(!e.success) {
                    //alert('error in BG service: ' + e.error);
                    //if lose signal alert user and figure out what to do
                    //alert('gps error in bg:' + e.error.message);
                    return;
                }
                //every 2 seconds
                if(TimerCount - oldTimerCount >= 4) {
                    //if(TimerCount != oldTimerCount) {
                    i++;
                    Ti.App.Properties.setInt('gpshitcount', i);
                    SQLInsertFromBGService('insert into tblRunRecords (runid,lat,lon,altitude,altitudeaccuracy,speed,heading,createdon) values (' + RunID + ',\'' + e.coords.latitude + '\',\'' + e.coords.longitude + '\',' + e.coords.altitude + ',' + e.coords.altitudeAccuracy + ',' + e.coords.speed + ',\'' + e.coords.heading + '\',\'' + new Date() + '\')');
                    //SQLExecute('insert into tblRunRecords (runid,lat,lon,altitude,altitudeaccuracy,speed,heading,createdon) values (' + RunID + ',\'' + e.coords.latitude + '\',\'' + e.coords.longitude + '\',' + e.coords.altitude + ',' + e.coords.altitudeAccuracy + ',' + e.coords.speed + ',\'' + e.coords.heading + '\',\'' + new Date() + '\')',false);
                    oldTimerCount = TimerCount;
                }

                if(Points.length == 0) {
                    //insert twice on first time so distHaversine formula below doesnt break
                    Points.push({
                        latitude : e.coords.latitude,
                        longitude : e.coords.longitude
                    });
                }

                Points.push({
                    latitude : e.coords.latitude,
                    longitude : e.coords.longitude
                });

                //distance
                var length = Points.length;
                var distance = LatLon.distHaversine(Points[length - 2].latitude, Points[length - 2].longitude, Points[length - 1].latitude, Points[length - 1].longitude);
                TotalDistanceRan = TotalDistanceRan + distance;
                Ti.App.Properties.setDouble('TDR', TotalDistanceRan);

                if(convertMilesToKilometers(TotalDistanceRan) >= intBGMileAVG) {

                    var timedif = timeDifference(new Date(),new Date(oldDiffTime));
                    milesArray.push({
                        'time' : timedif + ' Min/' + Ti.App.Properties.getString('DistanceUnits', 'Miles'),
                        'distance' : intBGMileAVG
                    });
                    Ti.App.Properties.setList('milesArray', milesArray);
                    oldDiffTime = new Date();
                    Ti.App.Properties.setString('oldDiffTime', oldDiffTime.toString());

                    intBGMileAVG++;
                    Ti.App.Properties.setInt('intBGMileAVG', intBGMileAVG);
                }
            } catch(ex) {
                //alert('inner bg error: ' + ex);
            }
        }
        Titanium.Geolocation.addEventListener('location', locationCallback);

        Ti.App.currentService.addEventListener('stop', function(e) {
            clearInterval(TimerID);
            Titanium.Geolocation.removeEventListener('location', locationCallback);
        });
    }

} catch(ex) {
    //alert('outer bg error:' + ex);
}
