  • Josh,
    Thanks for getting this started. I cleaned it up and corrected a couple of typos. Hope this helps others, too. If anyone finds an error, just fix it and post it here. :D

    
// Create state picker
var statePicker = Titanium.UI.createPicker(
  {
  selectionIndicator:true    
  });

//INITIALIZE THE ARRAY
var data = [
'AL','AK','AZ','AR','CA','CO','CT','DE','FL','GA',
'HI','ID','IL','IN','IA','KS','KY','LA','ME','MD',
'MA','MI','MS','MN','MO','MT','NE','NV','NH','NJ',
'NM','NY','NC','ND','OH','OK','OR','PA','RI','SC',
'SD','TN','TX','UT','VT','VA','WA','WV','WI','WY'
];

//ASSIGN VALUES TO PICKER ROWS
for (var i = 0, j = data.length; i < j; i++){  
  statePicker.add(Titanium.UI.createPickerRow({title:data[i]}));
}
