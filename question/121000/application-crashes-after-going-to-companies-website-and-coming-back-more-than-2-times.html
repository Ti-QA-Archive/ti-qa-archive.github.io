    The issue is that in your back butten event listener you just keep recreating the aboutus.js window

    back_button.addEventListener('click',function(){

 var window = Ti.UI.createWindow({
            url:"AboutUs.js",

        });
    Titanium.UI.currentTab.open(window,{animated:false});
   var emptyView = Titanium.UI.createView({});
window.leftNavButton = emptyView
  window.title='About Us';
});

    All you need to do is:

    back_button.addEventListener('click',function(){
    win.leftNavButton = null;
    webview.url = 'about_us.html';
});
