Event Listeners for items(imageview,button) in scrollable view

Application type:mobile,titanium sdk:3.1.3,platform:android 2.3.3,device:android emulator

How to have event listeners for items in scrollable view.

var scroll=Ti.UI.createScrollableView({
    currentPage:1,
    showPagingControl:true,
    touchEnabled:true
});
win.add(scroll);
var view=Ti.UI.createView({backgroundColor:'#fff'});
scroll.addView(view);
var nextb=Ti.UI.createButton({
    top:Titanium.Platform.displayCaps.platformHeight*0.8,
    left:Titanium.Platform.displayCaps.platformWidth*0.8,
    title:'Next'
});
view.add(nextb);
//event listener for button
 var img_map = Ti.UI.createImageView({
        backgroundColor: '#bdbdbd',
        borderRadius: parseInt(size / 2, 10),
        height: size,
        left:point.x,
        top:20+point.y,
        width: size
    });
    view.add(img_map);
//event listener for imageview.
