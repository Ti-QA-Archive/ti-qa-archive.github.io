  • Hi Sangana,
    If you want to add the images to the scroll view dynamically and scroll horizontally. Then please look this example

    var window =Ti.UI.createWindow({
height:Ti.UI.FILL,
width:Ti.UI.FILL,
backgroundColor:'white'
});

var scrollView=Ti.UI.createScrollView({
height:50,
width:Ti.UI.FILL,
layout:'horizontal',
scrollType:'horizontal',
top:30,
});
window.add(scrollView);

var imagesArray=['img1','img2','img3','img4','img4','img4','img5'];//Path of the images

for(var i=0;i<imagesArray.length;i++){
var image=Ti.UI.createImageView({
height:50,
width:50,
left:20,
image:imagesArray[i]
});

scrollView.add(image);
}

window.open();

    Please feel free to ask and mark the question if your query is solved to help others.
