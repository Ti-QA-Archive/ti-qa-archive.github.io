  • var win = Ti.UI.createWindow({
    height : 480,
    width : 320,
    backgroundColor : 'white',
    })
    var table = Ti.UI.createTableView({
    data : Ti.Contacts.getAllPeople(),
    height : 480,
    width : 320
    });
    win.addEventListener('focus', function(e) {
    Mobile();
    });

    win.add(table);
    function Mobile() {
    //var mobile1 = JSON.stringify(Titanium.Contacts.getAllPeople()[0].getPhone().mobile[0])
    var length = JSON.stringify(Titanium.Contacts.getAllPeople().length)
    Ti.API.info(JSON.stringify(Titanium.Contacts.getAllPeople()[1].getPhone().mobile));
    //Ti.API.info(JSON.stringify(Titanium.Contacts.getAllPeople().length));
    Ti.API.info(length);
    var data = []
    if (length == 0) {
    alert('no contacts saved')
    } else {
    for (var i = 0; i < length; i++) {
    if ((JSON.stringify(Titanium.Contacts.getAllPeople()[i].getPhone().mobile)) ==null) {
    Ti.API.info('error');
    } else {
    var row = Ti.UI.createTableViewRow({
    height : 50,
    width : 320,
    });
    var mNumber = Ti.UI.createLabel({
    text : JSON.stringify(Titanium.Contacts.getAllPeople()[i].getPhone().mobile[0].toString()),
    height : 40,
    top : 5
    });
    row.add(mNumber);
    data.push(row);
    Ti.API.info(Titanium.Contacts.getAllPeople()[i].getFullName() + ' ' + JSON.stringify(Titanium.Contacts.getAllPeople()[i].getPhone().mobile[0].toString()));
    }

        }
}

data.push(row);
table.setData(data);

//Titanium.Contacts.getAllPeople()[0].getPhone({phone});

    }

    win.open();
