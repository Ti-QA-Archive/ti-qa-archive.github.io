Simple app crashes only on device

app.js:

var startwin = Ti.UI.createWindow({
    backgroundImage: 'iphone/default.png',
    width:320,
    height:480
});
startwin.open();

startwin.addEventListener('click', function(e) {
    var xpos = parseInt(e.x/80,10);
    switch (xpos) {
        case 0:
            var win = Ti.UI.createWindow({
                url:'planer.js'
            });
            win.open();
        break;
    }
});

This simple app runs in simulator, but not on device.

I cannot see my error. The app also crashes without the addEventListener-block.

Any ideas? Is it possible to build an app without tabgroup
