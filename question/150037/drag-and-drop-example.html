    Hi nazir,

    I just provide sample code : 

    var view = Ti.UI.createView({
    width : 100,
    height : 100,
    backgroundColor : 'green'
});
var olt = Titanium.UI.create3DMatrix(), curX, curY;

view.addEventListener('touchstart', function(e) {
    curX = e.x;
    curY = e.y;
});
view.addEventListener('touchmove', function(e) {
    var deltaX = e.x - curX, deltaY = e.y - curY;
    olt = olt.translate(deltaX, deltaY, 0);
    view.animate({
        transform : olt,
        duration : 100
    });
});

win.add(view);

  • hi,

    try this one…

    var win = Titanium.UI.createWindow();

var circle = Titanium.UI.createView({
    height : 100,
    width : 100,
    borderRadius : 50,
    backgroundColor : '#336699',
    // top : 10
});

circle.addEventListener('touchmove', function(e) {
    Ti.API.debug('Our event tells us the center is ' + e.x + ', ' + e.y);
    var newX = e.x + circle.animatedCenter.x - circle.width / 2;
    var newY = e.y + circle.animatedCenter.y - circle.height / 2;
    circle.animate({
        center : {
            x : newX,
            y : newY
        },
        duration : 1
    });
});

win.add(circle);
win.open();
