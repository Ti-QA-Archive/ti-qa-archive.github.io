Can't access variable?

Hello there. I am working with my little app and i smashed into a problem whitch i cannot solve. This is my code:

    var Insect = m.extend({
        name : 'Insect',
        size : Size.medium,
        imagePath : '/m/insects/insect.png',
        speed : 2,
        scoreValue : (this.size * Size.factor),
        sprite : false, //This will return a sprite after this.init
        timer : false, //This will return a timer after this.startFlying

        init : function() {

            this.scoreValue = this.size * Size.factor;
            this.initSprite();
            window.add(this.sprite);
        },


        initSprite : function() { 
            this.sprite = Titanium.UI.createImageView({
                left : 20,
                top : 150,
                size : {
                    width : 'auto',
                    height : 'auto'
                },
                image : this.imagePath
            });
            return true;
        },

        startFlying : function() {
            this.timer = setTimeout(this.flyingStep,20);
        },

        flyingStep : function() {
            alert(this.sprite.left);
            this.timer = setTimeout(this.flyingStep,20);
        },

        stopFlying : function() {
            clearTimeout(this.timer);
        },

    })  

    var Bee = Insect.extend({
        name : 'Bee',
        size : Size.small,
        imagePath : '/m/insects/bee.gif',
        speed : 1,
        init : function() {
            this._super();
        },
    })

    var newBee = new Bee();
    newBee.startFlying();

And this is the error I will get while executing the code:

[ERROR] While executing Timer, received script error. 'Result of expression 'this.sprite' [undefined] is not an object. at app_beewhacker.js (line 52)'

Line 52 is this line:
alert(this.sprite.left);
in the startFlying function.

The m.extend is extending another object, so that I can extend it to the Insect. Don't worry about that. Thanks :)

  • I think Insect.init() is not beeing executed.

    Two questions:

    • Don't you need to call Bee.init() before Bee.startFlying()?

    • When you do this._super() inside Bee declaration, are you sure is executing parent init() ?

    One thing you can do is check if sprite exists and, if not, create before start flying:

    startFlying : function() {
     if(!this.sprite) {
         this.init(); //or initSprite(), depends on you
     }
     this.timer = setTimeout(this.flyingStep,20);
},
