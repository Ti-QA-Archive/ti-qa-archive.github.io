  • This is how a list view looks like.
    I found a way to animate list items on iOS, but still looking forward to how do this on Android =(

    
//THE TEMPLATE
    var tp = {
        properties: {
            accessoryType: Ti.UI.LIST_ACCESSORY_TYPE_NONE
        },
        childTemplates: [
            {
                type: 'Ti.UI.Label',
                bindId: '_title',
                properties: {
                    color : 'black',
                    left  : '15dp',
                }
            }
        ]
    }
        var item = {
            _title : {
                text : 'test',
                color : 'black',
                left : '15dp'
            }, 
            properties: {
                accessoryType: Ti.UI.LIST_ACCESSORY_TYPE_DISCLOSURE,
                height   : '60dp',
                backgroundColor : '#696969',
                click    : ''
            },
            template : tp
        };

    //THE SECTION
    var listSection = Titanium.UI.createListSection({
        items: [item],
        headerTitle: "Section A"
    });

    var listView = Titanium.UI.createListView({
        sections: [listSection],
        headerTitle: "Shopping List"
    });

    win.add(listView);

    On iOS you can do like this:

    var an = Ti.UI.createAnimation({duration:500});
item._title.animation = an;

    But this doesn't work for Android…
