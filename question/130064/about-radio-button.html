  • if you have 2 or more radio button then keep first as checked and other as a unchecked,

    var Radio_joke = Ti.UI.createView({
    top:60,
    left:25,
    height:25,
    width:25,
    backgroundImage:'../images/checked.png'
    });
    win_setting.add(Radio_joke);

    var Radio_quotes = Ti.UI.createView({
    top:95,
    left:25,
    height:25,
    width:25,
    backgroundImage:'../images/unchecked.png'
    });
    win_setting.add(Radio_quotes);
