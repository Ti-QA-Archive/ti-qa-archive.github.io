  • this code worked for me I've grouped some cods togther now i am able to use custom listview + pull to refresh , you can modify this code as your needs

    `var template = {
    childTemplates : [{
    type : 'Ti.UI.ImageView',
    bindId : 'avatar',
    properties : {
    width : 100,
    height : 100,
    left : 0,
    image : "/1.jpg",
    backgroundColor : "#000000"
    }
    },{
    type : 'Ti.UI.Label',
    bindId : 'label_utente',
    properties : {
    width : 452,
    height : 30,
    left : 10,
    right : 50,
    minimumFontSize : 8,
    color : "white"
    }
    }, {
    type : 'Ti.UI.View',
    bindId : 'check_utente',
    properties : {
    width : 30,
    height : 30,
    right : 10,
    backgroundImage : "/1.jpg"
    }
    }],
    properties : {
    //height : '150dp',
    height : Ti.UI.SIZE
    }
    };

        var header_utenti = Ti.UI.createView({
        backgroundImage : "/images/img_new/blur_bianco2.png",
        height : 24,
        width : 512
    });

    var label_utenti = Ti.UI.createLabel({
        text : "-----",
        left : 10,
        color : "#424242",
        font : {
            fontSize : 14
        }
    });

    header_utenti.add(label_utenti);

    var control = Ti.UI.createRefreshControl({
    tintColor:'red'
    });

        var listView_utenti = Ti.UI.createListView({
        width : Ti.UI.FILL,
        top : 70,
        height : Ti.UI.FILL,
        refreshControl:control,
        left : 0,
        backgroundColor : "transparent",
        templates : {
            'default' : template
        },
        defaultItemTemplate : 'default',
        separatorStyle: Titanium.UI.iPhone.TableViewSeparatorStyle.NONE,
        separatorInsets : {
            left : 0,
            right : 0,
        }
    });

    var sezione_utenti = Ti.UI.createListSection({
    headerView : header_utenti
    });

    control.addEventListener('refreshstart',function(e){
    Ti.API.info('refreshstart');
    setTimeout(function(){
    Ti.API.debug('Timeout');
    sezione_utenti.appendItems(genData());
    control.endRefreshing();
    }, 2000);
    });

        var data_utenti = [];
    var json  = {"user" : [{"name":"jack"},{"name":"jacksss"}] };

    function genData(){

        for (var i = 0; i < json.user.length; i++) {
        data_utenti.push({
            label_utente : {
                text : json.user[i].name
            },
            check_utente : {
                backgroundImage : "/1.jpg"
            },
            properties : {
                backgroundColor : "transparent",
                selectionStyle : Ti.UI.iPhone.ListViewCellSelectionStyle.NONE,
                id : json.user[i].id
            }
        });
    }

return data_utenti;

    }

        for (var i = 0; i < json.user.length; i++) {
        data_utenti.push({
            label_utente : {
                text : json.user[i].name
            },
            check_utente : {
                backgroundImage : "/1.jpg"
            },
            properties : {
                backgroundColor : "transparent",
                selectionStyle : Ti.UI.iPhone.ListViewCellSelectionStyle.NONE,
                id : json.user[i].id
            }
        });
    }
    sezione_utenti.setItems(data_utenti);

    listView_utenti.sections = [sezione_utenti];

    $.index.add(listView_utenti);`
