Hi guys,

I've made a for loop for creating 8 tumbnail images for video's. edn.ui.createSmallImageTumbnail is a selfmade function for creating the tumbnails, the first Ti.API.info returns the idWithParseShizzle, but in the clickedTumb function it won't return the idWithParseShizzle. 

for(var i = 0; i < 8; i++) {
    var image2 = edn.ui.createSmallImageTumbnail("images/tumbs/"+ i +".png");
    Ti.API.info(image2.idWithParseShizzle);

    image2.left = i * 120;

    image2.addEventListener("click", clickedTumb);

    function clickedTumb(e) {

        var view = e.source;

        Ti.API.info(view.idWithParseShizzle);

    }

    scrollViewTumbs.add(image2);
}

Is there somebody who could tell me what I'm doing wrong?
