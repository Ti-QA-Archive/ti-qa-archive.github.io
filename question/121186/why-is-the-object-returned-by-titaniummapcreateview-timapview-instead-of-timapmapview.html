  • Does this work for you? (ie: not using the addAnnotations method)

    
var game_map2 = Titanium.Map.createView({
    mapType: Titanium.Map.SATELLITE_TYPE,
    region: {
        latitude:42.359432, longitude:-71.090941, latitudeDelta:0.005, longitudeDelta:0.005
    },
    animate:true,
    regionFit:true,
    userLocation:true,
    annotations:[
        {
            latitude: 42.359083,
            longitude: -71.094804,
            title: "Player1",
            pincolor: Titanium.Map.ANNOTATION_GREEN,
            animate: true
        },
        {
            latitude: 42.362318,
            longitude: -71.090448,
            title: agent_two.name,
            subtitle: 'Koch Institute',
            pincolor: Titanium.Map.ANNOTATION_PURPLE,
            animate: true
        },
        {
            latitude:42.360526,
            longitude: -71.088753,
            title: agent_one.name,
            subtitle:'East Campus',
            pincolor:Titanium.Map.ANNOTATION_RED,
            animate:true
        }
    ]
});
win.add(game_map2);
