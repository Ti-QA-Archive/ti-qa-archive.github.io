    var win = Ti.UI.createWindow({
    backgroundColor : "white"
});

  var dialog = Ti.UI.createAlertDialog({
    title: 'Enter text',
    style: Ti.UI.iPhone.AlertDialogStyle.PLAIN_TEXT_INPUT,
    buttonNames: ['OK']
  });

  dialog.addEventListener('click', function(e){
    Ti.API.info('initial value = ' + e.text);
e.text = e.preDefineValue;    
Ti.API.info('changed value = ' + e.text);
  });

  dialog.show();

dialog.fireEvent('click', {preDefineValue:"Alex"} );

win.open();

    I know this could be a laughing at code :) , but i don't see any other way to do what you are trying to do.

    If this does not work then i would suggest you to create a custom alert dialog in which you can do anything…..

                </View>
            </View>
        </View>
    </Window>
</Alloy>
