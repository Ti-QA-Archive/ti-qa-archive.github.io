    Hi,
    I tested this code in iOS simulator and it works for me:

    
var win1 = Titanium.UI.createWindow({  
    backgroundColor:'#ccc'
});

var scrollMenuView = Ti.UI.createScrollView({
  top: 160,
  height: '250',
  width: '80%',
  layout: 'vertical',
  showHorizontalScrollIndicator: true,
  scrollType:'vertical',
  showVerticalScrollIndicator: true,
});

 win1.add(scrollMenuView);
for (var i=0;i<30;i++)
{ 
var menuButton = Titanium.UI.createButton({
       width: 100,
       height: 100,
       borderRadius: 10,
       backgroundColor: '#FFFFFF',
     title:'hello',
    });

scrollMenuView.add(menuButton);
}

win1.open();

  • Hello Korcan ,

    if you want to add scrollView after all button/view add in scrollview.

    try this :

    var win1 = Titanium.UI.createWindow({  
    backgroundColor:'#ccc'
});

var scrollMenuView = Ti.UI.createScrollView({
  top: 160,
  height: '250',
  width: '80%',
  layout: 'vertical',
  showHorizontalScrollIndicator: true,
  scrollType:'vertical',
  showVerticalScrollIndicator: true,
});

for (var i=0;i<30;i++)
{ 
var menuButton = Titanium.UI.createButton({
       width: 100,
       height: 100,
       borderRadius: 10,
       backgroundColor: '#FFFFFF',
     title:'hello',
    });


win1.add(scrollMenuView);   // add here after all button add
scrollMenuView.add(menuButton);
}

win1.open();

    i hope this helps. thanks.
