  • Hi Francesco Pallottini

    You can use tableView i think it,s more easy than listView and you can add something easily like this code just write somethings in textfield and press enter it will add bellow.

    var win = Ti.UI.createWindow({
    backgroundColor : '#fff',
    layout : 'vertical'

});

// Create a TextField.
var aTextField = Ti.UI.createTextField({
    height : Titanium.UI.SIZE,
    top : 10,

    width : Titanium.UI.FILL,
    hintText : 'This is hint text',
    softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_DEFAULT_ON_FOCUS, // Android only
    keyboardType : Ti.UI.KEYBOARD_DEFAULT,
    returnKeyType : Ti.UI.RETURNKEY_DEFAULT,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});

var tableView = Ti.UI.createTableView({
    top : 10,

});

// Listen for return events.
aTextField.addEventListener('return', function(e) {
    tableView.appendRow({
        title : aTextField.value,
        backgroundColor : 'transparent',
        color : "red"
    });
    aTextField.blur();
    aTextField.setValue('');
});


win.add(aTextField);
win.add(tableView);

win.open();
