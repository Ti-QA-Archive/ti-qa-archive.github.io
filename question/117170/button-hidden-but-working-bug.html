Button hidden but working bug?

I'm using a navigationGroup with two windows to show up on top of a tabGroup. In the first navgroup window I have a "CLOSE"-button which closes the navgroup.

When first opening the navgroup the button appears as it should but moving to the next window and going BACK to first window again the button is not longer there. Although, I can click on it, and the correct event is fired.

It seems like there is a stack issue here. Can I somehow not add the first window to the stack again when hitting back from the second window, or somehow, rerender the "CLOSE"-button?

var closeBtn = Titanium.UI.createButton({
    title: 'Close'
});
closeBtn.addEventListener('click', function() {    
    main.close({ animated: true });
});
var first = Titanium.UI.createWindow({
    leftNavButton: closeBtn
});
var label = Ti.UI.createLabel({text:"poke me to open the next window"});
first.add(label);

var navGroup = Ti.UI.iPhone.createNavigationGroup({
       window:first
});
var second = Ti.UI.createWindow({
  backgroundColor:"#fff",
  title:"Second Window"
});
var label2 = Ti.UI.createLabel({text:"Here's the child"});
second.add(label2);

//When the label on the first window receives a touch, open the second
label.addEventListener("click", function(e) {
  navGroup.open(second);
});

//This is the main window of the application
var main = Ti.UI.createWindow();
main.add(navGroup);
main.open();
