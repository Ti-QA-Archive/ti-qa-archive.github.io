  • ok, here is my own answer … (again !)

    x.html 

    <!-- all your import here -->
<body>
<canvas id="myLine" width="500'" height="'400'">[No canvas support]</canvas>
<script>
function myClick (e, shape){Titanium.App.fireEvent("click_chart","{d: shape}"}
window.onload = function (){
// Your RGraph here like (var myLine = new RGraph.Line ...)
}
</script>
</body>

    filewhereyoucreateyourwebview.js

    //Somewhere in your file
Ti.App.addEventListener("click_chart", function(e){
    alert("row clicked was: " + JSON.stringify(e));
});
