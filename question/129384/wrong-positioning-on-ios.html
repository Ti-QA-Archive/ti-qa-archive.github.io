  • Here is some code for one of the windows where the position is messed up on iOS but looks great on Android

    Everything basically appears like it wouldn't have been positioned at all…

    function infoWindow(){

    var win = Ti.UI.createWindow({
        backgroundColor: '#000',
        navBarHidden: true
    });
    win.orientationModes=[Titanium.UI.PORTRAIT];
    win.add(Ti.UI.createImageView({
        url: 'images/logo.jpg',
        width: '60%',
        top:0
    }));
    var scrollView = Ti.UI.createScrollView({
        width: '90%',
        height: '26%',
        top: '40%'
    });
    scrollView.add(Ti.UI.createLabel({
        text: 'Some text here',
        showVerticalScrollIndicator: true
    }));
    var sponsorer = Ti.UI.createImageView({
        url: 'images/sponsorer.jpg',
        width: '90%',
        bottom:'2%'
    });
    var sponsorerLabel = Ti.UI.createLabel({
        text: 'Våra samarbetspartners:',
        font: {fontSize: '22px'},
        color: '#FFF',
        bottom: '25%'
    });
    win.add(sponsorerLabel);
    win.add(scrollView);
    win.add(sponsorer);
    return win;
}
