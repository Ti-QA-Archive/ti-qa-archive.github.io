Add additional data to tableviewrow

I'm creating a tableview with some rows in them, and I have a "title" and "index" for each of the rows. What I want now is to click a row and show a new screen with all the details for that one row, like "title" or "username" and "email"

Can I attach "email" data to a tableviewrow, just the way you can add a title to a row? That way I can easily know which email address is linked to a row that has been clicked in the tableview.

If there is a better way to accomplish this, I'd love to hear it of course! Thanks for helping out.

    Yah It's possible. Try this sample code:

    var win = Ti.UI.createWindow();
var table = Ti.UI.createTableView();

win.add(table);
win.open();

// Set Data to table
var rows = [];

for ( var i = 0; i < 10; i++ ) {
    rows.push( Ti.UI.createTableViewRow({
        title: 'Row ' + i,
        email: 'email' + i + '@abc.com',
        whatever: 'Whatever ' + i 
    }) );    
}

table.setData(rows);

// Click Event
table.addEventListener('click', function(e){
    var row = e.row;
    alert(row.title + ' - ' + row.email + ' - ' + row.whatever);
});
