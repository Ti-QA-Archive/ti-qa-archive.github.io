    • Opacity of view, goes from 0.0 (transparent) to 1.0 (opaque), I'd suggest setting 'duration' property as needed, i.e. sample below:

      var win = Ti.UI.createWindow({backgroundColor: 'white'});

var view = Ti.UI.createView({
    width: 100,
    height: 100,
    backgroundColor: 'red',
    opacity: 0
});

win.addEventListener('click', function(e)
{
    Ti.API.info('Should be animating now.');

    view.animate({
        opacity: 0.6,
        duration: 6000
    }, function(){
        view.setOpacity(1);
        Ti.API.info('After opacity (should be 1): ' + view.opacity);
    });
});

win.add(view);
win.open();
