  • Hi Urvashi,

    Try this code :
    i am just put sample code of custom dialog.

    var win = Ti.UI.createWindow({
    backgroundColor : 'black',
    fullscreen : true,
    width : Ti.Platform.displayCaps.platformWidth,
    height : Ti.Platform.displayCaps.platformHeight
});

win.open()

var customDialog = Ti.UI.createView({
    height : 150,
    width : 300,
    backgroundColor : "#0c7598",
    borderRadius : 5
})
win.add(customDialog);

var titleLabel = Ti.UI.createLabel({
    top : 10,
    text : "ALERT",
    color : "#FFF",
    font : {
        fontSize : 24,
        fontWeight : "bold"
    }
})
customDialog.add(titleLabel);

var cancelBtn = Ti.UI.createButton({
    bottom : 5,
    left : 10,
    width : 135,
    height : 40,
    title : "Cancel",
    color : "#000"
})
customDialog.add(cancelBtn);

cancelBtn.addEventListener('click',function(){
    customDialog.hide()
})

var saveBtn = Ti.UI.createButton({
    bottom : 5,
    right : 10,
    width : 135,
    height : 40,
    title : "Save",
    color : "#000"
})
customDialog.add(saveBtn);

    Thanks SP

  • function createAlert(_args) {
    //283x170
    var alert = Ti.UI.createView({
    width:283,
    height:170,
    visible:false,
    backgroundImage:'/images/alert.png'
    });

    var label = Ti.UI.createLabel({
    text:'This is a custom alert box.\n\nAre you sure that you really want to do that?',
    width:263,
    height:100,
    top:10,
    textAlign:'center',
    color:'#fff',
    font:{
        fontWeight:'bold',
        fontSize:16
    }
});
alert.add(label);

var cancel = Ti.UI.createButton({
    width:127,
    height:42,
    bottom:10,
    left:10,
    title:'Wait a tick ...',
    backgroundImage:'/images/cancel.png'
});
cancel.addEventListener('click', function(e) {
    alert.hide();
});
alert.add(cancel);

var ok = Ti.UI.createButton({
    width:127,
    height:42,
    bottom:10,
    right:10,
    title:'Lets do it!',
    backgroundImage:'/images/ok.png'
});
ok.addEventListener('click', function(e) {
    alert.hide();
});
alert.add(ok);

return alert;

    }
