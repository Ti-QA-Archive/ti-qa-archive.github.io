  • Hi Shraddha,

    I guess you mean the Action bar of the android. You can customize it with your appicon and title. Not a big deal. Follow this code-

    
var win = Ti.UI.createWindow({
    navBarHidden: false
});

var actionBar;

win.addEventListener("open", function() {
    if (Ti.Platform.osname === "android") {
        if (! win.activity) {
            Ti.API.error("Can't access action bar on a lightweight window.");
        } else {
            actionBar = win.activity.actionBar;
            if (actionBar) {
                actionBar.icon = "/your_app_icon.png";
                actionBar.title = "Your App Title";
                   };
            }


        }

});

win.open();
