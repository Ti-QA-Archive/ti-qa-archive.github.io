  • removing the width value like this works for me

        var sv = Ti.UI.createScrollView({
        left: 10,
        top: 10,
        bottom: 10,
        backgroundColor: '#550000',
        contentWidth: 'auto',
        contentHeight: 1200,
        showVerticalScrollIndicator:true,
        showHorizontalScrollIndicator:false,
        zIndex: 104
    });
