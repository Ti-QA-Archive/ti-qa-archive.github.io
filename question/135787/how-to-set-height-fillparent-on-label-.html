  • Suppose you have the following:

    var myView = Ti.UI.createView({
    backgroundColor:"red",
    width:300,
    height:100
})

//set your label height to match your myView's height like that.

var l2 = Titanium.UI.createLabel({ 
    text:'label', 
    height:myView.height
 });

    Good luck!
