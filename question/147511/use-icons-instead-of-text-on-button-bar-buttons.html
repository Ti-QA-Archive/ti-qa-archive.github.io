Use icons instead of text on button bar buttons

I have a button bar with two labels "Prev" and "Next" I want to use two pictures instead of words.

View:

 <ButtonBar id="bb1" platform="ios" backgroundColor="grey" height="25" width="100" right="5dp" >
                    <Labels>
                        <Label id="btnPrev">Prev</Label>
                        <Label id="btnNext">Next</Label>
                    </Labels>
            </ButtonBar>

tss

"#btnPrev[formFactor=tablet platform=ios]":{
    style: Ti.UI.iPhone.SystemButtonStyle.PLAIN,
    image: '/images/arrow_down.png',
},
"#btnPrev[formFactor=handheld platform=ios]":{
    style: Ti.UI.iPhone.SystemButtonStyle.PLAIN,
    image: '/images/arrow_down.png',
},
"#btnNext[formFactor=tablet platform=ios]":{
    style: Ti.UI.iPhone.SystemButtonStyle.PLAIN,
    image: '/images/arrow_up.png',
},
"#btnNext[formFactor=handheld platform=ios]":{
    style: Ti.UI.iPhone.SystemButtonStyle.PLAIN,
    image: '/images/arrow_up.png',
},

controller

$.bb1.addEventListener('click', function(e) {
    //Ti.API.info('Total: ' + totalIndexLength);
    if(e.index == 0) { // prev
        if(clickedIndex > 0) {
            clickedIndex--;
        }    
    }
    else { // next
        if(clickedIndex < totalIndexLength -1) {
            clickedIndex++;    
        }
    }
    //Ti.API.info('current: ' + clickedIndex);
    var commData             = communications[clickedIndex];
    var communicationDetailInfo = getCommunicatonDetails(commData.id);
    displayCommunicationDetails(communicationDetailInfo);
});

http://docs.appcelerator.com/titanium/3.0/#!/api/Titanium.UI.ButtonBar
