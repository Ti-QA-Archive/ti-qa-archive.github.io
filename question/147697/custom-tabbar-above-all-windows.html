    Hi

    This might help you out. I have created a simple proof of concept app for you where you can use the normal tabGroup but "fake" your own tabs with your own styles and even differences in heights and width of the tabs you create.

    You can decorate this simple example any way you wish, add images, add different states when selected and all the time using the basic tabGroup mechanism to manage your windows so you do not have to.

    To test this code - create a new default project and replace the contents of the app.js file with the code below.

    function getData(intChoice) {
    var info = [];
    info.push(['One', 'Two', 'Three', 'Four', 'Five', 'Six', 'Seven', 'Eight', 'Nine', 'Ten', 'Eleven', 'Twelve', 'Thirteen', 'Fourteen', 'Fifteen']);
    info.push(['Red', 'Green', 'Blue', 'Yellow', 'Orange', 'Brown', 'Purple', 'Black', 'White', 'Grey']);
    info.push(['Andrew', 'Brian', 'Charles', 'David', 'Edward', 'Frank', 'Gary', 'Henry', 'Ian', 'James']);

    var data = [], intItem = 0, intItems = info[intChoice].length;

    for (intItem = 0; intItem < intItems; intItem = intItem + 1) {
        data.push({
            title: info[intChoice][intItem]
        });
    }
    return data;
}

var win1 = Ti.UI.createWindow({
    backgroundColor: '#fff',
    title: 'Window 1'
});
var tab1 = Ti.UI.createTab({  
    icon: 'KS_nav_ui.png',
    title: 'Tab 1',
    window: win1
});
var tbl1 = Ti.UI.createTableView({
    data: getData(0),
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
win1.add(tbl1);

var win2 = Ti.UI.createWindow({
    backgroundColor: '#fff',
    title: 'Window 2'
});
var tab2 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 2',
    window: win2
});
var tbl2 = Ti.UI.createTableView({
    data: getData(1),
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
win2.add(tbl2);

var win3 = Ti.UI.createWindow({
    backgroundColor: '#fff',
    title: 'Window 3'
});
var tab3 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 3',
    window: win3
});
var tbl3 = Ti.UI.createTableView({
    data: getData(2),
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
win3.add(tbl3);


var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);
tabGroup.addTab(tab2);
tabGroup.addTab(tab3);

var fakeTabs = Ti.UI.createView({
    backgroundColor: 'transparent',
    bottom: 0,
    height: 60,
    layout: 'horizontal',
    left: 0,
    width: Ti.UI.FILL
});
tabGroup.add(fakeTabs);

var fakeTab1 = Ti.UI.createView({
    backgroundColor: 'red',
    bottom: 0,
    height: 50,
    left: 0,
    width: 80
});
fakeTab1.addEventListener('click', function (e) {
    tabGroup.setActiveTab(0);
});
fakeTabs.add(fakeTab1);

var fakeTab2 = Ti.UI.createView({
    backgroundColor: 'green',
    bottom: 0,
    height: 60,
    left: 0,
    width: 160
});
fakeTab2.addEventListener('click', function (e) {
    tabGroup.setActiveTab(1);
});
fakeTabs.add(fakeTab2);

var fakeTab3 = Ti.UI.createView({
    backgroundColor: 'blue',
    bottom: 0,
    height: 50,
    left: 0,
    width: 80
});
fakeTab3.addEventListener('click', function (e) {
    tabGroup.setActiveTab(2);
});
fakeTabs.add(fakeTab3);

tabGroup.open();

    This uses the little known ability to add ANY view over the top of the tabGroup which itself sits over the windows it manages.

    Have a play I think you will see the potential of this solution.
