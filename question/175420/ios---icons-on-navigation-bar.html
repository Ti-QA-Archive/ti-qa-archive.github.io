  • Hi Jiale He,

    Thanks for your concern.

    You can't add multiple button in navigation bar. You can use custom view to add multiple icons in navigation bar.

    Please check my test code below.

    var win2 = Titanium.UI.createWindow({
    backgroundColor: 'white',
});

var win1 = Titanium.UI.iOS.createNavigationWindow({
    backgroundColor:"#496986",
   window: win2
});

var leftView = Ti.UI.createView({
    top: 0,
    left: 0,
    height: Ti.UI.SIZE,
    width:'50%',
});

var rightView = Ti.UI.createView({
    top: 0,
    right: 0,
    height: Ti.UI.SIZE,
    width:'50%',
});

var rb1 = Ti.UI.createButton({ 
    right:10,
    height:30,
        width:40,
        backgroundImage: "3.jpg" //just change this property
});
rightView.add(rb1);

var rb2 = Ti.UI.createButton({ 
    right: 70,
    height:30,
        width:40,
        backgroundImage: "2.jpg" //just change this property
});
rightView.add(rb2);

var rb3 = Ti.UI.createButton({ 
    right: 130,
    height:30,
        width:40,
       backgroundImage: "1.jpg" //just change this property
});
rightView.add(rb3);

var lb = Ti.UI.createButton({ 
    left: 10,
    height:30,
        width:40,
        backgroundImage: "0.jpg" //just change this property
});
leftView.add(lb);


win2.setLeftNavButton(leftView);
win2.setRightNavButton(rightView);

win1.open();

    Hopefully it'll help you.

    Thanks
