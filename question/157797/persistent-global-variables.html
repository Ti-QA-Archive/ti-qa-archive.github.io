  • You got it the wrong way. You should use getString to read the value, like this:

    
var local;
    pickerLocal.addEventListener('change',function(e) {
        Ti.API.info("change "+e.selectedValue);
        Ti.App.Properties.setString("branche", e.selectedValue);
        local = e.selectedValue;
    });

button.addEventListener('click',function() {
    var branche = Ti.App.Properties.getString("branche");
})
