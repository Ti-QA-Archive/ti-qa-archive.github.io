WEIRD - Buttons inside toolbar doesn't fire click event when using oop

I have this code that is inside a method of a class

self.createToolbar = function() {
var flexSpace = Titanium.UI.createButton({
    systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

var retakeButton = Ti.UI.createButton({
    title: "Retake",
    left: 5,
    style:Titanium.UI.iPhone.SystemButtonStyle.BORDERED
});

var useButton = Ti.UI.createButton({
    title: "Use",
    right: 5,
    style:Titanium.UI.iPhone.SystemButtonStyle.BORDERED
});

retakeButton.addEventListener('click', function() {
  alert("button called.");
});
useButton.addEventListener('click', function() {
  alert("another button called");
});

var toolbar = Titanium.UI.createToolbar({
    items:[retakeButton, flexSpace, useButton],
    bottom:0,
    borderTop: true,
    borderBottom: false,
    translucent: true,
    barColor: '#999'
});    
win.add(toolbar);
}

This renders the toolbar but when buttons inside it is clicked, click event is not fired.

When moving the code at the very bottom of my js file outside the self.createToolbar method, this seems to work well.

I've tested this on 1.7.0 and 1.7.1 continous build.

  • Put your buttons outside of the method. Make them global. It helped me.

    button1,
    button2,
    button3,

    var toolbar = Titanium.UI.createToolbar({
    zIndex:1,
    items:[button1,button2,button3],
    padding:{paddingTop:0},
    bottom:0,
    left:0,
    borderTop:true,
    borderBottom:false,
    horizontalGap:0,
    translucent: true,
    height:35

});

    function addToolbar(){

    Titanium.UI.currentWindow.add(toolbar);

    }
