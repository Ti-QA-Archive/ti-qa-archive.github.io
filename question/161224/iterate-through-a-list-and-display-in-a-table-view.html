  • Simple answer. you have to iterate through the second loop.

    var list = {"name":"John","items":{"123":"apples","456":"bananas"},
"name":"Jake","items":{"789":"apples"},
"name":"Mike","items":{"890":"apples","901":"bananas","012":"cherries"}
};

for (var i in list){
    alert(list.name[i] + " - " + items[i] ???);  //using alert just to see the results

    var output = list[i].name;
    if (list[i].items) {
        for (var b = 0, b < list[i].items.length, b++) {
           output = output + " - " + list[i].items[b];   
        }
    }
    //will put result in a tableview here
}

    Have fun
