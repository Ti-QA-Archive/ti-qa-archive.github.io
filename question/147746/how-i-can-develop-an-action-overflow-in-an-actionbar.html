  • I have developed a possible solution. Here is the code:

    var subMenu, 
    win = Ti.UI.createWindow({
        fullscreen : false,
        color : 'white',
        backgroundColor : 'gray',
        title : 'Collapse'
    });

    win.activity.onCreateOptionsMenu = function(e) {
        var menu = e.menu, 
        collapseActionView = Ti.UI.createView({
            width : 100,
            height : 20,
            backgroundColor : 'gray'
        }), 
        helpItem = menu.add({
            icon : "images/action_about.png",
            showAsAction : Ti.Android.SHOW_AS_ACTION_ALWAYS,
            order : 1
        }), 
        overflowItem = menu.add({
            icon : "images/ic_action_overflow.png",
            actionView: collapseActionView,
            showAsAction : Ti.Android.SHOW_AS_ACTION_ALWAYS | Ti.Android.SHOW_AS_ACTION_COLLAPSE_ACTION_VIEW,
            order : 2
        });

        collapseActionView.addEventListener('click', function() {
            overflowItem.expandedActionView();
        });

        overflowItem.on('expand', function() {
            Ti.API.info(' -------------> Expanded menu item!');
            subMenu = Titanium.UI.createView({
                        backgroundColor : 'black',
                        color:'white',
                        width : 100,
                        height :  200,
                        right : 0,
                        top : 0
            });
            var tableData = [ {title: 'Submenu 1'}, {title: 'Submenu 2'}, {title: 'Submenu 3'}, {title: 'Submenu 4'} ];

            var table = Ti.UI.createTableView({
              data: tableData
            });
            subMenu.add(table);
            win.add(subMenu);
        });

        overflowItem.on('collapse', function() {
            Ti.API.info(' -------------> Collapsed menu item!');
            win.remove(subMenu);
        });

        overflowItem.on('click', function() {
            Ti.API.info(' -------------> Click menu item!');

        });

        helpItem.addEventListener("click", function(e) {
            alert("Help Item Clicked!");
        });

    }

    win.open();

    I hope any suggestions.
