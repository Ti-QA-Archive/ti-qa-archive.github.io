  • Hi Suresh,

    There is no such type of property available like subTitle for tableViewRow. You have to create custom row for that using label like below code.

    while (rows.isValidRow()) {
    var row = Ti.UI.createTableViewRow({
        backgroundColor : 'transparent',
        height : Ti.UI.SIZE
    });

    var lblTitle = Ti.UI.createLabel({
        text : obj.business_name,
        top : 5,
        left : 15,
        width : 250,
        height : 20
    });

    row.add(lblTitle);
    var lblDesc = Ti.UI.createLabel({
        text : "subTitle",
        top : 30,
        left : 15,
        width : 215,
        bottom : 10,
        font : {
            fontSize : 12
        }
    });
    row.add(lblDesc);
    dataArray.push(row);
    rows.next();
}

    Try this code

    var ViewbyCategoryWin = Ti.UI.currentWindow;
//var db = Ti.Database.install('../Bu1.sqlite','Bu1');
// set the data from the database to the array
var db = Ti.Database.install('../Bu1.sqlite', 'Bu1');

function setArray() {

    var rows = db.execute('SELECT DISTINCT ECategory FROM Bu1');
    var rows1 = db.execute('SELECT DISTINCT EItem FROM Bu1');
    var prodCat = Ti.UI.currentWindow.prodCat;

    // create the array
    var dataArray = [];

    while (rows.isValidRow()) {
        var row = Ti.UI.createTableViewRow({
            backgroundColor : 'transparent',
            height : Ti.UI.SIZE,
            leftImage : "",
            hasChild : true,
            path : '../History/ViewByCategory1.js'
        });

        var lblTitle = Ti.UI.createLabel({
            text : rows.fieldByName('ECategory'),
            top : 5,
            left : 45,
            width : 250,
            height : 20
        });

        row.add(lblTitle);
        var lblDesc = Ti.UI.createLabel({
            text : rows.fieldByName('EItem'),
            top : 30,
            left : 45,
            width : 215,
            bottom : 10,
            font : {
                fontSize : 12
            }
        });
        row.add(lblDesc);
        dataArray.push(row);
        rows.next();
    };

    // set the array to the tableView
    tableview.setData(dataArray);
};
// create table view
var tableview = Ti.UI.createTableView({
});

    if (e.rowData.path) {

            title : e.rowData.title
        });
        var prodCat = e.rowData.title;
        win.prodCat = prodCat;
        Ti.UI.currentTab.open(win);
    }
});
ViewbyCategoryWin.add(tableview);
