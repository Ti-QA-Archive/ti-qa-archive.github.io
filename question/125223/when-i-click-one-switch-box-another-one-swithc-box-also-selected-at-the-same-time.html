  • This code creates a table and fills it with some data.
    Corresponding switches are created too.

    A click on a switch will fill a log label with the row's title and will turn off the other switches. Only one switch can be turned on at any time.

    Tell me if it helps.

    var win = Ti.UI.createWindow
({
    title: 'Playground',
    backgroundColor: '#fff'
});

var table = Ti.UI.createTableView();

var log = Titanium.UI.createLabel
({
    height: 'auto',
    color: '#000',
    text: 'no selection',
    textAlign: 'center',
    bottom: 0
});

win.add(table);

win.add(log);

var source_data =
[
    {title: 'google'},
    {title: 'yahoo'},
    {title: 'lycos'},
    {title: 'msn'}
];

var table_data = [];

var current_checkbox = '';

for( var i = 0 ; i < source_data.length ; i++ )
{
    var source = source_data[i];

    var row = Ti.UI.createTableViewRow();

    var title = Titanium.UI.createLabel
    ({
        text: source.title,
        font:
        {
            fontSize: 16,
            fontWeight: 'bold'
        },
        width: 'auto',
        textAlign: 'left',
        left: 10
    });

    var checkbox = Ti.UI.createSwitch
    ({
        value: 0,
        right: 10,
        data:
        {
            title: source.title
        }
    });

    checkbox.addEventListener
    (
        'change', function(e)
        {
            if( e.value )
            {
                if(current_checkbox) current_checkbox.value = 0;

                current_checkbox = e.source;

                log.text = e.source.data.title;
            }
            else
            {
                if( current_checkbox && current_checkbox == e.source )
                {
                    e.source.value = 0;

                    current_checkbox = '';

                    log.text = 'no selection';

                    return;
                }
            }
        }
    );

    row.add(title);

    row.add(checkbox);

    table_data.push(row);
}

table.data = table_data;

win.open();
