ActivityIndicator

The indicatorDiameter property isn't working. Has anyone tried this? I cannot enlarge the spinner. I have a custom background and I want the spinner to rotate around it.

exports.getActivityIndicator= function(){
var actInd = Titanium.UI.createActivityIndicator({
height:100,
width:100,
backgroundColor:'transparent',
style:Titanium.UI.iPhone.ActivityIndicatorStyle.BIG_DARK,
indicatorDiameter:'100'
});
var actView=Ti.UI.createView({
backgroundImage:'/images/H2.png',
height:100,
width:100,
})
actView.add(actInd);
return {actInd:actInd,actView:actView};
}
