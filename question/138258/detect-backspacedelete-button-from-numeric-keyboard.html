    • Hi Mark,

      Try this code.Hope it works for you….Best Luck

      pin1.addEventListener('change', function(e) {
    if (this.value.length == 1) {
        pin2.focus();
    }
    if (this.value.length == 0) {
        txt1.value = "";
        txt1.focus();
    }
});

pin2.addEventListener('change', function(e) {
    if (this.value.length == 1) {
        pin3.focus();
    }
    if (this.value.length == 0) {
        pin1.value = "";
        pin2.value = "";
        pin4.value = "";
        pin1.focus();
    }

});
pin3.addEventListener('change', function(e) {
    if (this.value.length == 1) {
        pin4.focus();
    }
    if (this.value.length == 0) {
        pin1.value = "";
        pin2.value = "";
        pin4.value = "";
        pin1.focus();
    }

});

pin4.addEventListener('change', function(e) {
    if (this.value.length == 1) {
        pin4.blur();
    }
    if (this.value.length == 0) {
        pin1.value = "";
        pin2.value = "";
        pin3.value = "";
        pin1.focus();
    }
});
