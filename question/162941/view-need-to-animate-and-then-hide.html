View need to animate and then hide

I have issue for hide view after the animation.

when i click add button view will bring up from bottom to top.

for the view i added close button btn_ClosePopUp;

when click btn_ClosePopUp need to animate view to bring top to bottom and then hide.

Issue is hide is calling immediately before animation. 



// add button action listener
 var view_PopUp = Titanium.UI.createView({
      width:'100%',
     height:'100%',
     left:0,
     bottom:'-100%',
     backgroundColor:'transparent'
 });

 var img_PopUp = Titanium.UI.createImageView({
      width:'100%',
     height:'100%',
     backgroundImage:'images/bgMain@2x.png',
     left:0,
     opacity:'0.6f'
 });

var anim_PopUp = Titanium.UI.createAnimation();
var anim_PopDown = Titanium.UI.createAnimation();

var btn_ClosePopUp = Ti.UI.createButton({
    title:'Close',
    width:90,
    height:35,
    right:12,
    top:10,
    backgroundColor:'yellow',
    font:{fontFamily:'Arial',fontWeight:'bold',fontSize:14},
    opacity:'1.0f'
    });

view_PopUp.add(img_PopUp);
view_PopUp.add(btn_ClosePopUp);


$.btn_Add.addEventListener('click',function(e){        
// // you could then add any interface elements, labels, etc to this.
view_PopUp.show();

anim_PopUp.bottom = 0;
anim_PopUp.duration = 1000;
view_PopUp.animate(anim_PopUp);

$.Window.add(view_PopUp);

});
btn_ClosePopUp.addEventListener('click',function(e){
    anim_PopDown.bottom = '-100%';
    anim_PopDown.duration = 1000;
    view_PopUp.animate(anim_PopDown);

});

    Hello Kiran,

    try this :

    anim_PopDown.addEventListener('complete',function(e){
    view_PopUp.hide();
});

    i hope this helps. thanks.
