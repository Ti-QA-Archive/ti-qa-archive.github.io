    var animation = Titanium.UI.createAnimation({
    duration : 700
});

var categoryView = Ti.UI.createView({
    height : 260,
    bottom : -260,
    backgroundColor :'white'
});

var btncatCancel = Ti.UI.createButton({
    title : 'Cancel',
    style : 1
});

var btncatDone = Ti.UI.createButton({
    title : 'Done',
    style : 1
});
var btnflexSpace = Titanium.UI.createButton({
    systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});
var cattoolview = Ti.UI.createToolbar({
    height : 40,
    width : '100%',
    left : 0,
    top : 0,
    items : [btncatCancel, btnflexSpace, btncatDone],
    barColor : '#000000'
});

var CatAnimation = Ti.UI.createAnimation({
    duration : 700
});

var categoryPicker = Ti.UI.createPicker({
    left : 0,
    top : 40,
    width : 320,
    height : 230,
});

var data = [];
var arrCat = [];
arrCat.push({
    id : 0,
    title : 'a'
});
arrCat.push({
    id : 1,
    title : 'b'
});
arrCat.push({
    id : 2,
    title : 'c'
});
for(var i = 0; i < arrCat.length - 1; i++) {
    var obj = arrCat[i];
    var row = Ti.UI.createPickerRow({
        id : i
    });
    var lbl = Ti.UI.createLabel({
        left : 25,
        text : 'C' + i + '-' + obj.title,
        font : {
            fontSize : 16,
            fontWeight : 'bold'
        }
    });
    row.add(lbl);

    data.push(row);

}

categoryPicker.add(data);

categoryPicker.selectionIndicator = true;
categoryPicker.addEventListener('change', function(e) {
    catValue = e.row.id;
});

btncatCancel.addEventListener('click', function(e) {
    animation.bottom = -260;
    categoryView.animate(animation);
});

btncatDone.addEventListener('click', function(e) {
    animation.bottom = -260;
    categoryView.animate(animation);
});
// categoryPicker.addEventListener('change', function(e) {
// categoryValue = e.value;
// });
categoryView.add(categoryPicker);
categoryView.add(cattoolview);
win.add(categoryView);

win.addEventListener('click', function(e) {
    animation.bottom = 0;
    categoryView.animate(animation);
});

  • You have to add a close button by yourself ontop of the picker.

    var cancel = Ti.UI.createButton({
        systemButton : Ti.UI.iPhone.SystemButton.CANCEL
    });
    cancel.addEventListener('click', function () {
        modalWin.close();
    });

    var done = Ti.UI.createButton({
        systemButton : Ti.UI.iPhone.SystemButton.DONE
    });
var spacer = Ti.UI.createButton({
        systemButton : Ti.UI.iPhone.SystemButton.FLEXIBLE_SPACE
    });

    var toolbar = Ti.UI.iOS.createToolbar({
        items :
                [
                    cancel,
                    spacer,
                    done
                ],
        height : 43,
        backgroundColor : '#bbb'
    });
currentWin.add(toolbar);

    you have to customize this example code for yourself. but that's the way, you can do it
