Window click event not fire when tableview is inside

Hallo,

i have the following problem. I want to trigger a click event on a window. This works if there is no tableview inside it. but if i put one inside the window. the window events didn't work anymore.
Does anyone know about this case?

Example:

function ApplicationTabGroup(Window) {
    //create module instance
    var self = Ti.UI.createTabGroup();

    //create app tabs
    var win1 = new Window(L('home')), win2 = new Window(L('settings'));

    win1.addEventListener('click', function() {
        alert("<< WINDOW CLICKED >>");
    });

    var table = Ti.UI.createTableView();

    win1.add(table);

    var tab1 = Ti.UI.createTab({
        title : L('home'),
        icon : '/images/KS_nav_ui.png',
        window : win1
    });
    win1.containingTab = tab1;

    var tab2 = Ti.UI.createTab({
        title : L('settings'),
        icon : '/images/KS_nav_views.png',
        window : win2
    });
    win2.containingTab = tab2;

    self.addTab(tab1);
    self.addTab(tab2);

    return self;
};

module.exports = ApplicationTabGroup;

  • HI Thomas, your table view is not clickable until you add some good rows to it, thats why you are not getting window click event.

    add this code 

    win1.addEventListener('click', function() {
        alert("<< WINDOW CLICKED >>");
    });
     var datat = [{
         title:"Ashish"
     },{
         title:"nigam"
     }]
    var table = Ti.UI.createTableView({
        data:datat
    });

    table.addEventListener('click', function(e) {
        alert(JSON.stringify(e));
    });
