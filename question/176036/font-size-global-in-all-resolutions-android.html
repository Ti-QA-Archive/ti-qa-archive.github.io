  • Hi Andres,

    Thanks for your concern.

    If I'm not mistaken to understand your problem, you want fantSize that varies with screen resolution like increase screen resolution increases the fontSize and decrease resolution decreases the the fontSize.

    To accomplish this task you can check my test code below and see if it helps you.

    TEST CODE

    var win = Ti.UI.createWindow({
    backgroundColor:'#fff',
});

var screenWidth = Ti.Platform.displayCaps.platformWidth;
var screenHeight = Ti.Platform.displayCaps.platformHeight;

function GetHeight(value) {
    var temp = (value * 100) / 480;
    return parseInt((screenHeight * temp) / 100);
}

function GetWidth(value) {
    var temp = (value * 100) / 320;
    return parseInt((screenWidth * temp) / 100);
}

var label = Ti.UI.createLabel({
        center:'0',
        color: '#019FCE',
        text: "Appcelerator Titanium Rockz !!!",
        font: {
            fontSize:GetHeight(15),
            fontFamily:'ABeeZee-Regular'
        },
        textAlign: 'center',
        wordWrap:true
});

win.add(label);
win.open();

    You can also use GetHeight and GetWidth function for height and width property. It works in all resolutions.
