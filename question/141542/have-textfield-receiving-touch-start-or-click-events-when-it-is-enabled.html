  • I'm not sure about the touchstart event, but I was able to get the swipe event to register correctly:

    var win = Ti.UI.createWindow({  
    backgroundColor:'#fff',
    barColor: '#1194e2',
    layout: 'vertical',
});

var txtField = Ti.UI.createTextField({
    hintText:'hintText',
    borderColor:'#000',
    touchEnabled:true
});
txtField.addEventListener('swipe', function(e) {
    Ti.API.info("swipe: "+JSON.stringify(e));
});
win.add(txtField);

win.open();

    output:

    [INFO] swipe: {"y":14,"type":"swipe","globalPoint":{"x":153,"y":34},"direction":"down","x":23.5,"source":{"borderColor":"#000","touchEnabled":true,"horizontalWrap":true,"hintText":"hintText"}}
