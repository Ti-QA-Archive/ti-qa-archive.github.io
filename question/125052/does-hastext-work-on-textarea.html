  • There is no "HasText" method in "TextArea" component, but you can write your own function that does that. For example:

    function addHasTextFunc(textArea) {
    if (textArea) {
        textArea.hasText = function() {
            return !!this.value;
        };
    }
}


***

var myTextField = Ti.UI.createTextField({ ... });
addHasTextFunc(myTextField);
Ti.alert(myTextField.hasText());
