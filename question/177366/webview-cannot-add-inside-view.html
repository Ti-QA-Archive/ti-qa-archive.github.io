  • Hi Pangwisitpanya,

    Thanks for your concern.

    Please check my test case below and see if it helps you.

    var window = Ti.UI.createWindow({
    backgroundColor:'#fff',
    exitOnClose:true
});

var view = Titanium.UI.createView({
   top:'10%',
   borderRadius:10,
   borderColor:'green',
   borderWidth:5,
   backgroundColor:'red',
   width:'80%',
   height:'80%'
});
window.add(view);

var webview = Titanium.UI.createWebView({
     url:'http://www.appcelerator.com'
});
view.add(webview);

webview.addEventListener('click', function() {
    Ti.API.info('WebView Clicked !!!');
});

window.open();

    Thanks
