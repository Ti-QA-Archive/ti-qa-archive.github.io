function showDeleteItemDialog(id, type) {

    Ti.API.info('showDeleteItemDialog');

    var opts = {
         cancel: 1,
          options: ['Delete', 'Cancel'],
    };

    var dialog = Ti.UI.createOptionDialog(opts);

    dialog.addEventListener('click', function(e) {
        switch ( e.index ) {
            case 0:
                deleteDiaryEntry(e, id, type);
                break;
            default:
                break;
        }
    });

    dialog.show();

}





function showEditScreen(e) {
//Ti.API.info('e: ' + JSON.stringify(e.source));
    if ( !isLoading ) {
        isLoading = true;    
        Ti.API.info("e: " + e.source.parentRow);    
        var row = e.source.parentRow;
        var    editParams = {
                editing: true,
                date: currentDate,
                exerciseID: row.id,
                title: row.exerciseTitle,
                exerciseType: row.parentExerciseType,        
                source: 'planner'
            };

        Ti.API.info('type: ' + row.exerciseType);

        if ( row.parentExerciseType === 'cardio' ) {
            editParams.cals = row.cals;
            editParams.calsPerMinute = row.calsPerMinute;
            editParams.duration = row.duration;
            editParams.hours = row.hours;
            editParams.minutes = row.minutes;
        } else {
            editParams.sets = row.sets;
            editParams.reps = row.reps;
            editParams.weight = row.weight;

        }

        Ti.API.info('editParams: ' + JSON.stringify(editParams) );


        xpng.openWin(Alloy.Globals.parentNav, 'fitness/addToDiary', editParams);
        setTimeout(function() {
            isLoading = false;
        }, 2000);
    }

}
