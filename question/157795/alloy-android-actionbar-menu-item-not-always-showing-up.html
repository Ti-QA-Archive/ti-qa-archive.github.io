Alloy: Android ActionBar Menu Item not always showing up

Hi,

I've got the below code as part of my application:

TabGroup.js

function doOpen(){
    var activity = $.TabGroup.activity;
        activity.actionBar.title = "My Application";
        activity.actionBar.backgroundImage = Alloy.CFG.app.barImage;
        activity.actionBar.displayHomeAsUp = false;
};

function doClickNewStory() {
    Alloy.createController('NewStory');
};

$.TabGroup.open();

Alloy.Globals.TabGroup = $.TabGroup;

TabGroup.xml


<Alloy>
    <Window id="TabGroup" navBarHidden="false" exitOnClose="true" onOpen="doOpen">
        <Menu id="menu">
            <MenuItem id="btnNewStory" onClick="doClickNewStory" />
        </Menu>    
    </Window>
</Alloy>

TabGroup.tss

'#btnNewStory': {
    title: 'New Story',
    showAsAction: Ti.Android.SHOW_AS_ACTION_ALWAYS
}

I'm seeing that sometime when the app loads up the 'New Story' is is not shown on the action bar and I have to exit out of the application and go back in for it to show up, sometimes more than once

Does anyone know why this happens?
