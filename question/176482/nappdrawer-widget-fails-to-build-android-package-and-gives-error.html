    iOS and Android are not created equal. Not all drawer options are available to each.

    following works adjust for your own use.

    "#drawer": {
    openDrawerGestureMode: "OPEN_MODE_ALL",
    closeDrawerGestureMode: "CLOSE_MODE_ALL",
    leftDrawerWidth: '250dp',
    rightDrawerWidth: '320dp',
    barColor: "white",
    statusBarStyle: 1,
    navTintColor: '#fff',
}
"#drawer[platform=android]" : {
  openDrawerGestureMode: "OPEN_MODE_ALL",
    closeDrawerGestureMode: "CLOSE_MODE_MARGIN",
    leftDrawerWidth: '250dp',
    rightDrawerWidth: '320dp',
    barColor:"white",
    statusBarStyle: 1,
    navTintColor: '#fff',
    backgroundColor:"#ff6600"
}
