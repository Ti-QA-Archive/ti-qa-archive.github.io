Global variables on Android

Can anyone tell me why this simple global variable test doesn't work on the Android simulator but works perfectly ok on the iOS simulator (I've not tried devices but I assume I would get the same result). In iOS the global variable is detected in the function ok but on Android I get an error of "myGlobalVariable is undefined"

I'm using SDK 1.8.1 and Android 2.2

app.js

var ApplicationWindow = require('ApplicationWindow');
var myGlobalVariable = 1;
var myWindow = new ApplicationWindow;
myWindow.open();

ApplicationWindow.js

function ApplicationWindow() {

    //create object instance    
    var self = Titanium.UI.createWindow({
        title : 'My Window',
        backgroundColor : '#fff'
    });

    alert(myGlobalVariable);

    //return instance from constructor
    return self;
}

module.exports = ApplicationWindow;
