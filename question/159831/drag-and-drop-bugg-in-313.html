    Try this code and i hope that it will help full for you

    
Titanium.UI.setBackgroundColor('#000');

var posx=null;
var posy=null;


var win = Titanium.UI.createWindow({
    backgroundColor:'gray',
    title:'Drag and Drop'
});


 var contentView=Ti.UI.createView({
    height:100,
    backgroundColor:'#fff',
    top:300,
    left:50,
    width:200,
    });

 win.add(contentView);


var temp = Titanium.UI.createView({
    width:50, 
    height:50,
    top:20, 
    left:20,
    borderRadius:5,backgroundColor:'green',
});
win.add(temp);

var view2 = Titanium.UI.createView({
    width:50, 
    height:50,
    top:20 , 
    left:20,borderRadius:5,
});
win.add(view2);


var store = {};


view2.addEventListener('touchstart', function(e){
    store.x = e.x - temp.left;
    store.y = e.y - temp.top;
});


view2.addEventListener('touchmove', function(e){
    temp.left = e.x - store.x;
    temp.top = e.y - store.y;
});


view2.addEventListener('touchend', function(e){
    callForPlace(e.x - store.x,e.y - store.y);
    view2.left =posx;
    view2.top = posy;
    temp.left=posx;
    temp.top =posy;
});


var antemp = Titanium.UI.createView({
    width:50, height:50,top:20, left:80,
    backgroundColor:'blue',borderRadius:5,
});
win.add(antemp);

var anview2 = Titanium.UI.createView({
    width:50, height:50,top:20 , left:80,borderRadius:5,
});
win.add(anview2);


var anstore = {};


anview2.addEventListener('touchstart', function(e){
    anstore.x = e.x - antemp.left;
    anstore.y = e.y - antemp.top;
});


anview2.addEventListener('touchmove', function(e){
    antemp.left = e.x - anstore.x;
    antemp.top = e.y - anstore.y;
});


anview2.addEventListener('touchend', function(e){

    callForPlace(e.x - anstore.x,e.y - anstore.y);
    anview2.left =posx; 
    anview2.top = posy;
    antemp.left=posx;
    antemp.top =posy;
});

win.open();

function callForPlace(chX,chY)
{

    if((chX>=50 && chX<=225) && (chY>=225 && chY<=400))
    {
        posx=50;posy=310;

    }
    else
    {
        posx=chX;posy=chY;
    }


}
