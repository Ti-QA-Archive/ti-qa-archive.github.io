Click a view while animating

Is it possible to interact (click) with a view while this is animated? The click event is fired only when the view is stopped or at the end of the animation, but not while the animation is running.

The code below is just a test. What I want to do is to replace the image of an imageView when this is touched while animating.

target: iOS, Titanium SDK 3.1.0

    var win = Ti.UI.createWindow({height: '100%', width: '100%', backgroundColor: '#fff'});    

    var view = Ti.UI.createView({top: 100, left: 0, width: 100, height: 100, backgroundColor: 'red'});
    win.add(view);

    view.addEventListener('click', function() {
        alert('click');
    });

    var button = Ti.UI.createButton({top: 10, title: 'Start'});
    win.add(button);

    button.addEventListener('click', function() {
        view.animate({
            left: 600,
            duration: 8000,
            //transform: Ti.UI.create2DMatrix().translate(600, 100)
        });
    });

    win.open();

  • Hi I have modified the sample, where you can find an alternative approach for getting the touch event while animating.

    Sample Code

    var win = Ti.UI.createWindow({
    height : '100%',
    width : '100%',
    backgroundColor : '#fff'
});

var view = Ti.UI.createView({
    top : 100,
    left : 0,
    width : 100,
    height : 100,
    backgroundColor : 'red',
    id : "viewObj1"
});
win.add(view);

view.addEventListener('touchmove', function() {
    Ti.API.info('########################## click');
});

var button = Ti.UI.createButton({
    top : 10,
    title : 'Start'
});
win.add(button);

button.addEventListener('click', function() {
    view.animate({
        left : 600,
        duration : 8000,
        //autoreverse:true
        //transform: Ti.UI.create2DMatrix().translate(600, 100)
    });

    var int1 = setInterval(function() {
        view.left += 1;
    }, 10);

    setTimeout(function() {
        clearInterval(int1);
        iterateAgain();
    }, 8000);

    function iterateAgain() {
        var int2 = setInterval(function() {
            view.left -= 1;
        }, 10);

        setTimeout(function() {
            clearInterval(int2);
        }, 8000);

    }
    //Ti.API.info('########################## click');
});

win.addEventListener('touchstart', function(e) {

    if (e.source.id == "viewObj1") {
        Ti.API.info("view clicked");
    }
    Ti.API.info('########################## win s click' + "         ::       " + view.left + "          ::        " + e.source.toString());
});

win.addEventListener('touchmove', function() {
    Ti.API.info('########################## win click');
});

win.open();
