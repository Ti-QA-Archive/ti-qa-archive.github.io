  • You need to use a javascript: link on your html to call a function in the head of the html, which can then fire a custom event back to your application JS.

    
<!DOCTYPE html PUBLIC "-//W3C//DTD HTML 4.01//EN"
"http://www.w3.org/TR/html4/strict.dtd">
<html lang="en" dir="rtl">
    <head>


    function link(u){
        // fire an event with the url back to your application js
        Ti.App.fireEvent('link_click',{url:u});
    }



    </head>
    <body>
        <img src="../images/LogoO.png"/>
        <p>
            some text
        </p>

        <p>
            some text
        </p>
        <p>
            some text
        </p>
            <a href="nojavascript...link('http://abcd.com')" target="_blank">www.abcd</a> 
            <a href =" " >service@abcds.com</a>

    </body>
</html>

    then in your application JS put something like:

    

var back_button = Titanium.UI.createButton({
    title:'< back',
    style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN
});

back_button.addEventListener('click',function(){

    Ti.API.info('the back button was clicked... ');

}):

// set up the custom event listener.. which is called from the webview html
Titanium.App.addEventListener('link_click', function(e){

    Ti.API.info('a link to '+e.url+' was clicked');

    // do something ...
    // ie: 
    Ti.UI.currentWindow.setLeftNavButton(back_button);


});

    hope this helps.
