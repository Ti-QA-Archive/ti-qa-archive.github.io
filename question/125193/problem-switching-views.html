Problem switching views

I am trying my darndest to create a basic screen that has three tabs across the bottom. When a user clicks a tab, I want to show a corresponding view. However, no matter what I do, I get an error and the iPhone emulator exits. Can somebody please tell me what I'm doing wrong? Here is my code:

var view = Titanium.UI.createTableView();
if (Titanium.Platform.name == 'iPhone OS') {
    view.appendRow({
        hasChild:false,
        header:'[result name]',
        separatorColor:'#fff'
    });
}
else {
    view.appendRow({
        hasChild:false,
        header:'[result name]',
    });    
}

var view1 = createView1();
var view2 = createView2();
var view3 = createView3();
var tabbedBar = createTabbedBar();
var currentSelectedView = null;


Titanium.UI.currentWindow.add(view);
function createView1() {
    var temp = Titanium.UI.createView({
        height:100
    });

    var l = Titanium.UI.createLabel({
        text:'About'
    });
    temp.add(l);

    return temp;
}

function createView2() {
    var temp = Titanium.UI.createView({
        height:100
    });

    var l = Titanium.UI.createLabel({
        text:'Details'
    });
    temp.add(l);

    return temp;
}

function createView3() {
    var temp = Titanium.UI.createView({
        height:100
    });

    var l = Titanium.UI.createLabel({
        text:'reviews'
    });
    temp.add(l);

    return temp;
}

function createTabbedBar() {
    var temp = Titanium.UI.createTabbedBar({
        labels:['View 1', 'View 2', 'View 3'],
        index:0
    });

    temp.addEventListener('click', function(e)
    {
        view.remove(currentSelectedView);
        if (e.index == 0) {
            view.add(view1);
            currentSelectedView = view1;
        }
        else if (e.index == 1) {
            view.add(view2);            
            currentSelectedView = view2;
        }
        else {
            view.add(view3);
            currentSelectedView = view3;
        }
    });

    var flexSpace = Titanium.UI.createButton({systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE});
    Titanium.UI.currentWindow.setToolbar([flexSpace,temp,flexSpace]);        

    view.add(view1);
    currentSelectedView = view1;
}
