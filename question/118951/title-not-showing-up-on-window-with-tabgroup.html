Title not showing up on window with tabgroup

Hi,

I'm using Android APIs 2.2, Titanium 1.6.2.

I created a window with tabgroups, and gave it a title. However, The window's title does not show up. Is there any way to make it show?

Here is my code:

app.js

var win = Ti.UI.createWindow({    // create new window when deal row is click
    url:'../main/index.js',
    backgroundImage:'../images/bkg.png',
    title:'My Index',
});

function openApp(){
    win.open();
}

openApp();

index.js

var tabGroup = Titanium.UI.createTabGroup();

var win1 = Ti.UI.createWindow({
    url: '../main/win1.js',
    title: 'Win1',
});
var tab1 = Ti.UI.createTab({
    icon: '../images/tabs/win1.png',
    window:win1,
});

var win2 = Ti.UI.createWindow({
    url:'../main/win2.js',
    title:'Win2',
});
var tab2 = Ti.UI.createTab({
    icon:'../images/tabs/win2.png',
    title:'Categories',
    window:win2
});

tabGroup.addTab(tab1);
tabGroup.addTab(tab2);

tabGroup.setActiveTab(0); 
tabGroup.open();

Thanks.
