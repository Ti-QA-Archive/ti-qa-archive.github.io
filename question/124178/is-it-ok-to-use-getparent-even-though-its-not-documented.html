Is it OK to use getParent() even though its not documented?

Can anyone explain a better way of doing this?

I am capturing propagated clicks from children components, by using the getParent() in the parent objects eventListener.

This keeps my code clean and I do not have to put an Event listener on each child object, but the getParent() method is not documented in the API

// Example
var homeBtn = function(){

var data = {
    id:1
};

homeBtn = Ti.UI.createWindow({
    width:100,
    height:100,
    data: data    // Static test data    
});

homeBtn.addEventListener('click',function(e){
    if(e.source.data === undefined){
        Ti.API.info('data getParent: ' + this.getParent().data.id);
    } else {
        Ti.API.info('data: ' + e.source.data.id);
    }
});

var image = Ti.UI.createImageView({
    image: 'KS_nav_ui.png',
    width:48,
    height:48
});

var label = Ti.UI.createLabel({
    text: 'test',
    top: 70,
    left: 10
});

// children    
homeBtn.add(image);
homeBtn.add(label); 

return homeBtn;

}

    • var image = Ti.UI.createImageView({
    image: 'KS_nav_ui.png',
    width:48,
    height:48,
    touchEnabled:false
});

var label = Ti.UI.createLabel({
    text: 'test',
    top: 70,
    left: 10,
    touchEnabled:false
});

      With this, even if you click on the label or image, the click is ignored by them and passed on to the parent window.
