  • the original window definition is wrong… you should define modal: true when creating the window

    
    var w = Titanium.UI.createWindow({
        title:'Add Location',
        backgroundColor:'#000000',
        opacity:1,
        barColor:'#000000',
        modal: true // <--- change
    });
