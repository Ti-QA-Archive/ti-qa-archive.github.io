  • not sure what you mean by "displays only in the menubar" but if i understand you correctly, set your main window visibility to false on startup and toggle it on/off.

    var app = Titanium.UI.getMainWindow();
var tray = Titanium.UI.addTray(iconURL="images/appicon.png", function() {
if(app.visible) {
  app.visible = false;
} else {
  app.visible = true;
}
});
