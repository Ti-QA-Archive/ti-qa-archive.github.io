Hi,

I'm using this moduel to realize a video widget. After starting a (long) black screen appears.How can I implement an interims view?

exports.create = function(_args) {
    var url = _args.mp4;
    var win = Ti.UI.createWindow();
    win.add(Ti.UI.createImageView({
        image : _args.poster
    }));
    var videoplayer = Ti.Media.createVideoPlayer({
        autoplay : true,
        fullscreen : true,
        url : url,
        mediaControlStyle : Ti.Media.VIDEO_CONTROL_DEFAULT,
        scalingMode : Ti.Media.VIDEO_MODE_FILL
    });
    videoplayer.addEventListener('playbackstate', function(_e) {
    });
    videoplayer.addEventListener('complete', function(e) {
        if (e.reason == 0) {
            win.close();
        };
    });
    videoplayer.addEventListener('fullscreen', function(e) {
        if (e.entering == 0) {
            win.close();
        };
    });
};
