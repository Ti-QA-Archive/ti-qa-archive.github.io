View animation

I have two views and on click of first view I have changed the position of those view using animate property.
Here is my sample code given below.

var win = currentWindow;

var view1 = Ti.UI.createView({
    width:100,
    height:20,
    left : 10,
    backgroundColor:'#f00'
});
win.add(view1);

var view2 = Ti.UI.createView({
    width:100,
    height:20,
    right : 10,
    backgroundColor:'yellow'
});
win.add(view2);

view1.addEventListener('click', function(){
    view1.animate({ 
            duration:3000, 
            right    : 10
        });

        view2.animate({
            duration:3000, 
            left    : 10
        });
});

Problem is that when I click on the 1st view (view1) it changes position but after that view1 gets disappeared. Can't got the reason.

Please suggest.

  • Accepted Answer

    It has to do with setting left then right (or the opposite) as it holds both.

    Try this instead:

    var win = currentWindow;
var right = Ti.Platform.displayCaps.platformWidth - 110;

var view1 = Ti.UI.createView({
    width:100,
    height:50,
    left : 10,
    backgroundColor:'#f00'
});
win.add(view1);

var view2 = Ti.UI.createView({
    width:100,
    height:50,
    left : right,
    backgroundColor:'yellow'
});
win.add(view2);

view1.addEventListener('click', function(){
    view1.animate({ 
            duration:3000, 
            left   : right
        });

        view2.animate({
            duration:3000, 
            left    : 10
        });
});

    • Thanks David, You are correct. Its relates with right/left position while animating. Here my code where I have make it fixed and it runs well. I have tested it on Android.

      var win = Ti.UI.currentWindow;

var view1 = Ti.UI.createView({
    width:100,
    height:50,
    left : 65,
    top        : 0,
    opacity : 0.5,    
    backgroundColor:'red'
});
win.add(view1);

var view2 = Ti.UI.createView({
    width:100,
    height:50,
    left : 165,
    top        : 0,
    opacity : 0.5,
    backgroundColor:'yellow'
});
win.add(view2);

view1.addEventListener('click', function(){
    view1.animate({ 
            duration:2000, 
            left    : 165,
            top    : 0
        });

        view2.animate({
            duration:2000, 
            left    : 65,
            top    : 0
        });
});
