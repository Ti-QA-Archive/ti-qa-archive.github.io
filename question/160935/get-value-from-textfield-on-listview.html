Get value from textfield on listview

Hi,
I'm not able to read value from textfield on listview.
I've created a listview with this code

    var myTemplate = {
        properties : {
            height : 50
        },
        childTemplates : [{
            type : 'Ti.UI.TextField',
            bindId : 'bindField',
            properties : {
                left : 10,
                width : Ti.UI.FILL,
                borderStyle : Ti.UI.INPUT_BORDERSTYLE_NONE,
                color : 'black'
            }
            , events : { change :textFieldChanged
            }

        }]
    };

    var sectionsArray = [];
    var section1 = Ti.UI.createListSection();
    sectionsArray.push(section1);
    var data = [];
    data.push({
        template : 'myCell',
        bindField : {
            hintText : 'Insert Text'
        },
        properties : {
            selectionStyle : Ti.UI.iPhone.ListViewCellSelectionStyle.NONE,
            itemId : 'TextField1'
        }
    });

    section1.setItems(data);
    var listView = Ti.UI.createListView({
        templates : {
            'myCell' : myTemplate
        },
        allowsSelection : true,
        style : Ti.UI.iPhone.ListViewStyle.GROUPED,
        sections : sectionsArray
    });

Now from a click event on a button I would like to get and process text value on textfield1, but I cannot do it.
I tried to get it with

var textFieldItem = sectionsArray[0].getItemAt(0);

but I received only the original object : 

{"template":"myCell","bindField":{"hintText":"Insert Text"},"properties":{"itemId":"TextField1","selectionStyle":0}}

Using function textFieldChange related to Change event.
If i print object e and the same textFieldItem, I get: 

{"sectionIndex":0,"value":"T","section":{},"itemIndex":0,"itemId":"TextField1","bindId":"bindField","bubbles":true,"type":"change","source":{"left":10,"hintText":"Insert Text","horizontalWrap":true,"borderStyle":0,"color":"black","width":"FILL","bindId":"bindField","value":"T"},"cancelBubble":false} 
and 
{"template":"myCell","bindField":{"hintText":"Insert Text"},"properties":{"itemId":"TextField1","selectionStyle":0}}

I'm sure that I'm missing something, but I don't understand what.. :-\
Please may you help to retrieve value of TextField1?

Many thanks

    • try this sample code…

      var txtField = null;

function textReturn(e) {
    txtField = e;
}

var myTemplate = {
    properties : {
        height : 50
    },
    childTemplates : [{
        type : 'Ti.UI.TextField',
        bindId : 'bindField',
        properties : {
            left : 10,
            width : Ti.UI.FILL,
            borderStyle : Ti.UI.INPUT_BORDERSTYLE_NONE,
            color : 'black'
        },
        events : {
            change : textFieldChanged,            
            'return' : textReturn
        }

    }]
};

var sectionsArray = [];
var section1 = Ti.UI.createListSection();
sectionsArray.push(section1);
var data = [];
data.push({
    template : 'myCell',
    bindField : {
        hintText : 'Insert Text'
    },
    properties : {
        selectionStyle : Ti.UI.iPhone.ListViewCellSelectionStyle.NONE,
        itemId : 'TextField1'
    }
});

section1.setItems(data);
var listView = Ti.UI.createListView({
    templates : {
        'myCell' : myTemplate
    },
    allowsSelection : true,
    style : Ti.UI.iPhone.ListViewStyle.GROUPED,
    sections : sectionsArray,
    height : 400
});
win.add(listView);

var btn = Ti.UI.createButton({
    width : 60,
    height : 40,
    title : 'Get Value',
    bottom : 10
});

btn.addEventListener('click', function(e) {
    if (txtField) {
        alert(txtField.value);
    }
});

win.add(btn);
win.open();
