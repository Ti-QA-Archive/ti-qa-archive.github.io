    If I understand right, is more or less what you want:

    .

    The code for the above image is below:

    // create window //
var myWin = Ti.UI.createWindow();





// Header for Photo and Name //
var myHeader = Ti.UI.createView({
    height:100
});
// Photo //
var myPicture = Ti.UI.createImageView({
    image: 'myphoto.png',
    height:80,
    width:80,
    left:10,
    borderWidth:1,
    borderColor:'#999',
    borderRadius:6,
    top:0
});
// Name //
var myLabel = Ti.UI.createLabel({
    text:'My Name',
    height:20,
    left:100,
    font:{fontSize:20},
    top:30
});
// Add Picture and Name to Header //
myHeader.add(myPicture);
myHeader.add(myLabel);






// Footer Row for 3 buttons
var myFooter = Ti.UI.createView({
    height:50
});
// My Buttons //
var myBtn1 = Ti.UI.createButton({
    title:'Button 1',
    width:'30%',
    left:10,
    height:40,
    top:10
});
var myBtn2 = Ti.UI.createButton({
    title:'Button 2',
    width:'30%',
    left:'35%',
    height:40,
    top:10
});
var myBtn3 = Ti.UI.createButton({
    title:'Button 3',
    width:'30%',
    right:10,
    height:40,
    top:10
});
// Add all 3 buttons to Footer //
myFooter.add(myBtn1);
myFooter.add(myBtn2);
myFooter.add(myBtn3);




// Data content //
var data = [];

// Create Section
var section = Ti.UI.createTableViewSection();
// Add Header and Footer to Section
section.headerView = myHeader;
section.footerView = myFooter;

// Add Section to Data content
data[0] = section;

// Finaly add your fields to Section
section.add(Ti.UI.createTableViewRow({title:'Row 1'}));
section.add(Ti.UI.createTableViewRow({title:'Row 2',hasChild:true}));



// Table View //
var myTable = Ti.UI.createTableView({
    data:data,
    style: Titanium.UI.iPhone.TableViewStyle.GROUPED,
    top:0,
    left:0,
    height:'100%',
    width:'100%'
});

// Add Table to Window //
myWin.add(myTable);


// Open Window //
myWin.open();
