var bar1 = Ti.UI.createView({
    backgroundColor : '#ffffff',
    width : 254,
    height : 7,
    transform : t3
});
win1.add(bar1);
var bar2 = Ti.UI.createView({
    backgroundColor : '#d90012',
    width : 250,
    height : 3,
    transform : t3
});
win1.add(bar2);
win1.open();
