  • that's my function

    var places = '';

var ottieniVia = function(latitude,longitude){

    Titanium.Geolocation.reverseGeocoder(latitude,longitude,function(evt)
        {

            //address = JSON.stringify(evt);
            places = evt.places;
            Ti.API.info("reverse geolocation result = "+JSON.stringify(evt));

  });

    return places; 

};
