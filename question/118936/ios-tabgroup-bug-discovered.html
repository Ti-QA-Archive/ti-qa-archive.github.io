IOS tabgroup bug discovered

I have an application that has UI similar to the facebook app at start where there is a tabgroup and then buttons on the page. If you click the buttons you load up a different tabgroup for that section. I have discovered a bug, the code is at the bottom of the post. Basically if you load and close a tabgroup 6 times you get the following error and app closes

2011-04-25 16:11:45.472 test[8016:207] -[NSRecursiveLock dealloc]: lock (<NSRecursiveLock: 0x6d11d80> '(null)') deallocated while still in use
2011-04-25 16:11:45.473 test[8016:207] Break on _NSLockError() to debug.
test(8016,0xa0a21540) malloc: error for object 0x7819000: pointer being freed was not allocated set a breakpoint in malloc_error_break to debug

I though it was a memory issue at first but i included Ti.Platform.availableMemory to check it and it wasnt a problem. I dont know what the magically 6 button is. The code i provided was a simplified version of my application but on mine as well as another other code i try it is always 6 attempts on opening and closing tabgroup that errors out even thought the memory is fine.

// this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');

var firstGroup = Titanium.UI.createTabGroup();

var secondGroup = Titanium.UI.createTabGroup();

var firstWindow = Titanium.UI.createWindow({
title:"firstwindow"
});
var backButton = Titanium.UI.createButton({
title: 'Back',
style: Titanium.UI.iPhone.SystemButtonStyle.PLAIN
});
var button = Titanium.UI.createButton({
top:5,
left:5,
text:"go to other tabgroup"
});

firstWindow.add(button);

var secondWindow = Titanium.UI.createWindow({
title:"secondwindow",
leftNavButton: backButton
});

var firstTab = Titanium.UI.createTab({
icon:'KS_nav_views.png',
title:'Home',
window:firstWindow
});

var secondTab = Titanium.UI.createTab({
icon:'KS_nav_views.png',
title:'Home',
window:secondWindow
});

firstGroup.addTab(firstTab);

secondGroup.addTab(secondTab);

firstGroup.open();

button.addEventListener('click', function(e)
{
firstGroup.close({transition:Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});
secondGroup.open();
Ti.API.info(Ti.Platform.availableMemory);
});

backButton.addEventListener('click',function(e)
{
secondGroup.close({transition:Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});
firstGroup.open();
Ti.API.info(Ti.Platform.availableMemory);
});
