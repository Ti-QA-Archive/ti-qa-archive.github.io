  • see , in my application on a table row i m creating 4 labels and an imageview , n tried everything on them but tha last thing that worked is events on the tables(touchstart,touchend,touchmove,click).apply index to all child views of rows according to row index.

    tblCards.addEventListener('touchstart', function(e) {
            currentEvent = e.type;
            moreThanTime = false;
            touched = setTimeout(function() {
                if(currentEvent == 'touchstart') {
                    moreThanTime = true;
            showOptionDialog1(e.source.index);
                }
            }, 1500);
        });
tblCards.addEventListener('touchend', function(e) {
            clearTimeout(touched);
            if(!moreThanTime) {
                searchBar.blur();
                DetailScreen.setData(cardsArray[e.source.index]);
                DetailScreen.loadImages(cardsArray[e.source.index]);
                }
        });
        tblCards.addEventListener('click', function(e) {
            searchBar.blur();

        });
        tblCards.addEventListener('touchmove', function(e) {
            currentEvent = e.type;
            moreThanTime = false;
            clearTimeout(touched);
        });
