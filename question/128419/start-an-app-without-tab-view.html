Start an app without tab view

Sorry for this really dumb question. I've built my first app, where I have three windows opened using a tab group. I want to see how this looks/functions whereby the app opens with tableview, from where users can access click a row to open one of the three individual windows.

Despite using tableviews extensively in my app, I have always done this within a tab group. I can't get it to work without a tab group. I've looked the nav group example in kitchensink, but still no joy.

Does anyone have a simple example of how to implement this?

  • something like this:

    
var init_win = Ti.UI.createWindow();

var win_one = Ti.UI.createWindow({title:'one'});
var win_two = Ti.UI.createWindow({title:'two'});
var win_three = Ti.UI.createWindow({title:'three'});



var nav = Titanium.UI.iPhone.createNavigationGroup({
    window:win_one
});


var b1 = Ti.UI.createButton({
    title:'open win_two',
    height:40,
    width:200
});


b1.addEventListener('click', function() {
    nav.open(win_two);
});

win_one.add(b1);

var b2 = Ti.UI.createButton({
    title:'open win_three',
    height:40,
    width:200
});

win_two.add(b2);

b2.addEventListener('click', function(){
    nav.open(win_three);
});




init_win.add(nav);

init_win.open();

  • There's nothing mysterious about the way a Titanium application opens. The tab group is just one way. The goal is to create one window in app.js from which the application is launched. I like to create window modules in their own files and/or namespaces and just include them to be opened in whatever other window needs them. To create the main window, I have a base level window created in base.js which acts as the main window. It will then include whatever windows it needs.

    This creates a very structured single-context application. Each window then opens, does its thing and then is closed. Neat and clean without using global variables.

    For this example, you wanted a base window with a tableview that opens multiple windows.

    We'll need 6 files for this:

    app.js (bootstrap and open the main window)
    ui.js commonly used ui functions
    base.js main window creation routine
    window1.js first window to be opened by base.js
    window2.js second window to be opened by base.js
    window3.js third window to be opened by base.js

    In turn, window1, window2, window3 need to include and open any other windows they need to do for your app.

    You asked for a tableview, so here are the files, but this could easily be done using buttons or any other object that has an event that can be fired to open up the window.

    app.js

    // this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');

var app = {};
app.isAndroid = (Titanium.Platform.osname == 'android');
app.platform = Titanium.Platform.osname;
app.version = Titanium.App.getVersion();

Ti.include("ui.js");
Ti.include("base.js");

app.mainWindow = app.UI.createMainWindow();
app.mainWindow.open();

    base.js

    (function()
{    
    Ti.include("window1.js");
    Ti.include("window2.js");
    Ti.include("window3.js");

    app.UI.createMainWindow = function()
    {    
        var win = Titanium.UI.createWindow({
            backgroundColor:'white',
            navBarHidden:true,
            exitOnClose:true,
            data:{},
            orientationModes: [Titanium.UI.PORTRAIT]
        });

        app.UI.winHeight = win.height;
        app.UI.winWidth = win.width;

        // leverage javascript custom properties to give us the window open function
        var data = [
            {title:'Open Window 1', hasChild:true, openWindowFunction: app.UI.createWindow1, extraData:{foo:'bar1'}},
            {title:'Open Window 2', hasChild:true, openWindowFunction: app.UI.createWindow2, extraData:{foo:'bar2'}},
            {title:'Open Window 3', hasChild:true, openWindowFunction: app.UI.createWindow3, extraData:{foo:'bar3'}},
        ];

        // create table view
        var tableview = Titanium.UI.createTableView({
            data:data
        });

        // create table view event listener
        tableview.addEventListener('click', function(e)
        {
            // get window function and pass it any open options
            // we use this so we don't need to pass around any global variables'
            app.UI.openWindow(e.rowData.openWindowFunction, e.rowData.extraData);
        });
        win.add(tableview);
        return win;        
    }
})();

    ui.js

    (function()
{
    if (app.UI != null && app.UI != undefined)
    {
        Ti.API.info("main window already exists, not recreating");
        return;
    }
    app.UI = {};

    app.UI.openWindow = function(createWindowFunction, extraData)
    {
        extraData = (extraData != null && extraData != undefined ) ? extraData : {};

        var newWindow = createWindowFunction(extraData);

        if (Titanium.UI.currentTab != null)
        {
            Titanium.UI.currentTab.open(newWindow,{animated:anim});
        }
        else
        {
            newWindow.open({animated:true});
        }
        return newWindow;    
    }

})();

    window1.js

    (function()
{    
    if (app.UI.createWindow1 != null && app.UI.createWindow1 != undefined)
    {
        Ti.API.info("window1 function already exists, not recreating");
        return;
    }

    app.UI.createWindow1 = function(data)
    {
        var win = Titanium.UI.createWindow({
            backgroundColor: 'red',
            width:'100%',
            height:'100%',
        });

        // use this data however you like
        win.data = data;

        var closeButton = Ti.UI.createButton({
            top: 10,
            left: 10,
            width: 50,
            height: 30,
            title: 'Close'
        });
        closeButton.addEventListener('click', function()
        {
            win.close();
        });
        win.add(closeButton);

        return win;
    }
})();

    window2.js

    (function()
{    
    if (app.UI.createWindow2 != null && app.UI.createWindow2 != undefined)
    {
        Ti.API.info("window2 function already exists, not recreating");
        return;
    }

    app.UI.createWindow2 = function(data)
    {
        var win = Titanium.UI.createWindow({
            backgroundColor: 'green',
            width:'100%',
            height:'100%',
        });

        // use this data however you like
        win.data = data;

        var closeButton = Ti.UI.createButton({
            top: 10,
            left: 10,
            width: 50,
            height: 30,
            title: 'Close'
        });
        closeButton.addEventListener('click', function()
        {
            win.close();
        });
        win.add(closeButton);

        return win;
    }
})();

    window3.js

    (function()
{    
    if (app.UI.createWindow3 != null && app.UI.createWindow3 != undefined)
    {

        return;
    }

    app.UI.createWindow3 = function(data)
    {
        var win = Titanium.UI.createWindow({

    • The goal for global variables should be to avoid polluting the global scope. If needed, they should be used with scrutiny as their use leads to harder to maintain programs and possible undesired side-effects as they are changed throughout the life of the program.

      The beauty of the approach I have shown above is that window variables are kept in scope of the createWindow function. Any data the window needs can be passed into the function at the time of instantiation avoiding the need to use global variables.

      — commented November 19th 2011 by Scott Mason

    • For data persistence between window openings, it depends on your needs. If only needed during the currently running application, I would recommend using a very structured JSON object that is placed in the app namespace.

      Here's an example of using the app namespace to store data:

      Assuming the app namespace is set in app.js as in my example and you want to store a user id:

      // somewhere you do only once (or whenever you want to clear the data)
app.importantData = {};

// somewhere else to store the data
app.importantData.userName = "Arthur Dent";
app.importantData.userId = 42;

      For data that needs to live between running instances you can either use the Properties API to store lightweight data or the SQLite db otherwise. The database gives you larger capacity and search/sort functionality.

      As with any persistent data, just make sure you organize it well so you can expect how it will behave.

      Sounds like you've got a good handle on all of this anyway, but hoping this may help you or others code better.
