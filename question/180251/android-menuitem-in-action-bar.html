  • If you are looking for a way to show a popup menu on a menu item like here: Creating a Popup Menu then you can use this little module: https://github.com/azwan082/nc_popup_android

    menuItem2.addEventListener('click', function(e) {
  var menu = popup.createPopupMenu({
      options: ['User', 'Project', 'Item'],
    view: menuItem2
  });

  menu.addEventListener('click', function(e) {
     Ti.API.info('Selected item: '+e.index);
  });
  menu.show();
});
