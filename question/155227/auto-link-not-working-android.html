    A couple of options;

    var win = Ti.UI.createWindow({
    backgroundColor : 'black'
});

var lbllink1 = Ti.UI.createLabel({
    top : 50,
    color : 'white',
    text : 'Click: http://www.google.com',
    autoLink : Ti.UI.Android.LINKIFY_WEB_URLS
});

var lbllink2 = Ti.UI.createLabel({
    top : 150,
    color : 'white',
    html : "<a href='http://www.google.com'>Click me</a>",
});

lbllink2.addEventListener('click', function() {
    Ti.Platform.openURL('http://www.google.com');
});
win.add(lbllink1);
win.add(lbllink2);
win.open();
