  • Hi May,

    Thanks for your concern.

    You can use Switch or custom button for your Radio Button.

    For custom Radio Button you can use this code.

    var win = Ti.UI.createWindow({
    backgroundColor:'#fff',
});
var checkbox = Ti.UI.createButton({
    title: '',
    top: 10,
    left: 10,
    width: 30,
    height: 30,
    borderColor: '#666',
    borderWidth: 2,
    borderRadius: 3,
    backgroundColor: '#aaa',
    backgroundImage: 'none',
    color: '#fff',
    font:{fontSize: 25, fontWeight: 'bold'},
    value: false //value is a custom property in this case here.
});

//Attach some simple on/off actions
checkbox.on = function() {
    this.backgroundColor = '#007690';
    this.title='\u2713';
    this.value = true;
};

checkbox.off = function() {
    this.backgroundColor = '#aaa';
    this.title='';
    this.value = false;
};

checkbox.addEventListener('click', function(e) {
    if(false == e.source.value) {
        e.source.on();
    } else {
        e.source.off();
    }
});
win.add(checkbox);
win.open();

  • Hi, Titanium however currently does not support it, but planning to support in 4.0. Now you can implement Android native RadioButtons in titanium using this NativeControlsModule for Android.

    Also here is a custom Radio button control. Hope it helps.

    RadioButton.js

    
function createButton(label, gradient) {
    return Ti.UI.createButton({
            title: label,
            borderRadius: "2px",
            borderWidth: "1px",
            borderColor: "black",
            backgroundSelectedColor: '#1EE81E',
            backgroundGradient: gradient,
            left: "0px",
            width:'150dp',
            height: '50dp',
            isClicked:false
    }); 
}

var radioButtonItems = [];

function createRadioButton(_params) {
    var buttons = [];
    var self = Ti.UI.createView({
        layout: 'horizontal',
        width: Ti.UI.SIZE,
        height: Ti.UI.SIZE,
        value: null
    });
    //Greyish default gradient
    self.backgroundGradient = {
        type:'linear',
        colors:[{color:'#d4d4d4',position:0.0},{color:'#c4c4c4',position:0.50},{color:'#b4b4b4',position:1.0}]
    }
    //Selected button - greenish gradient
    self.selectedBackgroundGradient = {
        type:'linear',
        colors:[{color:'#4DE33D',position:0.0},{color:'#17C705',position:0.50},{color:'#0E8102',position:1.0}]
    }
    self.labels = _params.labels || [];

    self.getSelectedButton = function() {
        for (var i=0; i < buttons.length; i++) {
            var btn = buttons[i];
            if (btn.isClicked) {
                return btn; 
            }
        };
        return null;
    }

    //Initiate buttons and add to self view
    for (var i=0; i < self.labels.length; i++) {
        var btnLabel = self.labels[i];
        var ibtn = createButton(btnLabel);

        buttons.push(ibtn);

        //Add event listener for button toggling.
        ibtn.addEventListener('click', function(e) {
            var srcButton = e.source;
            var changeOccured = false;
            for (var j=0; j < buttons.length; j++) {
              var jbtn = buttons[j];
              if (srcButton == jbtn) {              
                if (jbtn.isClicked === false) {
                    changeOccured = true;
                }
                jbtn.isClicked = true;
                jbtn.backgroundGradient = self.selectedBackgroundGradient;
              } else {
                jbtn.backgroundGradient = self.backgroundGradient;
                jbtn.isClicked = false;
              }
            };

            if (changeOccured === true) {
                self.value = self.getSelectedButton().title;
                self.fireEvent('change', self);
            }
        });

        self.add(ibtn);
    };


    return self;
}
exports.createRadioButton = createRadioButton;

    app.js

    
var RadioButton = require('ui/common/RadioButton');

    //create window
    var self = Ti.UI.createWindow({
        backgroundColor:'#ffffff',
        navBarHidden:true
    });

    //Create RadioButton
    var radioButton = new RadioButton.createRadioButton({
        labels: ['Yes', 'No', 'Maybe', 'For sure!']
    });
    radioButton.addEventListener('change', function(e) {
        Ti.API.debug("Received RadioButton change event");
        Ti.API.debug("Selected button: " + JSON.stringify(e.getSelectedButton(), null, 4))
    });
    self.add(radioButton);
    self.open();
