  • Try with scale greater then 1 if you want to increase height/weight:

    var win = Ti.UI.createWindow();

var transformPicker = Titanium.UI.create2DMatrix().scale(1.1);
var picker = Ti.UI.createPicker({
    backgroundColor : '#ff9900',
    top : 0,
    transform : transformPicker
});

var data = [];
data[0] = Ti.UI.createPickerRow({
    title : 'Bananas'
});
data[1] = Ti.UI.createPickerRow({
    title : 'Strawberries'
});
data[2] = Ti.UI.createPickerRow({
    title : 'Mangos'
});
data[3] = Ti.UI.createPickerRow({
    title : 'Grapes'
});

picker.add(data);
picker.selectionIndicator = true;

win.add(picker);
win.open();
