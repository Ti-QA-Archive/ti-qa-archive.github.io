  • var ingredentTotal = Ti.UI.createLabel({
            color:'#222',
            font:{
        fontSize:Titanium.Platform.displayCaps.platformWidth*10/100,
        fontWeight:'bold', 
            fontFamily:'Arial'
         },
        textAlign:Ti.UI.TEXT_ALIGNMENT_CENTER,
            left:photo.width + (boxIngredients.width/2),
            top: photo.height /2  ,
        width:Ti.UI.SIZE,
            clickName:'comment',
            verticalAlign: Ti.UI.TEXT_VERTICAL_ALIGNMENT_CENTER,
            text: results.matches[c].ingredients.length
        });
