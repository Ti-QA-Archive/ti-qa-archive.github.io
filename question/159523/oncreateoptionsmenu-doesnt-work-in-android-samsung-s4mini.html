onCreateOptionsMenu doesn't work in Android Samsung S4mini

In iPhone i create a tabBar and in Android i create the menu.

if(Ti.Platform.name == "android" ){
    var activity = Ti.Android.currentActivity;
    //var activity = Ti.activity;
    activity.invalidateOptionsMenu();

        alert('Menu created');
        var menu = e.menu;
        var home_menu = menu.add({ title: "Home" });
        home_menu.setIcon('home_menu_pic.png');
        home_menu.addEventListener("click", function(e) {
            home.open();
        });

}
