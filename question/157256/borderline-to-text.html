  • A label with borderColor and textColor

    var lbl = Titanium.UI.createLabel({

    text : 'This is label',
top : 50,
left : 10,
borderColor : 'green',
borderWidth : 1,
width : 100,
height : 30,
color : 'maroon'

    });
