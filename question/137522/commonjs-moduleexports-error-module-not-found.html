CommonJS module.exports error: module not found

I am having a problem with using CommonJS. I get this error when running in the iPhone simulator:

[ERROR] Script Error = {
    expressionBeginOffset = 368;
    expressionCaretOffset = 374;
    expressionEndOffset = 374;
    line = 26;
    message = "Can't find variable: module";
    name = ReferenceError;
    sourceId = 228689432;
}

The happens in all the predefined templates when creating a project, but here is an example from my project.

//listEvents.js file

function listEvents() {
        this.win = Ti.UI.createWindow({
            backgroundColor: 'green'
        })

        var label = Ti.UI.createLabel({
            text:'Hello World',
            color: 'white',
            height: 'auto'
        })

        this.win.add(label);

    }

listEvents.prototype.open = function() {
    this.win.open();    
}

module.exports = listEvents;

//app.js

var theapp = {};
theapp.ui = {};

theapp.ui.listEvents = require('ui/shared/listEvents');

theapp.ui.theEventsList = new theapp.ui.listEvents();
theapp.ui.theEventsList.open();

The line mentioned in the error is at:

module.exports = listEvents;

I tried cleaning the project, but that didn't help. Do I need to include or link something to pull in 'module'?

  • I will say that I've never tried the module.exports = composition, but the exports.variable = composition works well. Also, while you are on iOS, begin using a leading slash require(&#39;&#x2F;ui&#x2F;shared&#x2F;listEvents&#39;) so if you move to Android it doesn't bite you later.
