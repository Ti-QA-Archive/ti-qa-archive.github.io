window does not display the components in it

Hi I have a button that call to create a window with a tableView but the tableView doesn't show

here is the code

var listButton = Titanium.UI.createButton({
    title:'Listar',
    height:50,
    width:'auto',
    top:10
});
listButton.addEventListener('click',function(e){
    var win2 = Ti.UI.createWindow({
        title:'Lista Tigo Restaurants',
        layout:'vertical',
        backgroundColor:'#fff',
        url:'list_restaurant.js'
    });
    var tab = Titanium.UI.currentTab;

       tab.open(win2, {animated:true});
});
win.rightNavButton=listButton;

here is the code for the tableView

var win2 = Titanium.UI.currentWindow;
var data = [];
var xmlstr = "<?xml version=\"1.0\" encoding=\"utf-8\"?>"+
"<root>"+
    "<item>"+
        "<name>Brazilian Coffee</name>"+
        "<dir>Av. Ballivian (El Prado)  # 537 entre Mexico y Reza</dir>"+        "<url>images/brazilian_coffee.png</url>"+
    "</item>"+
"</root>";        
var doc =Ti.XML.parseString(xmlstr);
var x = 0;
var items = doc.getElementsByTagName("item");
for (var c=0;c<items.length;c++)
{
    var item = items.item(c);
    var name = item.getElementsByTagName("name").item(0).text;
    var dir = item.getElementsByTagName("dir").item(0).text;
    var url = item.getElementsByTagName("url").item(0).text;
    var row = Ti.UI.createTableViewRow({height:100});
    var label = Ti.UI.createLabel({text:name});
    row.add(label);
    var label2 = Ti.UI.createLabel({text:dir});
    row.add(label2);
    var img;
    img = Ti.UI.createImageView({image:url});
    row.add(img);
    data[x++] = row;
}
var tableview2 = Titanium.UI.createTableView({data:data});
win2.add(tableview2);
