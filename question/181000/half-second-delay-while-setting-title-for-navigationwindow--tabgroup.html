Half second delay while setting title for NavigationWindow / TabGroup

I have noticed a strange delay when setting the title property for a window contained into a NavigationWindow or TabGroup. The title appears after around half second after the opening of another window inside it.
Please find here a test code that demonstrates it:

var win = Ti.UI.createWindow({
    backgroundColor: "white"
});

var navWin = Ti.UI.iOS.createNavigationWindow({
    window: win
});


var win2 = Ti.UI.createWindow({
    backgroundColor: "green",
    title: "ciao"
});

var btn = Ti.UI.createButton({
    title: "ciao"
});

win.add(btn);
btn.addEventListener("click", function() {
    navWin.openWindow(win2);
});


navWin.open();

I am using TiStudio on Mac OS X with SDK 3.5.0.

  • Try the below code and let me know if you are still facing the issue. What I did is, I used the titleControl instead of title property of window.

    
var win = Ti.UI.createWindow({
    backgroundColor: "white"
});

var navWin = Ti.UI.iOS.createNavigationWindow({
    window: win
});

var label = Ti.UI.createLabel({
    text: "ciao"
});

var win2 = Ti.UI.createWindow({
    backgroundColor: "green",
    titleControl: label,
});

var btn = Ti.UI.createButton({
    title: "ciao"
});

win.add(btn);
btn.addEventListener("click", function() {
    navWin.openWindow(win2);
});

navWin.open();

    Note - I have tested on iPhone 5S simulator and its working fine at my end.
