  • Hi Raffaele,

    Try this code.

    var testImgView = Ti.UI.createImageView({
    height : 320,
    width : 240,
    image : 'imagePath',//event.media, //
    top : 50
});
win.add(testImgView);

var imgName = 'first.png';
var gf = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, imgName);
var blb = testImgView.toImage();
gf.write(blb);
