  • Hi

    Is this what you are looking for? This will rotate the device 180 degrees depending on which button you press. I guess you want this so you can have someone at either end of the device and the swap is so they can take their turn to perform whatever operation and then press again so it goes back to the other party.

    Same process as before; create a blank project remove the contents of the app.js and past this code in its place.

    Ti.UI.setBackgroundColor('#000');

var tab9;
(function () {
    var win = Ti.UI.createWindow({  
        backgroundColor: '#fff',
        height: Ti.UI.FILL,
        title: 'Text',
        width: Ti.UI.FILL
    });
    var txt = Ti.UI.createTextField({
        borderColor: '#666',
        height: 50,
        left: 20,
        paddingLeft: 10,
        paddingRight: 10,
        right: 20,
        top: 50,
        width: Ti.UI.FILL
    });
    //txt.transform = Ti.UI.create2DMatrix().rotate(-180);
    win.add(txt);
    var btnTestL = Ti.UI.createButton({
        title: 'Normal'
    });
    win.setLeftNavButton(btnTestL);
    btnTestL.addEventListener('click', function (e) {
        win.setOrientationModes([
            Ti.UI.PORTRAIT
        ]);
    });
    var btnTestR = Ti.UI.createButton({
        title: 'Upside Down'
    });
    win.setRightNavButton(btnTestR);
    btnTestR.addEventListener('click', function (e) {
        win.setOrientationModes([
            Ti.UI.UPSIDE_PORTRAIT
        ]);
    });
    tab9 = Ti.UI.createTab({  
        icon: 'KS_nav_views.png',
        title: 'Text',
        window: win
    });
})();
var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab9);
tabGroup.open();

    Error free
