the problem about release() of Media.createSound

I am making application for Android.

This is my source which plays mp3 file very simply.

    function playsound(){

        if (Sound0 != undefined){Sound0.release();}
        if (Sound1 != undefined){Sound1.release();}

        Sound0 = Titanium.Media.createSound({
            url : PfSoundUrl[0]
        });
        Sound1 = Titanium.Media.createSound({
            url : PfSoundUrl[1]
        });
        Sound0.play();
        Sound1.play();
    }
