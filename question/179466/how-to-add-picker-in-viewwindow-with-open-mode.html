  • i don't know what you exactly need . please look this code. which is dynamically add picker into an open window
    var picker = Ti.UI.createPicker({
    top:50
    });

    var data = [];
    data[0]=Ti.UI.createPickerRow({title:'Bananas'});
    data[1]=Ti.UI.createPickerRow({title:'Strawberries'});
    data[2]=Ti.UI.createPickerRow({title:'Mangos'});
    data[3]=Ti.UI.createPickerRow({title:'Grapes'});

    picker.add(data);
    picker.selectionIndicator = true;

    $.Mainwindow.add(picker); // replace Mainwindow with id of your parent window

    picker.setSelectedRow(0, 2, false);

    Please check the link for more details.
    http://docs.appcelerator.com/titanium/3.0/#!/api/Titanium.UI.Picker
