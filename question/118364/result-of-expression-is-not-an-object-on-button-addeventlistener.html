Result of expression is not an object On button addEventListener

Hi,

I have in the head of my application a button with "Start"/"Stop" titles which with the user can start or stop the app process.
When I'm trying to launch the app, I get this error message:

[ERROR] Script Error = Result of expression 'startStopButton' [undefined] is not an object.

startStopButton.addEventListener('click', function() {

    var appStatus = Ti.App.Properties.getString('appStatus','');

    if (appStatus == "active") {
        //so, the user clicked on "STOP"...
        stopProcess();
        return;
    }

    //else, appStatus = "inactive", so the user clicked on "START"...
    Ti.App.Properties.setString('appStatus','active');
    startStopButton.title = "Stop";

    startProcess();  //this function starts a new process or load an active one

    reloadData();    //this function refresh data every 0.5 second

});
