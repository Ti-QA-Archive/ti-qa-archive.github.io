    Hi,

    Please check the following code. It works well in my emulator 

    var win = Ti.UI.createWindow({
    backgroundColor: '#000', 
    fullscreen: true,
    layout:'vertical'
});
var createHeaderView = function (args) {
    var headerView = Ti.UI.createView({
        top:0,
        left:0,
        height:72,
        width:768,
        backgroundColor:'white'
    });
    var text1 = Ti.UI.createLabel({
        text:args.title,
        top: 12,
        left: 20,
        height: 45,
        color:'#666',
        font: {
            fontSize: 30,
        }
    });
    headerView.add(text1);
    return headerView;
};

     //----------
var sectionOne = Ti.UI.createTableViewSection({
        headerView : createHeaderView({title: 'Header text'})
});

var theBalanceDataOne = [
    {title: 'text1'},
    {title: 'text2'},
    {title: 'text3'}
];
var balanceData = [];

for (var i = 0; i < theBalanceDataOne.length; i++){

    var tableRowView = Ti.UI.createView({
        top:0,
        left:10,
        right: 10,
        //height: Ti.Platform.displayCaps.platformHeight*0.09//'13.7%',
    });

    var text1 = Ti.UI.createLabel({
        text: theBalanceDataOne[i].title,// + i,
        top: 30,//'25%',
        left: 20,//'3%',
        height: 32,
        color: '#666',//#c0c0be
        font:{
            fontSize: 30,
        }
    });
    tableRowView.add(text1);

    var row = Ti.UI.createTableViewRow({
        height:106
    });
    row.add(tableRowView);
    sectionOne.add(row);
 }

balanceData.push(sectionOne);

var tableViewBalance = Ti.UI.createTableView({
    top: 30,
    height: Ti.UI.FILL,
    data:balanceData,
    backgroundColor: "#fff"
});
win.add(tableViewBalance);
win.open();

  • Hi

    As you have not provided any code I cannot see if the problem is code related. But there is one likely cause that you may not be aware of. Section Headers and Footers in Android require an explicitly set height value, so no Ti.UI.SIZE - has to be a number or other dimension value.

    As a test set the heights to something generic - just so you can see if it changes things or not.
