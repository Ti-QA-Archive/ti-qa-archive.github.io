Toolbar button does not work on return from window

I have a toolbar button which opens a window (as part of a tabGroup). The button works fine the first time you click on it. On return from the opened window though, it does not work at all (the click event is not activated).

I have just tried something very similar in the default "Hello World" app and the same thing happens. Can anyone see why this button only works once?

This is the "Hello World" app.js with my code added right at the end:



Titanium.UI.setBackgroundColor('#000');

var tabGroup = Titanium.UI.createTabGroup();

var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});

var label1 = Titanium.UI.createLabel({
    color:'#999',
    text:'I am Window 1',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
});

win1.add(label1);

var win2 = Titanium.UI.createWindow({  
    title:'Tab 2',
    backgroundColor:'#fff'
});
var tab2 = Titanium.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Tab 2',
    window:win2
});

var label2 = Titanium.UI.createLabel({
    color:'#999',
    text:'I am Window 2',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
});

win2.add(label2);

tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);  

// MY ADDED CODE

var myToolBar = Titanium.UI.createTabbedBar({
            labels : [{
                title : 'Open Window 2',
                enabled : false
            }]
});

var flexSpace = Titanium.UI.createButton({
        systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});


win1.setToolbar([flexSpace, myToolBar, flexSpace]);

myToolBar.addEventListener('click', function(e) {
tabGroup.setActiveTab(1);
tabGroup.open()
});    

// open tab group
tabGroup.open();

    Since you have a single button, you'll want to use a button bar instead of a tabbed bar, as the tabbed bar will remain 'pressed' after you click it, with the assumption that another button on the bar will eventually be pressed, and make the first button 'unpressed'. Since the button remains pressed, it didn't respond to future click events.

    Try this:

    var myToolBar = Titanium.UI.createButtonBar({
    labels : [{
        title : 'Open Window 2'
    }]
});

var flexSpace = Titanium.UI.createButton({
    systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

win1.setToolbar([flexSpace, myToolBar, flexSpace]);

myToolBar.addEventListener('click', function(e) {
    tabGroup.setActiveTab(1);

});

    I also removed the 'enabled' property from the label and the additional tabGroup.open() from the event listener.
