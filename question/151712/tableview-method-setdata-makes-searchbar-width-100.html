TableView method SetData makes searchBar width 100%

Hello,

i'm trying to add a button (a lbl that acts like a button) at the right of a searchBar. Similar to this post: http://stackoverflow.com/questions/13974616/customize-cancel-button-of-searchbar-in-titanium

It works well if I only have the searchBar and the button but if I add a tableview and then use the setData method (in order to see the results of the search), suddenly the searchBar occupies de 100% of the width windows and left the button lblCancel in the top of the search bar.

Is there a way to have in the same line the searchbar and the button, and in the bottom of this the tableview or it's a bug?

Here is the code:



        dataRows = [];

        winSearch = Ti.UI.createWindow({
            left : 0,
            width: percentWidth(100),
            zIndex: 1,
            layout: 'vertical'
        });

//the view where I put the searchbar and the button
       searchHolder = Ti.UI.createView({
          height   : 55,
          width    : percentWidth(100),
          top      : 0,
          left: 0,
          backgroundColor: '#a3e',
          layout   : 'horizontal'
        });

//the searchBar that ocuppies the 65% of the window
        searchBarRows = Titanium.UI.createSearchBar({
           barColor: '#ddd',
           height: percentHeight(12),
           hintText: 'search',
           width: percentWidth(65),
           showCancel: false,
           top: 0,
            left: 0
        });

//the button that ocuppies the 35% of the window
        lblCancel = Ti.UI.createLabel({
            height   : 45,
              width    : percentWidth(35),
              backgroundColor: '#1a3',
            text: 'cancel'
        });

        lblCancel.addEventListener('click',function(){
            winSearch.close();
        });

        tbl = Titanium.UI.createTableView({
               search: searchBarRows,
               scrollable: true,
            backgroundColor: '#fff',
               top:0
        });    


        for(contI = 0; contI < data.length; contI = contI + 1){
            row = Ti.UI.createTableViewRow({
                  height: percentHeight(10),
                  title: data[contI].name,
                  hasDetail:true
            });

            lblRow = Ti.UI.createLabel({
                text: data[contI].name,
            });

            row.add(lblRow);
            dataRows.push(row);
        } 


        searchHolder.add(searchBarRows);
        searchHolder.add(lblCancel);
        win.add(searchHolder);

        tbl.setData(dataRows); //the sentence that make the searchbar ocuppy all the width of the window
        win.add(tbl);

Thanks.
