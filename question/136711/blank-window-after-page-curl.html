Blank Window after page curl

For some reason, after I click the "Close Hint" button in view2, the window that it goes back to is blank - I can't seem to figure out why view1 isn't showing up again:


var win = Titanium.UI.currentWindow;

win.backgroundColor = '#FFF';

var view2 = Ti.UI.createView({
    backgroundColor:'#FFF'
});
win.add(view2);

view = Ti.UI.createView({
    backgroundColor:'#fff'
});
win.add(view);

var b1 = Titanium.UI.createButton({
    title:'OK as is',
    width:100,
    height:50,
    left:10,
    top:225,
});

b1.addEventListener('click', function()
{
    l2.text = "You were able to click on the box. Good!";
});

win.add(b1);


var b2 = Titanium.UI.createButton({
    title:'Not sure, need a hint',
    width:180,
    height:50,
    left:125,
    top:225,
});

b2.addEventListener('click', function()
{
    // create close button for our window
    var b = Ti.UI.createButton({title:'Close Hint',width:200,height:40});
    b.addEventListener('click',function()
    {
        win.animate({view:view,transition:Ti.UI.iPhone.AnimationStyle.CURL_DOWN});
    });
    view2.add(b);

    // transition to view
    win.animate({view:view2, transition:Ti.UI.iPhone.AnimationStyle.CURL_UP});
});

win.add(b2);

var b3 = Titanium.UI.createButton({
    title:'Wrong',
    width:100,
    height:50,
    left:10,
    top:285,
});

b3.addEventListener('click', function()
{
    l2.text = "You were able to click on the box. Good!";
});

win.add(b3);
