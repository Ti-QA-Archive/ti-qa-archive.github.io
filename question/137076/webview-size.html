    Hi Raffaele,

    can you just see this demo in ipad. It is dynamically changes the size of its object and it does not require to change size when orientation change, try to make your view like dynamic rather than static, it will help full to you.

    if this demo can not helps you then advisable that put some code here so we can help you to improve reliability.

    var a = Ti.UI.createView({
    top : 10,
    bottom : 10,
    left : 10,
    right : 10,
    backgroundColor : 'gray'
});
var b = Ti.UI.createView({
    top : 10,
    bottom : "50%",
    left : 10,
    right : 10,
    backgroundColor : 'white'
});
var c = Ti.UI.createView({
    height : 200,
    width : 200,
    backgroundColor : 'blue'
});
win.add(a);
a.add(b);
b.add(c);

  • Try something like this:

    Ti.Gesture.addEventListener('orientationchange', function(e){
    if((Titanium.Gesture.orientation == Titanium.UI.LANDSCAPE_LEFT)||(Titanium.Gesture.orientation == Titanium.UI.LANDSCAPE_RIGHT))
    {       
        webview.left ='350dp';
        webview.top  = '-75dp';
        webview.left = '350dp'
        webview.top = '5dp';

    }
    else
    {        
        webview.left ='30dp';
        webview.top  = '55dp';
        webview.left = '30dp'
        webview.top = '5dp';
     }
});

  • Hello Raffaele,

    See this…

    Titanium.Gesture.addEventListener('orientationchange', function(e) {
    switch (Titanium.Gesture.orientation) {

        // For LandScap.......

        case Titanium.UI.LANDSCAPE_LEFT:
        case Titanium.UI.LANDSCAPE_RIGHT:

                // Change Size

            break;

        // For Portrait.......

        case Titanium.UI.PORTRAIT:
        case Titanium.UI.UPSIDE_PORTRAIT:

            // Change Size

            break;
    }
});

Titanium.Gesture.fireEvent('orientationchange');
