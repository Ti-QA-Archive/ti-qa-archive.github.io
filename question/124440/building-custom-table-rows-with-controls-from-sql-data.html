    following adam's approach

    var rightButton = Titanium.UI.createButton({
   width : 50,
   height : 30,
   right : 10,
   title : 'Add',
   _row : row // add row as property of button
});

    in event handler

    function rightButtonClickHandler(e)
{
    Ti.API.info('button click index = '+e.type);
    var row = e.source._row;
};
