    • you can use window's postlayout event to animate or you can animate view in open event after some interval to do that use setTimeout method.

      var win = Titanium.UI.createWindow({
    title : "Window",
    exitOnClose : true,
    backgroundColor : "red"
});

var v1 = Titanium.UI.createView({
    top : 180,
    backgroundColor : 'yellow',
    height : 160,
    width : 80,
});

win.add(v1);

function Animateview() {
    v1.animate({
        height : 220,
        duration : 2000
    });
}

win.addEventListener("open", function(e) {

    setTimeout(function(e) {
        Animateview();
    }, 1000);

});

win.open();
