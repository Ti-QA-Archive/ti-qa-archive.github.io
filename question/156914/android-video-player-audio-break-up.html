Android Video Player audio break-up

I am trying to play videos uploaded by my client (a TV broadcaster) to the servers of livestream.com which are encoded with Livestream's Procaster encoder which produces a number of 'mobile' friendly and other streams.

And I am playing it using the Android version of Appcelerator's VideoPlayer. But the sound is unusable, being very 'choppy'. So something is seriously amiss here. I presume it's a bitrate issue, but there appears to be no way of checking it or changing it within the VideoPlayer.

My test stream is the official Android one (the 'androidUrl') at rtsp://mobilestr3.livestream.com/onDemand/ls/gbctelevision/flv_288b74f2-5c9b-401d-9526-1288ffcdbcc3 but plays as described on my Android test tablet (An Arnova 7G3). The sound does play without breaking up on the Emulator - but on that there's no video picture! The 'iPhoneUrl' version of the same stream plays as it should on iOS.

If any kind soul with video streaming knowledge would care to test it with the standard Appcelerator VideoPlayer and see if they get the same result on another Android device I would be very grateful as I am left scratching my head here.

It might be worth adding that livestream.com's 'progressiveUrl' version of the same video clip plays just fine on the Android VideoPlayer, but ideally it's the 'androidUrl' one I'd need to use.

Thanks.

  • Just a warning: you're going to have a lot of issues with different devices on older versions of the Android OS when it comes to using H.264 video.

    Hard to say what this particular issue is, but livestream.com's servers are pretty good.

    Looking at their stream url, I'm guessing they're re-encoding or re-streaming an FLV (Flash video) file, which could be part of the problem.

    • OK. this worked on my Samsung Galaxy SII with 4.0.3

      Save it as a .js file and use it as the source of a heavyweight fullscreen window:

      
var win = Ti.UI.currentWindow;

Ti.API.info('loading video')

/*
win.orientationModes = [
    Titanium.UI.PORTRAIT,
    Titanium.UI.LANDSCAPE_LEFT,
    Titanium.UI.LANDSCAPE_RIGHT,    
    Titanium.UI.UPSIDE_PORTRAIT
];
*/

var fullScreenMovie = Titanium.Media.createVideoPlayer({
    url:'rtsp://mobilestr3.livestream.com/onDemand/ls/gbctelevision/flv_288b74f2-5c9b-401d-9526-1288ffcdbcc3',
    autoplay:true,
    mediaControlStyle:Titanium.Media.VIDEO_CONTROL_DEFAULT,
    scalingMode:Titanium.Media.VIDEO_SCALING_ASPECT_FIT
});    

win.add(fullScreenMovie);

fullScreenMovie.addEventListener('load',function(e){
    Ti.API.info('VIDEO LOADED..');
    hideIndicator();

});

// autoplay it
fullScreenMovie.play();


showIndicator('HOLD ON, TIGER...','LOADING VIDEO');

var indView, actInd, message, message2;


function showIndicator(txt1,txt2){

    indView = Titanium.UI.createView({
        bubbleParent:false,
        backgroundColor:'transparent',
        top:0,
        left:0,
        right:0,
        bottom:0,
        zIndex:100
    });

    var actIndView  = Titanium.UI.createView({
        backgroundColor:'#000',
        opacity:0.5,
        top:0,
        left:0,
        right:0,
        bottom:0,
        zIndex:1
    });   

    // loading indicator
    actInd = Titanium.UI.createActivityIndicator({
        style:Titanium.UI.ActivityIndicatorStyle.BIG_DARK,
        zIndex:2
    });

    indView.add(actIndView);
    indView.add(actInd);

    message = Titanium.UI.createLabel({
        text:txt1.toUpperCase(),
        color:'#fff',
        font:{fontSize:22,fontWeight:'bold'},
        textAlign:'center',
        bottom:100,
        zIndex:10,
        height:38,
        width:240
    });
    message2 = Titanium.UI.createLabel({
        text:txt2.toUpperCase(),
        color:'#eee',
        font:{fontSize:14,fontWeight:'bold'},
        textAlign:'center',
        bottom:70,
        zIndex:10,
        height:16,
        width:240
    });

    indView.add(message);
    indView.add(message2);
    actInd.show();
    win.add(indView);

}

function hideIndicator(){

    actInd.hide();
    win.remove(indView);
    indView = null;
    message = null;
    message2 = null;
}


win.addEventListener('close',function()
{
    Ti.API.info("window was closed");
});
