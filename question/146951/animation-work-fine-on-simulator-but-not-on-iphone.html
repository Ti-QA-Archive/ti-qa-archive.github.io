Animation work fine on simulator but not on iPhone

Hi,
I use this animation for scrolling a text :

var left = true;
    var animation = Titanium.UI.createAnimation();
    animation.duration = 8000;
    animation.left = '10dp';
    var animationHandler = function() {
        if (left) {
            animation.left = '310dp';
            left = false;
        } else {
            animation.left = '10dp';
            left = true;
        }

        titleLabel.animate(animation);
    };
    animation.addEventListener('complete', animationHandler);
    titleLabel.animate(animation);

SDK 3.0

any idea ?
