var view =  Ti.UI.createView({
    top:200,
    left:150,
    width:40,
    height:40,
    backgroundColor:'#f00'
});


win.add(view);

view.animate({top:0,left:0,duration:2000});

    this should do the trick.

  • If the window is not open yet, you'll need to wait until it opens. You can do this by adding an event listener, for the 'open' event:

    win.add(view1)

win.open();

win.addEventListener('open', function(e){
  view1.animate({top:0,left:0,duration:2000});
});
