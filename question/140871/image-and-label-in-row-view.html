    This code below will give you the base layout for what you are trying to do. You'll need to alter it a bit for your app of course. Style and name the labels, change the icons, add event listeners, etc. I would suggest putting the code into a commonjs module to make all this easier and then require() it from your home window. There is plenty that you'll need to do to make it look and behave exactly how you want it to, but this should give you the base to start from.

    var win = Ti.UI.createWindow({
    backgroundColor:'#fff'
});

var container = Ti.UI.createView({
    top:'50%',
    left:20,
    right:20,
    height:Ti.UI.FILL,
    width:Ti.UI.FILL,
    layout:'horizontal',
    backgroundColor:'#f0f0f0'
});

var i = 0, view, img, label;
for (i ; i < 5; i++) {
    view = Ti.UI.createView({
        left:0,
        top:20,
        width:'50%',
        height:Ti.UI.SIZE,
        layout:'horizontal'
    });

    img = Ti.UI.createImageView({
        image:'KS_nav_ui.png',
    });
    view.add(img);

    label = Ti.UI.createLabel({
        text:'Label',
        font:{fontSize:14},
        left:20
    });
    view.add(label);

    container.add(view);
}

win.add(container);

win.open();

  • Im pretty sure that the issue is going to lie in the code of your dash creation, and you haven't provided that. One basic idea though is that you're going to want to create a parent view for each label+icon. On the click of that view, you'll want to highlight or darken the icon image. Another approach is to simply create custom button images for each one and set the button background to that image. Then, when you click on the button, the darkening will happen automagically.

function newIcon(_x){
    return Ti.UI.createView({

        left: _x,
        top: 10,
        height: 40,
        width: 40,
        backgroundColor: '#fff'
    });
}

function newViewRow(_y){
    return Ti.UI.createView({
        layout: 'horizontal',
        left: 0,
        top: _y,
        height: 60,
        width: '100%',
        backgroundColor: '#000',
        opacity: 0.8
    });
}
