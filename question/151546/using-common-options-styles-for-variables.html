      function createLabel(left, top, value)
     var newLabel = Ti.UI.createLabel({

            width: "50%",
            backgroundColor: "blue",
            height: 20,
            text: value,
            textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER
        });

    return newLabel;

  • You can achieve this by creating a simple function that will merge two JavaScript objects. So you could declare a simple dictionary object containing your default attributes and then merge this object with the ones you want to add as extra parameters.

    ###Here's an example

    function mergeObjects(obj1, obj2) {
    var key, newObj = {};

    for (key in obj1) {
        newObj[key] = obj1[key];
    }
    for (key in obj2) {
        newObj[key] = obj2[key];
    }

    return newObj;
}

var defaultParams = {
    height: 0,
    somekey: "customAttribute"
};

var customParams = {
    height: 30,
    width: 30
};

var mergedParams = mergeObjects(defaultParams, customParams);

Ti.API.info("mergedParams: "+JSON.stringify(mergedParams));
    // --> {"height":30, "somekey":"customAttribute", "width":30}
Ti.API.info("defaultParams: "+JSON.stringify(defaultParams));
    // --> {"height":0, "somekey":"customAttribute"}

    Hope this helps
