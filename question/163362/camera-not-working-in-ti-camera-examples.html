  • btnTakePhoto.addEventListener('click', function(e) {Ti.Media.showCamera({
            success : function(event) {
                var image = event.media;

                var imageView = Ti.UI.createImageView({
                    image: image //just an example of use for event.media
                });
            },
            cancel : function() {
            },
            error : function(error) {
                var alertDialogBox = Titanium.UI.createAlertDialog({
                    title : 'Camera'
                });
                if (error.code == Titanium.Media.NO_CAMERA) {
                    alertDialogBox.setMessage(L('no_camera'));
                } else {
                    alertDialogBox.setMessage('Unexpected error: ' + error.code);
                }
                alertDialogBox.show();
            },
            autohide : false,
            saveToPhotoGallery : true,
            showControls : true,
            allowImageEditing : true,
            allowEditing : true,
            mediaTypes : ['public.image']
        });
    });
