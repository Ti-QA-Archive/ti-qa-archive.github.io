    Hi,Steven Marshall

    Try this code it helps you to solve your problem

    var main_win = TI.UI.createWindow({backgroundColor: 'red'});

var some_view = Ti.UI.createView({width:100,height:100,backgroundColor:'yellow'});
     main_win.add(some_view);

var new_view = Ti.UI.createView({width:100,height:100,backgroundColor:'blue',visible:false});
    main_win.add(new_view);

some_btn.addEventListener('click',function(e) {
    new_view.visible = true;
    main_win.remove(some_view); 
});
