Ti.App.addEventListener('navi:scroll', function(_e) {
    if (STATE.lastpinindex) {
        var lastbig = STATE.bigannotations[STATE.lastpinindex];
        var last = STATE.annotations[STATE.lastpinindex];
    }
    var currentbig = STATE.bigannotations[_e.ndx];
    var current    = STATE.annotations[_e.ndx];
    singletonMap.removeAnnotation(current);
    if (STATE.lastpinindex)  singletonMap.removeAnnotation(lastbig);
    singletonMap.addAnnotation(currentbig);
    if (STATE.lastpinindex) singletonMap.addAnnotation(last);
    STATE.lastpinindex = _e.ndx;
});
