  • Can you post some example code? I ran a quick test and it worked just fine.

    var win = Ti.UI.createWindow({
    backgroundColor: '#fff'
    });

    var tableView = Ti.UI.createTableView({
    backgroundColor: '#fff'
    });
    win.add(tableView);

    var tableViewRow = Ti.UI.createTableViewRow({
    backgroundColor: 'red'
    });

    tableViewRow.add(Ti.UI.createLabel({
    text: 'I am a label',
    backgroundColor: 'blue',
    color: '#fff',
    left: 5
    }));

    tableView.appendRow(tableViewRow);

    win.open();
