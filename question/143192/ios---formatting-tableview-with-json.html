iOS - Formatting TableView with JSON

I have this code that I modified from the sample projects. I have figured out that I can do stuff like "leftImage" and "color" in the JSON and the TableView will be changed accordingly. My question is where can I find a list of the other properties, I can set in the JSON for the table view? 

function HomeView() {
    //create object instance, parasitic subclass of Observable
    var self = Ti.UI.createView({
        backgroundColor:'white'
    });

    //some dummy data for our table view
    var tableData = [
        {title:'Exhibitors', hasChild:true, color: '#000', leftImage: '/images/group.png'},
        {title:'Schedule of Events', hasChild:true, color: '#000', leftImage: '/images/calendar.png'},
        {title:'Floor Map', hasChild:true, color: '#000', leftImage: '/images/map.png'},
        {title:'News', hasChild:true, color: '#000', leftImage: '/images/newspaper.png'},

    ];

    var table = Ti.UI.createTableView({
        data:tableData


    });



    self.add(table);
    //table.separatorColor
    //add behavior
    table.addEventListener('click', function(e) {
        self.fireEvent('itemSelected', {
            name:e.rowData.title
        });
    });

    return self;
};

module.exports = HomeView;
