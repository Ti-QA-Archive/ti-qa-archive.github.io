   color: '#333'

        containerWindow.animate(animateLeft);
        isToggled = true;

        containerWindow.animate(animateRight);
        isToggled = false;
    }
});

function switchPages(oldPage, newPage) {
    if( !_.isEqual(oldPage, newPage) ) {
        containerWindow.add(newPage);
        containerWindow.remove(oldPage);
        setTimeout(function() {
            containerWindow.animate(animateRight);
            isToggled = false;
        }, 500);
        currentPage = newPage;
    } else {
        Ti.API.info('page already open');
    }
    Ti.API.info('oldPage: ' + oldPage.window.title + ', newPage: ' + newPage.window.title);
}

/****************************************************************
 * Facebook style menu

var leftMenu = Ti.UI.createWindow({
    top:   0,

for ( var i=0; i<3; i++ ){
