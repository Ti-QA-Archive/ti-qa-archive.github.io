  • Hi,
    You need to add view not the virtualScroller object.

    Double check the documentation.

    var VirtualScroller = require('ui/common/VirtualScroller');
var virtualScroller = VirtualScroller({
    itemCount: 10,
    start: 0,
    getView: function(i) {
        return Titanium.UI.createLabel({
            width: Titanium.UI.FILL,
            height: Titanium.UI.FILL,
            text: "This is item " + (i + 1)
        });
    },
    isInfinite: false
});?
window.add(virtualScroller.view);

    Note: window.add(virtualScroller.view);
