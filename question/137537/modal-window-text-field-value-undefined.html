    Hello Jared,

    See this…

    var win = Ti.UI.currentWindow;

var moreInformationButton1 = Ti.UI.createButton({
    style : Titanium.UI.iPhone.SystemButton.DISCLOSURE,
    top : 75
});

win.add(moreInformationButton1);

moreInformationButton1.addEventListener('click', function() {

    var signerInformationWindow1 = Ti.UI.createWindow({
        url : 'ui/SignerModal.js',
        title : 'More Signer Information',
        backgroundColor : '#FFFFFF'
    });

    signerInformationWindow1.open({
        modal : true
    });
});

    SignerModal.js

    var signerModal = Ti.UI.currentWindow;

var containerScrollView = Ti.UI.createScrollView({
    top : 0,
    left : 0,
    height : 480,
    width : 320,
    contentHeight : 'auto',
    layout : 'vertical',

});
signerModal.add(containerScrollView);

var _marginLeft = (signerModal.getWidth() - 668) / 2;

var signerInformationView = Ti.UI.createView({
    top : 0,
    left : _marginLeft,
    left : 0,
    width : 668,
    height : 450,
});
containerScrollView.add(signerInformationView)

var firstNameLabel = Ti.UI.createLabel({
    text : 'First Name',
    width : 'auto',
    font : {
        fontSize : 14
    },
    color : '#000000',
    top : 55,
    left : 25
});

signerInformationView.add(firstNameLabel);

var firstNameText = Ti.UI.createTextField({
    width : 250,
    height : 30,
    top : 75,
    left : 25,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
    returnKeyType : Ti.UI.RETURNKEY_DONE,
});

signerInformationView.add(firstNameText);

var cancel = Ti.UI.createButton({
    title : 'Cancel',
    style : Ti.UI.iPhone.SystemButtonStyle.PLAIN
});

var done = Ti.UI.createButton({
    title : "Done",
    style : Ti.UI.iPhone.SystemButtonStyle.PLAIN
});

signerModal.setLeftNavButton(cancel);
signerModal.setRightNavButton(done);
signerModal.add(signerInformationView);

cancel.addEventListener('click', function() {
    signerModal.close();
});

done.addEventListener('click', function() {
    Ti.API.log('INFO', 'signerInformationWindow1.firstNameText is: ' + firstNameText.value);
    signerModal.close();
});
