  • Old question, but for the sake of others who find it…

    showProgress: function (message) {

        if (!utils.progress) {
            utils.progress = new ActivityIndicator({ height: 50, width: 10, message: utils.isAndroid ? message : "" });
            if (!utils.isAndroid) {
                tt.ui.Win.setRightNavButton(utils.progress);
            }
        }
        else if (utils.isAndroid) {
            utils.progress.message = message;
        }
        if (!utils.isAndroid)
            utils.progress.style = Titanium.UI.iPhone.ActivityIndicatorStyle.BIG; //PLAIN, DARK, BIG
        utils.progress.show();

    },
