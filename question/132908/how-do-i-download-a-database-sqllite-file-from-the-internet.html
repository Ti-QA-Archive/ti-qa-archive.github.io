  • Try something like this : 

    
// Downloading remote file and saving it to your application data directory. 

var remote_url = 'http://some.url.to.a.file';

// set up where you need the file to be saved and give it a file name.
var f = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory+'/the_name_of_the.file');

// delete a previous version
if(f.exists()){
    f.deleteFile();
    return;
}

var xhr = Titanium.Network.createHTTPClient();
xhr.setTimeout(10000);
xhr.onload = function()
{
    // write the downloaded file in to the previously create File object
    f.write(this.responseData);
};
xhr.onerror = function(e)
{
    //Ti.API.info('XHR Error ' + e.error);
};    

// download the file 
xhr.open('GET',remote_url);
xhr.file = f;
xhr.send();
