  • I'm developing apps for both iOS and Android, and I'm using this function to get the element size the same on different screen sizes. This function is only for Android devices. Not sure how this works on Android tables, but I think you can give it a try. Give scaleElements function value from 0-320 for width, because 320 should mean that the element size is the same as the device width (but I'm not 100% sure about that). 

    function scaleElements(number){
    var calculatedNumber = 0;

    if (!Ti.App.iOS) {
        var dpi = Ti.Platform.displayCaps.dpi;

        switch(dpi) {
            case 120: {
                calculatedNumber = number * 0.75;
                break;
            }
            case 160: {
                calculatedNumber = number;
                break;
            }
            case 213: {
                calculatedNumber = number * 1.33;
                break;
            }
            case 240: {
                calculatedNumber = number * 1.5;
                break;
            }
            case 320: {
                calculatedNumber = number * 2;
                break;
            }
            case 480: {
                calculatedNumber = number * 3;
                break;
            }

        }

    };
    return calculatedNumber;
};
