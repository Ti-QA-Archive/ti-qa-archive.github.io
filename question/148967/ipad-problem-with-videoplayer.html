iPad problem with videoplayer

This is all code i have

var win = Ti.UI.createWindow();
win.orientationModes = [Titanium.UI.LANDSCAPE_LEFT,Titanium.UI.LANDSCAPE_RIGHT];
win.backgroundImage = 'bg.png';
win.fullscreen = true;

var mv1 = Ti.UI.createView();
mv1.width = 306;
mv1.height = 168;
mv1.top = 184;
mv1.left = 134;
mv1.backgroundImage = 'mv1.png';
mv1.addEventListener('touchstart',function(){mv1.opacity=0.8;});
mv1.addEventListener('touchend',function(){mv1.opacity=1;});
mv1.addEventListener('click',function(){videowindow('mv1.m4v','Video 1');});
win.add(mv1);

win.open();

function videowindow(url,title){
    var winVideo = Titanium.UI.createWindow({
        title:title,
        backButtonTitle: 'Tillbaka',
        barColor: '#000',
        backgroundColor:'#000',
        orientationModes:[Ti.UI.LANDSCAPE_LEFT, Ti.UI.LANDSCAPE_RIGHT]
    });

    var videoPlayer = Titanium.Media.createVideoPlayer({
        url:url,
        backgroundColor: '#000',
        fullscreen:true,
        scalingMode: Titanium.Media.VIDEO_SCALING_ASPECT_FIT,
        mediaControlStyle: Titanium.Media.VIDEO_CONTROL_DEFAULT     
    });

    winVideo.add(videoPlayer);

    videoPlayer.addEventListener('complete', function(e){
        if (e.reason == 0) {
            winVideo.close();
        }
    });

    videoPlayer.addEventListener('fullscreen', function(e){
        if (e.entering == 0) {
            winVideo.close();
        }
    });

    winVideo.open({modal:true});
}

The videoplayer starts up fine but i have two other problems.

  1. The controls doesn't turn up when i touch the screen.
  2. The window doesn't close when the video is finished.

The Titanium console gives me this warning upon the end of the video

[WARN] :   ing: Attempt to dismiss from view controller <UINavigationControll
er: 0xb22af90> while a presentation or dismiss is in progress!
