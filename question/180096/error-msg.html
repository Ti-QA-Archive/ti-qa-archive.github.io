  • This is a try catch block

    try{
    /// here I put code that might throw an error


    }catch(ex){
///instead of getting the red screen of death I catch that error and do something with is
    alert("Something went wrong!!" + ex.message)
}

    Used like this for example

    function getCurrentLocation() {
    Titanium.Geolocation.purpose = 'Display the local weather information.';
    Titanium.Geolocation.accuracy = Titanium.Geolocation.ACCURACY_HIGH;
    var geoObject = null;

    Titanium.Geolocation.getCurrentPosition(function(e) {
        if (e.success) {

            try{

                ///my first try block to see if I can get the coordinates

                geoObject = e.coords;

                try{
                    /// my 2nd try block inside of the first one to see if I can apply long and lat if they exist
                    latitude = e.coords.latitude;
                    longitude = e.coords.longitude;
                }catch(ex){
                    ///my catch for my second try block

                    //alert("Error assigning lat and long returned value is" + JSON.stringify(e) + " THE ERROR RETURNED IS" + JSON.stringify(ex));
                    // another bizarre error to catch     
                    //cant determine location hide the weather
                //    $.allweather.visible = false;
                //    $.winBG.visible = false;

                    $.allweather.setVisible(false);
                    $.winBG.setVisible(false);
                    latitude = 0;
                    longitude = 0;
                }

            }catch(e){
                ////my catch for my first try block

                //cant determine location hide the weather
                $.allweather.setVisible(false);
                $.winBG.setVisible(false);
                latitude = 0;
                longitude = 0;
            }finally{
                ////after the first try catch block is done always run this code


                populateData();
            }






        }

    });
};

    Thats how you use try catch blocks! Marked as answered plz!
