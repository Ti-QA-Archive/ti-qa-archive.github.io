input.addEventListener('focus', function(e) {
    if ( this.value == 'Please input name' ) {
        this.value = null;
    }
});

input.addEventListener('blur', function(e) {
    if ( !this.value) this.value = 'Please input name';
});

  • You fake the hint with a normal value. When the user focuses the field, if the field's value is equal with your hint text you empty the field.
    When the user blurs the input, if the value is null u set it to your hint text.

    Not very difficult :)

  • Please try this

    var win = Ti.UI.createWindow({
    backgroundColor : 'pink'
});

var hintText = Ti.UI.createLabel({
    text : 'Enter your name',
    color : 'green',
    font : {
        fontSize : 12
    },
    backgroundColor : 'transparent',
    height : 35,
    width : 240,

});

var textbox = Ti.UI.createTextField({
    height : 35,
    top : 10,
    left : 40,
    width : 240,
    color : hintText.color,

});
textbox.value = hintText.text;
var flag = false;
var count = true;

textbox.addEventListener('click', function(e) {

    if (textbox.value == hintText.text && e.source.color == 'green') {
        textbox.setSelection(0, 0);
        count = false;
    }

});

textbox.addEventListener('change', function(e) {

    if (!flag) {
        flag = true;
        textbox.color = 'black';
        var txt = textbox.value;
        var actual = textbox.value.length;
        var hint = hintText.text.length;
        textbox.value = "";
        for (var i = 0; i < actual - hint; i++) {
            textbox.value += txt[i];
        }
    }

    if (textbox.value.length == 0) {
        textbox.value = hintText.text;
        textbox.color = hintText.color;
        textbox.setSelection(0, 0);
        flag = false;
    }

});

textbox.addEventListener('return', function(e) {
    if (textbox.value == hintText.text) {
    } else {
        e.source.color = 'black';
    }
});

win.add(textbox);
win.open();
