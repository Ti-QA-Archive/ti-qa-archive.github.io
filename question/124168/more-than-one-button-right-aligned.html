  • Ok, here my working example tested and working on Android (emulator)

    Here is the screenshot (android):

    Maybe somebody can try on ios too!

    var win = Ti.UI.createWindow();

var horizontalBar = Ti.UI.createView({
    backgroundColor: '#008800',
    height: 40,
    left: 10,
    right: 10,
    font: {
        fontWeight: 'normal'
    },
    borderColor: '#006600',
    touchEnabled: false
});

var label = Ti.UI.createLabel({
    gpId : 'label1',
    text: 'Mobile API',
    color: '#FFFFFF',
    left: 10,
    touchEnabled: true
});
horizontalBar.add(label);

// this view holds all buttons, layouted vertical. Margin between the buttons is done by buttonOption.left!
var buttonContainer = Ti.UI.createView({
    layout: 'horizontal',
    right: 0,
    touchEnabled: true
});

// general button options, later overwrite only title
var buttonOption = {
    left: 5, // left margin
    height: '30',
    width: 'auto',
    color: '#EEEEEE',
    backgroundColor: '#004400',
    borderColor: '#AACCAA',
    font: {
        fontWeigth: 'bold'
    },
    borderRadius: 2,
    borderWidth: 1,
    touchEnabled: true
}

// init calculation with option left (because i only use left for margin)
var buttonContainerWidthCalculated = buttonOption.left;

// create button 1 and calculate width
buttonOption.title = 'Android';
buttonOption.gpId  = 'button1';
var button = Ti.UI.createButton(buttonOption);
buttonContainer.add(button);
buttonContainerWidthCalculated += button.toImage().width + buttonOption.left;

// create button 2 and calculate width
buttonOption.title = 'iOS';
buttonOption.gpId  = 'button2';
var button = Ti.UI.createButton(buttonOption);
buttonContainer.add(button);
buttonContainerWidthCalculated += button.toImage().width + buttonOption.left;

// create button 3 and calculate width
buttonOption.title = 'Blackberry';
buttonOption.gpId  = 'button3';
var button = Ti.UI.createButton(buttonOption);
buttonContainer.add(button);
buttonContainerWidthCalculated += button.toImage().width + buttonOption.left;

// set calculated width of all buttons as button container width
buttonContainer.width = buttonContainerWidthCalculated;

horizontalBar.add(buttonContainer);

win.add(horizontalBar);

win.open();

// add a simple eventlistener
horizontalBar.addEventListener('click', function(event){
    if (event.source.gpId) {
        alert(String.format('clicked element with gpId "%s"', event.source.gpId));
    }
});

  • Version 0.2 - Ellipsize

    I added disabled wordwrap and added ellipsize by calculating/guessing the labels widht … i really like to make a better more generic version.

    Maybe somebody have tipps to make this better…. maybe im on a complete wrong track!

    var win = Ti.UI.createWindow();

var horizontalBar = Ti.UI.createView({
    backgroundColor: '#008800',
    height: 40,
    left: 10,
    right: 10,
    font: {
        fontWeight: 'normal'
    },
    borderColor: '#006600',
    touchEnabled: false
});

var label = Ti.UI.createLabel({
    gpId : 'label1',
    text: 'Titanium Mobile API - Choose one of the following  ',
    color: '#FFFFFF',
    backgroundColor: '#007700',
    left: 10,
    touchEnabled: true,
    wordWrap: false,
    ellipsize: true
});
horizontalBar.add(label);

// this view holds all buttons, layouted vertical. Margin between the buttons is done by buttonOption.left!
var buttonContainer = Ti.UI.createView({
    layout: 'horizontal',
    right: 0,
    touchEnabled: true
});

// general button options, later overwrite only title
var buttonOption = {
    left: 5, // left margin
    height: '30',
    width: 'auto',
    color: '#EEEEEE',
    backgroundColor: '#004400',
    borderColor: '#AACCAA',
    font: {
        fontWeigth: 'bold'
    },
    borderRadius: 2,
    borderWidth: 1,
    touchEnabled: true
}

// init calculation with option left (because i only use left for margin)
var buttonContainerWidthCalculated = (buttonOption.left || 0) + (buttonOption.right || 0);

// create button 1 and calculate width
buttonOption.title = 'Android';
buttonOption.gpId  = 'button1';
var button = Ti.UI.createButton(buttonOption);
buttonContainer.add(button);
buttonContainerWidthCalculated += button.toImage().width + (buttonOption.left || 0) + (buttonOption.right || 0);

// create button 2 and calculate width
buttonOption.title = 'iOS';
buttonOption.gpId  = 'button2';
var button = Ti.UI.createButton(buttonOption);
buttonContainer.add(button);
buttonContainerWidthCalculated += button.toImage().width + (buttonOption.left || 0) + (buttonOption.right || 0);

// create button 3 and calculate width
buttonOption.title = 'Blackberry';
buttonOption.gpId  = 'button3';
var button = Ti.UI.createButton(buttonOption);
buttonContainer.add(button);
buttonContainerWidthCalculated += button.toImage().width + (buttonOption.left || 0) + (buttonOption.right || 0);

// set calculated width of all buttons as button container width
buttonContainer.width = buttonContainerWidthCalculated;

horizontalBar.add(buttonContainer);

// set the real (... guessed) label width
var labelWidth = Ti.Platform.displayCaps.platformWidth - buttonContainerWidthCalculated - (horizontalBar.left || 0) - (horizontalBar.right || 0) - (label.left || 0) - (label.right || 0);
label.width = labelWidth;

win.add(horizontalBar);

win.open();

// add a simple eventlistener
horizontalBar.addEventListener('click', function(event){
    if (event.source.gpId) {
        alert(String.format('clicked element with gpId "%s"', event.source.gpId));
    }
});
