Pass two o more arrays in event listener

This example has one array, but how i can send two or more arrays?

Try this..

In list.js:
Titanium.UI.currentWindow.addEventListener('send_data', myDataFunction);
function myDataFunction(e)
{
Titanium.API.info("Data: " + e.dataArr);
};

In app.js:
var data = [1,2,3];
var w = Titanium.UI.createWindow({
title: 'List',
url: 'list.js'
});
w.open();
w.fireEvent('send_data', {dataArr:data});

    The data you send can be anything within that object. ie:

    {
    dataArr1: [1,2,3],
    dataArr2: [4,5,6]
}
