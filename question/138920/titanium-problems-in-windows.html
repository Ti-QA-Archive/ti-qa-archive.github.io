// create tab group
var tabGroup = Titanium.UI.createTabGroup();
var backgroundColor = '#333000';
var fontColor = '#FFFFFF';
var fontType = 'Times New Roman';
var fontSize = 32;

//
// create base UI tab and root window
//
var mainScreen = Titanium.UI.createWindow({
    title : 'Weather',
    backgroundColor : backgroundColor,
});
var tab1 = Titanium.UI.createTab({
    title : 'Current Weather',
    window : mainScreen
});

var prompt = Titanium.UI.createLabel({
    color : fontColor,
    text : 'Enter Zip Code:',
    font : {
        fontSize : fontSize,
        fontFamily : fontType
    },
    top : 10,
    left : 10,
    width : 'auto'
});

// Do researched for max allowed zipcode number lenght and minimun if there is such

var zipCode = Titanium.UI.createTextArea({
    color : '#0000',
    hintText : 'Type Zip Code Here',
    top : 40,
    left : 10,
    height : 30,
    width : 100
});

var locationLabel = Titanium.UI.createLabel({
    text : 'Location:',
    color : fontColor,
    font : {
        fontSize : fontSize,
        fontFamily : fontType
    },
    top : 40,
    left : 300,
    width:400
});

var weatherLabel = Titanium.UI.createLabel({
    text : 'Weather:',
    color : fontColor,
    font : {
        fontSize : fontSize,
        fontFamily : fontType
    },
    top : 75,
    left : 300,
    width:400
});


var setLocation = Titanium.UI.createButton({
    top : 300,
    left : 10,
    width : 150,
    height : 50,
    title : 'Locate',
});


//Get Weather Info and return Location and current weather
setLocation.addEventListener('click', function(click) {
    locationLabel.text = 'Location: '+ zipCode.value;
    weatherLabel.text = 'Weather: '+'Raining';
});


zipCode.addEventListener('change', function(e){
    if(e.value.length >= 5) {
        zipCode.value = e.value.substr(0,5);
    }
});



mainScreen.add(prompt);
mainScreen.add(zipCode);
mainScreen.add(locationLabel);
mainScreen.add(weatherLabel);
mainScreen.add(setLocation);

tabGroup.addTab(tab1);

// open tab group
tabGroup.open();
