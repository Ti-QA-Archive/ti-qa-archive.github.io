self.open not working

How can I get my links in my table to work? When I click I get nothing. The links do not work.

The problem I am having is with:

self.open(win,{animated:true});

I Have tried: 'Titanium.UI.currentWindow' as well as ''Titanium.UI.currentTab' [undefined]'and get:

'Titanium.UI.currentWindow' [undefined]

Below is the entire page of code in case I am missing something. This is my first project with accelerator so there is no telling what I have done!

function AppWindow(title) {
    var self = Ti.UI.createWindow({
        title:title,
        backgroundColor:'white'
    });

    var data = [

    { title:"Catalog", hasChild:true, test:'CatalogWindow.js', header:'' },
    { title:"Service Calculator", hasChild:true, header:'' }
    ];

// create table view
var tableview = Titanium.UI.createTableView({
    data:data,
    style:Titanium.UI.iPhone.TableViewStyle.GROUPED
    });

// create table view event listener
tableview.addEventListener('click', function(e)
{
    if (e.rowData.test)
    {
        var win = Titanium.UI.createWindow({
            url:e.rowData.test,
            title:e.rowData.title
        });
        self.open(win,{animated:true});
    }
});
    self.add(tableview);

    return self;
};

module.exports = AppWindow;
