    Hi, onreadystatechange call back and how to use DONE with readyState.

    var client = Ti.Network.createHTTPClient({
     // function called when the response data is available
     onload : function(e) {
         Ti.API.info("Received text: " + this.responseText);
         alert('success');
     },
     // function called when an error occurs, including a timeout
     onerror : function(e) {
         Ti.API.debug(e.error);
         alert('error');
     },
     onreadystatechange : function(e) {
         Ti.API.debug(this.readyState);  //Must be set before calling open.When the callback is invoked, this.readyState is set to one of the Titanium.Network.HTTPClient ready-state constants, OPENED, HEADERS_RECEIVED, LOADING, or DONE.
         alert(this.readyState);
     },
     timeout : 5000  // in milliseconds
 });
 // Prepare the connection.
 client.open("GET", url);
 // Send the request.
 client.send();
