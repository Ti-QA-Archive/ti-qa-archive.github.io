  • here is my code

    
var PlainTemplate = { 
    childTemplates: [
        { 
             type: 'Ti.Media.VideoPlayer',
             bindId:'videoPlayer',
             properties: {
             zIndex : -1,
             },
             events:{load : function(e){
                 Titanium.API.info('Loaded');
             },play : function(e){
                 e.source.play();
                 Titanium.API.info('hellooooooo');
             }
             }
        },

        {
            type: 'Ti.UI.Button',
            bindId: 'playButton',
            properties:{
                    backgroundImage:'/images/newImages/play_btn.png',
                    width:48,
                    height:34,
                    top: 133,
                    left:140,
                    zIndex:10
          },
          events : { click : function(e){

              Ti.API.info(e.section.getItemAt(0).videoPlayer);
              e.source.fireEvent('play');

          }}

        }

    ], properties:{width : 320, height:320}
};

    • Hi,

      I have done this on tableview in the following way, you can try this.

      var win = Ti.UI.createWindow(
    {
        backgroundColor:'white'
    }
);

var tbl_data = [];
for (var i = 0; i < 5; i++) {
    var row = Ti.UI.createTableViewRow();
    var videoPlayer = Titanium.Media.createVideoPlayer({
    top : 2,
    left: 10,
    height : 60,
    width : 70,
    mediaControlStyle : Titanium.Media.VIDEO_CONTROL_DEFAULT,
    scalingMode : Titanium.Media.VIDEO_SCALING_ASPECT_FIT
});

videoPlayer.url = 'http://assets.appcelerator.com.s3.amazonaws.com/video/media.m4v';

    var button = Ti.UI.createButton({
        right: 10,
        height: 50,
        width: 80,
        backgroundImage:'/KS_nav_ui.png'
    });
    row.add(videoPlayer);
    row.add(button);
    tbl_data.push(row);
}

// // now assign that array to the table's data property to add those objects as rows
var table = Titanium.UI.createTableView({
    data:tbl_data
});

 button.addEventListener('click', function(e) {
    videoPlayer.play();
   });


win.add(table);
win.open();
