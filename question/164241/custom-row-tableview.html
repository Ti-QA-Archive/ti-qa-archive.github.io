  • Hi,

    Please run the below code. It may help to you a little bit.

    var win=Titanium.UI.createWindow({ 
    backgroundColor:'#000',
    title:'Testing', 
    layout:'vertical' 
}); 
var rowData=[];
for (var i = 5 - 1; i >= 0; i--){

    var row=Ti.UI.createTableViewRow({
        height:Ti.UI.SIZE,
        className :'row_task',
        //clickName:CustomData[i].id
    });

    var rowView=Ti.UI.createView({
        height:Ti.UI.FILL,
        top:5,
        left:13,
        right:10,
    });

    var checkbox = Ti.UI.createSwitch({
        style: Ti.UI.Android.SWITCH_STYLE_CHECKBOX,
        textAlign:Ti.UI.TEXT_ALIGNMENT_LEFT,
        top:4, className :'checkbox_row',
        //clickName: CustomData[i].id,
        value:false,
        width: 25,
        left:20
    });

    var label1 = Titanium.UI.createLabel({
        text:'labelOne'+i,//CustomData[i].item,
        font:{fontSize:20,fontWeight:'bold'},
        width:'auto',
        textAlign:'left',
        top:2,
        left:55,
        height:40,
        className :'row_task',
        //clickName:CustomData[i].id,
    });

    var label2=Ti.UI.createLabel({
        top : 2,
        text:'label2',
        height : 40,
        visible:false,
        className :'row_task',
        //clickName:CustomData[i].id,
        left:55,
    });

    rowView.add(checkbox);
    rowView.add(label1);
    rowView.add(label2);

    row.add(rowView);

   // row.title='row'+i;//taskitem;

    rowData.push(row);

}
var table = Ti.UI.createTableView();
table.setData(rowData);
table.addEventListener('click',function(e){
    var rowElement=e.row.children[0];
    if(e.source.className=='checkbox_row')
    {
        if(rowElement.children[0].value==true)
        {
            rowElement.children[2].visible=true;
            rowElement.children[1].visible=false;
            alert(rowElement.children[2].text);
        }
    }
});
win.add(table);
win.open();
