 var newMapMarkerIcon = Ti.UI.createView({
    height:20,
    width: 20,
     backgroundColor: '#ff0000'
});

var f = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, 'newMapIcon.png');
f.write(newMapMarkerIcon);

var newMapMarker = Titanium.Map.createAnnotation({
    image:f 
});

newMap.annotations = [newMapMarker];

How would you get that to work?
