Drag and Drop a Window with elements inside it

I have been trying to implement a drag and drop of a window that has a button attached to that window. I am using TiDrop and it works if I drag the window, but for some reason the button I have is also captured by TiDrop, and causes an error:

Script Error = 'undefined' is not an object (evaluating '_event.globalPoint.y') at TiDrop.js (line 34).

My implementation is like this: 

var tileMenu= Titanium.UI.createWindow({  
       url:'tileMenu.js',
       zIndex:10,
    top:430,
    left:-270,
    width:320,
    height:480,
    backgroundColor:'#11CAAF'
});
tileMenu.open();
win.add(tileMenu);

function yay(e) {
    if(e.contained) {
        e.source.top = 0;
        e.source.left = 0;
    }
}

TiDrop.init(tileMenu, win, yay);

My tileMenu.js

var win = Ti.UI.currentWindow;

var tileMenuClose = Titanium.UI.createButton({
    height:20,
    width:20,
    left:270,
    top:0
});

win.add(tileMenuClose);

I have tried also just adding an event listener for touchmove of the window by itself, not using TiDrop, but it still fires the globalPoint error. Any ideas how I can get around this?
