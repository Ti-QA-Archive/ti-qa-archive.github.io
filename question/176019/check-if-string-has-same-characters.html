if (!String.prototype.ContrainSameChars) {
  String.prototype.ContrainSameChars = function (bar {
     if (this.length != bar.length) return false;
     if (this == bar) return true;
     return (this.split('').sort() == bar.split('').sort()) ? true : false;     
  };
}
