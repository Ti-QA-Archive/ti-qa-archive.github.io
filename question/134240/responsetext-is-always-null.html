responseText is always null

I write a code to download JSON file from server but this.responseText is always null

var url = 'http://someurl.com/resource_file.json';
var client = Ti.Network.createHTTPClient({
    onload: function(e) {
        Ti.API.info(this.status + ' - ' + this.statusText);
        Ti.API.info('Data : ' + this.responseText);
    },
    onerror: function(e) {
        Ti.API.error(this.status + ' - ' + this.statusText);
    }
});
client.open('GET', url);
client.send();

result is

I/TiAPI   (  781): 200 - OK
I/TiAPI   (  781): Data : null

Its status is 200 but while responseText is null.

I use Windows7 and build to Android platform.

    Did you try this.responseData.text? Below code works for me:

    var url = 'http://remixman.net/laika/menu_init.json';
var client = Ti.Network.createHTTPClient({
    onload: function(e) {
        Ti.API.info( JSON.stringify(this) );
        Ti.API.info('Data : ' + this.responseData.text);

        // Ti.API.info(this.status + ' - ' + this.statusText);
        // Ti.API.info('Data : ' + this.responseText);
    },
    onerror: function(e) {
        Ti.API.error(this.status + ' - ' + this.statusText);
    }
});
client.open('GET', url);
client.send();
