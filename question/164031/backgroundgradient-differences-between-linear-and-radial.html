  • Hi andres,

    Please try the following code. Its working fine for both type linear and radial.

    var self = Ti.UI.createWindow({
        backgroundColor: '#f9f9f9',
        exitOnClose : true, 
        navBarHidden : true,
        height : '100%',
        width : '70%',
        left : '0%',
        layout : 'vertical',
        orientationModes: [Ti.UI.PORTRAIT],
        backgroundGradient: {
           type: 'linear',
           //type: 'radial',
            startPoint: { 
                x: '0%', 
                y: '0%' 
            },
            endPoint: { 
                x: '100%', 
                y: '100%' 
            },
            colors: [ 
                { color: '#D7E6EF', offset: 0}, 
                { color: '#FEFEFF', offset: 4 }
            ],
        }   
    }); 

self.open();

    Thanks.
