How to update custom object through ACS?

Hi,can anyone help me with ACS updating of custom object. When i tried to change my rating value, my alert get the default value of 0 and not the value i clicked. 

        var dishName = Ti.UI.createLabel({
            text : dish.name,
            font : {
                fontSize : 20
            },
            height : "50%",
            textAlign : 'left',
            width : '43%',
            top : '10%',
            left : '15%',
            color : 'black'
        });

        var ratingBar1 = ratingbar.createRatingBar({
            top : '50%',
            left : '14%',
            rating : dish.rate,
            stars : 5,
            isIndicator : false
        });

        ratingBar1.addEventListener('change', function(e) {
            Cloud.Objects.update({
                classname : 'dishes',
                fields : {
                    rate : dishes.rate
                }
            }, function(e) {
                alert(ratingBar1.rating);
            });
        });
