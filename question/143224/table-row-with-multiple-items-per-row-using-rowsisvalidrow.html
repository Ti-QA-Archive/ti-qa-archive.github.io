  • I would handle it something like this. NOT TESTED

    var cnt = 0,
    data = [],
    tblrow, wrapper, label;

while (rows.isValidRow()) {
    // if the count is 0, then create a wrapper for the row
    if (cnt === 0) {
        wrapper = Ti.UI.createView({
            width:Ti.UI.FILL,
            layout:'horizontal'
        });
    }

    // create a label using the field you want and add it to the wrapper
    label = Ti.UI.createLabel({
        text:rows.fieldByName('fieldname')
    });
    wrapper.add(label);

    // If we're on the 4th item, create a row, add the wrapper to the row
    // and reset the count. otherwise, just increment the count
    if (cnt === 3) {
        tblrow = Ti.UI.createTableViewRow();
        tblrow.add(wrapper);
        data.push(tblrow);
        cnt = 0;
    } else {
        cnt++
    }

    // go to the next row.
    rows.next();
}
