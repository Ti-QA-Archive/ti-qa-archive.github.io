    • The short answer is no - not using only standard properties.

      But you can achieve your goal by playing a little, a quick example of how to achieve what you want;

      • the row responds to the selected state as a toggle on and off - I am guessing that was your wish, but easy to change to momentary
      • the left image toggles between two images
      • the background color of the row toggles between two colors
      • the buttons are not affected by the toggle
      • the two buttons on the right act independent of the row itself
      function addRow(obj) {
    var row = Ti.UI.createTableViewRow({
        type: 'row',
        className: 'standard',
        height: Ti.UI.SIZE,
        isSelected: false,
        selectionStyle: Ti.UI.iPhone.TableViewCellSelectionStyle.NONE,
        width: Ti.UI.FILL
    });
    var view = Ti.UI.createView({
        backgroundColor: '#fff',
        height: 60,
        left: 0,
        touchEnabled: false,
        width: Ti.UI.FILL
    });
    row.add(view);
    var img = Ti.UI.createImageView({
        height: Ti.UI.SIZE,
        image: 'checkoff.png',
        left: 10,
        touchEnabled: false,
        width: Ti.UI.SIZE        
    });
    row.add(img);    
    var lbl = Ti.UI.createLabel({
        color: '#000',
        height: Ti.UI.SIZE,
        highlightedColor: '#fff',
        left: 70,
        text: obj.title,
        textAlign: 'left',
        touchEnabled: false,
        width: Ti.UI.SIZE
    });
    row.add(lbl);
    var viewButtons = Ti.UI.createView({
        height: Ti.UI.SIZE,
        layout: 'horizontal',
        right: 10,
        width: Ti.UI.SIZE
    });
    row.add(viewButtons);
    var btn1 = Ti.UI.createButton({
        type: 'btn',
        height: 40,
        left: 10,
        title: 'One'
    });
    viewButtons.add(btn1);
    var btn2 = Ti.UI.createButton({
        type: 'btn',
        height: 40,
        left: 10,
        title: 'Two'
    });
    viewButtons.add(btn2);
    row.toggleSelected = function () {
        var isSelected = !row.isSelected;
        Ti.API.info('isSelected', isSelected);
        row.isSelected = isSelected;
        img.setImage(isSelected ? 'checkon.png' : 'checkoff.png');
        view.setBackgroundColor(isSelected ? '#ccc' : '#fff');
    };
    return row;
}
var data = [
    { title: 'One' },
    { title: 'Two' },
    { title: 'Three' },
    { title: 'Four' }
];
var intRow = 0, intRows = data.length, row, rows = [];
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    row = addRow({
        title: data[intRow].title
    });
    rows.push(row);
}
var tbl = Ti.UI.createTableView({
    data: rows,
    height: Ti.UI.FILL,
    minRowHeight: 60,
    width: Ti.UI.FILL
});
tbl.addEventListener('click', function (e) {
    if (e.source.type === 'row') {
        e.source.toggleSelected();
    }
});
win.add(tbl);

      This assumes you have "on" and "off" state images, see below for my bad ones, just name them;
