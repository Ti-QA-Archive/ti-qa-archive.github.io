  • The frame src value is unquoted. Try this instead

    var win = Ti.UI.createWindow();
var html = '<html><head></head><body><iframe src="\`https://www.google.com/calendar/embed?mode=AGENDA&height=600&wkst=1&bgcolor=%23FFFFFF&\` style=" border-width:0 " width="320dp" height="600" frameborder="0" scrolling="no"></iframe></body></html>';

var webview = Ti.UI.createWebView({
    html : html
});
win.add(webview);
win.open();
