  • here is some code with 2 pickers, with the eventListeners you should be able to do what you want to do.

    //
// Pickers
//
var pickerWindow = uiElements.Window('');

var minDate = new Date();
minDate.setMonth(0);
minDate.setDate(1);

var maxDate = new Date();
maxDate.setFullYear( (minDate.getFullYear()+50) );
maxDate.setMonth(11);
maxDate.setDate(31);

value = new Date();

var datePicker = Ti.UI.createPicker({
    type:Ti.UI.PICKER_TYPE_DATE,
    minDate: minDate,
    maxDate: maxDate,
    value: value,
    top: 0
});

var typePicker = Ti.UI.createPicker({top: 0});
var pickerData = [];
    pickerData[0]=Titanium.UI.createPickerRow({title:'1337'});
    pickerData[1]=Titanium.UI.createPickerRow({title:'A'});
    pickerData[2]=Titanium.UI.createPickerRow({title:'B'});
    pickerData[3]=Titanium.UI.createPickerRow({title:'C'});
    pickerData[4]=Titanium.UI.createPickerRow({title:'D'});
    pickerData[5]=Titanium.UI.createPickerRow({title:'E'});
    pickerData[6]=Titanium.UI.createPickerRow({title:'F'});
    pickerData[7]=Titanium.UI.createPickerRow({title:'G'});

typePicker.add(pickerData);

// turn on the selection indicator (off by default)
typePicker.selectionIndicator = true;
datePicker.selectionIndicator = true;
var typeValue = typePicker.value;
var dateValue = datePicker.value;

typePicker.addEventListener('change',function(e){
    typeValue = e.row.title;
});
datePicker.addEventListener('change',function(e){
    dateValue = e.value;
});

var btnSelect = uiElements.Button('');
    btnSelect.bottom = 50;
    btnSelect.height = 50;
    btnSelect.width = '95%';
    btnSelect.font = {fontSize: 18, fontFamily: 'Arial'};
btnSelect.addEventListener('click', function(e) {
    if(e.source.title == 'Select date'){
        date.value = dateValue.getDate()+' / '+(dateValue.getMonth()+1)+' / '+dateValue.getFullYear();
    }
    else if(e.source.title == 'Select type'){
        type.value = typeValue;
    }
    pickerWindow.close();
});
