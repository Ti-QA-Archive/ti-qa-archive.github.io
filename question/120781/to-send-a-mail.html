    • your email link… try this, rather then a href:

      onClick="Ti.App.fireEvent('send_email', {address: 'service@domain.com'}); return false;"

      Then in your about.js BEFORE you open the webview, add something like:

      Titanium.App.addEventListener('send_email',function(e){
    var emailDialog = Titanium.UI.createEmailDialog();

     emailDialog.subject = "Hello from My App";
     emailDialog.toRecipients = [e.address];
     emailDialog.messageBody = 'Appcelerator Titanium Rocks!';
     emailDialog.open();
});
