  • First, please use the markdown syntax to make your code readable. The syntax is detailed right above the box where you enter the question.

    Here is an example of a scrollableview with a "home" button on views 2 and 3:

    app.js:

    Ti.UI.setBackgroundColor('#000');

var win1 = Ti.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});


var v1 = Ti.UI.createView ({
    backgroundColor: '#f66',
    top: 0,
    bottom: 0,
    left: 0,
    right: 0
});

var v2 = Ti.UI.createView ({
    backgroundColor: '#6f6',
    top: 0,
    bottom: 0,
    left: 0,
    right: 0
});

var btn1 = Ti.UI.createButton ({
    title: 'Home'    
});

btn1.addEventListener ('click', function (e) {
    sv.scrollToView (0);
});

v2.add (btn1)

var v3 = Ti.UI.createView ({
    backgroundColor: '#66f',
    top: 0,
    bottom: 0,
    left: 0,
    right: 0
});

var btn2 = Ti.UI.createButton ({
    title: 'Home'    
});

btn2.addEventListener ('click', function (e) {
    sv.scrollToView (0);
});

v3.add (btn2)

var sv = Ti.UI.createScrollableView ({
    top: 0,
    left: 0,
    right: 0,
    bottom: 100,
    views: [v1, v2, v3]
});

win1.add (sv);

var bv1 = Ti.UI.createView ({
    height: 40,
    left: 0,
    right: 0,
    bottom: 50,
    layout: 'horizontal'
});

var btn3 = Ti.UI.createButton ({
    height: 40,
    width: '33%',
    title: 'View 1'    
});

btn3.addEventListener ('click', function (e) {
    sv.scrollToView (0);
});

var btn4 = Ti.UI.createButton ({
    height: 40,
    width: '33%',
    title: 'View 2'    
});

btn4.addEventListener ('click', function (e) {
    sv.scrollToView (1);
});

var btn5 = Ti.UI.createButton ({
    height: 40,
    width: '33%',
    title: 'View 3'    
});

btn5.addEventListener ('click', function (e) {
    sv.scrollToView (2);
});


bv1.add (btn3);
bv1.add (btn4);
bv1.add (btn5);

win1.add (bv1);

win1.open ();
