Scrollable View background image issue

I have a scrollable view for which i allow the user to select a background image from his Gallery. Seems pretty simple but for some reason the image is not getting set. I am absolutely lost….. Titanium 3.3.0 , iPhone Simulator.

var win = Ti.UI.createWindow({
    backgroundColor : 'black',
    layout : 'vertical',

});

var tempscrollableView = Ti.UI.createScrollableView({
    top : 0,
    left : 0,
    width : 320,
    height : 200,
    backgroundColor : 'pink',
});
win.add(tempscrollableView);

var lbltemp = Ti.UI.createLabel({
    text : 'CLICK Me',
    width : 100,
    height : 100,
    backgroundColor : 'white',
});
lbltemp.addEventListener('click', function() {

    Titanium.Media.openPhotoGallery({
        success : function(event) {

            var image = event.media;

            if (event.mediaType == Ti.Media.MEDIA_TYPE_PHOTO) {

                var filename = Titanium.Filesystem.applicationDataDirectory + "/CoverPic.jpg";
                var imageFile = Titanium.Filesystem.getFile(filename);

                if (imageFile.exists() !== false) {
                    imageFile.write(image);
                } else {
                    imageFile.createFile();
                    imageFile.write(image);
                }
                tempscrollableView.setBackgroundImage(imageFile);

            }
        },
        cancel : function() {
            return null;
            //user cancelled the action fron within
            //the photo gallery
        }
    });

});
win.add(lbltemp);
win.open();

    Hello Pritish Try this code to " Take a photo from the gallery and use the photo".

    var win = Ti.UI.createWindow({
    backgroundColor : 'black',
    layout : 'vertical',

});

var tempscrollableView = Ti.UI.createScrollableView({
    top : 0,
    left : 0,
    width : 320,
    height : 200,
    backgroundColor : 'pink',
});
win.add(tempscrollableView);

var lbltemp = Ti.UI.createLabel({
    text : 'CLICK Me',
    width : 100,
    height : 100,
    backgroundColor : 'white',
});
lbltemp.addEventListener('click', function() {
    Titanium.Media.openPhotoGallery({

        success:function(event)
        {
            var cropRect = event.cropRect;
            var image = event.media;

            // set image view
            Ti.API.debug('Our type was: '+event.mediaType);
            if(event.mediaType == Ti.Media.MEDIA_TYPE_PHOTO)
            {
                tempscrollableView.backgroundImage = image;
            }

        },
        mediaTypes:[Ti.Media.MEDIA_TYPE_VIDEO,Ti.Media.MEDIA_TYPE_PHOTO]
    });
});
win.add(lbltemp);
win.open();
