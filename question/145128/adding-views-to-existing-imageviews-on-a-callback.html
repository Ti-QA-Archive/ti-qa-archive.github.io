Adding Views to existing ImageViews on a callback

Hello, I'm running into a problem where on a 'click' callback, I need to add an imageview to the view which was clicked on. When i do so, I see nothing. My code looks something like this:

for(var i = 0; i < data.length; i++)
{
    var thumb = Ti.UI.createImageview({
        image: data[i].thumbnail
    });

    var callback = (function(index, thumbnail){
        return function(){
            var playbtn = Ti.UI.createImageView({
                image: 'blah'
        });
        thumbnail.add(playbtn);
    })(i, thumb);

    thumb.addEventListener('click', callback);

    horizontalView.add(thumb);
}

I know the closure syntax is funky, I stuck an alert() in there, so I know its working but the playbtn imageView doesn't show up.
