    I think you should use scalesPageToFit property.

    var webView = Ti.UI.createWebView({
    url : "http://developer.appcelerator.com",
    top : 38,
    scalesPageToFit : false
});

    • Hi Rickard, It is possible, I found some help on github and modify an example according to your need. I hope this will help you.

      var webView = Ti.UI.createWebView({
    url : 'http://www.appcelerator.com',
    scalesPageToFit : true,
    top : 35,
    left : 0
});
win.add(webView);
var htmlHack = '';
htmlHack += 'var element = document.createElement("meta");';
htmlHack += 'element.name = "viewport";';
htmlHack += 'element.content = " initial-scale = 2.5";';
htmlHack += 'var head = document.getElementsByTagName("head")[0];';
htmlHack += 'head.appendChild(element);';

webView.addEventListener('load', function(e) {
    webView.evalJS(htmlHack);
});
win.add(webView);

      you have to change zoom level according to your need in variable htmlHack

      'element.content = " initial-scale = 2.5";'
