mymedia.openPhotoGallery not working with version 3.3.0.GA

I have just downloaded and built the MyMedia module and it does not appear to be working with Version 3.3.0.GA. I have built it with XCode version Version 5.1.1 (5B1008) and am running it with Studio version Titanium Studio, build: 3.3.0.201407100905.

When I try and open the gallery on the iphone simulator as well as two different Iphones, I don't get any error but nothing happens, just a blank screen.

Here's my test code (straight from the github example) which I know is getting called since I see the alert.

function init()

{

 var mymedia = require("my.media");

 mymedia.openPhotoGallery({
success:function(e) {
    if (e.mediaType === Ti.Media.MEDIA_TYPE_PHOTO) {
        Ti.API.debug("Photo was uploaded");
    }
    Ti.API.debug('openPhotoGallery SUCCESS: ' + JSON.stringify(e));
    if (e.metadata) {
        var dialog = Ti.UI.createAlertDialog({
            message: e.metadata,
            ok: 'OK',
            title: 'Metadata'
        }).show();
    }
    else 
       alert( JSON.stringify(e) );
},
cancel:function() {
    Ti.API.debug("openPhotoGallery CANCEL");
},
error:function(error) {
    alert(JSON.stringify(error));
    Ti.API.error("openPhotoGallery ERROR: " + JSON.stringify(error));
},
animated:true,
allowEditing:true,
videoMaximumDuration:15000, // 15 seconds
videoQuality: Ti.Media.QUALITY_MEDIUM,
mediaTypes:[Ti.Media.MEDIA_TYPE_PHOTO, Ti.Media.MEDIA_TYPE_VIDEO]

});

alert('Done');
}
