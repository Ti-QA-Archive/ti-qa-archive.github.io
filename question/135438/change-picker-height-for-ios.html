  • Hey Marcel,
    Look at this code. this code shrink the picker view.

    var win = Ti.UI.createWindow({title : "Pickerview Demo"});

var picker = Ti.UI.createPicker({
    top : 50
});

var data = [];
data[0] = Ti.UI.createPickerRow({
    title : 'Bananas'
});
data[1] = Ti.UI.createPickerRow({
    title : 'Strawberries'
});
data[2] = Ti.UI.createPickerRow({
    title : 'Mangos'
});
data[3] = Ti.UI.createPickerRow({
    title : 'Grapes'
});

picker.add(data);
picker.selectionIndicator = true;
win.add(picker);
var twoDMetrix = Ti.UI.create2DMatrix({
    scale : 0.5
})
var scaleAnimate = Ti.UI.createAnimation({
    transform : twoDMetrix,
    duration : 1000
})

picker.animate(scaleAnimate);

var win.open();
