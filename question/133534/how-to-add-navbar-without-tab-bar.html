  • my code

    // this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');

// create tab group
var tabGroup = Titanium.UI.createTabGroup();

//
// create base UI tab and root window
//
var win1 = Titanium.UI.createWindow({
    title : 'Tab 1',
    backgroundColor : '#ccc',
    navBarHidden : false,
    zIndex : 1,

});
var win2 = Titanium.UI.createWindow({
    title : 'Tab 2',
    backgroundColor : '#fff',
    navBarHidden : true,
    zIndex : 0,
});

////table
var RegData = [{
    title : "Spain",
    hasChild : true
}, {
    title : "United Kingdom",
    hasChild : true
}, {
    title : "United States",
    hasChild : true
}, {
    title : "France",
    hasChild : true
}];
var tabledata = Ti.UI.createTableView({
    data : RegData
})
win2.add(tabledata);

var isSliding = false;
var baseX = null;
var conX = null;
var diff = null;

win1.addEventListener('touchstart', function(e) {
    baseX = e.globalPoint.x;
    isSliding = true;
});

win1.addEventListener('touchmove', function(e) {
    if(isSliding) {

        if(e.globalPoint.x < 300) {

            win1.left = e.globalPoint.x;

        }

    }
});

win1.addEventListener('touchend', function(e) {
    if(isSliding) {
        diff = e.globalPoint.x - baseX;

        if(diff > 0 && win1.left < 50) {
            win1.animate({
                left : 0,
                duration : 300
            });
        } else if(diff <= 300 && diff > 0 && win1.left > 50) {
            win1.animate({
                left : 300,
                duration : 300
            });
        } else if(diff < 0) {
            win1.animate({
                left : 0,
                duration : 300
            });
        }
    }
    isSliding = false;
});
win2.open();
win1.open();
