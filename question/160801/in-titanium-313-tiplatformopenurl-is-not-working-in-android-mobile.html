In Titanium 3.1.3 Ti.Platform.openURL is not working in android mobile

detailHtmlContent.addEventListener('beforeload', function(e) {
if (e.url.match(/^file:\/\//)) {
Ti.API.debug('beforeload.local file.');
} else {
Ti.API.debug('remote url. open with browser!');
Ti.Platform.openURL(e.url);
detailHtmlContent.stopLoading();
}
});
detailHtmlContent.addEventListener('load', function(e) {
if (e.url.match(/^file:\/\//)) {
Ti.API.debug('load. local file.');
} else {
if (detailHtmlContent.canGoBack()) {
Ti.API.debug('force first URL.');
detailHtmlContent.goBack();
}
}
});
