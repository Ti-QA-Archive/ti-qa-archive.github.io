    Hello alex santos,

    You have to add a custom property to the row and add the label text to that property. And at the tableView eventListner catch the label text value using “e.rowData.customProperty”. Here is the sample code to clearly understand.

    
Ti.UI.setBackgroundColor('#000');
var win = Ti.UI.createWindow({
    backgroundColor : 'black',
    exitOnClose : true,
    fullscreen : false,
    title : 'TableView Demo'

});

var tableData = [];

for (var i = 1; i <= 20; i++) {
    var row = Ti.UI.createTableViewRow({
        className : 'forumEvent', // used to improve table performance
        selectedBackgroundColor : 'white',
        rowIndex : i, // custom property
        height : 50,
        value : 'Fred Smith ' + i, //Custom property , useful for determining the row during events
    });

    var labelUserName = Ti.UI.createLabel({
        color : '#576996',
        font : {
            fontFamily : 'Arial',
            fontSize : 20,
            fontWeight : 'bold'
        },
        text : 'Fred Smith ' + i,
        left : 70,
        top : 6,
        width : 200,
        height : 30
    });

    row.add(labelUserName);

    tableData.push(row);
}

var tableView = Ti.UI.createTableView({
    backgroundColor : 'white',
    data : tableData
});

tableView.addEventListener('click', function(e) {
    alert('lave Value:' + e.rowData.value);  //For getting row value.
});

win.add(tableView);
win.open();

    You you have any concern, then let me know.
