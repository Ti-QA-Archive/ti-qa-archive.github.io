    • Okey..
      Try this code 

      var url = "http://seerah.designers99.com/songs.xml";
function loadXML(url) {
    Ti.API.info('>>>> loading RSS feed ' + url);
    xhr = Titanium.Network.createHTTPClient();
    xhr.open('GET', url);

    xhr.onload = function() {

        var doc = this.responseXML.documentElement;
        var deals = doc.getElementsByTagName('song');
        if(deals != null && deals.length > 0) {
            try {
                for(var i = 0; i < deals.length; i++) {
                    var id = deals.item(i).getElementsByTagName('id').item(0).text;
                    Ti.API.info('value of Id =' + id);

                    var title = deals.item(i).getElementsByTagName('title').item(0).text;
                    Ti.API.info('value of title =' + title);

                    var filename = deals.item(i).getElementsByTagName('fileName').item(0).text;
                    Ti.API.info('value of filename =' + filename);

                    var status = deals.item(i).getElementsByTagName('status').item(0).text;
                    Ti.API.info('value of status =' + status);

                }
            } catch(e) {

            }
        }
    }
    xhr.send();
}

loadXML(url);
