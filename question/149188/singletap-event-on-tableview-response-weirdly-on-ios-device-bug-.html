"singletap" event on TableView response weirdly on IOS device (BUG ???)

I am using a TableView and TableViewRow to display the information (image and labels) retrieved from localhost database. I used "singletap" event to respond to the user tap on each row and then display detail information of that selected row. But "singletap" event does not response correctly for the 1st and 2nd row, but normally ok after the 2nd row. The row information (e.index, e.source etc) respond incorrectly !!! If i use "click" event, the code worked perfectly and respond correctly…….This problem does not happen on IOS simulator but on IOS device.

I tried on Ti SDK 2.1.4 and 3.0.2.GA.5a77fe7, IOS SDK 6.1 and iPhone 4 (iOS 5.1.1) but the problem happened.

Anyone has any idea? or solution ? Any help is much appreciated !!

  • Hi, attached please find my test code…

    Let me know if you need further information.

    
var nf_win = Ti.UI.currentWindow;

nf_mod();


function nf_mod() {

        var content_data = [];

        for (var c=0;c<20;c++)
            {                    

                var    row = Ti.UI.createTableViewRow({
                    height:'auto',
                    click_name:"table_row",
                    backgroundColor:'white',
                    selectedBackgroundColor:'white',
                    });

                var back_view = Ti.UI.createView({
                    height:200,        
                    backgroundColor:'red',
                    opacity: 1,
                    click_name: 'back_overlay',
                });


                var content_text = Ti.UI.createLabel({
                    left:5,
                    right:5,
                    top:65,        
                    font:{fontSize:18, fontFamily:'Avenir-Light'},
                    color:'white',
                    textAlign:Ti.UI.TEXT_ALIGNMENT_LEFT,
                    text:'testing.......'+c,                
                    height:150,                    
                    click_name:'content_text_label',
                    backgroundColor:'blue',
                    });

                var front_view = Ti.UI.createView({
                    width:150,
                    height:400,
                    backgroundColor:'green',
                    opacity: 0.5,
                    click_name: 'front overlay',
                });


                row.add(back_view);
                row.add(content_text);
                row.add(front_view);                

                content_data.push(row);                            

            };     

        var table = Ti.UI.createTableView({
            rowHeight:'auto',
            scrollable:true,
            data:content_data,
            backgroundColor:'white',
            click_name:'table_view',
            });

        nf_win.add(table);  
        nf_win.open();


//== 'SingleTap' event ===

        table.addEventListener('singletap',function(e){

            Ti.API.info('SingleTap - Content Selected :' + e.index + ' '+ e.source.click_name);                    

            alert("SingleTap - Content Selected :" + e.index + ' '+ e.source.click_name);            

        });


    return nf_win;
};
