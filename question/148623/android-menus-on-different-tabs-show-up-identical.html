Android: Menus on different tabs show up identical

Dear all,

seems like there is a bug in Androids menu handling. If I create a standard empty project (File, New Project, Tabbed Application) and use code below for ui/common/ApplicationTabGroup.js the project compiles OK.

With Titanium SDK 2.1.4 all works fine, menus on both windows show the correct menu.
With Titanium SDK 3.0.0 AND 3.0.2 the general activity is obviously filled with the menu entry on first menu call. So if I press <menu> on win2, menu2 is displayed BOTH on win1 and win2.

I learned that this bug (https://jira.appcelerator.org/browse/TIMOB-1512) was supposed to be fixed in 3.0.2, but it isn't.

Anyone made the same experiences ? Any workarounds that really work ? (invalidateOptionsMenu() did not work for me…)

Appreciate your help,
Ditmar

function ApplicationTabGroup(Window) {
    //create module instance
    var self = Ti.UI.createTabGroup();

    //create app tabs
    var win1 = new Window(L('home')),
        win2 = new Window(L('settings'));

    // ***************************************************************
    // *** win1
    // ***************************************************************   
    win1.activity.onCreateOptionsMenu = function(e){
        var menu = e.menu;

        // ***************************************************************
        // *** menu on win1
        // ***************************************************************   
        var menuItem = menu.add({
            title: 'menu win 1', 
            itemId : 1,
            groupId : 0,
            order : 1
        });
        menuItem.setIcon(Titanium.Android.R.drawable.ic_menu_add);
        menuItem.addEventListener("click", function(e) {
            alert('alert on win1');
        });
    };


    // ***************************************************************
    // *** win 2
    // ***************************************************************   
    win2.activity.onCreateOptionsMenu = function(e){
        var menu = e.menu;

        // ***************************************************************
        // *** menu on win2
        // ***************************************************************   
        var menuItem = menu.add({
            title: 'menu win 2', 
            itemId : 2,
            groupId : 0,
            order : 1
        });
        menuItem.setIcon(Titanium.Android.R.drawable.ic_menu_delete);
        menuItem.addEventListener("click", function(e) {
            alert('alert on win2');
        });
    };

    var tab1 = Ti.UI.createTab({
        title: L('home'),
        icon: '/images/KS_nav_ui.png',
        window: win1
    });
    win1.containingTab = tab1;


    var tab2 = Ti.UI.createTab({
        title: L('settings'),
        icon: '/images/KS_nav_views.png',
        window: win2
    });
    win2.containingTab = tab2;

    self.addTab(tab1);
    self.addTab(tab2);

    return self;
};

module.exports = ApplicationTabGroup;

Platform: Mac OSX 10.8.1, Android Emulator with 2.3.3 and 4.0

      function ApplicationTabGroup(Window) {
    //create module instance
    var self = Ti.UI.createTabGroup();

    //create app tabs
    var win1 = new Window(L('home')),
        win2 = new Window(L('settings'));

    // ***************************************************************
    // *** self
    // ***************************************************************   
    self.activity.onCreateOptionsMenu = function(e){
        var menu = e.menu;

        // ***************************************************************
        // *** menu on win1
        // ***************************************************************   
        var menuItem = menu.add({
            title: 'menu win 1', 
            itemId : 1,
            groupId : 0,
            order : 1
        });
        menuItem.setIcon(Titanium.Android.R.drawable.ic_menu_add);
        menuItem.addEventListener("click", function(e) {
            alert('alert on win1');
        });

        // ***************************************************************
        // *** menu on win2
        // ***************************************************************   
        var menuItem = menu.add({
            title: 'menu win 2', 
            itemId : 2,
            groupId : 0,
            order : 1
        });
        menuItem.setIcon(Titanium.Android.R.drawable.ic_menu_delete);
        menuItem.addEventListener("click", function(e) {
            alert('alert on win2');
        });
    };


    // ***************************************************************
    // *** tabs onFocus
    // ***************************************************************   
    var tab1 = Ti.UI.createTab({
        title: L('home'),
        icon: '/images/KS_nav_ui.png',
        window: win1
    });

    tab1.addEventListener('focus', function(e) {
         var menu = e.menu;

         menu.findItem(1).setVisible(true);
         menu.findItem(2).setVisible(false);

         e.tab1.window.activity.invalidateOptionsMenu();
    })
    win1.containingTab = tab1;


    var tab2 = Ti.UI.createTab({
        title: L('settings'),
        icon: '/images/KS_nav_views.png',
        window: win2
    });
    tab2.addEventListener('focus', function(e) {
         var menu = e.menu;

         menu.findItem(1).setVisible(false);
         menu.findItem(2).setVisible(true);

         e.tab2.window.activity.invalidateOptionsMenu();
    })
    win2.containingTab = tab2;

    self.addTab(tab1);
    self.addTab(tab2);

    return self;
};

module.exports = ApplicationTabGroup;
