Titanium ScrollView Doesn't work When TableViewRow is Added

In my code at the bottom of the scrollview, I have a tableview. The user can press a button to insert a new row at the top of the tableview. However, when they do this, the scrollview's content area doesn't grow and therefore, the first row (at the bottom of the tableview) doesn't show anymore. Each time the user add's a new row, the oldest one on the bottom doesn't show anymore. Is there any way to automatically grow the scrollview to fit it's children, including new tableview rows?

    Hi,

    Please check the below code. I hope, it will help to you

    var win = Ti.UI.createWindow({ backgroundColor: '#222', layout: 'vertical' });

var scrollView = Ti.UI.createScrollView({
  contentWidth: 'auto',
  contentHeight: 'auto',
  layout:"vertical",
  showVerticalScrollIndicator: true,
  showHorizontalScrollIndicator: true,
  height: '80%',
  width: '100%'
});
win.add(scrollView);

var aButton = Ti.UI.createButton({
    title : 'Click',
});

var i=1;
aButton.addEventListener('click', function() {
    var row=Ti.UI.createTableViewRow({ title: 'Apples'+i });
    aTableView.insertRowBefore(0,row);
    i++;
});

scrollView.add(aButton);

var aTableView = Ti.UI.createTableView();

var data = [
    {title:'Row 1'},
    {title:'Row 2'},
    {title:'Row 3'},
    {title:'Row 4'}
];
aTableView.setData(data);

scrollView.add(aTableView);

win.open();
