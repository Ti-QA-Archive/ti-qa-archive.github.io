  • GET Example

    var xmlhttp = Titanium.Network.createHTTPClient();
xmlhttp.onload = function(e)
{
};
xmlhttp.open('GET', 'www.abc.com?param1=var1&param2=var2');
xmlhttp.send();

    POST Example

    var xmlhttp = Titanium.Network.createHTTPClient();
xmlhttp.onload = function(e)
{
};
xmlhttp.open('POST', 'www.abc.com');
xmlhttp.send({param1:var1,param2:var2});
