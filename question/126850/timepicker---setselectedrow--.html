  • this is taken from KitchenSink:

    var value = new Date();
value.setFullYear(2009);
value.setMonth(0);
value.setDate(1);

var picker = Ti.UI.createPicker({
    type:Ti.UI.PICKER_TYPE_DATE_AND_TIME,
    value:value
});
