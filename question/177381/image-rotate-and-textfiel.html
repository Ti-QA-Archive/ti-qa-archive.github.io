    Hi Jofre,

    this code should work.

    var win = Ti.UI.createWindow({
    backgroundColor : 'black'
});
var textf = Ti.UI.createTextField({
    backgroundColor : 'white',
    top : 58,
    width : 80,
    height : 28,
    keyboardType : Titanium.UI.KEYBOARD_NUMBER_PAD
});
var image = Ti.UI.createImageView({
    image : "appicon.png",
    width : 64,
    height : 64,
    top : 130
});
var button = Ti.UI.createButton({
    title : "Presionar para girar",
    backgroundColor : '4BCCCC',
    color : 'black',
    width : 148,
    height : 48,
    top : 256
});
textf.addEventListener('change', function(e) {
    e.source.value = e.source.value.replace(/[^0-9]+/, "");
});
button.addEventListener('click', function(e) {
    var matriz = Ti.UI.create2DMatrix();
    image.animate({
        transform : matriz.rotate(textf.value),
        duration : 5000
    });
});
win.add(textf);
win.add(image);
win.add(button);
win.open();

    Matthias

    • Ok here you go.

      I added a button for toggling the plus-/minus-sign.
      The calculation is also implemented.

      var win = Ti.UI.createWindow({
    backgroundColor : 'black'
});
var togglePlusMinus = Ti.UI.createButton({
    top : 58,
    left : 100,
    title : '+',
    width : 30,
    color : 'black',
    backgroundColor : 'white'
});

togglePlusMinus.addEventListener('click', function(e) {
    if (togglePlusMinus.title === '+') {
        togglePlusMinus.title = '-';
    } else {
        togglePlusMinus.title = '+';
    }
});
win.add(togglePlusMinus);

var textf = Ti.UI.createTextField({
    backgroundColor : 'white',
    top : 58,
    left : 140,
    width : 80,
    height : 28,
    keyboardType : Titanium.UI.KEYBOARD_NUMBER_PAD
    //Titanium.UI.KEYBOARD_NUMBERS_PUNCTUATION
});
var image = Ti.UI.createImageView({
    image : "appicon.png",
    width : 64,
    height : 64,
    top : 130
});
var button = Ti.UI.createButton({
    title : "Presionar para girar",
    backgroundColor : '4BCCCC',
    color : 'black',
    width : 148,
    height : 48,
    top : 256
});
textf.addEventListener('change', function(e) {
    e.source.value = e.source.value.replace(/[^0-9]+/, "");
    if (e.source.value > 90) {
        Alert('max 90Â° is allowed');
        textf.value = '90';
    }
});
button.addEventListener('click', function(e) {
    var matriz = Ti.UI.create2DMatrix();
    // calculate rotation, depending on plus minus
    image.animate({
        transform : matriz.rotate((togglePlusMinus.title === '-') ? textf.value * -1 : textf.value),
        duration : 5000
    });
});
win.add(textf);
win.add(image);
win.add(button);
win.open();
