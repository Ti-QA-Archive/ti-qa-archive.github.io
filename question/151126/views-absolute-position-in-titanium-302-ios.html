  • Hi Bharat ,

    Try this code, you will get the X and Y coordinate of the screen:

    var win = Ti.UI.createWindow({
 backgroundColor : "red",
});

win.addEventListener('touchstart', function(e) {
  alert("x--->"+e.x + " y--->" + e.y);

 Ti.API.info("x--->"+e.x+ " y--->" + e.y)
});

win.open();
