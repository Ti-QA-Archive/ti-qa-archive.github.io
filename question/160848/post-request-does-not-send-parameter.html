POST Request does not send Parameter

Hi,

I have problems to send parameter via POST HTTPRequest. My source code calls the url but no parameter is send ;-/ Here my configuration: Titanium Studio, build: 3.2.0.201312191547 and my source code snipplet:

var loginReq = Titanium.Network.createHTTPClient();

function doClick(e) {
if ($.username.value != '' && $.password.value != '') {
loginReq.open("POST","http://httpbin.org/post");
loginReq.setRequestHeader('content-type', 'application/json');
var params = {
username: $.username.value,
password: Ti.Utils.md5HexDigest($.password.value)
};
loginReq.send(JSON.stringify(params));
} else {
alert("Username/Password are required");
}
}

Can anyone help me?

  • Curious how you know its 'not working'?

    I ask, because you have no onload function to tell you the results of a successful connection and you have no onerror to let you know of a failure.

    Following simplified test function works fine, 

    function doClick(e) { 
var loginReq = Titanium.Network.createHTTPClient();
loginReq.open("POST","http://httpbin.org/post");
loginReq.setRequestHeader('content-type', 'application/json'); 
loginReq.onload = function(e){
    console.log(loginReq.status);
    console.log(loginReq.responseText);
}
var params = {username: 'testUser', password: Ti.Utils.md5HexDigest('testPassword')};
loginReq.send(JSON.stringify(params));
}
