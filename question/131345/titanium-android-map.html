  • var win = Ti.UI.createWindow();

    Create Map view and add in window

    var mapview = Titanium.Map.createView({
    mapType : Titanium.Map.STANDARD_TYPE,
    animate : true,
    regionFit : true,
    userLocation : true
});
win.add(mapview);
win.open();

    Create Annotation and add in mapview

    var atlantaParams = {
    latitude:37.389569,
    longitude:-122.050212,
    title:'your location',
    subtitle:'your address',
    animate:true,
    leftButton: Titanium.UI.iPhone.SystemButton.DISCLOSURE,
    rightButton: Titanium.UI.iPhone.SystemButton.DISCLOSURE,
    myid:3 // CUSTOM ATTRIBUTE THAT IS PASSED INTO EVENT OBJECTS
};

if (!isAndroid) {
    atlantaParams.pincolor = Titanium.Map.ANNOTATION_PURPLE;
} else {
    atlantaParams.pinImage = "images/map-pin.png";
}
var atlanta = Titanium.Map.createAnnotation(atlantaParams);

mapview.addAnnotation(atlanta);

    This works fine ………..
