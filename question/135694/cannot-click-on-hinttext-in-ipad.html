  • Hi Dhamodaran R
    Try This

    var win = Ti.UI.createWindow({
    backgroundColor: '#fff'
});
var aTextField = Ti.UI.createTextField({
    height : 35,
    top : 10,
    left : 40,
    width : Ti.UI.FILL,
    hintText : 'This is hint text',
    borderColor:'red'

});
aTextField.addEventListener('return', function(e) {
    aTextField.blur();
    alert('Input was: ' + aTextField.value);
});
win.add(aTextField);
win.open();
