    • should be

      
var win = Ti.UI.createWindow({
   fullscreen:true,
   backgroundImage:'yourimage.jpg'
});


function timeOutWin(){
   //wait 10 seconds, then close win and open tabgroup
   setTimeout(function() {
     win.close(); 
     tabGroup.open();
   }, 10000); 

}

//launch timeout when window is open
win.addEventListener('open', function(){
   timeOutWin();
});
