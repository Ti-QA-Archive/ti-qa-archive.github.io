adding a button on a textfield does not work on android

Hello,

Ik want to add a button inside a textfield. This does work on IOS but off course it does not work for Android. Here is my code:

var usernameField = Ti.UI.createTextField({
    color:'black',
    height:textFieldHeight,
    top:5,
    left:left,
    width:250,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
    value: Settings.getUsername()
});
usernameField.addEventListener("change", function(e){
    Settings.setUsername(e.value);
});
var addUsernameButton = Ti.UI.createButton({
    height: 30,
    width: 30,
    right: 5,
    top:2,
    bottom: 2,
    backgroundImage:'../../images/add-off.png',
    backgroundSelectedImage:'../../images/add-on.png',
    backgroundDisabledImage: '../../images/add-off.png',
});
addUsernameButton.addEventListener("click", function(e){
    Settings.addUsername(usernameField.value);
});
usernameField.add(addUsernameButton);

Does anybody knows why i can't add another view to a textfield on Android?
