  • Answer repeated here for completeness

    var fb = require('facebook');
fb.appid = 'xxxxxxx';
fb.permissions = ['user_photos'];

fb.addEventListener('login', function(e) {
    if (e.success) {
        fb.requestWithGraphPath('me?fields=picture', {}, 'GET', function(e) {
            if (e.success) {
                var meData = JSON.parse(e.result);
                alert(meData.picture.data.url);
                $.profilePicture.setImage(meData.picture.data.url);
            } else if (e.error) {
                alert(e.error);
            } else {
                alert('Unknown response');
            }
        });
    } else if (e.login) {
        alert('you already signed in');
    } else if (e.cancelled) {
        alert('cancelled')
    }
});

fb.authorize();
