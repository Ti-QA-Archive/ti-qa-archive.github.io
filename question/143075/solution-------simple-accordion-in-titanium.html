Solution ----- Simple Accordion In Titanium


var window = Titanium.UI.createWindow();

var view = Titanium.UI.createView
({
   backgroundColor:'red',
   layout:'horizontal'
});
window.add(view);

for(var i = 0 ; i < 5 ; i++ )
{
    var view_b = Titanium.UI.createView
    ({
       backgroundColor:'green',
       top:10,
       height:50
       // layout:'horizontal'
    });
    view.add(view_b);

    var but1 = Titanium.UI.createButton
    ({
        title:'show',
        height:20
    });
    view_b.add(but1);

    but1.addEventListener('click' , function()
    {
        var par = this.getParent();

        for(var i = 0 ; i < par.children.length ; i++ )
        {
            par.children[i].show();
            par.height = 50;
        }

    });


    var view_c = Titanium.UI.createView
    ({
       backgroundColor:'green',
       top:0,
       height:50,
       zIndex:1
       // layout:'horizontal'
    });
    view_b.add(view_c);

    var but = Titanium.UI.createButton
    ({
        title:'del'
    });
    view_c.add(but);

    but.addEventListener('click',function(e)
    {
        var par = this.getParent();
        var p_par = par.getParent();

        p_par.height = 20;
        par.hide();

    });
}
