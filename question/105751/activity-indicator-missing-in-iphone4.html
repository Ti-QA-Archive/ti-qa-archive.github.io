Activity Indicator missing in Iphone4

Hi,

I'm trying to show an Activity Indicator in a webview while my page is loading. It works well in the Simulator on low-res iphone, but not in Retina-display mode or on my actual iphone4.

Any thoughts?

var wv = Titanium.UI.createWebView({  
    url: Titanium.App.Properties.getString("baseurl")+'specials.html',
    top:0,
    scalesPageToFit: false,
    backgroundColor:'#fff'
});
var actind;
wv.addEventListener('beforeload',function(e){
    actInd = Titanium.UI.createActivityIndicator({
      style:Titanium.UI.iPhone.ActivityIndicatorStyle.BIG,
      height:120,
      width:120
    });
    wv.add(actInd);
    actInd.show();
});
wv.addEventListener('load',function(e){ actInd.hide(); });
