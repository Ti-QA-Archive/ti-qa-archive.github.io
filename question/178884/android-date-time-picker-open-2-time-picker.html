    You are opening the same Picker instance in its own success callback. You'll have to separate them pickers into two instances. E.g. you could do this:

    var date;
var datePicker = Ti.UI.createPicker();

var time;
var timePicker = Ti.UI.createPicker();

datePicker.showDatePickerDialog({
  callback: onDatePicked
});

function onDatePicked(e) {
  if (!e.cancel){

    date = e.value;

    timePicker.showTimePickerDialog({
      callback: onTimePicked
    });

  }
}

function onTimePicked(e) {
  if (!e.cancel){
    time = e.value;
  }
}

    I hope this makes sense. Didn't test the code but it should work something like this (apart from processing the picked values which I left out here).
