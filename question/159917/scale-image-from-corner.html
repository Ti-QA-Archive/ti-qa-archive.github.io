Scale image from corner

Am animating (scaling) an imagen like:

var zoomScale = false;
imagenNota.addEventListener('doubletap', function(e) {
        var t = Titanium.UI.create2DMatrix();
        if (!zoomScale){    
            t = t.scale(2.0);
              $.viewContenidoFoto.animate({
                transform:t,
                right: 10,
                bottom: 100,
                duration:500
            });
            zoomScale = true; 
        }else{
            $.viewContenidoFoto.animate({
                transform:t,
                duration:500
            });
            zoomScale = false;
        }
    });

    anchorPoint

    permalink
    1 Comment

    • If someone else need it:

      imagenNota.addEventListener('doubletap', function(e) {
        var t = Titanium.UI.create2DMatrix();
        $.viewContenidoFoto.anchorPoint = {x:1, y:1};
        if (!zoomScale){    
            t = t.scale(2.5);
              $.viewContenidoFoto.animate({
                transform:t,
                duration:500
            });
            zoomScale = true; 
        }else{
            $.viewContenidoFoto.animate({
                transform:t,
                duration:500
            });
            zoomScale = false;
        }
    });
