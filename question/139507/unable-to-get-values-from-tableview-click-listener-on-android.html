Unable to get Values from TableView Click Listener on Android

I am binding multiple rows in Table View and successfully able to get values along with ID on iPhone from TableView Click Listener but my application is crashing on Android when I am retrieving values on Table View Click. My code as below


    var table = Titanium.UI.createTableView({
        //data:data,
        allowsSelection:'YES',
        top: 87,
        height:_tableHeight,
        filterAttribute:'filter',
        borderColor:'#fff'
    });

    var tableRowdata = [];

    function AddToTable() {

        for (var i = 0; i < catArr.length ; i++)
        {

                var row = Ti.UI.createTableViewRow({
                hasChild: true,
                height: 40,
                test:'Check',
                top: 'Business',
                catId: catArr[i].catID,
                rowIndex: i,
                //filter: catAllArr[i].catName,
                category_name: catArr[i].catName
            });

            var titleLabel = Ti.UI.createLabel({
                text: catArr[i].catName,
                font: {fontSize: 14, fontFamily:'Arial'},
                left: 20,
                textAlign: 'left',
                color: '#000',
                height: 20,
                width: 260,
            });
            row.add(titleLabel);

            tableRowdata.push(row);        
        }


        if (tableRowdata.length > 0) {
            table.data = [];
            table.data = tableRowdata;
        }

    }

    table.addEventListener('click', function(e)
    {
        Ti.API.info(e.index);
        Ti.API.info(e.row.rowIndex);
        Ti.API.info(e.rowData.test);    
        Ti.API.info(e.rowData.rowIndex);
        Ti.API.info(e.rowData.catId);
    }
