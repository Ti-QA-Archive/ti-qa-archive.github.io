    i think you are looking at the wrong example in kitchen sink, look at picker_date2.js

    
var value = new Date();
value.setMinutes(0);
value.setHours(12);
value.setSeconds(00);

var picker = Ti.UI.createPicker({
    type:Ti.UI.PICKER_TYPE_TIME,
    value:value
});
