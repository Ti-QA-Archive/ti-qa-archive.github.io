ACS Cloud.Users.login callback function doesn't get called on Android

I have an issue using Cloud.Users.login method. My code is as follows. The callback function is called and works fine on iOS simulator but it doesn't get called on Android. Anyone encounter such an issue before? How to solve this? 

            // add event listener to call login function
    firstView.addEventListener('login', onLogin);

    // onLogin function
    function onLogin(e) {
        Cloud.Users.login({
            login: e.email,
            password: e.password
        }, function(e) {    // callback function not called on Android
            if (e.success) {
            var user = e.users[0];
                alert('Success:\n' +
                    'id: ' + user.id + '\n' +
                    'first name: ' + user.first_name + '\n' +
                    'last name: ' + user.last_name);
                mainView.setUser(user.first_name + user.last_name);
                mainWindow.open();
            } else {
                alert('Error:\n' +
                    ((e.error && e.message) || JSON.stringify(e)));
            }
        });
    }
