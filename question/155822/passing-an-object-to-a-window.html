Passing an object to a window

Hello all, I would like to pass an object (user) to a window (mapWindow) so I tried to consider my window as an object. My object mapWindow.js start with a constructor :
mapWindow.js:
Ti.include('user.js');

function MapWindow(user){
this.screen = Titanium.UI.createWindow({
title:'Carte',
backgroundColor:'#fff',
});
Titanium.UI.setBackgroundColor('#000');
this.screen.open();
this.user = user;
}

in app.js:
Ti.include ("user.js");
Ti.include("mapWindow.js");
var me = new User("JP","0619232516");
var mapWindow = new MapWindow(me);

I have this error:
[ERROR] : message = "-[NSNull krollObjectForBridge:]: unrecognized selector sent to instance 0x3a22678";

Someone could advice me?
