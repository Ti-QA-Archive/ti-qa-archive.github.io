  • Hi Graham,

    Have you tried with fb.requestWithGraphPath?? You should give it a try. Here is a sample/example snippet for that. 

    var data = {
        link : "http://www.appcelerator.com",
        name : "Appcelerator Titanium",
        tags : "1234,5678",
        accesstoken : "XXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX",
        message : "Tagging is much easier now  ",
        place : "XXXXXXXX",
        caption : "Appcelerator Titanium",
        picture : "http://developer.appcelerator.com/assets/img/DEV_titmobile_image.png",
        description : "You've got the ideas, now you've got the power. Titanium translates your hard won web skills..."
    };
    fb.requestWithGraphPath('me/feed', data, 'POST', function(e) {
        if (e.success) {
            alert("Success!  From FB: " + e.result);
        } else {
            if (e.error) {
                alert(e.error);
            } else {
                alert("Unkown result");
            }
        }
    });
});

    Report back with your result please.
