  • Wrap the main view into a scrollview would help:

    var win = Titanium.UI.createWindow({
    backgroundColor : 'white'
});

var mainView = Ti.UI.createView({width: '100%', height: '100%'});
var tf = Ti.UI.createTextField({
    height: 40,
    bottom: 100,
    borderColor: 'black'
});
mainView.add(tf);

var scrollView = Ti.UI.createScrollView({width: '100%', height: Ti.Platform.displayCaps.platformHeight + 10});
scrollView.add(mainView);
win.add(scrollView);
win.open();

tf.addEventListener('blur', function(){
    scrollView.scrollTo(0, 0);    
});
