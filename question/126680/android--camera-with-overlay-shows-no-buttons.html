  • Here is my code. I have no problem with the landscape mode of the camera, i can rotate my images. But when i use the overlay, camera buttons don't appear. 

    var cameraoverlay = Titanium.UI.createImageView({
    width:'480dp',
    height:'320dp',
    top:0,
    touchEnabled:false,
    image:'images/skins/camera/'+Ti.App.Properties.getInt('skin')+'.png'
});

var cameraTransform = Ti.UI.create2DMatrix();
cameraTransform = cameraTransform.scale(1);

Titanium.Media.showCamera({
    success: function(event){
        manipulate(event.media,selectView,true);    //function that creates a new view with the picture taken
    },
    cancel: function(){
        win1.close();
    },
    error: function(error){
        var a = Titanium.UI.createAlertDialog({
            title: 'Camera'
        });
        if (error.code == Titanium.Media.NO_CAMERA) {
            a.setMessage('? ??????? ??? ??? ???????? ??????????? ??????');
        }
        else {
            a.setMessage('Unexpected error: ' + error.code);
        }
        a.show();
        win1.close();
    },
    saveToPhotoGallery: false,
    showControls:true,
    overlay: cameraoverlay,
    transform:cameraTransform
});
