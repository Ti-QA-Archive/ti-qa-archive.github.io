  • Have you read the shareKit module configuration.html file?
    To start with, your sharekit.configure is incomplete.
    my_app_name and my_app_url are REQUIRED according to the docs

    Code I know works:

    In app.js setup sharekit:

    var sharekit = require("com.0x82.sharekit");
    Ti.API.info("module is => " + sharekit);
    sharekit.configure({
        my_app_name: 'MyCoolApp',
        my_app_url: 'http://mycoolapp.url',
        share_menu_alphabetical_order: true,
        shared_with_signature: false,
        ipad_modal_presentation_style: "UIModalPresentationFormSheet",
        ipad_modal_transition_style: "UIModalTransitionStyleCoverVertical",
        bar_style: "UIBarStyleDefault",
        form_font_color_red: -1,
        form_font_color_green: -1,
        form_font_color_blue: -1,
        form_bg_color_red: -1,
        form_bg_color_green: -1,
        form_bg_color_blue: -1,


        facebook_key: 'my_facebook_key',
        facebook_secret: 'my_secret_agent_facebook_key',
        facebook_use_session_proxy: false,
        facebook_session_proxy_url: '',
      });

    In the file where we want to call shareKit:

    btnShareKit = Titanium.UI.createButton({
        title: L('share_the_app'),
        backgroundColor: '#262626',
        color: '#fff',
        height: 35,
        width: 300,
        top: 278,
        borderRadius: 0,
        borderWidth: 0,
        backgroundImage : 'button.png'
    });

btnShareKit.addEventListener('click',function(){
        sharekit.share({
            title: "Download My Ultra Cool App!",
            view: btnShareKit,
            link: 'http://itunes.apple.com/us/app/conversion-center/id474744543?mt=8'
        });
    });

            if(sharer == 'Facebook') {
            Ti.Facebook.appid = 'MYAPPID';

                        title : MYTITLE,

                        sharer : sharer

                    alert(e.error);

                    alert("Canceled");

            });
        } else {
            sharekit.share({
