    there is no way for doing it with buttons,you can use labels with one height or width as auto to gain line breaks
    I have done this using label ,have a look at this,may it help you :)

    //Enput text in text are and make it label text on button click

    var txtfldvalue;
    var btn = Titanium.UI.createButton({
    title:"It is long",
    width: 'auto',
    height: 'auto',
    top:10
    });
    var btntext= Titanium.UI.createLabel({
    height:80,
    width:'auto',
    text:'i am here for long data',
    backgroundColor:'pink'
    })

    var txtarea= Ti.UI.createTextArea({
    top:100,
    hintText:'Entr Long Text value ',
    value:'',
    backgroundColor:'blue',
    height:'auto',
    width:'auto'
    })

    btn.addEventListener('click',function(){
    btntext.text=txtarea.value;
    })

    win1.add(btn);
    win1.add(txtarea);
    win1.add(btntext);
