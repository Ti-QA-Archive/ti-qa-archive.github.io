        deviceToken = e.deviceToken
        loginDefault();
    },
    error : function deviceTokenError(e) {
        alert('Failed to register for push! ' + e.error);
    }
});
function loginDefault(e) {
    //Create a Default User in Cloud Console, and login
    Cloud.Users.login({
        login : 'jesse',
        password : 'jesse'
    }, function(e) {
        if (e.success && Ti.Platform.osname == 'android') {

            defaultSubscribe();
        } else if (e.success && Ti.Platform.osname == 'iphone') {
            registerForPush();
        } else {
            alert('Error: ' + ((e.error && e.message) || JSON.stringify(e)));
        }
    });
}

function defaultSubscribe() {
    Cloud.PushNotifications.subscribe({
        channel : 'alert',
        device_token : deviceToken,
        type : 'android'
    }, function(e) {
        if (e.success) {

        } else {
            alert('Error:' + ((e.error && e.message) || JSON.stringify(e)));
        }
    });
}

function registerForPush() {
    Cloud.PushNotifications.subscribe({
        channel : 'demo_alert',

    });
}

var rurl;
Ti.App.rssurl = '';
if (Ti.App.rsscounter == 1)
    rurl = "http://riqapp.be/index.html";
else if (Ti.App.rsscounter == 2)
    rurl = "http://riqapp.be/brisbane-quaddie.html";
else if (Ti.App.rsscounter == 3)
    rurl = "http://riqapp.be/lays-of-the-day.html";
else if (Ti.App.rsscounter == 4)
    rurl = "http://riqapp.be/sports-best-bet.html";

Ti.App.rssurl = rurl;
var tableView1 = Titanium.UI.createTableView();
QLDRssWin.add(tableView1);
var Row = [];
Titanium.Yahoo.yql('select content from html where url="' + rurl + '" and xpath="//h5"', function(e) {
    var results = e.data.h5;

    var length = 0;
    if ( typeof results === "string") {
        length = 1;
    } else {
        length = results.length;
    }

    for (var i = 0; i < length; i++) {
        var rss = results[i];

        var rssRow = Titanium.UI.createTableViewRow({
            height : 100,
            className : 'rssrow'

        });
        var titleLabel = Titanium.UI.createLabel({
            text : rss,
            font : {
                fontSize : 30,
                fontWeight : 'bold'
            },
            width : 'auto',
            top : '25%',
            left : '20%',
            height : 40,
            color : 'black'
        });

        rssRow.add(titleLabel);

        Row.push(rssRow);
    };

    tableView1.setData(Row);
});
tableView1.addEventListener('click', function(e) {
    Ti.App.index = e.index
