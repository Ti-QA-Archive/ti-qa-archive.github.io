    You should check to make sure longitude & latitude are valid arguments for createAnnotation()

    if( sitesDataArr[j].rows[i].extraValue && 
    sitesDataArr[j].rows[i].extraValue.longitude && 
    sitesDataArr[j].rows[i].extraValue.latitude ) {

    siteAnnotation = Titanium.Map.createAnnotation({
        longitude : sitesDataArr[j].rows[i].extraValue.longitude,
        latitude : sitesDataArr[j].rows[i].extraValue.latitude,
        title : sitesDataArr[j].rows[i].value,
        siteId : sitesDataArr[j].rows[i].extraValue.siteId,
        pincolor : Ti.Map.ANNOTATION_RED,
        animate : true,
    });
}
