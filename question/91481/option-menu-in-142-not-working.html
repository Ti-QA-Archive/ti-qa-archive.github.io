Option Menu in 1.4.2 Not Working

why doesn't this work? The only way I can get a menu to work is if I create the menu in a .js file associated to a window


win2.addEventListener('open', function(e) {
    win2.menu = Titanium.UI.Android.OptionMenu.createMenu();
    Ti.API.info('menu ' + win2.menu);
    var _shareMenu = Titanium.UI.Android.OptionMenu.createMenuItem({
        title:'Share Item 2', 
        icon: 'Resources/KS_nav_views.png'});
    Ti.API.info('shareMenu 2' + _shareMenu);

    _shareMenu.addEventListener('click', function() {
        Ti.API.info('menu clicked');
    });

    win2.menu.add(_shareMenu);
    Titanium.UI.Android.OptionMenu.setMenu(win2.menu);
});
