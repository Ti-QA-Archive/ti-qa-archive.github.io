Open "popup" window after splash screen

I am trying have a window automatically open after the splash screen on ios so that the user can read some text and then close it and go on to the application but I can't seem to get it to work as the "popup" window opens but then goes behind the application window.

Titanium.UI.setBackgroundColor('#fff');

var tabGroup = Titanium.UI.createTabGroup();

var win1 = Titanium.UI.createWindow({  
    title:'Tab1',
       url:'/main_windows/tab1',
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
   title:'Tab1',
    window:win1
});

var win2 = Titanium.UI.createWindow({  
    title:'Tab2',
       url:'/main_windows/tab2.js',
});
var tab2 = Titanium.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Tab2',
    window:win2
});

tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);  


var w = Titanium.UI.createWindow({
    height:0,
    backgroundColor:'#000',
    bottom:0
});

var a = Titanium.UI.createAnimation();
a.height = 420;
a.duration = 300;

var b = Titanium.UI.createButton({
    title:'Close',
    height:30,
    width:150
});
    w.add(b);
    b.addEventListener('click', function()
{
    a.height = 0;
    w.close(a);
});

    w.open(a);

tabGroup.open();
win1.add(w);
