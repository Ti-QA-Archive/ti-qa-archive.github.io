    It would work something like this:

    var win = Ti.UI.createWindow({
    backgroundColor:'#fff'
});

var view = Ti.UI.createView({
    width:100,
    height:100,
    backgroundColor:'red'
});
win.add(view);

space.addEventListener('click', function(e) {
    var isTop = (e.y < (view.height / 2));
    var isBottom = (e.y > (view.height / 2));
    var isLeft = (e.x < (view.width / 2));
    var isRight = (e.x > (view.width / 2));
    // branches second
    if (isTop && isLeft) {
       // do things that ONLY apply if top and left
       alert('This is top and left');
    }
    if (isBottom) {
       // do things that ONLY apply if bottom
       alert('this is bottom');
    }
})


win.open();
