  • I think you are struggling in parsing of xml file.To parse xml, Titanium provides API in Titanium.XML package. You can get many helpful functions in this.For a sample code you can refer this

    var result = this.responseText;
var xml = Ti.XML.parseString(result);

var params = xml.documentElement.getElementsByTagName("tag1");
var name = xml.documentElement.getElementsByTagName("tag2");
var value = xml.documentElement.getElementsByTagName("tag3");
var data = [];

for (var i=0;i<params.item.length;i++) {
    Ti.API.log('Param '+i+': Name: '+n.item(i).text);
    Ti.API.log('Param '+i+': Value: '+v.item(i).text);

    // Add to array
    data.push({"name":n.item(i).text,"value":v.item(i).text});
}
