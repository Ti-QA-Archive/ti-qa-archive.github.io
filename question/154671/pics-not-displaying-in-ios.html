  • var cellWidth = Ti.Platform.displayCaps.platformWidth / 4;
var cellHeight = Ti.Platform.displayCaps.platformHeight / 6;
var xSpacer = 20;
var ySpacer = 20;
var xGrid = 3;
var yGrid = lib.datamephotos.length / 3;

var tableData = [];
var colorSetIndex = 0;
var cellIndex = 0;

for (var y = 0; y < yGrid; y++) {
    var thisRow = Ti.UI.createTableViewRow({
        className : "grid",
        layout : "horizontal",
        height : cellHeight + (2 * ySpacer)

    });
    for (var x = 0; x < xGrid; x++) {
        var thisView = Ti.UI.createView({
            objName : "grid-view",
            objIndex : cellIndex.toString(),
            backgroundImage : lib.datamephotos[colorSetIndex].picture,
            left : "19sp",
            height : cellHeight,
            width : cellWidth,
            top : "30sp"

        });
        //var images=Ti.UI

        thisRow.add(thisView);
        cellIndex++;
        colorSetIndex++;

        if (colorSetIndex === lib.datamephotos.length) {
            colorSetIndex = 0;
        }
    }
    tableData.push(thisRow);
}
var tableview = Ti.UI.createTableView({
    data : tableData,
    separatorColor : 'transperent',
    top:"5%"

});
tableview.addEventListener("click", function(e) {
    if (e.source.objName) {
        Ti.API.info("---> " + e.source.objName + " " + e.source.objIndex + " " + e.source.backgroundImage + " " + "was clicked!");

        lib.selectedimage = e.source.backgroundImage;

        //alert(lib.selectedimage);

        var Alloy = require('alloy');

        var con = Alloy.createController('selectpicture');
        con.getView().open();

    }

});

$.gridwindow.add(tableview);
