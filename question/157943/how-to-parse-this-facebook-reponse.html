    var xhr = Ti.Network.createHTTPClient();
    xhr.setTimeout(5000);
    xhr.open("GET", "https://graph.facebook.com/233729955181?fields=feed&access_token=210665145671703%7C2GFZgdD3y5uR2I-GfxvzIncSjRY");
    xhr.onerror = function(e) {
        alert('Oh Dear, There was a problem with Facebook\n\n' + e.error);
    };

    xhr.onload = function() {

        var data = JSON.parse(this.responseText);

        Ti.API.info('Data'+this.responseText);
        Ti.API.info('Data.ID'+data.id);
        Ti.API.info('Data.ID'+data.feed.data.length);

        for(var i=0; i<data.feed.data.length;i++){
            Ti.API.info('Picture'+data.feed.data[i].picture);
            Ti.API.info('Link'+data.feed.data[i].link);
        }


        Ti.API.debug(data);
    }
    xhr.send();
