Gallery realized with transitions

Hi,
i try a little gallery like examplein KitchenSink (transitions):

function createGallery(total) {
    var v = Ti.UI.createView({});
    var imageView = Titanium.UI.createView({  //container
        height : '36%',
        width : '100%',
        bottom : 30
    });
    var images = [];
    for(var i = 0; i < total; i++) {
        images[i] = Ti.UI.createView({
            backgroundImage : Ti.Filesystem.resourcesDirectory + 'assets/k' + i + '.png',
            height : '100%',
            width : '100%',
            borderRadius : 6
        });
    }
    for(var i = 0; i < total - 1; i++) {
        images[i].addEventListener('click', function() {
            imageView.animate({
                view : images[i + 1],
                transition : Ti.UI.iPhone.AnimationStyle.CURL_UP
            });
        });
    }
    images[total - 1].addEventListener('click', function() {
        imageView.animate({
            view : images[0],
            transition : Ti.UI.iPhone.AnimationStyle.CURL_UP
        });
    });
    v.add(imageView);
    for(var i = total - 1; i >= 0; i--) {
        imageView.add(images[i]);
    }
    return v;
}

But on every click comes the same image. I dont see the mistake…

Rainer
