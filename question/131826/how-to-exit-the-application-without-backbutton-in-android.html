    • win.addEventListener('android:back', function(e) {

      var dialog = Titanium.UI.createAlertDialog({

      message: 'Want to Exit Application !',
buttonNames: ['Yes','No'],

      });
      dialog.show();

      dialog.addEventListener('click',function(e)
      {

      if(e.index == 0){

          win.close();

    }

      });

      });
