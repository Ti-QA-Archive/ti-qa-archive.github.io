set_tab.addEventListener('click', function(e){
    if(e.index==4) {    

        if(Titanium.Network.online) {
    var rd = e.rowData;
    var uri = rd.webs;

    Titanium.API.info(uri);
      var win = Titanium.UI.createWindow();
      var wview = Titanium.UI.createWebView({url:uri});
     win.add(wview);
     Titanium.UI.currentTab.open(win);
     }
     else {
         alert("You are not connected to Internet");
     }
    }
});

    Here rd.webs has the url to the website.
