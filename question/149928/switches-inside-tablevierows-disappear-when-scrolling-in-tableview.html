    Hi

    I have not experienced the problem you are seeing, so as your code has some dependant functions that are not included, so I thought I would create a test app for you to try.

    To test this simply create a new default project and replace the contents of app.js with the code below;

    Ti.UI.setBackgroundColor('transparent');

var win = Ti.UI.createWindow({
    backgroundColor: '#fff',
    exitOnClose: true,
    height: Ti.UI.FILL,
    width: Ti.UI.FILL
});
win.open();

var addRow = function (o) {
    var obj = o || {};
    var row = Ti.UI.createTableViewRow({
        backgroundColor: '#fff',
        height: Ti.UI.SIZE,
        width: Ti.UI.FILL
    });
    var lbl = Ti.UI.createLabel({
        font: {
            fontSize: 14,
            fontWeight: 'normal'
        },
        height: Ti.UI.SIZE,
        left: 10,
        text: obj.title,
        width: Ti.UI.SIZE
    });
    row.add(lbl);
    var sw = Ti.UI.createSwitch({
        right: 10,
        value: obj.value
    });
    row.add(sw);
    return row;
};
var intSection = 0, intSections = 10, section, sections = [];
var intRow = 0, intRows = 50, row, rows = [];
var intRand = 0;
for (intSection = 0; intSection < intSections; intSection = intSection + 1) {
    section = Ti.UI.createTableViewSection({
        headerTitle: 'Section ' + (intSection + 1)
    });
    for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
        intRand = Math.floor((Math.random()*10)+1); 
        row = addRow({
            title: 'Section ' + (intSection + 1) + ', Line ' + (intRow + 1),
            value: ((intRand < 3) ? true : false)
        });
        //rows.push(row);
        section.add(row);
    }
    sections.push(section);
}
var tbl = Ti.UI.createTableView({
    data: sections,
    height: Ti.UI.FILL,
    minRowHeight: 44,
    width: Ti.UI.FILL
});
win.add(tbl);
win.open();

    You will see that even with 10 sections and 500 total rows, each with a switch and a random true/false answer - it works without hiding at any scroll speed.
