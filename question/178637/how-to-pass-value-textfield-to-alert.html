    Hi Sameera Mudalige,

    Thanks for your concern.

    Please check my test code below.

    var win = Ti.UI.createWindow({
    backgroundColor:'#fff',
});

var aTextField = Ti.UI.createTextField({
    height : 35,
    top : 10,
    left : 40,
    width : 240,
});

var aButton = Ti.UI.createButton({
    title : 'Click',
    height : 'auto',
    width : 'auto',
    top : 55,
});

aButton.addEventListener('click', function() {
    alert(aTextField.value);
});

win.add(aTextField);
win.add(aButton);
win.open();

    This is the way of getting the text field value. Now make sure you that you are doing it right or wrong and text field is on the scope where you use it.

    If you have any concern, then feel free to let us know.

    Thanks

    • Or you can follow this pattern of code.

      var win = Ti.UI.createWindow({
    backgroundColor:'#fff',
});

var txtFld = Ti.UI.createTextField(textFieldProperty());

var aButton = Ti.UI.createButton(buttonProperty());
aButton.title="click";

aButton.addEventListener('click', function() {
    alert(txtFld.value);
});

function textFieldProperty() {
    var proper = {
        color : '#000',
        height : Ti.UI.SIZE,
        width : '50%',
        top: 10,
    };
    return proper;
}

function buttonProperty() {
    var proper = {
        height : 'auto',
        width : 'auto',
        top : 55,
    };
    return proper;
}

win.add(txtFld);
win.add(aButton);
win.open();
