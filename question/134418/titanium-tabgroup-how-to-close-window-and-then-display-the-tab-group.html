Titanium TabGroup: How to close window and then display the tab Group?

I have following flow of app :

First screen is Login screen and if Login gets success the tab-group opens.

here is code :

app.js

var win = Titanium.UI.createWindow
({
    title:'User Login',
    url:'Login.js',
    tabBarHidden:true,
    backgroundColor:'gray',
    navBarHidden:false
});
win.open();

Login.js

var win = Ti.UI.currentWindow;

// some UI controls

loginBtn.addEventListener('click', function(e)
{
   //calling web service     
   if(isSuccess == 1)
   {
       var tabGroup = Titanium.UI.createTabGroup();    
       // code to create Tab 
       tabGroup.open();   
   }
}

Now if I hide the currentWindow (win) every thing works fine But Login view is get displayed in background of all the time !!! So I want to close Login window and then open the tab group. So I tried :

win.close();
tabGroup.open();

But doesn't work application gets crashed.

So, How to close window and then display the tab Group ???

  • Hi Maulik,

    Try to do like this that may help you.

    In app.js file do like this.

    var tabGroup = Titanium.UI.createTabGroup();

// create base UI tab and root window

var win1 = Titanium.UI.createWindow({
    title : 'Tab 1',
    url : 'first.js',
    backgroundColor : '#fff',
    navBarHidden : true,
    tabBarHidden : true,
    tabGroup : tabGroup,
    backgroundImage : '/images/background.png'
});



var tab1 = Titanium.UI.createTab({
    icon : 'tab1.png',
    title : 'New',
    window : win1
});

var win2 = Titanium.UI.createWindow({
    title : 'second',
    url : 'Second.js',
    navBarHidden : true,
    tabBarHidden : true,
    backgroundImage : 'images/background.png',
    backgroundColor : '#fff',

});
var tab2 = Titanium.UI.createTab({
    icon : 'tab2.png',
    title : 'tab2',
    window : win2
});

function showTab() {

    tabGroup.setActiveTab(0);
    tabGroup.show();
}

//----------------------------- Login Window
var winLogin = Ti.UI.createWindow({
    url : 'login.js'
});

winLogin.addEventListener('blur', function(e) {
    showTab();
});
function openLoginWin() {
    winLogin.open({
        transition : Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
    });
}

//  add tabs

tabGroup.addTab(tab1);
tabGroup.addTab(tab2);

tabGroup.hide();


tabGroup.open();

winLogin.open();

    In login.js file.

    btnLogin.addEventListener('click', function(e) {
    win.close({
        transition : Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
    });
});
