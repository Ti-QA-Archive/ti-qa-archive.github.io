When rotating, it looks like custom pin (using the image property) moves on the map and doesn't stay anchored to the proper location. Any thoughts?

  • Hello, Paul, Here is a code using Custom pin with an image and "ti.map" module. It's working fine. I have tested this with CLI version 3.3.0, Titanium SDK version 3.3.0.GA.

    var Map = require('ti.map');

var win = Ti.UI.createWindow();

var mountainView = Map.createAnnotation({
    latitude : 37.390749,
    longitude : -122.081651,
    title : "Appcelerator Headquarters",
    subtitle : 'Mountain View, CA', 
    image  : 'KS_nav_views.png',
});

var mapview = Map.createView({
    mapType : Map.STANDARD_TYPE,
    region : {
        latitude : 37.390749,
        longitude : -122.081651,
        latitudeDelta : 0.04,
        longitudeDelta : 0.04
    },
    animate : true,
    regionFit : true,
    userLocation : true,
    annotations : [mountainView]
});

win.add(mapview);

win.open();

    Hope it helps. If not, please provide the code so i can reproduce the issue and work on it. thank you.
