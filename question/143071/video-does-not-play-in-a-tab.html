    It works now with this code:

    var tabGroup = Ti.UI.createTabGroup({
       id: 'tabGroup1'
});

//Tab 1//
var win1 = Ti.UI.createWindow({backgroundColor:"red"});
var tab1 = Ti.UI.createTab({title:L('home'), window:win1});
var lab1 = Ti.UI.createLabel({top:20, textAlign:'left', text: L('lab1')});


//Tab 2//
var win2 = Ti.UI.createWindow({backgroundColor:"#ccc"});
var tab2 = Ti.UI.createTab({title:L('video'), window:win2});

if(Ti.Platform.name!='android') {
        var contentURL = 'http://assets.appcelerator.com.s3.amazonaws.com/video/media.m4v';
        win2.add(activeMovie);
    } else {
        var contentURL = 'http://assets.appcelerator.com.s3.amazonaws.com/video/media.3gp'; 
    }

    var activeMovie = Titanium.Media.createVideoPlayer({
        url: contentURL,
        backgroundColor: '#000',
        movieControlMode: Titanium.Media.VIDEO_CONTROL_DEFAULT,
        scalingMode: Titanium.Media.VIDEO_ASPECT_FILL,
