TableView not displaying unless height is set.

Hi all! I'm pretty certain I'm doing something wrong here, but I cant figure out what.

I have a table view to set up a login form, but the tableView will not show on the screen unless I set a height for it. That doesn't really seem very dynamic? I want to reuse the same code for multiple tableViews so I really need it to automatically set the height relating to how many Rows are in the Table?

I have tried height:&#39;auto&#39; and layout:&#39;vertical&#39; as well as setting heights on each row. All of which fail. The only way I've found is to set a height on the TableView.

Here is my code:


var formName = Ti.UI.createTableViewRow({ backgroundColor:'#fff' });
var formNameInp = Ti.UI.createTextField({ hintText:'Username', autocorrect:false, left:10 });
formName.add(formNameInp);

var formPassword = Ti.UI.createTableViewRow({ backgroundColor:'#fff' });
var formPasswordInp = Ti.UI.createTextField({ hintText:'Password', passwordMask:true, autocorrect:false, left:10 });
formPassword.add(formPasswordInp);

var formSection = Ti.UI.createTableViewSection();
formSection.add(formName);
formSection.add(formPassword);

data = [];
data.push(formSection);

var loginTable = myApp.menuUI.createFormTable(data);

win.add(loginTable);

In my UI.js - THIS ONE DOESNT WORK


myApp.menuUI.createFormTable = function(formData){
    var ft = Ti.UI.createTableView({
        width:300, top:0,
        data:formData,
        backgroundImage:'none', backgroundColor:'transparent',
        style: Ti.UI.iPhone.TableViewStyle.GROUPED,
        scrollable:false
    });
    return ft;
};

In my UI.js - THIS ONE WORKS (But is not dynamic to how many rows there are)


myApp.menuUI.createFormTable = function(formData){
    var ft = Ti.UI.createTableView({
        width:300, height:80, top:0,
        data:formData,
        backgroundImage:'none', backgroundColor:'transparent',
        style: Ti.UI.iPhone.TableViewStyle.GROUPED,
        scrollable:false
    });
    return ft;
};
