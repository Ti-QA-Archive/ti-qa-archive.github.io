createTableView not showing in window even after add()

I'm using the controller code from forging Titanium. The table just doesnt show regardless if I do window.add(tableView);
I can add labels and buttons but I'm missing something about tables. I appreciate your help.

My App.js

//require the UI components necessary to drive the test
var NavigationController = require('NavigationController').NavigationController;

    home     = require('windows/home').HomeScreen;


//create NavigationController which will drive our simple application
var controller = new NavigationController();

//open initial window
controller.open(new home(controller));

then my home.js


exports.HomeScreen = function(navController) {
    var win = Ti.UI.createWindow({
        title:'Home',
        backgroundImage:'images/glowbackground.png',
        layout:'vertical'

    });

    var data = [
    { title: 'My Title' }

    ];

    Ti.API.info(data.length);
    var menuTable = Ti.UI.createTableView({
        backgroundColor: '#fff',
        data:data
    });

    win.add(menuTable);


    return win;
};

    For vertical layout ( HomeScreen ), you should set table's height to a specific value. W/O a specific value, I wont work for me.

    Try this:

    home.js

    var menuTable = Ti.UI.createTableView({
    backgroundColor : '#fff',
    data : data,
    height: Ti.Platform.displayCaps.platformHeight/2 // 50% of window's height
});
