Image in AlertBox

How to show alert along with icon/image? I am using alloy framework and develop app for android device. I have create dialog as follow, but help me to show image/icon.


var dialog = Ti.UI.createAlertDialog({
                message : 'This is alert dialog.',
                ok : 'OK',
                title : 'AlertDialog'
            }).show();

    Hi Preety :

    Try this :

    var dialog = Ti.UI.createAlertDialog({
    message : 'This is alert dialog.',
    ok : 'OK',
    title : 'AlertDialog',
    apiName : Ti.UI.createImageView({
        image : "/img.jpg"
    })
}).show();

    for more information see this apiName.

    • you can create above alertbox using customview something like this :

      
var view = Ti.UI.createView({
    top : 0,
    height : 150,
    width: 200,

});

var image = Ti.UI.createImageView({
    image : "/img.jpg",
    height : "50dp",
    width : "50dp",
    top : "10dp"
});

var lbl = Ti.UI.createLabel({
    text : "Alert !",
    left : '35dp',
    top : '10dp'
});


var lbl1 = Ti.UI.createLabel({
    text : "This is an alert dialog",
    left : '35dp',
    bottom : '30dp'
});
view.add(image);
view.add(lbl);
view.add(lbl1);
var dialog = Ti.UI.createAlertDialog({
    androidView : view,
    ok : 'OK',
   }).show();
