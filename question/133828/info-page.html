    hi anand,

    try this one

    var txt = 'EMC\nUser Interface Enhancement to Customer Interactions\nCopyright 2012\nEntries pending submission'
var infowin = Ti.UI.createWindow({
    //specify window's configuration whatever u want

});
var lbl = Ti.UI.createLabel({
    text : txt,
    font : {
        fontSize : 13
    },
    textAlign : 'center'
});
infowin.add(lbl);
infowin.open({
    transition : Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
});
