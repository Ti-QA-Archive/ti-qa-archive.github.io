  • Hello Rama,

    Here is a sample for doing what you want, I can add rows without any issues.

    Code Sample

    var tableData = [];

var index = 0;

var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

var table = Ti.UI.createTableView({
    top : 0,
    left : 0,
    bottom : 43,
    objName : 'table',
});

var button = Ti.UI.createButton({
    title : 'add',
    bottom : 44,
    left : 0,
});

button.addEventListener('click', function() {
    Titanium.Media.openPhotoGallery({

        success : function(event) {
            var cropRect = event.cropRect;
            var image = event.media;
            var filename = Titanium.Filesystem.applicationDataDirectory + "/" + 'camera_photo' + new Date().getTime() + ".png";
            var f = Titanium.Filesystem.getFile(filename);
            if (f.exists()) {
                Ti.API.info('The file exist , trying to delete it before using it :' + f.deleteFile());
                f = Titanium.Filesystem.getFile(filename);
            }
            f.write(image);

            var row = Ti.UI.createTableViewRow({
                className : 'row',
                objName : 'row',
                touchEnabled : true,
                height : 44,
            });

            var icon = Ti.UI.createImageView({
                backgroundColor : '#008FD5',
                objName : 'enabledWrapperView',
                image : f.nativePath,
                left : 0,
            });

            var label = Ti.UI.createLabel({
                text : 'Row: ' + index,
                left : 45,
            });

            index++;

            row.add(icon);
            row.add(label);

            table.appendRow(row);
        },
        cancel : function() {

        },
        error : function(error) {
            // create alert
            var a = Titanium.UI.createAlertDialog({
                title : 'Camera'
            });

            // set message
            if (error.code == Titanium.Media.NO_CAMERA) {
                a.setMessage('Device does not have video recording capabilities');
            } else {
                a.setMessage('Unexpected error: ' + error.code);
            }

            // show alert
            a.show();
        },
        allowEditing : true
    });

});

win.add(table);
win.add(button);
win.open();

    Testing

    You will need to have some images in the gallery, open the simulator, open safari, open images.google.com and then save some images into gallery.

    1. Create a new mobile project (classic titanium)
    2. Paste the code into app.js
    3. Run it

    Let us know how it goes.

    Mauro
