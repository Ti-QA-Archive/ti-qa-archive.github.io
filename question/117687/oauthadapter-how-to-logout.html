  • or you could edit the oauth_adapter to add a function like this.

    this.deleteAccessToken = function(pService){
        Ti.API.debug('Deleting access token [' + pService + '].');
        var file = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, pService + '.config');
        if(file.exists() == true){
             file.deleteFile();
         }
    };

    I put it right after this.saveAccessToken function.
