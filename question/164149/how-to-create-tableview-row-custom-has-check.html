  • Hi,

    here is the small demo of custom has check.

    var win = Ti.UI.createWindow({
    backgroundColor : 'white',
    layout : 'vertical'
});

var data = [{
    title : "1"
}, {
    title : "2"
}, {
    title : "3"
}, {
    title : "4"
}, {
    title : "5"
}];
var rowData = [];
var tempRow = null;

var tbl = Ti.UI.createTableView({

});

tbl.addEventListener('click', function(e) {
    if (tempRow) {
        tempRow.img.backgroundColor = "black";
    }
    e.row.img.backgroundColor = "red";
    tempRow = e.row;
});

for (var i in data) {
    var row = Ti.UI.createTableViewRow({
        title : data[i].title,
        height : 50
    });
    row.img = Ti.UI.createImageView({
        right : 10,
        width : 40,
        height : 40,
        // image : Ti.Filesystem.resourcesDirectory + '1.png',
        // bubbleParent : false,
        backgroundColor : "#000"
    });
    row.add(row.img);
    rowData.push(row);
}

tbl.data = rowData;
win.add(tbl);

win.open();

  • Hi

    You can use the right image property to set a custom image for your checkbox.

    Then use setRightImage method to replace the image for a checked image on your click event.

    By using this property you will be able to provide density specific image assets to suit the various devices, reading through the following answer will provide you with any further information relating to image density folders for Android: images-in-android

    Hope this helps!
