Why is it so had to just create a simple menu to link some file.js ??

var tabGroup = Titanium.UI.createTabGroup();

    backgroundColor: "#FFF"

    icon:'KS_nav_ui.png',
    title:'Testing',
    window:win
});
var regItems = [ 
    {title: "Menu 1", font:{fontSize: 25, fontWeight: 'bold'}, color: "#45165C", leftImage:"images/regImg.gif", className: "tableRow", hasDetail: true},

    Hello Mario,

    You open only newWIn window on tableview's click event . you need to do like below.

    Try this : 

    var tabGroup = Titanium.UI.createTabGroup();
var win = Titanium.UI.createWindow({
    backgroundColor : "#FFF"
});
var tab = Titanium.UI.createTab({
    icon : 'KS_nav_ui.png',
    title : 'Testing',
    window : win
});
var regItems = [{
    title : "Menu 1",
    font : {
        fontSize : 25,
        fontWeight : 'bold'
    },
    color : "#45165C",
    leftImage : "images/regImg.gif",
    className : "tableRow",
    hasDetail : true,
    openWin : "newWin.js"

}, {
    title : "Menu 2",
    font : {
        fontSize : 25,
        fontWeight : 'bold'
    },
    color : "#45165C",
    leftImage : "images/regImg.gif",
    className : "tableRow",
    hasDetail : true,
    openWin : "newWindow.js"
}];

var regItemList = Titanium.UI.createTableView({
    data : regItems
});

regItemList.addEventListener('click', function(e) {
    Ti.API.info('>>>>' + e.source.openWin);
    var win2 = Titanium.UI.createWindow({
        url : e.source.openWin,
        title : e.rowData.title,
        backgroundColor : "#273691"
    });
    win2.open();

});
win.add(regItemList);

tabGroup.addTab(tab);
tabGroup.open();

    i hope this helps. thanks.
