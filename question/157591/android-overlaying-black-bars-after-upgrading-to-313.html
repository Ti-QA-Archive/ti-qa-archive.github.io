    • I was able to create a blank alloy project that reproduced the problem. The index.xml and index.js files are below. I used some code form the original project to build the small sample app.

      index.xml

      <Alloy>
    <Window id="loadingWindow">

        <Label id="versionNumber" />
    </Window>
</Alloy>

      index.js

      var scale = OS_ANDROID ? Ti.Platform.displayCaps.logicalDensityFactor : 1;

$.loadingWindow.backgroundColor = "#7CB2E5";
$.loadingWindow.orientationModes = [Ti.UI.PORTRAIT];
$.loadingWindow.navBarHidden = true;
$.loadingWindow.width = Ti.Platform.displayCaps.platformWidth / scale;
$.loadingWindow.left = 0;


// Version Number
$.versionNumber.color = '#FFF';
$.versionNumber.text = "Version: 1.0.0";
$.versionNumber.top = (Ti.Platform.displayCaps.platformHeight / scale) - 35;


$.loadingWindow.open();
