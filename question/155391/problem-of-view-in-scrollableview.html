    • based on the number of images you have in the array images are created n view to scrollableview
      each view will have a maximum of 9 images, like you wanted, 3x3

      this is the code

      var win = Titanium.UI.createWindow();
var url_image = "KS_nav_ui.png";
var array_images = [url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image, url_image];
var num_img = array_images.length;
var array_view = [];
var top = -100;
var left = -100;
var c1 = -1;

var scroll = Ti.UI.createScrollableView({
    width : 300,
    height : 300,
    borderColor : 'blue'
});

for (var i = 0; i < num_img; i++) {

    if (i % 9 == 0) {

        var view = Ti.UI.createView({
            width : 300,
            height : 300
        });
        array_view.push(view);

        c1++;
        left = -100;
        top = -100;

    }

    left += 100;

    if (i % 3 == 0) {
        top += 100;
        left = 0;
    }

    var view_img = Ti.UI.createView({
        width : 100,
        height : 100,
        backgroundImage : array_images[i],
        borderColor : 'red',
        top : top,
        left : left
    });

    array_view[c1].add(view_img);

}

scroll.views = array_view;

win.add(scroll);

win.open();
