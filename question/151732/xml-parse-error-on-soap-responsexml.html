  • I'm sorry. I couldn't give you any answer because I wasn't in work so I had no access to the code.

    Here is an example of how I do to stablish my connection: the error happens when I try to access the document via this.responseXML, this is, always after the message "webservice connection ok', inside of the "on load" event.

    Example:

    // Definitions for the HttpClient
    var xhr = Ti.Network.createHTTPClient();

    var authstr = 'Basic ' + Titanium.Utils.base64encode(user+':' + pass);

    xhr.open(operation,url);
    xhr.setRequestHeader('Authorization',authstr);
    xhr.setRequestHeader('Content-Type',typeContent);
    xhr.setRequestHeader('Content-Length', SoapRequest.length);
    xhr.setRequestHeader('Soapaction',soapAction);

    xhr.setUsername(user);
    xhr.setPassword(pass);

    xhr.send(SoapRequest);

    xhr.setTimeout(15000);

    xhr.onload = function(e) {

    Ti.API.info ('Countries::  WEBSERVICE CONNECTION OK');                

try {
    Ti.API.info (' XML '+this.responseXML);
}
catch (e) {
    alert (' ERROR ON PARSING ->>>>> '+e.message)
}

    }

    xhr.onerror= function(e) {
    alert ('Countries:: WEBSERVICE CONNECTION ERROR – '+e.error+' '+xhr.status);
    }
