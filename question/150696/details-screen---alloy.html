    • Ok. Got it now.

      Problem was my detail.xml file had database refs in it, and that causes that $model issue.

      detail.xml:

      <Alloy>
    <Window id="detailWindow" layout="Vertical">
        <!-- header -->
        <View id="header">
            <Label id="title">Toilet Details</Label>
            <View class="divider"/>
            <View id="addView" onClick="closeScreen">
                <ImageView id="addImage"/>
            </View>
        </View>
        <View layout="Vertical">
            <Label id="toiletName" text="" />
            <Label id="toiletAddress" text="" />
        </View>
    </Window>
</Alloy>

      detail.js:

      var args = arguments[0] || {};
$.toiletName.text = args.name || '';
$.toiletAddress.text = args.address || '';

function closeScreen() {
    $.detailWindow.close();
}

      index.js:

      var toilets = Alloy.Collections.toilets;

function printObject(obj) {
    Ti.API.info("Printing Object...");
    // central_object.debugApp("Printing Object...", null);
    for (property in obj) {
        Ti.API.info(property + ': ' + obj[property]);
        // central_object.debugApp(property, obj[property]);
    }
    // central_object.debugApp("Object...Done", null);
    Ti.API.info("Object...Done");
};

// open detail window
function openDetail(e) {
    //get the detail controller and window references
    var controller = Alloy.createController('detail', {
        'name' : e.rowData.rowName,
        'address' : e.rowData.rowAddress
    }).getView();

    controller.open();
}
