  • make your window heavy,… either using modal or fullscreen attribute.
    this code worked for me.
    example.

    var win1 = Titanium.UI.createWindow({  
    title:'myWindow',
    exitOnClose:true,
    backgroundColor:'#fff',
    modal:true

});




var b6=Titanium.UI.createButton({
    title:'Exit',
    top:440,
    width:100
});

b6.addEventListener('click',function(){
    win1.close();
});


win1.add(b6);
