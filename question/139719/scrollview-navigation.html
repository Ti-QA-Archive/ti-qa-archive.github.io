    similar code, use scrollView not scrollable view

    var win = Ti.UI.createWindow();

var scrollView = Titanium.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    top : '100dp',
    height : '64dp',
    zIndex : 100,
});

var vArray = [];

for ( var x = 0; x < 10; x++) {
    vArray.push(Ti.UI.createView({
        borderRadius : 10,
        backgroundColor : '#f888',
        height : '50dp',
        width : '70dp',
        isActive : false
    }));

    if (x !== 0) {
        vArray[x].left = 10 + (80 * x) + 'dp';
    } else {
        vArray[x].left = '10dp';
    }

    vArray[x].borderColor = 'transparent';

    scrollView.add(vArray[x]);
}

win.add(scrollView);
