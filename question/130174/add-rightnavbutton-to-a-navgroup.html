    var first = Ti.UI.createWindow({
  backgroundColor:"#f00",
  title:"DEMO",
  barColor:"#000",
  navBarHidden:false
});
var btn=Ti.UI.createButton({title:"test",height:'20',width:'30'});

first.setRightNavButton(btn);
first.add(tableView);

var navGroup = Ti.UI.iPhone.createNavigationGroup({
  window:first
});
