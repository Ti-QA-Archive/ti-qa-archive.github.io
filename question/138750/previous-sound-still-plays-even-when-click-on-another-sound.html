previous sound still plays even when click on another sound

Hi all,

I have several buttons, which plays a song when clicked. But i am facing an issue that the previous song
still being played even if i click another button to play another song.

Here is my code.

var button2_sound=Titanium.Media.createSound({
        url:'sound/annam.mp3',
        preload:true
    });

    button2.addEventListener('click',function(){
        button2_sound.play();
    });

    var button3_sound=Titanium.Media.createSound({
        url:'sound/Ninem.mp3',
        preload:true
    });

    button3.addEventListener('click',function(){
        button3_sound.play();
    });

When i first click button2 then song plays but if i click button 3 while the button2 song is being played , button3 song and button2 both plays simultaneously.
Please tell me the possible solution. I have 20 buttons like this on the same window which plays different songs.

  • Accepted Answer

    var playing_sound;

var button2_sound = Titanium.Media.createSound({
    url : 'sound/annam.mp3',
    preload : true
});

button2.addEventListener('click', function() {

    if (playing_sound) {
        playing_sound.stop();
    }

    playing_sound = button2_sound;
    button2_sound.play();
});

var button3_sound = Titanium.Media.createSound({
    url : 'sound/Ninem.mp3',
    preload : true
});

button3.addEventListener('click', function() {
    if (playing_sound) {
        playing_sound.stop();
    }
    playing_sound = button3_sound;
    button3_sound.play();
});
