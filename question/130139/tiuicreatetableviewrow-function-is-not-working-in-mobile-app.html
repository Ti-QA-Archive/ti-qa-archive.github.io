  • Try this:

    var mywindow = Ti.UI.createWindow({

    top:0,
    left:0,
    width:320,
    height:480

});

var TheTable = Ti.UI.createTableView();
mywindow.add(TheTable);

alert("Came 1");
var data = [];
var row = Ti.UI.createTableViewRow({});

alert("Came 2");
//instead of adding the objects to 'row'
//created a view to add them to
var rowView = Ti.UI.createView();
var watsOnTitle = Ti.UI.createLabel({
            text:"Test",
            font:{fontSize:16},
            width:'auto',
            textAlign:'left',
            left:70,
            color:'#535353'
        });
var arrow =  Ti.UI.createImageView({
            url:'images/arrow_white.png',
            width:10,
            height:10,
            right:10,
            top:20
        });

var thumbnail =  Ti.UI.createImageView({
            image:"http://....my image path...",
            width:50,
            height:45,
            left:5
        });

rowView.add(thumbnail);//added to a 'view'
rowView.add(watsOnTitle);//added to a 'view'
rowView.add(arrow);//added to a 'view'
row.add(rowView);//view added to row
data.push(row);

TheTable.setData(data);
