how to display long string of text (from database) in the tableview

I have the following code to display a string of text pulled out from the database. However the 'address' string is not completely displayed in the tableview. How do i wrap the long string of text to display properly in table view?

// create var for the currentWindow
var currentWin = Ti.UI.currentWindow;

var db = Ti.Database.install('tuitionlist.sqlite','north');

var addName = Ti.UI.currentWindow.addName;

var rows = db.execute('SELECT * FROM north WHERE name="' + addName + '"');
var data = [
{title:'' + rows.fieldByName('address') + '', header:'Address'},
{title:'' + rows.fieldByName('website') + '', header:'Website'},
{title:'' + rows.fieldByName('contact') + '', header:'Contact'}
];

var tableview = Ti.UI.createTableView({
data:data
});

currentWin.add(tableview);
