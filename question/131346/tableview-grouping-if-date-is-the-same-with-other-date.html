//Add that row to your section
section.add(row);

if (currentMonth != previousMonth){
    // We need to write out our previous section to the data array
    Blogdata[x++]= section;
    //Null out our section
    section = null;
    // And create a new section.
    var section = Titanium.UI.createTableViewSection();
    // Create our next row
    var row = Ti.UI.createTableViewRow();
    //Add row to the section
    section.add(row);
} else {
    // Create our next row
    var row = Ti.UI.createTableViewRow();
    //Add row to the section
    section.add(row);
}

{end your loop here}
//write out your last section to the dataset
Blogdata[x++]= section;
