  • Hi

    You should create a simple latch function to check against the new version and if that test shows a different version then show the new version message/code.

    I have simplified the process below by splitting them up into a two simple functions as below. This will provide a true match right from the start when this is the very first install.

    function showNewVersionMessage() {
    Ti.UI.createAlertDialog({
        buttonNames : ['OK'],
        message : 'The app version is '+ Ti.App.getVersion() + '.',
        title : 'Info'
    }).show();
}
function isNewVersion() {
    var lastVersion = Ti.App.Properties.getString('lastVersion', '');
    var currentVersion = Ti.App.getVersion();
    if (lastVersion !== currentVersion) {
        showNewVersionMessage();
        Ti.App.Properties.setString('lastVersion', currentVersion);
    }
}
// usage
isNewVersion();

    You can tweak this to show a different message on first install - this can be useful sometimes.

    function showNewVersionMessage() {
    Ti.UI.createAlertDialog({
        buttonNames : ['OK'],
        message : 'The app version is '+ Ti.App.getVersion() + '.',
        title : 'Info'
    }).show();
}
function showFirstVersionMessage() {
    Ti.UI.createAlertDialog({
        buttonNames : ['OK'],
        message : 'Hope you like this app.',
        title : 'Info'
    }).show();
}
function isNewVersion() {
    var lastVersion = Ti.App.Properties.getString('lastVersion', '');
    var currentVersion = Ti.App.getVersion();
    if (lastVersion !== currentVersion) {
        if (lastVersion === '') {
            showFirstVersionMessage();
        } else {
            showNewVersionMessage();
        }
        Ti.App.Properties.setString('lastVersion', currentVersion);
    }
}
// usage
isNewVersion();
