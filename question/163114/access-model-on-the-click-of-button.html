  • Well i got the solution for it. It was a small correction in following function :

    (
function nextButtonClicke(e){
    e.source.parent.hide();
    e.source.parent.parent.children[1].show();
    e.source.parent.parent.parent.children[0].children[0].visible=true;
    Ti.App.Properties.setInt("pageIndex",1);

    var questionList = Alloy.createCollection('questionList');
    questionList.fetch({query:'SELECT * FROM questionList',
    });
    alert('questionList = '+JSON.stringify(questionList));
}

    In the last alert i m getting the desired result.
