    var self = Ti.UI.createWindow({ backgroundColor : "#fff", navBarHidden : true, tabBarHidden : true });
var loginView = Ti.UI.createView();

var title = Ti.UI.createLabel({
    text : 'MyTitle',
    font : {
        fontSize : 16,
        fontWeight : 'bold'
    }
});

var flexSpace = Titanium.UI.createButton({
    systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

var toolbar = Titanium.UI.iOS.createToolbar({
    items : [flexSpace, title, flexSpace],
    top : 20,
    height : 44
});

self.add(toolbar);

var welcomeView = Ti.UI.createView({
    height : 80,
    top : 70
});

var welcomeContent = "Welcome";

var welcomeLabel = Ti.UI.createLabel({
    left : 5,
    right : 5,
    horizontalWrap : true,
    text : welcomeContent,
    textAlign : Ti.UI.TEXT_ALIGNMENT_CENTER,
    font : {
        fontSize : 14
    },
});

welcomeView.add(welcomeLabel);

loginView.add(welcomeView);

var cancelUsernameButton = Ti.UI.createButton({
    systemButton : Ti.UI.iPhone.SystemButton.CANCEL
});

cancelUsernameButton.addEventListener('click', function(e) {
    title.text = 'MyTitle';
    welcomeView.show();
    loginDetailsTableView.top = 170;
    createButtonView.show();
    forgotButtonView.top = 330;
    usernameTextField.blur();
});

var cancelPasswordButton = Ti.UI.createButton({
    systemButton : Ti.UI.iPhone.SystemButton.CANCEL
});

cancelPasswordButton.addEventListener('click', function(e) {
    title.text = 'MyTitle';
    welcomeView.show();
    loginDetailsTableView.top = 170;
    //createButtonView.show();
    //forgotButtonView.top = 330;
    passwordTextField.blur();
});

var nextButton = Ti.UI.createButton({
    title : 'Next'
});

nextButton.addEventListener('click', function(e) {
    passwordTextField.focus();
});

var loginButton = Ti.UI.createButton({
    title : 'Login',
    enabled : false
});

loginButton.addEventListener('click', function() {
    //logic for login user
});

var loginDetailsTableView = Ti.UI.createTableView({
    style : Ti.UI.iPhone.TableViewStyle.PLAIN,
    borderColor : '#D8D8D8',
    borderRadius : 10,
    top : 170,
    right : 20,
    left : 20,
    height : 100,
    separatorInsets : {
        left : 10,
        right : 10
    },
    scrollable : false
});

var tableSection = Ti.UI.createTableViewSection();

var usernameLabel = Ti.UI.createLabel({
    top : 15,
    text : 'Username',
    left : 10
});

var usernameTextField = Ti.UI.createTextField({
    top : 16,
    hintText : 'required',
    autocapitalization : Ti.UI.TEXT_AUTOCAPITALIZATION_NONE,
    autocorrect : false,
    left : 120,
    width : 150,
    borderStyle : Titanium.UI.INPUT_BORDERSTYLE_NONE,
    color : 'black',
    backgroundColor : '#fff',
    value : ''
});

var usernameToolbar = Titanium.UI.iOS.createToolbar({
    items : [cancelUsernameButton, flexSpace, nextButton],
    borderBottom : false
});
usernameTextField.keyboardToolbar = usernameToolbar;

var usernameRowView = Ti.UI.createView({
    height : 50
});

usernameRowView.add(usernameLabel);
usernameRowView.add(usernameTextField);

var usernameRow = Ti.UI.createTableViewRow({
    selectionStyle : Ti.UI.iPhone.ListViewCellSelectionStyle.NONE,
    height : 50,
    hasChild : false
});

usernameRow.addEventListener('click', function() {
    usernameTextField.focus();
});

usernameRow.add(usernameRowView);
tableSection.add(usernameRow);

var passowordLabel = Ti.UI.createLabel({
    top : 15,
    text : 'Password',
    left : 10
});

var passwordTextField = Ti.UI.createTextField({
    passwordMask : true,
    top : 16,
    hintText : 'required',
    autocapitalization : Ti.UI.TEXT_AUTOCAPITALIZATION_NONE,
    autocorrect : false,
    left : 120,
    width : 150,
    borderStyle : Titanium.UI.INPUT_BORDERSTYLE_NONE,
    color : 'black',
    backgroundColor : '#fff',
    value : ''
});

var passwordToolbar = Titanium.UI.iOS.createToolbar({
    items : [cancelPasswordButton, flexSpace, loginButton],
    borderBottom : false
});
passwordTextField.keyboardToolbar = passwordToolbar;

usernameTextField.addEventListener('change', function(e) {
    loginButton.enabled = (e.value != passwordTextField.value != '');
});

passwordTextField.addEventListener('change', function(e) {
    loginButton.enabled = (e.value != usernameTextField.value != '');
});

usernameTextField.addEventListener('focus', function(e) {
    title.text = 'Username';
    welcomeView.hide();
    loginDetailsTableView.top = 100;
    //createButtonView.hide();
    //forgotButtonView.top = 230;
});

passwordTextField.addEventListener('focus', function(e) {
    title.text = 'Password';
    welcomeView.hide();
    //loginDetailsTableView.top = 100;
    //createButtonView.hide();
   // forgotButtonView.top = 230;
});

usernameTextField.addEventListener('return', function(e) {
    loginButton.fireEvent('click');
});

passwordTextField.addEventListener('return', function(e) {
    loginButton.fireEvent('click');
});

var passwordRowView = Ti.UI.createView({
    height : 50
});

passwordRowView.add(passowordLabel);
passwordRowView.add(passwordTextField);

var passwordRow = Ti.UI.createTableViewRow({
    selectionStyle : Ti.UI.iPhone.ListViewCellSelectionStyle.NONE,
    height : 50,
    hasChild : false
});

passwordRow.addEventListener('click', function() {
    passwordTextField.focus();
});

passwordRow.add(passwordRowView);
tableSection.add(passwordRow);
loginDetailsTableView.setData([tableSection]);

loginView.add(loginDetailsTableView);

//loginView.add(createButtonView);
//loginView.add(forgotButtonView);

self.add(loginView);
self.open();
