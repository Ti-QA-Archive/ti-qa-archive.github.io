TableView search property not working?

Hi all,
I tried to insert a UI.SearchBar in Tableview but when I try to search for something, the result is always "no resuts"!

Here is the code:

        var window = Ti.UI.createWindow({
        title: 'List job',
        backgroundColor: 'white'
    });

    var search = Titanium.UI.createSearchBar({
        barColor: '#000', 
        showCancel: true,
        height: 43,
        top: 0,
    });

    var tableView = Ti.UI.createTableView({
        backgroundColor:'white',
        top: 43,
        bottom: 0,
        search: search,
        width: '100%'
    });

    window.addEventListener('focus', function() {
        soap.getJob(
        function(dati, numDati){
            var dataRow = [];

            for(var i=0;i<numDati; i++) {
                var row = Ti.UI.createTableViewRow({
                    className: dati[i][0], 
                    selectedBackgroundColor:'white',
                    rowIndex: dati[i][2],
                    height: 30
                });
                var lavoro = Ti.UI.createLabel({
                    text: dati[i][0],
                    left: 7,
                    width: 100,
                    height: '100%',
                    font: {fontSize: 12},
                    wordWrap: false,
                    ellipsize: true,
                    color: 'black'
                });
                var utente = Ti.UI.createLabel({
                    text: dati[i][1],
                    left: 107,
                    width: 80,
                    height: '100%',
                    font: {fontSize: 12},
                    wordWrap: false,
                    ellipsize: true,
                    color: 'black'
                });
                var numero = Ti.UI.createLabel({
                    text: dati[i][2],
                    left: 187,
                    width: 70,
                    height: '100%',
                    font: {fontSize: 12},
                    wordWrap: false,
                    ellipsize: true,
                    color: 'black'
                });
                var stato = Ti.UI.createLabel({
                    text: dati[i][4],
                    left: 257,
                    right: 0,
                    height: '100%',
                    font: {fontSize: 12},
                    wordWrap: false,
                    ellipsize: true,
                    color: 'black'
                });
                row.add(lavoro);
                row.add(utente);
                row.add(numero);
                row.add(stato);
                dataRow.push(row);
            }
            tableView.data = dataRow;
        },
        function(error) {
            Ti.API.error("Errore in get job!\n"+error);
        });
    });

    window.add(tableView);

    if(Ti.Platform.Android) {
        window.open();
    }
    else{
        navigWindow.openWindow(window, {animated:true});
    }

Missing something?

Titanium SDK: 3.2.3.GA

    Hi,

    Try this code it works

    var window = Ti.UI.createWindow({
        title: 'List job',
        backgroundColor: 'white'
    });

    var search = Titanium.UI.createSearchBar({
        barColor: '#000', 
        showCancel: true,
        height: 43,
        top: 0,
    });

    var tableView = Ti.UI.createTableView({
        backgroundColor:'white',
        top: 43,
        bottom: 0,
        search: search,
        filterAttribute : 'name' // FIlter By its Name
        width: '100%'
    });

    window.addEventListener('focus', function() {
        soap.getJob(
        function(dati, numDati){
            var dataRow = [];

            for(var i=0;i<numDati; i++) {
                var row = Ti.UI.createTableViewRow({
                    className: dati[i][0], 
                    selectedBackgroundColor:'white',
                    rowIndex: dati[i][2],
                    height: 30,
                    name : dati[i][1] // search by this value dati[i][1]
                });
                var lavoro = Ti.UI.createLabel({
                    text: dati[i][0],
                    left: 7,
                    width: 100,
                    height: '100%',
                    font: {fontSize: 12},
                    wordWrap: false,
                    ellipsize: true,
                    color: 'black'
                });
                var utente = Ti.UI.createLabel({
                    text: dati[i][1],
                    left: 107,
                    width: 80,
                    height: '100%',
                    font: {fontSize: 12},
                    wordWrap: false,
                    ellipsize: true,
                    color: 'black'
                });
                var numero = Ti.UI.createLabel({
                    text: dati[i][2],
                    left: 187,
                    width: 70,
                    height: '100%',
                    font: {fontSize: 12},
                    wordWrap: false,
                    ellipsize: true,
                    color: 'black'
                });
                var stato = Ti.UI.createLabel({
                    text: dati[i][4],
                    left: 257,
                    right: 0,
                    height: '100%',
                    font: {fontSize: 12},
                    wordWrap: false,
                    ellipsize: true,
                    color: 'black'
                });
                row.add(lavoro);
                row.add(utente);
                row.add(numero);
                row.add(stato);
                dataRow.push(row);
            }
            tableView.data = dataRow;
        },
        function(error) {
            Ti.API.error("Errore in get job!\n"+error);
        });
    });

    window.add(tableView);

    if(Ti.Platform.Android) {
        window.open();
    }
    else{
        navigWindow.openWindow(window, {animated:true});
    }
