  • Hello roy,

    Try below sample : 

    Titanium.Facebook.dialog("apprequests", {
            app_id : 'YOUR_APP_ID',
            message : "YOUR MESSAGE",
            redirect_uri : '****'
        }, function(rr) {
            if (rr.success) {
                var query = "facebook query";
                Titanium.Facebook.request('fql.query', {
                    query : query
                }, function(r) {
                    if (!r.success) {
                        if (r.error) {
                            //error
                        } else {
                            //error
                        }
                        return;
                    }
                    var result = JSON.parse(r.result);
                });

            }
        });

    Let me know if it works

    Thanks
    Jigar
