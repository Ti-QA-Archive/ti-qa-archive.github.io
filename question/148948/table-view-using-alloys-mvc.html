    <Model src="assetMaster">
    </Model>
    <Collection src="assetMaster" instance="true" id="coll_assetmaster">
    </Collection>
    <Window class="damagedAssetList" id="damList" title="Damaged Asset List" fullscreen="true">
    <TableView id="damagedScroll" layout="vertical" dataCollection="assetMaster" onClick="damagedAssetList">
    <TableViewRow class="listRow" layout="vertical">
    <Label id="assetID" text="{assetId}"/>
