      Here's the final version for anyone looking answer to this:

      var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff',
    barColor:"#6fab70"
});
win1.orientationModes = [Ti.UI.PORTRAIT, Titanium.UI.LANDSCAPE_LEFT,Titanium.UI.LANDSCAPE_RIGHT];

var subwindow = Titanium.UI.createWindow({
    top:0,
    height:250,
    width:315,
    borderWidth:1,
    borderColor:"#000",
    borderRadius:6,
    //url:"gethttp.js"

});
//xml starts here

var xhr = Ti.Network.createHTTPClient();
xhr.open('GET', "http://dl.dropbox.com/u/19609983/MapXML%20copy.xml");

//when it opens, perform a function

xhr.onload = function() 
{

    try
    {

    var doc = this.responseXML.documentElement;
    var items = doc.getElementsByTagName("item");
    var x = 0;
                var row = [];
            for (var c = 0; c < items.length; c++) 
            {
        var item = items.item(c);
        row.title = item.getElementsByTagName("title").item(0).text;
        row.desc = item.getElementsByTagName("description").item(0).text;



        var label = Ti.UI.createLabel
        ({
        text : row.title
        });

        subwindow.add(label);

               }            
    }


        catch(E) 
        {
        alert(E);
        }     


}

xhr.send();

win1.add(subwindow);
