    Hi

    I would create my exports like so:

    var first = function (args) {
    // code
}
var second = function (args) {
    // code
}

exports.first = first;
exports.second = second;

    By doing it this way, you can use your functions how you would like.

    var first = function (args) {
    alert(JSON.stringify(args));
}
var second = function (args) {
    first(args);
}

second({
    firstname: "John",
    lastname: "Doe"
});

exports.first = first;
exports.second = second;

    Hope this helps.

  • Hi antonio,

    you can simply call the exported functions like this, the same rules like in every other js file are used. 

    exports.first = function(args) {
    //code

}

exports.second = function(args) {
    //code
    exports.first(args);
}

    or declare a variable outside the function scopes

    var some_data = {};

exports.first = function(args) {
    //code
     some_data.test =args;
}

exports.second = function(args) {
    //code
    some_data.test =args;
}
