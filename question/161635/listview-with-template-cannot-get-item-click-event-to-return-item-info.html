Listview with template cannot get item click event to return item info

Application type: Mobile
Titanium SDK: 3.2.0
Platform & version: Andriod 4.0.2
Device : Andriod Emulator
Host Operating System: Windows 8.1
Titanium Studio: 3.2.0.201312191547

Hi Guys we need to get the info from the selected listview item.

XML
<ListView id="parcels" itemclick="getSelectedItem" defaultItemTemplate="template">
<Templates>
<ItemTemplate name="template">
<Label left="10" bindId="parcelNumber" id="parcelnumber" />
<Label left="100" bindId="status" id="status" />
</ItemTemplate>
</Templates>
<ListSection headerTitle="Parcels">
<ListItem parcelNumber:text="AD654321" status:text=" : Loaded "/>
<ListItem parcelNumber:text="AD654322" status:text=" : HUB"/>
<ListItem parcelNumber:text="AD654323" status:text=" : HUB"/>
<ListItem parcelNumber:text="AD654324" status:text=" : Loaded "/>
<ListItem parcelNumber:text="AD654325" status:text=" : HUB"/>
</ListSection>
</ListView>

JS

function getSelectedItem(e)
{
var section = $.parcels.sections[e.sectionIndex];

var item = section.getItemAt(e.itemIndex);

if(item.properties.title == 'Delivery')
{
    alert('Going to Deliver');
}
else if(item.properties.title == 'Collection')
{
    alert('Going to Collect');
}

};

    Try this:

    XML

    <Alloy>
    <Window id="win">        
        <ListView id="parcels" defaultItemTemplate="template"> 
            <Templates> 
                <ItemTemplate name="template"> 
                    <Label left="10" bindId="parcelNumber" id="parcelnumber" /> 
                    <Label left="100" bindId="status" id="status" /> 
                </ItemTemplate> 
            </Templates> 

            <ListSection headerTitle="Parcels"> 
                <ListItem title="Delivery" parcelNumber:text="AD654321" status:text=" : Loaded "/> 
                <ListItem title="Delivery" parcelNumber:text="AD654322" status:text=" : HUB"/> 
                <ListItem title="Collection" parcelNumber:text="AD654323" status:text=" : HUB"/> 
                <ListItem title="Delivery" parcelNumber:text="AD654324" status:text=" : Loaded "/> 
                <ListItem title="Collection" parcelNumber:text="AD654325" status:text=" : HUB"/> 
            </ListSection> 
        </ListView>
    </Window>
</Alloy>

    JS

    var getSelectedItem = function (e) {    
    var section = $.parcels.sections[e.sectionIndex];
    var item = section.getItemAt(e.itemIndex);

    if (item.properties.title === 'Delivery') {
        alert('Going to Deliver');
    } else if (item.properties.title === 'Collection') {
        alert('Going to Collect');
    }    
};
$.parcels.addEventListener('itemclick', getSelectedItem);

$.win.open();

    I have moved the event listener from your XML to the JS as I don't think it was firing from the XML.
