Is it better to declaring variables outside a module.exports function.

Which is a better practice.

var utility            = require('/ui/common/utilityFunctions');
var appDevice        = require( '/settings/interogateDevice');
var appServer        = require( '/settings/serverSetup' );

function afunc() {
    //code
      return something;
}

module.exports = afunc;

or…

function afunc() {
     var utility            = require('/ui/common/utilityFunctions');
     var appDevice        = require( '/settings/interogateDevice');
     var appServer        = require( '/settings/serverSetup' );
    //code
      return something;
}

module.exports = afunc;

    It depends on whether you want the variables redefined each time afunc() is called or just the first time the module is parsed. Declare them inside afunc() to redefine them with each call; outside to declare just once.
