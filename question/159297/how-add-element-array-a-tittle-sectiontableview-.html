    function compare(a, b) {
    if (a.competition_name < b.competition_name)
        return -1;
    if (a.competition_name > b.competition_name)
        return 1;
    return 0;
}

var listaAux = {
    "matches" : [{
        "id" : "27794",
        "round" : "13",
        "local" : "aaal",
        "visitor" : "bbb",
        "competition_name" : "A"
    }, {
        "id" : "27794",
        "round" : "13",
        "local" : "aaaal",
        "visitor" : "bbbb",
        "competition_name" : "B"
    }, {
        "id" : "27794",
        "round" : "13",
        "local" : "aaaaal",
        "visitor" : "bbbbb",
        "competition_name" : "A"
    }]
}

var w = Ti.UI.createWindow();

var tabla = Ti.UI.createTableView({
    backgroundGradient : {
        type : 'linear',
        startPoint : {
            x : "50%",
            y : "0%"
        },
        endPoint : {
            x : "50%",
            y : "100%"
        },
        colors : ["white", "#C3C3C3"]
    },
    separatorColor : '#2DBA43',
    top : '50dp'
});

var datos = [];
var prevSection;
var idx = -1;

listaAux.matches.sort(compare);

for (var i = 0; i < listaAux.matches.length; i++) {

    var row = Ti.UI.createTableViewRow({
        color : 'black',
        height : '75dp',
        backgroundSelectedColor : '#2DBA43',
    })

    var local = Ti.UI.createLabel({
        text : listaAux.matches[i].local,
        textAlign : 'center',
        font : {
            fontSize : 30
        },
        color : 'black',
        left : 0

    });

    var visitante = Ti.UI.createLabel({
        text : listaAux.matches[i].visitor,
        textAlign : 'center',
        font : {
            fontSize : 30
        },
        color : 'black',
        right : 0

    });

    row.add(local);
    row.add(visitante);

    if (prevSection !== listaAux.matches[i].competition_name) {
        prevSection = listaAux.matches[i].competition_name;
        idx++;

        datos[idx] = Ti.UI.createTableViewSection({
            headerTitle : listaAux.matches[i].competition_name
        });
    }

    datos[idx].add(row);
};

tabla.data = datos;

w.add(tabla);
w.open();
