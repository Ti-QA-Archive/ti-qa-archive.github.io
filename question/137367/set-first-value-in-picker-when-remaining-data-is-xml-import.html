    hi,

    try this

    var column = Ti.UI.createPickerColumn();

var index = Ti.UI.createPickerRow({title:'Select Interior Color'});
column.addRow(index);

var interiorRow = Ti.UI.createPickerRow({
    title:dataArrayResults
});

column.addRow(interiorRow);
data[x++] = interiorRow;

var picker = Ti.UI.createPicker();

picker.add([column]);
