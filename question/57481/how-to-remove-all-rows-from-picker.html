  • this is my solution for pickers with one column:

    picker.clear = function(){
    if(picker.columns[0]) {
        var col = picker.columns[0];
        var len = col.rowCount;
            for(var x = len-1; x >= 0; x-- ){
                    var row = col.rows[x];
                    col.removeRow(row);
            }
    }
}
