  • hi guy!, this works for me:

    var win = Ti.UI.createWindow({
    backgroundColor:'white'
    });

    var imageView = Titanium.UI.createImageView({
    image:'http://www.appcelerator.com/wp-content/uploads/2009/06/titanium_desk.png',
    width:261,
    height:178,
    top:20
    });

    imageView.addEventListener('click',function(e){

    Titanium.Facebook.appid = 'your facebook app id';
Titanium.Facebook.permissions = ['publish_stream']; // Permissions your app needs

var data = {
    link: "http://www.appcelerator.com",
    name: "Wall up app",
    message: "The e-commerce social network",
    caption: "Wal up",
    picture: "http://developer.appcelerator.com/assets/img/DEV_titmobile_image.png",
    description: "App de wall up en construccion"
};
Titanium.Facebook.dialog("feed", data, function(e) {
    if (e.success) {
        //alert("Success!  From FB: " + e.result);
         alert('Ha sido posteado en tu muro');
    } else {
        if (e.error) {
            alert(e.error);
        } else if (e.cancelled) {
            alert('Cancelled');
        } else {
            alert("Unkown result");
        }
    }
});

    });
    win.add(imageView);

    win.open();
