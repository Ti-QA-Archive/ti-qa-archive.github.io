Here is the window with the button:

login.js

function login(){


var createUserWin = require('createUser');


var loginWindow = Ti.UI.createWindow({
    backgroundColor: '#cecece'
});


var createUserButton = Ti.UI.createButton({
    title: 'Create a new account',
    bottom: 60
});

loginWindow.add(createUserButton);

createUserButton.addEventListener('click', function() {
    var createUser = new createUserWin();

    return createUser;
});


return loginWindow;

}

window that is opened as a result of button click:

createUser.js

function createUserWindow(){


var createUserWin = Ti.UI.createWindow();

var usernameText = Ti.UI.createLabel({
    text: 'Username:',
    left: 20,
    top: 15
});

createUserWin.add('usernameText');

var usernameTextField = Ti.UI.createTextField({
     attributedHintText: 'Username',
     borderStyle: Ti.UI.INPUT_BORDERSTYLE_ROUNDED,
     width: 250, 
     height: 60

});

createUserWin.add('usernameTextField');


return createUserWin;

}

module.exports = createUserWindow;

    Hi,

    please use 

    createUserWin.add(usernameText);

createUserWin.add(usernameTextField);

    instead of

    createUserWin.add('usernameText');
createUserWin.add('usernameTextField');
