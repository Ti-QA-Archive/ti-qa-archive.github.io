  • Hi,

    Please run the below code in your emulator. I hope, it will help to you to solve your problem 

    var win = Ti.UI.createWindow({backgroundColor: 'white'});
var listView = Ti.UI.createListView();
var country = Ti.UI.createListSection({ headerTitle: 'Country'});
var sections = [];
var listData=[];

var db=Ti.Database.open("firstDB");
db.execute('create table if not exists country(id text,con text)');    
var row=db.execute('select * from country');

if(row.rowCount==0)
{
    db.execute('insert into country values(?,?)','1','Bangladesh');
    db.execute('insert into country values(?,?)','2','India');
    db.execute('insert into country values(?,?)','3','Pakistan');
    db.execute('insert into country values(?,?)','4','USA');
    db.execute('insert into country values(?,?)','5','Canada');
    db.execute('insert into country values(?,?)','6','Japan');
    db.execute('insert into country values(?,?)','7','China');
    db.execute('insert into country values(?,?)','8','US');
    row=db.execute('select * from country');
}
while (row.isValidRow())
{
  listData.push({
      properties:{
          title:row.fieldByName('con')+'***'+row.fieldByName('id'),
          id:row.fieldByName('id')
      }
  });
  row.next();
}
country.setItems(listData);
sections.push(country);

listView.sections = sections;
win.add(listView);
win.open();

listView.addEventListener('itemclick', function(e) {
    var item=e.itemIndex;
    //alert(e.section.items[item].properties.id)
    deleteItem(e.section.items[item].properties.id);
    country.deleteItemsAt(e.itemIndex,1);

});

function deleteItem(id)
{
    db.execute('DELETE FROM country WHERE id=?',id);
    alert('delete From Database')
}
