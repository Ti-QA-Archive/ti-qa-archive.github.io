  • Hello Matt,

    Simple Demo for slider. You can also create widget for it.

    ###index.xml

    <Alloy>
    <Window id="winHome" >
        <View id="leftView">
            <TableView id="tableView" class="tblView"/>
        </View>
        <View id="mainView">
            <View id="navView">
                <Button id="btnLeft" onClick="showLeftPanel" color="black"/>
            </View>
        </View>
    </Window>
</Alloy>

    ###index.tss

    "#winHome":{
    backgroundColor: 'red',
    zIndex: 0
}
"#leftView":{
    height: '100%',
    width:'50%',
    top: 0,
    left: 0,
    backgroundColor: 'blue',
    zIndex: 0
},

".tblView":{
    left: 0,
    top: 0,//'45dp',
    bottom :0,
    scrollable: false,
    backgroundColor: '#282725',
    width: '100%',
},

"#mainView":{
    top: 0,
    zIndex: 2,
    height: '100%',
    width : '100%',
    left : 0,
    backgroundColor: 'pink',
}

"#navView":{
    top: 0,
    backgroundImage: 'barImg.png',  //   put your image path here
    height: "50dp",
    width: '100%',
},

"#btnLeft":{
    image: ' btnImage.png',  //  put your image path here
    left: "10dp",
    top: "14dp",
    title : "left",
    height: Ti.UI.SIZE,
    width: Ti.UI.SIZE,
    style: 'PLAIN',
    backgroundColor: 'transparent'
}

    ###index.js

    $.winHome.open();

var animateToLeft = Ti.UI.createAnimation({
    left : 0,
    curve : Ti.UI.ANIMATION_CURVE_EASE_OUT,
    duration : 500,
});

var animateToRight = Ti.UI.createAnimation({
    left : '50%',
    curve : Ti.UI.ANIMATION_CURVE_EASE_OUT,
    duration : 500,
});

var position = "left";
var showLeftPanel = function(e) {
    if (position == "left") {
        $.mainView.animate(animateToRight);
        position = "right";
    } else {
        $.mainView.animate(animateToLeft);
        position = "left";
    }
};

    i hope this helps. thanks.
