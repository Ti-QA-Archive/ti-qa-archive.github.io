    Try the following, I just ran a check and it looks like the background modes were not written to the info.plist using the older <iphone> tags:

    <ios>
 <plist>
  <dict>
    <key>UIBackgroundModes</key>
      <array>
       <string>audio</string>
      </array>
  </dict>
 </plist>
</ios>

    • @Stephen Feather, finally got both working using the following:

      <ios>   
    <plist>
        <dict>
    <!--<dict>-->
        <key>UIBackgroundModes</key>
          <array>
           <string>audio</string>
          </array>
    <!--</dict>-->

        <key>CFBundleURLTypes</key>

        <array>
            <!--<dict>-->
                <key>CFBundleURLName</key>
                <string>com.grabradioworld.4fm</string>
                <key>CFBundleURLSchemes</key>
                <array>
                    <string>com.grabradioworld.4fm</string>
                </array>
            <!--</dict>-->
        </array>
       </dict>
    </plist>
    </ios>
