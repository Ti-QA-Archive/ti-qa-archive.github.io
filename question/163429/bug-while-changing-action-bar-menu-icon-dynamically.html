bug while changing action bar menu icon dynamically

i am trying to toggle the menu icon dynamically in android action bar. it's working only in first time. In notifiImage.addEventListener block of below code, the else condition never executed, why ?

var win = Ti.UI.createWindow({
    backgroundColor : '#36A9E1',
    barColor : '#66B2FF',
    fullscreen : true
});

var actionBar;
win.addEventListener("open", function(e) {
    actionBar = win.activity.actionBar;
    if (actionBar) {
        actionBar.icon = "/images/TitleHorse.png";
        actionBar.title = "";
    }

    win.activity.onCreateOptionsMenu = function(e) {
        var menu = e.menu;
        var notifiImage = menu.add({
            icon : "/images/blurnotify.png",
            showAsAction : Ti.Android.SHOW_AS_ACTION_ALWAYS
        });

        notifiImage.addEventListener("click", function(e) {
            if (e.source.icon = "/images/blurnotify.png") {
                Ti.API.info('always executed');
                e.source.icon = "/images/focusnotify.png";
            } else {
                Ti.API.info('never executed');
                e.source.icon = "/images/blurnotify.png";
            }
        });
    };
});

    Hi Nuibb,

    Please copy and paste this code in your app.js and run it its working perfectly fine.

    var win = Ti.UI.createWindow({
    backgroundColor : '#36A9E1',
    barColor : '#66B2FF',
    fullscreen : true
});

var actionBar;
win.addEventListener("open", function(e) {
    actionBar = win.activity.actionBar;
    if (actionBar) {
        actionBar.icon = "/images/TitleHorse.png";
        actionBar.title = "";
    }

    var activity = this.activity;

    win.activity.onCreateOptionsMenu = function(e) {
        var menu = e.menu;
        var notifiImage = menu.add({
            icon : "KS_nav_ui.png",
            showAsAction : Ti.Android.SHOW_AS_ACTION_ALWAYS
        });
        var bool = false;
        notifiImage.addEventListener("click", function(e) {
            if (!bool) {
                Ti.API.info(menu.getItems());
                this.icon = "KS_nav_views.png";
                bool = true;
            } else {
                Ti.API.info('never executed');
                this.icon = "KS_nav_ui.png";
                bool = false;
            }
        });
    };
});

win.open();
