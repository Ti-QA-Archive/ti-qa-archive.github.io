function createRadiobutton(q,line, n, yVal, arr) { // Create the buttons
    for (var i = 0; i < n; i++) {
        var question = Titanium.UI.createLabel({
            text: q, // Sets the question
            value: i, // Give a unique value to the question
            color: '#000000', // Color of the text
            height: 150, // set the height of the question view
            width: 250, // set the width of the question view
            top: yVal, // gets the y value to place the questions
            left: 30, // set 30 px from the left
            font: {fontSize: 13} // Font styling
        });
        // Create Radiobutton
        var rBtn = Titanium.UI.createImageView({
            image: './images/radio-off.png',
            width: 57,
            height: 57,
            top: yVal + 45, // use the same y value as the question place it 45 px lower to center
            left: 320 + (65 * i), // set the left value based on the nr of buttons
            value: false, // set default value -- leave false
            name: i, // give a unique name for checking
            line:line // add line numbre to found this object where it is in the array
        });
        qArray.push(rBtn); // add the buttons to the window
        rBtn.addEventListener('click', toggleButton); // add the click listener
    }
    win1.add(question); // add the questions only once
    //win1.add(qArray);
    return qArray;
};

//////////////////////////////////////////////////

function getOrientation(o) {
    switch (o) {
        case Titanium.UI.PORTRAIT: {return 'portrait';}
        case Titanium.UI.UPSIDE_PORTRAIT: {return 'portrait';}
        case Titanium.UI.LANDSCAPE_LEFT: {return 'landscape';}
        case Titanium.UI.LANDSCAPE_RIGHT: {return 'landscape';}
        default : return 'unknown'
    }
}

Ti.Gesture.addEventListener('orientationchange',function(e){
    var orientation = getOrientation(e.orientation);
    // Do stuff for Portrait and Lanscape
    if (orientation == 'portrait') { 
        topBar.width = pWidth; // Switch the width, cause the device does not change the size
        bottomBar.width = pWidth; // Switch the width, cause the device does not change the size
        bottomBar.top = pHeight - barHeight; // Reposition it for portrait
    } else if (orientation == 'landscape') { 
        topBar.width = pWidth; // Switch the width, cause the device does not change the size
        bottomBar.width = pWidth; // Switch the width, cause the device does not change the size
        bottomBar.top = pHeight - barHeight; // Reposition it for landscape       
    }
});
