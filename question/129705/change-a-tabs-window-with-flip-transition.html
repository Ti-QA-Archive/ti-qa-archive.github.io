  • In app.js…

    
var winMYOECC = Titanium.UI.createWindow({  
    url: 'window_myoecc.js'
});
var winMYOECC_login = Titanium.UI.createWindow({  
    url: 'window_myoecc_login.js'
});

var tabMYOECC = Titanium.UI.createTab({  
    icon:'icons/icon-myoecc.png',
    title:'My OECC'
    window:winMYOECC_login
});



//-- snip --

tabGroup.addTab(tabMYOECC); 
tabGroup.open();

    In window_myoecc_login.js … I've attached an event listener to a login button in the window…

    
button_logIn.addEventListener('click', function()
{

    Titanium.UI.currentTab.open(winMYOECC,{animated:true, transition:Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT});
    fetchMyData();

});

    winMYOECC will show data specific to the user's account.
    winMYOECC_login is a login form.
