Getting the position of views in scrollView

Hey guys,

I was wondering is it possible to get the y position of views(I got 10 for them) in the scrollView!

Is it possible? If so, how?

Many thanks!

var self = Ti.UI.createWindow({
        backgroundColor : '#ffffff',
        navBarHidden : true,
        layout : 'horizontal'
    });

var scrollView = Ti.UI.createScrollView({
        backgroundColor : 'pink',
        contentWidth : '60%',
        contentHeight : 'auto',
        showVerticalScrollIndicator : true,
        scrollType : 'vertical',
        layout : 'vertical'
    });

for ( i = 0; i <= 10; i++) {
    var view = Ti.UI.createView({
            backgroundColor : '#336699',
            top : paddingCommonTop,
            height : squareValue,
            width : squareValue,
            bottom : paddingCommonBtm,
            title : 'I am View ' + i,
            viewNo : i
        });

    scrollView.add(view);
}

self.add(scrollView);
