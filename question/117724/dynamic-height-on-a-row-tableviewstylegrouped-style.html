Dynamic height on a row TableViewStyle.GROUPED style

Hi. So i've been looking in the Q&A to fix my problem, but well, i didn't find anything… I mean, i've seen that some people had the same problem, but questions were asked like 12 months ago…

Here my problem. I want to use a TableView with a TableViewSection and TableViewRow. The Kitchen Sink explains very well how to do that, and the following code work as expected :

var dataInfo = [];      
dataInfo[0] = Ti.UI.createTableViewSection({
    headerTitle:'my title'
});
var row = Ti.UI.createTableViewRow(
{
    height:'auto'
});

var label = Ti.UI.createLabel({
        text:text,
        height:'auto',
        width:'auto',
        left:5,
        right:5,
        top:5,
        bottom:5
});

row.add(label);

dataInfo[0].add(row);

var tableview = Titanium.UI.createTableView({
        data:dataInfo,
        style:Titanium.UI.iPhone.TableViewStyle.PLAIN
});

The height of the row is dynamic according to the size of the text of the label, thanks to height:'auto' (in the label AND the row)

The thing is I want to use 

style:Titanium.UI.iPhone.TableViewStyle.GROUPED

and not the PLAIN style… When i'm using the GROUPED style for iPhone, i have some artefact that i don't understand. Some row looks fine, and other not. Worst, sometimes, scrolling in the table view make some text appears…

here's a picture of my problem :

http://img683.imageshack.us/img683/8530/pbmtable.png

You can see that the label don't fit in the 2d row, i don't understand why.

Is this a bug ? Am i doing something wrong ? Is there a fix ?

Thanks for help in advance !

  • The following reproduce the problem. 

    var currentWin = Ti.UI.currentWindow;

var data = []; 

var tableView = Ti.UI.createTableView({
    top:10,
    style:Ti.UI.iPhone.TableViewStyle.GROUPED,        
    // minRowHeight:64,
});

var txt_data = ['text in one line', 'text in more lines - text in more lines - text in more lines - text in more lines - text in more lines'];
var curr_row;
for (var i=0; i < 1000; i++) {
    curr_row = Ti.UI.createTableViewRow({
        height:'auto',
        className:'row',                    
    }); 


    var textView = Ti.UI.createView({
        height:'auto',
        layout:'vertical',
        left:70,
        top:5,
        bottom:5,
        right:10,
        touchEnabled:'false'
    });    

    var title = Ti.UI.createLabel({
        height:'auto',
        // top:0,
        left:0,
        text:txt_data[Math.random() < 0.3 ?0 :1],
        textAlign:'left',
        color:'red',    
        font:{fontWeight:'bold',fontSize:13},

        touchEnabled:'false'
    });
    textView.add(title);

    var subtitle = Ti.UI.createLabel({
        height:'auto',
        left:0,
        top:2,

        text:txt_data[Math.random() < 0.3 ?0 :1],
        textAlign:'left',
        color:'black',

        font:{fontWeight:'bold',fontSize:15},

        touchEnabled:'false'
    });
    textView.add(subtitle);

    curr_row.add(textView);

    var imageView = Ti.UI.createImageView({
        height:54,
        width:54,
        left:5,
        top:5,

        image:'appicon.png',
        touchEnabled:'false'
    });

    curr_row.add(imageView);

    data.push(curr_row);
};

tableView.setData(data);
currentWin.add(tableView);

    The only solution I found is to disable style:Ti.UI.iPhone.TableViewStyle.GROUPED.
    If someone find other solution, please, let me know!
