  • Hi,

    In your posted code, the request is never sent, as the request.send() is within the onload.

    Here's an example of sending a request and handling the JSON response:

    var xhr = Titanium.Network.createHTTPClient();
var url = "http://www.myurl.com/getStatus.php?ticketcode=" + code;

xhr.onload = function(){
    var response = JSON.parse(this.responseText);
    Ti.API.log(response);
    //alert status of first element in list
    alert(response[0].status);        
};

xhr.open("GET", url);
xhr.setRequestHeader('Content-Type', 'application/json; charset=utf-8');
xhr.send();
