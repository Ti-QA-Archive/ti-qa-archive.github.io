  • You should use something like below.

    var window = Ti.UI.createWindow({
    backgroundColor: "#fff"
});
var view = Ti.UI.createView({
    layout: 'vertical'
});

for(i=1;i<=5;i++){
    var lbl = Ti.UI.createLabel({
        text: 'Label ' + i
    });
    lbl.addEventListener("click",function (e){
        alert(e.source.text);
    });
    view.add(lbl);
}

window.add(view);
window.open();
