    • we have a little different code

      ~~~

          function subscribe(_token) {
        Cloud.PushNotifications.subscribeToken({
            device_token : _token,
            channel : 'mainChannel',
            type : (OS_ANDROID) ? "gcm" : "ios"
        }, function(e) {
            Ti.API.debug(e);
        });
    }

    if (OS_ANDROID) {
        CloudPush.enabled = true;
        CloudPush.singleCallback = true;

        CloudPush.retrieveDeviceToken({
            success : function(e) {
                Ti.App.Properties.setString("DEVICE_TOKEN", e.deviceToken);
                subscribe(e.deviceToken);
            },
            error : function(e) {
            }
        });
    } else {
        Ti.Network.registerForPushNotifications({
            types : [Ti.Network.NOTIFICATION_TYPE_BADGE, Ti.Network.NOTIFICATION_TYPE_ALERT, Ti.Network.NOTIFICATION_TYPE_SOUND],
            success : function(e) {
                Ti.App.Properties.setString("DEVICE_TOKEN", e.deviceToken);
                subscribe(e.deviceToken);
            },
            error : function(e) {
                Ti.API.debug("Register For Push Notifications Error: " + e.error);
            },
            callback : function(e) {
                App.handlePush(e.data);
            }
        });

      ~~
