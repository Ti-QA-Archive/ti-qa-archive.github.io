ACS Push Notifications

Can somebody please help and explain how I get ACS Push notifications to work with my IOS app.
Here is my code:

//REGISTER FOR PUSH NOTIFICATIONS
Ti.Network.registerForPushNotifications({
types: [
     Ti.Network.NOTIFICATION_TYPE_BADGE,
     Ti.Network.NOTIFICATION_TYPE_ALERT,
     Ti.Network.NOTIFICATION_TYPE_SOUND
],
 success:function(e) {
    Ti.API.info('Device Token :  '+e.deviceToken);
},
 error:function(e) {
    Ti.API.warn("push notifications failed: "+e);
},
 callback:function(e) {
    alert ('Push received');
}
});

This is at the start of my app.js file. When running my app on iPad I'm not even prompted about push notifications and get an error in Console saying: [WARN] push notifications failed: [object Object]
I really need this to work and I have searched everywhere and still found nothing :(
