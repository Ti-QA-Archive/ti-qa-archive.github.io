    try this

    orderReq.open('GET', 'http://mywebservices.net/myapi.axd');

var params = {
    method : 'sendmail',
    sender : 'value',
    receiver : 'value',
    subject : 'crash report',
    message : 'this is the crash report send for test',
};

orderReq.send(params);
