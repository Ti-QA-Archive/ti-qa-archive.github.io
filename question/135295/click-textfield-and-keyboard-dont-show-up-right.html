  • That code i have included to my win1, it's not all because it's big, but less only the button save code. I think this is all the code for keyboard!

    var rssNewLinkField = Ti.UI.createTextField({
  value:'http://',
  width: 450,
  height: 60,
  left: 5,
  top: 15,
  hintText: 'URL',
  borderRadius: 10,
  returnKeyType:Titanium.UI.RETURNKEY_NEXT,
  borderStyle:Titanium.UI.INPUT_BORDERSTYLE_NONE
});

var table = Ti.UI.createTableView({
  rowHeight: 60
});

var section = Ti.UI.createTableViewSection({headerTitle: "Add your favorite source!"});

var row = Ti.UI.createTableViewRow();
row.add(rssNewLinkField);
row.hasChild = false;
row.className = "field";
section.add(row);

table.setData([section]);
win1.add(table);
