  • for all resolutions >= 320*480 use these generic formulas to set height and width 

    var screenWidth =  Ti.Platform.displayCaps.platformWidth;
var screenHeight = Ti.Platform.displayCaps.platformHeight;

function GetHeight(value) {

    var temp = (value * 100) / 480;
    return parseInt((screenHeight * temp) / 100);
}

function GetWidth(value) {
    var temp = (value * 100) / 320;
    return parseInt((screenWidth * temp) / 100);
}

  • Maybe this will help. The way I dealt with this in a recent app was to utilize the dp unit, but calculate the number based on a percentage. The result is that it scales nicely to the screen size. I used this both for font sizes and screen positioning and the app worked great from phone to tablet and kept its form very well (with trial and error to get the numbers right at different sizes).

    Me.percentWidth = function(relative) {
    return Math.round(Ti.Platform.displayCaps.platformWidth * (relative/100));
};
Me.percentHeight = function(relative) {
    return Math.round(Ti.Platform.displayCaps.platformHeight * (relative/100));
};
Me.percentWidthDp = function(relative, plus) {
    if (!plus) plus = 0;
    var px = Ti.Platform.displayCaps.platformWidth * (relative/100);
    var dp = Math.round(px / (Titanium.Platform.displayCaps.dpi / 160));
    return (dp + parseInt(plus))+ 'dp';
};
Me.percentHeightDp = function(relative, plus) {
    if (!plus) plus = 0;
    var px = Ti.Platform.displayCaps.platformHeight * (relative/100);
    var dp = Math.round(px / (Titanium.Platform.displayCaps.dpi / 160));
    return (dp + parseInt(plus))+ 'dp';
};

  • Thank you all, I found a better solution which is to use this code and add it to
    tiapp.xml

        <supports-screens android:resizeable=["true"| "false"]
                      android:smallScreens=["true" | "false"]
                      android:normalScreens=["true" | "false"]
                      android:largeScreens=["true" | "false"]
                      android:xlargeScreens=["true" | "false"]
                      android:anyDensity=["true" | "false"]
                      android:requiresSmallestWidthDp="integer"
                      android:compatibleWidthLimitDp="integer"
                      android:largestWidthLimitDp="integer"/>

    you can chose either false or true
