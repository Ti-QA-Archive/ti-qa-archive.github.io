    so your table Data is in sample.js and you want to use it in main.js??
    in sample.js

    var tableData = [ ] ;//assuming it is array

exports.poList = function()
{
    //your all stuff code etc
}

//one more function that wil return your tableData to main.js

exports.getTableData = function()
{
    return tableData;
}

    In your main.js

    var mysample = require('sample');
//call that function to get that table data

var myTableData = mysample.getTableData(); //now your myTableData has the table data from the sample.js file
