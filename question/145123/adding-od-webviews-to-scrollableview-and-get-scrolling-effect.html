Adding od webViews to ScrollableView and get scrolling effect

Hi I am developing an application using Titanium which uses a scrollable View, contains views pushed to the 'views' property of scrollable View. Each view contains a web view. When I add empty views to 'views' property of scrollable View I am able to achieve scrolling. But when I add webViews to views and if I bind that views to the scrollabelView I am not able to scroll. Below is my code:

var win = Ti.UI.createWindow();

var view1 = Ti.UI.createView({

backgroundColor : 'white'

});
var view2 = Ti.UI.createView({

backgroundColor : 'red'

});

var WebView1 = Titanium.UI.createWebView({

width : '90%',

height : '90%',

backgroundColor : 'red',    

showScrollbars : false,

scalesPageToFit : false,

enableZoomControls : false

});

var WebView2 = Titanium.UI.createWebView({

width : '90%',

height : '90%',

backgroundColor : 'white',    

showScrollbars : false,

scalesPageToFit : false,

enableZoomControls : false

});

view1.add(WebView1);

view2.add(WebView2);

var scrollableView = Ti.UI.createScrollableView({

views : [view1, view2],

showPagingControl : true,

currentPage : 0

});

win.add(scrollableView);

win.open();

Can anyone share a possible solution….

Thanks in Advance…
