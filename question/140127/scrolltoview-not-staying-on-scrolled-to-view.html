scrollToView not staying on scrolled to view

I have a scrollview with 2 views sitting side by side. I can swipe to scroll but I also want to put a next/forward button on the first view and a backward/previous button on the second view.

I have done this but when the scrollToView is called it does slide the screen like a swipe would do, but instead of staying on the second view, it instantly slides back. The weird thing is….if I HOLD the button….it stays on the second view.

Is there a bug? Am I doing something wrong, what can I do to fix it?

var MyScrollView = Titanium.UI.createScrollableView({
    views:[MyView1,MyView2],
    showPagingControl:true
});
win.add(MyScrollView);

forward.addEventListener('click',function()
{
MyScrollView.scrollToView(1);
});

I also tried 2 in the scrollToView just in case :-)

    Try this sample code. All 5 buttons in this example take you to the desired views, and they stay there. You might have to look deeper in your code for other things that might be causing the effect you're describing. Maybe you have an unexpected event listener that's getting triggered?

    // this sets the background color of the master UIView (when there are no windows/tab groups on it)
Ti.UI.setBackgroundColor('#000');

var win1 = Ti.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});


var v1 = Ti.UI.createView ({
    backgroundColor: '#f66',
    top: 0,
    bottom: 0,
    left: 0,
    right: 0
});

var v2 = Ti.UI.createView ({
    backgroundColor: '#6f6',
    top: 0,
    bottom: 0,
    left: 0,
    right: 0
});

var btn1 = Ti.UI.createButton ({
    title: 'Home'    
});

btn1.addEventListener ('click', function (e) {
    sv.scrollToView (0);
});

v2.add (btn1)

var v3 = Ti.UI.createView ({
    backgroundColor: '#66f',
    top: 0,
    bottom: 0,
    left: 0,
    right: 0
});

var btn2 = Ti.UI.createButton ({
    title: 'Home'    
});

btn2.addEventListener ('click', function (e) {
    sv.scrollToView (0);
});

v3.add (btn2)

var sv = Ti.UI.createScrollableView ({
    top: 0,
    left: 0,
    right: 0,
    bottom: 100,
    views: [v1, v2, v3]
});

win1.add (sv);

var bv1 = Ti.UI.createView ({
    height: 40,
    left: 0,
    right: 0,
    bottom: 50,
    layout: 'horizontal'
});

var btn3 = Ti.UI.createButton ({
    height: 40,
    width: '33%',
    title: 'View 1'    
});

btn3.addEventListener ('click', function (e) {
    sv.scrollToView (0);
});

var btn4 = Ti.UI.createButton ({
    height: 40,
    width: '33%',
    title: 'View 2'    
});

btn4.addEventListener ('click', function (e) {
    sv.scrollToView (1);
});

var btn5 = Ti.UI.createButton ({
    height: 40,
    width: '33%',
    title: 'View 3'    
});

btn5.addEventListener ('click', function (e) {
    sv.scrollToView (2);
});


bv1.add (btn3);
bv1.add (btn4);
bv1.add (btn5);

win1.add (bv1);

win1.open ();
