Auto Height not working correctly on iOS

Okay, so this page looks and works correctly on android, but the buttons and input fields don't show up on iOS… Why is 'auto' not working for height? is there a fix for this? am I missing something?

MainView.js

//MainView Component Constructor
function MainView() {
    var self = Ti.UI.createScrollView({
            showVerticalScrollIndicator:true,
            showHorizontalScrollIndicator:false
    });

    /*
     * START UI ITEMS
    */
    var applicationNameLabel = Ti.UI.createLabel({
        top:'13.33dp',
        color:'#FFFFFF',
        font:{
            fontSize:'16sp'
        },
        text:L('app_name'),
        height:'auto',
        width:'auto'
    });
    self.add(applicationNameLabel);

    var newsletterLabel = Ti.UI.createLabel({
        top:'260dp',
        font:{
            fontSize:'16sp'
        },
        color:'#FFFFFF',
        text:L('newsletter'),
        height:'auto',
        width:'284dp'
    });
    self.add(newsletterLabel);

    var nameInputView = Ti.UI.createView({
        layout:'horizontal',
        top:'330dp',
        width:'277dp',
        height:'auto'
    });

    var nameLabel = Ti.UI.createLabel({
        color:'#FFFFFF',
        text:L('name'),
        width:'50dp',
        height:'auto'
    });

    var nameField = Ti.UI.createTextField({
        left:'27dp',
        hintText:L('enter_name'),
        width:'200dp',
        height:'auto',
        borderStyle:Ti.UI.INPUT_BORDERSTYLE_ROUNDED
    });

    nameInputView.add(nameLabel);
    nameInputView.add(nameField);
    self.add(nameInputView);

    var emailInputView = Ti.UI.createView({
        layout:'horizontal',
        top:'380dp',
        width:'277dp',
        height:'auto'
    });

    var emailLabel = Ti.UI.createLabel({
        color:'#FFFFFF',
        text:L('email'),
        width:'50dp',
        height:'auto'
    });

    var emailField = Ti.UI.createTextField({
        left:'27dp',
        hintText:L('enter_email'),
        width:'200dp',
        height:'auto',
        borderStyle:Ti.UI.INPUT_BORDERSTYLE_ROUNDED
    });

    emailInputView.add(emailLabel);
    emailInputView.add(emailField);
    self.add(emailInputView);

    var joinInputView = Ti.UI.createView({
        layout:'horizontal',
        top:'430dp',
        width:'284dp',
        height:'auto'
    });

    var joinButton = Ti.UI.createButton({
        font:{
            fontWeight:'bold',
            fontSize:'16sp'
        },
        title:L('join'),
        height:'auto',
        width:'67dp',
        left:'217dp'
    });

    joinInputView.add(joinButton);
    self.add(joinInputView);

    /*
     * END UI ITEMS
    */

    /*
     * START UI ACTIONS
     */

    joinButton.addEventListener('click',function(e) {
    });

    /*
     * END UI ACTIONS
     */

    return self;
}

module.exports = MainView;
