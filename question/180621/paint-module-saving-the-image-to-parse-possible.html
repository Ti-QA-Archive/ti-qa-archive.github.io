  • Yes It is possible. Here is the sample code.

    
var note_id = "Note";
var i = 1;

var file = Titanium.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory + '/' + note_id + '_' + i + '.png');
var blob = file.read();

var vw_main = Titanium.Painter.createPaintView({
        top : 0,
        left : 0,
        right : 0,
        height : 850,
        width : 750,
        strokeColor : '#0000FF',
        backgroundColor : '#FFF',
        image : blob    
});
win.add(vw_main);

function saveDoc() {
    var file_obj = vw_main.toImage();            
    var f = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory + '/' + note_id + '_' + i + '.png');    
    f.write(file_obj);
}

    Hope this helps.
