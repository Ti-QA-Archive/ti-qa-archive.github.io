  • this should get you started

    var xhr_request = Titanium.Network.createHTTPClient();
xhr_request.setTimeout(5000);
xhr_request.onerror = function(e) { alert(e);
}
xhr_request.open("GET", 'http://www.gurkanbulbuloglu.net/gurkanim.js');
xhr_request.onload = function() {
    try {
        var json_data = JSON.parse(this.responseText);
        var imageCount = json_data.images.length;
        for(var i in json_data.images) {
            Ti.API.info("square: " + json_data.images[i].square)
            Ti.API.info("image: " + json_data.images[i].image)
            Ti.API.info("note: " + json_data.images[i].note)
        }
    } catch(e) {
        alert(e);
    }
}
xhr_request.send();
