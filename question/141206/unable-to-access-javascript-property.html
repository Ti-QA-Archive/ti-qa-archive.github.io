  • When you add an item to a view, you cannot access it via the dot notation.

    You will have to add it as a property:

    function CommonOneUi(){
    var baseView = Ti.UI.createView({
        height:'auto',
        width:'auto',
        backgroundColor:'#000',
        layout:'composite'
        // right:'1 dip'
    })
    this.songsBut = Ti.UI.createButton({
        // textAlign:Ti.UI.TEXT_ALIGNMENT_CENTER,
        // title:'SONGS',
        height:'45 dip',
        width:'33%',
        backgroundImage:'/images/drawable/song_grey.png',
        backgroundSelectedImage:'/images/drawable/song_red.png'

    })

    baseView.add(songsBut);
    baseView.songsBut = songsBut;

    return baseView
}

module.exports = CommonOneUi;
