When is better to close DB?

Hi, I have a little dilemma - When is better to close database? Example:

var db = Ti.Database.open("xy");

db.execute("SOME SQL QUERY...");
// ...

someButton.addEventListener("click", function(){
    db.execute("SOME SQL QUERY...");
};

someButton_1.addEventListener("click", function(){
    db.execute("SOME SQL QUERY...");
};

// ...

db.close();

OR

var db = Ti.Database.open("xy");

db.execute("SOME SQL QUERY...");
// ...
db.close();


someButton.addEventListener("click", function(){
    var db = Ti.Database.open("xy");
    db.execute("SOME SQL QUERY...");
    db.close();
};

someButton_1.addEventListener("click", function(){
    var db = Ti.Database.open("xy");
    db.execute("SOME SQL QUERY...");
    db.close();
};

// ...

And one more question. Do I need for better App performance something like this?

// ...

function closeDB(){
    if(db)
        db.close();
}

win.addEventListener("close", closeDB);
Ti.App.addEventListener("pause", closeDB);
Ti.App.addEventListener("resume", openDB);

    Keep your database open only as long as you are reading/writing then close it again to keep the memory footprint as low as possible
