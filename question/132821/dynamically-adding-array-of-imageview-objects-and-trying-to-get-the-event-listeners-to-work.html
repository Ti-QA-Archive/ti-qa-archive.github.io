Dynamically adding array of imageView objects and trying to get the event listeners to work

Hello,

I am trying to do a drag and drop solution and use a code i got from one of the developers. It works if I use one single object. but when I try to use it with an array of object, the event would trigger an error saying the array is undefined. Can anyone help to find a way around this problem?

var circle = [];
var circle2 = [];

var circleX = [];
var circleY = [];

for(var i = 0; i < balance; i++){

    circleX[i] = Math.floor(Math.random()*(Ti.Platform.displayCaps.platformWidth-50));
    circleY[i] = aligny/2+Math.floor(Math.random()*40)-20;

circle[i] = Titanium.UI.createImageView({
    image:'money.png',
    height:50,
    width:50,
    anchorPoint:{x:25,y:25},
    borderRadius:25,
    backgroundColor:'#336699',
    opacity:0.3,
    top:circleY[i],
    left:circleX[i]
});
circle2[i] = Titanium.UI.createImageView({
    image:'money.png',
    height:50,
    width:50,
    anchorPoint:{x:25,y:25},
    borderRadius:25,
    backgroundColor:'#003366',
    top:circleY[i],
    left:circleX[i]
});

win.add(circle[i]);
win.add(circle2[i]);

circle2[i].addEventListener('touchmove', function(e)
{
    circle[i].setTop(e.y + circleY[i] - 25);
    circle[i].setLeft(e.x + circleX[i] - 25);

});

circle2[i].addEventListener('touchend', function(e)
{
    circleY[i] = circle[i].getTop();
    circleX[i] = circle[i].getLeft();

    circle2[i].setTop(circle[i].getTop());
       circle2[i].setLeft(circle[i].getLeft());

    Titanium.API.info('x:'+circleX[i]);  
});

}

The problem is I cannot call using the e.source object as I need to call an object array which is circle which is not the source of object of the event listener.

thank you in advance

    Try this in for loop:

    for(var i = 0; i < balance; i++){

    win.add(circle[i]);
    win.add(circle2[i]);

    (function(i){
        circle2[i].addEventListener('touchmove', function(e) {
            circle[i].setTop(e.y + circleY[i] - 25);
            circle[i].setLeft(e.x + circleX[i] - 25);

        });

        circle2[i].addEventListener('touchend', function(e) {
            circleY[i] = circle[i].getTop();
            circleX[i] = circle[i].getLeft();

            circle2[i].setTop(circle[i].getTop());
            circle2[i].setLeft(circle[i].getLeft());

            Titanium.API.info('x:' + circleX[i]);
        });
    }(i));
}
