    Hi

    This should help you simplify the way you add items to your window, by moving your rows into this format if you are using a scrollView it will increase the the readability and manage the styles of how it is presented.

    var addRow = function (obj) {
    var view = Ti.UI.createView({
        bottom: 5,
        height: Ti.UI.SIZE,
        layout: 'vertical',
        left: 10,
        right: 10,
        top: 5,
        width: Ti.UI.FILL
    });
    if (obj.field) {
        view.add(Ti.UI.createLabel({
            color: '#999',
            font: {
                fontSize: 12
            },
            height: Ti.UI.SIZE,
            left: 0,
            text: obj.field,
            textAlign: 'left',
            width: Ti.UI.FILL
        }));
    }
    if (obj.value) {
        view.add(Ti.UI.createLabel({
            color: '#000',
            font: {
                fontSize: 14,
                fontWeight: 'bold'
            },
            height: Ti.UI.SIZE,
            left: 0,
            text: obj.value,
            textAlign: 'left',
            width: Ti.UI.FILL
        }));
    }
    return view;
};

    Usage;

    scrollview.add(addRow({
    field: 'First Name',
    value: 'Malcolm'
}));
scrollview.add(addRow({
    field: 'Last Name',
    value: 'Hollingsworth'
}));
// specifically your
scrollview.add(addRow({
    field: 'Navn',
    value: sql.fieldByName('navn')
}));
scrollview.add(addRow({
    field: 'Adresse',
    value: sql.fieldByName('adresse')
}));
// also possible
scrollview.add(addRow({
    value: 'Just the value if you want'
}));

    Tweak your scrollVIew code as follows;

    var scrollview = Titanium.UI.createScrollView({
    color: '#000',
    contentHeight: Ti.UI.SIZE,
    contentWidth: Ti.UI.SIZE,
    height: Ti.UI.FILL,
    layout: 'vertical',
    showHorizontalScrollIndicator: true,
    showVerticalScrollIndicator: true,
    width: Ti.UI.FILL
});
