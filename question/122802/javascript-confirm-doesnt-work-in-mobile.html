    Use an alertWindow instead

    var alertWindow = Titanium.UI.createAlertDialog({
                    title: 'Warning',
                    message: 'This will delete all filled in forms from your device.  Are your sure?',
                    cancel:1,
                    buttonNames: ['OK','Cancel']
                });

                alertWindow.addEventListener('click', function(ev) {
                    switch(ev.index) {
                        case 0:
                            alert('ok pressed');
                            break;
                        case 1:
                            alert('cancel pressed');
                            break;

                    }
                });
                alertWindow.show();

    Hope this helps!
