@SuppressLint({ "WrongViewCast", "WrongViewCast" })
public class MainActivity extends Activity implements OnClickListener
{
private Button Calculate;
EditText Distance;
EditText Angle;
private RadioButton dtoa;
private RadioButton atod;

@Override
public void onCreate(Bundle savedInstanceState)
{
    super.onCreate(savedInstanceState);
    setContentView(R.layout.activity_main);

    Distance=(EditText)findViewById(R.id.dtoa);
    Angle=(EditText)findViewById(R.id.atod);
    dtoa = (RadioButton)this.findViewById(R.id.dtoa);

    dtoa.setChecked(true);

    atod = (RadioButton)this.findViewById(R.id.atod);

    Calculate = (Button)this.findViewById(R.id.button1);

    Calculate.setOnClickListener(this);

}

public void onClick(View arg0) 
{

   if (dtoa.isChecked()) 
      {

          convertDistanceToAngle();

      }

   if (atod.isChecked()) 
        {

           convertAngleToDistance();

        }

}

protected void convertDistanceToAngle() 
{

   double val = Double.parseDouble(Angle.getText().toString());
   Distance.setText(Double.toString(val/0.67));

};  

protected void convertAngleToDistance() 
{

    double val = Double.parseDouble(Distance.getText().toString());
    Angle.setText(Double.toString(val*0.67));

};

}
