  • You can pass the array of images path to next view controller. Below is the sample code.

    index.js

    
var smallView = Ti.UI.createImageView({
    images: ['/test1.png', '/test2.png']
});

btn.addEventListener('cliclk', function(e) {
    var imagesArr = smallView.images;
    Alloy.createController('NextController', {
        images: imagesArr
    }).getView().open();
});

    NextController.js

    
var args = arguments[0] || {};
