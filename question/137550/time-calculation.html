  • Given a unix timestamp, you can use this js function:

        prettyDate = function(time){
        var monthname = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];
        var date = new Date(time*1000);
        var diff = (((new Date()).getTime() - date.getTime()) / 1000);
        var day_diff = Math.floor(diff / 86400);
        if ( isNaN(day_diff) || day_diff < 0 ){
            return '';
        }
        if(day_diff >= 31){
            var date_year = date.getFullYear();
            var month_name = monthname[date.getMonth()];
            var date_month = date.getMonth() + 1;
            if(date_month < 10){
                date_month = "0"+date_month;
            }
            var date_monthday = date.getDate();
            if(date_monthday < 10){
                date_monthday = "0"+date_monthday;
            }

            return month_name + " " + date_monthday + ", " + date_year;
        }
        return day_diff == 0 && (
            diff < 60 && "just now" ||
            diff < 120 && "1 minute ago" ||
            diff < 3600 && Math.floor( diff / 60 ) + " minutes ago" ||
            diff < 7200 && "1 hour ago" ||
            diff < 86400 && "about " + Math.floor( diff / 3600 ) + " hours ago") ||
        day_diff == 1 && "Yesterday" ||
        day_diff < 7 && day_diff + " days ago" ||
        day_diff < 31 && Math.ceil( day_diff / 7 ) + " week" + ((Math.ceil( day_diff / 7 )) == 1 ? "" : "s") + " ago";
    }
