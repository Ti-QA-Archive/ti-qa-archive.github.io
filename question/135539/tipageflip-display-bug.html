Ti.PageFlip display bug?

I am making a flip book composed of several Views, each of them contains 6 images and labels.

The module works great, except that if the user interacts in the middle of a flip (like trying to flip pages quickly) the module flips out and goes blank until it is swiped again at a slower speed.

It looks like it is trying to display the page after the one it's transitioning to but it shows a blank page instead.

Tried disabling enableBuiltInGestures when flip stars and re enabling when the the change even is triggered but the page it gets caught mid-flip, unresponsive because the change event is never generated.

Any ideas?

  • Absolutely!

    Ti.include("functions.js");
var PageFlip = require('ti.pageflip');
var win = Ti.UI.createWindow({backgroundColor:"#fff"});
var itemsPerPage = 9;
var exhibitItems = parseXMLData();
var pages = [];
var pointer = 0;
for( b = 0; b < Math.ceil(exhibitItems.length / itemsPerPage); b++) {
    var page = Ti.UI.createView({
        backgroundColor : "white"
    });
    var oneLayout = Ti.UI.createView({
        layout : "horizontal",
        width : Titanium.Platform.displayCaps.platformWidth,
        height : Titanium.Platform.displayCaps.platformHeight,
        layout : "horizontal"
    });
    cards = [];
    for( i = 0; i < 9; i++) {
        theItem = exhibitItems.item(pointer);
        if(theItem != null) {
            pointer++;
            Ti.API.info(theItem.getElementsByTagName('image').item(2).getAttribute('file'));
            cards[i] = GetCard();
            cards[i].add(GetImage(theItem));
            cards[i].add(GetLabel(theItem));
            oneLayout.add(cards[i]);
        }
    }
    page.add(oneLayout);
    pages[b] = page;
}
var pageflip = PageFlip.createView({
    transition : PageFlip.TRANSITION_CURL,
    pages : pages
});
win.add(pageflip);
win.open();
