  • Hi Ronald Vogel,

    You can follow this code for playing sound when launching apps.

    var win = Ti.UI.createWindow({
    backgroundImage: '#fff',
    layout :'vertical'    
});

var file = Titanium.Filesystem.getFile(Titanium.Filesystem.resourcesDirectory,'songLocationFrom ResourceFile');

    // load from file object
var sound = Titanium.Media.createSound({url:file});

win.addEventListener('open',function(e){

    sound.play();

});

var stop = Titanium.UI.createButton({
        title:'Stop',
        height:40,
        width:145,
        right:10,
        top:60
    });
    stop.addEventListener('click', function()
    {
        sound.stop();
        //pb.value = 0;
    });
    win.add(stop);


win.open();

    Thanks
