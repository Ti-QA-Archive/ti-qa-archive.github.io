    Try this one :

    var arrRows = [];

var length = 2 ;  // as your data length

for(var i=0; i<length; i++)
{
    var row = Ti.UI.createTableviewRow({
    height  : '50dp',
    rowId : i,
    obj : 'row',
    layout : 'horizontal'
});

    var lblDate = Ti.UI.createLabel({
    left : '5dp',    // set this as your UI
    text :  '12/2/2013 8:28:30 AM'  //  as your data
    height : Ti.UI.SIZE,
    width :  Ti.UI.SIZE,
});

    row.add(lblDate);
    var lblValue = Ti.UI.createLabel({
    left : '15dp' ,  // set this as your UI
    text :  '140'  //  as your data
    height : Ti.UI.SIZE,
    width :  Ti.UI.SIZE,
});
    row.add(lblValue);

    var lblPrice = Ti.UI.createLabel({
    left : '15dp' ,  // set this as your UI
    text :  '66$'  //  as your data
    height : Ti.UI.SIZE,
    width :  Ti.UI.SIZE,
});

    row.add(lblPrice);
    var imgCar = Ti.UI.createImageView({
    left : '5dp' ,  // set this as your UI
    image :  'image.png'  //  as your image path
    height : Ti.UI.SIZE,
    width :  Ti.UI.SIZE,
});
    row.add(imgCar);
    var btnDelete = Ti.UI.createButton({
    left : '5dp' ,  // set this as your UI
    image :  'image.png'  //  as your image path
    height : Ti.UI.SIZE,
    width :  Ti.UI.SIZE,
});

    row.add(btnDelete);
arrRows.push(row);
}

$.tableview.data = arrRows;
