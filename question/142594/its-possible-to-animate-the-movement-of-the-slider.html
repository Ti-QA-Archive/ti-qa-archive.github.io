button.animate ({left:100,duration:500});

you can do this? thanks

    • my result 

      var win = Ti.UI.createWindow({
    title:'ciao'
});

var slider = Ti.UI.createSlider({
    top:50,
    left:20,
    height:10,
    width:500,
    value:0.0,
    min:0,
    max:40

});

var label = Ti.UI.createLabel({
    text:'0.0',
    top:50,
    left:540,
    width:'auto',
    color:'red'
});


var button = Ti.UI.createButton({
    title:'animazione',
    top:100,
    left:20
})

var ciao=0;
button.addEventListener('click',function(e){
    var interv = setInterval(function(e){
        if(slider.value<20)
        {ciao+=1;
        label.text=""+ciao;
        slider.value=ciao;}else {clearInterval(interv)}
    },20);
});



slider.addEventListener('change',function(e){
    newValue = parseValue(e.source.value, 1);
    label.text=""+newValue;
});


parseValue = function(value, value_to_round) {   //funzione arrotonda valori slider ad interi
        value /= value_to_round;
        value = Math.round(value);
        value *= value_to_round;
        return value;
    };

win.add(slider);
win.add(label);
win.add(button);


win.open();
