function random()

random.getRandom = function (aleatorio)
{
var arreglo=[];
var num;
var repe;    
for(var i=0; i<aleatorio; i++)
{
    do{
        num=(Math.floor(Math.random()*aleatorio)+1);
        repe=repetido(num,aleatorio,arreglo);
    }while(repe!=false);
    arreglo.push(num);
}
return arreglo;
}

function repetido(num,con,arreglo)
{
var repe=false;
for(var i=0; i<con; i++)
{
if(num==arreglo[i])
repe=true;
}    
return repe;
}

random.validarEjercicio=function(arreglo)
{
    //var corre=true;
    var arregloValidar=[];

    {arregloValidar[i]=i}
    for(var i=0; i<arreglo.length; i++)

    return true;
}
