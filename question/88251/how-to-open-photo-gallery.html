How to open Photo Gallery

I'm trying to have a options dialog box allow a user to open their Photo Gallery. That's all. I can't seem to figure out what I'm doing wrong?

var opengallery = Ti.UI.createOptionDialog({
    title:'Image Saved',
    options:['Open Photo Gallery', 'Close'],
    buttonNames:['Close']
    }).show();

opengallery.addEventListener('click', function(e) {
    Titanium.Media.openPhotoGallery({});
});

    try it like this

    
var opengallery = Ti.UI.createOptionDialog({
    title:'Image Saved',
    options:['Open Photo Gallery', 'Close']
    });

opengallery.addEventListener('click', function(e) {
    Titanium.Media.openPhotoGallery({});
});

opengallery.show();
