updating tableview data. howto?

I want to change my tableview data on an event listener. that means, I want to change my tableview data, after I added it to the tableview.

I tried 2 different ways.

First (changing row directly):

collectionWin.addEventListener('refreshBackground', function () {
    tableView.backgroundImage = Ti.Filesystem.resourcesDirectory + '/images/background/' + lightapps.userConfig.getParam('design') + '.png';
    var data = tableView.data;
    for (var c = 0; c < data.length; c++) {
        data[c].backgroundImage = Ti.Filesystem.resourcesDirectory + '/images/background/' + lightapps.userConfig.getParam('design') + '.png';
    }
});

Second (changing row, collect them into array and set the tablview data afterwards):

collectionWin.addEventListener('refreshBackground', function () {
    tableView.backgroundImage = Ti.Filesystem.resourcesDirectory + '/images/background/' + lightapps.userConfig.getParam('design') + '.png';
    var result = [];
    var data = tableView.data;
    for (var c = 0; c < data.length; c++) {
        data[c].backgroundImage = Ti.Filesystem.resourcesDirectory + '/images/background/' + lightapps.userConfig.getParam('design') + '.png';
        result.push(data[c]);
    }
    tableView.data = result;
});

Both will not really work. Anybody has an idea where I do a mistake?
