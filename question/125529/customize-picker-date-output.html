    e.value is a javascript Date() object, so you can just format that to get what you want. Note this isn't a very clean example, but will get you started:

    picker.addEventListener('change', function(e) {
    var myDate = e.value;
    var month = myDate.getMonth();
    month += 1;
    if(month < 10) {
        month = "0" + month
    }
    var day = myDate.getDate();
    if(day < 10) {
        day = "0" + day;
    }
    var year = myDate.getFullYear().toString();
    year = year.substr(2, 2);
    label.text = day + month + year;
});

    Learn more about the Date Object.
