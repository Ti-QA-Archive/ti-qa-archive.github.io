Adding image to button on android action bar

Hi, I was wondering if there was a way to add an image to a button on the action bar. Currently I have tried background Image and Image but none of them seem to work. Any Ideas?

This is the code that i used to craete a button on the action bar:

var activity = newReleases.activity;

activity.onCreateOptionsMenu = function(e){
  var menu = e.menu;
  var menuItem = menu.add({ 
    title: "Back", 
    //backgroundImage:  "/images/back_btn.png", Does not work
    showAsAction: Ti.Android.SHOW_AS_ACTION_IF_ROOM,
    //image: "/images/back_btn.png"
  });
  menuItem.addEventListener("click", function(e) {
    newReleases.close();
  });
};
