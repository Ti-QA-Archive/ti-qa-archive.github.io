opentok on SDK 3.2.0

Hi,

in this beta version of opentok it is possible to work in the newest SDK branche.

I got a connection, but after adding the videoview to parent it crashes. Here the log:

-[NSNull krollObjectForBridge:]: unrecognized selector sent to instance 0x3b201090";

It happens in the marked line:

exports.create = function(_actionsbutton) {
    var publisher, publisherView;
    var CONF = {
        sessionId : Ti.App.Properties.getString('opentok_sessionId'),
        apiKey : Ti.App.Properties.getString('opentok_apiKey'),
        token : Ti.App.Properties.getString('opentok_token')
    };
    var self = Ti.UI.iPad.createPopover({
            width : 360,
            height : 540,
            title : 'Videokonferenz',
    });
    var OpenTok = require("com.tokbox.ti.opentok");
    var session = OpenTok.createSession({
        sessionId : CONF.sessionId
    });
    session.connect(CONF.apiKey, CONF.token);
    session.addEventListener("sessionConnected", function(event) {
        publisher = session.publish();
        publisherView = publisher.createView({
            width : 360,
            height : 540
        });
        self.add(self.publisherView);  // <== here it crashes
    });
    return self;
}

Any ideas what going wrong?
