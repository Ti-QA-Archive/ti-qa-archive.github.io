    var viewHelp = Ti.UI.createView({
        opacity: 0.5,
        backgroundColor: "#db5b33",
        height: 65,
        top: 207,
        width: "100%",
        borderColor: "#752f1b",
        borderWidth: 1,
        backgroundGradient:{type:'linear',
            colors:['#f16437','#b74a2a'],
            startPoint:{x:'0%',y:0},
            endPoint:{x:'0%',y:65},
            backFillStart:false}        
    });

    viewHelp.addEventListener("click", function(){
        this.opacity = 1;    
        lblHelp.opacity = 1;    

        lblGreat.opacity = 0.5;
        viewGreat.opacity = 0.5;
        lblFine.opacity = 0.5;
        viewFine.opacity = 0.5;
    });
