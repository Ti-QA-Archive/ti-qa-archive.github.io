    you can use this two property to take the two screen size

    Ti.Platform.displayCaps.platformHeight (height in px)
Ti.Platform.displayCaps.platformWidth (width in px)

    with this method you can figure out when the device is turned

    
function getOrientation(o) {
    switch (o) {
        case Titanium.UI.PORTRAIT: return 'portrait';
        case Titanium.UI.UPSIDE_PORTRAIT: return 'portrait';
        case Titanium.UI.LANDSCAPE_LEFT: return 'landscape';
        case Titanium.UI.LANDSCAPE_RIGHT: return 'landscape';
    }
}

Ti.Gesture.addEventListener('orientationchange',function(e){
    var orientation = getOrientation(e.orientation);

    }

});
