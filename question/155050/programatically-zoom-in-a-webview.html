  • Tried this ugly hack:

    function zoomIn(webview){
    var html = webview.getHtml();

    var htmlHack = '';
    htmlHack += 'var element = document.createElement("meta");';
    htmlHack += 'element.name = "viewport";';
    htmlHack += 'element.content = " initial-scale = 2.5";';
    htmlHack += 'var head = document.getElementsByTagName("head")[0];';
    htmlHack += 'head.appendChild(element);';
    webview.evalJS(htmlHack);
}

    That does actually change the zoom level. So i guess i could keep track of the scale myself and change it when the button is pressed. And set the initial-scale to the current zoom level.
