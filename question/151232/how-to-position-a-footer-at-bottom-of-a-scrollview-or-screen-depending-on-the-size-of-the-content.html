How to position a footer at bottom of a scrollView OR screen depending on the size of the content

I want to have a footer, which behaves as follows:
(1) When the scrollView.height is larger than the screen height, the footer should be positioned end of the scrollView.
(2) When the scrollView.height is smaller than the screen, the footer should be positioned at the bottom of the screen.

The following results in the footerView being always directly below myScrollView (also in case myScrollView does not extend beyond the size of the screen).

var myScrollView = Titanium.UI.createScrollView({
     width : Ti.UI.FILL,
       height: Ti.UI.FILL,
    contentHeight: Ti.UI.SIZE,
      contentWidth: Ti.UI.SIZE,
      layout:'vertical',
      showVerticalScrollIndicator: true
});

var myView = Ti.UI.createLabel({
    height: Ti.UI.SIZE,
    width: Ti.UI.FILL,
    top: 20,
    text: "text of flexible length"
});

myScrollView.add(myView);

var footerView = Ti.UI.createView({
    width : Ti.UI.FILL,
    height: 50,
    bottom: 0
});    
myScrollView.add(footerView);    

win.add(myScrollView);
