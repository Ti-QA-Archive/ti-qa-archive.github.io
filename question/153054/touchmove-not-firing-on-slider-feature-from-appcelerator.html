  • Hi Malcolm,

    Thanks for ur reply.i am using following code to deal with that.

    var win = Ti.UI.createWindow({
    backgroundColor : 'white',
    //layout : 'Horizontal',
});

var sliderLabel = Titanium.UI.createLabel({
    text : '0',
    color : '#000',
    font : {
        fontFamily : 'Helvetica Neue',
        fontSize : 15
    },
    textAlign : 'center',
    top : 0,
    width : 40,
    height : 40,
});
var slider = Titanium.UI.createSlider({
    min : 10,
    max : 50,
    value: 10,
    width : 268,
    height : 40,
    top : 90,
});

var contentBlock = Ti.UI.createView({
    top : 55,
    width : 40,
    height : 40,
    backgroundImage: 'popup-floor-build-view.png',
    left : 10,
}); 


contentBlock.add(sliderLabel);
win.add(slider);
win.add(contentBlock);


 var olt = Titanium.UI.create3DMatrix(), curX, curY;
 slider.addEventListener('touchmove', function(e) {
    var currentValue = Math.round(e.value);
    if (currentValue <= 10) {
        currentValue = 10;
    } else if (currentValue <= 25) {

        currentValue = 25;
    } else if (currentValue <= 40) {
        currentValue = 40;
    } else {
        currentValue = 50;
    }

    slider.value = currentValue;
    sliderLabel.text = currentValue;

    var deltaX = e.x - curX, deltaY = e.y - curY;
    olt = olt.translate(deltaX, deltaY, 0);
    contentBlock.animate({
        transform : olt,
        duration : 100
    });

});

slider.addEventListener('touchstart', function(e) {
    curX = e.x;
    curY = e.y;
});

slider.addEventListener('touchend', function(e) {
    var deltaX = e.x - curX, deltaY = e.y - curY;
    olt = olt.translate(deltaX, deltaY, 0);
    contentBlock.animate({
        transform : olt,
        duration : 100
    });
});

win.open();
