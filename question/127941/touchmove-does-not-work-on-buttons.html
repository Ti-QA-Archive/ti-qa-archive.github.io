Touchmove does not work on buttons?..

I have not been able to get the touchmove event to work with buttons even though I can get it to work with labels. Here is some testing code:

var win = Titanium.UI.createWindow({
    title:'Window',
    left:0,
    top:0,
    backgroundColor:'#FFF'
});

var label = Titanium.UI.createLabel({
    height:40,
    width:200,
    top:60,
    text:'Label (can touchmove)',    
    textAlign:'center',
    borderWidth:1,
    borderRadius:10
});
var button = Titanium.UI.createButton({
    height:40,
    width:200,
    top:10,
    title:'Button (cant touchmove)'
}); 
label.addEventListener('touchmove', function(e)
{
    alert("test"); 
});
button.addEventListener('touchmove', function(e)
{    
    alert("test"); 
}); 

win.add(label);
win.add(button);
win.open();
