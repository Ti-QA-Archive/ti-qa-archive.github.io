Adding method to my buttons

Hi, i normally create a buttons like this (this is simplified):

function bt(text) {
        var bt             = Ti.UI.createView(stylesheet.bt);

        var label         = Ti.UI.createLabel(stylesheet.label);
        label.text         = text;

        bt.add(label);
        return bt;
}
module.exports = bt;

And i call it: 

var bts                = require('ui/common/default/bt');
var bt                 = new bts('testButton');
container.add(bt);

Now i need to add methods to this button for make changes inside it in click events for example:

bt.addEventListener('click', function (e){
    bt.changeSomething();
});

How do you recommend to do that?

thanks

    Try this:

    function bt(text) {
        var bt          = Ti.UI.createView(stylesheet.bt);

        var label       = Ti.UI.createLabel(stylesheet.label);
        label.text      = text;

        bt.add(label);

    bt.changeSomething = function(params) {
        // Do whatever you want e.g.
        label.text  = "Changed Tex"
    }

        return bt;
}
module.exports = bt;
