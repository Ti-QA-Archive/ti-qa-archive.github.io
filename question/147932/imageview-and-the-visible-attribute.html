ImageView and the visible attribute

Hello everyone,

I am have this code of creating an Image view + a button:

var brep = Titanium.UI.createImageView({ 
    url: '/images/picture1.png',
    width:430,
    height:90,
    visible:false,
    });

var button = Titanium.UI.createButton({

        title:"Hit",
        height:80,
        width:80,
        top:40,
        left:150    
});

When I click on the button, I want that the imageView appears, I tried with that code, but in vain:

button.addEventListener("click", function(){
        //Here I couldn't find the visible attribute
    // brep.!!!!!
    });

How should I do?

  • try this ?

    var brep = Titanium.UI.createImageView({ 
    image:'/images/picture1.png',
    width:430,
    height:90,
    visible:false,
    });

var button = Titanium.UI.createButton({

        title:"Hit",
        height:80,
        width:80,
        top:40,
        left:150    
});

button.addEventListener('click', function(){
    brep.show();
});
