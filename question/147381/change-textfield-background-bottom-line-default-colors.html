var searchView = Ti.UI.createView({
                width : Ti.UI.FILL,
                height : screenHeight,
                bottom : screenHeight,
                backgroundColor : 'black',
                opacity : 0.7,
                zIndex : 5,
                touchEnabled : true,
                bubbleParent : false
            });

var searchCont = Ti.UI.createView({
                    width : Ti.UI.FILL,
                    height : Ti.UI.FILL,
                    bottom : 80,
                    top : 0,
                    opacity : 1,
                    backgroundColor : 'white'

                });

                var searchText = Ti.UI.createTextField({
                    width : Ti.UI.FILL,
                    left : 30,
                    right : 30,
                    top : 20
                });
                searchCont.add(searchText);
                searchView.add(searchCont);
win.add(searchView);
