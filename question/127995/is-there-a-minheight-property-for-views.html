  • Set the height to 'auto' and add a 'bottom' value to get your padding at the bottom. For example, here's a table view row with 10 pixels padding at the top, bottom, left, and right:

    var table = Ti.UI.createTableView();

var row = Ti.UI.createTableViewRow({
  height: 'auto'  
});
row.add(Ti.UI.createLabel({
  top: 10,
  bottom: 10,
  left: 10,
  right: 10,
  height: 'auto',
  text: "Lorem ipsum..." // long text deleted
}));

table.setData([row]);
