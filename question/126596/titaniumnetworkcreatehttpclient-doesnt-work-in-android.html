Titanium.Network.createHTTPClient() doesn't work in Android

I have a php script that return data from a remote database. It's working in ios flawlessly. But it doesn't work in Android at all. It always returns empty result.
I'm now suspecting the "xhr.send" doesn't send the right data or right format to the server.

From the server side, I use $POST_[SALESPERSON] to fetch the data.

Please advise. 

function getSales(passcode, salesperson){
    var phpName='www.url.com/somescript.php';

    var xhr = Titanium.Network.createHTTPClient();
    xhr.setTimeout(10000);
    xhr.onerror=function(e){
        alert(e);
    }
    xhr.onload = function(){
        Titanium.API.info(this.responseText);
        var json = JSON.parse(this.responseText);

        if (!json) { 
            Titanium.API.info('Error - Null return!'); 
            return false;
        }
        else{
        Titanium.API.info(json);    
        }


    }
    xhr.open('GET', phpName);
    xhr.send({ 
    SALESPERSON: salesperson,
    BASE64:Titanium.Utils.base64encode(passcode).toString()
    });

}
