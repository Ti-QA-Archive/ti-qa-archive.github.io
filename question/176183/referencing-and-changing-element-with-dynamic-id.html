Referencing and changing element with dynamic id

I have a tableview with plenty of tablerows. Below are the relevant part of the codes:

<TableView id="myTable"></TableView>

then I populate rows from json and pushed the table row to the alloy table. tablerows are inside a loop so it populates lots of rows depending on the number of json entries.

data = [];

var t = Ti.UI.createTableViewRow({
   id : array[i].name,
   ...
});
var l = Ti.UI.createLabel({
   text:  array[i].title,
...
});
t.add(l);

data.push(t);

$.myTable.data = data;

Then I set up the listener and this is where my problem starts:

$.myTable.addEventListener('touchend', function(e){
    var i = e.rowData.id;

   $[i].backgroundColor = '#fff';   // This gives an undefined error.

});

My problem is how to reference the ID i declared in the loop so I can change the row's properties when touched. $.i.backgroundColor = &#39;#fff&#39;; doesn't work because it will actually look for element with ID 'i'.

    Hello Nash,

    check this :

    index.xml

    <Alloy>
    <Window id="win">
        <TableView id="tableview"></TableView>
    </Window>
</Alloy>

    index.js

    
var dataArray = [];

for(var i=0; i<10; i++)
{
    var row = Ti.UI.createTableViewRow({
        backgroundColor : "#484848",
        height : "50dp",
        id : i,
        title:"Title " + i
    });
    dataArray.push(row);
};
$.tableview.setData(dataArray);

$.tableview.addEventListener('click', function(e){
    var sections = $.tableview.data; // grab the array of sections
    var section = sections[0]; // use just the first section
    var theRow = section.rows[e.indx];

    theRow.backgroundColor = "#484848";
});

$.win.open();
