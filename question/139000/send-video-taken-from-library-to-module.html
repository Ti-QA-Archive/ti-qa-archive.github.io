     mediaTypes: [Titanium.Media.MEDIA_TYPE_VIDEO],

    {
        alert('success');
        var customModule = require('com.mymodule');
        customModule.source=event.media;
        customModule.Start='0';
        customModule.end='10';
        var kno=customModule.example();
    }
