No menu bars when I open the example map_view.js from a modal window

I am attempting to show the Kitchen Sink example 'map_view.js' by clicking on a button in a webview window. When I click my button it shows the map but there are no top and bottom menu bars like there are when opened by Kitchen Sink. Here is the snippet where I try to do this:

clientTableView.addEventListener('click',function(e){            
            var w3 = Ti.UI.createWindow({title:e.row.heading});
                var wb3 = Ti.UI.createWebView();
                wb3.html = composeclientHTML(e.row.client_id,e.row.client_address,e.row.client_phone,e.row.client_city,e.row.client_logo,e.row.client_description);
                w3.add(wb3);
            var b3 = Ti.UI.createButton({
                title:'Close',
                style:Ti.UI.iPhone.SystemButtonStyle.PLAIN
                });
            w3.setLeftNavButton(b3);
            var b4 = Ti.UI.createButton({
                title:'Map',
                style:Ti.UI.iPhone.SystemButtonStyle.PLAIN
                });
            w3.setRightNavButton(b4);    
            w3.open({modal:true});            
            b4.addEventListener('click',function(){
                var w4 = Ti.UI.createWindow({
                url:'map_view.js'
                });
                w4.open();
            });    
            b3.addEventListener('click',function(){
                w3.close();
            });
        });                    
};

I don't know why the menu bars don't show up. This is an iPhone app, Titanium SDK 1.6.2, iPhone SDK 4.3. I do not get any errors or warnings from Titanium Developer 1.2.2. Any help on this would be greatly appreciated.
