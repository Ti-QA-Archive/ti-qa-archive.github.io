Why window animation not work while beginning top>=460?

I found an interesting behavior of Window animation. Would anybody please help to explain that?

My intent was to animate the window from bottom to top.
When I set the top attribute to 459 or less, the animation worked perfectly.
However, If I set it to be equal or larger than 460, the window did not appear at all.

Here's the code


var TOP_ANIMATE_START = 459;

var win = Titanium.UI.createWindow({
    layout: 'vertical',
    backgroundColor:'#003300'
});

var btn = Ti.UI.createButton({
    top: 100,
    left: 30,
    title: 'Open child window with animation'
});
btn.addEventListener('click', function() {
        var childWin = Titanium.UI.createWindow({
            layout: 'vertical',
            backgroundColor:'blue'
        });

        childWin.top = TOP_ANIMATE_START;

        var btn1 = Ti.UI.createButton({
            top: 50,
            left: 100,
            title: 'close child window'
        });
        btn1.addEventListener('click', function() {
            childWin.close();
        });

        childWin.add(btn1);
        childWin.open();

        childWin.animate({top:0, duration:1000}, function(){
        });
});

win.add(btn);
win.open();
