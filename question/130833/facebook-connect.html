Titanium.Facebook.permissions = ['publish_stream,manage_pages,user_photos,offline_access']; // Permissions your app needs

    if (e.success) {
        alert('Logged In');
    } else if (e.error) {
        alert(e.error);
    } else if (e.cancelled) {
        alert("Cancelled");
    }
