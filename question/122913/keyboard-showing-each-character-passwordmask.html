  • This code is straight forward. Android OS currently but plan on iOS once working.

    // the password text field
 var tf2 = Titanium.UI.createTextField({
    value:'',
    passwordMask:true,
    keyboardType:Titanium.UI.KEYBOARD_ASCII,
    top:191,
    width:240,    
    borderStyle:Ti.UI.INPUT_BORDERSTYLE_ROUNDED
  }); 

win.add(tf2);

// force landscape mode
win.orientationModes = [Ti.UI.LANDSCAPE_LEFT]; 
Ti.UI.orientation = Titanium.UI.LANDSCAPE_LEFT;
