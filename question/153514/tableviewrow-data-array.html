tableViewRow data array

Hi, am just scratching the surface on tableViewRow for Android and I wish to ask about how to create array of links (example row1.js, row2.js, row3.js) inside a tableViewRow such that when one clicks using the EventListener for each row, it corresponds to the url inside that table view rows. I did it using the conventional tableView with hasChild + urls inside but would be faster using a data array + tableViewRow. Please help.

Here is the reference:
SDK:3.1.0.GA, OSX Mountain Lion, Device is Android 2.3.

code sample:

var win=Ti.UI.currentWindow;

//I created  data array with three rows. Each row must link to row1.js, row2.js and row3.js respectively.
var data=[
    {text:"row1", url:"row1.js"},
    {text:"row2", url:"row2.js"},
    {text:"row3", url:"row3.js"}
];

//forloop
var rowData=[];

for(var i=0; i<data.length; i++){
    var content=Ti.UI.createLabel({
        text:data[i].text,
        font:{fontSize:17, fontFamily:"helvetica nueue",fontWeight:"bold"},
        color:"#000",
        textAlign:"left"
    });

    var row=Ti.UI.createTableViewRow({
        hasChild:true, 
        height:"auto", 
        backgroundColor:"red",
//how to insert url with array of links (row1.js, row2.js, row3.js)???In this case, it only links to row1.js
        url: "row1.js"
        });
    row.add(content);
    rowData.push(row);
};



var tableView=Ti.UI.createTableView({
    data:rowData
});

tableView.addEventListener("click",function(e){
    if(e.rowData.hasChild){
     var w=Ti.UI.createWindow({
         title:"hello",
         url:e.rowData.url
     });
    }
    w.open();
});

win.add(tableView);
