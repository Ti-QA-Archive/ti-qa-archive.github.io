    function DisplayTags(){
        //for tags data
        for(var i=tagsContainer.children.length - 1; i>=0; i--){
            tagsContainer.remove(tagsContainer.children[i]);
        }

        //tags
        if(media.tags.length > 0){
            for(var i=0; i<media.tags.length; i++){

                var length = require('BusinessLogic/util').getSizeTag(media.tags[i]);

                var tagButton = Ti.UI.createButton({
                    title: media.tags[i],
                    backgroundImage: 'images/tag.png',
                    color: 'black',
                    font: {fontSize: 13},
                    width: length + 18,
                    textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER,
                    left: 6,
                    top: 6,
                    height: 25
                });

                tagsContainer.add(tagButton);
            }

            tagsContainer.height = Ti.UI.FILL;
        }
        else{
            var tagsEmptyLabel = Ti.UI.createLabel({
                text: L('empty_tags'),
                font: {fontSize: 13},
                color: '#fff',
                top: 10
            });

            tagsContainer.add(tagsEmptyLabel);
        }
        }
