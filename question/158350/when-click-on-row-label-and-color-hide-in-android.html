When click on row, label and color hide in Android

I've some rows with 2 label. When I click on one row I would to color the text of first label and write other text in last label. Then if I click again, the color and the text change again. But in Android the second Label hides some time and the color of first Label back to initial state (text = Start, color = black), Why? This is my code:

row.xml

<Alloy>
    <TableViewRow id="row">
        <View id="view">
            <Label id="title" />
            <Label id="text" />
        </View>
    </TableViewRow>
</Alloy>

row.tss

"#row": {
      objName:                "row",
    selectedBackgroundColor:    "white",
    selectedColor:            "black",
    backgroundColor :        "white",
    color:                "black",
    height:                30
},
"#title": {
    left:                    0,
    touchEnabled:            false,
    color:                "black",
    text:                    "Color"
},
"#text": {
    right:                    0,
    touchEnabled:            false,
    color:                "black",
    height:                30,
    width:                70,
    backgroundColor:        "white",
    text:                    "Start"
}

row.js


$.row.addEventListener("click",function(e)
    {
        Alloy.Globals.changeColorAndText(e);
    });

index.js

Alloy.Globals.changeColorAndText = function(e)
{
    if (e.row.children[0].children[1].text == "Text Green")
    {
        e.row.children[0].children[0].color     = "orange"; //first label
        e.row.children[0].children[1].text        = "Text Orange"; //second label
    }
    else
    {
        e.row.children[0].children[0].color     = "green"; //first label
        e.row.children[0].children[1].text        = "Text Green"; //second label
    }
};
