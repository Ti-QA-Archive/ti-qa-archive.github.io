  • Taken exact from doc

    var win2 = Titanium.UI.createWindow({
    backgroundColor: 'red',
    title: 'Red Window'
});

var win1 = Titanium.UI.iOS.createNavigationWindow({
   window: win2
});

var win3 = Titanium.UI.createWindow({
    backgroundColor: 'blue',
    title: 'Blue Window'
});

var button = Titanium.UI.createButton({
    title: 'Open Blue Window'
});
button.addEventListener('click', function(){
    win1.openWindow(win3, {animated:true});
});

win2.add(button);
var button2 = Titanium.UI.createButton({
    title: 'Close Blue Window'
});
button2.addEventListener('click', function(){
    win1.closeWindow(win3, {animated:false}); //win3.close() will also work!!
});

win3.add(button2);
win1.open();

    Here is reference also

    http://docs.appcelerator.com/titanium/3.0/#!/api/Titanium.UI.iOS.NavigationWindow

    • and how you will use this from ANOTHER .js ?

      on the same .js it's not problem… I'm facing this problem because I use ANOTHER.js to show the next window's objects and stuffs…

      anyway… I already figure it out…

      I have to pass my navigation window through the windows…

      /////////////////////////////////////////////////////////////////////////////
// app.js
/////////////////////////////////////////////////////////////////////////////

var win = Ti.UI.createWindow({
    url: 'someURL.js'
});


var navWin = Ti.UI.iOS.createNavigationWindow({
    window: win
});

win.navWin = navWin;

navWin.open();


/////////////////////////////////////////////////////////////////////////////
// someURL.js
/////////////////////////////////////////////////////////////////////////////
var win  = Ti.UI.currentWindow;
var navWin = win.navWin;

var nextWindow = Ti.UI.createWindow({
    url: 'anotherURL.js',
    navWin : navWin
});

navWin.openWindow( nextWindow );
