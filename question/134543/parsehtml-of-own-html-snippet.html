parseHTML of own HTML-snippet

Hi,
I try to parse a remote HTML-document:

ctrl.tides = ( function() {
    var api = {};
    parseHTML = function(html) {
        var table = html.getElementsByTagName("table")[0];
    };
    api.getForcast = function(id, callback) {
        var url = 'http://mobile.bsh.de/cgi-bin/gezeiten/was_mobil.pl?ort='+ id + '&zone=Gesetzliche+Zeit&niveau=KN';
        var xhr = Titanium.Network.createHTTPClient({});
        xhr.onload = function() {
            try {
                var html = this.responseText;
                var data = parseHTML(html);
                callback(data);
            } catch(e) {
            };
        };
        xhr.open('GET', url);
        xhr.send();
    };
    return api;
}());

My problem: message = &quot;&#39;undefined&#39; is not a function (evaluating &#39;html.getElementsByTagName(\&quot;table\&quot;)&#39;)&quot;;

  • self solved:

    
ctrl.tides = ( function() {
    var api = {};
    api.getForcast = function(id, callback) {
        var url = 'http://mobile.bsh.de/cgi-bin/gezeiten/was_mobil.pl?ort='+ id + '&zone=Gesetzliche+Zeit&niveau=KN';
        Ti.Yahoo.yql("select * from html where url=\"" + url + "\" and xpath=\"//table\"",function(res) {
            callback(JSON.stringify(res));
        });
    };
    return api;
}());
