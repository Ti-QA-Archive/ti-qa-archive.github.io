    • Hi Jérémie, how did you got RGraph to work on Android? I have an app that should display a simple pie chart, I get and empty page. I'm using Titanium SDK 2.1.4, Android Device 2.3.4, Android Device 4.0.3.

      This is the code I'm using:

      htmlContent = "<!DOCTYPE HTML>" +
            "<html>" +
            "<head>" +
            "<meta name='viewport' content='user-scalable=yes, width=device-width, initial-scale = 1.0, minimum-scale = 1.0, maximum-scale = 10.0' /> <meta name='apple-mobile-web-app-capable' content='yes' />" +
            "<script src='lib/RGraph/RGraph.common.core.js.lib'></script>" +
            "<script src='lib/RGraph/RGraph.common.tooltips.js.lib'></script>" +
            "<script src='lib/RGraph/RGraph.pie.js.lib'></script>" +
            "</head>" +
            "<body>" +
            "<p>...Chart <b>here</b>....</p>"+
            "<canvas id='canvas' class='canvasStyle' width='400' height='400' style='border:1px solid blue;align:center'>This should not be seen</canvas>" +
            "<script type='text/javascript' language='JavaScript'>" +
            "window.onload = function(){" +
            "var graph = new RGraph.Pie('canvas', [1000.00, 1200.00, 1400.00]); " +
            " graph.Set('chart.labels', ['Ago', 'Sep', 'Oct']); " +
            " graph.Set('chart.tooltips.effect', 'fade');" +
            " graph.Set('chart.exploded', [15])" +
            " graph.Set('chart.title', 'Totales de ordenes'); " +
            " graph.Draw();" +
            "}" +
            "</script>" +
            "</body>" +
            "</html>";
            Ti.API.info('html content : ' + htmlContent);
            _webview.html = htmlContent;
