    I experienced the same issue with my previous project. Here's a workarround that worked for my app:

    • Set row0's backgroundColor to selectedColor - That's just a cosmetic change to archive the goal :)
    • Reset row0's backgroundColor on first click.
    var myData = [
    {title:'Row 1', hasChild:true, test:'', backgroundColor: 'blue'},
    {title:'Row 2', hasChild:true, test:''},
    {title:'Row 3', hasChild:true, test:''},
    {title:'Row 4', hasChild:true, test:''},
    {title:'Row 5', hasChild:true, test:''},
    {title:'Row 6', hasChild:true, test:''},
    {title:'Row 7', hasChild:true, test:''},
    {title:'Row 8', hasChild:true, test:''},
    {title:'Row 9', hasChild:true, test:''}
];

var tableView = Titanium.UI.createTableView({
    top: 0,
    width: 320,
    height: 350,
    data: myData,
     allowsSelection: true
});

tableView.selectRow(0); // this will select your 1st row by default
var w = Ti.UI.createWindow();
w.add(tableView);
w.open();

var firstClick = true;
tableView.addEventListener('click', function(e) {
    if (firstClick) {
        tableView.sections[0].rows[0].backgroundColor = 'white';
        firstClick = false;
    }
});

    Hope this can give you some hints,

    Minh
