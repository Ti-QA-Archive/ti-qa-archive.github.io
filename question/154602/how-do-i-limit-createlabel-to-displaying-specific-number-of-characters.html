How do I limit createLabel to displaying specific number of characters

I have the user input a name and then display it on the screen using Titanium.UI.createLabel… if the name is too long it spills over the designated area … how can I limit the number of characters displayed not to exceed a certain number say 65 characters ?

var label2 = Titanium.UI.createLabel({
    color:'white',
    text: 'Your Player' ,
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    left:180,
    top:645,
    width:'auto'
});

  • Why not just truncate your text at that ?

    var labelText = 'This is a very long piece of text This is a very long piece of text This is a very long piece of text This is a very long piece of text ' ;

var displayText = labelText.length > 65 ? labelText.substring(0,64) : labelText;

var label2 = Titanium.UI.createLabel({
    color:'white',
    text: displayText ,
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    left:180,
    top:645,
    width:'auto'
});
