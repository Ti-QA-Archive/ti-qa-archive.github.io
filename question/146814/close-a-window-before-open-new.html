Close a window before open new

Hello everyone =]

Imagine that I have 3 windows: 1, 2 and 3.
In window 1 I have a button that open window 2. In window 2 a button that open window 3. And in window 3 a button that closes it and a button that open window 1 again.
If I still opening windows my app will be slow as hell.
Is there a way to close current window before open the new one?
Thanks!

    Hi Carlos, Here is the complete code which is working fine.. just edit it as per your need.

    var Win1 = Ti.UI.createWindow({
    title:"win1",  // title will not be shown, as its a bug in titanium with single window. To show title either use navGroup or tabGroup
    backgroundColor:"red"
});

var Win2 = Ti.UI.createWindow({
    title:"win2",
    backgroundColor:"blue"
});

var Win3 = Ti.UI.createWindow({
    title:"win3",
    backgroundColor:"green"
});


var button1 = Ti.UI.createButton({
    title:"button1",
    height:40,
    width:50,
    left:50,
    top:150
});

button1.addEventListener("click",function(){
    Win1.close();
    setTimeout(function(){
        Win2.open();
    },300);

});

Win1.add(button1);

var button2 = Ti.UI.createButton({
    title:"button2",
    height:40,
    width:50,
    left:50,
    top:150
});

button2.addEventListener("click",function(){
    Win2.close();
    setTimeout(function(){
        Win3.open();
    },300);
});

Win2.add(button2);

var button3 = Ti.UI.createButton({
    title:"button3",
    height:40,
    width:50,
    left:50,
    top:150
});

button3.addEventListener("click",function(){
    Win3.close();
    setTimeout(function(){
        Win1.open({transition:Titanium.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});
    },300);
});

Win3.add(button3);


Win1.open();
