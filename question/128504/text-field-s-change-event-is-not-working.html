    • Mayuri

      There may be things you need to change to get this to work fully, as it has not been tested (because your code isn't a test case), however, this it is probably what you want:

      var win = Ti.UI.currentWindow;

var searchText = Ti.UI.createTextField({
  hintText:'Search for a hospital',
  top: 10,
  width: 200
});

win.add(searchText);

var tableview = Ti.UI.createTableView({
  top: 10
});

win.add(tableview);

searchText.addEventListener('change',function(e){
  var db = Ti.Database.open('example');
  var rows = db.execute('SELECT hospital_name FROM Hospital_name WHERE hospital_name LIKE ?', e.value);

  var dataArray = [];

  while (rows.isValidRow()){
    dataArray.push({title:rows.fieldByName('hospital_name'), hasChild:true, path: 'hospital_info.js'});
    rows.next();
  }
  rows.close();
  db.close();
  tableview.setData(dataArray);
});

tableview.addEventListener('click', function(e) {
  if (e.rowData.path){
    var win1 = Ti.UI.createWindow({
      url:e.rowData.path,
      title:e.rowData.title
    });

    var prodCat = e.rowData.title;
    win.prodCat = prodCat;
    Ti.UI.currentTab.open(win);
  }
});

      Hope it helps
