    var win = Titanium.UI.currentWindow;
win.backgroundColor = '#FFF';

var b1 = Titanium.UI.createButton({
    width : 145,
    height : 40,
    left : 10,
    top : 250
});
win.add(b1);
var b1Label = Ti.UI.createLabel({
    text : 'OK as is',
    left : 53,
    width : 100,
    height : 20,
    color : '#336699',
    font : {
        fontSize : 18,
        fontWeight : 'bold'
    },
});
b1.add(b1Label);

var b1ImageView = Ti.UI.createImageView({
    image : '../images/check.png',
    left : 10,
    height : 30,
    width : 33
});
b1.add(b1ImageView);

// Create Correct Window
b1.addEventListener('click', function() {

    var w = Titanium.UI.createWindow({
        height : 0,
        backgroundColor : '#FFF',
        bottom : 0
    });

    // create window open animation
    var a = Titanium.UI.createAnimation();
    a.height = 416;
    a.duration = 275;

    // create a button to move on window
    var b = Titanium.UI.createButton({
        title : 'Next Question',
        height : 30,
        width : 150
    });
    w.add(b);
    w.open(a);

    b.addEventListener('click', function() {
        var nextwin = Ti.UI.createWindow({
            url : 'modifiers2.js',
            title : 'Next Questions',
            backgroundColor : 'pink'
        });
        nextwin.open({
            animated : true
        });
    });
});
