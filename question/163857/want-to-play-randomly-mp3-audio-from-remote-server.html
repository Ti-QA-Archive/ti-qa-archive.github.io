Want to play Randomly mp3 audio from remote server

I want to play mp3 music from remote server. The musics are in the music folder in the server. How to play one by one music from there. I use the following code:

var soundeng = Titanium.Media.createSound({
        url : 'http://example.com/music/01.mp3',
        preload : true
    });
soundeng.play();

The above code works for only one music but I need to play all musics from that(music) folder.
Testing environment Titanium SDK 3.2.0

    You want to play Randomly mp3 audio from remote server.

    try this one :

    var soundeng = Titanium.Media.createSound({
        url : genrateRandomAudioURL(),
        preload : true
    });
soundeng.play();

function genrateRandomAudioURL() {
    var url = "http://example.com/music/";
    url = (url + (Math.floor(Math.random() * 10) + 1) + ".mp3");
    return url;
}

    as above code : you have audio file like : 1.mp3 , 2.mp3 in your server url. and genrateRandomAudioURL function generate random number between 1-10. try this code and let me know if any query.

    i hope this helps. thanks.

    • try this:

      var win = Ti.UI.createWindow({
    backgroundColor : "white"
});

var btnPlay = Ti.UI.createButton({
    height : 100,
    width : 200,
    title : "Play Sound"
});

win.add(btnPlay);

btnPlay.addEventListener("click", function(e) {
    var audioURL = genrateRandomAudioURL();
    Ti.API.info('>>> audio url is :>> ' + audioURL);
    var playerC = Ti.Media.createSound({
        url : audioURL
    });
    playerC.addEventListener("error", function(e) {
        playerC = null;
    });
    playerC.addEventListener("complete", function(e) {
        playerC = null;
    });

    if (playerC.playing) {
        playerC.stop();
    }
    playerC.play();
});

function genrateRandomAudioURL() {
    var url = "http://example.com/music/";
    url = (url + (Math.floor(Math.random() * 10) + 1) + ".mp3");
    return url;
}

win.open();

    Thanks for writing. I have tested your issue. Please test my following code. I use local files for testing. I hope its solve your problem.

    Test Code:

      var win = Ti.UI.createWindow({
            backgroundColor : 'orange', title : 'Red Window',layout:'vertical'

 });
 var mp3_array = [];
 mp3_array.push('music/sound1.mp3');
 mp3_array.push('music/sound2.mp3');
 mp3_array.push('music/sound3.mp3');


 Titanium.Media.audioSessionMode = Titanium.Media.AUDIO_SESSION_MODE_AMBIENT;

 var current_index = 0;

function playSound(index){
    var sound = Titanium.Media.createSound();   

    sound.url = mp3_array[index];

    sound.play();

    Ti.API.info('Now Playing '+index+': '+mp3_array[index]);



    sound.addEventListener('complete',function(){


        // increment the array index
        current_index++;

        if(current_index < mp3_array.length){

            // play the next mp3 in the array 

            playSound(current_index);
        } else {
              Ti.API.info('all sounds complete!');
            }
     });
}
// play the first mp3 in the array @ index 0
playSound(current_index);
win.open();
