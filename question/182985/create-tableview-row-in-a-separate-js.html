  • Try this

    disponibleRow.tss

    "#label": {
    font:{fontSize:12,fontWeight:'light'},
    width:Ti.UI.SIZE
    textAlign : Titanium.UI.TEXT_ALIGNMENT_LEFT,
    color: '#72828c',
    bottom:2,
    left:45,
    height:12
}

"#image":{
    image:"/images/puntos_azules.png",
    width:16,
    height:16,
    right:10
}

    disponibleRow.xml

    <Alloy>
    <TableViewRow height="65" id="disponibleRow">
        <View width="Ti.UI.FILL" height="Ti.UI.FILL">
            <Label id="label" />
            <ImageView id="image" />
    </TableViewRow>
</Alloy>

    disponibleRow.js

    var args = arguments[0] || 0;

$.label.text = args.get('date');

    *No change in listadoCuponesVC.js
