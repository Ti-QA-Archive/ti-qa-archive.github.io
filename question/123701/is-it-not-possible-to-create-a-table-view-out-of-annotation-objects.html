Is it not possible to create a table view out of annotation objects?

I am trying to create a table view out of an array of annotation objects. Is this not possible?

Here is some sample code of what I'm trying to do:

bb.ui.createTableWindow = function () {

    var win = Ti.UI.createWindow({
        title:'Some Title',
        backgroundColor:'white'
    });

// The next line should populate the results array with
// annotation objects chosen based on my search criteria.

    var results = bb.map.showPins('searchCriteria');  

    for (var i=0; i < results.length; i++) {
        // Through the console I can see that the results array
        // does indeed contain the correct objects!
        Ti.API.info('results['+i+']: '+results[i].title);
        // I add the hasDetail property to get the little blue buttons...
        results[i].hasDetail = true;
    };

    var tv = Ti.UI.createTableView({
        top:15,
        scrollable:false,
        style:Ti.UI.iPhone.TableViewStyle.GROUPED,
        //data:results
    });
    tv.addEventListener('click', function(_e){
        // as of yet unimplemented
        // open (create detail window( _e.rowData))
    });

    tv.setData(results);

    win.add(tv);
    return win;
};

When the window opens, there is no table view with rows, however I can see 15px of white at the top, and then there is the vertical grey stripe background of what I can only assume is an empty table view?

Any suggestions?

    Couple of suggestions, try

    width:'auto'

    in your createTableView()

    also what happens if you create the rows array explicitly

        var results = bb.map.showPins('searchCriteria');  
    var data = [];
    for (var i=0; i < results.length; i++) {
        // Through the console I can see that the results array
        // does indeed contain the correct objects!
        Ti.API.info('results['+i+']: '+results[i].title);
        // I add the hasDetail property to get the little blue buttons...
    var row = { title:results[i].title, hasDetail:true };
    data.push(row);
    };

    tv.setData(data);
