  • Here is an example of list-view.I have customized it for you

    var win = Titanium.UI.createWindow();

var tableData = [ {title: 'Spring'}, {title: 'Summer'}, {title: 'Autumn'}, {title: 'Winter'} ];

var table = Ti.UI.createTableView({
  data: tableData
});
win.add(table);
win.open();

  • The mobile equivalent of Radio Buttons is an OptionDialog (link to the documentation)

    ###Here is a short sample on how to use it:

    var win = Ti.UI.createWindow({
  title: 'Click window to test',
  backgroundColor: 'white',
  exitOnClose: true,
  fullscreen: false
});

var opts = {
  title: 'Select Season?',
  options: ['winter', 'summer', 'fall', 'spring'],
  selectedIndex: 2
};

win.addEventListener('click', function(e){
  var dialog = Ti.UI.createOptionDialog(opts).show();
});
win.open();
