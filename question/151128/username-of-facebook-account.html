    Hi chung,

    Try this ,If any user logged in u can get the all the details of the logged user :

    if (Ti.Facebook.loggedIn) {

        params = {
            access_token : Ti.Facebook.accessToken
        };
        Ti.Facebook.requestWithGraphPath('/me', params, 'GET', function(e) {
            var result = JSON.parse(e.result)
            Ti.API.info("Result is : " + e.result);
            Ti.API.info("Id is : " + result.id);
            Ti.API.info("Username is : " + result.username);
            Ti.API.info("Full Name is : " + result.name);
            Ti.API.info("First name is : " + result.first_name);
            Ti.API.info("Last name is : " + result.last_name);
            Ti.API.info("Gender : " + result.gender);

        });
    }

  • Try this 

    var facebookButton = Ti.UI.createButton({
        title:'Facebook',
    height : Math.round(((winWidth * 0.50) - 20) / 3),
    width : Math.round(winWidth * 0.50) - 10,
});
win.add(facebookButton);
facebookButton.addEventListener('click', function() {
    if (Titanium.Facebook.loggedIn) {
        Titanium.Facebook.logout();
    } else {
        Ti.Facebook.authorize();
    }
});

function checkStatus(e) {
    if (e.success) {
        Titanium.Facebook.requestWithGraphPath('me', {}, 'GET', function(e) {
            if (e.success) {
                var data = JSON.parse(e.result);
                Ti.API.info("Name:" + data.name);
                Ti.API.info("email:" + data.email);
                Ti.API.info("facebook Id:" + data.id);
                alert(data.email+"  " +data.name+" "+ 'FBL');
            } else if (e.error) {
                alert(e.error);
            } else {
                Ti.UI.createAlertDialog({
                    title : 'Facebook',
                    message : 'There is a Facebook responce error'
                }).show();
            }
        });
    } else {
        if (e.error) {
            Ti.UI.createAlertDialog({
                title : 'Facebook Error',
                message : e.error
            }).show();

        } else {
            Ti.UI.createAlertDialog({
                title : 'Facebook Error',
                message : 'Unkown error while trying to login to facebook'
            }).show();
        }
    }
}

function checkLogout() {
    Ti.UI.createAlertDialog({
        title : 'Facebook Logout',
        message : 'You are loged out from facebook.'
    }).show();
}

Titanium.Facebook.addEventListener('login', checkStatus);
Titanium.Facebook.addEventListener('logout', checkLogout);
