    • Hi Karl,
      I found some problem while parsing your datum string to date object.

      I change custom sort function that works for your datum string.

      //custom sort function
function comp(a, b) {
     //Sort by recent date to old date
    var aDatum=a.datum.split(" ")[0].split("-");
    var bDatum=b.datum.split(" ")[0].split("-");
   return (new Date(bDatum[0],bDatum[1],bDatum[2]).getTime() - new Date(aDatum[0],aDatum[1],aDatum[2]).getTime());
}

}
