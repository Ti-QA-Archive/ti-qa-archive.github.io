    Hi khodour,

    You can test this code. I think it will be helpful for you and your problem will be solved. But I have tested only android devices, please check it with your iOS device.

    var win = Titanium.UI.createWindow({
    title : 'Photo Upload',
    backgroundColor : '#fff',
    layout : 'vertical'
});

// Create a Button.
var takePhoto = Ti.UI.createButton({
    title : 'Take Photo',
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,
    top : 20,
    color : 'pink'
});

win.add(takePhoto);

// Create an ImageView.
var imaveView = Ti.UI.createImageView({
    image : 'location',
    width : 100,
    height : 100,
    top : 40,
});

// Add to the parent view.
win.add(imaveView);

//var path = [];

var imageDir = Ti.Filesystem.getFile(Ti.Filesystem.externalStorageDirectory, 'images');
if (! imageDir.exists()) {
    imageDir.createDirectory();
}
// Listen for click events.
takePhoto.addEventListener('click', function() {

    Titanium.Media.showCamera({
        success : function(event) {
            imaveView.setImage(event.media);
            var filename = new Date().getTime() + "-ea.jpg";
            if (Ti.Filesystem.isExternalStoragePresent()) {
                var imageFile = Ti.Filesystem.getFile(imageDir.resolve(), filename);
                Ti.API.info("ImageFile path is: " + imageFile.resolve());
                imageFile.write(event.media);
                if (imageFile.write(event.media) === false) {
                    alert('error');
                } else {
                    alert('see your sd Card project folder');
                }

            } else {
                alert("Your Memory Card is not Present");
            }

        },

        cancel : function() {
            // called when user cancels taking a picture
        },
        error : function(error) {
            // called when there's an error
            var a = Titanium.UI.createAlertDialog({
                title : 'Camera'
            });
            if (error.code == Titanium.Media.NO_CAMERA) {
                a.setMessage('Please run this test on device');
            } else {
                a.setMessage('Unexpected error: ' + error.code);
            }
            a.show();
        },

        mediaTypes : [Ti.Media.MEDIA_TYPE_VIDEO, Ti.Media.MEDIA_TYPE_PHOTO]
    });
});

win.open();

    If you have any question about this query, then let me know.

    Thanks.
