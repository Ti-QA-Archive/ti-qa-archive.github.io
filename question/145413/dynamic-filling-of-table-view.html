dynamic filling of table view

Hi,
in a tableview I like to fill the sections and the rows in sections dynamicly.
In result I only see the section titles – the content leaves empty:

var tv = Ti.UI.createTableView();
var channelObj = require('/modules/channels');
var channels = new channelObj('veranstaltungen', function(data) {
    function insertRows(section, fakultaetId) {
        channels.getBereiche(fakultaetId, function(data) {
            for (var bereichId in data) {
                var row = Ti.UI.createTableViewRow({title : data[bereichId],});
                section.add(row);
            }
        });
    }
    var sections = [];
    for (var fakultaetId in data) {
        var section = Titanium.UI.createTableViewSection({
            headerTitle : data[fakultaetId],
            fakultaetId : fakultaetId
        });
        insertRows(section, fakultaetId);
        sections.push(section);
    }
    tv.data = sections;
});
