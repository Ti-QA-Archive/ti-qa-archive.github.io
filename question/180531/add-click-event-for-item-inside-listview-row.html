  • You have to add events differently for custom item templates.

    /*
      *  Raw Template
      */
      var rawTemplate = { childTemplates : [rowTempalteLoad()] };

      function rowTempalteLoad()
      {
          var views = [];
          var fk = {
              type: 'Ti.UI.Label',
              bindId: 'info',
              properties: {
                  color: '#000000',
                  text: "a a a a",
                  height: Ti.UI.SIZE,
                  bottom: 5,
                  left: "3%",
                  font: {
                      fontSize: 22
                  },
              }
              events: { click: /* your function here */ }
          };

          //contact title 
          views.push(fk);
          return views;
      }

  • Hello khodour fouani,

    For that you have to use custom listView template. Here is my sample test code just follow this and let me know. 

    var myTemplate = {

    childTemplates : [{// Title
        type : 'Ti.UI.Label', // Use a label for the title
        bindId : 'info', // Maps to a custom info property of the item data
        properties : {// Sets the label properties
            color : 'black',
            font : {
                fontFamily : 'Arial',
                fontSize : '14',

            },
            left : 10
        },
        events : {
            click : titleClick
        }
    }, {// Subtitle
        type : 'Ti.UI.Label', // Use a label for the subtitle
        bindId : 'es_info', // Maps to a custom es_info property of the item data
        properties : {// Sets the label properties
            color : 'black',
            font : {
                fontFamily : 'Arial',
                fontSize : '14dp'
            },
            left : '100',
        },
        events : {
            click : NumberClick
        }
    }]
};

var itemData = [];

for (var i = 0; i < 20; i++) {
    itemData.push({

        info : {
            text : 'Hi Guys'
        },
        es_info : {
            text : '123'
        }
    });
}

var section = Ti.UI.createListSection({});
section.setItems(itemData);

var list = Ti.UI.createListView({
    sections : [section],
    templates : {
        'template' : myTemplate
    },
    defaultItemTemplate : 'template',
    top : 20,

});

function titleClick(e) {

    var item = e.section.getItemAt(e.itemIndex);
    alert(item.info.text);

}

function NumberClick(e) {
    var item = e.section.getItemAt(e.itemIndex);
    alert(item.es_info.text);

}

var win = Ti.UI.createWindow({
    title : 'ListView',
    backgroundColor : '#fff'
});
win.add(list);
win.open();

    If you have any concern, then let me know.

    Thanks.
