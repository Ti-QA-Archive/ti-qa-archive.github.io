  • exports.create =function() {
    var self = Ti.UI.createView({height:100,bottom:-100});
    self.slideup = function() {
        self.animate({bottom:0});    
    }
}

// usage:
var myview = require('mymodule');

mybutton.addEventlistener('click',function() {
    myview.slideup();    
});
