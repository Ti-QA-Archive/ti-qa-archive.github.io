Cannot Open views with Alloy

I am having an impossible time opening a Alloy view from within a controller. I have read the documentation several times and downloaded the git examples. The examples will work fine but I am unable to get Alloy to work on code I write.

Here is the code: index.js

var Alloy = require('alloy');

function doClick(e) {  
    var w = Alloy.createController('a').getView();
    w.open();   
}

$.index.open();

index.xml

<Alloy>
    <Window class="container">
        <Label id="label" onClick="doClick">Hello, World</Label>
    </Window>
</Alloy>

a.js

//Blank
~~~http://www.appcelerator.com/company

a.xml

<Alloy>
<View class="container">
<Button id="mainButton">
</View>
</Alloy>


a.tss

".container": {
backgroundColor: "gray"
}

"mainButton":{
title:"Hello A",
height:"50px",
width:"100px"
}

~~~

The Error I am getting is below:

Script Error = 'undefined' is not a function (evaluating 'w.open()') at index.js (line 4).

I tried the ezti lib from Ricardo's git. The Git examples will work but I cannot get the library to work in a new project generated by my machine. Same error as above.

    height: Ti.UI.SIZE,
    color: "#000"
}
