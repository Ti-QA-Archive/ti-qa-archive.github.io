    Hello Narasimha,

    Try this :

    var priceversionurl = "http://173.230.148.65/openbravo/org.openbravo.service.json.jsonrest/PricingPriceListVersion?_where=priceList=%27" + pricelistglobal + "%27";
var priceversionvhr = Titanium.Network.createHTTPClient({

    onload : function() {
        var result = JSON.parse(this.responseText);
        var priceversionidname = result.response.data[0]; 

        var priceversiontype = priceversionidname.name;
        var priceversionid = priceversionidname.organization;
        priceversionglobal = priceversionidname.id;

    }, //--price version onload --//
    onerror : function(e) {
        Ti.API.debug(e.error);
        alert('error' + e.error);
    }
});
//--   pricevhr http client --//
priceversionvhr.open('GET', priceversionurl);
priceversionvhr.send();

    i hope this help to solved your problem. thanks.
