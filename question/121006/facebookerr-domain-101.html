  • Kosso - I tried what you posted above and that worked. It looks like the facebook module now expects you to be logged in before trying to send a feed request where as before it would prompt you to login if not already done so. Here is my fix:

    var win = Ti.UI.currentWindow;

Titanium.Facebook.appid = "xxxxxxxxxxxx";
Titanium.Facebook.permissions = ['publish_stream', 'read_stream', 'offline_access'];

Ti.API.info(Titanium.Facebook.loggedIn);
if (Titanium.Facebook.loggedIn) {
    Titanium.Facebook.dialog("feed", win.data, function(e) {
        if (e.success) {
            alert("Done!");
        } else {
            if (e.error) {
                alert(e.error);
            } else if (e.cancelled) {
                alert('Cancelled');
            } else {
                alert("Unkown result");
            }
        }
        win.close();
    });
} else {
    Titanium.Facebook.authorize();
}

Titanium.Facebook.addEventListener('login', function() {
    if (Titanium.Facebook.loggedIn) {
        Titanium.Facebook.dialog("feed", win.data, function(e) {
            if (e.success) {
                alert("Done!");
            } else {
                if (e.error) {
                    alert(e.error);
                } else if (e.cancelled) {
                    alert('Cancelled');
                } else {
                    alert("Unkown result");
                }
            }
            win.close();
        });
    } else {
        alert("You must be logged in to Post!");
        win.close();
    };
});
