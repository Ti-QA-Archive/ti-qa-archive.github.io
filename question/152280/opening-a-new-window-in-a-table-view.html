  • Accepted Answer

    Use something like below.

    tableView.addEventListener('click', function(e) {
    var win=Ti.UI.createWindow({
        title: e.row.title
    });
    var label = Ti.UI.createLabel({
        text: e.row.rowContent; //Custom property you could set to row.
    });
    win.add(label);
    win.open();
});

function OpenNewWindow(tileText) {
    var swin = Titanium.UI.createWindow({
        url :'/Discriptiontext.js',
        desc : tileText,
        fullscreen : true,
        navBarHidden : true,
        backgroundColor : 'gray',
    });
    swin.orientationModes = [Titanium.UI.PORTRAIT, Titanium.UI.UPSIDE_PORTRAIT, Titanium.UI.LANDSCAPE_LEFT, Titanium.UI.LANDSCAPE_RIGHT];

    if (isAndroid) {
        swin.open({
            navBarHidden : true,
            fullscreen : true,
            animated : true,
            modal : true
        });
    } else {
        swin.hideTabBar();
        Titanium.UI.currentTab.open(swin, {
            animated : true
        });
    }
}
