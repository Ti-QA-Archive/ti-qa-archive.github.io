    Hello Prashaanth,

    Try the below code

    index.xml

    <Alloy>
    <Window id="win" backgroundColor="white" layout="vertical" exitOnClose="true">
        <View id="viewMenu" borderWidth = 1 borderColor = "blue" height = 35 width = 100 >
            <Label id="selectedItem" text = "" width = 80 height = 35 top = 0></Label>
            <ImageView id="img"     image = "http://png-5.findicons.com/files//icons/2198/dark_glass/128/1downarrow.png" height = 20 width=20 right = 0></ImageView>
        </View>
        <ScrollView id="viewContainer" height = 150 width = 100 contentHeight = "auto" contentWidth = "Ti.UI.SIZE" visible = "false" borderColor = "gray" layout = "vertical"></ScrollView>
    </Window>
</Alloy>

    index.js

    $.viewMenu.addEventListener('click',function(e){
    $.viewContainer.show();
});

var arr = ["One","Two","Three","One","Two","Three","One","Two","Three","One","Two","Three","One","Two","Three"];

for (i=0;i<arr.length;i++){
    var item = Ti.UI.createLabel({
        text : arr[i],
        height : 20,
        width : 100
    });
    item.addEventListener('click',function(e){
        $.viewMenu.children[1].text=this.text;
        $.viewContainer.hide();
    });
    $.viewContainer.add(item);
}

$.win.open();

    Hope this helps. Thanks.
