  • "Furthermore is it possible any kind of communication/control between my own App and the device's GPS navigator?"

    Other apps have custom url schemes and intents that allow you to pass information to them for processing. Many of the navigation apps have this functionality.

    The problem lies with the fact that you dont know what app is installed on the user's system.

    You don't know if there is a default app for navigation.

exports.NavigateTo = function(latitude, longitude){
    var url = 'waze://?ll='+latitude+','+longitude+'&navigate=yes';
    if (Ti.Android){
        try {
            Ti.API.info('Trying to Launch via Intent');
            var intent = Ti.Android.createIntent({
                action: Ti.Android.ACTION_VIEW,
                data: url

            });
            Ti.Android.currentActivity.startActivity(intent);
        } catch (e){
            Ti.API.info('Caught Error launching intent: '+e);
            exports.Install();
        }
    }else{
        if (Ti.Platform.canOpenURL(url)){
            Ti.Platform.openURL(url);
        } else {
            exports.Install();
        }

    }

};

exports.Search = function(query){
    var url = 'waze://?q='+query;
    if (Ti.Android){
        try {
            Ti.API.info('Trying to Launch via Intent');
            var intent = Ti.Android.createIntent({
                action: Ti.Android.ACTION_VIEW,
                data: url

            });
            Ti.Android.currentActivity.startActivity(intent);
        } catch (e){
            Ti.API.info('Caught Error launching intent');
            exports.install();
        }
    }else{
        if (Ti.Platform.canOpenURL(url)){
            Ti.Platform.openURL(url);
        } else {
            exports.install();
        }

    }
};

exports.Install = function(){
    var alert = Ti.UI.createAlertDialog({
        title: "Waze Not Installed",
        message: "This action requires you install the Waze application",
        buttonNames: [ "OK", "Cancel" ],
        cancel: 1
    });

    alert.addEventListener("click", function(_event) {
        if(_event.index === 0) {
            var installURL;
            if (Ti.Android){
                installURL = 'market://details?id=com.waze';
            } else {
                installurl = 'http://itunes.apple.com/us/app/id323229106';
            }
            Ti.API.info("Installing Waze application");
            Ti.Platform.openURL(installURL);
        } else {
            Ti.API.info("User aborted Waze installation");
        }
    });

    alert.show();




};

exports.NavigateTo = function(latitude, longitude){
    var url; 
    if (Ti.Android){
        try {
            Ti.API.info('Trying to Launch via Intent');
            var intent = Ti.Android.createIntent({
                action: Ti.Android.ACTION_RUN,
                //data: 'vznavigator:app2app?version=1.0&credential=123&commands=M&lat='+latitude+'&lon='+longitude
                packageName: 'com.nim.vznavigator.app2app'
            });
            intent.putExtra('context','vznavigator:app2app?version=1.0&commands=M&lat='+latitude+'lon='+longitude);
            Ti.Android.currentActivity.startActivity(intent);
        } catch (e){
            Ti.API.info('Caught Error launching intent: '+e);
            exports.Install();
        }
    }else{
        url = 'vznavigator:app2app?version=1.0&commands=M&lat='+latitude+'&lon='+longitude;
        if (Ti.Platform.canOpenURL(url)){
            Ti.Platform.openURL(url);
        } else {
            exports.Install();
        }

    }
};

    • You can avoid intents and knowing what the user has installed on Android by using openURL().

      if(OS_IOS) {
  // for iOS, you can use either maps.apple.com or maps.google.com to launch the native Apple maps app, the user can then choose to navigate to the address
  Ti.Platform.openURL('maps://maps.google.com/maps?daddr=123+Main+St+City+State+Zip');
  // to specifically launch Google's Maps app (if it's installed), use
  // comgooglemaps://
  //Ti.Platform.openURL('comgooglemaps://maps.google.com/maps?daddr=123+Main+St+City+State+Zip');
} else if(OS_ANDROID) {
  // the following will prompt the user what to open: browser, native maps app,
  // google earth, waze, etc.
  Ti.Platform.openURL('maps://maps.google.com/maps?daddr=123+Main+St+City+State+Zip&directionsmode=driving');
  // can also do by lat/long
  //Ti.Platform.openURL('http://maps.google.com/maps?daddr=99.9999,-99.9999&directionsmode=driving');
}
