    On success(event) callback you'll get image via event.media

    Below is sample code for Gallary. Camera should be the same :)

    var imageViewMe = Titanium.UI.createImageView();

Titanium.Media.openPhotoGallery({
    success : function(event) {
        imageViewMe.image = event.media;
    },
    cancel : function() {

    },
    error : function(error) {
    },
    mediaTypes : [Ti.Media.MEDIA_TYPE_PHOTO]
});
