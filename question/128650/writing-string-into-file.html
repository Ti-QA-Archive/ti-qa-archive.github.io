Writing string into file

I am trying to write a long string to file. But I have got such error
Error:

[ERROR] The iOS reported an error. Could not open file stream for file at path: /Users/arkady/Library/Application Support/iPhone Simulator/4.2/Applications/790D6F78-C220-450F-A016-152FF8D4294B/Documents

Code:

function writeIntoFile(stream) {

        var fname = buildFileName();

        //Get File
        var file = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, fname);

        //Create Buffer
        var buffer = Titanium.createBuffer({
            length : 1024,
            value : stream
        });
        //Create Stream
        var outStream = file.open(Titanium.Filesystem.MODE_WRITE);
        var inStream = Titanium.Stream.createStream({
            mode : Titanium.Stream.MODE_READ, // There is also MODE_APPEND for writing
            source : buffer
        });

        //File Writing
        var size = 0;
        while(( size = inStream.read(buffer)) > -1) {
            outStream.write(buffer);
            Titanium.API.info('Wrote ' + size + ' bytes');
        }
    }

Problematic line:

var outStream = file.open(Titanium.Filesystem.MODE_WRITE);

Any ideas? Thanks
