  • Hi guys,

    Try this sample code and let me know if you have any question about this issue.

    var win = Ti.UI.createWindow({
    title : "Control Window",
    backgroundColor : "white",
    layout : "vertical"
});

win.activity.onCreateOptionsMenu = function(e) {
    var menu = e.menu;
    var menuItem = menu.add({
        title : "Compose",
        icon : "appicon.png",
        showAsAction : Ti.Android.SHOW_AS_ACTION_NEVER
    });
    menuItem.addEventListener("click", function(e) {
        alert("Menu 0 clicked");
    });
    var menuItem1 = menu.add({
        title : "menu1",
        icon : "appicon.png",
        showAsAction : Ti.Android.SHOW_AS_ACTION_NEVER
    });
    menuItem1.addEventListener("click", function(e) {
        alert("Menu 1 clicked");
    });
    var menuItem2 = menu.add({
        title : "menu2",
        icon : "appicon.png",
        showAsAction : Ti.Android.SHOW_AS_ACTION_NEVER
    });
    menuItem2.addEventListener("click", function(e) {
        alert("Menu 2 clicked");
    });
};
win.open();

    Thanks.
