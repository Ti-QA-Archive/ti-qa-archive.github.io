fetch data from database to a table

hi , how can i fetch data from database to a table. I tried it but still not working :

var a = db.execute('SELECT page FROM book');

var table1 =  Titanium.UI.createTableView({});

    while (a.isValidRow())
    {
        table1.data=[{ title:a.fieldByName('page') }];
         a.next();    
    }    

a.close();

it takes only the last row in the table , i want to build a table with the content of the database

var data = [];
    while (a.isValidRow())
    {
        data.push({ title:a.fieldByName('page') });
        a.next();   
    }   

table1.setData(data);
