    Hi Shekhar Saraf,

    Auto Complete property is not present in titanium like native android . But you can do this by using tableView search like this sample code.

    Test Environment:
    Ti SDK: 3.2.3
    Android,IOS

    var win = Titanium.UI.createWindow({
    title : "Table Search",
});

// create table view data object
var data = [];

for (var i = 0; i < 10; i++) {
    var row = Ti.UI.createTableViewRow({
        hasChild : true,
        title : 'Row ' + i
    });

    data.push(row);
};

var search = Titanium.UI.createSearchBar({
    barColor : '#385292',
    showCancel : false,
    hintText : 'search'
});

search.addEventListener('return', function(e) {
    search.blur();
});
search.addEventListener('cancel', function(e) {
    search.blur();
});

// create table view
var tableview = Titanium.UI.createTableView({
    data : data,
    search : search,
    filterAttribute : "title"
});

win.add(tableview);

win.open();
