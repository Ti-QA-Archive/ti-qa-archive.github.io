    Hello Sergio,

    Here is the full test case of handling JSON file to alloy project. This is almost similar to the classic project, just run this following test code.

    index.xml

    <Alloy>
    <Window class="container">
        <TableView id="table" top="30"></TableView>
    </Window>
</Alloy>

    index.js

    var data = [];

var url = "http://ficouincrivel.com.br/estudio/data/JSON/galleriesbg/books.json";
var client = Ti.Network.createHTTPClient({
    // function called when the response data is available
    onload : function(e) {
        Ti.API.info("Received text: " + this.responseText);

        var json = JSON.parse(this.responseText);

        for (var i = 0; i < json.length; i++) {
            //alert(json[i].title);
            var row = Ti.UI.createTableViewRow({
                height : 100
            });

            // Create a Label.
            var title = Ti.UI.createLabel({
                text : json[i].title,
                color : '#000',
                font : {
                    fontSize : 20
                },
                height : Ti.UI.SIZE,
                width : Ti.UI.SIZE,
                top : 10

            });

            // Add to the parent view.
            row.add(title);

            var author = Ti.UI.createLabel({
                text : json[i].author,
                color : '#000',
                font : {
                    fontSize : 15
                },
                height : Ti.UI.SIZE,
                width : Ti.UI.SIZE,
                top : 50

            });

            // Add to the parent view.
            row.add(author);

            data.push(row);

            $.table.setData(data);
        };

    },
    // function called when an error occurs, including a timeout
    onerror : function(e) {
        Ti.API.debug(e.error);
        alert('error');
    },
    timeout : 5000 // in milliseconds
});
// Prepare the connection.
client.open("GET", url);
// Send the request.
client.send();

$.index.open();

    If you have any question then feel free to ask me.

    For HTTPClient.
