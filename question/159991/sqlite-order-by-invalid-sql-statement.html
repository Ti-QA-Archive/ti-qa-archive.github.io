    First thing to note is the variable query is already a string so there is no need to use toString in this case.

    Without knowing the columns in this table I need to to verify the following things to be true based on your query;

    • Your table is called messages
    • This table has column (field) user_name
    • This table has column (field) data

    I suspect you do not have a column called data if the error is suggesting the problem is with that part of the query.

    Titanium.UI.setBackgroundColor('#000');
var win=Ti.UI.createWindow({
    title:'Moble Web Test',
    backgroundColor:'#fff'
});
var datas=[];
var db=Ti.Database.open('messages');

    db.execute('drop table if exists messages');

    db.execute('create table if not exists messages( type text, user_name text, user_id text, message text, data date)');
    db.execute('insert into messages values(?,?,?,?,?)','t1','Simone Puglia1','id1','m1','2013-12-01');
    db.execute('insert into messages values(?,?,?,?,?)','t1','Simone Puglia2','id2','m1','2013-12-02');
    db.execute('insert into messages values(?,?,?,?,?)','t1','Simone Puglia3','id3','m1','2013-12-03');
    db.execute('insert into messages values(?,?,?,?,?)','t1','Simone Puglia4','id4','m1','2013-12-04');


    var row=db.execute("SELECT * FROM messages WHERE user_name='Simone Puglia2'ORDER BY data DESC");

    while(row.isValidRow())
    {
        datas.push(
            {
                title: row.fieldByName('user_name')
            }
        );
        row.next();
    }

    db.close();    


var aTableView = Ti.UI.createTableView();

aTableView.setData(datas);

win.add(aTableView);


win.open();
