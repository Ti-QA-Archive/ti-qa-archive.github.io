    Regarding ellipsize, it is easy to test it with just 20 lines of code:

    var win1 = Titanium.UI.createWindow({
    backgroundColor:'#ccc',
    exitOnClose:true,
    title:'win1: Main Window'
});
var label1 = Ti.UI.createLabel({
    backgroundColor:'black',
    color:'white',
    ellipsize:false,
    text:'This is some text to demonstrate the use of the ellipsize property of a label(click me!)',
    height:66,
    width:200
});
win1.add(label1);
win1.open();
var ellipsizeEnabled = false; // initial label1.ellipsize value set above
label1.addEventListener('click', function() {
    label1.ellipsize =  !ellipsizeEnabled;
    ellipsizeEnabled = !ellipsizeEnabled;
});
