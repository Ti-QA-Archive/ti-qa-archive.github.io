How to create a country list in android using scroll view

I'm using Titanium Studio, build: 3.2.1.201402041146, and my operating system is Windows 8. I have an android alloy mobile application project and in that for my registration form I have to get the users' country list for that I'm using a window to popup when clicked on to the country text field. As table views are not supported in android I'm using a scroll view to add the countries here is my code

    var scrollView = Ti.UI.createScrollView({
        top : '90sp',
        contentHeight : 'auto',
        layout : 'vertical'
    });

    var country = content.arrCountry;

    for (var i = 0; i <= country.length; i++) {

        var row = Ti.UI.createView({
            backgroundColor : 'white',
            borderColor : '#bbb',
            borderWidth : 1,
            width : '100%',
            height : 70,
            top : 0,
            left : 0,

        });
        row.index = i;
        var label1 = Ti.UI.createLabel({
            color : '#900',
            font : {
                fontSize : 12
            },
            shadowColor : '#aaa',
            shadowOffset : {
                x : 5,
                y : 5
            },
            shadowRadius : 3,
            text : country[i],
            textAlign : Ti.UI.TEXT_ALIGNMENT_CENTER,
            top : 30,
            width : Ti.UI.SIZE,
            height : Ti.UI.SIZE
        });

        label1.addEventListener('singletap', function(e) {

        });

        row.add(label1);
        scrollView.add(row);
    }
    popup.add(scrollView);

};

There is an array containing 245 countries in content.arrCountry it will be assigned to country:
var country = content.arrCountry;

The code is working but I want to get the country or the index of the country on that addEventListener

label1.addEventListener('singletap', function(e) {
            //get the value of the clicked label
        });

Can anybody suggest a way of doing this. Thanks in advance!

  • I got the code working to my requirements and now its working

    var country = content.arrCountry;
    var row=[245];
    var label=[245];

    for (var i = 0; i <= country.length; i++) {

            row[i] = Ti.UI.createView({
            backgroundColor : 'white',
            borderColor : '#bbb',
            borderWidth : 1,
            width : '100%',
            height : 70,
            top : 0,
            left : 0,

        });

            label[i] = Ti.UI.createLabel({
            color : '#900',
            font : {
                fontSize : 12
            },
            shadowColor : '#aaa',
            shadowOffset : {
                x : 5,
                y : 5
            },
            shadowRadius : 3,
            text : country[i],
            textAlign : Ti.UI.TEXT_ALIGNMENT_CENTER,
            top : 30,
            width : Ti.UI.SIZE,
            height : Ti.UI.SIZE
        });

        label[i].addEventListener('singletap', function(e) {
            alert(this.text);
        });

        row[i].add(label[i]);
        scrollView.add(row[i]);
    }
    popup.add(scrollView);

};
