set tab to not active when other tab is active

I am creating a custom tabBar for Android that looks like the iOS tabBar.

Now I have the tabBar at the bottom and each tab is has a dynamic width so that you can have virtually as many as you want as well I have the active states for each button working. The problem Im having is when switching tabs the currently active tab's active state doesn't change and Im not sure how I should detect and change the state of the currently active tab.

Hope I explained well, code is below

tabBar.js

function tabBar(w, c) {

    var main = w;
    var content = c;

    //---------Custom Tab Bar---------//

    var tabBar = Titanium.UI.createView({
        bottom: 0,
        height: 75,
        width: Ti.Platform.displayCaps.getPlatformWidth,
        backgroundImage: 'images/tabBarGrad.png'
    });

    //-------End Custom Tab Bar-------//

    if (content != null) {
        content.height =  content.height - tabBar.height;
    }

    var aboutTab = createTab('About', '../images/tabIcons/info_48.png', 1, 2, tabBar);
    var helpTab = createTab('Help', '../images/tabIcons/help_48.png', 2, 2, tabBar);

    tabBar.add(aboutTab);
    tabBar.add(helpTab);

    main.add(tabBar);

}

function createTab(text,img, num, total, tb) {

    var tabBar = tb;
    var tabWidth = Ti.Platform.displayCaps.getPlatformWidth() / total;
    var tabHeight = tabBar.height;

    if (num != 1) {
        tabLeft = tabWidth * (num - 1); 
    } else {
        tabLeft = 0;
    }

    var tab = Ti.UI.createView({
        width: tabWidth,
        height: tabHeight,
        left: tabLeft
    });

    var text = Ti.UI.createLabel({
        text: text,
        bottom: 5
    });

    var image = Ti.UI.createImageView({
        image: img,
        width: 34,
        height: 34,
        top: 10
    });

    tab.add(image);
    tab.add(text);

    tab.addEventListener('click', function(e){
        this.backgroundImage = '../images/tabBarActive.png';
    });


    return tab;

}

This file is included in the app.js file and functions will be called to create the tabBar almost like the default tabBar functions.
