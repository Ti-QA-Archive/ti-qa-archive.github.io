Making an eventlistener of Picker

Hi
I am making an eventlistener of picker Here is my code:

var YahooScreen = require("YahooScreen");
var ESPNScreen = require("ESPNScreen");

});
var ApiLabel = Titanium.UI.createLabel({
    text : 'I am Fantasy Expert at',
    top : '28%',
    left : '10%',
    right : '10%',
    height : 'auto',
    width : 'auto',
    color : '#000',
    font : {
        fontSize : 28,
        weight : 'bold'
    },
    textAlign : 'center'
});
ApiWin.add(ApiLabel);
var ApiPicker = Ti.UI.createPicker({
    Top : '35%',

    width : '50%'

ApiPicker.addEventListener('change', function(e) {
    if (e.row == 'Yahoo!') {
     alert('I am in Yahoo')

    alert('I am in CBS')
    }

var data = [];
data.push(Titanium.UI.createPickerRow({

ApiWin.add(ApiPicker);
