I want to display photo and message from the status resource in ACS using a vertical layout in the row.

So to summarize: I am adding an imageview that I have resized using height and added it to a row (actually a view in a row that uses layout:vertical).

  • exports.TableMessageRow = function (args) {

    var instance = Ti.UI.createTableViewRow ({

        width: 320,
        height: Ti.UI.SIZE,
        backgroundColor: 'transparent',
        layout: 'vertical'

    });

    var message = Ti.UI.createView ({

        width: 300,
        backgroundColor: '#fff',
        height: Ti.UI.SIZE,
        layout: 'vertical'

    });

    var spacing = Ti.UI.createView ({

        height: 10

    });

    if (args.photo_data) {

        if (args.photo_data.processed) {

            var photo = Ti.UI.createImageView ({

                image: args.photo_data.urls.small_640,
                width: 320,
                height: 320,
                top: 10,
                left: 10

            });

        }
    }

    var text = Ti.UI.createLabel ({

        text: args.message,
        width: 280,
        top: 10,
        left: 10

    });

    var timestamp = Ti.UI.createLabel ({

        text: '2 min',
        textAlign: 'right',
        color: '#999',
        width: 280,
        bottom: 10,
        left: 10

    });

    instance.add (message);

    instance.add (spacing);

    if (args.photo_data) {

        if (args.photo_data.processed) {

            message.add (photo);

        }

    }

    message.add (text);

    message.add (timestamp);

    return instance;

};
