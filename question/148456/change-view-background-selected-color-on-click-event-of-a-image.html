  • Hi jai,

    backgroundSelectedColor only works on Android and mobile Web.

    Please tell us more about your environment if you ask a new question

    You can achieve BackgroundColor switching for your view with the attached code.

    var win = Ti.UI.createWindow();

var likeView = Titanium.UI.createView({
    borderWidth : '1',
    borderColor : '#dcdcdc',
    backgroundColor : '#F0F0F0',
    originalBackgroundColor : '#F0F0F0',
    borderRadius : '1',
    width : '30%'
});
likeView.addEventListener('toggleBackgroundColor', function(e) {
    Ti.API.info('toggleBackgroundColor fired');

    // set new BackgroundColor
    likeView.backgroundColor = '#ff9900';

    // wait 500 milliseconds and switch back to the original BackgroundColor
    setTimeout(function() {
        likeView.backgroundColor = likeView.originalBackgroundColor;
    }, 500);
});

var likeImage = Titanium.UI.createImageView({
    backgroundImage : '/images/likes.png',
    backgroundSelectedImage : '/images/likes1.png',
    left : '10%',
    bottom : 7,
    height : '15',
    width : '15'
});

likeImage.addEventListener('click', function() {
    Ti.API.info('image click');
    likeView.fireEvent('toggleBackgroundColor');
});
likeView.add(likeImage);

win.add(likeView);
win.open();

    Hope that helps,

    Matthias

  • Try this…. this should work….

    
var button = Ti.UI.createView({
        width : 100,
        height : 100,
        top : 10,
        backgroundColor : '#ff000000'
    });

    button.addEventListener('click', function(){
        Ti.API.info('Clicked!');
    });

    button.addEventListener('touchstart', function() {
        button.backgroundColor = '#ffffffff';
    });

    button.addEventListener('touchend', function() {
        button.backgroundColor = '#ff000000';
    });
