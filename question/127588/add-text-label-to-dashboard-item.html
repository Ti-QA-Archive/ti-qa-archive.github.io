Add Text Label to dashboard Item?

Question Short:

Is there a way so I can add a text label to a dashboard item? I am using the dashboard kitchensink example.

Thank you very much!

  • Accepted Answer

    You can, using custom dashboard item, just like you would do for custom tableviewrows:

    var win = Titanium.UI.createWindow();

var data = [];
for (var i=0;i<9;i++)
{
    var item = Titanium.UI.createDashboardItem();
    var container = Titanium.UI.createView({
        layout:"vertical"
    });
    var image = Titanium.UI.createImageView({
        image:'KS_nav_ui.png',
        width:40,
        height:40
    });
    container.add(image);
    var label = Titanium.UI.createLabel({
        text:"Label" + i,
        width:80,
        height:20,
        color:"#ffffff"
    });
    container.add(label);
    item.add(container);
    data.push(item);
}

var dashboard = Titanium.UI.createDashboardView({
    data:data
});

win.add(dashboard)

win.open();
