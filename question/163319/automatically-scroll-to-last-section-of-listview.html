Automatically scroll to last section of listView

I was wondering if it is possible to automatically scroll to the last section of a list when adding the new section to the list. I have been trying with the scrollToItem method of a listview, but this does not work on Android (as is mentioned in the documentation).

    Hello Jasper,

    I tested this issue with the following code in Android. It is working well as expected. Can you please check my sample code in your environment and update your test result. If you have this problem, then please attach your sample code with snap or any error clip for testing.

    var myTemplate = {
    childTemplates : [{
        type : 'Ti.UI.Label',
        bindId : 'one',
        properties : {
            color : 'black',
            font : {
                fontFamily : 'Arial',
                fontSize : '14dp'
            },
            left : '10dp'
        }
    }, {
        type : 'Ti.UI.Label',
        bindId : 'two',
        properties : {
            color : 'black',
            font : {
                fontFamily : 'Arial',
                fontSize : '14dp'
            },
            left : '50dp'
        }
    }, {
        type : 'Ti.UI.Label',
        bindId : 'three',
        properties : {
            color : 'gray',
            font : {
                fontFamily : 'Arial',
                fontSize : '14dp'
            },
            left : '90dp'
        }
    }, {
        type : 'Ti.UI.Label',
        bindId : 'four',
        properties : {
            color : 'gray',
            font : {
                fontFamily : 'Arial',
                fontSize : '14dp'
            },
            left : '130dp'
        }
    }, {
        type : 'Ti.UI.Label',
        bindId : 'five',
        properties : {
            color : 'gray',
            font : {
                fontFamily : 'Arial',
                fontSize : '14dp'
            },
            left : '170dp'
        }
    }, {
        type : 'Ti.UI.Label',
        bindId : 'six',
        properties : {
            color : 'gray',
            font : {
                fontFamily : 'Arial',
                fontSize : '14dp'
            },
            left : '210dp'
        }
    }, {
        type : 'Ti.UI.Label',
        bindId : 'seven',
        properties : {
            color : 'gray',
            font : {
                fontFamily : 'Arial',
                fontSize : '14dp'
            },
            left : '240dp'
        }
    }]
};

var genData = function(section) {
    var data = [];
    for ( i = 0; i < 20; i++) {
        data.push({
            one : {
                text : section + ' one'
            },
            two : {
                text : 'two'
            },
            three : {
                text : 'three'
            },
            four : {
                text : 'four'
            },
            five : {
                text : 'five'
            },
            six : {
                text : 'six'
            },
            seven : {
                text : 'seven'
            },
            template : 'template'
        });
    }

    return data;
};

var genSections = function() {
    var sections = [];
    for ( j = 0; j < 5; j++) {
        var section = Ti.UI.createListSection({
            headerTitle : 'Section ' + j
        });
        section.setItems(genData(j));
        sections.push(section);
    }

    return sections;
};

var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

var mainView = Ti.UI.createView({
    backgroundColor : "white"
});

var buttonsView = Ti.UI.createView({
    backgroundColor : "gray",
    bottom : 0,
    height : 40, 
});

var listView = Ti.UI.createListView({
    height : 420,
    top : 0
});

listView.templates = {
    'template' : myTemplate
};
listView.setSections(genSections());

var down = Ti.UI.createButton({
    title : 'Add Section '
});
down.addEventListener('click', function() {
    j = listView.getSectionCount();
    var section = Ti.UI.createListSection({
        headerTitle : 'Section ' + j
    });
    section.setItems(genData(j));
    listView.appendSection(section);
    listView.scrollToItem(listView.getSectionCount() - 1, 0, {
        animated : false,
        //position : Titanium.UI.iPhone.TableViewScrollPosition.TOP
    });
});

buttonsView.add(down);

mainView.add(buttonsView);
mainView.add(listView);

win.add(mainView);
win.open();

    I hope it will solve your problem.

    Thanks
