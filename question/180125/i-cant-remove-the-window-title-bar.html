I can't remove the window title bar

Hi,

In my app.js I create a new window this way:

mainWindow = Titanium.UI.createWindow({
    title : '',
    backgroundColor : '#eee',
    backgroundImage : '/images/bg-menue@2x.png',
    tabBarHidden : true,
    navBarHidden : true
});

mainWindow.open();

But I can't hide the bar on top of the window on android. The bar is always there. I am talking about the bar with the app-icon on the left and the text which is in the property title of the window.

Is there any way to hide this bar?

Thanks,
Joscha

    Create a theme 

    <?xml version="1.0" encoding="utf-8"?>
<resources>
     <style name="Theme.NoActionBar" parent="@style/Theme.AppCompat">
        <item name="android:windowActionBar">false</item>
        <item name="android:windowNoTitle">true</item>
    </style>
</resources>

    Refer this theme in your tiapp.xml 

    <android xmlns:android="http://schemas.android.com/apk/res/android">
        <manifest >    
            <uses-sdk android:minSdkVersion="14" android:targetSdkVersion="19"/>
            <application android:debuggable="false" android:theme="@style/Theme.NoActionBar">
            </application>
        </manifest>
  </android>

    Add theme in your window

    mainWindow = Titanium.UI.createWindow({
    title : '',
    backgroundColor : '#eee',
    backgroundImage : '/images/bg-menue@2x.png',
    tabBarHidden : true,
    navBarHidden : true,
   theme : ''Theme.NoActionBar"
});

mainWindow.open();

    If the above method not worked then try this one also

    var actionBar;
mainWindow.addEventListener("open", function() {
    if (Ti.Platform.osname =="android") {
        if (mainWindow.activity) {
            actionBar = mainWindow.activity.actionBar;
            if (actionBar) {
                  actionBar.hide();              
        }
    }
    }
});
