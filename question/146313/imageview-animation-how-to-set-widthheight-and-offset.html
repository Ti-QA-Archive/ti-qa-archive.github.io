    var win = Titanium.UI.currentWindow;

//
// REUSABLE ANIMATION WITH TRANSFORM
//

var arrImgs = [];

//
// TOP LEFT
//
var view1 = Titanium.UI.createView({
    backgroundColor : '#336699',
    top : 10,
    left : 220,
    height : 50,
    width : 50,
    anchorPoint : {
        x : 0,
        y : 0
    }
});
arrImgs.push(view1);
win.add(view1);

//
// TOP RIGHT
//
var view2 = Titanium.UI.createView({
    backgroundColor : '#336699',
    top : 80,
    left : 220,
    height : 50,
    width : 50,
    anchorPoint : {
        x : 1,
        y : 0
    }
});
arrImgs.push(view2);
win.add(view2);

//
// BOTTOM LEFT

var view3 = Titanium.UI.createView({

        y : 1
    }

var view4 = Titanium.UI.createView({

//

var view5 = Titanium.UI.createView({
    backgroundColor : '#336699',
    top : 290,
    left : 220,
    height : 50,

        x : 0.5,
        y : 0.5
    }
});
arrImgs.push(view5);

var btnAnimate = Titanium.UI.createButton({
    title : 'Animate',
    height : 40,
    width : 80,
    left : 50
});

btnAnimate.addEventListener('click', function() {
    // view5.animate(a);
    for (var i = 0; i < arrImgs.length; i++) {
        var img = arrImgs[i];
        var t = Ti.UI.create2DMatrix();
        t = t.rotate(180);

        var a = Titanium.UI.createAnimation();
        a.transform = t;
        a.duration = 500;
        a.autoreverse = true;
        img.animate(a);
    }
});

win.add(btnAnimate);
