slider values in decimals

Hi,

I am new to titanium. I am using slider using following code

var slider = Titanium.UI.createSlider({
top: 50,
min: 0,
max: 55.25,
width: '100%',
value: 50
});

var label = Ti.UI.createLabel({
text: slider.value,
width: '100%',
height: 'auto',
top: 30,
left: 0,
textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER
});

slider.addEventListener('change', function(e) {
label.text = String.format("%3.1f", e.value);
});

Can some please tell me how would I get all the values of slider in decimals.
