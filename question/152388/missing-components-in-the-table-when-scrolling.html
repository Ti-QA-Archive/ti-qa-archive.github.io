function buildTable() {
    var tableOBJ = Ti.UI.createTableView({
        editable : false,
        top : 0,
        rowHeight : 44
    });

    tableOBJ.style = Titanium.UI.iPhone.TableViewStyle.GROUPED;

    return tableOBJ;
}

function buildTf(width) {
    var tfOBJ = Ti.UI.createTextField({
        right : 10,
        width : 100, //width || 80,

        //clearButtonMode : 1,
        borderStyle : Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,

        color : '#385487',
        //minimumFontSize : 8,
        height : 30,
        value : '687777'
    });

    return tfOBJ;
}
