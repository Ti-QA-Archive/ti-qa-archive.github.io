Referencing labels created in a loop

I have rows with views and labels attached to them, they are constructed in a loop.

I want to be able to change the value of certain labels when clicking a button AFTER they have been created.

How can I access each individual label?

Each row is constructed like this:


 for (var x=1; x<8; x++){


            var thisView = Ti.UI.createView({
                 backgroundColor:'#C0C0C0',
                borderColor:'#3f3f3f',
                borderRadius:0,
                borderWidth:0.5,
            height: 45,
            width: 45
            });
            thisRow.add(thisView);

            var thisLabel = Ti.UI.createLabel({
                        color:"#3f3f3f",
                        cellID:cellID,
                        height:'auto',
                        width:'auto',
                        font:{fontSize:21,fontWeight:'bold'},
                        text:'',
                        touchEnabled:false
            });
              thisView.add(thisLabel);
}
tableData.push(thisRow);

    myTable.addEventListener('click', function(e){
    if(e.source == 'TiUILabel'){
        e.source.text = 'new text';
    }
    else if(e.source == 'TiUIView')
    {
           e.source.children[0].text = 'hello world';
    }

});

    You also can add a listener to the label

    for (var x=1; x<8; x++){


            var thisView = Ti.UI.createView({
                backgroundColor:'#C0C0C0',
                borderColor:'#3f3f3f',
                borderRadius:0,
                borderWidth:0.5,
            height: 45,
            width: 45
            });
            thisRow.add(thisView);

            var thisLabel = Ti.UI.createLabel({
                        color:"#3f3f3f",
                        cellID:cellID,
                        height:'auto',
                        width:'auto',
                        font:{fontSize:21,fontWeight:'bold'},
                        text:'',
                        touchEnabled:false
            });

            // ________________________________HERE
             thisLabel.addEventListener('click', function(e){
                  e.source.text = 'my new text';
            })

              thisView.add(thisLabel);
}
tableData.push(thisRow);
