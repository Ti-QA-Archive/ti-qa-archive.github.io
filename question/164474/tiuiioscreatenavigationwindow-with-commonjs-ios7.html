I am trying to make my navigation group with commonjs but couldn't make it, how can make it work ?

Thanks

app.js

var navAc = require('/navWinTest').navAc, win2 = require('/win2').win2;

var rootWin2 = Ti.UI.createWindow({
    backgroundColor : 'red',
    navBarHidden : false
});
var button = Ti.UI.createButton({
    title : 'Open Second'
});
rootWin2.add(button);

button.addEventListener('click', function() {
    Ti.API.log('--->>  Clicked!');
    var mySecondWindow = new win2();
    new navAc(rootWin2, mySecondWindow);

});

win2.js

exports.win2 = function() {'use strict';

        backgroundColor : 'green',

    });
    //win.open();

};

navWinTest.js

exports.navAc = function(rootWin, secondWin) {'use strict';

    var navWin = Ti.UI.iOS.createNavigationWindow({
        window : rootWin
    });

    var b2 = Ti.UI.createButton({
        title : "Go Back"
    });

    secondWin.add(b2);

    navWin.openWindow(secondWin, {
        animated : true
    });
    b2.addEventListener('click', function() {
        navWin.closeWindow(secondWin);
    });

    return navWin;
};
