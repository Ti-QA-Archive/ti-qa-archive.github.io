  • This will not handle the case of versions of IE lower than 7, but if your accessing your own personal server code

    function customXHR(action, postData){
var url = 'server.php'; //just the local path
var req = new XMLHttpRequest();
var method = action;
req.open(method,url,true);

if(action == "POST"){
req.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
}
req.setRequestHeader('User-Agent','Mozilla/5.0');

    req.onreadystatechange = function () {
        if (req.readyState != 4) {
            return;
        }
        else if(req.status != 200 && req.status != 304) {
            return;
        }
        handleThis(this);

    }
    if (postData){
        req.send(postData);
    }
    else{
        req.send();
    }
}

function handleThis(req) {
    alert(req.responseText);
}
