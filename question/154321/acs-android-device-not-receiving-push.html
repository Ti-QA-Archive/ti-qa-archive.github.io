      getDeviceToken();

function getDeviceToken(){
    var CloudPush = require('ti.cloudpush');
    CloudPush.enabled = true;    //Very important stuff, solve "Cannot get specific MQTT Dispatcher to send message to" for me
    CloudPush.retrieveDeviceToken({
            success: function deviceTokenSuccess(e) {
                alert('Device Token: ' + e.deviceToken);
                login();
            },
            error: function deviceTokenError(e) {
                alert('Failed to register for push! ' + e.error);
         }
    });
    CloudPush.addEventListener('callback', function (evt) {
        alert(evt.payload);
    });
}


function login(){
    Cloud.Users.login({
            login: username,
            password: password
        },
        function (e) {
            if (e.success) {
                var user = e.users[0];
                firstName = user.first_name;
                lastName = user.last_name;
                subscribeToServerPush();
            }
            else {
                registerUser();
            }
        }
    );
}

function registerUser(){
    Cloud.Users.create({
        username: username,
        password: password,
        password_confirmation: password,
        first_name: firstName,
        last_name: lastName
    }, function (e) {
        if (e.success) {
            alert("USER CREATED SUCCESSFULLY.");
            login();
        }
        else {
            alert("Error register");
        }
    });
}

function subscribeToServerPush(){
    Cloud.PushNotifications.subscribe({
            channel: channel,
            type: type,
            device_token: user_device_token
        }, function (e) {
            if (e.success) {
                alert('Welcome '+firstName+' '+lastName+' !');
            } else {
                alert('Error '+user_device_token+'(subscribeToServerPush) :\\n' + ((e.error && e.message) || JSON.stringify(e)));
            }
        }
    );
}

    • PLEASE NOTE!!

      I marked this as best answers, though Paul Lv came up with the real solution by studying our log files, so credits to Wei and Paul both.

      As mentioned I always used CloudPush.enabled = true; Apparently this is problematic, when used on device BEFORE retriveDeviceToken was called (seems to be no problem for ths simulator?).

      So always make sure to call Ti.CloudPush.enabled = true; within the success method of retrieveDeviceToken(), this solved our problem!
      When you study the example in C:\Users[User]\Application Data\Titanium\modules\android\ti.cloudpush\2.2.0\example you will see that it is always enabled after the DeviceToken is fetched.

      Thank you all for helping!

      function deviceTokenSuccess(e) {
    Ti.API.info('Device Token: ' + e.deviceToken);
    deviceTokenLabel.text = e.deviceToken;
    enablePush.enabled = true;
}

      
/*
 Push can be enabled or disabled.
 */
var enablePush = Ti.UI.createButton({
    top: '10dp', width: '320dp', height: '40dp',
    enabled: false
});
enablePush.addEventListener('click', function () {
    enablePush.title = CloudPush.enabled ? 'Disabling...' : 'Enabling...';
    CloudPush.enabled = !CloudPush.enabled;
    // NOTE: Push.enabled takes a moment to update after you change its value.
    setTimeout(syncButtons, 500);
});
win.add(enablePush);
