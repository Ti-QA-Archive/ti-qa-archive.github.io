Set events in ios

Hi,

I'm using event kit module to set events in ios device calender.
I'm able to set the event successfully but all the events are set on 1-Jan-2001 on 5:30AM(
may be default value). Instead i want to set the event on the date and time selected on picker.

I'm using titanium 3.1

Added the code below.
(dt,yr,mnth,shr,smin,ehr,emin are some variables which has the picker values).

        var eventBegins = new Date();
        //Insert values from picker
                eventBegins.setDate = dt;
                eventBegins.setFullYear = yr;
                eventBegins.setMonth = mnth;
                eventBegins.setHours = shr;
                eventBegins.setMinutes = smin;

                var eventEnds = new Date();
        //Insert values from picker
                eventEnds.setFullYear=yr;
                eventEnds.setMonth=mnth;
                eventEnds.setDate=dt;
                eventEnds.setHours=ehr;
                eventEnds.setMinutes=emin;

                    EK.requestAuthorization(function(e) {
                        if (e.authorized == true) {
                            EK.createCalendarEvent({
                                title: "Title",
                                notes: "note",
                                //location: "B385 - Foo",
                                begin: eventBegins,
                                end: eventEnds,
                            });
                           Ti.UI.createAlertDialog({
                                title: "Event",
                                message: "Event has been created successfully...",
                                buttonNames: ["OK"]
                            }).show();
                            win.close();
                        } else {
                            alert("You haven't permissions to add Events");
                            win.close();
                        }
                    });
