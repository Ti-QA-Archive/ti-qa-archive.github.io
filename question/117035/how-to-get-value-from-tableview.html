How to get value from tableview

I've succeeded to get values from table in case1 below.
But I have no idea how to get values in case2
There must be some mistakes but now way to catch it, cuz I am not an experienced programmer.

case.1

var win = Ti.UI.createWindow();

var table = Ti.UI.createTableView();

var array_row =[];

function read(x){

var row = Ti.UI.createTableViewRow();

var label = Ti.UI.createLabel({

text : 'result :' + x });

row.add(label);

}

function main(){

for ( i=0; i<10; i++) {

array_row[i] = read(i);

}

table.setData(array_row);

}

main();

win.add(table);

win.open();

table.addEventListener('click', function(e){

Ti.API.info(array_row[e.index].children[0].text);
});

// this event works good!

case.2

var win = Ti.UI.createWindow();

var table = Ti.UI.createTableView();

function read(x){

var row = Ti.UI.createTableViewRow();

var label = Ti.UI.createLabel({

text : 'result :' + x });

row.add(label);

table.appendRow(row);

}

function main(){

for ( i=0; i<10; i++) {

read( i );

}

}

main();

win.add(table);

win.open();

table.addEventListener('click', function(e){

Ti.API.info(table[e.index].children[0].text);

});

// this event doesn't work

// Should I create data storage like array_row to access and refer the value(label.text)? Is there any other easy way to access?

    You can add additional information to any tableViewRow

    var row = Ti.UI.createTableViewRow({myid:'a value'});

    that you can then retrive in the event

    table.addEventListener('click', function(e){
   Ti.API.info(e.rowData.myid);
});
