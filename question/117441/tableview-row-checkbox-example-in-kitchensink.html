  • this is a simple example that hopefully can get you started

    
var data = [
    {title:"Row 1"},
    {title:"Row 2"},
    {title:"Row 3"},
    {title:"Row 4"}
];
var itemList = Titanium.UI.createTableView({
    data: data,
    top: 0
});

itemList.addEventListener('click',function(e){
    e.row.hasCheck = !e.row.hasCheck;
    var _rowData = itemList.data[0].rows;
    for ( var x in _rowData) {
        Ti.API.debug(_rowData[x].title + " : " + _rowData[x].hasCheck);    
    }

});

    You can get more creative by creating a custom row and add a control to actually capture the click events instead of the whole row capturing it
