  • Hi Bunty

    try this one

    var actionView = Ti.UI.createView({
    width : "100%",
    height : 60,
    horizontalWrap : false,
    layout : "horizontal"
});
$.winIndex.add(actionView);

var seperatorView = Ti.UI.createView({
    backgroundColor : "red",
    bottom : 0,
    width : "100%",
    height : 3
});

var pre = undefined;
function doClick(e){
    if(pre != undefined){
        pre.remove(seperatorView);
    }    
    e.source.add(seperatorView);
    pre = e.source;
}

var btn1 = Ti.UI.createView({
    width : "33.33%",
    height : "100%",
    id : 1,
    backgroundColor : "yellow",
});
actionView.add(btn1);
btn1.add(seperatorView);

var btn2 = Ti.UI.createView({
    width : "33.33%",
    height : "100%",
    id : 2,
    backgroundColor : "blue",
    style : "plain"
});
actionView.add(btn2);

var btn3 = Ti.UI.createView({
    width : "33.33%",
    height : "100%",
    id : 3,
    backgroundColor : "yellow",
    style : "plain"
});
actionView.add(btn3);

btn1.addEventListener("click",doClick);
btn2.addEventListener("click",doClick);
btn3.addEventListener("click",doClick);
