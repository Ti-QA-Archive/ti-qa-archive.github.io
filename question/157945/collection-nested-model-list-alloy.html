  • Nested collections aren't really supported, but if you have all the data in your model, you could implement the above UI probably using labels and the dataTransform property to assemble your multiple likes and comments into a single UI component. Here's the docs and an example of dataTransform

    • http://docs.appcelerator.com/titanium/latest/#!/guide/Alloy_Data_Binding

    • Thanks. I was able to have the data from my model using the data transform like so.

      <Alloy>
    <Collection src="post">
    <Window id="win">
        <TableView id="table" dataCollection="post" data>
            <TableViewRow class="row" dataTransform="transformFunction">
                <Label text="{title}" class="title"></Label>
                <Label text="{summary}" class="summary"></Label>
                                <TableView data="{comments}" scrollable="false"></TableView>
                                <TableView data="{likes}" scrollable="false"></TableView>
            </TableViewRow>
        </TableView>
    </Window>
</Alloy>

      and in my transform function inside my controller i did.

      function transformFunction(model) {
    var transform = model.toJSON();

   comments_data = [];

   _.each(transform.comments, function(comment){
            var row = Ti.UI.createTableViewRow();
            var label = Ti.UI.createLabel({
                 text: comment.get('username')
            });
           var label_comment = Ti.UI.createLabel({
                text: comment.get('comment')
           });
           row.add(label);
           row.add(label_comment);
            comments_data.push(row);
    });

    transform.comments = comments_data;

   likes_data = [];

   _.each(transform.likes, function(like){
            var row = Ti.UI.createTableViewRow();
            var label = Ti.UI.createLabel({
                 text: like.get('username')
            });
           row.add(label);
            likes_data.push(row);
    });

    transform.likes = likes_data;

    return transform;
}
