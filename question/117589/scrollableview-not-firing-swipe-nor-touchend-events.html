ScrollableView not firing 'swipe' nor 'touchend' events

Documentation shows these events available: http://developer.appcelerator.com/apidoc/mobile/latest/Titanium.UI.ScrollableView-object

But they are not available. Also the 'scroll' event doesn't fire always either.

Sample code:

var win = Titanium.UI.createWindow({  
    backgroundColor:'#fff'
});

var views_arr = [];

for(var i=0; i<10; i++){
    views_arr[i] = Ti.UI.createView({
        width: 200,
        height: 100,
        borderColor:'#000' 
    });
    var lbl = Titanium.UI.createLabel({
        text:'view '+i,
        color:'#000',
    });
    views_arr[i].add(lbl);
}

scrollableView = Titanium.UI.createScrollableView(
{
    views: views_arr,
    top: 0,
    left: 0,
    right: 0,
    borderWidth: 0,
    showPagingControl:true
});

scrollableView.addEventListener('scroll', function(e)
{
    Titanium.API.info("Image Scrolled current page: " + e.currentPage);
});

scrollableView.addEventListener('touchstart', function(e)
{
    Titanium.API.info("Image touchstart: " + e.x);
});

scrollableView.addEventListener('touchmove', function(e)
{
    Titanium.API.info("Image touchmove: " + e.x);
});

scrollableView.addEventListener('touchend', function(e)
{
    Titanium.API.info("Image touchend: " + e.x);
});
win.add(scrollableView);
win.open();

  • Cannot swipe on ScrollableView.

    Not possible on Titanium SDK 4.0 RC or below

    ####use ScrollView inside ScrollableView for ANDROID

    ####use view inside ScrollableView for IOS

    On ANDROID

    <ScrollView scrollType="horizontal" layout="horizontal" id="scrollView">
    <ScrollableView>
        <View></View>
        .
        .
        <View></View>
    </ScrollableView>
</ScrollView>

    now listen to scrollView's swipe

    $.scrollView.addEventListener('swipe', function(e)
{
    if(e.direction == "left" || e.direction == "right"){
        // triggers left right swipe
    }
}

    On IOS

    <View id="view"  layout="vertical">
    <ScrollableView>
        <View></View>
        .
        .
        <View></View>
    </ScrollableView>
</View>

    now listen to view's swipe

    $.view.addEventListener('swipe', function(e)
{
    if(e.direction == "left" || e.direction == "right"){
        // triggers left right swipe
    }
}

    ###Make sure you use scrollType and layout params as i have used for android & ios.

    And that's Jenga!
