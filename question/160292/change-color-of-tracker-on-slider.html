Change color of tracker on slider

I'm trying to figure out the best way to get rid of the default gray color of the slider. I'm having no luck. Here is my code. Can't find any posts on it here in the forums. Anybody have experience with this?

var sl_01 = Titanium.UI.createSlider({
    top: 100,
    min: 0,
    max: 10,
    width: '25%',
    value: 50,
    backgroundImage: 'assets/sliderBar.png'
});


var label_sl01 = Ti.UI.createLabel({
    text: sl_01.value,
    width: '100%',
    height: 'auto',
    top: 30,
    left: 0,
    textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER
});
