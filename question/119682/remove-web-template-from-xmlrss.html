remove web template from xml_rss

Hey, been searching the forums and google for a while now to see if I can get some help…

I grabbed the xml_rss feed from the KS, and found a modified version on the forums here (source code at end of post), and made some edits to it, but one thing I cant figure out, is that how do I get rid of the websites template from around the post. If the posts has pictures, videos etc, then I want those to show up, but dont want like the web template with the navigation bar etc to show up?

any help is appreciated. Thanks!

code:

// create table view data object
var data = [];

var xhr = Ti.Network.createHTTPClient();
xhr.open("GET","http://mpatel.mpsites.com/?feed=rss2");
xhr.onload = function()
{
    try
    {
        var doc = this.responseXML.documentElement;
        var items = doc.getElementsByTagName("item");
        var x = 0;
        var doctitle = doc.evaluate("//channel/title/text()").item(0).nodeValue;
        for (var c=0;c<items.length;c++)
        {
            var item = items.item(c);
            var title = item.getElementsByTagName("title").item(0).text;
            var row = Ti.UI.createTableViewRow({height:40});
            var label = Ti.UI.createLabel({
                text:title,
                left:7,
                top:2,
                bottom:2,
                right:5             
            });
            row.add(label);
            data[x++] = row;
            row.url = item.getElementsByTagName("link").item(0).text;
        }

        var tableview = Titanium.UI.createTableView({data:data});
        Titanium.UI.currentWindow.add(tableview);
        tableview.addEventListener('click',function(e)
        {
            var w = Ti.UI.createWindow({title:doctitle});
            var wb = Ti.UI.createWebView({url:e.row.url});
            w.add(wb);
            var b = Titanium.UI.createButton({
                title:'Close',
                style:Titanium.UI.iPhone.SystemButtonStyle.PLAIN
            });
            w.setLeftNavButton(b);
            b.addEventListener('click',function()
            {
                w.close();
            });
            w.open({modal:true});
        });
    }
    catch(E)
    {
        alert(E);
    }
};
xhr.send();
