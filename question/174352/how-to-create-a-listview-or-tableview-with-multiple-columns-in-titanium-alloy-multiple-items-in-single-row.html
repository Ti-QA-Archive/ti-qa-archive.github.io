    Hi,

    To use 3 items in one row, please you can check the following code sample

    index.xml

    <Alloy>
    <Window class="container">
        <TableView id="table"></TableView>
    </Window>
</Alloy>

    index.js

    for (var i=0; i < 10; i++) {
    itemGeneration(i);
};

$.index.open();

function itemGeneration(i)
{
    var row=Ti.UI.createTableViewRow();
    var itemOne=Ti.UI.createLabel({
        text:'Item '+i,
        left:0,
    });
    var itemTwo=Ti.UI.createLabel({
        text:'Item '+i,
    });
    var itemThree=Ti.UI.createLabel({
        text:'Item '+i,
        right:0,
    });
    row.add(itemOne);
    row.add(itemTwo);
    row.add(itemThree);
    $.table.appendRow(row);

}
