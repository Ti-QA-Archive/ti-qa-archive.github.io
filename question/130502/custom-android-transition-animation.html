                nextMid = getNextMenuLeft(currentMid,mode);

                nextMid = getNextMenuRight(currentMid,mode);
            var nextView = AppUtil.ui.createPlaceView(nextMid,mode);  //creates the view
            nextView.visible = false;
            placewin.add(nextView);  //placewin is the window object
            //right swipe should flip from left
            if (!AppUtil.isAndroid()){
                var anim = Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT;
                if (e.direction=='left')
                    anim=Ti.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT;                
                placewin.animate({view:nextView,transition:anim});
            }
            nextView.visible = true;
            placewin.remove(bodyView);  //bodyview is the current view
            bodyView=nextView;
