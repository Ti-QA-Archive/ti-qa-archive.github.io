Why does my main window don't closes ??

My app.js is as belows…

var win2 = Titanium.UI.createWindow({
    url: 'win2.js',
    title: 'Win2',
    backgroundColor: '#fff',
    navBarHidden: true,
    fullscreen: true,
    exitOnClose: false,
});


var winMain = Ti.UI.createWindow({
    title: 'Wikiki audit',
    backgroundColor: '#fff',
    navBarHidden: false,
    exitOnClose: true,
});



Ti.App.addEventListener( 'audit', function(e) {
    win2.close();
    win2 = null;
    winMain.close();
});

winMain.open(); 
win2.open();

In win2.js, Ti.App.fireEvent('audit') successfully triggers eventlistener in app.js, and win2 closes, but winMain never closes. What's wrong with me ??

Thanks in advance.

  • I myself solved this.

    Ti.App.addEventListener( 'audit', function(e) {
    win2.close();
    win2 = null;
    setTimeout( function() { winMain.close(); }, 1000 );
});

    As above, delayed close can close the window. That might related with focusing of the window to be closed….
