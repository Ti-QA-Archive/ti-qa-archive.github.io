  • Try adding it after you open the window, such as:

    var win1 = Ti.UI.createWindow({
   url:'_views/calendar.js',
   title:'Calendar',
   tabBarHidden:true,
   barImage:'_images/bar_bg_empty.png',
   backgroundColor:'#dfdddb'
});

Ti.UI.currentTab.open(win1,{animated:true});

win1.setRightNavButton(win1BtnRight);
Ti.API.warn(win1.rightNavButton);
