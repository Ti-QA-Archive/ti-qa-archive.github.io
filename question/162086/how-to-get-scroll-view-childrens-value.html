    there are different ways to achieve that.
    One quick way is that you go through all of the children in your scrollview:

    for (var a in scrollView.children) {
    var toggler = scrollView.children[a];
    var checkbox = toggler.children[1]; //because it's the second child
    if (checkbox.selected === true) {
        alert("THE CHECKBOX WITH ID ="+checkbox.ID+" IS SELECTED!");
        checkbox.selected = false; //this deselects it
    }
}
