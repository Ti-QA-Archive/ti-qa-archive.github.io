{
case 0:
openOptionsDialog();
break;
case 1:

break;
}
return true;
}

.setMessage(R.string.app_about_message)
.setPositiveButton(R.string.str_ok,
new DialogInterface.OnClickListener()
{
public void onClick(DialogInterface dialoginterface, int i)
{
}
})
.show();
}

{

.setTitle(R.string.app_exit)
.setMessage(R.string.app_exit_message)

      var activity = Ti.Android.currentActivity;

// Here is an example of creating the menu handlers after window creation but before open.
b1.addEventListener('click', function(e) {
    var w = Ti.UI.createWindow({
        backgroundColor : 'blue',
        navBarHidden : false
    });


    w.activity.onCreateOptionsMenu = function(e) {
        var menu = e.menu;

        var m1 = menu.add({
            title : 'Close Window'
        });
        m1.addEventListener('click', function(e) {
            Ti.UI.currentWindow.close();
        });
    };

    w.activity.onPrepareOptionsMenu = function(e) {
        var menu = e.menu;

        menu.clear();

        var mi = menu.add({
            itemId : 2,
            order : 1,
            title : 'Toast'
        });
        mi.addEventListener('click', function(e) {
            Ti.UI.createNotification({
                message : "To you and yours."
            }).show();
        });
    };
    var l = Ti.UI.createLabel({
        backgroundColor : 'white',
        color : 'black',
        width : 'auto',
        height : 'auto',
        text : 'Press the menu button, then select Close Window'
    });
    w.add(l);

    w.open({
        animated : true
    });
});

win.add(b1);
