  • Hi

    I have also just tried the documentation above and you are correct, following those alloy example files - nothing, no app - annoying.

    So here is a really simple example that will get you started. To try this create a new default Alloy app and then use the bold filenames to create a new file and then copy the code in, add new files as required.

    Important; removed the contents of the index.tss but leave the file - just in case anything conflicts, will not kill anything, just may look weird.

    Here goes, there are a few;

    app/views/index.xml

    <Alloy>
    <Window id="index">
        <NavigationGroup id="navgroup" platform="ios,mobileweb">
            <Window title="Window 1" id="win1">
                <Label>This is the first window</Label>
            </Window>
        </NavigationGroup>
        <Label platform="android">NavigationGroup not available on Android</Label>
    </Window>
</Alloy>

    app/views/win2.xml

    <Alloy>
    <Window title="Window 2">
        <Label>This is window 2</Label>
    </Window>
</Alloy>

    app/views/win3.xml

    <Alloy>
    <Window title="Window 3">
        <Label>This is window 3</Label>
    </Window>
</Alloy>

    app/controllers/index.js

    if (OS_IOS || OS_MOBILEWEB) {
    // attach the navgroup to Alloy.CFG so it can be accessed globally
    Alloy.CFG.navgroup = $.navgroup;

    // We can do most of this in markup once we resolve:
    // https://jira.appcelerator.org/browse/ALOY-202
    //
    // TODO: find out why nav buttons and titles don't work for mobileweb
    var button = Ti.UI.createButton({ title: 'next' });
    button.addEventListener('click', function(e) {
        $.navgroup.open(Alloy.createController('win2').getView());
    });
    $.win1.rightNavButton = button;
}
$.index.open();

    app/controllers/win2.js

    // We can do most of this in markup once we resolve:
// https://jira.appcelerator.org/browse/ALOY-202
var button = Ti.UI.createButton({ title: 'next' });
button.addEventListener('click', function(e) {
    // Access navgroup from Alloy.CFG, set in index.js
    Alloy.CFG.navgroup.open(Alloy.createController('win3').getView());
});
$.win2.rightNavButton = button;

    app/styles/app.tss

    "Window": {
    backgroundColor: '#fff'
},
"Label": {
    color: '#000',
    font: {
        fontWeight: 'bold'
    }
}

    I found this example a while ago, not sure where but it works a treat as a starting point for you.

    • Here the documentation comes back to the rescue, I have taken the examples for the ListView and tweaked very slightly (the icon) and added a click event.

      Make the following changes to the above code;

      app/views/index.xml

      <Alloy>
    <Window id="index">
        <NavigationGroup id="navgroup" platform="ios,mobileweb">
            <Window title="Window 1" id="win1">
                <ListView id="listView" defaultItemTemplate="template" onItemclick="lvClick">
                    <!-- The Templates tag sets the ListView's templates property -->
                    <Templates>
                        <!-- Define your item templates within the Templates tags or use the
                             Require tag to include a view that only contains an ItemTemplate -->
                        <ItemTemplate name="template">
                            <ImageView bindId="pic" id="icon" />
                            <Label bindId="info" id="title" />
                            <Label bindId="es_info" id="subtitle" />
                        </ItemTemplate>
                    </Templates>
                    <ListSection headerTitle="Fruit / Frutas">
                        <!-- You can specify any ListItem or ListDataItem properties in ListItem -->
                        <!-- Specify data to bind to the item template with inline attributes
                             defined as <bindId>:<Ti.UI.Component.property> -->
                        <ListItem info:text="Apple" es_info:text="Manzana" pic:image="appicon.png" />
                        <ListItem info:text="Banana" es_info:text="Banana" pic:image="appicon.png" />
                    </ListSection>
                    <ListSection headerTitle="Vegetables / Verduras">
                        <ListItem info:text="Carrot" es_info:text="Zanahoria" pic:image="appicon.png" />
                        <ListItem info:text="Potato" es_info:text="Patata" pic:image="appicon.png" />
                    </ListSection>
                    <ListSection headerTitle="Grains / Granos">
                        <ListItem info:text="Corn" es_info:text="Maiz" pic:image="appicon.png" />
                        <ListItem info:text="Rice" es_info:text="Arroz" pic:image="appicon.png" />
                    </ListSection>
                </ListView>
            </Window>
        </NavigationGroup>
        <Label platform="android">NavigationGroup not available on Android</Label>
    </Window>
</Alloy>

      app/controllers/index.js

      if (OS_IOS || OS_MOBILEWEB) {
    // attach the navgroup to Alloy.CFG so it can be accessed globally
    Alloy.CFG.navgroup = $.navgroup;

    // We can do most of this in markup once we resolve:
    // https://jira.appcelerator.org/browse/ALOY-202
    //
    // TODO: find out why nav buttons and titles don't work for mobileweb
    var button = Ti.UI.createButton({ title: 'next' });
    button.addEventListener('click', function(e) {
        $.navgroup.open(Alloy.createController('win2').getView());
    });
    $.win1.rightNavButton = button;
}
function lvClick(e) {
    Ti.API.info('e', JSON.stringify(e));
    alert(JSON.stringify(e));
    $.navgroup.open(Alloy.createController('win2').getView());
}
$.index.open();

      New file app/styles/index.tss

      if (OS_IOS || OS_MOBILEWEB) {
