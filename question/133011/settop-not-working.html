setTop not working

I cant able to set top for a view using "setTop". Here is my sample code

var scrollView = Titanium.UI.createScrollView({ 
    contentWidth:300, 
    contentHeight:"auto", 
    top:10,
    left:10,
    right:10,
    bottom:10,
    borderColor:"#999",
    borderWidth:1,
    borderRadius:8,
    backgroundColor:'#fff',
    showVerticalScrollIndicator:true, 
    showHorizontalScrollIndicator:true
});
win.add(scrollView);

var titleContainer = Ti.UI.createView({
    height:'auto',
    width:300,
    top:10,
    backgroundColor:'blue',
    visible:true
})
scrollView.add(titleContainer);

var infoContainer = Ti.UI.createView({
    height:'auto',
    width:300,
    backgroundColor:'green',
    top:titleContainer.height+10
})
scrollView.add(infoContainer);

i am trying to set top using this line but it does not work. Anyone help me to solve this issue. I am using Titanium 1.7.5 and ios

infoContainer.setTop = 0;

or

infoContainer.top = 0;

Thanks in Advance

    infoContainer.top = 0; wont work unless you set height of titleContainer and infoContainer other than 'auto'.

    Try this:

    var scrollView = Titanium.UI.createScrollView({ 
    contentWidth:300, 
    contentHeight:"auto", 
    top:10,
    left:10,
    right:10,
    bottom:10,
    borderColor:"#999",
    borderWidth:1,
    borderRadius:8,
    backgroundColor:'#fff',
    showVerticalScrollIndicator:true, 
    showHorizontalScrollIndicator:true
});
win.add(scrollView);

var titleContainer = Ti.UI.createView({
    height:'40%',// or fix value, for example 200
    width:300,
    top:10,
    backgroundColor:'blue',
    visible:true
})
scrollView.add(titleContainer);

var infoContainer = Ti.UI.createView({
    height:'50%',// or fix value, for example 200
    width:300,
    backgroundColor:'green',
    top:titleContainer.height+10
})
scrollView.add(infoContainer);

win.open();

setTimeout(function(){
    infoContainer.top = 0;
}, 2000);
