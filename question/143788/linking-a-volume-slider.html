here is an example of what i have working so far.

var win1 = Titanium.UI.createWindow({
    title: 'win1',
    backgroundColor: 'blue',

});

win1.open();

var posLabel = Ti.UI.createLabel({
    bottom: 10, width:100, left: 120,
    text : 'Pos: 0'

    value: 0,
