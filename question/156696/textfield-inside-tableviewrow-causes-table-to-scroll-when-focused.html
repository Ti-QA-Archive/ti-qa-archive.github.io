TextField inside TableViewRow causes table to scroll when focused

On IOS :
When i focus on a textfield inside a tableView and start scrolling, at the end it will automatically move back to the focused textfield.

Is it possible to disable this behavior ?
Thanks

 var data = [];

    function addRow(idx) {             
        var field = Ti.UI.createTextField({
            width: 200,
            top: 0,
            value: idx,
            borderStyle: Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
        }); 
        var view = Ti.UI.createView(); 
        view.add(field);
        data[idx].add(view);
    }  

    // Create 20 rows
    for (var i=0; i <= 20; i++) {
        data[i] = Ti.UI.createTableViewRow({hasDetail:true,height:'50'});
        addRow(i);
    };

    // Create table view
     var tableview = Titanium.UI.createTableView({
         data:data,
        backgroundColor: 'green',
     });  
     win.add(tableview);
    win.open();
