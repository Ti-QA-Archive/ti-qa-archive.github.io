This is my App.js:


var win = Titanium.UI.createWindow({fullscreen : true,backgroundColor : '#fff'});
var itemsPerPage = 10;// How many fish cards per page
var f = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory, 'amazon.xml');
var xmltext = f.read().text;
var doc = Ti.XML.parseString(xmltext);
var exhibitItems = doc.getElementsByTagName("item");
var pages = [];//will hold a bunch of Views that hold all the cards for each fish
var rowCount = 0;//used to layout the cards within the views - PageFlip seems to break the Layout property and stuff just piles on stuff of each other otherwise
var colCount = 0;//used to layout the cards within the views - PageFlip seems to break the Layout property and stuff just piles on stuff of each other otherwise
var page = Ti.UI.createView();//Start with a blank page
var label = Ti.UI.createLabel({text:"Swipe it!"});page.add(label);//Add a label to it
for(var i = 0; i < exhibitItems.length; i++) {//make a card for each item on exhibit (about 260)
    if(i % itemsPerPage !== 0) {//if within a group of itemsPerPage
        fishCard = Ti.UI.createView({//make an enclosing view for the card
            width : 250,height : 300});
        item = Ti.UI.createImageView({//create an image view for the illustration
            image : "images/amazon/images/" + exhibitItems.item(i).getElementsByTagName('image').item(2).getAttribute('file'),width : 250,height : 250});
        fishLabel = Ti.UI.createLabel({//create an label for the fish name
            width : 250,height : 50,color : '#000', font : {fontSize : 18,fontFamily : 'Cochin'},textAlign : 'center',bottom : 0,text : exhibitItems.item(i).getAttribute('common'),});    
        fishCard.add(fishLabel); fishCard.add(item);        
        colCount++;
        if(colCount > 2) {colCount = 0;rowCount++;}
        if(rowCount > 2) {rowCount = 0;}
        fishCard.left = 250*colCount; fishCard.top = 300*rowCount;
        page.add(fishCard);//Add our card to the current page being generated
    } else {//must be done with that page, time for a new one
        pages[pages.length] = page;//copy done page into pages
        rowCount = 0; colCount = 0;//reset our row counts
        page = Ti.UI.createView();//start a new page
    }
}
var pageflip = require("com.infinery.pageflip");
pageflip.PageView(win, pages);
