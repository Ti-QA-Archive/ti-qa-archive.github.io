function createOptionsField(labelText) {
        var labelName = ui.createLabel({
            text : labelText,
        });
        view.add(labelName);
        viewForList = Ti.UI.createView({
            height : '40dp',
            width : '280dp',
        });
        var addButton = Ti.UI.createButton({
            title : "Add",
            right : 0
        });
        viewForList.add(addButton);

        var textField = Ti.UI.createTextField({
            value : "",
            width : '223dp',
            height : '37dp',
            top : 0,
            font : {
                fontSize : '14sp',
            },
            left : 0,
            focusable : false
        });
        viewForList.add(textField);
        view.add(viewForList);

        var postData = new Object();
        postData.field = textField;
        postData.button = addButton;
        postData.viewForList = viewForList;
        return postData;
    };
