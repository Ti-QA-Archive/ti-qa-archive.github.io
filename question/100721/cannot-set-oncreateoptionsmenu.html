in app.js: 

var main = Ti.UI.createWindow({
    titleid:'main',
    url: 'Views/Main_Window.js',
    PLATFORM: PLATFORM,
    hideNavBar: true
});
...
main.open();

and in Main_Window.js:

    var activity = Titanium.Android.currentActivity;
