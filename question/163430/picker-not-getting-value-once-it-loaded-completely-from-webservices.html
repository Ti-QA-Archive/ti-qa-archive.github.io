    //Creating a new window
var sendmessage = Titanium.UI.createWindow({
    title : 'Send Message',
    backgroundColor : '#fff',
    navBarHidden : true,
    orientationModes : [Titanium.UI.PORTRAIT],
    tabBarHidden : true
    // url:'main_windows/login.js'
});

//calculate the height and width of the Device and divide both the value by 6
var totheight = Ti.Platform.displayCaps.platformHeight / 6;

var fixedmessagepicker = Ti.UI.createPicker({
  top:totheight * 1 + 'px',  
  width: '80%',
  height: '8%'
});
sendmessage.add(fixedmessagepicker);

var data = [];
data[0]=Ti.UI.createPickerRow({title:'Hi'});
data[1]=Ti.UI.createPickerRow({title:'Hello'});
data[2]=Ti.UI.createPickerRow({title:'You'});
fixedmessagepicker.add(data); 

var txtMessage = Ti.UI.createTextArea({ 
    width: '80%',
    hintText:'Enter Message...',
    top:totheight * 1.5 + 'px',
    height: totheight * 2 + 'px',

});
sendmessage.add(txtMessage);

fixedmessagepicker.addEventListener('change',function(e){
    //var feeType=feeTypePicker.getSelectedRow(0).title;

    txtMessage.value=e.row.title;
});

sendmessage.open();
