  • You can addEventListener to the label as Mauro said and open there: Try the following code :

    
var link = 'http://www.youtube.com/';

var linklabel = Ti.UI.createLabel({

    top:'auto',
    left:10,
    text:'http://www.youtube.com/',

    color: "blue",
    font: {fontSize: 25, fontWeight: 'bold'},
    textAlign:'left',

});
if (Titanium.Platform.name == 'android'){
linklabel.addEventListener('click', function(){

    var intent = Ti.Android.createIntent({
        action: Ti.Android.ACTION_VIEW,
        data:link
    });
    Ti.Android.currentActivity.startActivity(intent);
});
}
else if(Titanium.Platform.name == 'iPhone OS'){

    linklabel.addEventListener('click', function(e){

        Titanium.Platform.openURL('http://www.youtube.com/watch?v=5d5iIoCiI8g');
    });

}

    With platform condition. You can use this code for both Android and iOS.
    Try and let me know if this worked for you.
