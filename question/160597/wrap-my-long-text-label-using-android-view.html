Wrap my long text label using android view

Hi I have this code:


var win = Ti.UI.currentWindow;
var view = Ti.UI.createView({
    height: 'auto',
    layout: 'vertical'
});
var movieSynopsisView = Ti.UI.createView({
    left: 10,
    right:10,
    backgroundColor: dataBG,
    borderRadius:10,
    borderColor: '#EEEEEE',
    height: 'auto',
    color: '#000'
});

var movieSynopsis = Titanium.UI.createLabel({
    left:15,
    right:10,
    font:{fontSize:12},
    text: movie.synopsis + '\n\n',
    height: 'auto',
  color: '#000'
});
movieSynopsisView.add(movieSynopsis);
view.add(movieSynopsisView);

win.add(view);

The movieSynopsisView should wrap the movieSynopsis label and should follow the size of the label. THe view should act as a container for the view.
