Getting values from a form

I'm trying to take the info from a form I created and create a flyer from it. For the life of me I can't get the values from the submitted text field to become a label in a separate window. I have only one .js file (I'm still very new to this and have not yet figured out using multiple files)

This was my latest attempt. 

ProfileSubmit.addEventListener('click', function(e){
    if (PetNameField.value == "" || OwnerNameField.value == "" || OwnerPhoneField.value == "")
    {
    alert('All fields must be filled out');
    }
    else
    {
    alert(PetNameField.getValue());
    ProfileWindow.close({
        transition: Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
    });
    }
});

var FlyerPetName = Ti.UI.createLabel({
    text: PetNameField.value,
    top: 200,
    height: 'auto',
    width: 'auto',
});

The alert does pop up on submission, but I can't get the value otherwise.

Any help is appreciated.

    function showNameInNewWindow(petName) {
    var w = Ti.UI.createWindow();
    var l = Ti.UI.createLabel({
        text : petName,
        top : 200,
        height : 'auto',
        width : 'auto',
    });
    w.add(l);
    w.open();
}

ProfileSubmit.addEventListener('click', function(e) {

    if (PetNameField.value == "" || OwnerNameField.value == "" || OwnerPhoneField.value == "") {
        alert('All fields must be filled out');
    } else {
        alert(PetNameField.getValue());
        ProfileWindow.close({
            transition : Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
        });

        showNameInNewWindow(PetNameField.value);
    }
});
