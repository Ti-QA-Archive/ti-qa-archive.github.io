  • Hi Devanjali,

    Add click event listener of thisView.
    e.g

    ...
for (var x = 0; x < xGrid; x++) {
        var thisView = Ti.UI.createView({
            objName : "grid-view",
            objIndex : cellIndex.toString(),
            //objUrl : topicsArray[cellIndex].url,
            backgroundColor : colorSet[colorSetIndex],
            left : ySpacer,
            height : cellHeight,
            width : cellWidth
        });

    thisView.addEventListener("click",function(e){
        txtFld.value = e.source.objIndex;
    });
