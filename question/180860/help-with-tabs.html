    Hello Francisco,

    Try the below code. Suppose you have three tabs in a application where each tab contain different things. and also each tabs code is in separate file.

    "app.js"

    // this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');

// create tab group
var tabGroup = Titanium.UI.createTabGroup();
//
// create base UI tab and root window
//

var RescueWin = require('Window'),
    win1 = new RescueWin();
    win1.open();

var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Window',
    window:win1
});

var RescueWin = require('Web'),
    win2 = new RescueWin();
    win2.open();


var tab2 = Titanium.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Web',
    window:win2
});

var RescueWin = require('Map'),
    win3 = new RescueWin();
    win3.open();


var tab3 = Titanium.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Map',
    window:win3
});

tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);
tabGroup.addTab(tab3); 


// open tab group
tabGroup.open();

    First tab is a window with a table view.
    "Window.js"

    function Window(){

Ti.UI.backgroundColor = 'white';
var win1 = Ti.UI.createWindow({
    title:'Window',
});

var tableData = [ {title: 'Apples'}, {title: 'Bananas'}, {title: 'Carrots'}, {title: 'Potatoes'} ];

var table = Ti.UI.createTableView({
    bottom:10,
      data: tableData
});
win1.add(table);
return win1;


}

module.exports = Window;

    Second tab is a webview.
    "Web.js"

    function Web() {

    Ti.UI.backgroundColor = 'white';
    var win2 = Ti.UI.createWindow({
        title : 'Web',
    });

    var webview = Titanium.UI.createWebView({
        url : 'https://www.google.com.bd/'
    });

    win2.add(webview);
    return win2;

}

module.exports = Web;

    Third tab is a map.
    "Map.js"

    function Map() {

    Ti.UI.backgroundColor = 'white';
    var win3 = Ti.UI.createWindow({
        title : 'Map',
    });

    var MapModule = require('ti.map');
    var map1 = MapModule.createView({
        userLocation : true,
        mapType : MapModule.NORMAL_TYPE,
        animate : true,
        region : {
            latitude : -33.87365,
            longitude : 151.20689,
            latitudeDelta : 0.1,
            longitudeDelta : 0.1
        },
        height : '100%',
        top : 0,
        left : 0,
        width : '100%'
    });
    win3.add(map1);
    return win3;

}

module.exports = Map;

    Hope this helps you. If it does, Please mark as best answer. Thanks.
