    • try this sample..

      var self = Ti.UI.createWindow({
    backgroundColor : 'white',
});

self.addEventListener('touchend', function(e) {
    var v = Ti.UI.createView({
        width : 1,
        height : 1,
        top : e.y,
        left : e.x,
        backgroundColor : 'red'
    });

    self.add(v);
});

self.open();
