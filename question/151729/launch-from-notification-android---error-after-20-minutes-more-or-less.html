  • OK, so, I've programmed a simple app that shows that fact. There is an app.js, a service.js and a main_UI.js.
    Important to remember: it works fine at the beginning, but, when passing 20 minutes of non activity, it crashes when notification try to start the app.

    app.js:

    var myData = {};
myData.intent = Titanium.Android.createServiceIntent({
    url : 'service.js'
});

myData.ui = require('main_UI').createUI();
myData.ui.main.open();
myData.fcounter = 0 ;
myData.scounter = 0;
Ti.App.Properties.setInt("scounter", myData.scounter);

(function createController() {
    myData.ui.main.addEventListener('blur',function(){
        myStart();
    });
    myData.ui.main.addEventListener('focus', function() {
        myStop ();
        myData.fcounter += 1;
        myData.ui.label1.setText ('Focus Counter = ' + myData.fcounter);
        myData.scounter = Ti.App.Properties.getInt("scounter");
        myData.ui.label2.setText ('Servie Counter = ' + myData.scounter);
    });
})();

function myStop() {
    if (Ti.Android.isServiceRunning(myData.intent)) {
        //Ti.Android.stopService(myData.intent);
        myData.service.stop();
    }
};

function myStart() {
    if (!Ti.Android.isServiceRunning(myData.intent)) {
        myData.service = Titanium.Android.createService(myData.intent);
        myData.intent.putExtra('interval', 2000);
        myData.service.start();
    }
};

    service.js

    var counter = Ti.App.Properties.getInt("scounter");
counter += 1;
Ti.App.Properties.setInt("scounter", counter);

var intent = Ti.Android.createIntent({
    action : Ti.Android.ACTION_MAIN,
    className : 'com.plasmafive.testnotif.TestnotifActivity',
    flags : Ti.Android.FLAG_ACTIVITY_RESET_TASK_IF_NEEDED | Ti.Android.FLAG_ACTIVITY_SINGLE_TOP
});

intent.addCategory(Ti.Android.CATEGORY_LAUNCHER);

var activity = Ti.Android.currentActivity;

var pending = Titanium.Android.createPendingIntent({
    activity : activity,
    intent : intent,
    type : Ti.Android.PENDING_INTENT_FOR_ACTIVITY,
    flags : Titanium.Android.FLAG_UPDATE_CURRENT
});

var notification = Titanium.Android.createNotification({
    contentTitle : 'myNotif',
    contentIntent : pending,
    flags : Titanium.Android.ACTION_DEFAULT | Titanium.Android.FLAG_AUTO_CANCEL | Titanium.Android.FLAG_SHOW_LIGHTS
});
Titanium.Android.NotificationManager.notify(1, notification);

    main_UI.js

    function createUI() {
