  • you may try these .. set height to section items as given below

    
data.push({
            properties : {

                                    color : '#c2c1c2',
                            itemId : rows.fieldByName('id'),
                        title : rows.fieldByName('brand_name'),
                            accessoryType : Ti.UI.LIST_ACCESSORY_TYPE_CHECKMARK,
                        height:60,

                    }

});

        var listSection = Titanium.UI.createListSection({
            items : data
        });

        var listView = Titanium.UI.createListView({
            sections : [listSection],
            top : 40,
            width:'90%',
        });
