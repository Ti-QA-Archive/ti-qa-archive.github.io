True rotating only works on simulator

Hi,
because standard rotation only works in range of +-180° Tony Lukasavage has coded a wonderful module with name path.animator. In my case it only works on simulator and I dont know why:

exports.create = function() {
    var w = Ti.Platform.displayCaps.platformWidth;
    var self = Ti.UI.createWindow({});
    var rota = require('path.animator');
    var logocontainer = rota.createView();
    var logo = Ti.UI.createImageView({
        image : '/assets/logo.png',
        width: w/3,
        height: w/6
    });
    logocontainer.add(logo);
    logocontainer.rotate({
        angle : 2880,
        duration : 12000,
        delay : 100,
        timingFunction : Ti.UI.iOS.ANIMATION_CURVE_EASE_OUT
    });
    self.add(logocontainer);
    return self;
}
— asked January 2nd 2013 by Rainer Schleevoigt

  • self solved with:

    var logocontainer = rota.createView({
        width : Ti.UI.SIZE,
        height : Ti.UI.SIZE
});
