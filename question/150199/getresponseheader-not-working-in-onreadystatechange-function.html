getResponseHeader not working in onreadystatechange function

Hi Community,

Is there any possibility to use getResponseHeader("Last-Modified") function in the onreadystatechange function? I tried it but I am getting NULL in response!

What I wanted to achieve is, download resource from server only if it has been modified since its last download.

var xhr = Titanium.Network.createHTTPClient();

xhr.onreadystatechange = function(e)
{
    switch(this.readyState)
    {
        case 0:
            // after HTTPClient declared, prior to open()
                        // though Ti won't actually report on this readyState
                        Ti.API.info('case 0, readyState = ' + this.readyState);
                        break;
        case 1:
                        // open() has been called, now is the time to set headers
                        Ti.API.info('case 1, readyState = ' + this.readyState);
                        break;
        case 2:
                        // headers received, xhr.status should be available now
                        Ti.API.error(JSON.stringify(this.getResponseHeader("Last-Modified")));
                        Ti.API.info('case 2, readyState = ' + this.readyState);
                        break;
        case 3:
                        // data is being received, onsendstream/ondatastream
                        // being called now
                        Ti.API.info('case 3, readyState = ' + this.readyState);
                        break;
        case 4:
                        // done, onload or onerror should be called now
                        Ti.API.info('case 4, readyState = ' + this.readyState);
                        break;
    }
};

xhr.open('GET', 'http://goo.gl/6IeXJ');

xhr.send();

Output log:

[INFO] :   case 2, readyState = 2

    • I am using that approach as of now but in that case I need to made another GET request to get updated data if the resource is modified since its last download. Pls check the below code snippet I am using currently.

      Helper.downloadAssets = function (requestType) {
    var xhr = Titanium.Network.createHTTPClient();

    xhr.onload = function () {
        if (this.readyState == 4 && this.status == 200) {

            var lastModified = JSON.stringify(this.getResponseHeader("Last-Modified"));

            if (lastModified != Titanium.App.Properties.getString('zipLastModified')) {

                Helper.downloadAssets('GET');

                Titanium.App.Properties.setString('zipLastModified', lastModified);
                return;
            } else {
                if (requestType == "GET") {
                    var file = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, 'assets.zip');
                    file.write(this.responseData);
                }
            }
        }
    };

    xhr.open(requestType, url);

    xhr.send();
};

Helper.downloadAssets('HEAD');

      The following sample code shows what I wanted to achieve.

      Helper.downloadAssets = function (requestType) {

    var xhr = Titanium.Network.createHTTPClient();

    xhr.onload = function (e) {
        // logic to save zip file goes here...
    };

    xhr.onreadystatechange = function (e) {

        switch (this.readyState) {

            case 2:
                // headers received, xhr.status should be available now
                Ti.API.info(JSON.stringify(this.getResponseHeader("Last-Modified"))); // NULL
                Ti.API.info(JSON.stringify(xhr.getResponseHeader("Last-Modified"))); // NULL

                var lastModified = JSON.stringify(this.getResponseHeader("Last-Modified"));

                if (lastModified == Titanium.App.Properties.getString('zipLastModified')) {

                    // stop moving ahead for downloading resource
                    // as zip has not been modified on the server
                    xhr.abort();
                }

                Ti.API.info('case 2, readyState = ' + this.readyState);
                break;

            case 3:
                // data is being received, onsendstream/ondatastream
                // being called now
                Ti.API.info('case 3, readyState = ' + this.readyState);
                break;

            case 4:
                // done, onload or onerror should be called now
                Ti.API.info('case 4, readyState = ' + this.readyState);
                break;
        }
    };

    xhr.open(requestType, url);

    xhr.send();
};
