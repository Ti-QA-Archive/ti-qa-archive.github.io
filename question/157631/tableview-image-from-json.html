tableView image from json

Application type: mobile
Titanium SDK 3.1.2GA
Platform: Android, iOS
Device: Nexus 7, iOS simulator
Host OS: OSX 10.8.5

Hi, i'm trying to get images from a json file and put them in a tableView, the connection is ok because i can get all the others value from the url but i'm not able to show images.

`
var jsonImage = [];
var loader = Ti.Network.createHTTPClient(); 

    loader.open("GET", url);
  loader.onload = function() 
  {

       var json = JSON.parse(this.responseText);

       for (var i = 0; i < json.features.length; i++) 
       {
               var imageUrl = json.features[i].properties.field_foto_rendered;
           jsonImage.push({
            id: i,
            title: imageUrl 
           });     

               var intImage = 0, intImages = jsonImage.length, av_image;

               var post_view = Titanium.UI.createView({ 
                height:'auto', 
                layout:'vertical', 
                top:5, 
                right:5,
                bottom:5, 
                left:5 
       }); 

            for (intImage = 0; intImage < intImages; intImage = intImage + 1) 
                {

                 av_image = Ti.UI.createImageView({  
                    url: jsonImage[intImage].title,
                    left: 0,
                    top: 0
                });      

            post_view.add(av_image);
            Ti.API.info('jsonImage[intImage] ' + jsonImage[intImage].title);

        }
            }
      };

`
