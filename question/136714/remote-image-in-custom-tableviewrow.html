Remote image in custom tableviewRow

Hi all,

I'm having troubles with an remote image in a custom tableviewRow. When I use this row in my tableview, the images aren't at 'top:0', and the image is much smaller than 60. It's centered between top and bottom, but I've set the hight the same as the row.height.

What am I doing wrong?

Regards,
Marc

        var row = Titanium.UI.createTableViewRow({
            height:60,
            width:'auto'
        });

        var view = Titanium.UI.createView({
            height : 60,
            width:'auto',
            top : 0,
            left : 0,
            backgroundColor : '#000000'
        });
        row.add(view);

        var img = Titanium.UI.createImageView({
            left : 0,
            top : 0,
            width : 60,
            height : 60,
            image : 'WORKING URL TO AN JPG IMG'
        });
        row.add(img);
