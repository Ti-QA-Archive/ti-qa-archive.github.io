Animation callback not called

Hi,

i am wondering why the callback of an animation is not called, when the animation is created by Ti.UI.createAnimation. The expected behavior of the code here should be, that "label1 anim finished" and "label2 anim finished" is printed out, but only the second message appears.

var win = Titanium.UI.createWindow({
    title: 'MyWindow'
});

var label1 = Titanium.UI.createLabel({
    backgroundColor: 'red',
    height:30,
    width:100,
    top:10
});

var label2 = Titanium.UI.createLabel({
    backgroundColor: 'blue',
    height:30,
    width:100,
    top:50
});

win.add(label1);
win.add(label2);


var anim = Ti.UI.createAnimation({
    left: Ti.Platform.displayCaps.platformWidth,
    curve: Ti.UI.iOS.ANIMATION_CURVE_EASE_IN_OUT,
    duration: 200
});


label1.animate(anim, function() {
    Ti.API.info('label1 anim finished');
});

label2.animate({left: Ti.Platform.displayCaps.platformWidth, curve: Ti.UI.iOS.ANIMATION_CURVE_EASE_IN_OUT, duration: 200}, function() {
    Ti.API.info('label2 anim finished');
});

win.open();

I am using Titanium mobile 2.0.1 and the iOS Simulator.

Is this a known bug or am i using this wrong?

Thanks, regards Joe.
