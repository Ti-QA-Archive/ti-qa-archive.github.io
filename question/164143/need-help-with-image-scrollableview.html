    Hello Dwayne Brown,

    Thanks for raising this issue. In ScrollableView have an addView method where you can add view. You want to add images in your scrollable view. Image view is one kind of view. Here have an image that takes up most of the screen on a window. Below this image on the window is a Side Scroll View that has many images. The active image when scrolling will display that image in the main image above the scroll view.

    Check the Attachment

    Test Code:

    index.js

    
 var images ;

 function scroll(e) {
   images = e.view.children[0].image ;
   $.imageView.image = images;

 }

 $.container.open();

    index.xml

    <Alloy>
    <Window id="container" >
        <ImageView image="/smile_-_ok-128.png" id ="imageView"/>
        <ScrollableView id="scrollableView" showPagingControl="true" onScrollend="scroll">
            <View id="view1" backgroundColor="#123">
                   <ImageView image="/smile_-_ok-128.png"/>

            </View>
            <View id="view2" backgroundColor="#246">

                 <ImageView image="/handshake_hands_deal_contractors-128.png"/>
             </View>
            <View id="view3" backgroundColor="#48b">

                 <ImageView image="/Untitlwaaaawwed.png"/>
             </View>
        </ScrollableView>
    </Window>
 </Alloy>

    • Hi Dwayne,
      If you have something to add then please add it in the comment of the answer, not in the answer box itself, Well you can try this in your xml

      <Alloy>
    <Window class="container">  
        <Label id="label" onClick="closeView">LabelView To show I am on this window</Label>
     <ImageView id="currentImage"/>
        <ScrollableView id="scrollView" showPagingControl="true" />
    </Window>
</Alloy>

      Now in your tss

      "#scrollView": {
    bottom: 0,
    borderColor: '#fff',
    width: Titanium.Platform.displayCaps.platformWidth,
    height: 100,
},
"#currentImage":{
    top:0,
    bottom:0,
    width:Ti.UI.FILL
}

      Now in you controller i.e .js file

      var images=[[image1,image2,image3,image4,image5],[image1,image2,image3,image4,image5],[image1,image2,image3,image4,image5],[image1,image2,image3,image4,image5]];

for (var i=0,j=images.length;i<j;i==){
var view=Ti.UI.createView({
    height:Ti.UI.FILL,
    width:Ti.UI.FILL,
    backgroundColor:'transparent',
    layout:'horizontal'
    });
$.scrollView.addView(view);

for(var k=0,l=images[i].length;k<l;k++){
var imageView=Ti.UI.createImageView({
    height:Ti.UI.FILL,
    width:'20%',
    left:0,
    image:images[i][j],
    });

imageView.addEventListener('click',function(e){
    $.currentImage.setImage(this.image);
});

view.add(imageView);

}

}
