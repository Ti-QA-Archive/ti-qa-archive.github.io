    • It goes against my better judgement to simply give you the code without anyone presenting the code they have tried. This is purely for the fact that you should have at least tried.

      @Nilesh the link I gave is an example of how to achieve it and not a complete solution.

      Nevertheless, here is an example (not tested on Android):

      //INDEX.XML

<Alloy>
    <Window id="win">
        <ScrollableView id="scrollableView" >
            <View id="view1" class="view">
                <Label>View 1</Label>
            </View>
            <View id="view2" class="view">
                <Label>View 2</Label>
            </View>
            <View id="view3" class="view">
                <Label>View 3</Label>
            </View>
        </ScrollableView>
    </Window>
</Alloy>

//INDEX.TSS

"#scrollableView": {
    backgroundColor: "#fff",
    height: Ti.UI.FILL,
    showPagingControl: "true"
},
"#view1": {
    backgroundColor: "#123"
},
"#view2": {
    backgroundColor: "#246"
},
"#view3": {
    backgroundColor: "#48b"
},
".view": {
    height: Ti.UI.FILL,
    width: Alloy.Globals.pageWidth
}

//INDEX.JS

$.scrollableView.width = Ti.Platform.displayCaps.platformWidth;

function setLayout () {
    var views = $.scrollableView.getViews(),
        currentPageNumber = $.scrollableView.getCurrentPage(),
        prev,
        next;

    var current = views[currentPageNumber];
    current.setLeft(20);
    current.setRight(20);

    if (currentPageNumber !== 0) {
        prev = views[currentPageNumber - 1];
        prev.setLeft(50);
    }
    if ((currentPageNumber + 1) < views.length) {
        next = views[currentPageNumber + 1];
        next.setLeft(-10);
    }
}
setLayout();

$.scrollableView.addEventListener('scroll', function (e) {
    setLayout();
});

$.win.open();

//ALLOY.JS

Alloy.Globals.pageWidth = Ti.Platform.displayCaps.platformWidth - 40;

    • In light of this not actually being marked as answered, I have realised that iOS only has a property called clipViews. If you read the docs about this, it will tell you what it does.

      But in summary, setting this property to false on creation and the width of the scrollableView to something smaller than the device width will allow the views to the left and right to appear.
