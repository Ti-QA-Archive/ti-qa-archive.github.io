  • Create a button or view, set background image for that looks like an drop down menu say menuView, and add new label for selected item, say selectedItem

    Create a view and add a label till item count in list say listItemView..

    On click event of button or view add list into listItemView and show it to bellow menuView.

    And on any label in the list you can set selectedItem to listItemView.label.text and hide listItemView.

      var win = Ti.UI.createWindow({
    title : "test",
    backgroundColor : "white"
});

var img = Ti.UI.createImageView({
    image : "http://png-5.findicons.com/files//icons/2198/dark_glass/128/1downarrow.png",
    height : 20,
    width  :20,
    right : 0
});
win.open();
var selectedItem = Ti.UI.createLabel({
    text : "",
    width : 80,
    height : 35,
    top : 0
});

var viewMenu = Ti.UI.createView({
    borderWidth : 1,
    borderColor : "blue",
    height : 35,
    width : 100,
    top : 50
});
viewMenu.addEventListener('click',function(e){
    viewContainer.show();
});

viewMenu.add(img);
viewMenu.add(selectedItem);

var arr = ["One","Two","Three","One","Two","Three","One","Two","Three","One","Two","Three","One","Two","Three"];

var viewContainer = Ti.UI.createScrollView({
    height : 150,
    width : 100,
    contentHeight : "auto",
    contentWidth : Ti.UI.SIZE,
    top : 51+35,
    visible : false,
    borderColor : "gray",
    layout : "vertical",
});

for (i=0;i<arr.length;i++){
    var item = Ti.UI.createLabel({
        text : arr[i],
        height : 20,
        width : 100
    });
    item.addEventListener('click',function(e){
         viewMenu.children[1].text=this.text;
         viewContainer.hide();
    });
    viewContainer.add(item);
}

win.add(viewMenu);
win.add(viewContainer);

    • Just in case anyone's interested: I used the Gaurang's code above to create a re-usable function that you can add to a UI class (or to a .js screen/window directly). When you call the function it will create the dropdown list, place it in a view and return that view.

      Note that I did this in a few minutes with only a few test runs, so it's not impossible that there may be a bug or so in there.
      However, what I tested (and works) includes:

      • setting the max height of the list (its still scrollable like the original code, but you can make it bigger/smaller if you know how many items you will have, etc.),
      • setting the width of the control,
      • setting the top and left positions,
      • passing in an array of data (strings),
      • changing the box border colour (i.e. the non-dropdown part),
      • changing the list border colour (i.e. the "dropdown" part).
        It will catch any non-supplied arguments and replace them with default values, making it (hopefully) uncrashable by forgetting parameters.
      function CreateDropDownMenu(data, a_width, listHeight, a_top, a_left, a_backgroundColour, a_borderColourBox, a_borderColourList)
{
    if (data == null || data == undefined)
    {
        data = ["no", "data"];
    }
    if (a_width == null || a_width == undefined)
    {
        a_width = 100;
    }
    if (listHeight == null || listHeight == undefined)
    {
        listHeight = 120;
    }
    if (a_top == null || a_top == undefined)
    {
        a_top = 0;
    }
    if (a_left == null || a_left == undefined)
    {
        a_left = 0;
    }
    if (a_backgroundColour == null || a_backgroundColour == undefined)
    {
        a_backgroundColour = "transparent";
    }
    if (a_borderColourBox == null || a_borderColourBox == undefined)
    {
        a_borderColourBox = "#000";
    }
    if (a_borderColourList == null || a_borderColourList == undefined)
    {
        a_borderColourList = "#333";
    }

    var ddView = Ti.UI.createView({backgroundColor: a_backgroundColour, top: a_top, left: a_left});
    ddView.selectedText = "";

    var img = Ti.UI.createImageView({
        image : "/img/downarrow.png", 
        height : 20,
        width  : 20,
        right : 0
    });
    //http://png-5.findicons.com/files//icons/2198/dark_glass/128/1downarrow.png

    var selectedItem = Ti.UI.createLabel({
        text : "",
        width : (a_width - 20),
        height : 35,
        top : 0
    });

    var viewMenu = Ti.UI.createView({
        borderWidth : 1,
        borderColor : a_borderColourBox,
        height : 35,
        width : a_width,
        top : 50
    });
    viewMenu.addEventListener('click',function(e){
        viewContainer.show();
    });

    viewMenu.add(img);
    viewMenu.add(selectedItem);

    var viewContainer = Ti.UI.createScrollView({
        height : listHeight,
        width : a_width,
        contentHeight : "auto",
        contentWidth : Ti.UI.SIZE,
        top : 51+35,
        visible : false,
        borderColor : a_borderColourList,
        layout : "vertical",
    });

    for (i = 0; i < data.length; i++){
        var item = Ti.UI.createLabel({
            text : data[i],
            height : 20,
            width : a_width,
            left : 2
        });
        item.addEventListener('click',function(e){
            viewMenu.children[1].text = this.text;
            ddView.selectedText = this.text;
            viewContainer.hide();
        });
        viewContainer.add(item);
    }

    ddView.add(viewMenu);
    ddView.add(viewContainer);

    return ddView;
}
