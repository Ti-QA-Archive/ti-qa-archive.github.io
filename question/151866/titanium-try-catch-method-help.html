tableview.addEventListener('click',function(e) {

    if(e.index == 0){

        try{

var instaFAILdialog = Ti.UI.createAlertDialog({

    //cancel:0

    • I see. Your are defining your AlertDialog outside of your table-click-listener but adding a new eventlistener to it again and again within the listener.

      Try this:

      tableview.addEventListener('click',function(e) {
    var mediaID = e.rowData.media;
    var instaDialog = Ti.UI.createAlertDialog({
        title:'Obs!',
        message:'You will open the instagram app',
        cancel: 1,
        buttonNames:['Continue','Cancel']
    });

    instaDialog.addEventListener('click',function(e){
        if(e.index == 0){
            if (Ti.Platform.canOpenURL('instagram://media?id='+mediaID)){
              Ti.Platform.openURL('instagram://media?id='+mediaID);
            } else {
              Ti.UI.createAlertDialog({
                title:'Obs again!',
                message:'You need to have the instagram app installed on your phone',
                ok:'OK',
                //cancel:0
              }).show();
            }
        }
    });
    instaDialog.show();

});
