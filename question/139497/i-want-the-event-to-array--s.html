  • One way to do it would be to simply change the videoPlayer's url once the video is done playing.

    Your code would look something like this.

    var videosToPlay = [ 'videoUrl1', 'videoUrl2', 'videoUrl3', 'videoUrl4' ];
var index = 0; // This variable keeps tabs on which video is currently playing.

var videoPlayer = Titanium.Media.createVideoPlayer({
    autoplay : true
    url = videosToPlay[index];
});

videoPlayer.addEventListener('complete', function (e) {
    videoPlayer.url = videosToPlay[index++];
    videoPlayer.play();

    if (index > 4) 
        index = 0;  // Retart looping through the array
});
