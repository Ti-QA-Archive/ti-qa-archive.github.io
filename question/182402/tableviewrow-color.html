    len = arr.length; i < len; i++) {
    var row = Ti.UI.createTableViewRow({
        height : 50,
        width : "100%",
        backgroundColor : (arr[i].team == 1) ? "gray" : "red"
    });

    // Create a Label.
    var aLabel = Ti.UI.createLabel({
        text : arr[i].title,
        color : '#000',
        left : 10,
        right : 10
    });

    // Add to the parent view.
    row.add(aLabel);
    arrRow.push(row);
}

tblList.data = arrRow;
