  • Well finally i get this working:

    var picker = {};

picker.selectedPerson = function(e) {

    var p     = e.person.phone;
    var phone = null;

    if (p.mobile != null)
        phone = p.mobile[0]; 
    else {
        if (p.home != null)
            phone = p.home[0];
        else {
            if (p.work != null)
                phone = p.work[0];
        } 
    }

    phone = phone.replace("-",""); // This doesn't work

};

Ti.App.addEventListener('contactPicker', function(e) {
    Ti.Contacts.showContacts(picker);
});
