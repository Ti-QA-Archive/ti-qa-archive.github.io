    backgroundColor: "#fff"

        visible: true,
        backgroundColor:'transparent',
        separatorColor:'transparent',
        backgroundRepeat: true
    });

musicListView.data=[{title:'test'}];
wrapperView.add(musicListView);
win.add(wrapperView);

win.open();
