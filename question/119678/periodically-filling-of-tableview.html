  • Hi,

    I solved the problem myself with events:

    velo.ui.createTableView = function() {
   function populateData(datas) {
    var mydata = datas.list;
    tv.setData(mydata);
   }
   var tv = Ti.UI.createTableView();
   velo.getListofnextSchrauber();
   setInterval(velo.getListofnextSchrauber,3000);
   Ti.App.addEventListener('new_event',populateData);
   return tv;
};

velo.getListofnextSchrauber = function() {
// logic to get datas ...
   Ti.App.fireEvent('new_event',{list:[item1,item2]});
});
