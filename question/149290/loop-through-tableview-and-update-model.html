Im having trouble figuring out how to update the model could someone take a look and point me in the right direction?

$.btnSave.addEventListener('click', function(e) {
    //loop through the table and save the checked values
        var categoryCollection = Alloy.Collections.Categories;


    var counter=0;
    for(var i=0; i < $.tblCategories.data[0].rows.length; i++){
        var categoryModel  = categoryCollection.get($.tblCategories.data[0].rows[i].model);
        if ($.tblCategories.data[0].rows[i].hasCheck == true){
            categoryModel.set("Selected", 1);
            counter++
        }else{
            categoryModel.set("Selected", 0);
        }

    }
    alert(counter + " were checked")

});
