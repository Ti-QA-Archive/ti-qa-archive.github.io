  • Mathew

    If you don't wish to use a tabGroup, you can achieve your objective as demonstrated in this video. The tested, working code to it is below:

    var win1 = Ti.UI.createWindow({
  title:'About Us',
  backgroundColor: 'green'
});

var launchButton= Ti.UI.createButton({
  title:'Go to appcelerator.com',
  top:'20%',
  width:'64%',
  height:'18%'
});

win1.add(launchButton);

win1.open();


launchButton.addEventListener("click", function(){
  var win2 = Ti.UI.createWindow({
    title:'Our Website',
    backgroundColor: '#eee',
    layout:'vertical',
    fullscreen:false
  });
  var backButton1 = Ti.UI.createButton({
    title:'Back',
    top:'2%',
    left: 10,
    height: '8%',
    width:80
  });
  win2.add(backButton1);

  var webview1 = Ti.UI.createWebView({
    top:'1%',
    height:'89%',
    url:'http://www.appcelerator.com'
  });

  win2.add(webview1);

  var activityIndicator = Ti.UI.createActivityIndicator({
    message:'Loading...',
    bottom:10,
    height:50,
    width:10
  });
  win2.add(activityIndicator);

  win2.open();

  activityIndicator.show();

  webview1.addEventListener('load', function(){
    activityIndicator.hide();
  });

  backButton1.addEventListener('click', function(){
    win2.close();
  });
});

    Cheers
