  • Well, you provide no code so we're left guessing what you are doing wrong. This code works fine

    var w1 = Ti.UI.createWindow();
var w2 = Ti.UI.createWindow();

var b1 = Ti.UI.createButton({
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
    title : 'Open Picker Window'
});

var b2 = Ti.UI.createButton({
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
    title : 'Get Selected Value'
});

var picker = Ti.UI.createPicker({
    top : 50
});

var data = [];
data[0] = Ti.UI.createPickerRow({
    title : 'Bananas'
});
data[1] = Ti.UI.createPickerRow({
    title : 'Strawberries'
});
data[2] = Ti.UI.createPickerRow({
    title : 'Mangos'
});
data[3] = Ti.UI.createPickerRow({
    title : 'Grapes'
});

picker.add(data);
picker.selectionIndicator = true;

w2.add(picker);
w2.add(b2);

b1.addEventListener('click', function() {
    w2.open();
});

b2.addEventListener('click', function() {
    alert(picker.getSelectedRow(0));
});

w1.add(b1);
w1.open();
