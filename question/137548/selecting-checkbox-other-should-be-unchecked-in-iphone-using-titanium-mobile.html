    Hi Juni Gupta,

    Try this code it's similar to your requirement you just need to change image with title which i used.

    var win = Ti.UI.currentWindow;

var t = Ti.UI.createButton({
    title : "1",
    height : 30,
    width : 30,
    top : 50,
    left : 10
});

var t2 = Ti.UI.createButton({
    title : "2",
    height : 30,
    width : 30,
    top : 50,
    left : 50

});

var t3 = Ti.UI.createButton({
    title : "3",
    height : 30,
    width : 30,
    top : 50,
    left : 90
});

t.addEventListener('click', function(e) {
    t.title = 'c';
    t2.title = '2';
    t3.title = '3';

});
t2.addEventListener('click', function(e) {
    t.title = '1';
    t3.title = '3';
    t2.title = 'c';

});
t3.addEventListener('click', function(e) {
    t.title = '1';
    t2.title = '2';
    t3.title = 'c';

});
var resetBtn = Ti.UI.createButton({
    title : "reset",
    height : 30,
    width : 180,
    top : 150
});
resetBtn.addEventListener('click', function(e) {
    t.title = '1';
    t2.title = '2';
    t3.title = '3';

});
win.add(t);
win.add(t2);
win.add(t3);
win.add(resetBtn);

  • hi,

    just try this

    for (var i = 0; i < 10; i++) {
    var star = Ti.UI.createImageView({
        height : 24,
        width : 24,
        left : 5 + (i * 25),
        backgroundColor : 'white',
        isChecked : false
    });

    star.addEventListener('click', function(e) {
        if (e.source.isChecked == false) {
            e.source.backgroundColor = 'red';
            e.source.isChecked = true;
        } else {
            e.source.backgroundColor = 'white';
            e.source.isChecked = false;
        }
    });
    win.add(star);
}
