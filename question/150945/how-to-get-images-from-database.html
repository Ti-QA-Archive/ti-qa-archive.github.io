how to get images from database?

app.js :
Ti.include('database.js');
Ti.UI.setBackgroundColor("#ffffff");
var win1=Ti.UI.createWindow({
backgroundColor:'#fff'
});
var button1=Ti.UI.createButton({
title:'image'
});
var db = Titanium.Database.open('abcdef');
var rows = db.execute('SELECT * FROM test1');
button1.addEventListener('click',function(){
while (rows.isValidRow()) {
var tabGroup=Ti.UI.createTabGroup();
var win2=Ti.UI.createWindow({
url:'window2.js',
backgroundColor:'#fff'
});
var tab1=Ti.UI.createTab({
window:win2,
title:'IMAGE1'
});
var image5 = Ti.UI.createImageView({
image :rows.fieldByName('image1'),
width : 240,
height : 100,
top : 110,
left : 40
});
win2.add(image5);
var win3=Ti.UI.createWindow({
url:'window3.js',
backgroundColor:'#fff'
});
var tab2=Ti.UI.createTab({
window:win3,
title:'IMAGE2'
});
var image6 = Ti.UI.createImageView({
image :rows.fieldByName('image2'),
width : 240,
height : 100,
top : 110,
left : 40
});
win2.add(image5);
win3.add(image6);
tabGroup.addTab(tab1);
tabGroup.addTab(tab2);
tabGroup.open();
//win2.open();
}
rows.close();
});
win1.add(button1);
win1.open();

database.js:

var img1=Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory,'fish.png');
var img2=Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory,'bird.png');
var db=Titanium.Database.open('abcdef');
db.execute('CREATE TABLE IF NOT EXISTS test1 (image1 TEXT,image2 TEXT)');
db.execute('INSERT INTO test1 (image1,image2) VALUES(?,?)',img1.read(),img2.read());
db.close();
