Barcode Scanner Android doesn't work

Hi I have the mwaysolutions barcode module 0.3 and when I have tested it works perfectly, but when I try to add this to a tab of my app, the scanner shows but doesn't call success, error or cancel, here is my code

var winQr=Ti.UI.currentWindow;
var titaniumBarcode = winQr.barcode;//This is require('com.mwaysolutions.barcode')

winQr.backgroundImage='../images/fondo_app.png';
winQr.barImage='../images/barra_superior_portrait.png';

var data=[];
var cont=0;
var btnScanner=Ti.UI.createButton({
    title:'Escanear',
    bottom:0,
    left:0,
    right:0,
    height:'15%'
});
var tableQr=Ti.UI.createTableView({
    data:data,
    left:0,
    right:0,
    top:0,
    height:'90%',
    backgroundColor:'transparent'
});
btnScanner.addEventListener('click',function(){
    titaniumBarcode.scan({
          success: function (data) {
              if(data && data.barcode) {
                var row=Ti.UI.createTableViewRow({
                    backgroundColor:'transparent',
                    title:data.barcode,
                    color:'#000',
                    height:80
                });
                data[cont++]=row;
                tableQr.data=data;
            } else {
                  alert(JSON.stringify(data));
               }
          },
        error: function (err) { 
            alert('error');
            alert(err.error); 
          },
          cancel: function () { 
            alert("cancel"); 
          }
    });
});
winQr.add(tableQr);
winQr.add(btnScanner);
