Database disk image is malformed (code 11)

I had built an iPhone app in native objective-c language, In which I had created a database. Now I am using the same database in my Titanium App. I used Ti.Database.install to use the database. Here is my code

var DATABASE_NAME = 'RevisionApp';

exports.installDB = function() {
Ti.Database.install('/data_source/RevisionApp.sqlite',DATABASE_NAME);
};

exports.fetchQuestionsForSubject = function(subject) {
var lessonList = new Array();
var db = Ti.Database.open(DATABASE_NAME);
Ti.API.info('SELECT FROM ZQUESTIONS WHERE ZMULTICHOICE='+ subject.getId());
var lessonRS = db.execute('SELECT FROM ZQUESTIONS WHERE ZMULTICHOICE=?',subject.getId());
while (lessonRS.isValidRow()) {
var Question = require('/data_source/Question');
var question = new Question();
question.setId(lessonRS.fieldByName('Z_PK'));
question.setQuestion(lessonRS.fieldByName('ZQUESTION'));
question.setOption1(lessonRS.fieldByName('ZOPONE'));
question.setOption2(lessonRS.fieldByName('ZOPTWO'));
question.setOption3(lessonRS.fieldByName('ZOPTHREE'));
question.setOption4(lessonRS.fieldByName('ZOPFOUR'));
question.setAnswer(lessonRS.fieldByName('ZANSWER'));

        lessonList.push(question);
        lessonRS.next();
    }
    lessonRS.close();
    db.close();
    return lessonList;

};

I am getting a Titanium Runtime error Uncaught Error: database disk image is malformed (code 11).
