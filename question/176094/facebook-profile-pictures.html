  • Hi , Nash

    Prefer this code it helps you

    

var api = 'http://graph.facebook.com/' + **USER ID** + '/?fields=picture&type=large';
  var xhr = Ti.Network.createHTTPClient({
   onload : function(e) {

    var data = JSON.parse(this.responseText);

    Ti.App.Properties.setString("ProfilePic", data.picture.data.url);
   },
   onerror : function(e) {
    Ti.API.debug(e.error);
    Ti.API.info('Error');
   },
   timeout : 5000
  });
  xhr.open("GET", api);
  xhr.send();
