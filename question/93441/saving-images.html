  • I use something like this, wrapped in a function.

    var dir = Ti.Filesystem.applicationDataDirectory;
var name = Ti.Utils.md5HexDigest(imageUrl); // make md5 hash of URL
var file = Ti.Filesystem.getFile(dir, name);
if ( file.exists() ) { //  check if we already downloaded it
    imgUrl = file.nativePath;
} else {
    var xhr = Titanium.Network.createHTTPClient();
    xhr.open('GET', imageUrl, false);
    xhr.send();
    if (xhr.readyState == 4 && xhr.status == 200) {
        try {
            var f = Ti.Filesystem.getFile(dir, name);
            f.write(xhr.responseData); 
            imgUrl = f.nativePath;
        } catch(e) {
            // whoops, cant save
        }
    } else {
        // cant get to the image online
    }
}
// use with:
var image = Ti.UI.createImageView({image:imgUrl});
