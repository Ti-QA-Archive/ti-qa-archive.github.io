Orientations of windows

Hi,
my standard view is portrait. From a gallery I want to open the gallery in fullscreen/landscape mode. Here my code:

galery.addEventListener('singletap', function(e) {
    var galwin = Ti.UI.createWindow({
        width : '100%',
        height : '100%',
        navBarHidden : true,
        orientationModes : [Titanium.UI.LANDSCAPE_LEFT, Titanium.UI.LANDSCAPE_RIGHT]
    });
    galwin.addEventListener('close', function() {
        w.orientationModes = [Titanium.UI.PORTRAIT, Ti.UI.UPSIDE_PORTRAIT];
        galwin = null; // kill it!
    });
    var biggalery = Ti.UI.createScrollableView({
        views : images,
        backgroundColor : '#000',
        maxZoomScale : 3.0,
        currentPage : galery.currentPage,
        width : '100%',
        height : '100%',
        top : 0,
        bottom : 0
    });
    galwin.add(biggalery);
    galwin.open({
        modal : true,
        fullscreen : true,
        modalTransitionStyle : Ti.UI.iPhone.MODAL_TRANSITION_STYLE_PARTIAL_CURL
    });
    galwin.addEventListener('singletap', function() {
        galwin.close();
    })
});

But after visiting of big gallery and jumping to parent window then the parent window is in wrong landscape mode.

Any ideas what is wrong?

Rainer
