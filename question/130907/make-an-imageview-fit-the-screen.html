Make an imageView fit the screen

Hi,
I use an imageView like this

var scrollView = Titanium.UI.createScrollView({
  contentWidth:'auto',
  contentHeight:'auto',
  top:0,
  bottom:0,
  showVerticalScrollIndicator:true,
  showHorizontalScrollIndicator:true,
  zoomScale : 1,
  maxZoomScale:2,
  minZoomScale:0.2
});

var planImg = Titanium.UI.createImageView({
    image : '../images/' + mainWin.image,
    left : 0,
    top : 0,
    width : 'auto',
    height : 'auto'
    //top : 2
});

scrollView.add(planImg);
mainWin.add(scrollView);

I would like when the windows open to have the imahge that fit the screen then the user zoom on the part he want.

How can I achieve this ?

Regards

    • Ok, I got it. I have the same behavior in an app, but requires you to know the real size of the image (height and width pixels)

      This is what I do

      
var img = Ti.UI.createImageView({
    image:element.mediaUrl, //url to your image
    top:'auto', left:'auto', right:'auto', bottom:'auto',
    width:element.width, //your image width
    height:element.height, //your image height
});

var scrollImg = Titanium.UI.createScrollView({
        contentWidth:img.width,
        contentHeight:img.height,
        top:0,left:0,right:0,bottom:0,
        maxZoomScale:1.5, //set to whatever you want
        minZoomScale:0.3,
});

scrollImg.add(img);
