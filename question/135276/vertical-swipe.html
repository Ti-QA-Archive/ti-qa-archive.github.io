    • Thanks for the suggestion:

      I created this structure just for the vertical swipe:

                  var swipeUp = function(start, end) {
            var dif = start - end;
            if(dif > 75) { // adjust this to determine the length of the swipe
                return true;
            } else {
                return false;
            }
        }

        var start = null;
        var end = null;

        container.addEventListener('touchstart', function(e1) {
            Ti.API.info('touchstart');
            start = e1.y;
        });

        container.addEventListener('touchend', function(e2) {
            Ti.API.info('touchend');
            end = e2.y;
            if(start) {
                if(swipeUp(start, end)) {
                    Ti.API.info('true swipeup');
                }
                start = null;
            }
        })
