    • Here is an example code for calling webservice function and showing data.
      in below code I have define a 'show_status' webservice function, 'pdfStatus' is root tag
      and 'status' is the tag that we are reading.
      suds.js is available in kitchen sink application.

      Ti.include('suds.js');
var callparams = {
};
var suds = new SudsClient({
    endpoint : 'your_site_url/server.php?wsdl',
    targetNamespace : 'your_site_url'
});
suds.invoke('show_status', callparams, function(xmlDoc) {
    var root_tag = xmlDoc.documentElement.getElementsByTagName('pdfStatus');
    var status = xmlDoc.documentElement.getElementsByTagName('status');
    alert(status.item(0).text);
    }
);
