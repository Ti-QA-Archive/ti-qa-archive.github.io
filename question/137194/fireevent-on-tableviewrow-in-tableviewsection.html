Fireevent on tableViewRow in tableViewSection

Hello, I get confused about how to fire tableViewRow into tableViewSection.

Here I create tableViewSection and tableViewRow inside.

var win = Ti.UI.currentWindow;
var data = [];
var sections = [];
var section1 = Ti.UI.createTableViewSection({
   headerTitle: 'Entries'
});
var row1 = Ti.UI.createTableViewRow({
    title: 'title',
    height: '60dp',
});
section1.add(row1);

var section2 = Ti.UI.createTableViewSection({
    headerTitle: 'entey'
});

Titanium.App.addEventListener('test123', function(){
    var row2 = Ti.UI.createTableViewRow({
        title: 'row2',
        height: '60dp',
    });
    section2.add(row2);
    sections.push(section2);
});

sections.push(section1);



var tableView = Ti.UI.createTableView({
  data:sections,
});

win.add(tableView);

and Row2 is supposed to fired by the button in other window bellow.

var win = Ti.UI.currentWindow;

var btn = Ti.UI.createButton({  
    title:'Insert Information',  
    top: '280dp',  
    width: '150dp',  
    height: '40dp',  
    borderRadius:1,  
    font:{fontFamily:'Arial',fontWeight:'bold',fontSize:14}  
});  

win.add(btn);

btn.addEventListener('click', function(){
    Titanium.App.fireEvent('test123');
});

Please tell me where I get wrong, thanks.

    Whats your issue? In your custom Eventlistener you push the New section into the Array but the tableView wont recognize that. so you need to tableView.setData(sections); again.

    • Thanks Alex,

      Sorry for not to mention, the problem is row2 is not fired. Probably it is not recognized.

      As you mentioned, I changed it like bellow, and it works.

      Titanium.App.addEventListener('test123', function(){
    var row2 = Ti.UI.createTableViewRow({
        title: 'row2',
        height: '60dp',
    });
    section2.add(row2);
    sections.push(section2);
    tableView.setData(sections);
});
