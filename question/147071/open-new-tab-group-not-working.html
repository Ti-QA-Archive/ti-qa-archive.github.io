Open new tab group not working

I'm trying to open a new window with a tab group and I cannot get it to work in iOS. I have a simple window with one button that opens a new window with a tab group in it. The tab group shows up but will not switch tabs - it basically has no response. The code works in Android, but not iOS. Also, I have tried it with SDK 2.13 and 3.0 and it still does not work. Is there something I'm missing with opening new tab groups in iOS?

initial window:

curWin = Ti.UI.currentWindow;
curWin.backgroundColor = 'white';

var btn_openTabs = Ti.UI.createButton({
    title:'Open Tabs',
});

curWin.add(btn_openTabs);

btn_openTabs.addEventListener('click',function(){
    var newTabWindow = Ti.UI.createWindow({
        url:'tabGroup_window.js',
    });
    newTabWindow.open();
});

tabGroup_window.js:

var newTabGroup = Ti.UI.createTabGroup();

var tabWin1 = Ti.UI.createWindow({
    title:'Tab Window 1',
    url:'tabWindow1.js',
});
var tab1 = Ti.UI.createTab({
    title:'Tab 1',
    window:tabWin1,
});
var tabWin2 = Ti.UI.createWindow({
    title:'Tab Window 2',
    url:'tabWindow2.js',
});
var tab2 = Ti.UI.createTab({
    title:'Tab 2',
    window:tabWin2,
});

newTabGroup.addTab(tab1);
newTabGroup.addTab(tab2);

newTabGroup.open();

tabWindow1.js and tabWindow2.js:

curWin = Ti.UI.currentWindow;
curWin.backgroundColor = 'white';

var testLabel = Ti.UI.createLabel({
    text:'Test'
});

curWin.add(testLabel);

    Hey Marlys,

    I made a small modification to your code and this should work: 

    curWin = Ti.UI.currentWindow;
curWin.backgroundColor = 'white';

var btn_openTabs = Ti.UI.createButton({
    title:'Open Tabs',
});

curWin.add(btn_openTabs);

btn_openTabs.addEventListener('click',function(){
    var newTabGroup = Ti.UI.createTabGroup();

    var tabWin1 = Ti.UI.createWindow({
            title:'Tab Window 1',
            url:'tabWindow1.js',
    });
    var tab1 = Ti.UI.createTab({
            title:'Tab 1',
            window:tabWin1,
    });
    var tabWin2 = Ti.UI.createWindow({
            title:'Tab Window 2',
            url:'tabWindow2.js',
    });
    var tab2 = Ti.UI.createTab({
            title:'Tab 2',
            window:tabWin2,
    });

    newTabGroup.addTab(tab1);
    newTabGroup.addTab(tab2);

    newTabGroup.open();
});
