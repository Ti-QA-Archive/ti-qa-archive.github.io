  • Accepted Answer

    there was not problems with your code, just the website… it was checking for User-Agent, and since the user agent was from a mobile device, it appears it was redirecting your request.

    I just put a dummy user agent in and it all works fine.

    var data = [];

var xhr = Ti.Network.createHTTPClient();
xhr.open("GET", "http://www.troyhigh.com/apps/events/events_rss.jsp?id=0");
xhr.setRequestHeader('User-Agent', 'my user agent')

xhr.onload = function() {
    try {
        var doc = this.responseXML.documentElement;
        var items = doc.getElementsByTagName("item");
        var x = 0;
        var doctitle = doc.evaluate("//channel/title/text()").item(0).nodeValue;
        for (var c = 0; c < items.length; c++) {
            var item = items.item(c);
            var title = item.getElementsByTagName("title").item(0).text;
            var url = item.getElementsByTagName("link").item(0).text;
            Ti.API.info("title: " + title);
            Ti.API.info("url: " + url);
        }
    }
    catch(E) {
        alert(E);
    }
};
xhr.send();
