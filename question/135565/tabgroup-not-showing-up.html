TabGroup not showing up

I'm trying to show a tab group when clicking on a button but it's not showing up. This is the code I have right now. Any ideas on how to fix this?

    btnGetStarted.addEventListener("click", function(){
        Ti.include("/windows/UsersWindow.js");
        Ti.include("/windows/FineWindow.js");
        var winUsers = new UsersWindow();
        var winFine = new FineWindow();

        // create tab group
        var tabGroup = Titanium.UI.createTabGroup();

        var tab1 = Titanium.UI.createTab({  
            icon:'../images/icons/users32.png.png',
            title:'Users',
            window:winUsers
        });

        var tab2 = Titanium.UI.createTab({
            icon:'../images/icons/check32.png',
            title:'I\m Fine!',
            window:winFine
        });

        tabGroup.add(tab1);
        tabGroup.add(tab2);
        tabGroup.open();
    });
