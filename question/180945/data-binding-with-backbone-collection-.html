  • Alloy collections are as same as backbone collections since alloy is using backbone for Model and Collection. You can bind the collection as shown in alloy. If you want to bind other data which is not part of collection (Backbone/Alloy). Here is a sample code.

    
var win = Ti.UI.createWindow({backgroundColor: 'white'});
var listView = Ti.UI.createListView();
var sections = [];

var fruitSection = Ti.UI.createListSection({ headerTitle: 'Fruits'});
var fruitDataSet = [
    {properties: { title: 'Apple'}},
    {properties: { title: 'Banana'}},
];
fruitSection.setItems(fruitDataSet);
sections.push(fruitSection);

var vegSection = Ti.UI.createListSection({ headerTitle: 'Vegetables'});
var vegDataSet = [
    {properties: { title: 'Carrots'}},
    {properties: { title: 'Potatoes'}},
];
vegSection.setItems(vegDataSet);
sections.push(vegSection);

listView.sections = sections;
win.add(listView);
win.open();

var fishSection = Ti.UI.createListSection({ headerTitle: 'Fish'});
var fishDataSet = [
    {properties: { title: 'Cod'}},
    {properties: { title: 'Haddock'}},
];
fishSection.setItems(fishDataSet);
listView.appendSection(fishSection);

    You can see the example on appcelerator.

    • You can refer the collection in you controller as shown below,

      index.xml

      
<Alloy>
    <Collection src="index"/>
    <Window title="Window 1" class="container">
        <TableView id="tblEmp" dataCollection="index" dataTransform="formatOutput">
            <TableViewRow empid="{empID}" title="{msg}" />
        </TableView>
    </Window>
</Alloy>

      index.js

      
var emp = Alloy.Collections.instance('index'); // This is singleton collection specify in the view (xml) file
emp.fetch();

function formatOutput(model) {
    // Need to convert the model to a JSON object
    var transform = model.toJSON();
    Ti.API.info("transform ==> " + JSON.stringify(transform));
    transform.msg = transform.Name + " - " + transform.Age + " year.";
    return transform;
}

$.index.open();

      Note - There is another way also to create new instance of collection in controller. Alloy.createCollection(&#39;index&#39;);
