  • Hi Heedoo,

    Try this code…

    
Table.addEventListener('click', function(e) {

    var WinNew = Ti.UI.createWindow({
        title : 'New Window'
    });

    var Ima = Ti.UI.createImageView({
       width:55,
       height:55,
       left:5,
       image : List[e.index]
      });
    WinNew.add(Ima);

    var button = Ti.UI.createButton({
        title : 'Close'
    });
    button.addEventListener('click', function(e) {
        WinNew.close();
    });
    WinNew.leftNavButton = button;
    WinNew.open({
        modal : true
    });
});
