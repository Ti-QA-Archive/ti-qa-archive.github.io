    // platform height/width
var platformWidth = Titanium.Platform.displayCaps.platformWidth;
var platformHeight = Titanium.Platform.displayCaps.platformHeight;

// MODEL
var lastTouchPosition;

// create window
var dragWindow = Ti.UI.createWindow({
    title: 'drag and drop',
    tabBarHidden:true,
});

// create moveable image
var revolver = Titanium.UI.createView({
    backgroundImage: "/images/revolver.png", // image
    top: 190,
    left: 110,
    width: 100,
    height: 100
});

var rotationTransform = Titanium.UI.create2DMatrix();

// create button
var hoverBtn = Ti.UI.createButton({
    title: 'drop',
    height: platformHeight * 0.2,
    width: platformWidth * 0.4,
    center: {
        x: platformWidth * 0.5,
        y: platformHeight * 0.48,
    }
});

// CONTROLLER
revolver.addEventListener('touchstart', function(e){
    var touchPos = {x: e.x, y: e.y};
    lastTouchPosition = revolver.convertPointToView(touchPos, dragWindow);
});

revolver.addEventListener('touchmove', function(e){
    var touchPos = {x: e.x, y: e.y};

    // Move revolver by difference between two touch positions
    revolver.top += newTouchPosition.y - lastTouchPosition.y;

// code for changing btn background when image is over the button
    if((revolver.bottom >= hoverBtn.top && revolver.top <= hoverBtn.bottom) || (revolver.right >= hoverBtn.left && revolver.left <= hoverBtn.right))
        hoverBtn.backgroundSelectedImage = '/images/blood.png'; // image

    lastTouchPosition = newTouchPosition;
});

// VIEW
