    • I think your test code is just not letting Ti process any events.

      Try this and you'll see the principal works. Just copy and paste into a blank app.js

      (function() {

    // Load the backgrounding module
    //require('bg');
    var stopService = false;

    var self = Titanium.UI.createWindow({
        title: 'Window',
        backgroundColor: 'white',
        layout: 'vertical',
        });

    var runServiceButton = Ti.UI.createButton({
        title:'run Background service',
        top: 10
    });

    var stopServiceButton = Ti.UI.createButton({
        title:'stop Background service',
        top: 60
    }); 

    self.add(runServiceButton);
    self.add(stopServiceButton);

    runServiceButton.addEventListener('click', function(e) {
         Ti.App.fireEvent("app:service.run");
         Ti.API.info('line after run service event fire');
    });

    stopServiceButton.addEventListener('click', function(e) {
         Ti.API.info('stopService clicked');
         Ti.App.fireEvent("app:service.stop");
    });

    function runService() {
        if (stopService==true) return;

        Ti.API.info('service run');
        setTimeout(function() { runService(); },500);
    }

    Ti.App.addEventListener('app:service.run', function(e) {
        runService();
    });

    Ti.App.addEventListener('app:service.stop', function(e) {
        Ti.API.info("Stop service event fired");
        stopService = true;
    });

    self.open();
})();
