    Okey,

    Then replace this code

    Ti.App.myCustomTabBar = new CustomTabBar({
    tabBar : tabGroup,
    imagePath : 'img/',
    width : 80,
    height : 40,
    items : [{
        image : 'home.png',
        selected : 'home_over.png'
    }, {
        image : 'cloud.png',
        selected : 'cloud_over.png'
    }, {
        image : 'cart.png',
        selected : 'cart_over.png'
    }, {
        image : 'settings.png',
        selected : 'settings_over.png'
    }]
});

    and put this line in your back button click event.

    Ti.App.myCustomTabBar.hide();
