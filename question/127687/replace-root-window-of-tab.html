Replace root window of tab

Hi,

I try to implement a single context solution with tabs using CommonJS scheme.

Currently the code looks like that: 

var globals = {};

var page1 = require('lib/page2'); 
var page2 = require('lib/page2'); 
var page3 = require('lib/page3'); 

globals.tabs = Ti.UI.createTabGroup();

var tab1 = Ti.UI.createTab({  
    icon: 'tab1.png',
    title: 'page1',
    window: new page1.win()
});

var tab2 = Ti.UI.createTab({  
    icon: 'tab2.png',
    title: 'page2',
    window: new page2.win()
});

var tab3 = Ti.UI.createTab({  
    icon: 'tab3.png',
    title: 'page3',
    window: new page3.win()
});

globals.tabs.addTab(tab1);
globals.tabs.addTab(tab2);
globals.tabs.addTab(tab3);

globals.tabs.open();

To avoid huge overhead on startup, I try to add tabs with "blank" windows and load the required modules only on demand.

But I seems that a default root window can't be overwritten, after the tab was added to the tabgroup.

Can anyone advice a good solution how to load modules on request and not on startup?
