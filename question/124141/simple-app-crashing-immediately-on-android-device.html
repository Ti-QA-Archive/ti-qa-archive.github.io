Along with some other goodies. There's nothing to the app but a few image files, so I'll post app.js here as well. It's very basic.

// this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');

// create tab group

var main = Titanium.UI.createWindow({
    title:'Main',
    opacity:0.75,
    backgroundImage:'images/wallpaper.png'
});

var face_label = Titanium.UI.createLabel({
    text:':|',
    color:'#11420a',
    font:{fontSize:145,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto',
    top:25
});

var button_happy = Titanium.UI.createButton({
    title:':)',
    color:'#11420a',
    width:105,
    height:45,
    backgroundImage:'images/button-up.png',
    left:45,
    top:275
});

button_happy.addEventListener('click', function(e){
    if (':)' != face_label.text) {
        face_label.animate({opacity:0, duration:250}, function() {
            face_label.text = ':)';
            face_label.animate({opacity:1, duration:250});
        });
    }
});

button_happy.addEventListener('touchstart', function(e) {
    button_happy.backgroundImage = 'images/button-down.png';
});

button_happy.addEventListener('touchend', function(e) {
    button_happy.backgroundImage = 'images/button-up.png';
});

var button_sad = Titanium.UI.createButton({
    title:':(',
    color:'#11420a',
    width:105,
    height:45,
    right:45,
    backgroundImage:'images/button-up.png',
    top:275
});

button_sad.addEventListener('click', function(e){
    if (':(' != face_label.text) {
        face_label.animate({opacity:0, duration:250}, function() {
            face_label.text = ':(';
            face_label.animate({opacity:1, duration:250});
        });
    }
});

button_sad.addEventListener('touchstart', function(e) {
    button_sad.backgroundImage = 'images/button-down.png';
});

button_sad.addEventListener('touchend', function(e) {
    button_sad.backgroundImage = 'images/button-up.png';
});

main.add(face_label);
main.add(button_happy);
main.add(button_sad);

main.open();

Could someone point out what might be the problem here? I've tried it on my brother's Android phone (the original Verizon Droid), and I'm met with the same results. Emulator shows no signs of issue.
