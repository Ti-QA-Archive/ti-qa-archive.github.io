  • I can provide an example that was used in an app in js. It also contains labels and buttons built as well.

    var homeurl="";
    homeurl=Alloy.Globals.homehosturl;
    fullScreen = "true";

    userinfo=Alloy.Globals.username;

    Ti.API.info ('userinfo / username from alloy globals = ' + userinfo);

    var win = Ti.UI.createWindow({
    navBarHidden : "true",
    backgroundColor: '#6de',
    exitOnClose: true,
    fullscreen: true,
    title: 'Current Patients'
    });

    var usern = Ti.UI.createLabel ({
    top: "5%",
    left : "80%",
    text : "User : ",
    font: {fontSize : "14", fontWeight: "bold"},
    color :"black"
    });
    win.add(usern);

    var uinfo = Ti.UI.createLabel ({
    top: "5%",
    left : "88%",
    text : userinfo,
    font: {fontSize : "14", fontWeight: "bold"},
    color :"black"
    });
    win.add(uinfo);

    var top = Ti.UI.createLabel ({
    top: "0%",
    text : "OB Reporter Application",
    font: {fontSize : "26", fontWeight: "bold"},
    color :"black"
    });
    win.add(top);

    var top2 = Ti.UI.createLabel ({
    top: "4%",
    text : "Current Patients",
    font: {fontSize : "20", fontWeight: "bold"},
    color :"black"
    });
    win.add(top2);

    var ptmedrec = Ti.UI.createLabel ({
    top: "10%",
    left: "3%",
    text : "Medrec# :",
    font: {fontSize : "16", fontWeight: "bold"},
    color :"black"
    });

    win.add(ptmedrec);

    var back = Ti.UI.createButton({
    top: "0%",
    left : "1%",
    height: "7%",
    width: "12%",
    textColor: "black",
    textAlign : "center",
    font : { fontSize : "12%", fontWeight: "bold"}

    });
    var ptlname = Ti.UI.createLabel ({
    top: "10%",
    left: "40%",
    text : "Patients Name :",
    font: {fontSize : "16", fontWeight: "bold"},
    color :"black"
    });
    win.add(ptlname);

    var ptdob = Ti.UI.createLabel ({
    top: "10%",
    left: "73%",
    text : "Patients DOB : ",
    font: {fontSize : "16", fontWeight: "bold"},
    color :"black"
    });
    win.add(ptdob);

    var xhr = Titanium.Network.createHTTPClient();
    xhr.onload = function() {

    var td1 = JSON.parse(this.responseText);

    var defaultFontSize = Ti.Platform.name === 'android' ? 16 : 14;

    var tableData = [];

    for (var i=0; i<td1.length; i++){
    var row = Ti.UI.createTableViewRow({
    className:'forumEvent', // used to improve table performance

    rowIndex:i, // custom property, useful for determining the row during events
height:50, // your height values for the individual labels cannot exceed this number  nor the top values

// this is where you want to put any lables and / or buttons... 
// this will allow each table row to have the correct variables for medrec etc.

    });
    Ti.API.info('td1 id = ' + td1[i].ptmedrec);

    row.rowidnum = td1[i].ptmedrec;
    row.ptlname = td1[i].ptlname;
    row.ptfname = td1[i].ptfname;
    row.ptdbirth = td1[i].ptdbirth;
    Ti.API.info('row.rowidnum = ' + row.rowidnum);

    var labelMedrec = Ti.UI.createLabel({
    font:{fontFamily:'Arial', fontSize:"12", fontWeight:'bold'},
    text: 'Medrec#: ' + td1[i].ptmedrec,
    top: "18%",
    left: "3%",
    width: "15%",
    // height: "2%",
    font: {fontSize : "16", fontWeight: "bold"},
    color:"black"
    });
    row.add(labelMedrec);

    var labelFirstName = Ti.UI.createLabel({

    font:{fontFamily:'Arial', fontSize:"12", fontWeight:'bold'},
text: td1[i].ptfname,
 top: "18%",
 left: "35%", 
 width: "22%",
// height: "2%",
font: {fontSize : "16", fontWeight: "bold"},
 color: "black"

    });
    row.add(labelFirstName);

    Ti.API.info ('td1[i].ptfname results = ' + td1[i].ptfname);
    Ti.API.info ('table label for first name = ' + labelFirstName.text); // run that code so that we can see if its making its way into the label

    var labelLastName = Ti.UI.createLabel({

    font:{fontFamily:'Arial', fontSize:"12", fontWeight:'bold'},
text: td1[i].ptlname,
top: "18%",
left: "55%", 
width: "22%",
// height: "2%",
font: {fontSize : "16", fontWeight: "bold"},
color: "black"

    });
    row.add(labelLastName);

    var labelBirthDate = Ti.UI.createLabel({
    // color:'#6de',
    font:{fontFamily:'Arial', fontSize:"12", fontWeight:'bold'},
    text:td1[i].ptdbirth,
    top: "18%",
    left: "75%",
    width: "22%",
    // height: "2%",
    font: {fontSize : "16", fontWeight: "bold"},
    color: "black"
    });
    row.add(labelBirthDate);

    tableData.push(row);

    function selectRow(e) {
    var rowid = e.row.rowidnum; // medrec
    var ptlname = e.row.ptlname;
    var ptfname = e.row.ptfname;
    var ptdbirth = e.row.ptdbirth;

    Alloy.Globals.medrec = rowid;
// this is the function that causes the row to be clicked .... this is where we assign a global 
// variable to medrec so that we can grab it on the next controller for the working pt 
// and this grabs the variable for the global from line 184. so what ever row they click on
// gets the right medrec number .... watch the console  medrec is a variable

Ti.API.info('row clicked medrec = ' + Alloy.Globals.medrec);



 var add = Alloy.createController ('reports').getView('add');
win.add(add);

    };

    }

    var tableView = Ti.UI.createTableView({
    backgroundColor:'#6de',
    // backgroundColor: 'white',
    top:"15%",
    // bottom: "20%",
    height : "70%",
    borderRadius: "5",
    data:tableData
    });
    tableView.addEventListener('click', selectRow);

    win.add(tableView);
    win.open();

    };
    xhr.open("GET", homeurl+"/demo/student/site/mobile/mobilecurrentpt.php");
    xhr.send();
