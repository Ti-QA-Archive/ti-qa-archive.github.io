ImageView wont follow 'top:0' configuration

I have no idea why the logo wont stick to the top!!!!
I think i am not seeing something really simple.

var currentWin = Titanium.UI.currentWindow;
var logo = Ti.UI.createImageView({
    image:'http://www.neonhusky.com/Images/boinc_logo.png',
    width:320,
    left:0,
    top:0,
});
currentWin.add(logo);

var username = Titanium.UI.createTextField({
    color:'#336699',
    top:80,
    left:10,
    width:300,
    height:40,
    hintText:'Username',
    keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
    returnKeyType:Titanium.UI.RETURNKEY_DEFAULT,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});

currentWin.add(username);

var password = Titanium.UI.createTextField({
    color:'#336699',
    top:username.top+username.height+10,
    left:10,
    width:300,
    height:40,
    hintText:'Password',
    passwordMask:true,
    keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
    returnKeyType:Titanium.UI.RETURNKEY_DEFAULT,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});
currentWin.add(password);

var loginBtn = Titanium.UI.createButton({
    title:'Login',
    top:password.top+password.height+10,
    width:90,
    height:35,
    borderRadius:1,
    font:{fontFamily:'Arial',fontWeight:'bold',fontSize:14}
});
currentWin.add(loginBtn);

    Hello,

    I have to say the "top" of the ImageView is correct.

    The problem that you did not put the "height".

    Without this, ImageView occupies the entire screen and image inside it, is centralized.
