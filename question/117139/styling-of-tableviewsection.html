Styling of TableViewSection

I've made a TableView where I add 2 sections separately, but I want to style the sections and it won't let me.

I've tried the documentet way and tried the solution in this question:
http://developer.appcelerator.com/question/80241/styling-tableviewsection

but neither works.

My code:

var defaultColor = "#000000";

//Create root window
var window = Titanium.UI.createWindow({
    backgroundColor:'#999'
});

//Create the tableview
var table = Ti.UI.createTableView({
    backgroundColor:"white",
    top:20,
     width:320
});

//Data for the tableview
var data = new Array();

//Create section1
var section1 = Ti.UI.createTableViewSection({
});

//Create headerview for section1
var headerView = Titanium.UI.createView({
    backgroundImage:'label_header.jpg',
    width:320,
    height:20
});

//Create headerlabel for headerview            
var headerLabel = Titanium.UI.createLabel({
    text: "This week",
    width:315,
    height:20,
    color:"#ffffff",
    font: {
        fontFamily:"Arial",
        fontSize:14,
        fontWeight:'bold'
    }

});

//add headerlabel to headerview
headerView.add(headerLabel);

//add headerview to section1            
section1.headerView = headerView;


//Make rows for section1
for(var i=0; i<18; i++){
    var row = Titanium.UI.createTableViewRow({
        title:"Event " + i
    });

section1.add(row);
}

//Create section2
var section2 = Ti.UI.createTableViewSection({
    headerTitle: "Next week"
});

//Make rows for section2
for(var i=0; i<18; i++){
    var row = Titanium.UI.createTableViewRow({
        title:"Event " + i
    });

section2.add(row);
}

//Push data to array
data.push(section1);
data.push(section2);

//Set table data
table.data = data;

//Add table to window and open it
window.add(table);
window.open();

Where do I go wrong ?

Edit: It needs to work on both iPhone and Android
