function parseData(d){

    var events = d.data;

    Ti.API.info('====================================');    
    Ti.API.info('Events count : '+events.length);
    var e;
    for(e in events){
        if(e){
            Ti.API.info('name : '+events[e].name);
            Ti.API.info('location : '+events[e].location);
            Ti.API.info('start_time : '+events[e].start_time);
            Ti.API.info('end_tme : '+events[e].end_time);
            Ti.API.info('--------------------------------------------');

        }
    }


}


function getData(){

    Titanium.Facebook.requestWithGraphPath('blahblah/events', {}, 'GET', function(e) {
        if (e.success) {

            //Ti.API.info(e.result);
            parseData(JSON.parse(e.result));

        } else if (e.error) {
            alert(e.error);
        } else {
            alert('Unknown response');
        }
    });
}
