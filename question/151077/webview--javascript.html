    just try this one

    webview.evalJS('call()');

    • because you have not set webview's height and width so it is added over the button just set webview's height and width..

      var win = Titanium.UI.createWindow
({
    backgroundColor:'#fff',
    title:'Window'
});

var webview = Titanium.UI.createWebView
({
    top : 0,
    width : 320,
    height : 400,
    url:'index.html'
});

var button = Titanium.UI.createButton
({
    width:120,
    height:50,
    bottom : 10,
    title:'Test'
});

button.addEventListener('click',function()
{
    webview.evalJS('call()');
});

win.add(button);
win.add(webview);
win.open();
