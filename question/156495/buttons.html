    Hi Uche,

    Please have a look on the following example of creating buttons and adding into a window and linking both to two different windows:

    var w = Ti.UI.createWindow({
    backgroundColor : '#cecece',
    title : 'added by click from label1',
    layout : 'vertical'
})
var button = Ti.UI.createButton({
    title : 'Click to Open modal'
});
button.addEventListener('click', function() {
    var wm = Ti.UI.createWindow({
        modal : true,
        navBarHidden : true,
        tabBarHidden : true,
        backgroundColor : 'red',
        title : 'Modal window one',
        layout : 'vertical'
    });
    wm.open();

});
w.add(button);

var button2 = Ti.UI.createButton({
    title : 'Click to Open regular window'
});
button2.addEventListener('click', function() {
    var wm = Ti.UI.createWindow({
        modal : true,
        navBarHidden : true,
        tabBarHidden : true,
        backgroundColor : 'blue',
        title : 'Modal window two',
        layout : 'vertical'
    });
    wm.open();

});
w.add(button2);

w.open();

    For more examples you can go through the Appcelerator Kitchesink and then try by yourself .
