  • AndroidManifest.xml is created based on tiapp.xml. So you should put your manifest elements there.

    <?xml version="1.0" encoding="UTF-8"?>
<ti:app xmlns:ti="http://ti.appcelerator.org">
    <property name="acs-push-type-development" type="string">gcm</property>
    <property name="acs-push-type-production" type="string">gcm</property>
    <property name="acs-push-type" type="string">gcm</property>
    <property name="ti.android.shake.factor" type="double">1.2</property>
    <property name="ti.android.shake.quiet.milliseconds" type="int">150</property>
    <property name="ti.android.shake.active.milliseconds" type="int">200</property>
    <id>com.galgomedia.planvitalapp</id>
    <name>Plan Vital</name>
    <version>1.0</version>
    <publisher>anfaguerrero</publisher>
    <url>http://www.galgomedia.com</url>
    <description>App Cotizador Plan Vital</description>
    <copyright>2014 by Galgomedia</copyright>
    <icon>appicon.png</icon>
    <persistent-wifi>false</persistent-wifi>
    <prerendered-icon>false</prerendered-icon>
    <statusbar-style>default.png</statusbar-style>
    <statusbar-hidden>true</statusbar-hidden>

    <ios>
        <plist>
            <dict>
                <key>UISupportedInterfaceOrientations~iphone</key>

                    <string>UIInterfaceOrientationPortrait</string>
