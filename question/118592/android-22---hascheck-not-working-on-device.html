         e.rowData.rightImage = 'images/blank.png' ;
      }
});

  • I found that this was a bug in the android implementation of titanium tableview. The problem is that e.rowData is a COPY of the row, not the ACTUAL row.

    The work-around is easy. Add a rowIndex property to each TableViewRow and use that to update the ACTUAL data array like this:

            var data = [];

        var i = 0;
        selections.forEach(function(selection){
        selection.rowIndex = i;
            data.push(Ti.UI.createTableViewRow(selection));
            i++;
        });

        var tableView = Ti.UI.createTableView({ data: data }); 


        tableView.addEventListener('click', function (e) {
            // toggle state of row checkmark
            //NOTE:  on android, e.rowData is a COPY of the row, NOT THE ACTUAL ROW
            var row = data[e.rowData.rowIndex];
            row.hasCheck = !row.hasCheck;
    }

    Now the checkmarks will ACTUALLY APPEAR and function correctly.

    You can look at the values in data[] to see what was selected.
