    • Or prefer this CODE

      It working for me.

      var win = Ti.UI.createWindow({
    backgroundColor:'#dfdfdf'
});


var image = Ti.UI.createImageView({
    image:'KS_nav_views.png',
    width:100,
    height:100,
    top:190
});

image.addEventListener('pinch', function(e) {
    var t = Ti.UI.create2DMatrix().scale(e.scale);
    image.transform = t;
});
win.add(image);


win.open();
