  • try this

        myslider.addEventListener('change', function(e) {

        // need to get the rounded value
        var rndValue = Math.round(myslider.value);

        if(rndValue == 3) {
            myslider.backgroundColor = 'green'
        } else if(rndValue == 2) {
            myslider.backgroundColor = 'red'
        } else {
            myslider.backgroundColor = 'orange'
        }

    });
