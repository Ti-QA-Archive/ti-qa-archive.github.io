  • I found out how to get the App Access Token. For all who need it also, here is the description:

    Before that you need to register a facebook app in the developer section of facebook. Then take your app id and your app secret key and follow the description shown on this link: http://developers.facebook.com/docs/opengraph/using-app-tokens

    With this access token it is possible to read public page feeds and use it in your mobile apps.

    • // first read the token from facebook
// you need to make a facebook app in the facebook developer section
// then you have to use the app ID and app secret to get your token
var xhr = Titanium.Network.createHTTPClient();
xhr.onload = function() {
   var TokenResponse = this.responseText;
   var TokenArray = TokenResponse.split("=");
   var Token = TokenArray[1];
   if (Token !="")
   {
        // with your token you can read public feeds   
    getFeeds(Token);
   }    

};

xhr.open("GET","https://graph.facebook.com/oauth/access_token?client_id=YOUR_FACEBOOK_APP_ID&client_secret=YOUR_FACEBOOK_APP_SECRET&grant_type=client_credentials");
xhr.send();




function getFeeds(Token)
{
    Ti.Facebook.requestWithGraphPath('THE_ID_OF_YOUR_PUBLIC_PAGE/feed', {access_token: Token}, 'GET', function(e){
        if (e.success) {
            alert("Success!  From FB: " + e.result);
            var resultText = e.result;
            var s = JSON.parse(resultText);
            var data = [];

            var header = Ti.UI.createTableViewRow({
                        height: 30,
                        title: "Feeds: " + s.data.length

            });
            data.push(header);
            for (var c=0;c<s.data.length;c++) 
            {
                   var row = Ti.UI.createTableViewRow({
                        height: 30,
                        title: s.data[c].message

                   });

                    data.push(row);
            }

            var tableview = Titanium.UI.createTableView();
            tableview.setData(data);
            win1.add(tableview);
        } else {
            if (e.error) {
                alert(e.error);
            } else {
                alert("Unkown result");
            }
        }
    });

}
