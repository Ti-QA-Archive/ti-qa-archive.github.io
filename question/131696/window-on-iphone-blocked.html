var winVideos = Ti.UI.currentWindow;


//source articles
var videos = [];
//view videos
var viewVideos = Ti.UI.createTableView({
    data: videos,
    separatorStyle:0,
    top : 1, bottom: 50
});

var act = Ti.UI.createActivityIndicator({
    height:250,
    width:'auto',
    message : 'Chargement...',
    style : Titanium.UI.iPhone.ActivityIndicatorStyle.DARK
});

function loadVideos(){
    var titreVideo, imageVideo, urlVideo, updateVideo;
    var client = Ti.Network.createHTTPClient();

    client.onerror = function(e) {
        alert("Connnexion impossible !!");
        act.hide();    
    };

    client.onload = function(e) {
        var doc = this.responseXML.documentElement;
        var elements = doc.getElementsByTagName("video");
        videos = [];
        for(var i=0;i<elements.length;i++)
        {
            titreVideo = elements.item(i).getElementsByTagName("titre").item(0).text;
            imageVideo = elements.item(i).getElementsByTagName("image").item(0).text;
            urlVideo = elements.item(i).getElementsByTagName("url").item(0).text;
            updateVideo = elements.item(i).getElementsByTagName("pubDate").item(0).text;

            var row = Ti.UI.createTableViewRow({
                backgroundImage: "middleRow.png",
                selectedBackgroundImage: "middleRowSelected.png",
                height : '80'
            });

            var titreVideoView = Ti.UI.createLabel({
                text : titreVideo,
                color : '#000',
                height : 'auto',
                left : 115, right: 8,
                font:{fontSize:13},
                touchEnabled : false
            });
            row.add(titreVideoView);    

            var imageVideoView = Ti.UI.createImageView({
                defaultImage: "default.jpg",
                image : imageVideo,
                left: 8,
                width : 100,
                height : 64,
                touchEnabled : false
            });
            row.add(imageVideoView);

            row.thisVideo = urlVideo;
            row.thisTitle = titreVideo;
            row.thisUpdate = updateVideo;

            videos.push(row);    
        }
        viewVideos.setData(videos);
        act.hide();
    };
    client.open('GET','http://feeds...');
    client.send();
}

//button refresh
