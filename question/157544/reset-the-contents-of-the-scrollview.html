  • Hi,

    try this demo.

    var win = Ti.UI.createWindow();
var scrollView = Ti.UI.createScrollView({
    top : 0,
    height : Ti.UI.FILL,
    width : Ti.UI.FILL,
    layout : 'vertical'
});

for (var i = 0; i < 5; i++) {
    var view = Ti.UI.createView({
        width : 300,
        height : 250,
        backgroundColor : 'gray'
    });

    var lbl = Ti.UI.createLabel({
        text : '>> ' + (i + 1),
        width : Ti.UI.SIZE,
        height : Ti.UI.SIZE
    });

    view.add(lbl);

    view.addEventListener('click', function(e) {
        scrollView.remove(e.source);
    });

    scrollView.add(view);
}

win.add(scrollView);

win.open();
