Problem navbar on grouptab

Hi. I can't view the title of my app on the phone. What I can do?
Image: http://imageshack.com/a/img802/9747/sfkp.png

var tabGroup = Titanium.UI.createTabGroup({
    title:'Calcolo percentuale'
});

var win1= Ti.UI.createWindow({
   backgroundColor:'white'
});

var tab1 = Titanium.UI.createTab({
    title: 'Calcola sconto',
    window: win1
});

var win2= Ti.UI.createWindow({
   backgroundColor:'white'
});

var tab2 = Titanium.UI.createTab({
    title: 'Calcola percentuale',
    window: win2
});
tabGroup.addTab(tab1);
tabGroup.addTab(tab2);

tabGroup.open();

  • TabGroup doesn't have a property of 'title' to pass.

    You should set the title's that you want on your Window's when you create them.

    — answered February 13th 2014 by Luke Taylor
    permalink
    2 Comments

    • Try adding this to your code:

      if (Ti.Platform.osname === 'android') {
    tabGroup.addEventListener("open", function() {        
        if (!tabGroup.activity) {
            Ti.API.error("Can't access action bar on a lightweight window.");
        } else {
            var actionBar = tabGroup.activity.actionBar;
            if (actionBar) {
                actionBar.title = "Calcolo percentuale";
            }
        }
    });
}
