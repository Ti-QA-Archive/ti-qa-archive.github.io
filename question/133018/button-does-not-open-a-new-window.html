Button does not open a new window

Hi guys I'm new to Titanium and I'm facing a big problem. I want to open a new window from the click event of a button. here is the code:

var newWin = Titanium.UI.createWindow({
url : '/scripts/telefonos.js',

});

var win = Ti.UI.createWindow({
backgroundColor : 'white',
navBarHidden : true,
title : 'dashboard',
layout : 'vertical',
modal : true,
});

var win1 = Ti.UI.createView({
top : 0,
left : 0,
width : windowWidth,

});

var item = Titanium.UI.createButton({
