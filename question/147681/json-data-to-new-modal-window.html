    Nick make a common js module and send the parameters in the form of function like this.This first is Home Screen

    var DetailScreen = require('DetailScreen');
var db = require('/db');
//db.titlefun();
function HomeScreen() {
    var Stories = new Array();
    var NewWords = new Array();
    var Exp = new Array();
    var AudioStory = new Array();
    var myfontsize = (Titanium.Platform.displayCaps.platformHeight * 4) / 100;
    var self = Ti.UI.createWindow({
        backgroundColor : 'White'
    });
    var JsonTable = Ti.UI.createTableView();
    var tableData = [];
    var Json, Story
    var xhr = Ti.Network.createHTTPClient({
        onload : function() {
            //To show it

            // Ti.API.debug(this.responseText);
            json = JSON.parse(this.responseText);
            for (var i = 0; i < json.nodes.length; i++) {
                Story = json.nodes[i].node;

                Stories[i] = Story.body;
                NewWords[i] = Story.field_new_words;
                Exp[i] = Story.field_explanation;
                AudioStory[i]=Story.field_audio_story;
                var row = Ti.UI.createTableViewRow({
                    height : '85dp'
                });
                var StoryImage = Ti.UI.createImageView({
                    image : Story.field_image, //
                    top : '2%',
                    left : '2%',
                    width : '20%',
                    height : '96%'
                });
                var labTitle = Ti.UI.createLabel({
                    text : Story.title,
                    font : {
                        fontSize : myfontsize,
                        fontWeight : 'bold'
                    },
                    height : '35%',
                    left : '25%',
                    top : '2%',
                    color : 'black',
                });
                var labCategory = Ti.UI.createLabel({
                    text : Story.field_tags,
                    font : {

                        fontSize : myfontsize * 0.7
                    },
                    height : '35%',
                    left : '25%',
                    top : '40%',
                    color : '#000',
                });
                var labSubscription = Ti.UI.createLabel({
                    text : Story.field_subscription,
                    font : {
                        fontSize : myfontsize * 0.6
                    },
                    height : '25%',
                    left : '25%',
                    bottom : '2%',
                    color : 'black',
                });
                row.add(labTitle);
                row.add(labCategory);
                row.add(labSubscription);
                row.add(StoryImage);
                tableData.push(row);

            }

            JsonTable.setData(tableData);

        },
        onerror : function(e) {
            Ti.API.debug("STATUS: " + this.status);
            Ti.API.debug("TEXT: " + this.responseText);
            Ti.API.debug("ERROR: " + e.error);
            alert('There was an error retrieving the remote data. Try again.');
        },
        timeout : 5000
    });
    JsonTable.addEventListener('click', function(e) {
        var index = e.index;
        new DetailScreen(Stories[index], NewWords[index], Exp[index],AudioStory[index]).open({
            modal : true
        });
    });
    xhr.open("GET", "http://dailyting.com/daily-ting/");
    xhr.send();
    self.add(JsonTable);
    return self;
}

module.exports = HomeScreen;

    //var AudioPlayer = require('AudioPlayer')
function DetailScreen(Stories, NewWords, Exp,AudioStory) {
    var self = Ti.UI.createWindow({
        backgroundColor : 'white'
    });
//    new AudioPlayer(self,AudioStory).open({
//        modal:true
//    });
    var scroll = Titanium.UI.createScrollView({
        contentWidth : Ti.UI.SIZE,
        contentHeight : Ti.UI.SIZE,
        top : '15%',
        layout : 'vertical',
        showVerticalScrollIndicator : true,
        showHorizontalScrollIndicator : false
    });
    self.add(scroll);
    var myfontsize = (Titanium.Platform.displayCaps.platformHeight * 4) / 100;
    var labStory = Ti.UI.createLabel({
        text : Stories,
        font : {
            fontSize : myfontsize,
            fontWeight : 'bold'
        },
        height : Ti.UI.SIZE,
        left : '2%',
        top : Ti.UI.SIZE,
        color : 'black',
    });
    scroll.add(labStory);
    var labNewWords = Ti.UI.createLabel({
        text : NewWords,
        font : {
            fontSize : myfontsize * 0.7
        },
        top : Ti.UI.SIZE,
        height : Ti.UI.SIZE,
        left : '2%',
        //  top: '12%',
        color : 'black',
    });
    scroll.add(labNewWords)
    var labExp = Ti.UI.createLabel({
        text : Exp,
        font : {
            fontSize : myfontsize * 0.6
        },
        top : Ti.UI.SIZE,
        height : Ti.UI.SIZE,
        left : '2%',
        //  bottom: '2%',
        color : 'black'
    });
    scroll.add(labExp)
    return self;
}

module.exports = DetailScreen;
