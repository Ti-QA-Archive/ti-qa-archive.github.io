    Please try this (for android only).

    // Get real px from dp
var convertDpToPx = function(dp) {

    return dp * (Ti.Platform.displayCaps.dpi / 160);
};
exports.convertDpToPx = convertDpToPx;

// Get dp from real px
var convertPxToDp = function(px) {

    return px / (Ti.Platform.displayCaps.dpi / 160);
};
exports.convertPxToDp = convertPxToDp;
