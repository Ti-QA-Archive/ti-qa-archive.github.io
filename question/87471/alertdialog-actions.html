alertDialog actions

alertDialog takes an array of button names during instantiation. I was wondering if it's possible to choose different actions based on which button is pressed during the dialog. For example, 'OK' would close it, but 'Help' would open another view. Is this possible?

Or is there a similiar functionality that can be implemented using another technique?

var alertWindow = Titanium.UI.createAlertDialog({
    title: 'Titanium',
    message: 'Simple Alert Message',
    cancel:1,
    buttonNames: ['OK','Cancel','Another Choice']
});

alertWindow.addEventListener('click',function(ev){
    Titanium.API.info( "cancel " + ev.cancel );
    Titanium.API.info( "index " + ev.index );
    Titanium.API.info( "index " + ev.source );
    switch(ev.index)
    {
    case 0:
      Titanium.API.info( "OK button was hit");
      break;
    case 1:
      Titanium.API.info( "cancel button was hit");
      break;
    case 2:
      Titanium.API.info( "Another Choice button was hit");
      break;
    }
});

alertWindow.show();
