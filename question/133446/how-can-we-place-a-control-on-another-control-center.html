How can we place a control on another control, center

I need to place an imageview on a button in exactly centre of button and then on this image i want to put a label on it, again exactly at the center of the imageview. How can we place a control on another control on its center. Any one can help me please.

    Try this:

    var win = Ti.UI.createWindow({backgroundColor: 'white'});

var container = Ti.UI.createView({
    width: 200,
    height: 60
});

var btn = Ti.UI.createButton({
    width: container.width,
    height: container.height
});

var imgWidth = imgHeight = 48;
var image = Ti.UI.createImageView({
    image: 'http://www.icondig.com/data/icons/oxygen/48x48/actions/contact-new.png',
    width: imgWidth,
    height: imgHeight,
    top: (container.height - imgHeight)/2,
    left: (container.width - imgWidth)/2,
    zIndex: 1
});

var lbl = Ti.UI.createLabel({
    text: 'Hello World',
    textAlign: 'center',
    zIndex: 2    
});

container.add(btn);
container.add(image);
container.add(lbl);
win.add(container);
win.open();
