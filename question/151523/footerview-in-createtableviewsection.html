footerView in createTableViewSection

Hi,
I'm using Titanium SDK 3.1 and my host operating system is Ubuntu 12.04.
I'm trying to use a TableViewSection to apply headerView and footerView to the section table. HeaderView works fine, but footerView does not seem to work.
Below, I list my code:

(function() {
    Netinventions.iconandsplashapp.ui = {};
    Netinventions.iconandsplashapp.ui.creaWindow = function() {

        var win = Titanium.UI.createWindow({
            fullscreen : true,
            navBarHidden : true,
            title : 'Table section',
            backgroundColor : '#FFFFFF'
        });



        var section1 = Titanium.UI.createTableViewSection({
            headerView: createCustomView('Section 1', '#222'),
            footerView: createCustomView('Close section 1', '#FF0000')
        });
        var section2 = Titanium.UI.createTableViewSection({
            headerView: createCustomView('Section 2', '#222'),
            footerView : createCustomView('Close section 2', '#FF0000')

        });

        for (var i = 0; i < 4; i++) {
            section1.add(Titanium.UI.createTableViewRow({
                title : 'Row ' + i
            }));
        }
        for (var i = 4; i < 10; i++) {
            section2.add(Titanium.UI.createTableViewRow({
                title : 'Row ' + i
            }));
        }
        var tabelView = Titanium.UI.createTableView({
            backgroundColor : 'trasparent',

            data : [section1,section2]
        });
        win.add(tabelView);

        return win;
    };
    var createCustomView = function(title, color) {
        var view = Titanium.UI.createView({
            backgroundColor : color,
            height : 40
        });
        var text = Titanium.UI.createLabel({
            text : title,
            left : 20,
            color : '#fff',
            width : '100%'
        });
        view.add(text);
        return view;
    };
})();

Where am I doing wrong?

Thank you.

  • Hi

    Without actually testing your code there are a couple of things I would do first;

    • Move your function createCustomView to be before you actually call it, whilst I do not know when and how you call the creaWindow function - it certinaly will not hurt
    • TableViewSections have an annoying need to have specific dimensions applied - unlike other objects, so add a width and height to dependant child views
    • You have moved into Ti SDK 3.1 you should be taking advantage of Ti.UI.FILL and Ti.UI.SIZE now
    • Fix a small typo trasparent becomes transparent - we have all done this one
    • I am a big fan of populating width and height properties as well
    (function() {
    var createCustomView = function(title, color) {
        var view = Titanium.UI.createView({
            backgroundColor : color,
            height : 40,
            width : Ti.UI.FILL
        });
        var text = Titanium.UI.createLabel({
            text : title,
            left : 20,
            color : '#fff',
            height : Ti.UI.SIZE,
            width : Ti.UI.FILL
        });
        view.add(text);
        return view;
    };
    Netinventions.iconandsplashapp.ui = {};
    Netinventions.iconandsplashapp.ui.creaWindow = function() {

        var win = Titanium.UI.createWindow({
            fullscreen : true,
            navBarHidden : true,
            title : 'Table section',
            backgroundColor : '#FFFFFF'
        });

        var section1 = Titanium.UI.createTableViewSection({
            headerView: createCustomView('Section 1', '#222'),
            footerView: createCustomView('Close section 1', '#FF0000')
        });
        var section2 = Titanium.UI.createTableViewSection({
            headerView: createCustomView('Section 2', '#222'),
            footerView : createCustomView('Close section 2', '#FF0000') 
        });

        for (var i = 0; i < 4; i++) {
            section1.add(Titanium.UI.createTableViewRow({
                title : 'Row ' + i
            }));
        }
        for (var i = 4; i < 10; i++) {
            section2.add(Titanium.UI.createTableViewRow({
                title : 'Row ' + i
            }));
        }
        var tabelView = Titanium.UI.createTableView({
            backgroundColor : 'transparent', 
            data : [section1,section2],
            height: Ti.UI.FILL,
            width: Ti.UI.FILL
        });
        win.add(tabelView);

        return win;
    };
})();

    Try this and see if anything improves.

    • @Mario You are the second person on this exact same subject in the last 30 minutes.

      Footers in sections are possible but you MUST set them at the creation of the section not afterwards.

      So;

      var viewFooter = Ti.UI.createView({
    height: 50,
    width: Ti.UI.Fill
});
var section1 = Titanium.UI.createTableViewSection({
    footerView: viewFooter // define here
});
section1.setFooterView(viewFooter); // NOT like this
section1.footerView = viewFooter; // and NOT like this
