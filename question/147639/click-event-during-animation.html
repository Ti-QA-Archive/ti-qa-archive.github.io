Click event during animation

Does anyone have studied the possibility to fire a specific function by clicking on a view that is animating?

It seem that I have to wait on end of animation in order to have eventlistner (click) activated….

var win = Ti.UI.createWindow({
    height : "100%",
    width : "100%",
    fullscreen : true,
    backgroundColor : "white"
})

var bersaglio = Ti.UI.createView({
    backgroundImage : "/myPic.png",
    width :25 ,
    height : 40,
    top : -100,
    left :100,
})

bersaglio.addEventListener ('focus', function(e){
    alert ("BINGO!");
});

bersaglio.animate({
    top : 600,
    left : 200,
    duration : 7000,
})

win.add (bersaglio);
win.open();
