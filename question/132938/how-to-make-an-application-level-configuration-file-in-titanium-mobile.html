    Create a CommonJS module that contains the definitions of your attributes, then "require" it as needed in other files; eg

        exports.theme = function() {

        this.baseBackgroundColor = '#274167';
        this.headerBackgroundColor = '#FFF';
        this.sectionBackgroundColor = '#2A528C';
        this.backgroundColor = 'transparent';
        this.tableViewRowBackgroundColor = '#313949';
        this.tableViewBackgroundColor = '#252C3A';
        this.tableViewRowTextColor = '#CCC';
        this.tableViewRowSelectedColor = '#3C4043';
        this.tableViewFont = {
            fontSize : 15,
            fontWeight : 'bold',
            fontFamily : 'Trebuchet MS'
        };


    }
