    this is a bit of code of my last app , used to create a listview with row with transparent background

    var template = {
            childTemplates : [{
                type : 'Ti.UI.Label',
                bindId : 'label_utente',
                properties : {
                    width : 452,
                    height : 30,
                    left : 10,
                    right : 50,
                    minimumFontSize : 8,
                    color : "white"
                }
            }, {
                type : 'Ti.UI.View',
                bindId : 'check_utente',
                properties : {
                    width : 30,
                    height : 30,
                    right : 10,
                    backgroundImage : "/images/img_new/c_accessi_viewcheck.png"
                }
            }]
        };

        var header_utenti = Ti.UI.createView({
            backgroundImage : "/images/img_new/blur_bianco2.png",
            height : 24,
            width : 512
        });

        var label_utenti = Ti.UI.createLabel({
            text : "-----",
            left : 10,
            color : "#424242",
            font : {
                fontSize : 14
            }
        });
        header_utenti.add(label_utenti);

        var listView_utenti = Ti.UI.createListView({
            width : 511,
            top : 61,
            height : 290,
            left : 0,
            backgroundColor : "transparent",
            templates : {
                'default' : template
            },
            defaultItemTemplate : 'default',
            separatorInsets : {
                left : 0,
                right : 0
            }
        });

       var sezione_utenti = Ti.UI.createListSection({
            headerView : header_utenti
        });

        var data_utenti = [];
        for (var i = 0; i < json.user.length; i++) {
            data_utenti.push({
                label_utente : {
                    text : json.user[i].name
                },
                check_utente : {
                    backgroundImage : "/images/kkkk.png"
                },
                properties : {
                    backgroundColor : "transparent",
                    selectionStyle : Ti.UI.iPhone.ListViewCellSelectionStyle.NONE,
                    backgroundImage : "/images/oooo.png",
                    id : json.user[i].id
                }
            });
        }
        sezione_utenti.setItems(data_utenti);

        listView_utenti.sections = [sezione_utenti];
        view_controllo_accessi_principale.add(listView_utenti);
