  • Still having some issues with oAuth and Foursquare API v2. Doing a venue search doesn't need authentication. So I figured I would update my venue search to V2 to keep things moving forward. I though it would be a simple switch but guess I was wrong.

    The below code works fine for V1 but won't work with V2 and I can't see why. Can anyone see what I am doing wrong?

    var xhr = Titanium.Network.createHTTPClient();
    // open the client
    xhr.open('GET','https://api.foursquare.com/v2/venues/search?ll='+lat+','+lon+'&oauth_token=TOKEN');
    Ti.API.info("xhr open");
    xhr.onload = function()
    {


        var resp =  eval('('+this.responseText+')');
        var venues = resp.groups[0].venues;
        var data = [];
        for (var i=0;i<venues.length;i++){

            var loca = Titanium.Map.createAnnotation({
                latitude:venues[i].geolat,
                longitude:venues[i].geolong,
                title:venues[i].name,
                image:"../images/mymarker.png",
                animate:true,
                rightButton: Titanium.UI.iPhone.SystemButton.DISCLOSURE,
                id:venues[i].id
            });
            data.push(loca);

        }
