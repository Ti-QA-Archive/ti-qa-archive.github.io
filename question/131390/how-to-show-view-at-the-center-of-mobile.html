How to show view at the center of mobile

I have created a view as below

var vview = Ti.UI.createView({
        height: 100,
        width: 100,
        layout: "vertical",
    backgroundColor:'#ff0000'
});


var fullview = Ti.UI.createView({
        height: '100%',
        width: '100%',
        backgroundColor:'#ffffff',
        layout:'vertical',
        top:'cente'
});

fullview.add(vview);
var win = Ti.UI.createWindow({  
    title:'win',
    backgroundColor:'#ffff00'
});

win.add(fullview);
win.open();

but thew view appear on top.
How do I make it to the center of the fullview?

  • Hi Avaya
    Try the below code too:

    var valign= (Titanium.Platform.displayCaps.platformHeight-100)/2;
var halign = (Titanium.Platform.displayCaps.platformWidth-100)/2;
var vview = Ti.UI.createView({
        height: 100,
        width: 100,
        top:valign,
        bottom:valign,
        right:halign,
        left:halign,
        layout: "vertical",
    backgroundColor:'#ff0000'
});

var fullview = Ti.UI.createView({
        height: Titanium.Platform.displayCaps.platformHeight,
        width: Titanium.Platform.displayCaps.platformWidth,
        backgroundColor:'#ffffff',
        layout:'vertical',
        top:'center'
});

    Regards

    Nikunj

  • Create View with top attribute.
    Ti.Platform.displayCaps.platformHeight this give height of device.

    var vview = Ti.UI.createView({
        height: 100,
        width: 100,
    top:Ti.Platform.displayCaps.platformHeight/2,
        layout: "vertical",
    backgroundColor:'#ff0000'
});
