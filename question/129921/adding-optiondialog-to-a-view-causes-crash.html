  • Tried this simple test under latest 1.8.0.1 SDK and iOS 5.0. Worked there: 

    var rootWindow = Ti.UI.createWindow({
    backgroundColor:'#000'
});

var win = Ti.UI.createWindow({
    title:'Option Dialog test'    
});

var opt = Ti.UI.createOptionDialog({
    title:'options',
    options:['option 1', 'option 2'],
    cancel:1    
});

var navGroup = Ti.UI.iPhone.createNavigationGroup({
    window:win
});

rootWindow.add(navGroup);
rootWindow.open();
opt.show();

    • I guess that might be the point. You don't need to attach the OptionDialog to a view or window. I'd be interested to see the code where it's being attached to an Android view. Here's a better example of triggering the dialog to show up. Kitchen Sink also has one in examples\options_dialog.js. 

      var rootWindow = Ti.UI.createWindow({
    backgroundColor:'#000'
});

var win = Ti.UI.createWindow({
    title:'Option Dialog test'    
});

var opt = Ti.UI.createOptionDialog({
    title:'options',
    options:['option 1', 'option 2'],
    cancel:1    
});

var vw = Ti.UI.createView({
    height: 100,
    width: 100,
    backgroundColor:'#fff'
});

var label = Ti.UI.createLabel({
    text:'click me to open dialog',
    height:'auto',
    width:'auto',
    color:'red'
});

label.addEventListener('click', function(e){
    opt.show();
});

vw.add(label);
win.add(vw);

var navGroup = Ti.UI.iPhone.createNavigationGroup({
    window:win
});

rootWindow.add(navGroup);
rootWindow.open();
