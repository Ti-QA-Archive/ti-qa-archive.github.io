  {title:i18n_string('controller_main_tab_about'), controller:'/controllers/about.js', shortTitle:i18n_string('controller_main_tab_about'), tabIconName:'nav_apropos.png'}
];

view_create_main_android_iphone(tabs_item);

function view_create_main_android_iphone(tabs_items) {
    var tabGroup = Titanium.UI.createTabGroup();
    for (i=0; i<tabs_item.length; i++) {
        var win = Titanium.UI.createWindow({
            url:tabs_item[i].controller,
            title:tabs_item[i].title,
            backgroundColor:view_default_background_color(),
            barColor:view_default_bar_color()
        });
        var platform_name = '';
        if (isAndroid()) {
            platform_name = 'android/';
        }
        else if (isIPhone()) {
            platform_name = 'iphone/';
        }
        var tab = Titanium.UI.createTab({
            title:tabs_item[i].shortTitle,
            window:win,
            icon:'images/'+platform_name+tabs_items[i].tabIconName
        });
        tabGroup.addTab(tab);  
    }
    tabGroup.open();
}
