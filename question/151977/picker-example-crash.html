I am testing the single picker and the multiple picker components through this code:

var win = Titanium.UI.createWindow({

    backgroundColor:"#FFFFFF",
    exitOnClose:true
});

var resultat = Ti.UI.createLabel({
    top:150,
    width:"auto", 
    height:"auto",
    font:{fontSize:40,fontWeight:"bold"},
    color:"red"
});

    top:10
});

var datasource=[];

datasource[0]=Ti.UI.createPickerRow({title:"Musique",val:"music"});

datasource[2]=Ti.UI.createPickerRow({title:"Voyages",val:"travel"});

mypicker.add(datasource);

mypicker.addEventListener("change",function(e){

    resultat.text=e.row.title+" - "+e.row.val;
});


var datasource1=[

    for(var i=0;i<datasource1.lentgh;i++){

        pickercolumn2.addRow(Titanium.UI.createPickerRow({title:datasource2[i].title,val:datasource2[i].val}));
