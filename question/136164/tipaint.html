    function undo() {
    undoButton.removeEventListener('click', undo);
        if(paintViewHistory.length > 0) {
            var lastPaintView = paintViewHistory.pop();
            paintView.image = lastPaintView;
        }
        setTimeout(function() {
                undoButton.addEventListener('click', undo);
        }, 100);
}
