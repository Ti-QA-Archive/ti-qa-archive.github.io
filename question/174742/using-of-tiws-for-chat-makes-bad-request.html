Using of TiWS for chat makes "Bad request"

Hi,
I'm using TiWS (websockets API) from marketplace to build a simple chat system without cloud.
Here my server code (inspired from Christian Brousseau):

var chatter = [];
var server = require('http').createServer();
server.on('request', function(req, res) {
    res.writeHead(200, {
        'Content-type' : 'text/plain'
    });
    res.end('Server rennt …');
});
server.listen(1333, function() {
    console.log('Listening at localhsot.1333');
});
var socketio = require('socket.io');
socketio.listen(server).on('connection', function(socket) {
    socket.on('join', function(newchatter) {
        console.log(newchatter);
    });
});

On client run this code:

exports.start = function() {
    var socket = require('vendor/socket.io').connect(Ti.App.Properties.getString('chaturi'));
    socket.emit('join', {
        id : Ti.Platform.id,
        avatar : 'myHead'
    });
};

On server I can test the server. He answers with "Server rennt …", but from android app comes:

[ERROR] :  TiHttpClient:     at ti.modules.titanium.network.TiHTTPClient$LocalResponseHandler.handleResponse(TiHTTPClient.java:258)

What can I do to find the error?
