function getLocation() {

Titanium.Geolocation.getCurrentPosition(function(e) {
    if (e.error) {
        Ti.API.log('error: ' + JSON.stringify(e.error));
        return;
    };

    Titanium.App.Properties.setString('lat', e.coords.latitude);
    Titanium.App.Properties.setString('long', e.coords.longitude);
    Titanium.App.Properties.setString('alt', e.coords.altitude);
    var region = {
        latitude : e.coords.latitude,
        longitude : e.coords.longitude,
        animate : true,
        latitudeDelta : 0.0001,
        longitudeDelta : 0.0001
    };
    mymap.setRegion(region);
});

}

Titanium.Geolocation.addEventListener('location', function() {
getLocation();
});
