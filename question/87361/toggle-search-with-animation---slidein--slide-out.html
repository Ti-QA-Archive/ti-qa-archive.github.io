  • take a look at the code below. Instead of using searchbar.show, search can be hidden or shown in the table view properties. So, instead use tableName.searchHidden==true/false

    schools_win.setRightNavButton(search_button);
    search_button.addEventListener('click', function()
    {
    if(tableName.searchHidden==false)
    {
    tableName.searchHidden=true;
    }
    else {
    tableName.searchHidden=false;
    search.focus();
    }
    });
