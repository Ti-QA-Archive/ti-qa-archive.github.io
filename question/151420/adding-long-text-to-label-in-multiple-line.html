adding long text to label in multiple line

Hi,
I want to put a long text in label. Its about 15 lines. So what is the procedure to add this text to label in multiple line. 

var label1 = Titanium.UI.createLabel({
        top:100,
        color:'black',
        text:"adfafds fdskjfk dsfjklsdjfkldsjfkl kdjf klsdjf ldsj fkldsjfkldsj ",
        textAlign:'center',
        font:{fontSize:size,fontFamily:'Helvetica Neue'},
        width:'auto'
    });

but I want to write this in this way.

var label1 = Titanium.UI.createLabel({
        top:100,
        color:'black',
        text:"adfafds fdskjfk 
        dsfjklsdjfkldsjfkl kdjf 
        klsdjf ldsj fkldsjfkldsj fkdsjfkl dsklfjdsklfjk ",
        textAlign:'center',
        font:{fontSize:size,fontFamily:'Helvetica Neue'},
        width:'auto'
    });

Please suggest me a way. Or is there any other method of adding long text?

    Hello Mahbubur,

    You can separate line by \n.

    var label1 = Ti.UI.createLabel({
    top : 100,
    color : 'black',
    width : 'auto',
    height : 'auto',
    text : 'adfafds fdskjfk' + '\n' + 'dsfjklsdjfkldsjfkl' + '\n' + 'kdjf klsdjf ldsj' + '\n' + 'fkldsjfkldsj fkdsjfkl dsklfjdsklfjk',
    textAlign : 'center',
    font : {
        fontSize : 12,
        fontFamily : 'Helvetica Neue'
    }
});
win.add(label1);
