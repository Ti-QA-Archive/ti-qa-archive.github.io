    hi,

    have you tried with httpClient…..?

    var winSplash = Ti.UI.createWindow({
    // left : 320,
    height : 460,
    width : 320,
    backgroundColor : 'black'
});

var img = Ti.UI.createImageView({
    height : 460,
    width : 320
});

var xhr = Titanium.Network.createHTTPClient({
    onerror : function(e) {

        alert('Error | No internet connection detected.');
        Ti.App.fireEvent('hide_indicator');
    },
    onload : function(e) {

        if (this.status == 200) {
            img.image = this.responseData;
            setTimeout(function(et) {
                var winAct = Ti.UI.createWindow({

                });

                winAct.open({
                    modal : true
                });

                winSplash.animate({
                    left : -320,
                    duration : 500
                }, function(evt1) {
                    winSplash.close();
                });

            }, 3000);
        }
    }
});

xhr.setTimeout(120000);

xhr.open('GET', 'http://www.alliphonewallpapers.com/images/wallpapers/o427bufnc.jpg');

xhr.send();

winSplash.add(img);

winSplash.open();
