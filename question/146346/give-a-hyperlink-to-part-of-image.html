  • hi,

    you can get x and y co-ordinate in touchend event so on that co-ordinates you can open different urls

    test this code…

    var self = Ti.UI.createWindow({
    // backgroundColor : '#666666'
    backgroundColor : 'white',
    // backgroundColor : 'stripped',
    titleImage : 'img/1-Star.png',
    // url : 'scrollView.js'
});

var img = Ti.UI.createImageView({
    height : Ti.UI.SIZE,
    width : Ti.UI.SIZE,
    image : 'http://www.osa-opn.org/opn/media/Images/ImageOfTheWeek/12-10-22.jpg?width=1024&height=1024&ext=.jpg'
});

img.addEventListener('touchend', function(e) {
    alert('x : ' + e.x + ' , Y : ' + e.y);
});

self.add(img);
self.open();
