Question about understanding this two functions

What does this two functions do?

function doTiming() {
Ti.API.info('timer is running: ');
Ti.API.info('Time: '+videoPlayer.currentPlaybackTime);
var time1 = videoPlayer.currentPlaybackTime;
if ((time1 > 15000) && (time1 < 15700) )
{
clearInterval(timer);
videoPlayer.pause();
Ti.App.fireEvent('VideoChoice1', {choice: 'choice1'});
}
if (time1 >= videoPlayer.duration){
clearInterval(timer);
}

}
videoPlayer.addEventListener('playbackState', function(e) {
    Ti.API.info('state is changing');
    if (myNum===0){
        myNum=1;
        timer = setInterval(doTiming,250);
        Ti.API.info("Video starting: "+e);
    }else {
        Ti.API.info("Video stopping: "+e);
        myNum = 0;
        clearInterval(timer);
    }        
});

  • Hi,

    playbackState is actually a event listener for videoPlayer object which is called when videoPlayer changes its state. For more detail you can check online doc…

    and doTiming is a custom function which is created to check the video playback time so i think it actually used to pause movie when video's playback time is more than 15000 and less than 15700.

    VideoChoice1 event is a custom event listener to call whenif block is called.
