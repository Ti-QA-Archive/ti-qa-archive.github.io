Android textfield getting focus issue

Hi All,

I am having the Window which is containing the scrollview and label, input text field in the below order.

3 Labels, and i am adding the value for these labels when selecting value from Custom Drop down,

And then having 3 Text fields under that label.
I had set the blur method to avoid the Text field getting focus when window getting focus. So the Softkeyboard is not appearing, Which is fine, But the Cursor is showing on the text field.

its working fine on Android Phone. Not showing the cursor on it, And not skipping the top 3 labels by scroll down the page to Text field

BUT The ANDROID TAB getting an issue by showing the Cursor on The First TextField (no softkeyboard ther), And its Skipping the first 3 labels, So this will be inconvenient to user. Because they are not able to see those 3 Labels which are above to the Textfield.

    function SellView(parentView, user_id, parentWindow) {

var selfWin = Ti.UI.createWindow({
        top : VARS.isOS7 ? 20 : 0,
        statusBarStyle : VARS.isOS7 ? Ti.UI.iPhone.StatusBar.GREY : "",
        navBarHidden : true,
        backgroundColor : 'white',
        exitOnClose : false
    });

var scrollView1 = Titanium.UI.createScrollView({
        contentWidth : 'auto',
        top : globalVariables.GetHeight(50),
        showVerticalScrollIndicator : true,
        showHorizontalScrollIndicator : false,
        touchEnabled : true,
        visible : true,
                layout:"vertical"
    });
selfWin.add(scrollView1);

var Label1 = Titanium.UI.createLabel({
        left : globalVariables.GetWidth(10),
        right : globalVariables.GetWidth(10),
        font : {
            fontSize : globalVariables.GetHeight(18),
        },
        color : VARS.pinkColor,
        top : globalVariables.GetHeight(10),
        text : "Label1",
        textAlign : 'center'
    });
    scrollView1.add(Label1);

var Label2 = Titanium.UI.createLabel({
        left : globalVariables.GetWidth(10),
        right : globalVariables.GetWidth(10),
        font : {
            fontSize : globalVariables.GetHeight(18),
        },
        color : VARS.pinkColor,
        top : globalVariables.GetHeight(10),
        text : "Label2",
        textAlign : 'center'
    });
    scrollView1.add(Label2);

var Label3 = Titanium.UI.createLabel({
        left : globalVariables.GetWidth(10),
        right : globalVariables.GetWidth(10),
        font : {
            fontSize : globalVariables.GetHeight(18),
        },
        color : VARS.pinkColor,
        top : globalVariables.GetHeight(10),
        text : "Label3",
        textAlign : 'center'
    });
    scrollView1.add(Label3);

var txt_brand = Titanium.UI.createTextField({
        hintText : 'Enter brand...',
        height : globalVariables.GetHeight(30),
        left : globalVariables.GetWidth(10),
        right : globalVariables.GetWidth(10),
        top : globalVariables.GetHeight(325), //315
        backgroundImage : 'none',
        borderColor : VARS.textbordercolor,
        backgroundColor : '#fff',
        borderRadius : VARS.textborderradius,
        borderWidth : VARS.textborderwidth,
        paddingLeft : VARS.padleft,
        color : VARS.globalfont_color,
        font : {
            fontSize : VARS.globalfont_size_android,
            fontFamily : VARS.globalfont_family
        },
        borderStyle : Ti.UI.INPUT_BORDERSTYLE_NONE
    });

    txt_brand.addEventListener('focus', function(evt) {
        txt_brand.borderColor = VARS.textboxfocus;
    });

    txt_brand.addEventListener('blur', function(evt) {
        txt_brand.borderColor = VARS.textboxblur;
    });
scrollView1.add(Label3);

selfWin.addEventListener('focus', function(e) {
        setTimeout(function() {        
            txt_brand.borderColor = "#6C6363";            
            txt_brand.blur();            
        }, 1000);
    });

}

module.exports = SellView;
