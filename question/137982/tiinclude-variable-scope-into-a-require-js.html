Hi,

I have a problem with the Ti.include() function, each time i try to make an include i'm running into this error : [ERROR] Script Error = Can't find variable:..

app.js

var Window1 = require('ui/Window1').init();
splash.open();

ui/Window1.js

exports.init = function () {
    var win = Titanium.UI.createWindow({
        some props
    });

    var button = Titanium.UI.createButton({
        some props
    });
    win.add(button);

    Ti.include('events/Window1Events.js'); //The Ti.include() call

events/Window1Events.js

button.addEventListener('click', function(e) {
    blablabla

Is this because the Ti.include() call is in the require js ?
