  • Hi David, this is an animation that you have a look,

    var win = Ti.UI.createWindow({title : "Animation window"});
var tView = Ti.UI.createView({
    top : 800,
    height : 200,
    width : '100%',
    backgroundColor : 'red'
});

var button = Ti.UI.createButton({
    top : 20,
    height : 20,
    width : 100,
    title : 'Up'
});
button.addEventListener('click', function(e) {
    if(button.title == 'Up') {
        button.title = 'Down'
        var anim = Ti.UI.createAnimation({
            top : 200,
            duration : 1000
        });

        tView.animate(anim);
    } else {
        button.title = 'Up'
        var anim = Ti.UI.createAnimation({
            top : 800,
            duration : 1000
        });

        tView.animate(anim);
    }
});

win.add(button);
win.add(tView);
