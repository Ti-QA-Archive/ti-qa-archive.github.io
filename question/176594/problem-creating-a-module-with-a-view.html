    The code of the DeWzlModulesTestWindowModuleView.h and .m file is the same as in the tutorial:

    DeWzlModulesTestWindowModuleView.h:

    #import "TiUIView.h"
#import "DeCompanyModulesTestWindowModuleViewProxy.h"

@interface DeCompanyModulesTestWindowModuleView : TiUIView {
    UIView *square;
}

@end

    DeWzlModulesTestWindowModuleView.m:

    
#import "DeCompanyModulesTestWindowModuleView.h"
#import "TiUtils.h"

@implementation DeCompanyModulesTestWindowModuleView

-(void)dealloc
{
  RELEASE_TO_NIL(square);
  [super dealloc];
}

-(UIView*)square
{
  if (square==nil)
  {
    square = [[UIView alloc] initWithFrame:[self frame]];
    [self addSubview:square];
  }
  return square;
}

-(void)frameSizeChanged:(CGRect)frame bounds:(CGRect)bounds
{
  if (square!=nil)
  {
    [TiUtils setView:square positionRect:bounds];
  }
}

-(void)setColor_:(id)color
{
  UIColor *c = [[TiUtils colorValue:color] _color];
  UIView *s = [self square];
  s.backgroundColor = c;
}
@end

    As said in the tutorial the ViewProxy is just the template code.
