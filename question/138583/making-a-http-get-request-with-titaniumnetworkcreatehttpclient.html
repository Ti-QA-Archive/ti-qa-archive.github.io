Making a HTTP-GET Request with Titanium.Network.createHTTPClient()

I am working with Titanium Studio and I have to make an HTTP get request to Couchdb. This is my code:

var url ="http://192.168.0.152:5985/hello_world/00a271787f89c0ef2e10e88a0c0001f4";
var json;

var xhr = Ti.Network.createHTTPClient({
onload: function() {
Ti.API.info("Received text: " + this.responseText);
json = JSON.parse(this.responseText); 
},
onerror: function(e) {
Ti.API.debug("STATUS: " + this.status);
Ti.API.debug("TEXT:   " + this.responseText);
Ti.API.debug("ERROR:  " + e.error);
alert('There was an error retrieving the remote data. Try again.');
},
timeout:5000
});
xhr.open("GET", url);
xhr.send();

The operation go in time out..

  • Hey Maria,

    Follow this format and you shouldn't have any issues:

              var url = "http://192.168.0.152:5985/hello_world/00a271787f89c0ef2e10e88a0c0001f4";
var json;
var appReq = Ti.Network.createHTTPClient();

       appReq.open('GET', url);

        //on load handler
        appReq.onload = function(){

            TI.API.info("Received text: " + this.responseText);
            json = eval('(' + this.responseText + ')');

        };

        //error handler
        appReq.onerror = function(e){

            Ti.API.debug("STATUS: " + this.status);
            Ti.API.debug("TEXT:   " + this.responseText);
            Ti.API.debug("ERROR:  " + e.error);
            alert('There was an error retrieving the remote data. Try again.');

        };

        appReq.send();

    I would try this without the the timeout first. I believe the order of the calls makes a difference too. Also make sure the JSON you're receiving is valid JSON. You can use JSON Lint to check for errors.

    Hope this helps.
