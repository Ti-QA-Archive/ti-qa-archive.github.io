      function update(videos){

                var playImage = Ti.UI.createImageView({
                    image: 'images/play.png',
                    width: 22, height: 22,
                    zIndex: 1
                });

                var content = Ti.UI.createView({
                    layout: 'vertical'
                });

                for(var i=0;i<videos.length-1;i=i+2){

                    var imagesContent = Ti.UI.createView({
                        layout: 'horizontal',
                        top: 15,
                        height: 64
                    });

                    var imageVideoViewGauche = Ti.UI.createImageView({
                        defaultImage: "images/default.jpg",
                        image : videos[i].image,
                        left: 30,
                        width : 100,
                        height : 64,
                        borderColor: '#fff',
                        urlVideos : videos[i].url,
                        titre : videos[i].titre,
                        pubdate : videos[i].pubdate,
                        zIndex:0
                    });

                    var imageVideoViewDroite = Ti.UI.createImageView({
                        defaultImage: "images/default.jpg",
                        image : videos[i+1].image,
                        left: 60,
                        width : 100,
                        height : 64,
                        borderColor: '#fff',
                        urlVideos : videos[i+1].url,
                        titre : videos[i+1].titre,
                        pubdate : videos[i+1].pubdate,
                        zIndex:0
                    });

                    imageVideoViewGauche.add(playImage);
                    imageVideoViewDroite.add(playImage);
                    imagesContent.add(imageVideoViewGauche);
                    imagesContent.add(imageVideoViewDroite);
                    content.add(imagesContent);
                }

                scroll.add(content);
            }
