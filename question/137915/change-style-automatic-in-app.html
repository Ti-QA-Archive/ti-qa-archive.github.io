    • Hi Rickard,

      Try this code…

      var win = Ti.UI.currentWindow;

var btn = Ti.UI.createButton({
    title : 'change'
});
win.rightNavButton = btn;
btn.addEventListener('click', function(e) {
    animation.bottom = 0;
    Locationview.animate(animation);
});
var strLocation = 'blue';
var animation = Titanium.UI.createAnimation({
    duration : 700
});

var Locationview = Ti.UI.createView({
    height : 230,
    bottom : -230,
    zIndex : 999999
});

var Locationpicker = Ti.UI.createPicker({
    //type : Ti.UI.PICKER_TYPE_DATE,
    left : 0,
    top : 40,
    width : 320
});

var flexSpace1 = Titanium.UI.createButton({
    systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

var Locationcancel = Ti.UI.createButton({
    title : 'Cancel',
    style : 1
});

Locationcancel.addEventListener('click', function(e) {
    animation.bottom = -230;
    Locationview.animate(animation);
});
var Locationdone = Ti.UI.createButton({
    title : 'Done',
    style : 1
});
Locationdone.addEventListener('click', function(e) {

    win.barColor = strLocation;
    animation.bottom = -230;
    Locationview.animate(animation);
});
var candonetoolview1 = Ti.UI.iOS.createToolbar({
    height : 40,
    width : '100%',
    left : 0,
    top : 0,
    items : [Locationcancel, flexSpace1, Locationdone],
    barColor : '#000000'
});
Locationpicker.addEventListener('change', function(e) {
    strLocation = e.row.title;
});
var data = [];

data[0] = Ti.UI.createPickerRow({
    title : 'red'
});
data[1] = Ti.UI.createPickerRow({
    title : 'green'
});
data[2] = Ti.UI.createPickerRow({
    title : 'blue'
});
data[3] = Ti.UI.createPickerRow({
    title : 'yellow'
});
data[4] = Ti.UI.createPickerRow({
    title : 'gray'
});
data[5] = Ti.UI.createPickerRow({
    title : 'black'
});
// turn on the selection indicator (off by default)
Locationpicker.selectionIndicator = true;

Locationpicker.add(data);
Locationview.add(candonetoolview1);
Locationview.add(Locationpicker);
win.add(Locationview)
