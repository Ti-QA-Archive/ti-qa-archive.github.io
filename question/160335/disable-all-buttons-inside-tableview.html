Disable all buttons inside tableView

Hello,
im creating dynamically some rows in a table view. Each row has a button inside it, i want when i click a button in a row to disable all the other buttons in the other rows
here is the code:

for(var i = 0; i < file_num; i++) {
    var row = Ti.UI.createTableViewRow({
        id:i,
        height: 180,
        className: "sound_row",
    });
    var button_ = Ti.UI.createButton({
        title : '',
        ctl: 'btn_play',
        backgroundImage: 'images/play.png',
        height : 110,
        width: 110,
        left :10,
        buttonid : i   
    });    

    row.add(button_);
    tableData.push(row);
}
var tableView = Ti.UI.createTableView({
    data : tableData,
    top: 94
});

tableView.addEventListener('click', function(e) {
    if (e.source.ctl == 'btn_play') {
        //disable all buttons except the one i clicked    
    }
}

    try this code. I hope, it will work.

    var win = Ti.UI.createWindow({backgroundColor: 'white'});

var file_num=10;
var tableData=[];
for(var i = 0; i < file_num; i++) {
    var row = Ti.UI.createTableViewRow({
        id:i,
        height:'auto',//180,
        className: "sound_row",
    });
    var button_ = Ti.UI.createButton({
        title : 'Click',
        ctl: 'btn_play',
        //backgroundImage: 'images/play.png',
        height : 110,
        width: 110,
        left :10,
        buttonid : i   
    }); 
    row.add(button_);
    tableData.push(row);
}
var tableView = Ti.UI.createTableView({
    data : tableData,
    top: 94
});

tableView.addEventListener('click', function(e) {
    if (e.source.ctl == 'btn_play') {
        //disable all buttons except the one i clicked 
        var current=e.index; 
        alert(current);
        for(var a=0;a<file_num;a++)
        {
            if(current!=a)
            {
                tableView.data[0].rows[a].children[0].visible=false;
            }
        }
    }
});
win.add(tableView);
win.open();
