Unable to change title color of TableViewRow

Hi,
I've a list which display a list of articles. What I want to do is change the color of title if it's already been clicked. HighlightColor does not work cause the information is retrieved from database, so it's not just changing the color when it's clicked. It has to show the row color in a different color if it has been visited before. Just like browser clicked links. I tried using row.color = "#FF0" and row.setColor()..but it does not work.
The following works..but i can't change it dynamically.
row = Titanium.UI.createTableViewRow({
title : postName,
backgroundColor : '#000',
color : 'red'
});

Please help.
thanks

    hey i got the solution…

    try this

    function createRow(title) {
    var tvr = Ti.UI.createTableViewRow({
    });
    //comment next two lines for it to work correctly.
    var lbl = Ti.UI.createLabel({
        text : title,
        color :'white',
        left : 0
    });
    tvr.lbl = lbl;
    tvr.add(lbl);
    return tvr;
}

var tableview = Ti.UI.createTableView({

});
var data = [];
for(var i = 0; i <= 20; i++) {
    data.push(createRow('Row ' + i));
}
tableview.setData(data);

tableview.addEventListener('click', function(e) {
    var lbl = e.row.lbl;
    lbl.color = 'blue';
});

win.add(tableview);
