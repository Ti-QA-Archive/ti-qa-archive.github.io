    Hi

    If you are using the built in leftImage property then the display of that image is handled by the OS and the only parameter you have available is the url of the image itself.

    You can however create a custom tableViewRow and add labels and image (or many other things inside each row).

    This is a very simple example of how to achieve this;

    var addRow = function (o) {
    var obj = o || {};
    var row = Ti.UI.createTableViewRow({
        backgroundColor: '#fff',
        data: obj,
        height: Ti.UI.SIZE,
        id: 'row',
        width: Ti.UI.FILL
    });
    var img = Ti.UI.createImageView({
        height: Ti.UI.SIZE,
        image: obj.image,
        left: 10,
        width: Ti.UI.SIZE
    });
    var lbl = Ti.UI.createLabel({
        color: '#000',
        font: {
            fontSize: 18
        },
        height: Ti.UI.SIZE,
        id: 'lbl',
        left: 100,
        text: obj.title || '',
        width: Ti.UI.SIZE
    });
    row.add(lbl);
    return row;
};
var data = [
    { title: 'One', image: 'one.png' },
    { title: 'Two', image: 'two.png' },
    { title: 'Three', image: 'three.png' },
    { title: 'Four', image: 'four.png' },
    { title: 'Five', image: 'five.png' },
    { title: 'Six', image: 'six.png' },
    { title: 'Seven', image: 'seven.png' }
];
var intRow = 0, intRows = data.length;
var row, rows = [];
for (intRow = 0; intRow < intRows; intRow = intRow + 1) {
    rows.push(addRow(data[intRow]));
}
var tbl = Ti.UI.createTableView({
    data: rows,
    height: Ti.UI.FILL,
    minRowHeight: 50,
    width: Ti.UI.FILL
});
tbl.addEventListener('click', function (e) {
    alert('data: ' + JSON.stringify(e.row.data));
});
win.add(tbl);
