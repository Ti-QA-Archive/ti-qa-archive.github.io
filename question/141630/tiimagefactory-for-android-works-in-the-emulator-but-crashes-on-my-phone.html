ti.imagefactory for Android works in the emulator, but crashes on my phone

I'm using ti.imagefactory to scale and crop images from the camera/gallery on android. It works great on the emulator, so that's good. But when deployed to my phone, it crashes:

Uncaught error: java Exception occurred

The specific usage is:

var ImageFactory = require( 'ti.imagefactory' );
...

source = ImageFactory.compress( source, 1.0 ) // crashes here

In the above line, source is the image returned from the camera or gallery. I'm doing this to get an image blob with width & height so I can crop later. I'm doing this because the image blob returned from the camera/gallery APIs has zero width and height. I'm using ti.imagefactory to work around this bug.

See TIMOB-8481

As I mentioned, this works beautifully in the emulator ( running 2.3 ). It crashes on my phone.

My setup:
Application type: mobile

Titanium SDK: 2.1.1.GA

Platform & version: Android 2.3

Device: Droid Incredible, Android 2.3

Host Operating System: Mac OSX Lion ( 10.7 )

Titanium Studio: 2.1.1.201207271312

  • I've found the solution. It happens that the success callback to the camera and gallery picker passes the image's width and height.

    Who knew? Certainly not me, since it's undodumented. But I'll take it, since I was about to give up and attempt to write my client's app from scratch as a native android app.

    Here's the final working code. It correctly determines the width and height of images coming in from the camera or gallery, and then makes a much smaller version in an offscreen ImageView ( thanks to Atilla's tip! ) and then passes the toBlob() of that view to ImageFactory for cropping.

    NOTE this code is only meaningful for Android, and for the problem of cropping images coming from the camera or gallery. On iOS blobs correctly report their width and height, and on iOS blob has the imageAsCropped and imageAsResized methods built-in.

    (function() {

    var ImageFactory = require( 'ti.imagefactory' );

    var window = Ti.UI.createWindow({
        backgroundColor:'#ffffff',
        navBarHidden:true,
        exitOnClose:true
    });

    var view = Ti.UI.createView({
        layout: 'vertical'
    });

    window.add( view );

    var row = Ti.UI.createView({
        left: 20,
        right: 20,
        top: 20,
        height: Ti.UI.SIZE,
        layout: 'horizontal'
    })

    view.add( row );

    var cameraButton = Ti.UI.createButton({
        title: 'Camera',
        left: 0,
        height: 50
    })

    row.add( cameraButton );

    var galleryButton = Ti.UI.createButton({
        title: 'Gallery',
        left: 20,
        height: 50
    })

    row.add( galleryButton );

    var SCREEN_WIDTH = Ti.Platform.displayCaps.platformWidth;
    var imageContainer = Ti.UI.createView({
        top: 20,
        width: SCREEN_WIDTH - 40,
        height: SCREEN_WIDTH - 40,
        layout: 'absolute',
        backgroundColor: '#ccc'
    })

    view.add( imageContainer );

    var imageView = null;

    /////////////////////////////////////////////////////////////

    var _openGallery = function()
    {
        Ti.Media.openPhotoGallery({
            mediaTypes: [Ti.Media.MEDIA_TYPE_PHOTO],
            success: function(event){
                alert( 'image width: ' + event.width + ' height: ' + event.height ); 
                _showPhoto( event.media, event.width, event.height ) 
            }
        })
    }    

    var _openCamera = function()
    {
        Ti.Media.showCamera({
            mediaTypes: [Ti.Media.MEDIA_TYPE_PHOTO],
            success: function(event){ 
                alert( 'image width: ' + event.width + ' height: ' + event.height ); 
                _showPhoto( event.media, event.width, event.height ) 
            }
        })
    }

    var _showPhoto = function( image, width, height )
    {    
        if ( imageView )
        {
            imageContainer.remove( imageView );
            imageView = null;
        }

        if ( true )
        {
            // compute minimum scale to fill the screen
            var scale = 1;
            if ( width < height )
            {
                scale = Ti.Platform.displayCaps.platformWidth / width;
            }
            else
            {
                scale = Ti.Platform.displayCaps.platformHeight / height;
            }

            // render to an offscreen view and then export a smaller blob        
            var offscreenView = Ti.UI.createImageView({
                image: image,
                width: width * scale,
                height: height * scale
            })

            image = offscreenView.toBlob();

            // now, the image is small enough to work in ImageFactory's methods
            image = ImageFactory.imageAsThumbnail( image, {
                size: 300,
                border: 0
            })
        }

        imageView = Ti.UI.createImageView({
            image: image,
            width: image.width,
            height: image.height        
        })

        imageContainer.add( imageView );
    }

    /////////////////////////////////////////////////////////////

    cameraButton.addEventListener('click', _openCamera );
    galleryButton.addEventListener('click', _openGallery );

    /////////////////////////////////////////////////////////////

    window.open();

}());
