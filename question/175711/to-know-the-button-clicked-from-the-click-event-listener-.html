    Hello kaushik,

    Try this :

    index.xml

    <Button class="numbut" id="bb4" onClick="bclick" left ="20" top="150" title="4"/>
<Button class="numbut" id="bb5" onClick="bclick" left ="80" top="150" title="5"/>

    index.js

    function bclick(e) {
    Ti.API.info('info ==> ' + JSON.stringify(e));
    Ti.API.info('Title ==> ' + e.source.title);
    Ti.API.info('id ==> ' + e.source.id);
}

    and output is : 

    info ==> {"x":15,"y":15,"bubbles":true,"type":"click","source":{"title":"4","id":"bb4","left":"20","enabled":true,"horizontalWrap":true,"top":"150"},"cancelBubble":false}
