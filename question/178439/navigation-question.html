navigation question

Hello,

Can anyone tell me what I'm doing wrong and why the 4th window will not open.

Thanks

index.js

$.index.open();

$.rightNavButtonWin1.addEventListener('click', function(){
    var win3 = Alloy.createController('win3').getView();      
    $.tab1.open(win3);
});

index.xml

<Alloy>
    <TabGroup>
        <Tab title="Tab 1" id="tab1" icon="KS_nav_ui.png">
            <Window id="win1" title="Win 1">
                <RightNavButton platform=ios>
                    <Label color="000000" id="rightNavButtonWin1">open win3</Label>
                </RightNavButton>
                <Label>I am Window 1</Label>
            </Window>
        </Tab>
        <Tab title="Tab 2" id="tab2" icon="KS_nav_views.png">
            <Window id="win2" title="Win 2">
                <Label>I am Window 2</Label>
            </Window>
        </Tab>

</Alloy>

  • Accepted Answer

    Change

    $.rightNavButtonWin3.addEventListener('click', function(){
    var win4 = Alloy.createController('win4').getView();      
    $.win3.open(win4);
});

    to

    $.rightNavButtonWin3.addEventListener('click', function(){
    var win4 = Alloy.createController('win4').getView();      
    win4.open();
});

    • In that case you need to put $.tab1 into Globals and use that to open the window:

      index.js

      Alloy.Globals.tab1 = $.tab1;

      win3.js

      $.rightNavButtonWin3.addEventListener('click', function(){
    var win4 = Alloy.createController('win4').getView();      
    Alloy.Globals.tab1.open(win4);
});

      Can I also suggest that if you don't need the win4 variable, just do this:

      $.rightNavButtonWin3.addEventListener('click', function(){
    $.win3.open(Alloy.createController('win4').getView());
});
