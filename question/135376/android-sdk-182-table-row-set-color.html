  • I experienced the same issue on my previous project. I cannot change the text color on Android. The workaround was adding Label to TableViewRow, then change Label's color.

    function createRow(title) {
    var row = Ti.UI.createTableViewRow();
    var label = Ti.UI.createLabel({
        left: 10,
        text: title,
        color: 'black'
    });

    row.add(label);    
    return row;
}

var win = Ti.UI.createWindow();

var rows = [
    createRow('Row 1'),
    createRow('Row 2'),
    createRow('Row 3'),
];
var table = Ti.UI.createTableView({
    data: rows
});

win.add(table);
win.open();

setTimeout(function(){
    for(var i2 = 0; i2 < table.data[0].rows.length; i2++) {
        table.data[0].rows[i2].children[0].color = 'red';
    }
}, 1000);
