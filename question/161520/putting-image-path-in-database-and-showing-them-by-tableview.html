    Hi Sharif Abu Darda!!!

    please try the below code. I hope, it will work 

    var win=Ti.UI.createWindow({
    backgroundColor:'white'
});
var db = Titanium.Database.open('db');

    db.execute('DROP TABLE IF EXISTS DATABASE');
    db.execute('CREATE TABLE IF NOT EXISTS DATABASE (ID INTEGER, NAME TEXT, STALL TEXT, IMAGE TEXT)');
    //db.execute('DELETE FROM DATABASE');

    db.execute('INSERT INTO DATABASE (ID, NAME , STALL , IMAGE) VALUES(?,?,?,?)',1,'Anno','Stall No: 10','/images/KS_nav_ui.png');
    db.execute('INSERT INTO DATABASE (ID, NAME , STALL , IMAGE) VALUES(?,?,?,?)',2,'Kotha','Stall No: 12','/images/KS_nav_ui.png');
    db.execute('INSERT INTO DATABASE (ID, NAME , STALL , IMAGE) VALUES(?,?,?,?)',3,'Khushi','Stall No: 110','/images/KS_nav_ui.png');
    db.execute('INSERT INTO DATABASE (ID, NAME , STALL , IMAGE) VALUES(?,?,?,?)',4,'Rodela','Stall No: 99','/images/KS_nav_ui.png');
    db.execute('INSERT INTO DATABASE (ID, NAME , STALL , IMAGE) VALUES(?,?,?,?)',5,'Kona','Stall No: 55','/images/KS_nav_ui.png');
    db.execute('INSERT INTO DATABASE (ID, NAME , STALL , IMAGE) VALUES(?,?,?,?)',6,'Koishor','Stall No: 50','/images/KS_nav_ui.png');
    var rows = db.execute('SELECT * FROM DATABASE');
    var data =[];
while (rows.isValidRow()){


    var images=rows.fieldByName('IMAGE');
    var row = Ti.UI.createTableViewRow();

    var image = Ti.UI.createImageView({
        image:images,
        left:0,
        top:0,
        width:50,
        height:50
    });
    var aLabel = Ti.UI.createLabel({
        text : rows.fieldByName('ID')+"  "+rows.fieldByName('NAME')+"   "+rows.fieldByName('STALL'),
        color : 'red',
        font : {fontSize:12},
        height : Ti.UI.SIZE,
        width : Ti.UI.SIZE,
        top : 0,
        left : 70,
    });

    row.add(image);
    row.add(aLabel);
    data.push(row);

rows.next();
 }
rows.close();
db.close();

var tableview = Titanium.UI.createTableView({
    data:data,
});

win.add(tableview);
win.open();
