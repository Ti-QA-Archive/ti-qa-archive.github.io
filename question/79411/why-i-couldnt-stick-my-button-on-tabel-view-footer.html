    var footerButton = Ti.UI.createButton({

    title:'Add Row',
    color:'#191',
        left:125,
    width:'auto', //cannot be filled with 'auto',  you must type the value for iphone
    height:'auto' //cannot be filled with 'auto',  you must type the value for iphone
});
