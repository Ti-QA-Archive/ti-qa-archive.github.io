Image Scaling - IPhone

function createImage(img) {
    /*
     * Create the left Image
     */
    var left_image = Ti.UI.createImageView({
        left : 5,
        width : 100,
        height : 40,
        image : img

    });
    return left_image;
}

I am using this function to add a left image to a table view, but the images don't scale.
Seems to work on Andriod.
