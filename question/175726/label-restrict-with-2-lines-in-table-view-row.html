    Try this :

    index.xml

    <Alloy>
    <Window id="index">
        <TableView id="listtable">
            <TableViewRow>
                <Label class="subtitle1">Established in 1999, The Tatami Shop Singapore lead to communicate the benefits of natural Igusa Tatami to local</Label>
            </TableViewRow>
            <TableViewRow>
                <Label class="subtitle1">Established in 1999, The Tatami Shop Singapore lead to communicate the benefits of natural Igusa Tatami to local</Label>
            </TableViewRow>
            <TableViewRow>
                <Label class="subtitle1">Established in 1999, The Tatami Shop Singapore lead to communicate the benefits of natural Igusa Tatami to local</Label>
            </TableViewRow>
        </TableView>
    </Window>
</Alloy>

    index.tss

    "TableViewRow":{
    height: Ti.UI.SIZE,
    hasChild : "true"
}

"#subtitle1":{
    color: "#000",
    height: "auto",
    textAlign : "left"
}
