Accessing a MasterView table from ApplicationView

Hi,

Here is the partial of MasteView code.

function MasterView() {
    var self = Ti.UI.createView({
        backgroundColor:'#ccc'
    });

    var table = Ti.UI.createTableView();

    self.add(table);
    table.addEventListener('click', function(e) {
        self.fireEvent('itemSelected', { id: e.row.id });
    });

    self.refreshRssTable = function(data) {
        if (Object.prototype.toString.apply(data) === '[object Array]') {
            var rows = [];
            for (var i = 0; i < data.length; i++) {
                rows.push(createTableRow(data[i]));
            }
            table.setData(rows);
        }
    };

    return self;
}
module.exports = MasterView;

I want to set the header for this table from the applicationview to enable pull to rrefresh. i am stuck because i have no idea how to access the table from the applicationview. I am new to this framework and commonJS.

I have tried to implement directly but it gives me an error message saying it's not a valid object.

any help is appreciated here.
