    James

    Why do you need a view? A scrollView is already a view.

    A label is designed to constrict to the boundaries of the text it contains by default, whereas a view consumes any unused area.

    The following code works:

    var window = Ti.UI.createWindow({  
    backgroundColor:'red'
});

var scrollView = Ti.UI.createScrollView({
    contentWidth: 'auto',
    contentHeight: 'auto',
    top: 0,
    showVerticalScrollIndicator: true,
    showHorizontalScrollIndicator: false,
    layout: 'vertical',
    scrollType: "vertical"
});

var title = Ti.UI.createLabel({
    text: "This is a title",
    backgroundColor: '#FFF',
    color: '#323232',
    font: {
        fontSize: 18,
        fontWeight: 'bold'
    },
    top: 20
});

var label = Ti.UI.createLabel({
    backgroundColor:"blue",
    text: 'Lorem ipsum dolor sit amet, consectetur adipisicing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua. Ut enim ad minim veniam, quis nostrud exercitation ullamco laboris nisi ut aliquip ex ea commodo consequat.     color: '#000',

    },

});

var footer = Ti.UI.createLabel({
    text: "This is a footer",
    backgroundColor: '#FFF',
    color: '#323232',
    font: {
        fontSize: 18,
        fontWeight: 'bold'
    },
    top: 20
});

scrollView.add(title);
scrollView.add(label);
scrollView.add(footer);
window.add(scrollView);
window.open();

  • Hi, have you tried to do a wordWrap = true on your label, then add the label to your scroll view. For both your UI labels and scroll views, set the height to 'auto'. You should be able to decide what you want the width of your label to be. If you just want to display text, you could set it to 'auto' as well.
