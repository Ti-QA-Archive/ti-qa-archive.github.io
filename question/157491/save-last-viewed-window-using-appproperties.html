  • try to do some if on your code to check it

    e.g:

    // app.js
var winDynamic = Ti.UI.createWindow({
    backgroundColor: 'white',
    fullscreeen: true
})


if( Ti.App.Properties.hasProperty( 'whatWindow' ) == 1 ){
    var result = Ti.App.Properties.getInt( 'whatWindow' );

    if( result == 1 ){
        winDynamic.url = 'window1.js';

        winDynamic.open();

    }else if( result == 2 ){
        winDynamic.url = 'window2.js';

        winDynamic.open();
    }

}else{
    // do not have property saved yet
}


///////////////////////////////////////////////////////////////
// on some on your windows.js
var currentWindow = 1;

Ti.App.Properties.setInt( 'whatWindow', currentWindow );
