    you probably want the web view in a container window, and add the close button to the container and not the webview

    var win1 = Titanium.UI.createWindow({
    title:'win1',
});

var webView = Ti.UI.createWebView({
   url: 'www.cnn.com',
   backgroundColor:'purple'
});
var b = Ti.UI.createButton({
   title:'Close',
   width:100,
   height:30
});
b.addEventListener('click',function(){
   win1.close();
});
win1.add(webView);
win1.rightNavButton(b);
win1.open({modal:true});
