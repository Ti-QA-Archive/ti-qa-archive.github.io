    function listObjects(ip_object) {

    if (ip_object.children && ip_object.children[0]) {
        for (var i = 0; i < ip_object.children.length; i++) {
    // Do your stuff with the ip_object.children[i]
        };
    };
};

    You can even call the function recursively to scan childs of childs etc …

    • Raju, you need to make sure that you keep the test for children existence 

      if (ip_object.children && ip_object.children[0])

      Some objects do not have children (like a button for example) so if you try to manipulate or get information about non existent child, you'll get the 'undefined' error message.
      If you are trying to call recursively the function pasted above, make sure you repeat the test for child existence. Please see below the complete code which would scan everything down to the last object.

      function listObjects(ip_object) {

    if (ip_object.children && ip_object.children[0]) {
            for (var i = 0; i < ip_object.children.length; i++) {
            if (ip_object.children[i].children && ip_object.children[i].children[0]) {
                listObjects(if (ip_object.children[i])
            }
            else {
                // Do your stuff with the ip_object.children[i]
            };
        };                
    };
};
