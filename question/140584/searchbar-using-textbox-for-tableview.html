var search = Titanium.UI.createTextField({
    color:'#336699',
    top:0,
    left:10,
    width:300,
    height:40,
    hintText:'Search for ...',
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
    keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
