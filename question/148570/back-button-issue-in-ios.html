app.js

var win = Ti.UI.createWindow({
    backgroundColor:'#fff',
    tabBarHidden:true
    });

/ create and set right button
var rBtn = Ti.UI.createImageView({title: 'Settings', image:'images/settings.png'});
win.setRightNavButton(rBtn);

// handle rBtn clicks
rBtn.addEventListener('click', function() {
    var win1 = Ti.UI.createWindow({
        barImage: "images/bar_image_1.png",
        backgroundColor:'#000',
        title: 'Settings',
        url: "settings.js",
        tabBarHidden:true
    });   
    tab.open(win1, {});

    image:'images/Blog_80x80.png',
    top:90,
    right:35

        barImage: "images/bar_image_1.png",
        backgroundColor:'#fff',
        title: 'Blogs',
        url: "blog.js",
        tabBarHidden:true
    });
    tab.open(win3, {});
});

var label2 = Ti.UI.createLabel({
    text:'Blog',
    color:'#6EBB1F',
    width:'auto',
    height:'auto',
    font:{fontSize:16,fontFamily:'Trebuchet MS',fontWeight:'bold'},
    top:180,
    right:58
});

win.add(image2);
win.add(label2);

var tg = Ti.UI.createTabGroup({});
var tab = Ti.UI.createTab({window:win});
tg.addTab(tab);
tg.open();

blog.js

var win3 = Ti.UI.currentWindow;

var webview = Titanium.UI.createWebView({
