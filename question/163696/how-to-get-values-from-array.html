    Hello Sabari,

    try this :

    var data = [{
    "key" : "0",
    "value" : "111",
    "id" : 1019
}, {
    "key" : "0",
    "value" : "111",
    "id" : 1019
}, {
    "key" : "0",
    "value" : "111",
    "id" : 1019
}];

var dic = [];

for (i in data) {
    dic.push(data[i].value);
}

alert(JSON.stringify(dic));

    i hope this helps. thanks.
