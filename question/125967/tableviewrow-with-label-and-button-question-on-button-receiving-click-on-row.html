tableViewRow with label and button... Question on button receiving click on row.

Hi… Simple TableView with TableViewRows… Each row has a label and a button. The button is firing its click event when the row in its entirety is clicked upon, rather just the button itself. How can this be resolved? As an example: If I click anywhere on the row, the button (which has backgroundImage and backgroundSelectedImage properties set) is alternating these states and firing click. Shouldn't this be limited to a click on the button itself?

  • Below… Like i said though, the code isn't complex (simple)…

    var tableView = Ti.UI.createTableView({
    minRowHeight:60,
    width:320,
    height:410,
    bottom:0,
    separatorStyle: Titanium.UI.iPhone.TableViewSeparatorStyle.NONE,
    backgroundColor:"transparent"
    });

    winInvitenco.add(tableView);

    var result = Titanium.Contacts.getAllPeople();
var data = [];

    for (var c=0;c<result.length;c++)

    {
        var row = result[c];

        var tvRow = Ti.UI.createTableViewRow({
            height:'auto',
            width:320,
            backgroundImage: "images/invitecellbkg@2x.png",
            selectedBackgroundColor:'transparent'

        });

        var imageView;
        imageView = Ti.UI.createImageView({
            image:row.image === null ? '../images/user.png' : row.image,
            left:22,
            width:34,
            height:32,
            bottom:15


        });

        tvRow.add(imageView);


        var userLabel = Ti.UI.createLabel({
            font:{fontSize:13, fontWeight:'bold'},
            left:70,
            right:5,
            height:20,
            color:'#000',
            text:row.firstName


        });

        tvRow.add(userLabel);

        var invButton = Ti.UI.createButton({
                backgroundImage: "images/invitebutton@2x.png",
                backgroundSelectedImage: "images/invitehibutton@2x.png",
                font:{fontSize:11, fontWeight:'bold'},
                width:60,
                height:25,
                right:15,
                title: "Invite",
                color: "#fff"

        });

        tvRow.add(invButton)

        tvRow.uid = row.uid;

        data[c] = tvRow;



}        
    tableView.setData(data, { animationStyle : Titanium.UI.iPhone.RowAnimationStyle.DOWN });

    winInvitenco.open({transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT});
