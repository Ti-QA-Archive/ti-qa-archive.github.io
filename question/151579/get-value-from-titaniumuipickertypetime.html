  • The picker's value property will give you the selected date or time with datetime pickers. Like this:

    inputHoraInicio.addEventListener('change', function(e) {
  Ti.API.info(inputHoraInicio.value);
}

  • Hi Fernando,

    you can get the time value using this :

    picker1 = Titanium.UI.createPicker({
     type:Titanium.UI.PICKER_TYPE_TIME,
    minDate:new Date(2009,0,1),
    maxDate:new Date(2014,11,31),
    value:new Date(),
});
win.add(picker1);

picker1.addEventListener('change', function(e) {
         var hours =  e.value.getHours();
});

  • Hello Ferbnando,

    Try this : 

    var inputHoraInicioValue, inputHoraFinValue;

var inputHoraInicio = Ti.UI.createPicker({
    type : Titanium.UI.PICKER_TYPE_TIME,
    //format24 : true,
    left : "12%",
    height : "auto",
    top : '420px',
    transform : t
});
inputHoraInicio.selectionIndicator = true;

inputHoraInicio.addEventListener('change', function(e){
    inputHoraInicioValue = e.value
});

var inputHoraFin = Ti.UI.createPicker({
    type : Titanium.UI.PICKER_TYPE_TIME,
    //format24 : true,
    left : "12%",
    height : "auto",
    top : '610px',
    transform : t
});
inputHoraFin.selectionIndicator = true;

inputHoraFin.addEventListener('change', function(e){
    inputHoraFinValue = e.value
});

nuevoEvento.addEventListener("click", function(e) {
        ...
        var hora_inicio = inputHoraInicioValue;
        var hora_fin = inputHoraFinValue;
        ...
}
