  • Here's a simple way to do a typewriter effect in a text label using susbstrings and intervals:

    
var win = Ti.UI.currentWindow;

var text_label = Ti.UI.createLabel({
    text:' ',
    top:245,
    left:10,
    right:10,
    borderWidth:1,
    height:'auto'
});
win.add(text_label);

var some_long_text = 'The quick brown fox jumps over the lazy dogs!';

var c = 0;
var timer;

function typewriter(){

    text_label.text = some_long_text.substr(0, c);

    if(c==some_long_text.length){        
        clearInterval(timer);
        Ti.API.info('done!');
    }

    c++;
}

timer = setInterval(typewriter,250); // trigger the typewriter function every 250ms

    ;) hope this helps.
