      // Create UI elements
var win = Ti.UI.createWindow({});
var btn = Ti.UI.createButton({title : 'Upload photo'});

// Executed when the user hits the upload photo-button
btn.addEventListener('click', function () {
    // Create the dialog
    var cameraPrompt = Ti.UI.createAlertDialog({
        // Insert title and empty message
        title : 'Upload photo',
        message : 'Do you want to upload a photo from your gallery or snap a new one with the camera.',

        // Add buttons
        buttonNames : ['From gallery', 'With camera']
    });

    // Bind an event-listener to the dialog
    cameraPrompt.addEventListener('click', function (e) {
        // Take photo with camera?
        if (e.index == 1) {
            // Open the camera
            Ti.Media.showCamera({
                // Only capture images
                mediaTypes : [Ti.Media.MEDIA_TYPE_PHOTO],

                // Bind events
                success : onSuccess,
                error : onError,
                cancel : onError
            });

        // ... Or select photo from gallery?
        } else if (e.index == 0) {
            // Open the gallery
            Ti.Media.openPhotoGallery({
                // Only capture images
                mediaTypes : [Ti.Media.MEDIA_TYPE_PHOTO],

                // Bind events
                success : onSuccess,
                error : onError,
                cancel : onError
            });
        }
    });

    // Show the dialog
    cameraPrompt.show();

    // When the user has successfully captured an image, save it temporarily to
    // the local filesystem - and upload it to the server 
    function onSuccess (event) {
        // Store the image-data into a temporary file
        var f = Ti.Filesystem.getFile(Ti.Filesystem.tempDirectory, 'photo.jpg');
        f.write(event.media);

        // Store the image-data into a temporary file
        var f = Ti.Filesystem.getFile(Ti.Filesystem.tempDirectory, 'thumb.jpg');
        f.write(event.media.imageAsThumbnail(40, 0));

        // Next we need to actually upload the thumb to the MoodPik API
        var uploadClient = Ti.Network.createHTTPClient({
            // We only want to register data upon load
            onload : function () {
                // Parse the response
                var response = JSON.parse(this.responseText);

                // Make sure everything went OK
                if (response.success) {
                    // Register the photo ID
                    photo_id = response.photoId;

                // Alert the user if an error occurred
                } else {
                    // Alert the user of the error
                    if (response.message) {
                        // Create the alert dialog
                        Ti.UI.createAlertDialog({
                            // Insert title and message
                            title : 'Error!',
                            message: response.message,

                            // Insert OK-button
                            buttonNames : ['OK']
                        }).show();
                    }
                }
            },

            // On error we alert the user
            onerror : function () {
                // Create the alert dialog
                Ti.UI.createAlertDialog({
                    // Insert title and message
                    title : 'Error!',
                    message: 'An error occurred while uploading your photo to the server.',

                    // Insert OK-button
                    buttonNames : ['OK']
                }).show();
            },

            // Set a timeout of 120 seconds
            timeout : 120000
        });

        // Prepare the request
        uploadClient.open('POST', '[API_BASE]/images/create.json');

        // Send request to the API
        uploadClient.send({
            photoToUpload : event.media
        });
    }

    // If an error occurs, then alert the user
    function onError (e) {
        // Create the alert dialog
        Ti.UI.createAlertDialog({
            // Insert title and message
            title : 'Error!',
            message: 'An error occurred while capturing a photo.',

            // Insert OK-button
            buttonNames : ['OK']
        }).show();
    }
});

// Open window
win.add(btn);
win.open();
