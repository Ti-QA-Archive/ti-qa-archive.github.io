    Hey May,

    Try this : 

    dataArray = [];
//Insert the JSON data to the table view
for (var i = 0; i < json.length; i++) {
    var row = Ti.UI.createTableViewRow({
        title : json[i].nom,
        hasChild : true,
        obj : json[i].titrePro,
        obj1 : json[i].adresse,

    });
    dataArray.push(row);
};
$.tableView.setData(dataArray);

$.tableView.addEventListener('click', function(e) {
    var alertDialog = Titanium.UI.createAlertDialog({
        title : 'A propos ' + e.row.title,
        message : '\nTitre Pro : ' + e.row.obj + '\n\nAdresse : ' + e.row.obj1 + '\n',
        buttonNames : ['Rejeter', 'Accepter'],
        cancel : 1

    });
    alertDialog.addEventListener('click', function(ev) {
        if (ev.cancel === true) {
            Titanium.API.info("Accept button was clicked !");
            $.tableView.deleteRow(e.index);
        }
    });
    alertDialog.show();
});
