    Ah yes - A spot of recursion is required…

    var idx = 0, filtrarSexo = [], tot = arrayUsuarios.length;

function doRequest() {

    if (idx == tot){
        populaTableView();
        return;
    }
    fb.requestWithGraphPath(arrayUsuarios[i].id, {}, 'GET', function(e) {

        if (e.success) {

            var t = JSON.parse(e.result);

            var row = {
                nome : t.name,
                sexo : t.gender,
                idFace : t.username,
                id : t.id
            };

            filtrarSexo.push(row);
        }

        idx++;
        doRequest();

    });
}
