TabGroup position

Hi,

Is there a way to place the TabGroup at the bottom of the page instead of the top?

No matter what I try, I can't change its position, or even its height/left/top parameters.. it seems like no matter what I try, it just stays at the same place.

This is what I have right now. I'm developing for Android.

// create tab group
var tabGroup = Titanium.UI.createTabGroup({
    top:40,
    left:10,
    height:230
});

var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#000'
});
var tab1 = Titanium.UI.createTab({  
    //icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});

var win2 = Titanium.UI.createWindow({  
    title:'Tab 2',
    backgroundColor:'#fff'
});
var tab2 = Titanium.UI.createTab({  
   // icon:'KS_nav_ui.png',
    title:'Tab 2',
    window:win2
});

tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);  
tabGroup.open();
