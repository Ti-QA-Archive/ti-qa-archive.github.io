  • Ok, I got it.

    It seems that you MUST fill the FROM field or the email never reach the destination. Just do this:

    var Cloud = require('ti.cloud');
Cloud.Emails.send({
    template: 'test',
    from: 'my@email.com',
    recipients: 'nabil.mughal00@gmail.com',
    //message:'nabeel'
}, function (e) {
    if (e.success) {
        alert('Success');
    } else {
        alert('Error:\\n' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});
