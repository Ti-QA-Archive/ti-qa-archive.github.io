Open url in safari when clicked on tab in tabgroup

Hi All,

I wonder if it is possible to open a url in mobile safari when there is clicked on a specific tab in the tabgroup.

var workplace_tab = Titanium.UI.createTab({
icon:'/assets/img/icon-workplace.png',
title: 'Studieplek'
});

workplace_tab.addEventListener('click', function(e) {
alert('w00t');
Ti.Platform.openURL("http://www.snelafstuderen.nl/methode/scriptielab/");
});

The code above doesn't work.
Could anyone tell me what I am doing wrong?

    Try something like this:

    
var workplace_tab = Titanium.UI.createTab({ 
    icon:'/assets/img/icon-workplace.png', 
    title: 'Studieplek',
    role:'open_web' 
});

// tabGroup is your tabgroup

tabGroup.addEventListener('focus', function(e) {
    if (e.tab.role == 'open_web') {
        alert('woot');
    }
});
