    • Here you go this works fine for increments of 10, if you make any changes in the number of ros you have to adjust the code. 

      {//DYNAMIC TABLE
    var updating = false;
    var lastRow = 10;
    var countingPages = 1;
    var loadingRow = Ti.UI.createTableViewRow({title:"Loading...", height: 65});

    function beginUpdate(){
        updating = true;

        tableView.appendRow(loadingRow);

        params.page++;

        // just mock out the reload
        setTimeout(loadPosts,2000);
    }

    var lastDistance = 0; // calculate location to determine direction
    var  params = {
            dev : 1,
            count : 10,
            id : 261,
            page: 1
    };
    var tabData = [];
    // Create the table view and set its data source to "rowData" array
    var tableView = Titanium.UI.createTableView({
        data : [],
        hasDetail : true,
    });

    tableView.addEventListener('scroll',function(e){
        var offset = e.contentOffset.y;
        var height = e.size.height;
        var total = offset + height;
        var theEnd = e.contentSize.height;
        var distance = theEnd - total;

        // going down is the only time we dynamically load,
        // going up we can safely ignore -- note here that
        // the values will be negative so we do the opposite
        if (distance < lastDistance){
            // adjust the % of rows scrolled before we decide to start fetching
            var nearEnd = theEnd * .75;

            if (!updating && (total >= nearEnd)){
                beginUpdate();
            }
        }
        lastDistance = distance;
    });


    //Add the table view to the window
    win.add(tableView);
    function loadPosts() {

        // Create our HTTP Client and name it "loader"
        var loader = Titanium.Network.createHTTPClient();

        // Runs the function when the data is ready for us to process
        loader.onload = function() {
            var postsdata = JSON.parse(this.responseText);
            Ti.API.info(countingPages)
            if (postsdata.pages <= countingPages){
                tableView.deleteRow(lastRow-params.count,{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.NONE});
                return;
            }else{
                countingPages++;
            }


            for(var i = 0; i < postsdata.posts.length; i++) {
                var id = postsdata.posts[i].id;
                var title = postsdata.posts[i].title;
                var content = postsdata.posts[i].content;

                var date = postsdata.posts[i].date;

                if(postsdata.posts[i].excerpt == "") {
                    var limit_excerpt = 'Excerpt not available';
                } else {
                    var excerpt = postsdata.posts[i].excerpt;
                    var limit_excerpt = excerpt.substr(0, 75);
                }

                // The screen name of the user
                //var Picture = postsdata.posts[i].profilepicture;
                // The profile image
                // Create a row and set its height to auto
                var row = Titanium.UI.createTableViewRow({
                    height : 65,
                    hasChild : true,
                    id : id,
                    posttitle : title
                });
                // Create the view that will contain the text and avatar
                var post_view = Titanium.UI.createView({
                    height : 65,
                    layout : 'vertical',
                    top : 5,
                    right : 5,
                    bottom : 5,
                    left : 5,
                    height : 65,
                });
                // Create image view to hold profile pic
                var av_image = Titanium.UI.createImageView({
                    image : 'imgs/admin.thumbnail.jpg', // the image for the image view
                    top : 0,
                    left : 0,
                    height : 45,
                    width : 45
                });
                post_view.add(av_image);

                // Create the label to hold the screen name
                var user_lbl = Titanium.UI.createLabel({
                    text : title,
                    left : 54,
                    width : 240,
                    top : -45,
                    bottom : 2,
                    height : 16,
                    textAlign : 'left',
                    color : '#0b324f',
                    font : {
                        fontFamily : 'Trebuchet MS',
                        fontSize : 16,
                        fontWeight : 'bold'
                    }
                });
                post_view.add(user_lbl);

                row.filter = user_lbl.text;
                // Create the label to hold the tweet message
                var tweet_lbl = Titanium.UI.createLabel({
                    text : limit_excerpt,
                    left : 54,
                    top : 0,
                    //bottom : 0,
                    color : '#333333',
                    //height : 'auto',
                    width : 240,
                    textAlign : 'left',
                    font : {
                        fontFamily : 'Arial',
                        fontSize : 11
                    }
                });
                post_view.add(tweet_lbl);

                var date = Titanium.UI.createLabel({
                    text : 'Posted on: ' + date,
                    left : 54,
                    top : -2,
                    bottom : 1,
                    //height : 'auto',
                    width : 200,
                    color : '#333',
                    textAlign : 'left',
                    font : {
                        fontFamily : 'Arial',
                        fontSize : 10
                    }
                });

                post_view.add(date);
                row.add(post_view);
                tabData[i] = row;
            }
            updating = false;

            tableView.deleteRow(lastRow-params.count,{animationStyle:Titanium.UI.iPhone.RowAnimationStyle.NONE});

            tableView.appendRow(tabData);

            lastRow += params.count;
        };

        loader.open("GET", "http://injustices.in/api/get_category_posts/");

        // Send the HTTP request
        loader.send(params);

    }

    loadPosts();


}

  • I'm not sure to get your problem.
    What I understood : you don't manage to send correct parameters values in order to get page 2, page 3, page 4, etc.

    What you have to do for this is simply to externalize your parameters data :

    var devVar = 1;
var countVar = 10;
var idVar = 261;
var pageVar = 1;

    And you send it this way : 

    xhr.send({dev:devVar,count:countVar,id:idVar,page:pageVar});

    In the "onload" of your HTTP client you simply have to increment your pageVar value.
    Then you can call the same send line : as page variable was updated, it will return the good result with the new page.
