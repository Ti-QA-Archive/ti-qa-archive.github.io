    ####app.js

    win = Ti.UI.createWindow({
    backgroundColor: '#eee',
    layout: 'vertical'
});

var btn1 = Ti.UI.createButton({
    title: 'sax',
    url: 'sax.mp3'
})
btn1.addEventListener('click', function(e) {
    playsound(btn1.url);
});
win.add(btn1);

var btn2 = Ti.UI.createButton({
    title: 'chime',
    url: 'chime.mp3'
})
btn2.addEventListener('click', function(e) {
    playsound(btn2.url);
});
win.add(btn2);

var btn3 = Ti.UI.createButton({
    title: 'tweet',
    url: 'tweet.mp3'
})
btn3.addEventListener('click', function(e) {
    playsound(btn3.url);
});
win.add(btn3);

var player = Ti.Media.createSound();

var playsound = function(url) {
    player.stop();
    player = Ti.Media.createSound({url:url});
    player.play();
};

win.open();

  • As your code is, you won't ever free up memory because you are creating the sounds and storing them in variables for the life of the program.

    var button2_sound = Titanium.Media.createSound({
    url:'cricket2.wav',
    preload:true
});

    On the button click event you need to dynamically create and play the sound. When that function completes the sound will be garbage collected.

    As for stoping an existing sound, keep an array of playingSounds. In the button click event stop any sounds that are playing before starting a new one.
