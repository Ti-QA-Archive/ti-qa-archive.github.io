table view does not work

I'm trying to develop a prototype of my application for which I need a list of rows to be displayed. The code is as below

// this sets the background color of the master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');

//create the root window

var win_root = Titanium.UI.createWindow();

//
// create practice selection window
//
var win_practice_select = Titanium.UI.createWindow({
title:'practice',
backgroundColor:'#fff'
});

//search bar

var search_practice = Titanium.UI.createSearchBar({
barColor:'#000',
showCancel:false,
height:43,
top:0,
});

search_practice.addEventListener("change", function(source, type){
search_practice.setShowCancel(true, {animated:true});
});

search_practice.addEventListener("cancel", function(source, type){
search_practice.setValue("");
search_practice.setShowCancel(false, {animated:true});

});

win_practice_select.add(search_practice);

//selected practices

var data = [{title:"Row 1"},{title:"Row 2"}];
var table = Titanium.UI.createTableView({data:data});

//win_practice_select.add(tbl_practice_history);

//navigation group
var nav_root = Titanium.UI.iPhone.createNavigationGroup({
window:win_practice_select
});

win_root.add(nav_root);

win_root.open();

// open the practice selction window
nav_root.open(win_practice_select, {animated:true});

The above code crashes with the error message saying

[WARN] attempted to load: TiUITableViewProxy
[ERROR] Script Error = invalid method (createTableView) passed to UIModule at app.js (line 45).

I read another post with a similar problem for which he did a reinstall of the titanium studio. I dont think that is a feasible solution in all cases. I'm using Iphone SDK 5.1, OSx LION, Titanium Studio, build: 2.0.1.201204132053 and have installed xcode 4.3.2.

Do we have a solution for this problem?

  • I did few things and it started working

    instead of the simulator I choose to run on the iphone device itself.
    I did the below changes in the code

    // this sets the background color of the master UIView (when there are no windows/tab groups on it)
    Titanium.UI.setBackgroundColor('#000');

    //create the root window

    var win_root = Titanium.UI.createWindow();

    //
    // create practice selection window
    //
    var win_practice_select = Titanium.UI.createWindow({
    title:'practice',
    backgroundColor:'#fff'
    });

    //search bar

    var search_practice = Titanium.UI.createSearchBar({
    barColor:'#000',
    showCancel:false,
    height:43,
    top:0,
    });

    search_practice.addEventListener("change", function(source, type){
    search_practice.setShowCancel(true, {animated:true});
    });

    search_practice.addEventListener("cancel", function(source, type){
    search_practice.setValue("");
    search_practice.setShowCancel(false, {animated:true});

    });

    var data = [{title:"Row 1"},{title:"Row 2"}];
    var table = Titanium.UI.createTableView({data:data});

    win_practice_select.add(search_practice);

    win_practice_select.add(table);

    //navigation group
    var nav_root = Titanium.UI.iPhone.createNavigationGroup({
    window:win_practice_select
    });

    win_root.add(nav_root);

    win_root.open();

    // open the practice selction window
    nav_root.open(win_practice_select, {animated:true});

    It started working
