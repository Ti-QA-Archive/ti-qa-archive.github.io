  • Accepted Answer

    Here is a quick mockup:

    var view = Ti.UI.createView({
    height: 100,
    layout: 'horizontal',
    width: Ti.UI.SIZE
});

function fixed() {
    var blueFixed = Ti.UI.createView({
        backgroundColor: 'blue',
        height: Ti.UI.FILL,
        width: 15 // set fixed width
    });
    return blueFixed;
}

var greenView = Ti.UI.createView({
    backgroundColor: 'green',
    height: Ti.UI.FILL,
    width: Ti.UI.SIZE // this will assume the width of  any contents, or you could set the value specifically
});

var pinkView = Ti.UI.createView({
    backgroundColor: 'pink',
    height: Ti.UI.FILL,
    width: 100 // fixed width
});

view.add(fixed());
view.add(greenView);
view.add(fixed());
view.add(pinkView);
view.add(fixed());
