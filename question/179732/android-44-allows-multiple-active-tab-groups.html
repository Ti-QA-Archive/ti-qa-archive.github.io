Android 4.4 allows multiple active tab groups

Android 4.4 allows multiple active tab groups, as verified by this code:

(function() {
    "use strict";

    function TabGroup(Window, tabs) {
        var i, self = Ti.UI.createTabGroup();

        function addTab(title) {
            var win = new Window(title),
                tab = Ti.UI.createTab({
                    title: title,
                    window: win
                });
            self.addTab(tab);
        }

        for (i = 0; i < tabs.length; i = i + 1) {
            addTab(tabs[i]);
        }

        return self;
    }

    function PhoneWindow(title) {
        var self = Ti.UI.createWindow({
            title:title,
            backgroundColor:'white'
        }),

        button = Ti.UI.createButton({
            height:44,
            width:200,
            title: 'New Tab Group',
            top:20
        });

        self.add(button);

        button.addEventListener('click', function() {
            var tabs = [title+'1', title+'2', title+'3'];
            new TabGroup(PhoneWindow, tabs).open();
        });

        return self;
    }

    new TabGroup(PhoneWindow, ['A', 'B', 'C']).open();
}());

When was this restriction removed and could we count on using multiple active tab groups without significantly limiting the user base? Thanks in advance.
