setLocation() only works in one way

Hi,
I try to zoomion and out – here is my snippet:

switch (status.zoomed) {
    case false:   // Zoom to my position:
        ctrl.stations.getPosition(function(e) {
            mapView2.setLocation({
                latitude : e.latitude,
                longitude : e.longitude,
                latitudeDelta : 0.04,
                animate : true,
                longitudeDelta : 0.04
            });
            optionsDialogOpts.options[0] = 'Kartenüberblick';
            dialog.setOptions(optionsDialogOpts.options);
            status.zoomed = true;
        });
    break;
    case true:   // Zoom to overView
        mapView2.setLocation({
            latitude : 54.2,
            longitude : 8.5007,
            animate : true,
            latitudeDelta : 0.5,
            longitudeDelta : 0.5
        });
        optionsDialogOpts.options[0] = 'Eigener Standort';
        dialog.setOptions(optionsDialogOpts.options);
        status.zoomed = false;
    break
}

My problem: to map only zoomed in. but not out. With other words, the second case doesnt work.

Rainer
