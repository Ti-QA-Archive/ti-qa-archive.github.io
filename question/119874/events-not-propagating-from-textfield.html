Events not propagating from TextField

Hi All,

I'm trying to set up a table with editable rows and the only way I know to do this is to add a TextField to the TableViewRow. I want to be able to know which row is being edited when the user begins and ends editing but the events from the TextField are not propagating either to the row or to the underlying table. Here's (roughly) how I'm coding it:

function fieldEvents(field) {
    field.addEventListener('click', function(e) {
        Ti.API.info("I haz da click." + JSON.stringify(e));
    });
    field.addEventListener('blur', function(e) {
        Ti.API.info("I been blurred." + JSON.stringify(e));
    });
}

function setData() {
    var data = [];
    for (var i=0; i<30; i++) {
        var row = Ti.UI.createTableViewRow({ 
            id: i,
            hasChild: true
        });
        var field = Ti.UI.createLabel({
            id: i,
            text: '', 
            font: { fontSize: 16, fontFamily: 'Trebuchet MS' }, 
            color: 'black', 
            left: 25
        });
        fieldEvents(field);
        row.add(field);
        data.push(row);
    }
    return data;
}

var tableView = Titanium.UI.createTableView({
    data: setData()
});

Any advise would be appreciated.
