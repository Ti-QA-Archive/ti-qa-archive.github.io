iOS doesn't send geolocation when in standby

Is there a fix/work around to get the iOS devices to start sending geolocations when they're on standby?
(Titanium SDK 3.0.0, iOS 4.3+ and Android 2.2+(But that works))
Relevant code:

Ti.Geolocation.purpose = 'Testing Location Services';
Ti.Geolocation.preferredProvider = Ti.Geolocation.PROVIDER_GPS;
var callback;
function locationHandler(e) {
    if (e.success === undefined || e.success) {
        if (typeof (callback) === 'function') {
            callback(e);
        }
    }
} 
function start(obj) {
    callback = obj.action;
    Ti.Geolocation.addEventListener("location", locationHandler);
}

function stop() {
    callback = undefined;
    Ti.Geolocation.removeEventListener("location", locationHandler);
}

exports.start = start;
exports.stop = stop;
