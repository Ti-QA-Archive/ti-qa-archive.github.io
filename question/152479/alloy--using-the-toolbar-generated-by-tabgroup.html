<Alloy>
    <TabGroup>
        <Tab title="Tab 1" icon="">
            <Window title="Tab 1">

                //THAT DOESNT WORK, IT ADD ANOTHER TOOL BAR IN THE VIEW
                <Toolbar top="0" platform="ios">
                    <Items>
                        <FlexSpace/>
                        <Button  width="20" heigth="20" onClick="logout" image="/ic_menu_home.png"/>
                    </Items>
                </Toolbar>    

                <Label>I am Window 1</Label>        
            </Window>
        </Tab>

        <Tab title="Tab 2" icon="">
            <Window title="Tab 2">
                <Label>I am Window 2</Label>
            </Window>
        </Tab>

    </TabGroup>
</Alloy>

    Like this?

    <Alloy>
    <TabGroup>
        <Tab title="Tab 1" icon="">
            <Window title="Tab 1">
                <RightNavButton>
                    <Button>Log Out</Button>    
                </RightNavButton>
                <Label>I am Window 1</Label>        
            </Window>
        </Tab>
        <Tab title="Tab 2" icon="">
            <Window title="Tab 2">
                <Label>I am Window 2</Label>
            </Window>
        </Tab>
    </TabGroup>
</Alloy>

    See more on properties that are assigned proxies.
