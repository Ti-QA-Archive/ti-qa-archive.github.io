  • Hi guys,

    Try this sample code and apply to your project. It works as expected, separator color show in Titanium SDK 3.3.0. I also used AppCompat.Light theme.

    Testing Environment:

    Titanium SDK: 3.3.0.GA

    Titanium CLI: 3.3.0,

    Android SDK: 4.3.0 galaxy s3, 4..2.2, 4.4.2

    app.js

    var win = Ti.UI.createWindow({backgroundColor: 'white'});
var listView = Ti.UI.createListView({
    separatorColor:'#000'
});
var sections = [];

var fruitSection = Ti.UI.createListSection({ headerTitle: 'Fruits'});
var fruitDataSet = [
    {properties: { title: 'Apple',
color:'#000'}},
{properties: { title: 'Banana',
color:'#000'}},
];
fruitSection.setItems(fruitDataSet);
sections.push(fruitSection);

var vegSection = Ti.UI.createListSection({ headerTitle: 'Vegetables'});
var vegDataSet = [
    {properties: { title: 'Carrots',
    color:'#000'}},
    {properties: { title: 'Potatoes',
    color:'#000'}},
];
vegSection.setItems(vegDataSet);
sections.push(vegSection);

listView.sections = sections;
win.add(listView);
win.open();

    platform/android/res/values/builtin_themes.xml

    <?xml version="1.0" encoding="utf-8"?>
<resources xmlns:android="http://schemas.android.com/apk/res/android">
    <style name="LightDarkBar" parent="Theme.AppCompat.Light.DarkActionBar"/>
    <style name="Light" parent="Theme.AppCompat.Light"/>
    <style name="Dark" parent="Theme.AppCompat"/>
</resources>

    tiapp.xml

    <android xmlns:android="http://schemas.android.com/apk/res/android">
    <manifest>
        <application android:theme="@style/Light"/>
        <uses-sdk android:minSdkVersion="14" android:targetSdkVersion="19"/>
    </manifest>
</android>

    If you have any question about this issue, then let me know.

    Thanks.
