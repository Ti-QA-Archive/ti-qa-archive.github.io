  • Here a module to retrieve infos from youtube:

    exports.get = function(_endpoint,_calback) {
    var self = Ti.Network.createHTTPClient({
        timeout : 15000
    });
    var url = 'http://gdata.youtube.com/feeds/api/' + endpoint + '?v=2&alt=json';
    xhr.onerror = function(_e) {
        _callback({
            success : false
        });
    };
    xhr.onload = function() {
        if (this.status == 200) {
            var res = JSON.parse(this.responseText);
            _callback({
                data : res,
                success : true
            });
        } else
            _callback({
                success : false
            });
    }
    xhr.open("GET", url);
    xhr.setRequestHeader("Content-Type", "application/json; charset=utf-8");
    xhr.send(null);
