    Hi a tak, could you try this to change your height of text area .

    var comp = Ti.UI.createTextArea(properties);

    done.addEventListener('click',function(e){

        comp.height = 300;
       // comp.setHeight(300);    
        value = comp.getValue();
        //comp.blur();
    });
