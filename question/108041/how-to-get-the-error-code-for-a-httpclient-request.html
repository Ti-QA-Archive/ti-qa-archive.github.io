  • I know this thread is pretty old, but since I had the same problem, this is my solution:

        function getErrorCode(e) {
        var errorInformations = e.error.split(" ");

        if( typeof String.prototype.startsWith != 'function') {
            String.prototype.startsWith = function(str) {
                return this.indexOf(str) == 0;
            };
        }

        for(var i in errorInformations) {
            var info = errorInformations[i];
            if(info.startsWith("Code=")){
                return info.slice(5,info.length);
            }
        }
    }
