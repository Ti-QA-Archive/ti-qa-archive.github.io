Continous Animation

here is my code.

var win1 = Titanium.UI.createWindow({
    title : 'Tab 1',
    backgroundColor : '#fff'
});

var tochangeleft = 10;
var tochangeright = 10;

var square = Titanium.UI.createView({  //this is the "character" controlled by the user
    backgroundImage : 'KS_nav_ui.png',
    width : 80,
    height : 80,
    top : 300,
    left : 200
})

var left_button = Titanium.UI.createButton({  //button to move left
    width : 80,
    height : 30,
    top : 10,
    left : 10,
    title : "left_button"
})
var right_button = Titanium.UI.createButton({  //button to move right
    width : 80,
    height : 30,
    top : 10,
    right : 10,
    title : "right_button"
})


var square_left = square.left

left_button.addEventListener('click', function() {
    animate_left(square)
})

right_button.addEventListener('click', function() {
    animate_right(square)
})
function animate_left(square) {             //makes the square animate to the left
    square_left = square_left - tochangeleft
    square.animate({
        left : square_left,
        duration : 500,
    });
}

function animate_right(square) {                //makes the square animate to the right
    square_left = square_left + tochangeright
    square.animate({
        left : square_left,
        duration : 500
    });
}

win1.add(right_button);
win1.add(left_button);
win1.add(square);

win1.open();

The problem with this code is that I have to repeatedly press the left or right buttons for the character to move to the left or right. I want to make it so that when the user presses and holds a button, the character moves continuously until the user releases the button. How do I do this? Thanks in advance.

    You could use touchstart/end events like Anthony said and set a boolean whether you are pressing the button or not
    Then in the Animation add a callback function which restarts animate_left/right if the boolean is true,

    You could give this a try :

    var win1 = Titanium.UI.createWindow({
    title : 'Tab 1',
    backgroundColor : '#fff'
});

var tochangeleft = 10;
var tochangeright = 10;

var square = Titanium.UI.createView({  //this is the "character" controlled by the user
    backgroundImage : 'KS_nav_ui.png',
    width : 80,
    height : 80,
    top : 300,
    left : 200
})

var left_button = Titanium.UI.createButton({  //button to move left
    width : 80,
    height : 30,
    top : 10,
    left : 10,
    title : "left_button"
})
var right_button = Titanium.UI.createButton({  //button to move right
    width : 80,
    height : 30,
    top : 10,
    right : 10,
    title : "right_button"
})


var square_left = square.left
var touching = false; 
left_button.addEventListener('touchstart', function() {
    animate_left(square)

    touching = true;
}) 
left_button.addEventListener('touchend', function() {
    touching = false;
}) 

right_button.addEventListener('touchstart', function() {
    animate_right(square)
    touching = true;
}) 
right_button.addEventListener('touchend', function() {
    touching = false;
}) 

function animate_left(square) {             //makes the square animate to the left
    square_left = square_left - tochangeleft
    square.animate({
        left : square_left,
        duration : 500,
    },function() {
        if(touching)
            animate_left(square);
    });
}

function animate_right(square) {                //makes the square animate to the right
    square_left = square_left + tochangeright
    square.animate({
        left : square_left,
        duration : 500
    },function() {
        if(touching)
            animate_right(square);
    });
}

win1.add(right_button);
win1.add(left_button);
win1.add(square);

win1.open();

      var animationLoop;

    animationLoop = setInterval(function(){animate_left(square)},100);

left_button.addEventListener('touchend', function() {

right_button.addEventListener('touchstart', function() {
