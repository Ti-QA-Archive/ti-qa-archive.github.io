    You can also use WebView to open .docx file

    var win = Ti.UI.createWindow();
var wv = Ti.UI.createWebView({
    url: 'http://examples.maartenballiauw.be/WordVisualizer/Documents/test2.docx'
})
win.add(wv);
win.open();
