var search = Titanium.UI.createSearchBar({
    showCancel:false,
});

var data = [
        {title:'Alan',filter:'filter cold'},
        {title:'Alice',filter:'filter coat'},
        {title:'Alexander',filter:'filter xander'},
        {title:'Brenda',filter:'filter bunny'},
        {title:'Brenda',filter:'filter hat'},
    ];
var mListaTableView = Titanium.UI.createTableView({
    data:data,
    search:search,
    separatorColor:'transparent',
    backgroundColor:'transparent',    
    allowsSelection:false,
    left:0,        
    top:0,
    height:Ti.UI.FILL,
    width:Ti.UI.FILL,    
    filterAtribute:'filter',                
});

  • Hello Ezoom,

    You can also try this…

    
var win = Ti.UI.currentWindow;

var data = ['Alan', 'Alice', 'Alexander', 'Brenda', 'Brendas'];
var newarr = [];

var search = Titanium.UI.createSearchBar({
    showCancel : false,
});

var tableview = Titanium.UI.createTableView({
    search : search,
    top : 0,
    backgroundColor : '#FFFFFF'
});

for (var i = 0; i < data.length; i++) {
    var row = Ti.UI.createTableViewRow({
        filter : data[i],
        height : 60,
        hasChild : true,
    });

    var labtitle = Ti.UI.createLabel({
        text : data[i],
        height : 20,
        left : 20,
        top : 20,
        width : 160,
        font : {
            fontSize : 14,
            fontWeight : 'bold',
            fontFamily : 'Minion Pro'
        }
    });
    row.add(labtitle);
    newarr.push(row);
}

tableview.setData(newarr);
tableview.filterAttribute = "filter";
win.add(tableview);
