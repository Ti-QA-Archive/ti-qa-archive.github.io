  • Here is a code snippet to get infos from weather server and display it in a tableview:

    https://gist.github.com/2557157

    and here the controler:

    api.getWeather = function(gps, callback) {
        var url = 'http://free.worldweatheronline.com/feed/marine.ashx?q=' + gps + '&format=json&key=XXXXX01246120104';
        var xhr = Titanium.Network.createHTTPClient();
        xhr.onerror = function() {
            Ti.API.log('xhrerror');
        };
        xhr.onload = function() {
            try {
                var response = JSON.parse(this.responseText);
                callback(response.data.weather);
            } catch(E) {
                Ti.API.log(E);
            }
        };
        xhr.open('GET', url);
        xhr.send();
};
