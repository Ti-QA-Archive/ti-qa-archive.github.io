  • Hi, you can try this code

    Ti.Gesture.addEventListener('orientationchange', function(e){
    if((Titanium.Gesture.orientation == Titanium.UI.LANDSCAPE_LEFT)||(Titanium.Gesture.orientation == Titanium.UI.LANDSCAPE_RIGHT))
    {       
       label.left = 100; // set label left for landscape mode

    }
    else
    {        
         label.left = 50; // set label left for portrait mode
     }
});
