  • The following code works by updating the value and message with a button. You just need to change the button to whatever you are using to iterate the progress.

    var win = Ti.UI.createWindow({
    title: "New Cool Window"
});
var pb=Titanium.UI.createProgressBar({
    top:10,
    width:250,
    height:'auto',
    min:0,
    max:10,
    value:0,
    color:'#fff',
    message:'Downloading 0 of 10',
    font:{fontSize:14, fontWeight:'bold'},
    style:Titanium.UI.iPhone.ProgressBarStyle.PLAIN,
});

win.add(pb);
pb.show();

var btn = Ti.UI.createButton({
    title:'Update'
});
btn.addEventListener('click', function() {
    pb.value = pb.value + 1;
    pb.setMessage("Downloading "+pb.value+" of 10");
});
win.add(btn);
win.open();
