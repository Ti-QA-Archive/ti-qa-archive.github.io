Different Labels In Table Rows

I currently have a table of rows and would like to have the row display one row in black (eg Example 1) and then have lines underneath that, that would then be a different font color. Below is the code I am using for the tab


var data = [];

data[0] = Ti.UI.createTableViewRow({hasChild:false, height:'190'});
data[1] = Ti.UI.createTableViewRow({leftImage: '../images/top1.jpg', hasChild:true, height:'80', title:'Technique 1', test:'technique1.js'});
data[2] = Ti.UI.createTableViewRow({leftImage: '../images/top2.jpg', hasChild:true, height:'80', title:'Technique 2', test:'technique2.js'});
data[3] = Ti.UI.createTableViewRow({leftImage: '../images/top3.jpg', hasChild:true, height:'80', title:'Technique 3', test:'technique3.js'});


function addRow(idx,image)
{
    data[idx].add(Ti.UI.createImageView({
        image:'../images/top9.png',
        height:'auto',
        width:'320',
        left:0,
        right:0,
        top:0,
        bottom:0,
    }));
}

addRow(0, 'image');


// create table view
var tableview = Titanium.UI.createTableView({
    data:data,
});

// create table view event listener
tableview.addEventListener('click', function(e)
{
    if (e.rowData.test)
    {
        var win = Titanium.UI.createWindow({
            url:e.rowData.test,
            title:e.rowData.title
        });
        Titanium.UI.currentTab.open(win,{animated:true});
    }
});

// add table view to the window
Titanium.UI.currentWindow.add(tableview);

    Just create your own TableViewRow:

    for example:

    var row = Titanium.UI.createTableViewRow();
var view = Titanium.UI.createView({backgroundColor:'red',width:20,height:20});
row.height = 'auto';
row.add(view);
