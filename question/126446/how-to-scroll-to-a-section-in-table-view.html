    ok i have a solution:

    1. add to data "name:"
    2. use getIndexByName
    3. scrolltoindex

    var data = [

    {title:'Alan', hasChild:true, header:'A'},

    {title:'Alice', hasDetail:true},

    {title:'Alexander'},

    {title:'Amos'},

    {title:'Alonzo'},

    {title:'Alonzo'},

    {title:'Brad', header:'B'},

    {title:'Brent'},

    {title:'Billy'},

    {title:'Brenda'},

    {title:'crad',name:'C', header:'C'},

    {title:'crent'},

    {title:'cilly'},
    ];

    label5km.addEventListener('click',function()
    {
    row = tableview.getIndexByName('C');
    tableview.scrollToIndex(row);
    });
