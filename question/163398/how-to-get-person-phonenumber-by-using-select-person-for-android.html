    Hello Sabari,

    Try this :

     Ti.Contacts.showContacts({
            selectedPerson : function(e) {
                alert(e);
                Ti.API.info('Phone Number : :  ' + e.person.phone);
                Ti.API.info('Email : :  ' + e.person.email);
            }
        });

    i hope this helps. thanks.

    • i tested below code in Ti SDK 3.2.1 and it working fine. it give me a full details of selected person. clean your project and check again.

      var btn = Ti.UI.createButton({
    title : "contact",
    height : 100,
    width : 200
});

btn.addEventListener("click", function(e) {
    Ti.Contacts.showContacts({
        selectedPerson : function(e) {
            var person = e.person;
            alert(JSON.stringify(person));
            Ti.API.info('Phone Number : :  ' + e.person.phone);
            Ti.API.info('Email : :  ' + e.person.email);
        }
    });
});
