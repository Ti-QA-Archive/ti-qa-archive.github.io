this is my code:
Window with TableView

function SignOutputWin(){

    this.win = Ti.UI.createWindow({
        title:'Output',
        backgroundColor: STYLE.BACKGROUND.COLOR,
        barColor:'black'
    });

    this.table = Ti.UI.createTableView({
        backgroundColor    : STYLE.BACKGROUND.COLOR,
        top                : '2%',
        width            : '95%'
    });

    this.win.add(this.table);

    return this;
};

SignOutputWin.prototype.update = function(){
        for(var i=0; i<100; i++){
            var newRow = Ti.UI.createTableViewRow({
                className        : 'SplitterRow'+i,
                //touchEnabled    : false,
                hasChild        : false,
                backgroundColor    : STYLE.SPLITTER.BACKGROUND,
                backgroundSelectedColor: STYLE.SPLITTER.BACKGROUND,
                selectionStyle    : 'none',
                height            : STYLE.SPLITTER.HEIGHT,
                title            : 'SplitterRow'+i
            });
            this.table.appendRow(newRow);
        }
};

module.exports = SignOutputWin;
