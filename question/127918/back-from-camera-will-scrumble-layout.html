Back from camera will scrumble layout

Hi all,
my app have to scan a QRcode using intent and zxing. So I have a view with a button (and some label and textinput) that start camera, if no code is scanned and user press back button my view will change labels and textinputs position.

Here is my view code:

var win = Titanium.UI.createView({
    layout: 'vertical',
    top: 60,
    height: '100%',
    width: '100%'
});

var cameraView = Titanium.UI.createView({
    touchEnabled: false,
    layout: 'vertical'
});

var manualView = Titanium.UI.createView({
    touchEnabled: false,
    layout: 'vertical',
    top: 150
});

var manualInput = Titanium.UI.createView({
    touchEnabled: false,
    layout: 'horizontal'
});

var cameraLabel = Titanium.UI.createLabel({
    text: 'By camera'
});

var manualLabel = Titanium.UI.createLabel({
    text: 'Manually',
    borderTop: 1
});

var guidInput = Titanium.UI.createTextField({
    width: '80%',
    hintText: 'Insert code',
});

var qrButton = Ti.UI.createButton({
    width: '95%',
    title: "Scan code"
});

var manualCodeButton = Titanium.UI.createButton({
    id: 'manualbutton',
    width: '15%',
    title: 'Ok'

    win.orientationModes = [Titanium.UI.PORTRAIT];
