    Hi

    Sounds like you are looking to open a normal window as a child where a back button automatically appears.

    Here is a simple example app to show you how, create a new classic default app and replace the contents of app.js with the code below.

    Ti.UI.setBackgroundColor('#000');

var win1 = Ti.UI.createWindow({  
    title: 'Tab 1',
    backgroundColor: '#fff'
});
var tab1 = Ti.UI.createTab({  
    icon: 'KS_nav_views.png',
    title: 'Tab 1',
    window: win1
});

var btn = Ti.UI.createButton({
    title: 'New Window'
});
btn.addEventListener('click', function (e) {
    var newWin = Ti.UI.createWindow({  
        title: 'New Window',
        backgroundColor: '#fff'
    });
    tab1.open(newWin, { animated: true});
});
win1.add(btn);

var tabGroup = Ti.UI.createTabGroup();
tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);  
tabGroup.open();
    • Press the button and the new window appears from the right and pushes the existing one off to the left.
    • Press the back button to go back to the first window

    • OK here you go. You have to specify the dimensions correctly to start with, these need to be fixed rather than something that is in relation to something else (no %, SIZE, FILL)

      pWidth = Titanium.Platform.displayCaps.platformWidth;
pHeight = Titanium.Platform.displayCaps.platformHeight;

var animOutLeft = Ti.UI.createAnimation({
    duration: 500,
    left: -pWidth
});
var animCenter = Ti.UI.createAnimation({
    duration: 500,
    left: 0
});

var winFirst = Ti.UI.createWindow({
    backgroundColor: 'orange',
    height: pHeight,
    left: 0,
    width: pWidth
});
var btn1 = Ti.UI.createButton({
    title: 'Show Green'
});
btn1.addEventListener('click', function (e) {
    winFirst.animate(animOutLeft);
    winSecond.animate(animCenter);
});
winFirst.add(btn1);
var winSecond = Ti.UI.createWindow({
    backgroundColor: 'green',
    height: pHeight,
    left: pWidth,
    top: 0,
    width: pWidth
});

winFirst.open();
winSecond.open();

      This will not work with Android heavyweight windows as they cannot be animated in the same way.
