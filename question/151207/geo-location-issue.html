Geo Location Issue

Central.prototype.getDeviceLocation = function(callback) {
    var location = new Object();
    location.lat = 0;
    location.lng = 0;
    Titanium.Geolocation.purpose = "Recieve User Location";
    Titanium.Geolocation.accuracy = Titanium.Geolocation.ACCURACY_BEST;
    Titanium.Geolocation.distanceFilter = 10;
    var location_services = Titanium.Geolocation.getLocationServicesEnabled();
    if (!location_services) {
        Ti.API.info("location_services disabled");
        callback(location);
    }
    Ti.API.info("getDeviceLocation waiting for getCurrentPosition event to return");
    Titanium.Geolocation.getCurrentPosition(function(e) {
        if (e.error) {
            Ti.API.info('error: ' + e.error);
            callback(location);
        }
        try {
            var current_longitude = e.coords.longitude;
            var current_latitude = e.coords.latitude;

            /*
             * My Home Location
             */
            if (isSimulator) {
                current_longitude = 138.668083;
                current_latitude = -34.820642;
            }

            location.lat = current_latitude;
            location.lng = current_longitude;

            Ti.API.info("TRUE location.lat:" + location.lat);
            Ti.API.info("TRUE location.lng:" + location.lng);
        } catch (e) {
            // alert("Can't fetch location");
            /*
             * Use default location, so app works at least
             */
            current_longitude = 138.668083;
            current_latitude = -34.820642;
            location.lat = current_latitude;
            location.lng = current_longitude;

            Ti.API.info("FAKE location.lat:" + location.lat);
            Ti.API.info("FAKE location.lng:" + location.lng);
        }

        Ti.API.info("Calling Master Function");
        callback(location);
    });
}

I have written this function to fetch the device location.

So the first one calls it and it returns the location fine.

Then I press the second button, and the geo location never returns.

Can I add a timeout or something like that?
