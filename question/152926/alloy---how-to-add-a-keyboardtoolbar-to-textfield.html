Alloy - How to add a keyboardToolBar to textfield

How to do it in the markup?

Traditional way :

var send = Titanium.UI.createButton({
    title : 'Send',
    style : Titanium.UI.iPhone.SystemButtonStyle.DONE,
});

var camera = Titanium.UI.createButton({
    systemButton : Titanium.UI.iPhone.SystemButton.CAMERA,
});

var cancel = Titanium.UI.createButton({
    systemButton : Titanium.UI.iPhone.SystemButton.CANCEL
});

var flexSpace = Titanium.UI.createButton({
    systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

var textfield = Titanium.UI.createTextField({
    borderStyle : Titanium.UI.INPUT_BORDERSTYLE_BEZEL,
    hintText : 'Focus to see keyboard with toolbar',
    keyboardToolbar : [cancel, flexSpace, camera, flexSpace, send],
    keyboardToolbarColor : '#999',
    keyboardToolbarHeight : 40,
    top : 10,
    width : 300, height : 35
});

  • hi,

    just try this one

    function createToolBar() {
    var send = Titanium.UI.createButton({
        title : 'Send',
        style : Titanium.UI.iPhone.SystemButtonStyle.DONE,
    });

    var camera = Titanium.UI.createButton({
        systemButton : Titanium.UI.iPhone.SystemButton.CAMERA,
    });

    var cancel = Titanium.UI.createButton({
        systemButton : Titanium.UI.iPhone.SystemButton.CANCEL
    });

    var flexSpace = Titanium.UI.createButton({
        systemButton : Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
    });
    var toolBar = Ti.UI.iOS.createToolbar({
        items : [cancel, flexSpace, camera, flexSpace, send],
        backgroundColor : '#000',
        height : 44,
        width : 320
    });
    return toolBar;
}

var textfield = Titanium.UI.createTextField({
    borderStyle : Titanium.UI.INPUT_BORDERSTYLE_BEZEL,
    hintText : 'Focus to see keyboard with toolbar',
    keyboardToolbar : createToolBar(),
    keyboardToolbarColor : '#999',
    keyboardToolbarHeight : 40,
    top : 10,
    width : 300, height : 35
});

  • Hello Jiale,

    Try this Way : 

    var win = Ti.UI.createWindow();
function getKeyboardToolbar() {
    //Toolbar
    var nextPrevBar = Titanium.UI.createButtonBar({
        labels : ['Prev', 'Next'],
        backgroundColor : '#666',
        style : Titanium.UI.iPhone.SystemButtonStyle.BAR,
        height : 30,
        width : 130
    });
    return nextPrevBar;
}     

var tf1 = Ti.UI.createTextField({
    top : "10dp",
    width : 100,
    height : 35,
    font : {
        fontSize : 15,
        fontWeight : 'bold',
    },
    keyboardToolbar : [getKeyboardToolbar()],
    keyboardToolbarColor : '#999',
    keyboardToolbarHeight : 40,
    borderColor : "black",
    borderRadius : "7dp",
    borderWidth : "1"
});
win.add(tf1);
win.open()
