var addRow=function(name,join) {
    var row = Ti.UI.createTableViewRow();
    row.hor = $.UI.create('View',{
        classes: 'rowView',
    });
    row.add(row.hor);
    row.nameLabel= $.UI.create('Label',{    
        classes: "LabelCell",
        text : name,
    });
    row.hor.add(row.nameLabel);
    row.statusLabel= $.UI.create('Label',{    
        classes: "LabelCell",
        text : join,
    });
    row.hor.add(row.statusLabel);

    return row;    
};
