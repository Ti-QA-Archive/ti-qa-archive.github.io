  • Hi Jayaprakash,

    var self = Ti.UI.createWindow({
    titleControl : view,
    barImage : "/images/topbar@2x.png",
    backgroundImage : "/images/bg@2x.png",
    navBarHidden : true  // using this property , we can hide the nav bar from window
});

    Please use this.
