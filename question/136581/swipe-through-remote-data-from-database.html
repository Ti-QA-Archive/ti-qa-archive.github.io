  • Problem solved, this is my code:

    var sendit = Ti.Network.createHTTPClient();
sendit.open('GET', '******');
sendit.send();
sendit.onload = function(){
var json = JSON.parse(this.responseText);
var json = json.spreuken;
var dataArray = [];
var pos;
    for( pos=0; pos < json.length; pos++){
        // Create a view    
        var container_view = Ti.UI.createView();
            var spreuk = json[pos].titel;
        // Create a label
         var spreuk_label = Ti.UI.createLabel({
           width: 320,
           height: 387,
               textAlign:'center',
            font:{fontFamily:'Helvetica Neue', fontWeight:'bold',fontSize:16},
            text:spreuk
    });
     // Add label to the view
    container_view.add(spreuk_label);
     // Push the view to the array
    dataArray.push(container_view);
}

var spreuken = Titanium.UI.createScrollableView({
    views:dataArray,
    showPagingControl:true,
    pagingControlHeight:30,
    maxZoomScale:4.0,
    currentPage:0

});

currentWin.add(spreuken);
};
