Ti.App.addEventListener cause crash

Hi,
I've troubled in addEventListener with Ti.App.
Sample is below.
I've sent HTTP request and got the XML formatted data.
So I used Ti.App.fireEvent and Ti.App.addEventListener.
But at addEventListener, app got crush.
It seems fireEvent's 2nd parameter cause crush.
And I've tried use window object instead Ti.App, it done successfully.
Ti.app.fireEvent and Ti.app.AddEventListerner can't use XML object?
It is successful useing responseText instead of response XML.
Someone have any information?

crash:—————————————

Ti.App.addEventListener('login',function(e)
{
var elements = e.xml.getElementsByTagName("auth");
var auth = elements.item(0).text;

Ti.API.info("api_test" + auth);

}
);

var xhr = Titanium.Network.createHTTPClient();

xhr.setTimeout(20000);
xhr.onload = function(e)
{
    xml = this.responseXML;
    Ti.App.fireEvent('login',{xml:xml});
};
xhr.open('POST','http://hoge.com/api/');
// send the data
xhr.send({id:'hoge',pwd:'hoge'});

success:—————————————

var win = Titanium.UI.createWindow();

win.addEventListener('login',function(e)
{
var elements = e.xml.getElementsByTagName("auth");
var auth = elements.item(0).text;

Ti.API.info("api_test" + auth);

}
);

var xhr = Titanium.Network.createHTTPClient();

xhr.setTimeout(20000);
xhr.onload = function(e)
{
    xml = this.responseXML;
    win.fireEvent('login',{xml:xml});
};
xhr.open('POST','http://hoge.com/api/');
// send the data
xhr.send({id:'hoge',pwd:'hoge'});
