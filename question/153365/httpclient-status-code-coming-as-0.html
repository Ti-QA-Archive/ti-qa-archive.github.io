Httpclient status code coming as 0

When I test my app on the android emulator and the post request failed, in the onerror hander this.status is coming as 0, wheres in iOS it seems to have a correct status code ie 401. What might be the problem? here is the code I am using to make the request -

exports.submit = function(data, cb){
    var xhr = Ti.Network.createHTTPClient();
    xhr.open('POST', getUrl());
    xhr.setRequestHeader('Authorization', getAuthHeader());
    xhr.setRequestHeader('Content-Type', 'application/json');
    var payload = JSON.stringify(data);
    console.log(payload);
    xhr.onload = function(){
        return cb(JSON.parse(this.responseText), this.status);
    };
    xhr.onerror = function(e){
        console.log('Error in submit. Status Code: ' + this.status + ', ' + e.code + e.error)
        cb(null,this.status);
    }
    xhr.send(payload);
}
