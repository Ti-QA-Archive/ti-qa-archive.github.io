    Try this:

    commonJS.js

    function commonJS()
{
    this.myFont= {
        fontSize:25,
        fontFamily:'Heiti TC'
    };
    this.myBackgroundColor = '#274167';

}

module.exports = commonJS;

    myView.js

    var commonJS = require('commonJS');
settings = new commonJS();
alert(settings.myBackgroundColor);
