Android date picker

Hi.

I'm using Android 2.2 on Samsung Galaxy Tab, Window XP, Titanium SDK 1.7.3 latest.

I created date picker:

var picker = Ti.UI.createPicker({ type: Ti.UI.PICKER_TYPE_DATE });

Now I have a problem because picker is huge and of course I want to make it smaller. How can I do that?
I tried to change width/height, but haven't succeed.

Thanks!

  • Is it a bad idea to create a new window and open this picker on it?
    Something like:

        var winDatePicker = Titanium.UI.createWindow({});
    winDatePicker.backgroundColor = 'black';

    var doneBtn = Ti.UI.createButton({
        title: 'Done',
    });

    doneBtn.addEventListener('click', function() {
        winDatePicker.hide();
    });

    winDatePicker.add(doneBtn);

    var datePicker = Ti.UI.createPicker({
        type:Ti.UI.PICKER_TYPE_DATE,
        bottom:0,
    });


    // turn on the selection indicator (off by default)
    datePicker.selectionIndicator = true;

    datePicker.addEventListener('change', function(e) {
        //your code
    });

    winDatePicker.add(datePicker);


    //open window
    winDatePicker.open();    

});
