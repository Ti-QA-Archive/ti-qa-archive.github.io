NullPointerException on window open on android

Hey folks,

I have a problem with window opening.
I'm using the same structure as the https://github.com/appcelerator/tweetanium project.

The application window has no properties but a background color.

In the application window I have a listener:

        addButton.addEventListener('click', function() {
            var w = Test.UI.createAddWindow();
            w.open();
        });

The opening of the window from out the application window works under ios but not under android.

The crash report: http://pastie.org/1973739

The AddWindow.js:

(function() {
    Test.UI.createAddWindow = function() {
        var win = Ti.UI.createWindow($$.backgroundColor),

        headLabel = Ti.UI.createLabel({
            top: '10%',
            bottom: '10%',
            text: 'Add',
            textAlign: 'center'
        }),

        header = Ti.UI.createView({
            top: 0,
            height: '10%',
            backgroundColor: 'red'
        }),

        footer = Ti.UI.createView({
            top: '80%',
            height: '20%',
            backgroundColor: 'red'
        }),

        addButton = Ti.UI.createButton({
            left: '3%',
            top: '10%',
            right: '3%',
            bottom: '10%',
            title: 'Add'
        });

        header.add(headLabel);
        win.add(header);

        footer.add(addButton);
        win.add(footer);

        return win;
    };
})();
