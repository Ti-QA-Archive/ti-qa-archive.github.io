  • Hi,
    It is not possible in Titanium till now. You have to use custom checkBox. You can try this

    var win = Ti.UI.createWindow({
  backgroundColor: 'white'
});

// Create a Button.
var check = Ti.UI.createButton({
    title : '',
    height : 30,
    width : 20,
    top : 50,
    left : 50,
     value:false,
     borderColor:'red',
     borderRadius:2
});

var check2 = Ti.UI.createButton({
    title : '',
    height : 30,
    width : 20,
    top : 90,
    left : 50,
     value:false,
     borderColor:'red',
     borderRadius:2
});

win.add([check,check2]);

// Listen for click events.
check.addEventListener('click', function(e) {
    if(e.source.value== false){check.value=true;e.source.title='V';}
    else {check.value=false; e.source.title=''; }
});
// Listen for click events.
check2.addEventListener('click', function(e) {
    if(e.source.value== false){check2.value=true;e.source.title='V';}
    else {check2.value=false; e.source.title=''; }
});

win.open();

  • Also look this function to create custom check box. You can change background image of one button and handle events. 

    exports.createCheckBox = function(boolVal) {
    var boolval;

    var btn = Ti.UI.createButton({
        boolValue : boolval,
        width : 32,
        height : 32,
    });

        // set initial background image 
    setImage();

    //private method that determines the image for the button
    function setImage() {

        if (boolval) {
            btn.backgroundImage = "/images/checkbox-checked.png";
        } else {
            btn.backgroundImage = "/images/checkbox.png";
        }
    }

    btn.addEventListener('click', function(e) {

        boolval = !boolval;
        btn.boolValue = !btn.boolValue;
        setImage();

    });

    return btn;

};
