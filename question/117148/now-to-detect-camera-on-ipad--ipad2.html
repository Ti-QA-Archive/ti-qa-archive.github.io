  • The following code snippet (which originally came from a Titanium app) shows how I handle the absence of a camera:

    Titanium.Media.showCamera ({

    saveToPhotoGallery: true,
    allowEditing: true,

    success: function(event){
        var a = Titanium.UI.createAlertDialog({
            title: 'You did it!'
        });
        a.show();
    },

    error: function(error){
        var a = Titanium.UI.createAlertDialog({
            title: 'Uh Oh...'
        });
        if (error.code == Titanium.Media.NO_CAMERA || 
                         error.code == Titanium.Media.NO_VIDEO) {
            a.setMessage('Sorry, this device doesn't have a camera.');
        }
        else {
            a.setMessage('Unexpected error: ' + error.code);
        }
        a.show();
    }


});
