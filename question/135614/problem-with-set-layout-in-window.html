Problem with set layout in window

 var b=Ti.UI.createLabel({text:'b' });
 var c=Ti.UI.createLabel({text:'c' });
 var d=Ti.UI.createLabel({text:'d' });

 var ab=Ti.UI.createView({ layout: 'vertical' });
 ab.add(a);

  • set the height to Ti.UI.SIZE to ab and cd views…

    
var introWin = Ti.UI.createWindow({
            title:'abc',
            layout: 'vertical',
            fullscreen:true,
            backgroundColor:'white',
            });


 var a=Ti.UI.createLabel({text:'a'});
 var b=Ti.UI.createLabel({text:'b' });
 var c=Ti.UI.createLabel({text:'c' });
 var d=Ti.UI.createLabel({text:'d' });

 var ab=Ti.UI.createView({ layout: 'vertical', height:Ti.UI.SIZE });
 ab.add(a);
 ab.add(b);
 var cd=Ti.UI.createView({ layout: 'vertical', height:Ti.UI.SIZE });
 cd.add(c);
 cd.add(d);

 introWin.add(ab);
 introWin.add(cd);
 introWin.open();

    (tested on iOS)
