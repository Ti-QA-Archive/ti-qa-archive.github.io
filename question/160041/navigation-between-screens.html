  • //try this code it may solve you problem to clean up the first window submit form when you back from second to first window
    //————————————————————————————————————————————————————-
    //app.js file code

    
Titanium.UI.setBackgroundColor('#000');
var win=Ti.UI.createWindow({
    title:'Test',
    backgroundColor:'#fff',
    layout:'vertical'
});

var aTextField = Ti.UI.createTextField({
    height : Ti.UI.SIZE,
    top : 10,
    left : 40,
    width : 240,
    hintText : 'Enter Text',
    softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_DEFAULT_ON_FOCUS, // Android only
    keyboardType : Ti.UI.KEYBOARD_DEFAULT,
    returnKeyType : Ti.UI.RETURNKEY_DEFAULT,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});

win.add(aTextField);

var aTextField1 = Ti.UI.createTextField({
    height : Ti.UI.SIZE,
    top : 10,
    left : 40,
    width : 240,
    hintText : 'Enter Text',
    softKeyboardOnFocus : Ti.UI.Android.SOFT_KEYBOARD_DEFAULT_ON_FOCUS, // Android only
    keyboardType : Ti.UI.KEYBOARD_DEFAULT,
    returnKeyType : Ti.UI.RETURNKEY_DEFAULT,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});

win.add(aTextField1);

var aButton = Ti.UI.createButton({
    title : 'submit',
    height : 'auto',
    width : 'auto',
    top : 10,

});
var ok = Titanium.UI.createWindow({
    url:'new.js', // create this file under the resource directory like (Resources/new.js)
    title:'Second Window',
    backgroundColor:'#fff',
    _parent:Ti.UI.currentWindow,
    modal:true,
    });
aButton.addEventListener('click', function() {
    win.close();
    ok.open();
});

win.add(aButton);


win.open();

    won.open();

so.add(aButton);
