  • try something like this :

    index.xml

    <Alloy>
    <Window id="win">
        <View id="mapview" ns="Ti.Map" animate="true" regionFit="true" userLocation="true"
            mapType="Ti.Map.STANDARD_TYPE" >
        </View>
    </Window>
</Alloy>

    index.js

    
var json = data;   // your json data

for (var i = 0; i < json.length; i++) {
    var annotation = Titanium.Map.createAnnotation({
        latitude : json[i].latitude,
        longitude : json[i].longitude,
        title : json[i].title,
        pincolor : Titanium.Map.ANNOTATION_RED,
        animate : true,
        myid : I // Custom property to uniquely identify this annotation.
    });

    $.mapview.addAnnotation(annotation);
}

    i hope it's helps. thanks.
