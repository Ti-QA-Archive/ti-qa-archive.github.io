var SL = {
    UI: {
        createUserLabel: function(props){

            var mask = Ti.UI.createScrollView({
                contentWidth:'auto',

                height: 'auto'

            control.left = 0;
            control.width = 'auto';
            control.height = 'auto';
            control.width = control.width;

  • So i figured it out. easiest way to see is to plop this code into an app.js and run it

    
var win = Ti.UI.createWindow({
    backgroundColor: "#000000"
});
win.open();



var SL = {
    UI: {
        createUserLabel: function(props){

            var mask = Ti.UI.createScrollView({
                contentWidth:'auto',
                   contentHeight:'auto',
                width: props.width,
                height: 'auto'
            })
            var control = Ti.UI.createLabel(props);
            control.top = 0;
            control.left = 0;
            control.width = 'auto';
            control.height = 'auto';
            control.width = control.width;


            mask.add(control);

            return mask;
        }
    }
}

var headerContainer = Ti.UI.createView({
    width: 220,
    height: 100,
    right: 1,
    top: 1,
    backgroundColor: '#292b37'
});
win.add(headerContainer);
var baseline = Ti.UI.createView({
    height: 1,
    width: 220,
    top: 49,
    backgroundColor: 'green'
});
headerContainer.add(baseline);

var postedBy = Ti.UI.createLabel({
    top: 30,
    left: 10,
    width: 130,
    height: 25,
    color: 'white',
    text: "Posted by ",
    font: {
        fontSize: 15,
        fontFamily: 'Helvetica Neue'
    }
});
headerContainer.add(postedBy);

var userWrapper = Ti.UI.createView({
    width: 120,
    top:32, left: 90
});
var userName = Ti.UI.createLabel({
    top:0, left:0,
    height:20,
    width: 'auto',
    color: '#b9e100',
    text: "DS Techroom Photography",
    font: {
        fontWeight: 'bold',
        fontSize: 15,
        fontFamily: 'Helvetica Neue'
    },
    touchEnabled: false,
    minimumFontSize: 8
});
userWrapper.add(userName);

headerContainer.add(userWrapper);

    An item of interest is the userWrapper View that I added. It sets the width that you want the userName Label to conform to. userName's width is set to 'auto', which will make it take up the available space inside of userWrapper without wrapping. In this case, if we don't set minimumFontSize on userName, iOS will truncate the Label for us and it will say "DS Techroâ¦", because we DO have minimumFontSize, the size reformats itself to fit within the width of userWrapper.

    And that's how you use minimumFontSize.
