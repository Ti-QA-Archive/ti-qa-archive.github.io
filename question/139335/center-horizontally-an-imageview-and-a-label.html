center horizontally an imageview and a label

This worked in SDK 1.8.2 but as of 2.0.1 alignment is now to the left and I cant figure out how to get it back in the center. My aim is to have a 'header' centered, with an imageview containing an icon, and a variable-width label containing the header.

This is my current code:


var header = Ti.UI.createView({ 
   width:"auto",
   height:35,
   top:30
});

var icon = Ti.UI.createImageView({
   width:35,
   height:35,
   left:0
});

var label = Ti.UI.createLabel({
   width:"auto",
   height:35,
   left:50,
   font:{fontSize:30, fontFamily:"Helvetica-Light"},
   color:"#000000"
});

icon.image = 'images/'+iconimg+'.png';
label.text = caption;

header.add(icon);
header.add(label);

Is there a better way to center a variable-width view?

I also tried using label.toImage().width to get the width of the label and use that to calculate the needed left and height, but the result was always 320. even though most of the time the actual width of the label was around 200.

    Since version 2.0, you don't really want to use "auto" for your dimensions. Make sure you read this document.

    Here's an example that does what you want:

    var win = Titanium.UI.createWindow({    
});

var header = Ti.UI.createView({ 
    width: Ti.UI.SIZE,
    height: 35,
    top: 30
});


var icon = Ti.UI.createImageView({
    width:35,
    height:35,
    left:0
});

var label = Ti.UI.createLabel({
    height:35,
    left:50,
    font:{fontSize:30, fontFamily:"Helvetica-Light"},
    color:"#fff"
});

icon.image = '/images/foo.png';
label.text = "this is a test";

header.add(icon);
header.add(label);

win.add (header);

win.open ();
