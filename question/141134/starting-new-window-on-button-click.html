    hi Ashwin, set a listener on your button like this

    btn.addEventListener('click',function(){
    var win2 = Ti.UI.createWindow({
    backgroundColor:'white',
    url:'window2.js',
    prop:data
    });
    win2.open();
});

    you can access your data in window2.js in following way

    window2.js

    var win = Ti.UI.currentWindow;
var previous data = win.prop;

    i hope this will solve your problem.

      ApplicationWindow.js

      
function ApplicationWindow() {
    var SongsUi = require('ui/common/baseUI/songs');

    var self = Ti.UI.createWindow({
        backgroundColor:'#ffffff',
        navBarHidden:true,
        exitOnClose:true,

    });
    var songsui = new SongsUi(); 
    self.add(songsui)
    return self;

}
module.exports = ApplicationWindow;

      songs.js

      function songUi() {
    var baseView = Ti.UI.createView({
        height : 'auto',
        width : 'auto',
        backgroundColor : '#000',
        layout : 'composite'
        // right:'1 dip'
    })
    addButtons(baseView);
    return baseView;
}

module.exports = songUi;

function addButtons(baseView) {
    var header = Ti.UI.createView({
        height : 'auto',
        width : 'auto',
        layout : 'horizontal',
        top : '43 dip'
    })

    var artistBut = Ti.UI.createButton({
        // textAlign:Ti.UI.TEXT_VERTICAL_ALIGNMENT_TOP,
        // title:'ARTIST',
        height : '45 dip',
        width : '33%',
        backgroundImage : '/images/drawable/artists_grey.png',
        backgroundSelectedImage : '/images/drawable/artists_red.png',
    })

    artistBut.addEventListener('click', function() {
        // var win2 = require('ui/handheld/android/ArtistWindow');
        var win2 = Ti.UI.createWindow({
            backgroundColor : 'white',
            url : 'ui/common/ArtistWindow.js'
            // prop : data
        });
        win2.open();
    })

    header.add(artistBut);
    baseView.add(header)
}


}

      ArtistWindow.js

      function ArtistWindow(){
    var ArtistUi = require('ui/common/artist');
    var self = Ti.UI.createWindow({
        backgroundColor:'#ffffff',
        // navBarHidden:true,

    });
    var artistui = new ArtistUi();
    self.add(artistui)
    Ti.API.info('view added to the window')
    return self;
}
module.exports=ArtistWindow;
