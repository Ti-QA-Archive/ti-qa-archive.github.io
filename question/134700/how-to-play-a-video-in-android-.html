How to play a video in android ?

I have copied the whole code from http://developer.appcelerator.com/apidoc/mobile/latest/Titanium.Media.VideoPlayer-object#methods. Here is my code 

Titanium.UI.setBackgroundColor('#000');
var win = Titanium.UI.createWindow({
    title : 'Test',
    backgroundColor : '#fff',
    exitOnClose : true
});

// Change to a valid URL
var contentURL = "media/movie.mp4";

var openButton = Ti.UI.createButton({
    title : "Start Video",
    top : "0dp",
    height : "40dp",
    left : "10dp",
    right : "10dp"
});

openButton.addEventListener('click', function() {
    var activeMovie = Titanium.Media.createVideoPlayer({
        url : contentURL,
        movieControlMode : Titanium.Media.VIDEO_CONTROL_DEFAULT,
        scalingMode : Titanium.Media.VIDEO_SCALING_ASPECT_FILL,
        autoplay : true
    });

    var closeButton = Ti.UI.createButton({
        title : "Exit Video",
        top : "0dp",
        height : "40dp",
        left : "10dp",
        right : "10dp"
    });

    closeButton.addEventListener('click', function() {
        activeMovie.hide();
        activeMovie.url= null;
    });

    activeMovie.add(closeButton);
});
win.add(openButton);
win.open();

When I click on openButton it opens a new view with total black background and closeButton but no video playback on that page. I am not able to figure out why this is not working. To verfiy i have uploaded this screenshot to show the directory of movie.mp4 and app.js.
