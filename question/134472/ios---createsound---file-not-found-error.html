IOS - CreateSound - File Not Found Error

The code below runs perfectly on Android… Can't load the MP3 in iPhone

If I turn the following, it says "File Found"

var downloadDir = null;
var audioFileName = 'instructions.mp3';
if (Ti.Platform.name == 'android') {
    downloadDir = Ti.Filesystem.getFile(Ti.Filesystem.externalStorageDirectory, L('mp3_dir'));
} else {
    downloadDir = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory, L('mp3_dir'));
}

var tempAudioLocation = Ti.Filesystem.getFile(downloadDir.resolve(), audioFileName);
if (tempAudioLocation.exists()) {
    alert('File Found');
}
var tempAudioPlayer = Ti.Media.createSound({ 
    url:tempAudioLocation.resolve(),
    //allowBackground:true,
    preload:false,
    volume:0
});
