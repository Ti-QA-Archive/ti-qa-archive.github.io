How to remove separator from list view in iPhone app.

I want to remove separator from listview in iphone. I used separatorColor:'transparent' which works in android but not in iphone. It gives me error undefined Titanium.UI.iPhone.ListViewSeparatorStyle Here is my sample code

var win = Ti.UI.createWindow({backgroundColor: 'white'});
var listView = Ti.UI.createListView({
separatorColor:'transparent',
separatorStyle: Titanium.UI.iPhone.ListViewSeparatorStyle
});

var sections = [];

var fruitSection = Ti.UI.createListSection({ headerTitle: 'Fruits'});
var fruitDataSet = [
{properties: { title: 'Apple'}},
{properties: { title: 'Banana'}},
];
fruitSection.setItems(fruitDataSet);
sections.push(fruitSection);

var vegSection = Ti.UI.createListSection({ headerTitle: 'Vegetables'});
var vegDataSet = [
{properties: { title: 'Carrots'}},
{properties: { title: 'Potatoes'}},
];
vegSection.setItems(vegDataSet);
sections.push(vegSection);

listView.sections = sections;
win.add(listView);
win.open();

Titanium SDK: 3.1.3GA
OS: Mac Mavericks
iOS SDK: 6.1
