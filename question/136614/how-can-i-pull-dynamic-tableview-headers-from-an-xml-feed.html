  • OK, the gist of flow is we wanted a new section for each month of blog posts. Couple custom functions format dates, ect. Wrap it as a function, and can be used in an event to update your table. For us its called from the onload handler of the httpClient call that loads an rss feed:

    var items = doc.getElementsByTagName("item");
var x = 0;
var newsection = 0;
var doctitle = doc.evaluate("//channel/title/text()").item(0).nodeValue;
                for (var c=0;c<items.length;c++)
                {
                    var item = items.item(c);
                    var title = item.getElementsByTagName("title").item(0).textContent;
                    var pubdate = item.getElementsByTagName("pubDate").item(0).textContent;
                    currentMonth = returnPubMonth(pubdate);
                    var content = item.getElementsByTagName("content:encoded").item(0).textContent;
                    var row = Ti.UI.createTableViewRow({className: 'datarow', height:60, backgroundColor: 'transparent',
                    backgroundGradient:{
                    type:'linear', 
                    colors:['#ffffff','#c2c2c2'],
                    startPoint:{x:0,y:0},
                    endPoint:{x:0,y:60}}
                    });
                    if (isAndroid) {
                    row.backgroundImage = 'images/c2gradient.png';
                    }

                    var DateGraphicView = createDateGraphic(pubdate);
                    //var DateImage = Ti.UI.createImageView({image: DateGraphicView.ToImage()});
                    row.add(DateGraphicView);

                    var label = Ti.UI.createLabel({
                        text:title,
                        left:72,
                        top:5,
                        bottom:5,
                        color: 'black',
                        right:5                
                    });
                    row.add(label);



                    row.pubdate = pubdate;
                    row.articletitle = title;
                    row.content = content;
                    //var media;
                    //media = item.getElementsByTagName("myimage").item(0).text;

                    row.hasChild = true;
                    row.url = item.getElementsByTagName("link").item(0).textContent;

                    if (.currentMonth != EAC.UI.previousMonth){
                        if (section){
                            Blogdata[x++]= section;
                        }
                        var customHeaderView = Ti.UI.createView({backgroundColor: '#3d3d3d', height:'auto'});
                        var customHeaderLabel = Ti.UI.createLabel({text: ShortMonthtoLong(currentMonth), height: 25, left: 5, color: 'white',font: {fontSize: 20, fontWeight: 'bold'}});
                        customHeaderView.add(customHeaderLabel);

                        /*
                         * Custom headerViews do not work on Android.
                         */
                        if (isAndroid) {
                            var section = Titanium.UI.createTableViewSection({backgroundColor: headercolor, headerTitle:ShortMonthtoLong(currentMonth)});

                        } else {
                            var section = Titanium.UI.createTableViewSection({backgroundColor: headercolor, headerView:customHeaderView}); 
                        }
                        previousMonth = currentMonth;
                        section.add(row);
                        newsection = 1;
                    } else {
                        if (!section){
                        var customHeaderView = Ti.UI.createView({backgroundColor: '#3d3d3d', height:'auto'});
                        var customHeaderLabel = Ti.UI.createLabel({text: ShortMonthtoLong(EAC.UI.currentMonth), height: 25, left: 5, color: 'white',font: {fontSize: 20, fontWeight: 'bold'}});
                        customHeaderView.add(customHeaderLabel);
                        if (isAndroid) {
                            var section = Titanium.UI.createTableViewSection({backgroundColor:headercolor, headerTitle: ShortMonthtoLong(EAC.UI.currentMonth)});

                        } else {
                            var section = Titanium.UI.createTableViewSection({backgroundColor: headercolor, headerView:customHeaderView}); 
                        }
                        }

                        section.add(row);
                        newsection = 0;
                    }




                }


Blogdata[x++]= section;

      Ok, programming workflow. Step away from the code. Grab your notebook and a pencil.

      Draw out the flow of xml as its parsed.

      • Is this a new element?
      • Did the day change from a previous element?
      • If it did we would have to start a new section.

      • If the date is the same, just add the new element to old section.
