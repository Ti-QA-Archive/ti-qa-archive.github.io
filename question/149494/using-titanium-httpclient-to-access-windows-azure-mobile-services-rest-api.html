  • For anyone else who is trying to connect to an Azure Mobile Service this is the code I used to successfully get data from an Azure Mobile Service. Replace the 'your' parts' eg: yourservice, yourtable etc. One of the main things to remember is to pass your applicationkey in a RequestHeader, and ensure you have the correct permissions set in Azure to allow access to anyone who has the application key.

    var xhr = Titanium.Network.createHTTPClient();
xhr.setTimeout(30000);
xhr.open('GET', 'https://yourservice.azure-mobile.net/tables/yourtable');

//set headers
xhr.setRequestHeader("Content-Type", "application/json; charset=utf-8");
xhr.setRequestHeader("X-ZUMO-APPLICATION", "yourapplicationkey");

//send request with parameters
xhr.send();

xhr.onerror = function() {
    var a = Titanium.UI.createAlertDialog({title:'Error', message:'An error occurred calling the web service.' + this.responseText});
    a.show();
};

xhr.onload = function() {
    var a = Titanium.UI.createAlertDialog({title:'Okay', message:'Data: ' + this.responseText});
    a.show();
};

  • Yes, I am able to login to Azure Mobile Services using REST API.

    var url = "http://www.google.com";

    var client = Ti.Network.createHTTPClient({
    onload : function(e) {
    }
    onerror : function(e) {
    }

    // Prepare the connection.
    client.open("GET", url);

    // setRequestHeader needs to be called after open() and before send()

    // see here: https://wiki.appcelerator.org/display/guides/HTTPClient+and+the+Request+Lifecycle

    client.setRequestHeader("Accept","application/json");
    client.send();
