    Hi Abdulmajeed,
    I change your code and it works for me. try this.

    var views = [];
var Episode_buttons = [];
var PlayPause_Button = [];
var c = 1;
var new_top = 10;

while(c < 4)// 3 here is just for test in the app there is counter count number of rows in the db
{
    views[c] = Ti.UI.createView({
        backgroundColor : '#B6E5FF',
        width : '90%',
        height : '9%',
        top : new_top + '%'
    });

    win.add(views[c]);

    //==========================================

    Episode_buttons[c] = Ti.UI.createButton({
        title : 'Episode : ' + c ,
    });

    Episode_buttons[c].addEventListener('click', function() {

        PlayPause_Button[c] = Ti.UI.createButton({
            title : 'Play / Pause',
        });
// 
        // views[c].add(PlayPause_Button[c]);
        this.getParent().add(PlayPause_Button[c]);        
        // the error here as in the second image

    });

    views[c].add(Episode_buttons[c]);
    new_top = new_top + 10;
    c++;
}
