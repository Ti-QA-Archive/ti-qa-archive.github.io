  • // open window, it is offscreen
var newWindow=Ti,UI.createWindow({
    left : '330dp', // starts offscreen to the right
    top : '0dp',
});
newwindow.open();

// do some stuff..
var a = Titanium.UI.createAnimation();
a.height = Titanium.Platform.displayCaps.platformHeight - 20;
a.width = Titanium.Platform.displayCaps.platformWidth;
a.duration = 300;
a.left = '0dp'; // this should be 0 so window slides in from right to left

// animate
newwindow.animate(a);
