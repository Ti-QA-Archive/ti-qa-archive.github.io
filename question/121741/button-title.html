`
for(var i=0;i<5;i++){
var followButton = Titanium.UI.createButton({
title: 'Follow',

                _index: i
            });




            }

            followButton.addEventListener('click',function(e)
            {

               followButton.title="unFollow";
               alert("button index is +e.source._index);

            });

}
`

  • Maybe something (pretty common) like this:

    var eventListenerHandler = function(index) {
    return function(e) {
        this.title = "unFollow";
        alert("button index is " + index);
    };
};

for (var i = 0; i < 5; ++i) {

   var followButton = Titanium.UI.createButton({
        title: 'Follow'
    });

    followButton.addEventListener('click', eventListenerHandler(i));  
}
