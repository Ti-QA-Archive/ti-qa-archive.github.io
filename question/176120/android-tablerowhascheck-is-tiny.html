      var section = table.data[0];

        for (var i=0;i<section.rows.length;i++) {
            section.rows[i].hasCheck = false;
            if (section.rows[i].children.length > 0) {
                section.rows[i].children[0].color = '#336699';
            }
        };

        // set current check
        if (section.rows[index]) {
            section.rows[index].hasCheck = true;
            if (section.rows[index].children.length > 0) {
                section.rows[index].children[0].color = '#fff';
            }
        }

      As you can see in the screenshot the tick does display, and on older lower density Android devices is looks fine. But it doesn't scale depending on density by the looks of things.
