      Look ay my code Below

      var win1 = Titanium.UI.createWindow({ 

       title : 'MapOverlay Example',

 backgroundColor : '#fff'

      });

      var nfm = require('netfunctional.mapoverlay');

      var mapView = nfm.createMapView({

      mapType : Titanium.Map.STANDARD_TYPE,

region : {
    latitude : Titanium.App.Properties.getString('lat'),
    longitude : Titanium.App.Properties.getString('long'),
    latitudeDelta : 0.01,
    longitudeDelta : 0.01
},

animate : true,

regionFit : false,

userLocation : true

      });

      Titanium.Geolocation.getCurrentPosition(function(e) {

      if (e.error) {
    Ti.API.log('error: ' + JSON.stringify(e.error));

    return;

}

Titanium.App.Properties.setString('lat', e.coords.latitude);
Titanium.App.Properties.setString('long', e.coords.longitude);
Titanium.App.Properties.setString('alt', e.coords.altitude);
var circleOverlayDef2 = {

    type : "circle",
    center : {
        latitude : e.coords.latitude,
        longitude : e.coords.longitude,
    },
    radius : 100, //approximates 1000 miles
    //strokeColor : "red",
    strokeColor : "#2E0854",
    strokeAlpha : 0.5,
    fillColor : "#147a00",
    fillAlpha : 0.2,
    width : 1
};
mapView.addOverlay(circleOverlayDef2);

      });

      win1.add(mapView);
