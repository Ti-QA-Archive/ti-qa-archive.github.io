    var formView3 = Ti.UI.createView({
    width : '100%',
    height : Ti.UI.SIZE,

//This will hold current place lat long
var lat;
var lng;

//This will hold ids of complain type and id
var complaintype = 0;
var siteid = 0;

//This is hint for complain type and site code, we will check if site code hint is passed we will call webserivce for that
var hintComplainType = "??? ??????";
var hintSiteCode = "????? ????????";

//Data for two pickers, this is demo change accordingly
var claimantTypeData = ['??? ??????', '???????'];
var siteCodeData = [];

//Call lat long finder method
getCurrentLatLong(function() {

    var title;
    var subtitle;

    var addrUrl = "http://maps.googleapis.com/maps/api/geocode/json?sensor=true&latlng=" + lat + "," + lng;
    /* web-service call */
    var addrReq = Titanium.Network.createHTTPClient();
    addrReq.open("GET", addrUrl);
    addrReq.send(null);

    addrReq.onload = function() {
        var response = JSON.parse(this.responseText);

        if (response.status == "OK") {
            var resLen = response.results[0].address_components.length;
            for (var i = 0; i < resLen; i++) {
                switch (response.results[0].address_components[i].types[0]) {
                    // case "street_number":
                    // title = response.results[0].address_components[i].long_name + ", ";
                    // Ti.API.info("street number : " + response.results[0].address_components[i].long_name);
                    // break;
                    // case "route":
                    // title = title + response.results[0].address_components[i].long_name + ", ";
                    // Ti.API.info("street name : " + response.results[0].address_components[i].long_name);
                    // break;
                    case "locality":
                        title = response.results[0].address_components[i].long_name + " ";
                        Ti.API.info("city name : " + response.results[0].address_components[i].long_name);
                        break;
                    case "administrative_area_level_1":
                        subtitle = response.results[0].address_components[i].long_name + " ";
                        Ti.API.info("state name : " + response.results[0].address_components[i].long_name + " ");
                        break;
                    case "postal_code":
                        subtitle = subtitle + response.results[0].address_components[i].long_name + " ";
                        Ti.API.info("zip code : " + response.results[0].address_components[i].long_name);
                        break;
                    case "country":
                        subtitle = subtitle + response.results[0].address_components[i].long_name + " ";
                        Ti.API.info("country name : " + response.results[0].address_components[i].long_name + " ");
                        break;
                }
            }

            var Map = require('ti.map');

            var annotation = Map.createAnnotation({
                latitude : lat,
                longitude : lng,
                title : title,
                subtitle : subtitle,
                rightButton : Ti.UI.iPhone.SystemButton.DISCLOSURE,
                pincolor : Map.ANNOTATION_RED,
            });

            var mapview = Map.createView({
                width : '85%',
                height : 200,
                top : 10,
                borderColor : 'red',
                borderWidth : 2,
                mapType : Map.STANDARD_TYPE,
                animate : true,
                regionFit : true,
                userLocation : true,
                touchEnabled : true
            });

            mapview.setAnnotations([annotation]);

            var region = {
                latitude : lat,
                longitude : lng,
                animate : true,
                latitudeDelta : 0.1,
                longitudeDelta : 0.1
            };
            mapview.setRegion(region);

            // Handle click events on any annotations on this map.
            mapview.addEventListener('click', function(evt) {
                //Ti.API.info("Annotation " + evt.title + " clicked, id: " + evt.annotation.myid);
            });

            mapview.addEventListener('regionchanged', function(e) {

                mapview.removeAllAnnotations();

                var annot = Map.createAnnotation({
                    latitude : e.latitude,
                    longitude : e.longitude,
                    region : {
                        latitude : e.latitude,
                        longitude : e.longitude,
                        latitudeDelta : 0.01,
                        longitudeDelta : 0.01
                    },
                    // title : "Appcelerator Headquarters",
                    pincolor : Map.ANNOTATION_RED,
                    animate : true
                });
                //alert('Latitude >> ' + e.latitude + ' , Longitude >> ' + e.longitude);
                mapview.addAnnotation(annot);
            });

            formView3.add(mapview);
        } else {
            showAlert('', 'Unable to find Address');
        }

    };


});

//This function finds current place latitude longitude and assign values to lat long variable
function getCurrentLatLong(callback) {

    //Purpose
    Titanium.Geolocation.purpose = "Showing current location on a map";

    //Get best accuracy that is available
    Titanium.Geolocation.accuracy = Titanium.Geolocation.ACCURACY_BEST;

    //Set Distance
    Titanium.Geolocation.distanceFilter = 10;

    //Get lat long
    Titanium.Geolocation.getCurrentPosition(function(e) {
        if (e.error) {
            alert('Can not get your current location');
            return;
        }

        lng = e.coords.longitude;
        lat = e.coords.latitude;
        var altitude = e.coords.altitude;
        var heading = e.coords.heading;
        var accuracy = e.coords.accuracy;
        var speed = e.coords.speed;
        var timestamp = e.coords.timestamp;
        var altitudeAccuracy = e.coords.altitudeAccuracy;

        callback();

    });

}
