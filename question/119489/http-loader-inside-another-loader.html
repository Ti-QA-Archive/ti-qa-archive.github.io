HTTP Loader inside another loader

Hi everybody.

Here is the thing: I make a normal loader, but i need to call another loader from the first one, like this:


var sendNumbers = function(selected) {

    var loadaux = Titanium.Network.createHTTPClient();
    loadaux.open("GET","http://192.168.21.102/ZipZend.php?acc=1&z="+selected);
    loadaux.onload = function() {

        var intent = Titanium.Android.createIntent({
            action : Titanium.Android.ACTION_VIEW,
            type   : 'vnd.android-dir/mms-sms',
        });

        intent.putExtra('sms_body','This is a message');

        numbers = JSON.parse(this.responseText);

        intent.putExtra("address", numbers.data);

        alert(numbers.data);

        try {
            Titanium.Android.currentActivity.startActivity(intent);
        } catch (ActivityNotFoundException) {
            Titanium.UI.createNotification({ message : "Error" }).show();
        }
    }
}


loader.open("GET","http://192.168.21.102/ZipZend.php?acc=1");

loader.onload = function() {

    var selected = JSON.parse(this.responseText);

    // A lot of other code for tableviews and more ...

    var button = Titanium.UI.createButton({
        title: 'Send Message',
        top: 300
    });
    button.addEventListener('click',function(e) {
        sendNumbers(selected.data);
    });
}

As you can see, i wanna call the sendNumbers function, that uses another loader, from inside the original loader.

This doesn't work, of course, don't even show an error.

Any suggestions?

Thanks in advance!
