    var text = "welcome wo  shi  hao ren ,,1,,llllllll";

        text: text,
        attributes: [{

        color: '#000000',
        height: 20,
        width: 200,
        attributedString: attr
    });

    label.addEventListener('link', function(e) {
        alert(e);
    });

    self.add(label);
    return self;
}
