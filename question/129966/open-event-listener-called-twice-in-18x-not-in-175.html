Open Event Listener Called Twice in 1.8.X not in 1.7.5

I seem to be having nothing but trouble with 1.8.X what am I doing wrong? I want to create a tableView, put some rows in it, add that tView to the window and open the window. When I do it in 1.8.X my open event listener gets called twice and I get double the rows. To see this in action, make a new project, replace all the app.js code with my code below. I change the SDK version in TiApp to 1.7.5 and all is well, I change it to 1.8.0.1 or 1.8.0.1.RC3 and I get double the rows. Nothing else changes except the SDK version. Is is my install of Studio, my SDKs, what??? I have all but wasted two full days of work with various problems that seem to all be 1.8.X related. Anyone else having issues?

Running Titanium Studio, build: 1.0.8.201112271627 on a Windows 7 box developing for Android only at this point.

//Array for tableView
var assessmentDataArray = [];

var debugLabel = Titanium.UI.createLabel({
    bottom:10,
    height:'auto',
    text:'info ',
    width:'auto',
    color:'black'
})

var winAssessments = Titanium.UI.createWindow({
    title:'Assessments',
    backgroundColor:'#FFFFFF',
    fullscreen:false//makes it a "heavyweight window"
});

//This gets called twice in 1.8.X but only once in 1.7.5
winAssessments.addEventListener('open',function(e){
    debugLabel.text = debugLabel.text + 'In Open Event Listener ';
    LoadAssessments(assessmentDataArray);
    tvAssessments.data = assessmentDataArray;
});

var tvAssessments = Titanium.UI.createTableView({
    backgroundColor:'#FFFFFF',
    separatorColor:'#000000',
    top:5,
    borderColor:'#0000FF'
});

winAssessments.add(tvAssessments);
winAssessments.add(debugLabel);
winAssessments.open();

function LoadAssessments(arrayToBeLoaded){    
    for(var i = 0; i < 3; i++) //simulate things from the DB
    {
        var row = Titanium.UI.createTableViewRow({
            title:'This is row ' + i, //title seen by user in the row
            hasChild:true, //want down arrow on the right of the row
            aID:i, //custom property to hold DB primary key        
        });                        
        arrayToBeLoaded.push(row);
    }
    return arrayToBeLoaded;
}
