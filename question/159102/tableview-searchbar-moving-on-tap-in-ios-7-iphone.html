TableView Searchbar moving on tap in iOS 7 iPhone

I have a pretty standard SearchBar on a TableView that I'm using for filtering. In iOS 6, it appears in the second row. In iOS 7 it starts in the first row (where I want it), but jumps to the second row as soon as you tap it to enter data.

This is happening both on the simulator and test devices (handheld only).

    var search = Titanium.UI.createSearchBar({
        hintText : 'Enter Acronym',
        height: 42,
        top: 0
    });

    var data = getData();

    var table = Ti.UI.createTableView({
        data : data,
        search : search,
        top: 42
    });

    table.add(search);

Any help would be greatly appreciated.

  • Hello Ryan,

    For fixing your searchBar position issue you need to add your own search function like my following code.

    var win = Ti.UI.createWindow({backgroundColor: 'white'});
var searchbar = Ti.UI.createSearchBar({ top: 0, height: 43, showCancel: false });
win.add(searchbar);

function dumpData() {
    var data = [];
    for(var i = 0; i < 20; i ++) {
        data.push({title: Math.random().toString(36).substring(10)});
    }
    return data;
}

var data = dumpData();
var tbl = Ti.UI.createTableView({
    data: data,
    top: searchbar.height
});

win.add(tbl);
win.open();

searchbar.addEventListener('change', function(e){
    tbl.setData([]);

    var searchResults = mysearch(e.value, data);

    if (searchResults.length == 0) {// "No Result"
        tbl.setData([

            {title: 'No Result'},
        ]); 
    } else {
        tbl.setData( searchResults );
    }
});

searchbar.addEventListener('return', function(e){
    searchbar.blur();   
});

tbl.addEventListener('click', function(e) {
    searchbar.blur();
});

function mysearch(value, data) {
    var searchResults = [];
    for(var i = 0, len = data.length; i < len; i ++) {
        if (data[i].title.indexOf(value) != -1) { // or == 0: for prefix search
            searchResults.push(data[i]);    
        }
    }
    return searchResults;
}

    I hope it will to you.

    Thanks
