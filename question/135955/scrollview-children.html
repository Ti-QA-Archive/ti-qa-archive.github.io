      function removeAllChildren(viewObject){
    //copy array of child object references because view's "children" property is live collection of child object references
    var children = viewObject.children.slice(0);

    for (var i = 0; i < children.length; ++i) {
        viewObject.remove(children[i]);
    }
}

  • i use this as part of a 'refresh' function to clear out all child objects in the entire window to help with Garbage Collection and memory woes:

    
var win = Ti.UI.currentWindow;

    var recursive_killing = function(ob, parent) {
            // Do we have children?
            if (ob.children) {
                for (var c = (ob.children.length - 1); c >= 0; c--) {
                    recursive_killing(ob.children[c], ob);
                }
            }
            // If we have a parent (ie not the root node)
            if (parent) {
                // Do whatever you need to kill the object here
                parent.remove(ob);
                ob = null;
            }
        }
        // note the underscore here...
    if (win._children) {
        for (var i = win._children.length - 1; i >= 0; i--) {
            if (win._children[i] != null) {
                win._children[i].hide();// stops the user 'seeing' it in progress
                recursive_killing(win._children[i]);
                //remove the 'parent' too
                win.remove(win._children[i]);
                win._children[i] = null;
            }
        }
        win._children = [];
    }

    Might help ;-)

  • one easy approach for removing all children from scrollView (or any component) is
    remove scrollView, create the scrollView with same properties as like former and refer to the same variable name, add to its parent.

    with this approach, there will not be any time lagging while looping through scrollView children and removing each, if there are many child components in scrollView.

    you can also ensure that its child components are getting garbage collected in Instruments

    for ex: 

    var scrollView = Ti.UI.createScrollView({ ----- });
//
// scrollView cleanup
parent.remove(scrollView);
scrollView =  null;

scrollView = Ti.UI.createScrollView({ ---- });

parent.add(scrollView);

    hope this helps..
