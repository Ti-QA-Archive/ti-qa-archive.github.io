    statusBarHidden : true,

    navBarHidden :true,

    exitOnClose : false,

    fullscreen : true,

    • for iPhone…..

      var officialWin2 = Ti.UI.createWindow({

      url : 'links.js',

      navBarHidden : false,

      });

      var officialBtn2 = Ti.UI.createButton({

      left : 10,

      height : 30,

      width : 55,

      title :'back'

      });

      officialWin2.leftNavButton = officialBtn2;

      officialBtn2.addEventListener('click', function(e) {
      officialWin2.close();
      });

      Titanium.UI.currentTab.open(officialWin2, {

      animation : true
      });
