  • So here's some code to close the parent of the window. I then re-open the sub window (which in effect means only the parent window is closed).

    var baseWin = Titanium.UI.createWindow({
    backgroundColor : '#0000ff'
});

var subWin = Titanium.UI.createWindow({
    backgroundColor : '#ff0000',
    width: 100
});
baseWin.add(subWin);

baseWin.open();

// Comment these lines out to see both windows
subWin.getParent().close();
subWin.open();
