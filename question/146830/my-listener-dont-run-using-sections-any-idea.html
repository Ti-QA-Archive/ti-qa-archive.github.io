  • Hi,

    Use rowData instead of row:

    table.addEventListener('click', function(e){
        tab[0].open(W.Hermandades(e.rowData.idhermandad,e.rowData.dia));
    });

    But for using this idhermandad and dia should be custom property of tableview row.

        // CREAMOS LAS SECCIONES CON LOS DÍAS DE LA SEMANA
    // ###############################################
    for (var i = 0; i<diassemana.length; i++) {    
        secciones[i] = Ti.UI.createTableViewSection();
        var customView = Ti.UI.createView({height:'auto'});
        var customLabel = Ti.UI.createLabel({
            height:40,
            textAlign: Ti.UI.TEXT_ALIGNMENT_CENTER,
            width:oswidth,
            text:diassemana[i],
            font:{fontSize:16, font:customFont},
            backgroundColor:diassemanacolorseccion,
            color:diassemanacolorfila
        });
        customView.add(customLabel);
        secciones[i].headerView = customView;
    }

    // LLAMAMOS A NUESTRA BBDD INTERNA PARA OBTENER LAS HERMANDADES E INCLUIRLAS EN CADA SECCIÓN O DÍA DE LA SEMANA
    // ############################################################################################################
    var queryRS = db.execute('SELECT * FROM hermandad');
    var i = 1; // llevamos el control del número de hermandades que se van almacenando
    var j = 1; // llevamos el control del día de la semana en el que estamos

    while (queryRS.isValidRow())
    { 
      if (queryRS.fieldByName('dia') == j) 
            secciones[queryRS.fieldByName('dia')-1].add(Ti.UI.createTableViewRow({
                backgroundColor:(i%2==0)?diassemanacolorfilapar:diassemanacolorfilaimpar,
                color:diassemanacolorfila,
                font: { fontSize: 16, font:customFont},
                title:queryRS.fieldByName('nhermandad'), 
                leftImage:'images/escudo'+queryRS.fieldByName('idhermandad')+'.jpg', 
                idhermandad: queryRS.fieldByName('idhermandad'),
                dia:queryRS.fieldByName('dia'),
                hasChild:true
            }));

      if (i==hermandadesdia[j-1]) {
          j++;
          i=0;
      }
      i++;
      queryRS.next();
    }

    queryRS.close();

    var table = Ti.UI.createTableView({
        top:0,
        left:0,
        right:0,
        bottom:50,
        backgroundColor:backgroundColor,
        separatorColor:diassemanacolorseparador,
        sections:secciones
    });

    table.addEventListener('click', function(e){
        tab[0].open(W.Hermandades(e.row.idhermandad,e.row.dia));
