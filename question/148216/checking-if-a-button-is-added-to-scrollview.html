    hi,
    you can check with for loop 

    
for (var i in scrollView.children) {

    var child = scrollView.children[i];

    if (child.valueOf() == '[object Button]') {
        alert('Button);
    }
}

      for (var i in scrollView.children) {

    var child = scrollView.children[i];

    if (child.valueOf() == '[object Button]' && child.typ=='specific') {
        alert(' Specific Button);
    }
}

      Provided if you declare button with,

      typ:'specific'
