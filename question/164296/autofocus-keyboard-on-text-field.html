  • Hi Subhan Zaidi, Please try this code , This is very similar to stop Android textField from auto-focusing

    var win = Ti.UI.createWindow({
  backgroundColor: 'white'
});
win.open();
var aTextField = Ti.UI.createTextField({
    height : 35,
    top : 10,
    left : 40,
    width : 240,
    hintText : 'This is hint text',

});
aTextField.addEventListener('return', function(e) {
    aTextField.blur();
    alert('Input was: ' + aTextField.value);
});
win.add(aTextField);

aTextField.hide();

win.addEventListener('open', function(e) {

    setTimeout(function(e){
        aTextField.show();
    },1000) ;
});
