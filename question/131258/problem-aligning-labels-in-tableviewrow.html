Problem aligning labels in TableViewRow

Hi guys,

var label = Ti.UI.createLabel({
        color: '#516691',
        height: 12,
        text: 'label',
        top: 12,
        left: 0,
        width: 75,
        textAlign: Ti.UI.TEXT_ALIGNMENT_RIGHT,
        font: {
            fontSize: 12,
            fontWeight: 'bold'
        }
    });
var text = Ti.UI.createLabel({
        text: 'text',
        top: 10,
        bottom: 10,
        left: 85,

            fontSize: 12,
            fontWeight: 'bold'
        },
        height: 'auto'
    });
