    With regards to the API, please consult the official API

    Here is some sample code taken from suds.js to create a HTTP connection.

    xhr = Titanium.Network.createHTTPClient();
xhr.onload = function() {
    /* do some hot diggity action here */
};

xhr.onerror = function(e) {
    if( typeof _errorCallback !== 'undefined') {
        /* oh noeesss, it's broken */
    }
};
xhr.timeout = 3000; //ms
xhr.open('POST', 'www.thatwebsite.com');
xhr.setRequestHeader('Content-Type', 'yourcontenttype');
xhr.send(dataToSend);
