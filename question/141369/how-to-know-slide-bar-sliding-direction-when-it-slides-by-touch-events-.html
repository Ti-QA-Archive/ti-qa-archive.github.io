  • simply save the old value of the slider in a variable and compare the old value to the new value.
    If the old value is smaller than the new value = was slided to the right (in normal position), if the old value is heigher than the new value = was slided to the left.

    not tested example:

    var oldValue=0;

sliderbar.addEventListener('touchend',function(e) 
{ 
    if (oldValue>sliderbar.value){
        //slided to the left

    }else if(oldValue<sliderbar.value){
        //slided to the right

    }else if{
        //no change

    }

    oldValue=sliderbar.value;
});
