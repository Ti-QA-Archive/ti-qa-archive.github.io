    Hello Benjamin Frazier,

    As per your requirement, I have added one example for split window application. Where I have added navigation buttons in master and details windows Please see run this I hope its OK for your requirements. 

    var myProfileWin = Titanium.UI.createWindow({
    top : 0,
    left : 0,
    backgroundColor : '#fff',
});
var masterView = Titanium.UI.createWindow({
    top : 0,
    left : 0,
    layout : 'absolute',
    backgroundColor : '#eee',
});

var nav = Ti.UI.iPhone.createNavigationGroup({
    window : myProfileWin
});
var masterViewnav = Ti.UI.iPhone.createNavigationGroup({
    window : masterView
});
var self = Ti.UI.iPad.createSplitWindow({
    detailView : nav,
    masterView : masterViewnav,
    showMasterInPortrait : true,

});

/***************************************************************************/

// Create a Button.
var aButton = Ti.UI.createButton({
    title : 'aButton',

});

// Listen for click events.
aButton.addEventListener('click', function() {
    alert('\'aButton\' was clicked!');
});

// Add to the parent view.
masterView.setLeftNavButton(aButton);

myProfileWin.setRightNavButton(aButton);
var list = Ti.UI.createListView({
    left : 10,
});

masterView.add(list);
var listSection = Ti.UI.createListSection();

list.sections = [listSection];

var data = [];

data.push({
    properties : ( {
        title : 'Edit Profile',
        color : '#cc9900',
        font : {
            fontSize : 16,
            fontWeight : 'bold',
        },
    })
});

data.push({
    properties : ( {
        title : 'Inbox',
        color : '#cc9900',
        font : {
            fontSize : 16,
            fontWeight : 'bold',
        },
    })
});

data.push({
    properties : ( {
        title : 'My Network',
        color : '#cc9900',
        font : {
            fontSize : 16,
            fontWeight : 'bold',
        },
    })
});

listSection.setItems(data);

// Create a Label.
var aLabel = Ti.UI.createLabel({
    text : 'aLabel',
    color : '#000',
    font : {
        fontSize : 30
    },

    textAlign : 'center'
});

// Add to the parent view.
myProfileWin.add(aLabel);

self.open();

    • By the way to get the item that is click you have to add the following onItemclick in the XML: 

      <ListView onItemclick="onItemClick" id="list">

      And then in the javascript: 

      function onItemClick(e) {
    // get the clicked section
    var section = $.list.sections[e.sectionIndex];

    // get the clicked item from that section
    var item = section.getItemAt(e.itemIndex);

    // print the item's title
    Ti.API.info('itemclick: ' + item.properties.title);
}
