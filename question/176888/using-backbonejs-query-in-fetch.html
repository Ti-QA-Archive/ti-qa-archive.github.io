Using Backbone.js query in fetch

Hi,

I have the following code to query name and date columns in my db. The dates are stored as strings. However, when I query for a date no result set is returned. But if I run the same query in SQLite Professional I get a result set:

curPage = 0;

var temp = Alloy.createCollection('estimateModel');
var offSet = curPage * pageSize;

var sql = "SELECT * FROM estimateModel WHERE (name || modified) LIKE \'%" + val + "%\' ORDER BY name LIMIT " + pageSize + " OFFSET " + offSet;

temp.fetch({
    query : sql
});

var data = [];
temp.each(function(item) {
    // Create table rows...
    data.push(addRow(item));
});

Here is an example of the sql query generated by the string above:
SELECT * FROM estimateModel WHERE (name || modified) LIKE &#39;%7&#x2F;11%&#39; ORDER BY name LIMIT 50 OFFSET 0

Thanks for your help in advance!
