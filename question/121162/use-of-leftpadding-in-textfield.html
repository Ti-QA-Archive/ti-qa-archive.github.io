  • It's used to set the space to the left and right of the whole field.

    use it like this:

    
var a_field = Titanium.UI.createTextField({
    color:'#222222',
    hintText:'this is a hint',
    height:35,
    top:10,
    right:5,
    left:5,
    paddingLeft:20,      // pad left by 20 pixels
    paddingRight:20,    // pad right by 20 pixels
    autocorrect:false,
    clearButtonMode:Titanium.UI.INPUT_BUTTONMODE_ALWAYS,
    returnKeyType:Titanium.UI.RETURNKEY_DONE,    
    autocapitalization:Ti.UI.TEXT_AUTOCAPITALIZATION_NONE,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED
});    

win.add(a_field);
