Errore including html file in a webview

This is the window file

function InfoContent() {

    var webview = Titanium.UI.createWebView({
        url: Titanium.Filesystem.resourcesDirectory + "/info.html"
        });

    var win5 = Ti.UI.createWindow({

        backgroundColor: '#790000',
        orientationModes : [Titanium.UI.PORTRAIT]
    });
    win5.navBarHidden = true;
    win5.add(webview);
    return win5;
}
module.exports = InfoContent;

And this is the info.html file

<!DOCTYPE html>
<html>
    <head>
        <title>Info</title>
    </head>
    <body>
        Direttore 
    </body>
</html>

the error is: "Can't find variable: webview (unknown file)." But the info.html file is in the Resources directory. So I can't understand why the app doesn't work!

Any suggestion? Thanks!
