How do I clear a textfield on edit?

I want to clear my textfield on edit.

code:

var tal3 = Titanium.UI.createTextArea({

editable: true,
value:'Skriv din kommentar här',
height:140,
width:280,
top:155,
font:{fontSize:16, fontWeight:''},
color:'#000000',
textAlign:'left',
appearance:Titanium.UI.KEYBOARD_APPEARANCE_ALERT,
keyboardType:Titanium.UI.KEYBOARD_NUMBERS_PUNCTUATION,
returnKeyType:Titanium.UI.RETURNKEY_EMERGENCY_CALL,
borderWidth:2,
borderColor:'#bbb',
borderRadius:5,
suppressReturn:false,

});

win.add(tal3);

    Hello David,

    you have to write 'focus' event of tal3.

    tal3.addEventListener('focus',function(e){
    tal3.value = '';
});
