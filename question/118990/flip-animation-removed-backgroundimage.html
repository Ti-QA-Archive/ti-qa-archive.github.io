Flip animation removed backgroundImage

I have some views with 2 labels. The original flip works perfect but when I do the flip back over the backgroundImage is missing. Any ideas or solutions are appreciated.

for(var i=1;i <= total; i++){
    view[i] = Titanium.UI.createView({
        id: i,
        height: height,
        width: width,
        top: this_top,
        left: this_left
    });

    face_up[i] = Titanium.UI.createLabel({
        id: i,
        name: 'fuc',
        backgroundColor:'blue'
    });


    face_down[i] = Titanium.UI.createLabel({
        id: i,
        name: 'fdc',
        backgroundColor:'pink',
        backgroundImage:'image.jpg'
    });

    view[i].add(face_up[i]);
    view[i].add(face_down[i]);
    window.add(view[i]);    
}

window.addEventListener('click', function(e){
    if(e.source.name == 'fuc'){    
        t = Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT;
        e.source.animate({view:face_down[e.source.id],transition:t});
    } else if (e.source.name == 'fdc'){
        t = Ti.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT;
        e.source.animate({view:face_up[e.source.id],transition:t});
    }
});

  • Here is the workaround: use modal window and modal animations instead:

    app.windows.settingsWindow.open({
    modal: true,
    modalTransitionStyle: Ti.UI.iPhone.MODAL_TRANSITION_STYLE_FLIP_HORIZONTAL
});
