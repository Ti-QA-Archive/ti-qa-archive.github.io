getting children from the table ViewRow 'click' event is giving error in 1.8.0 and 1.8.1

Titanium mobile project, titanium 1.8.1, android emulator 2.3.3,

Hi,

In my application , getting the row values in the row event is giving runtime error.It works fine in titanium 1.7.5. I dont know what is the problem in 1.8.0 and 1.8.1

The code is as follows:

var win = Ti.UI.currentWindow;
var table = Ti.UI.createTableView();
var row = Ti.UI.createTableViewRow({layout:'horizontal'});
var data = [];

var name = Ti.UI.createLabel({text:'raj'});
var addr = Ti.UI.createLabel({text:'hyderabad'});
var phone = Ti.UI.createLabel({text:'5454545454'});

row.add(name);
row.add(addr);
row.add(phone);

data.push(row);
table.setData(data);
win.add(table);

//Row event for getting the data
row.addEventListener('click', function(e){
    var Name = e.rowData.children[0].text; //error is coming here
    var Addr = e.rowData.children[1].text;
    var Phone = e.rowData.children[2].text;

    alert('Name:'+Name+'\nAddress:'+Addr+'\nPhone:'+Phone);
});

/*
//table event for getting the data
table.addEventListener('click', function(evt){
    var Name = evt.data[0].rows[evt.index].children[0].text; //error is coming here
    var Addr = evt.data[0].rows[evt.index].children[1].text;
    var Phone = evt.data[0].rows[evt.index].children[2].text;

    alert('Name:'+Name+'\nAddress:'+Addr+'\nPhone:'+Phone);
});
*/
win.open();

I am getting the following runtime error.
This is the console :

E/TiJSError: (main) [0,1968996] - Message: Uncaught TypeError: Cannot read property '0' of undefined
E/TiJSError: (main) [1,1968997] - Source: 
    var Name = e.rowData.children[0].text; //error is coming here
E/V8Exception: Exception occurred at app.js: Uncaught TypeError: Cannot read property '0' of undefined

Please help me..

Thanks.
