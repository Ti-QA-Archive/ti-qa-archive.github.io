back button within a url page using webview

I created links to open external pages using webview. The webview has back buttons (jquery mobile style) inside but does not work when viewed in the apps. How can I make a back button so that the webview can shift back one historical page in the native app? There were previous questions mentioning back button but no definite and great answers about it. Can someone share the best way to implement back buttons in a webview?

  • Here you go

    var self = Ti.UI.createWindow({
        title : 'Ya\'ll Lifestyle',
        navBarHidden : false,
        backgroundColor : 'white'
    });
    var webView = Titanium.UI.createWebView({
        url : 'http://www.yalllifestyle.com/',
        backgroundColor : '#000'
    });

    self.add(webView);
    var actInd = Ti.UI.createActivityIndicator({
        width : '10%',
        height : '10%',
    })
    self.add(actInd);
    actInd.show();
    webView.addEventListener("load", function(e) {
        actInd.hide();
    });
    var backButton = Titanium.UI.createButton({
        title : "<",
        bottom : '2%',
        left : '2%',
        width : '15%',
        height : '7%'
    });
    self.add(backButton);
    var forwardButton = Titanium.UI.createButton({
        title : ">",
        bottom : '2%',
        right : '2%',
        width : '15%',
        height : '7%'
    });
    self.add(forwardButton);
    backButton.hide();
    forwardButton.hide();
    backButton.addEventListener("click", function(e) {
        webView.goBack();
    })

    forwardButton.addEventListener("click", function(e) {
        webView.goForward();
    })
    if (webView.canGoBack()) {
        backButton.show();
    } else {
        backButton.hide();
    }

    if (webView.canGoForward()) {
        forwardButton.show();
    } else {
        forwardButton.hide();
    }
    self.open();

    Thanks
