    Hi Jon,
    What you need to do is add your event listener to your mask. But apply your touchmove logic to your underlying image.

    var win = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
}),
label = Titanium.UI.createLabel({
    color:'#999',
    text:'I a LABEL',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
}),
mask = Ti.UI.createView({
    width:200,
    height:50,
    backgroundColor:'#000',
    ID:'mask',
    opacity:0.9
});
win.add(label);
win.add(mask);
mask.addEventListener('touchmove', function(e) {
    //DO TOUCHMOVE LOGIC
});
win.open();
