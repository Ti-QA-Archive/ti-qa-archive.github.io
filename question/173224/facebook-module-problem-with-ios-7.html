  • Hi samuel,

    Have you add the below property in tiapp.xml..? As mention in doc 

     <property name="ti.facebook.appid">FACEBOOK_APP_ID</property>

    <modules>
    <!-- Add the appropriate line(s) to your modules section -->
    <module platform="android">facebook</module>
    <module platform="iphone">facebook</module>
</modules>

    var fb = require('facebook');
fb.appid = "your facebook_app_id";
fb.permissions = ['publish_stream', 'read_stream','email'];

fb.addEventListener('login', function(e) {
            if (e.success) {
                fb.forceDialogAuth = true;
                var params = {
                    access_token : fb.accessToken
                };
                fb.requestWithGraphPath('/me', params, 'GET', function(e) {
                    var result = JSON.parse(e.result);
                    alert('result-->'+result);

                });
            } else if (e.error) {
                alert('Error-->' + e.error);

            } else if (e.cancelled) {
                alert('cancelled-->' + e.cancelled);

            }

        });
    fb.authorize();
