Mobile Events won't fire at all!

Hi there.

I was just experiementing with Titaniums IDE and I just can't seem to figure out how to get a text field in iOS to respect a click event. Not only that, but it seems that NO event is being fired in the iPhone simulator.

Could someone explain to me why the text fields won't respond to a click event? I just want them to focus.

Here is my code:


var win = Ti.UI.currentWindow;

//-- Create the sub windows
var login = Ti.UI.createWindow();

login.layout = 'vertical';


var emailLabel = Ti.UI.createLabel({
    text:'Email:',
    color:'#444',
    font: {fontSize:13},
    width: 'auto',
    height: 'auto'
});


var email = Ti.UI.createTextField({
    color:'#444',
    height:35,
    width:250,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
    hintText: 'enter your email...',
    editable: true,
    enabled: true
});





var passLabel = Ti.UI.createLabel({
    text:'Password:',
    color:'#444',
    font: {fontSize:13},
    width: 'auto',
    height: 'auto'
});



var password = Ti.UI.createTextField({
    color:'#444',
    height:35,
    width:250,
    borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
    passwordMask: true,
    hintText: 'your password',
    editable: true,
    enabled: true
});



login.add(email);
login.add(password);

login.addEventListener("open", function(event,type){
    //email.focus();
});

login.open();

I have even tried things like:

email.addEventListener("click", function(event,type){
    email.focus();
});

Still nothing…..
