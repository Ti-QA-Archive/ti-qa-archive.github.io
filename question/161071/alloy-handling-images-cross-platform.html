  • I've created this function to help me out:

    Don't know if there is an easier way - for me / my case it's working perfect. :)

    Alloy.Globals.scaleObject = function(_object, _targetWidth, _sourceWidth, _sourceHeight  ) {

    var densityFactor;
    if (OS_ANDROID) {
        densityFactor = Ti.Platform.displayCaps.logicalDensityFactor;
    } else {
        densityFactor = 2;
        // IOS: If  width is equal to the platform width
        if (_targetWidth == Titanium.Platform.displayCaps.platformWidth) {
            _targetWidth = _targetWidth * densityFactor;
        }
    }    
    // Width
    _object.width = (_targetWidth / densityFactor );

    // Height
    var scaleFactor = _sourceWidth / _object.width;
    _object.height     = _sourceHeight / scaleFactor;

    Titanium.API.info( "Adjusted to: " + _object.width + " //// " + _object.height );
};
