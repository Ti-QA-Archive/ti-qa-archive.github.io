Window does not add imageView for android in Titanium 3.1.3

i have a problem for Titanium 3.1.3, but previous version of titanium is working correctly. if i try to add the below code for android, the window does not add imageview but for iphone it does.

function ApplicationWindow(title) {

    var homeWindow = Titanium.UI.createWindow({
        backgroundColor : '#fff',
        tabBarHidden : true,
        navBarHidden : true,
        //url : 'home.js',
        fullscreen : false,
        exitOnClose : true
    });

    var imageView = Titanium.UI.createImageView({
        image : 'football.png',
        width : 100,
        height : 100,
        left : 0
    }); 

    homeWindow.add(imageView);

    return homeWindow;
    };


module.exports = ApplicationWindow;

But if i try below code then it works for both android & iphone -

function ApplicationWindow(title) {

    var homeWindow = Titanium.UI.createWindow({
        backgroundColor : '#fff',
        tabBarHidden : true,
        navBarHidden : true,
        url : 'home.js',
        fullscreen : false,
        exitOnClose : true
    });

    return homeWindow;
    };


module.exports = ApplicationWindow;

and In home.js -

var homeWindow = Titanium.UI.currentWindow;

img = Titanium.UI.createImageView({
    image : 'home-icon.jpg',
    height : 250,
    width : 170,
    top : 5,
    bottom : 10
});

homeWindow.add(img);

now i want to know what is the problem with my first code. is it bug for Titanium 3.1.3 that only permits window to add imageView with url property ? please clarify……..

    Hello Nuibb Hasan,

    Please try this. For android you need to &#x2F; in image location but for iOS this is not needed.

     var homeWindow = Titanium.UI.createWindow({
        backgroundColor : '#fff',
        tabBarHidden : true,
        navBarHidden : true,
        //url : 'home.js',
        fullscreen : false,
        exitOnClose : true
    });

    var imageView = Titanium.UI.createImageView({
        image : '/football.png',
        width : 100,
        height : 100,
        left : 0
    }); 

    homeWindow.add(imageView);


homeWindow.open();
