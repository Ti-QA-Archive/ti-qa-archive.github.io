Scrollable Views Isse

Hello,

i'm trying to create a slide show using scrollable views:

var slide_path = path + 'slide.xml';
xhr.open("GET", slide_path);
xhr.onload = function()
{
    try
    {
        var doc = this.responseXML.documentElement;
        var items = doc.getElementsByTagName("image");
        for (var c=0;c<items.length;c++)
        {
            var item = items.item(c);
            url = item.getAttribute("url");
            var image_url  = path+url;
            //alert(path+url);
            views.push( Ti.UI.createImageView({
                image: image_url,
                top: 0, left: 0,
                width: 320,
                height: 50
                }));    
        }
    }
    catch(E)
    {
        alert(E);
    }
};
Ti.API.debug(views);
xhr.send();
var scrollView = Titanium.UI.createScrollableView({
    views:views,
    top: 0, right: 0, bottom: 0, left: 0,
    width:320,
    height:'auto'
});

win1.add(scrollView);
win1.open();

if i write alert( image_url) in the for, I can get the image url…

but when i run it no images are displayed and i get only a white screen
