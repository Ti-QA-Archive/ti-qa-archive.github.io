Can 2 views share the same backgroundImage?

I'm working off the open source struct project from Titanium. I started by modifiying the bazview and barview to empty views with a background image (the same one). Android forced a close. When I commented out one of the backgroundImage attributes it worked fine.

any ideas?

// bazview.js
S.ui.createBazView = function(){
        var view = Ti.UI.createView($$.stretch);

        view.add(Ti.UI.createLabel(bt.combine($$.headerText, {text:'Check in'})));
        view.add(Ti.UI.createView({
            top:$$.headerView.height,
            bottom:0,
            right:0,
            left:0,
            backgroundImage:'images/mock.png'
        }));

        return view;
};

//barview.js
S.ui.createBarView = function(){
        var view = Ti.UI.createView($$.stretch);

        view.add(Ti.UI.createLabel(bt.combine($$.headerText, {text:'Check in'})));
        view.add(Ti.UI.createView({
            top:$$.headerView.height,
            bottom:0,
            right:0,
            left:0,
            backgroundImage:'images/mock.png'
        }));

        return view;
};
