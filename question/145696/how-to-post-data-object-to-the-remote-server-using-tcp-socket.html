How to POST data object to the remote server using TCP Socket

Hi ,
Am looking to send JSON object to the remote server using TCP Socket.

JSON object : {"AuthType":"BASIC","Password":"password","UserName":"username"} to the remote server using TCP Socket.

Sample Code used for connecting Remote site using TCP Socket:

var controlSocket = Titanium.Network.Socket.createTCP({
host: localhost,
port: 8888,
timeout: 5000,
connected : function(e) {
Ti.Stream.pump(controlSocket, readCallback, 1024, true);
Ti.Stream.write(controlSocket, Ti.createBuffer({
value : 'POST /json.php\r\n'+ 'Host: localhost:8888\r\nContent-Type : application/json;charset=utf-8\r\n\r\n'
}), writeCallback);
},
error : function(e) {
Ti.API.info("Error " + e.errorCode);
},
});
controlSocket.connect();
function readCallback(e) {
Ti.API.info('Successfully read from socket.');
}
function writeCallback(e) {
Ti.API.info('Successfully wrote to socket.');
}
