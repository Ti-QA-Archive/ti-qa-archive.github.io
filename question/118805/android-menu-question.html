Android menu Question

How am i able to acces the .js file that includes the android menu?

Why do i need this? I have a tabgroup which allows users to fill in data to create a new user or company.

Now i want to have a Android menu button, that when pressed activates a function, or acces data from the .js file that included the menu.

Here is a sample code that should give the idea what i want to do, but this code gives me the following error: reference error: hello is not defined.

how can i fix it, code below:

// Window.js
Ti.include('menu.js'); 
// this sets the background color of the master UIView
Ti.UI.setBackgroundColor('#000');

var tabgroupNote = Ti.UI.createTabGroup();
var Window = Ti.UI.createWindow({
            backgroundColor:'#000',
            barColor:'#111',
            textAlign: 'center'
           // url: 'menu.js'
});

var label1 = Titanium.UI.createLabel({
    color:'#999',
    text:'I am Window 1',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
});

var tab1 = Ti.UI.createTab({
        icon: '../images/light_note.png',
        title: 'New note',
        window: Window
});

function hello(){
   Ti.API.info('Hello world');
}

Window.add(label1);
tabgroupNote.addTab(tab1);
tabgroupNote.open();

// menu.js

var new1 = null;

var menuHandler = function() {
    new1.addEventListener('click', function() {
        hello();
    });
};

var activity = Ti.Android.currentActivity;
activity.onCreateOptionsMenu = function(e) {
    var menu = e.menu;
    new1 = menu.add({title : 'Create new note'});
    menuHandler();
};

Can someone help me in this
