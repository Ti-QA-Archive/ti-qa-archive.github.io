Dynamically adjust spacing on Labels

Hi how can I dynamically set the top spaces of label.The text of movieTitle can be long so it can have two or more lines of words. and sometimes it overlaps the genre_label. How can I set properties for genre label so that it will be right next to moveTitle.

The 2 labels::

var view = Ti.UI.createView({
    height: 'auto'
});


var movieTitle = Titanium.UI.createLabel({
    top:10,
    left: 15,
    font: {
        fontFamily:'HelveticaNeue',
    fontSize:30,
    fontWeight: 100
    },
    color: '#000',
    text: movie.title,
    width: 'auto',
    height: 'auto'
});
view.add(movieTitle);

var genre_label =  Ti.UI.createLabel({
    text: (movie.genres ? movie.genres : "-"),
    left:15,
    top: 15,
    width:'auto',
    bottom:15,
    height:30,
    textAlign:'left',
    color:'#444444',
    touchEnabled: false,
    font: {
        fontSize:15
    }
});
view.add(genre_label);

Ti.UI.currentWindow.add(view);

  • Accepted Answer

    var w = Ti.UI.createWindow({backgroundColor:'white'});

var view = Ti.UI.createView({
    top:0,
    height : Ti.UI.SIZE,
    layout : 'vertical'
});

var movieTitle = Titanium.UI.createLabel({
    top : 0,
    left : 15,
    font : {
        fontFamily : 'HelveticaNeue',
        fontSize : 30,
        fontWeight : 100
    },
    color : '#000',
    text : 'This is looooooooooooooooooooooooooooooooooooong title',
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE
});
view.add(movieTitle);

var genre_label = Ti.UI.createLabel({
    text : 'THIS IS A GENRE',
    left : 15,
    top : 15,
    bottom : 15,
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
    textAlign : 'left',
    color : '#444444',
    touchEnabled : false,
    font : {
        fontSize : 15
    }
});
view.add(genre_label);

w.add(view);
w.open();
