  • Hi tareq alzhrani
    I think it will help you. Try this

    var win1 = Titanium.UI.createWindow({backgroundColor:'red'});

var win2 = Titanium.UI.createWindow({backgroundColor:'gray'});
win1.open();

win1.addEventListener('swipe', function() {
    win1.close();
    win2.open();
});

win2.addEventListener('swipe', function() {
    win2.close();
    win1.open();
});
