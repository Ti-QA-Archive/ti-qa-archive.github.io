ScrollView Touch Area Not Refreshing On Orientation Change

Hi

Hope someone can help me. I am trying to make a simple image scroller. The issue I have is that when I flip to landscape the left and right edges of the screen do not respond to the touch. I am guessing this is down to the fact that the scrollView size isnt being refreshed somehow. However I have tried numerous ways to solve this and resizing it. My code is below would really love some help.

var comicViewer = Titanium.UI.createView({
  top:0,
  width:320,
  height:460,
  opacity:1,

});



var comicHolder = Titanium.UI.createView({
  top:0,
  left:0,
  right:0,
  opacity:1,

});

comicViewer.add(comicHolder)


var scrollView = Titanium.UI.createScrollView({
    contentWidth:'auto',
    contentHeight:'auto',
    top:0,
    left:0,
    borderColor:'#ccc',
      borderWidth:3,
    showHorizontalScrollIndicator:true
});


var comic = Titanium.UI.createImageView({
    image:'images/cartoons/1.jpg',
    width:1114,
    height:320,
    left:0,
    top:0,
    opacity:1
});

scrollView.add(comic);
comicHolder.add(scrollView)
comicViewer.add(comicHolder)

// Add Event Listner To listen for double click to back to menu
scrollView.addEventListener('dblclick',function(e)
{
    Ti.API.info('double clicked');
}) 



// Make sure we resize on rotate
Ti.Gesture.addEventListener('orientationchange', function(e){
    //Ti.API.info(Titanium.UI.orientation);


     if(Titanium.UI.orientation == 1 || Titanium.UI.orientation == 2){
          // potrait
          scrollView.width=320;
          scrollView.height=320;
          comicViewer.remove(comicHolder); 
         comicViewer.add(comicHolder); 

      }
      if(Titanium.UI.orientation == 3 || Titanium.UI.orientation == 4){
           //landscape

          comicHolder.width=480;
          scrollView.width=480;
          scrollView.left=0;
        scrollView.right=0;
        comicHolder.left=0;
        comicHolder.right=0;
        comicHolder.remove(scrollView); 
         comicHolder.add(scrollView);

      }



});
