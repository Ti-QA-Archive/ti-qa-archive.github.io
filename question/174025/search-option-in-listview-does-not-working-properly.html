search option in listview does not working properly

Hi, I have problem with searchView in listview. i want to write a keyword in the searchbar and some value related to that keyword will be populated in the listview. I tried like below code but that does not working. when i write a keyword (suppose 'a'), no values are populated in the listview but if i erase that keyword then some value (in my code "p, pp, ppp") are populated in the listview. i want to - when i write 'a' (according to my code), then 'a, aa, aaa' should populated in the listview but thats not happening. why ?

var win = Ti.UI.createWindow();

var search = Ti.UI.createSearchBar({
    hintText : "Write Search Keyword",
    showCancel : true
});

search.addEventListener('change', function(e) {
        renderData(e.value.toString());
});

//Listview template
var plainTemplate = {
    childTemplates : [{
        type : 'Ti.UI.Label',
        bindId : 'title',
        properties : {
            color : '#000',
            height : '30dp',
            width : Ti.UI.FILL,
            left : '15dp',
            font : {
                fontFamily : 'Arial',
                fontWeight : 'bold',
                fontSize : '14dp'
            }
        }
    }],
    properties : {
        height : '30dp',
        backgroundColor : '#FFF'
    }
};

var listView = Ti.UI.createListView({
    //backgroundColor:'transparent',
    left : '5dp',
    right : '5dp',
    top : '1%',
    height : '98%',
    borderRadius : 5,
    templates : {
        'default' : plainTemplate
    },
    defaultItemTemplate : 'default',
    searchView : search,
    caseInsensitiveSearch : true
});

var _section = Ti.UI.createListSection();

win.add(listView);

function dataToLoad(key) {
    if (key === 'a') {
        var _array = ['a', 'aa', 'aaa'];
        return _array;
    } else if (key === 'b') {
        var _array = ['b', 'bb', 'bbb'];
        return _array;
    } else if (key === 'c') {
        var _array = ['c', 'cc', 'ccc'];
        return _array;
    } else if (key === 'd') {
        var _array = ['d', 'dd', 'ddd'];
        return _array;
    }else{
        var _array = ['p', 'pp', 'ppp'];
        return _array;
    }
}

//For Edit/View and Add(All)
function renderData(_key) {
    _section.setItems([]);
    var data = [];
    var _array = dataToLoad(_key);

    Ti.API.info(_key + '  ' + _array);
    for (var i = 0, j = _array.length; i < j; i++) {
        data.push({
            title : {
                text : _array[i]
            },
            // Sets the regular list data properties
            properties : {
                itemId : i,
                accessoryType : Ti.UI.LIST_ACCESSORY_TYPE_NONE
            }
        });
    }

    _section.setItems(data);
    listView.setSections([_section]);
}

win.open();

    Hi

    This is easier than you think but only because the searchBar has a hidden life doing something slightly different.

    The searchBar actually does two different jobs;

    • returns events to you so you can make choices - for example populate a listview yourself.
    • filter the existing entries on a listview or tableView

    You have currently got it in filter mode. This happens when you set the listview property searchView or the tableView property search to your search bar variable. This the connects the two and acts as a filter

    Easy solution remove the property searchView (or search with a tableView) and add the searchBar to the win as you have done with listview. Make sure the listview leaves space at the top if the window for it as this will no longer be automatic.

    Your event listener will now respond as the filtering is not happening automatically now.
