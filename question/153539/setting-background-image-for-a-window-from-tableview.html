setting background image for a window from tableview

Hi everybody
i create a table view and every row take me to another window the problem is that idon't know how to set a backgroundImage to the new window or change the color of the row's title !! can someone tell me how please
this is the code i did 


var win=Titanium.UI.currentWindow;
var data = [];
data[1] = Ti.UI.createTableViewRow({path:'rechercheproduit.js', subtitle:'subtext1',hasChild:true,title:'Produits',opacity:2.5,
font:{fontSize:19}});
data[2] = Ti.UI.createTableViewRow({path:'promotion.js', hasChild:true,title:'Promotions',font:{fontSize:19}});

data[3] = Ti.UI.createTableViewRow({path:'centrecommercial.js', hasChild:true,title:'Magasins',font:{fontSize:19}});

// create table view
var tableview = Titanium.UI.createTableView({

    data:data
});
 tableview.addEventListener('click', function(e){
    if (e.rowData.path){
        newWindow = Titanium.UI.createWindow({

        url:e.rowData.path
        });
    }
    Titanium.UI.currentTab.open(newWindow);
});

win.add(tableview);

thanks in advance

  • set backgroundImage of newWin in tableView's Click listner

    tableview.addEventListener('click', function(e){
    if (e.rowData.path){
        newWindow = Titanium.UI.createWindow({
        backgroundImage : '/images/bg.png',,  // bgImage of opening window
        url:e.rowData.path
        });
    }
    Titanium.UI.currentTab.open(newWindow);
});

  • var win=Titanium.UI.currentWindow;
var data = [];
data[1] = Ti.UI.createTableViewRow({path:'rechercheproduit.js', subtitle:'subtext1',hasChild:true, color:"#f00", title:'Produits',opacity:2.5,
font:{fontSize:19}});
data[2] = Ti.UI.createTableViewRow({path:'promotion.js', hasChild:true, color:"#ff0", title:'Promotions',font:{fontSize:19}});

data[3] = Ti.UI.createTableViewRow({path:'centrecommercial.js', hasChild:true, color:"#00f", title:'Magasins',font:{fontSize:19}});

// create table view
var tableview = Titanium.UI.createTableView({

    data:data
});
 tableview.addEventListener('click', function(e){
    if (e.rowData.path){
        newWindow = Titanium.UI.createWindow({
         backgroundColor:"#f00",
            url:e.rowData.path
        });
    }
    Titanium.UI.currentTab.open(newWindow);
});

win.add(tableview);
