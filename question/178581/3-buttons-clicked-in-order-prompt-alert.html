  • You can achieve that using "click" event listener on buttons. Below is a sample code to achieve that goal.

    var aClicked = false,bClicked = false;
var ButtonA = Ti.UI.createButton ({
    backgroundImage:"buttons/buttonA.png",
    top:110,
});

var ButtonB = Ti.UI.createButton ({
    backgroundImage:"buttons/buttonB.png",
    top:200,
});

var ButtonC = Ti.UI.createButton ({
    backgroundImage:"buttons/buttonC.png",
    top:300,
});

ButtonA.addEventListener('click',function(e){
    aClicked = true;
});
ButtonB.addEventListener('click',function(e){
    if(aClicked){
        bClicked = true;
    }
});
ButtonC.addEventListener('click',function(e){
    if(aClicked && bClicked){
        alert("Three Buttons Clicked in Order");
        aClicked = bClicked = false; // to reset
    }
});
