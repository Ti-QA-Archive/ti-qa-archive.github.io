PickerRow only works with 'change' and not 'click'

Hi

I cant seem to get a picker/picker row to work with a click rather a change. The Titanium.UI.PickerRow has an Event of click.

Any ideas???

var picker = Titanium.UI.createPicker({bottom:0, zIndex:100});

listButton.addEventListener('click', function()
{
    picker.add(Titanium.UI.createPickerRow({title:'List 1',list_id:'1'}));
    picker.add(Titanium.UI.createPickerRow({title:'List 2',list_id:'2'}));
    picker.add(Titanium.UI.createPickerRow({title:'List 3',list_id:'3'}));
    picker.add(Titanium.UI.createPickerRow({title:'List 4',list_id:'4'}));
    picker.selectionIndicator = true;
    win.add(picker);
});

picker.addEventListener('click',function(e)
{
    Titanium.API.info("You selected id: "+e.row.list_id);
});

  • I don't think you can get the selected row at the click event.

    The value is set after the picker changes it's value and that it's done after the click on a row occurs - it's done when the animation ends. So you are somehow forced to use the change event.
