  • this one removes itself when finished ;)

    var isPlaying = false;
if (!isPlaying)
{
  isPlaying = true;
  var sound = Titanium.Media.createSound({url:'yoursound.mp3'});
  endListener = sound.addEventListener('complete', function()
  {
    isPlaying = false;
    sound.removeEventListener('complete',endListener);
    sound.release();
  });
  sound.volume = 1;
  sound.play();
}
