  • with Titanium SDK 1.6.2 you can navigate the DOM with the function:

    var xmlDOM = Titanium.XML.parseString('file.xml').documentElement;
var branch = xmlDOM.getElementsByTagName('my_tag');
var leaf = branch.getElementsByTagName('my_subs_tag');
//..
var attr = leaf.getAttribute('custom_attribute');
//..
var branches = branch.item(0).getChildNodes();
//..
for(var i=0; i<branch.length; i++)
    var element = branch.item(i);
    Ti.API.info("Element content: " + element.getText() );
}
