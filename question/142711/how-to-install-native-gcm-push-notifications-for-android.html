How to install native GCM push notifications for android?

Question first: Is it possible to make the android push notifications work without a custom modules and if so can you provide a tutorial or example?

Explanation: Hello all, I am trying to get the GCM push notifications working within my titanium project. I've read dozens of articles/posts/Q&A's on the subject but they all point to a custom module. I am by all means a newbie when it comes to titanium and don't exactly get how I am suppose to implement a custom module much less build one. It seems to me though that using the Ti.cloud and ti.cloudpush services I should be able to get my very straight forward push notifications to work without a custom module.

What I've done: So far I've been able to get the emulator to log in and register but it will not receive any push notifications. I am also getting a weird error in the log cat saying it cannot find the ti.cloud and ti.cloud push modules even though they have been declared. I am sending my push notifications from the online appcelerator console and noticed the emulator device is registered for channel "alert" but the notifications I've sent have a blank channel. Please help!!

  • This is the code i used that is now working in a emulator. I added the two modules on using the GUI of the tiapp.xml. The acs oath keys are entered automatically if you create a new titanium mobile project and check the box for acs services (when asked). Otherwise, you can obtain them from appcelerator - my apps - manage acs - developer/production. The acs app username and password will be the usernames created via either the online appcelerator cloud services or via the app. Ensure you replace [your package name] with the application id. Feel free to comment with your questions if you are trying to do the same.
    Please note that this is only relevant code. If you try and create a project with ONLY this code it will not run.

    tiapp.xml

    <modules>
        <module platform="commonjs" version="2.3.0">ti.cloud</module>
        <module platform="android" version="2.0.7">ti.cloudpush</module>
</modules>
<property name="acs-oauth-secret-production" type="string">[your key]</property>
<property name="acs-oauth-key-production" type="string">[your key]</property>
<property name="acs-api-key-production" type="string">[your key]</property>
<property name="acs-oauth-secret-development" type="string">[your key]</property>
<property name="acs-oauth-key-development" type="string">[your key]</property>
<property name="acs-api-key-development" type="string">[your key]</property>
<android xmlns:android="http://schemas.android.com/apk/res/android">
        <tool-api-level>8</tool-api-level>
        <manifest>
            <uses-sdk android:minSdkVersion="8"/>
            <permission
                android:name="[your package name].permission.C2D_MESSAGE" android:protectionLevel="signature"/>
            <uses-permission android:name="[your package name].permission.C2D_MESSAGE"/>
            <uses-permission android:name="com.google.android.c2dm.permission.RECEIVE"/>
            <uses-permission android:name="android.permission.WAKE_LOCK"/>
            <uses-permission android:name="android.permission.VIBRATE"/>
            <uses-permission android:name="android.permission.INTERNET"/>
            <uses-permission android:name="android.permission.GET_ACCOUNTS"/>
            <uses-permission android:name="android.permission.USE_CREDENTIALS"/>
            <uses-permission android:name="android.permission.ACCESS_NETWORK_STATE"/>
            <application>
                <receiver
                    android:name="com.google.android.gcm.GCMBroadcastReceiver" android:permission="com.google.android.c2dm.permission.SEND">
                    <!-- Start receiver on boot -->
                    <intent-filter>
                        <action android:name="android.intent.action.BOOT_COMPLETED"/>
                        <category android:name="android.intent.category.HOME"/>
                    </intent-filter>
                    <!-- Receive the actual message -->
                    <intent-filter>
                        <action android:name="com.google.android.c2dm.intent.RECEIVE"/>
                        <category android:name="[your package name]"/>
                    </intent-filter>
                    <!-- Receive the registration id -->
                    <intent-filter>
                        <action android:name="com.google.android.c2dm.intent.REGISTRATION"/>
                        <category android:name="[your package name].permission"/>
                    </intent-filter>
                </receiver>
            </application>
        </manifest>
</android>

    app.js

    if(Ti.Platform.osname == "android"){
    var CloudPush = require("ti.cloudpush");
    CloudPush.setEnabled(true);
    CloudPush.addEventListener('callback', androidPush);
    CloudPush.addEventListener('trayClickLaunchedApp', function (evt) {
        Ti.API.info('Tray Click Launched App (app was not running)');
    });
    CloudPush.addEventListener('trayClickFocusedApp', function (evt) {
        Ti.API.info('Tray Click Focused App (app was already running)');
    });
    var Cloud = require("ti.cloud");
    Cloud.debug = true;
}
function androidPush(evt){
    var arr_from_json = JSON.parse(evt.payload); 
        var dialog = Ti.UI.createAlertDialog({
        message : arr_from_json['android'].alert,
        ok : 'OK',
        title : arr_from_json['android'].title
    }).show();
}

    settings.js (for registration)

    var Cloud, CloudPush;
var button = Ti.UI.createButton({
    height:60,
    width:200,
    title:'Login',
    top:20
});
self.add(button);
Cloud = require("ti.cloud");
CloudPush = require("ti.cloudpush");
button.addEventListener('click', function() {
    CloudPush.retrieveDeviceToken({
        success: function deviceTokenSuccess(e) {
            alert('Device Token: ' + e.deviceToken);
            var deviceToken = e.deviceToken
            loginDefault(deviceToken);
        },
            error: function deviceTokenError(e) {
                alert('Failed to register for push! ' + e.error);         
        }
    });            
});
function loginDefault(deviceToken){
    Cloud.Users.login({
        login: [acs app username],
        password: [acs app password]
    }, function (e) {
        if (e.success) {
            alert("login success");
            defaultSubscribe(deviceToken);
        } else {
            alert('Error: ' +((e.error && e.message) || JSON.stringify(e)));
        }
    });
}

function defaultSubscribe(deviceToken){
    Cloud.PushNotifications.subscribe({
        channel: 'alert',
        device_token: deviceToken,
        type: 'android'
    }, function (e){
        if (e.success) {
           alert('Subscribed for Push Notification!');
        }else{
            alert('Error:' +((e.error && e.message) || JSON.stringify(e)));
        }
    });
}
