Alert Box - Issue

Ti.App.addEventListener('openLink', function(e) {
    // alert("Fired");
    if(e.linkUrl.toString().indexOf("tel:", 0) != -1) {
        var link = e.linkUrl.toString().trim();
        var link_length = link.length;
        var phone_num = link.substring(4,link_length-1);
        var alert_dialog = Ti.UI.createAlertDialog({
            cancel : 1,
            buttonNames : ['Cancel', 'Call'],
            message : 'Call '+phone_num
        }).show();

        alert_dialog.addEventListener('click',function(e){
            if (!e.cancel) {
                alert("1");
                Ti.Platform.openURL(e.linkUrl);
            }
        });
    } else {
        Ti.Platform.openURL(e.linkUrl);
    }
});
