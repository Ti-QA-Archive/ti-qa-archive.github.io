  • This code works on device (Ti.sdk 3.2.2, iOS 7.1):

    var micPeak = 0,
    micPower = 0;

function monitorInput(){
    // Ti.Media.startMicrophoneMonitor();
    micPeak = Ti.Media.peakMicrophonePower;
    micPower = Ti.Media.averageMicrophonePower;
    Ti.API.info('peak: '+micPeak+' avg: '+micPower);
    if (micPeak > 0.01 || micPower > 0.01){
        txtArea.value += '\nMicrophone  |  peak: '+micPeak.toFixed(3)+' avg: '+micPower.toFixed(3);
    }
}

Ti.Media.audioSessionMode = Ti.Media.AUDIO_SESSION_MODE_PLAY_AND_RECORD;
Ti.Media.requestAuthorization(function(e){
    alert(e);
    Ti.Media.startMicrophoneMonitor();
    // var recorder = Ti.Media.createAudioRecorder();
    // recorder.start();
    setInterval(monitorInput, 250);
});
