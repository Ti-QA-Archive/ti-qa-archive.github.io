    • I made aflew assumption on the json array but youll get the idea

      var american = Titanium.UI.createTableViewSection();
    american.headerTitle = 'American';

    var japanese = Titanium.UI.createTableViewSection();
    japanese.headerTitle = 'Japanese';

    var dataProducts = [{
        category: ['performance', 'performance', 'n/a', 'performance', 'performance', 'performance'],
        cars: [
            {title: 'Hub Cap', car:'honda', options:'Silver', qty:'1', price:'50' },
            {title: 'Hub Cap', car:'ford', options:'Silver', qty:'1', price:'50' },
            {title: 'Hub Cap', car:'ford', options:'Silver', qty:'1', price:'50' },
            {title: 'Hub Cap', car:'chevy', options:'Silver', qty:'1', price:'50' },
            {title: 'Hub Cap', car:'nissan', options:'Silver', qty:'1', price:'50' },
            {title: 'Hub Cap', car:'nissan', options:'Silver', qty:'1', price:'50' }
        ]
    }];
    for(var i = 0, len = dataProducts.length; i < len; i++) {
        var type = dataProducts[i];

            for(var j = 0, lenj = type.category.length; j < lenj; j++) {

                if ( type.category[j] === 'performance' ) {

                    var row = Titanium.UI.createTableViewRow({title: type.cars[j].car});

                    switch (type.cars[j].car){
                        case 'chevy':
                        case 'ford':
                            american.add(row);
                        break;
                        case 'honda':
                        case 'nissan':
                            japanese.add(row);
                        break;
                    }
                }
            }
    }
var tableview = Titanium.UI.createTableView({
        data: [american, japanese]
    });
