  • Hi PieterJan,

    You can create your custom view and add your buttons and on button click event you can do as per your requirement.Try below code as sample,

    var animation = Titanium.UI.createAnimation({
    duration : 700
});

var view = Ti.UI.createView({
    height : 230,
    bottom : -230,
    zIndex : 9999,
    layout : "vertical",
    backgroundColor : 'gray'
});

var btn1 = Ti.UI.createButton({
    title : 'Delete',
    backgroundColor : 'red',
    height : 40,
    width : 320,
    top : 10
});
btn1.addEventListener('click', function() {
    animation.bottom = -230;
    view.animate(animation);
});
var btn2 = Ti.UI.createButton({
    title : 'Save',
    backgroundColor : 'white',
    height : 40,
    width : 320,
    top : 10
});
btn2.addEventListener('click', function() {
    animation.bottom = -230;
    view.animate(animation);
});
var btn3 = Ti.UI.createButton({
    title : 'Cancel',
    height : 40,
    backgroundColor : 'gray',
    width : 320,
    top : 10
});
btn3.addEventListener('click', function() {
    animation.bottom = -230;
    view.animate(animation);
});
view.add(btn1);
view.add(btn2);
view.add(btn3);

var btn4 = Ti.UI.createButton({
    title : 'open',
    height : 30,
    width : 100,
    top : 5
});
btn4.addEventListener('click', function() {
    animation.bottom = 0;
    view.animate(animation);
});
win.add(btn4);
wn.add(view);
