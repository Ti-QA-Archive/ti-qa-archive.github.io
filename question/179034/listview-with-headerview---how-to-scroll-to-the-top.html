Listview with headerview - how to scroll to the top

I have a listview which a attach a headerview to. Using listView.scrollToItem(0, 0) the list scrolls to the first item leaving the headerview invisible just above.

Is there no way to scroll to the top of the headerview?

(Building an iPhone and Android app using Studio 3.3.1).

  • Hello Anders,

    Normally listView header does not show when we scroll to any position.
    So you can remove a headerView property of the listView and add a custom view top of the listView .
    Here is the sample code.
    Try to use update environment

    My Testing Environment:
    Titanium SDK: 3.4.0,
    Titanium CLI: 3.4.0,
    iOS Version: 8.1,
    Android SDK : 4.2.2,
    OS X Version: 10.9.5,
    Appcelerator Studio: 3.4.1

    Sample Code

    var win = Ti.UI.createWindow({
    backgroundColor: 'white',
    layout:'vertical'

    });

var view = Ti.UI.createView({
    width:Ti.UI.FILL,
    height:50,
});

// Create a Label.
var header = Ti.UI.createLabel({
    text : 'List View Header',
    color : 'red',
    font : {fontSize:20},
});
view.add(header);
win.add(view);

var listView = Ti.UI.createListView({
    height:410,
    top:0
});

var sections = [];

var fruitSection = Ti.UI.createListSection({ headerTitle: 'Fruits'});
var fruitDataSet = [];


for (var i=0; i <100; i++) {
    fruitDataSet.push( {properties: { title: 'Carrots'}},
    {properties: { title: 'Potatoes'}}) ;

};
fruitSection.setItems(fruitDataSet);
sections.push(fruitSection);


var vegSection = Ti.UI.createListSection({ headerTitle: 'Vegetables'});
var vegDataSet = [
    {properties: { title: 'Carrots'}},
    {properties: { title: 'Potatoes'}},
];
vegSection.setItems(vegDataSet);
sections.push(vegSection);

listView.sections = sections;
win.add(listView);


// Create a Button.
var scrollTo = Ti.UI.createButton({
    title : 'scrollTo',
    height : 60,
    top:20,
    width : 100,

});

// Listen for click events.
scrollTo.addEventListener('click', function() {
    listView.scrollToItem(0,0);
});

// Add to the parent view.
win.add(scrollTo);

win.open();

    If you have any question about this query, then let me know.
    For more info scrollToItem Doc.

    Thanks.
