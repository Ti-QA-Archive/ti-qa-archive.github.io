  • This is the code, I usually use for this purpose:

    var shareButton = Ti.UI.createButton();
shareButton.addEventListener('click', function() {
    if (Ti.Facebook.loggedIn) {
        var f = Ti.UI.createImageView({image:'appicon.png', width:100, height:100}).toBlob();
            var blob = f;
            var data = {
            message: L('facebook_sharing_text'),
            link:L("appstore_url"),
            picture: blob
        };
        Ti.Facebook.requestWithGraphPath('me/feed', data, 'POST', showRequestResult);
        fbWin.close();
    } else {
            alert(L("facebook_note_header"), L('facebook_note_message'));
    }
});
viewFacebook.add(shareButton);

    I am using the Ti.Facebook.createLoginButton() for Login.
    Be sure to set also: 

    Ti.Facebook.appid = <your facebook app id>;
Ti.Facebook.permissions = ['publish_stream'];
