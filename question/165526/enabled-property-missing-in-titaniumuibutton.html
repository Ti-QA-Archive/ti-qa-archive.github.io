  • Code working fully fine for me

    var win = Ti.UI.createWindow({
    title : 'Background Services Example',
    backgroundColor : 'white',
});
var btn = Ti.UI.createButton({
    title : 'I am button',
    width : Ti.UI.SIZE,
    height : Ti.UI.SIZE,
    enabled : false
});
win.add(btn);
win.open();
