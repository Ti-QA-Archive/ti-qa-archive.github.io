  • Hi pintu Patel, thanks for your question.

    To find Current Location in android Device, you can use the following code.

    var mapPage = Titanium.UI.currentWindow;

var map=require('ti.map');

var annotations = [Ti.Map.createAnnotation({
    latitude: 23.7272512,
    longitude: 90.39666650000004,
    title: 'Central Shoheed Minar',
    subtitle: 'Dhaka,Bangladesh',
    animate : true,
    pincolor : Ti.Map.ANNOTATION_RED,
})];

var mapview = Titanium.Map.createView({
    mapType : Titanium.Map.STANDARD_TYPE,
    animate : true,
    regionFit : true,
    userLocation : true,
    region : {
        latitude: 23.7272512,
        longitude: 90.39666650000004,
        latitudeDelta : 0.1,
        longitudeDelta : 0.1
    },
    annotations : annotations
});

var longitude;
var latitude;
var points=[]; 
points.push({latitude: 23.7272512,longitude: 90.39666650000004});

Titanium.Geolocation.getCurrentPosition(function(e) 
{
    Ti.Geolocation.accuracy = Ti.Geolocation.ACCURACY_BEST;
    if (e.error)
    {
        alert('Current location not found.');
        return;
    }
    else 
    {
        latitude = e.coords.latitude;
        longitude = e.coords.longitude;
        annotations.push(Ti.Map.createAnnotation({
            latitude: e.coords.latitude,
            longitude: e.coords.longitude,
            title: 'My Current Location',
            animate : true,
            pincolor : Ti.Map.ANNOTATION_RED,
            }));
    points.push({latitude: e.coords.latitude,longitude: e.coords.longitude});
       var route = {
        name : "Bangladesh",
        points : points,
        color : "green",
        width : 3
    };

    mapview.addRoute(route);
    }
});

mapPage.add(mapview);

    I've tested it and works fine.
