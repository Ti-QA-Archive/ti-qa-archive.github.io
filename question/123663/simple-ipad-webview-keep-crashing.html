simple iPad webview keep crashing

Hello,

Below is a code for a simple webview on iPad, which I took outside of a bigger app.

Each of the 1-6 buttons loads a different website.

If you click on any button, wait a few seconds, than click on button 2, the app will crash about 1-2 minutes after website 2 finished loading. This happens with several other website that seem to crash the iPad.

By looking at the console it seems that this is a memory issue - I'm getting Level 1 & 2 memory warnings before it crashes.

This of course happens only on the iPad device and not on the emulator.

Any idea how to resolve this?

Thanks
Idan

var win1 = Ti.UI.createWindow({  
    backgroundColor:'#fff',
    barColor: '0096c7'
});


var webView = Ti.UI.createWebView({
  top:0,
  height:964, 
  //!//scalesPageToFit:true,
  backgroundColor:'#fff',
  //url:postLink
});
win1.add(webView);

var bb1 = Titanium.UI.createButtonBar({
    labels:['1', '2', '3', '4', '5', '6'],
    backgroundColor:'#336699',
    bottom:0,
    style:Titanium.UI.iPhone.SystemButtonStyle.BAR,
    height:100,
});
win1.add(bb1);

bb1.addEventListener('click', function(e) {
    //Ti.API.info('title '+bb1.labels[e.index]);
    switch (bb1.labels[e.index])
    {
        case '1':
            Ti.API.info('1');
            webView.url = "http://www.ynet.co.il/articles/0,7340,L-4101653,00.html";
            break;
        case '2':
            Ti.API.info('2');
            webView.url = "http://www.ynet.co.il/articles/0,7340,L-4102071,00.html";
            break;
        case '3':
            Ti.API.info('3');
            webView.url = "http://www.calcalist.co.il/markets/articles/0,7340,L-3526213,00.html";
            break;
        case '4':
            Ti.API.info('4');
            webView.url = "http://www.calcalist.co.il/local/articles/0,7340,L-3526220,00.html";
            break;
        case '5':
            Ti.API.info('5');
            webView.url = "http://news.nana10.co.il/Article/?ArticleID=818640";
            break;
        case '6':
            Ti.API.info('6');
            webView.url = "http://news.walla.co.il/?w=/1/1845660";
            break;            
    }
webView.reload();    
});

win1.open();
