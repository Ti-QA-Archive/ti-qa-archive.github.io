  • This is the code for the event listeners which dosent work on android bt works on ios

    
Ti.API.addEventListener('proposalListTap', function(e){

    var singleProposalWin = Alloy.createController('singleBusinessProposal',e.rowData);

    $.tabgroup.getActiveTab().open(singleProposalWin.getView());

});


Ti.API.addEventListener('singleBusinessProposalDetail', function(e){
    var detailsWin = Alloy.createController(e.winName,{header : e.header, businessID : e.businessID, ismine : e.ismine});

    $.tabgroup.getActiveTab().open(detailsWin.getView());
});
