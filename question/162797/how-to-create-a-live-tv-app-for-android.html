  • Hi, Jude Addo
    First of all you need a content URL for live TV

    Here is an Example:

    var contentURL = "http://~~~~~~~~~~~~~~Your TV URL~~~~~~~~~~~~~~~~~~~";

var win = Titanium.UI.createWindow({      
        title:'Test',
        backgroundColor:'#fff',
        layout: 'vertical'
});

var movie = Titanium.Media.createVideoPlayer
({
    url:contentURL,
    top : 20,
    autoplay : true,
    height : 300,
    width : 300,
    mediaControlStyle : Titanium.Media.VIDEO_CONTROL_DEFAULT,
    scalingMode : Titanium.Media.VIDEO_SCALING_ASPECT_FIT
});
movie.play();
win.add(movie);
win.open();

    Thanks
