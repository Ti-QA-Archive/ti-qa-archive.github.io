    Try This custom code for switch effect in Android. set your own images for ON and OFF buttons.
    var lblMessage = null;
    switchButtons() ; // calling of switch control

    function switchButtons() {
    var arrview = [];
    lblMessage = Ti.UI.createLabel({
    text : 'Default is ON',
    width : 200,
    height : 30,
    top : 80,
    color : 'black'

        });

    win1.add(lblMessage);

    var onBtn = Ti.UI.createButton({
        width : 65,
        height : 24,
        id : 1,
        backgroundImage : '/images/on.png'
    });
    var offBtn = Ti.UI.createButton({
        width : 65,
        height : 24,
        id : 2,
        backgroundImage : '/images/off.png'
    });
    arrview.push(onBtn);
    arrview.push(offBtn);
    var switchView = Ti.UI.createScrollableView({
        top : 150,
        width : 64,
        height : 24,
        views : arrview,
        backgroundColor : 'white',
        showPagingControl:false

    });

    switchView.addEventListener('scroll', function(e) {
        var indexpage = e.currentPage;
        if (indexpage == 0) {
            lblMessage.text = "Default is ON";
            // Your Custom Logic

        } else {
            lblMessage.text = "Default is OFF";
                // Your Custom Logic
        }

    });

    win1.add(switchView);
}

    • brilliant. Also apart from scroll you need to add click on the view. if the singletap event on scrollable view does not work then u can add clicks to the individual buttons to get your work done. eg:

      onBtn.addEventListener('click',function(){
              switchView.scrollToView(1);
          });

offBtn.addEventListener('click',function(){
                  switchView.scrollToView(0);
              });
