    Hi,

    first thing is you can not access object $.navbar in winflow window because its decalred in index.xml so you need to pass navbar object to winflow window.

    function openNextWindow(e) {
    var winflow = Alloy.createController("winflow",$.navbar).getView(); 
    $.navbar.open(winflow, {animated : true}); 
}

    then get that object in winflow.js by the below code

    winflow.js

    var navbar = arguments[0];

    then use the below code to open about window.

    function about(e) {
    var about = Alloy.createController("about").getView(); 
    navbar.open(about, {animated : true}); 
}
