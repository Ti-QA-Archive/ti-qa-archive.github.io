var player = Titanium.Media.systemMusicPlayer;

function playAudioFile(file_name, callback) {
    var player = Ti.Media.createSound({
        allowBackground : true
    });
    player.addEventListener('complete', function(e) {
        callback();
    });
    player.url = file_name;
    player.play();
}
