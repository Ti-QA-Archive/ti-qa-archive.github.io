  • Long Press is not yet implemented through Titanium (we are working on it), however in the mean time you could do something like:

    var touched = false;
tableview.addEventListener('click', function(e)
{
    touched = false;
});
tableview.addEventListener('touchstart', function(e){
    touched = true;
    setTimeout(function(){
        if (touched)
            alert("Long press");
    },1000);
});
tableview.addEventListener('touchmove', function(e){
    touched = false;
});
tableview.addEventListener('touchend', function(e){
    touched = false;
});

    To emulate the call.
