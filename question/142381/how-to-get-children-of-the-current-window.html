    backgroundColor:'#fff'
});

    backgroundColor:'red',
    width:100,
    height:100
});
win.add(view);

var c = win.children;
Ti.API.info(c);

win.open();

    • Ah, I see. When you use the url property of the Window object, you actually create a new js context in which everything is running in. If you put your code in a commonjs module, you'd still be able to access the children of the window.
