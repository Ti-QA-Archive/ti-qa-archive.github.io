    Hello Prema,

    i re posting full code

    just paste this into your app.js file and click on any box so you will get it's row and column and from that you can change any property of this box.

    
var win = Ti.UI.createWindow({

});

// Create a TableView.
var aTableView = Ti.UI.createTableView();

// Populate the TableView data.
var data = [];

// Listen for click events.
aTableView.addEventListener('click', function(e) {

});

for (var i=0; i < 10; i++) {
  var row = Ti.UI.createTableViewRow({
      height : 50
  });

  var label1 = Ti.UI.createLabel({
      top : 5,
      left : 5,
      height : 40,
      width : 40,
      backgroundColor : 'red',
      column : '1',
      rowid : i
  });

  var label2 = Ti.UI.createLabel({
      top : 5,
      left : 55,
      height : 40,
      width : 40,
      backgroundColor : 'green',
      column : '2',
      rowid : i
  });

  var label3 = Ti.UI.createLabel({
      top : 5,
      left : 105,
      height : 40,
      width : 40,
      backgroundColor : 'blue',
      column : '3',
      rowid : i
  });

  var label4 = Ti.UI.createLabel({
      top : 5,
      left : 155,
      height : 40,
      width : 40,
      backgroundColor : 'purple',
      column : '4',
      rowid : i
  });

  var label5 = Ti.UI.createLabel({
      top : 5,
      left : 205,
      height : 40,
      width : 40,
      backgroundColor : 'red',
      column : '5',
      rowid : i
  });

  var label6 = Ti.UI.createLabel({
      top : 5,
      left : 255,
      height : 40,
      width : 40,
      backgroundColor : 'green',
      column : '6',
      rowid : i
  });

  row.add(label1);
  row.add(label2);
  row.add(label3);
  row.add(label4);
  row.add(label5);
  row.add(label6);

  data.push(row);
};

aTableView.addEventListener('click',function(e){
    alert('row : '+e.source.rowid +'\n column : '+e.source.column);
});

aTableView.data = data;
// Add to the parent view.
win.add(aTableView);


win.open();
