Create a graphic MapRoute with Titanium.Geolocation.getCurrentPosition.

Hi,
I wonder how I can make a Maproute that updates when Im moving position (a visual route for my travel etc)?

Code:
var w_Map = Ti.UI.currentWindow;

Ti.Geolocation.purpose = "Get the users location";

Titanium.Geolocation.distanceFilter = 10;

Titanium.Geolocation.getCurrentPosition(function(e)
{
if (e.error)
{
// manage the error
return;
}

    var longitude = e.coords.longitude;
    var latitude = e.coords.latitude;
    var altitude = e.coords.altitude;
    var heading = e.coords.heading;
    var accuracy = e.coords.accuracy;
    var speed = e.coords.speed;
    var timestamp = e.coords.timestamp;
    var altitudeAccuracy = e.coords.altitudeAccuracy;

    var v_Map = Titanium.Map.createView({
        mapType: Titanium.Map.TERRAIN_TYPE,
        region: longitude.latitude,
        latitudeDelta:0.1, 
        longitudeDelta:0.1,
        animate:true,
        regionFit:true,
        userLocation:true,
        annotations: longitude.latitude
    })  
    w_Map.add(v_Map);          
});

Titanium.Geolocation.addEventListener('location',function(e)
{
    if (e.error)
    {
         return;
    }

    var longitude = e.coords.longitude;
    var latitude = e.coords.latitude;
    var altitude = e.coords.altitude;
    var heading = e.coords.heading;
    var accuracy = e.coords.accuracy;
    var speed = e.coords.speed;
    var timestamp = e.coords.timestamp;
    var altitudeAccuracy = e.coords.altitudeAccuracy;

});

Thanks in advance!
