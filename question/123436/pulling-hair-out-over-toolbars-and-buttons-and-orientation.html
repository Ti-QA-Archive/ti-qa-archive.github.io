    The toolbar looks the same, other than the widths?
    Why not just have the toolbar added then use the code to change the toolbars item widths?

                                         var button1 = Ti.UI.createButton({systemButton: Ti.UI.iPhone.SystemButton.FIXED_SPACE });

                var button2 = Ti.UI.createButton({systemButton: Ti.UI.iPhone.SystemButton.FIXED_SPACE });

                var button3 = Ti.UI.createButton({systemButton: Ti.UI.iPhone.SystemButton.FIXED_SPACE });


var toolbar1 = Titanium.UI.createToolbar({
                items:[
                mapButton, 
                button1,
                helpButton,
                button2,
                mainTitleLabel,
                button3,
                navigationButton,
                flexSpace, envelopeButton, flexSpace, settingsButton, flexSpace, refreshButton, flexSpace, infoButton ],
                top:0,
                borderTop:true,
                borderBottom:true,
                barColor:'#222'
            });
win.add(toolbar1);

if (orientation == 'portrait')
            {  
               button1.width = 15;
               button2.width = 324;
               button3.width = 180;
            } else {
               button1.width = 15;
               button2.width = 185;
               button3.width = 120;
            }

    Not tested but should work?
