Locking headers on table view

Is it possible to lock the headers in the table view? Just like when you are in contacts on iPhone and scrolling through them and the current letter header stays locked up top. I have the data building out ok just not sure of the best way to deal with locking the headers.

function setData(){
    var len = data.length;
    var myData = [];
    var oldLetter = null;
    var curLetter;

    for(var i=0; i<len; i++)
    {
        var letter = data[i].title.substr(0,1);
        if(letter != oldLetter)
        {
            oldLetter = letter;
            curLetter = letter;
        } 
        else curLetter = null; 
        var row = TI.UI.createTableViewRow({
            title: data[i].title,
            height:40,
            header: curLetter
        });
        myData[i] = row;
    }
    return myData;
}

var table = TI.UI.createTableView({
    width:'100%',
});    

table.setData(setData())
