TabGroup with NavigationWindow has 2 headers

Hi All, I am trying to add a Navigation window to a tabgroup but I am getting 2 tab headers, one with tab title and another with window title. I have tried setting the tabBarHidden to true on the window but it doesnot work because as per documentation
tabBarHidden : "This property is only valid when the window is the child of a tab."
and the window is not directly added to tab but a navigationwindow.
I want to have a single header for the window, on which I can set navButtons (left & right). Any help is appreciated.
SDK Used : 3.2.2 GA

Ti.UI.setBackgroundColor('#fff');
/**
 * Master view with tabGroup.
 */
var tabGroup = Ti.UI.createTabGroup();

var window1 = Ti.UI.createWindow({title : "Window 1"});
var window2 = Ti.UI.createWindow({title : "Window 2"});

var winNav1 = Ti.UI.iOS.createNavigationWindow({window : window1});
var winNav2 = Ti.UI.iOS.createNavigationWindow({window : window2});

var tab1 = Ti.UI.createTab({
    title: 'Tab 1',
    window: winNav1
});

var tab2 = Ti.UI.createTab({
    title: 'Tab 2',
    window: winNav2
});

tabGroup.addTab(tab1);
tabGroup.addTab(tab2);


/**
 * Details view with navigationGroup.
 */
var winDetail = Ti.UI.createWindow();
var navDetail = Ti.UI.iOS.createNavigationWindow({
    window: winDetail
});

var splitWindow = Ti.UI.iPad.createSplitWindow({
    masterView: tabGroup,
    detailView: navDetail
});

/**
 * Popover menu.
 */
splitWindow.addEventListener('visible', function(e) {
    if (e.view == 'detail') {
        e.button.title = 'Menu';
        winDetail.setLeftNavButton(e.button);
    }
    else if (e.view == 'master') {
        winDetail.setLeftNavButton(null);
    }
});

splitWindow.open();
