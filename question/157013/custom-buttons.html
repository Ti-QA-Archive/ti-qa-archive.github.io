  • var fb = require('facebook');

    fb.appid =xxxxxxxxx;

    fb.permissions = ['publish_stream', 'read_stream', 'email'];

    fb.forceDialogAuth = true;

    var fblogin = Ti.UI.createButton({

        top : 200,
    left : 52,
    width : 373,
    height : 68,
    backgroundImage : "/facebookloginButton.png"
});
fblogin.addEventListener('click', function(e) {
    if (Titanium.Network.online) {
        fb.authorize();
    } else {
        alert('turn on internet');
    }
});
win.add(fblogin);
