  • I don't like sections, so I create my own headers. As easy as adding a new row at the beggining of your data array, customized as you like, for example:

    //layout view for the header
var v =  Ti.UI.createView({
    height:50, left:10, right:10, top:10,
    layout:'horizontal'
});

//Create header row
row = Ti.UI.createTableViewRow({
    height:50, backgroundColor:'#ddd',
    selectionStyle:Titanium.UI.iPhone.TableViewCellSelectionStyle.NONE,
    className: 'rowHeader'
});

v.add(Ti.UI.createLabel({text:'Title', width:260, height:'auto', color:'#fff'}));
v.add(Ti.UI.createLabel({text:'Credits', width:100, height:'auto', color:'#fff', textAlign:'center'}));
v.add(Ti.UI.createLabel({text:'%', width:100, height:'auto', color:'#fff', textAlign:'center'}));
v.add(Ti.UI.createLabel({text:'Comments', left:20, width:60, height:'auto', color:'#fff', textAlign:'center'}));

row.add(v);

tableData.push(row);

//Now feed your table content
//...

    (I use style files, so the code usually is seen cleaner)
