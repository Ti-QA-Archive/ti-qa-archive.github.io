    One way to accomplish this is to add whatever data you want to retrieve as a custom property to the view or control that will be receiving the click event. Here is a trimmed down example of your code. If you only want to detect the click event on the image, then you would add the custom property to the imageview. See the code and comments for more info.

    var win = Ti.UI.createWindow();

var images = [];

for(var i=0;i<3;i++){

    var test_url = 'url ' + i;

    container_view = Ti.UI.createView({
        top:0,
        bottom:50,
        url:test_url  //add as custom property then look for it in the eventListener -- e.source.url
    });

    image_label = Ti.UI.createLabel({
        bottom:100,
        left:10,
        right:10,
        height:18,
        textAlign:'center',
        color:'#FFF',
        backgroundColor:'ff0000',  //turning on background color just to see where the label is actually located
        font:{fontFamily:'Helvetica Neue', fontWeight:'bold',fontSize:12},
        text: test_url,
        touchEnabled:false //turning off touch events here so they dont get detected
    });

    container_view.add(image_label);    
    images.push(container_view);

}

var imageGallery = Titanium.UI.createScrollableView({
    views:images,
    top:0,
    height:450,
    backgroundColor:'transparent',
    showPagingControl:false,
    pagingControlColor:'transparent',
    pagingControlHeight:30,
    maxZoomScale:4.0,
    currentPage:0
});

imageGallery.addEventListener("singletap",function(e){
    alert(e.source.url);
});

win.add(imageGallery);

win.open();
