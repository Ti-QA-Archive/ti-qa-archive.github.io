var checkrate = Titanium.UI.createButton();
checkrate.addEventListener('click', function(){
 var option_rates = Ti.UI.createAlertDialog({
        title: "Recharge message",
        message: "This would take some minutes for downloading data from server? Do you want to continue?",
        buttonNames: ["Yes", 'No']
    });
option_rates.show();
});

  • This works fine for me:

    var checkrate = Titanium.UI.createButton({
    title:'test',
    width:100,
    height:50,
    top:30
});
win.add(checkrate);

checkrate.addEventListener('click', function(){
    var option_rates = Ti.UI.createAlertDialog({
        title: "Recharge message",
        message: "This would take some minutes for downloading data from server? Do you want to continue?",
        buttonNames: ["Yes", 'No']
    });
    option_rates.show();
});

    Are you sure you are not adding an event listener elsewhere in your application?

    Depending on the structure of your code in the rest of your application, it is possible that the event listener is being set up more than once, or it is not being removed when it should.
