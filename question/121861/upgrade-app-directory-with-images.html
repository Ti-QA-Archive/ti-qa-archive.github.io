  • i think you can't compile without resources data. put your images in memory card and set your application to sync any folder in memory card where you will keep your images. so when you update your app your images will be safe in memory card and then you can update some off them or all of them, depend on what you need.

    Note: if the images are not for app design, if you use them like a gallery put them in memory card(external storage).

    this is my idea :P.

    var file_obj = {file:filename, url:url, path: null};

    var file = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory,filename);
    if ( file.exists() ) {
        file_obj.path = Titanium.Filesystem.applicationDataDirectory+Titanium.Filesystem.separator;
        fn_end(file_obj);
    }
    else {

        if ( Titanium.Network.online ) {
            var c = Titanium.Network.createHTTPClient();

            c.setTimeout(10000);
            c.onload = function()
            {

                if (c.status == 200 ) {


                    var f = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory,filename);
                    f.write(this.responseData);
                    file_obj.path = Titanium.Filesystem.applicationDataDirectory+Titanium.Filesystem.separator;
                }

                else {
                    file_obj.error = 'file not found'; // to set some errors codes
                }
                fn_end(file_obj);

            c.ondatastream = function(e)
            {

                if (fn_progress) {
                    fn_progress(e.progress);
                }
            };
            c.error = function(e)
            {

                file_obj.error = e.error;
                fn_end(file_obj);
            };
            c.open('GET',url);
            c.send();           
        }
        else {
            file_obj.error = 'no internet';
            fn_end(file_obj);
