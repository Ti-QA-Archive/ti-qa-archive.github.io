How to get a form with numerous fields to be scrollable?

Hi - I have a form with multiple fields and cannot get the screen to scroll in order to see all the fields. I have tried using scrollView - but my form elements are fixed as opposed to the scrollable background.

/*--------------------- Creates  Wndow ---------------------*/
gg.ui.createWindow = function(){
        var win = Titanium.UI.createWindow({
            title:'theTitle',
            backgroundColor:'green',
            tabBarHidden:true,
            barColor:'#88bb50',
            titleimage:'iphone/appicon.png',
            });
/* ---------------------Creates Scrollable area for window---------------------*/
        var scrollView = Titanium.UI.createScrollView({
                contentWidth:'auto',
                contentHeight:'auto',
                top:0,
                showVerticalScrollIndicator:true,
                showHorizontalScrollIndicator:true
            });
/* ---------------------Creates view for scrollView for window---------------------*/
        var view = Ti.UI.createView({
            backgroundColor:'#000000',
            height:2000,
            width:300,
            });
/* --------------------------------Create Name Label -----------------------------------*/
        var nameLabel = Ti.UI.createLabel({
            text:'Name:',
            top:10,
            width:'250',
            height:'auto',
            textAlign:'left',
            color:'white'
            });

/* ---------------------Creates Name Text Field ---------------------*/
            var name = Ti.UI.createTextField({
            height:40,
            top:'auto',
            width:250,
            keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
            returnKeyType:Titanium.UI.RETURNKEY_DONE,
            borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
            hintText:L('Name')
        });

/* --------------------------------Create Label -----------------------------------*/
        var emailLabel = Ti.UI.createLabel({
            text:'Email Address:',
            top:'auto',
            width:'250',
            height:'auto',
            textAlign:'left',
            color:'white'
            });
/* --------------------------------Creates Email Text Field --------------------------------*/
            var email = Ti.UI.createTextField({
            height:40,
            top:'auto',
            width:250,
            keyboardType:Titanium.UI.KEYBOARD_DEFAULT,
            returnKeyType:Titanium.UI.RETURNKEY_DONE,
            borderStyle:Titanium.UI.INPUT_BORDERSTYLE_ROUNDED,
            hintText:L('Email_Address')
        });
/*---- There are about 5 more feilds (shortening for the purpose of clarity)  */

        scrollView.add(view);
        win.add(scrollView);

        win.add(nameLabel);
        win.add(name);
        win.add(emailLabel);
        win.add(email);
        return win;

Any ideas why the background is scrollable while the form fields stay fixed (on top of each other). I also tried changing all the 'top' properties to space them out, and that worked for spacing, but then it wasn't scrollable.

Using Titanium Studio, build: 1.0.2.201107130739 and the iPhone simulator

    you need to add the elements to the scrollview not the window, or add the elements to the view that you have added to the scrollview.
