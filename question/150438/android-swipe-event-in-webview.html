Android: Swipe event in WebView

Hi,

We are trying to handle the swipe event in WebView, but the event is not getting triggered.

var currentWindow = Ti.UI.createWindow({
    exitOnClose : true
});

var webview = Ti.UI.createWebView({
    url : 'http://google.co.in'
});

webview.addEventListener('swipe',function(e){
    Ti.API.info('i was swiped');
    alert('WebView swipe');
});



currentWindow.add(webview);
currentWindow.open();

The alert is not getting displayed, not the log. I have tried it with 3.0.2GA SDK on Android. Is the swipe event not triggered for WebView or do I need to do some more code changes.

  • HI Soumya,

    var currentWindow = Ti.UI.createWindow({
    exitOnClose : true
});

var webview = Ti.UI.createWebView({
    url : 'http://google.co.in'
});

webview.addEventListener('swipe',function(e){
if(e.direction=='up'){

    Ti.API.info('i was swiped up');
    alert('WebView swipe up');

}else if(e.direction == 'down'){

    Ti.API.info('i was swiped down');
         alert('WebView swipe down');
    }
});

currentWindow.add(webview);
currentWindow.open();
