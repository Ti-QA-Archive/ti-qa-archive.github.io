Blank tittle is not coming in the nav bar

I am using titanium sdk as 1.7.1, and Iphone sdk is 4.2.

There is a table view when a row is clicked in the table view ,a new window is created and the title of the newly created window should be the title given in the categories title. But my following code creates a window with a blank title..

My CODE is 

var win = Titanium.UI.currentWindow;
    win.barColor='orange';
    //Creating an Array named categories
    var categories = [
        {title: "All",hasChild:true,category_name:"All",test:'Show_result.js'},

        {title:'Honda',hasChild:true,category_name:"honda",test:'Show_result.js'},
        {title:'bmw',hasChild:true,category_name:"bmw",test:'Show_result.js'},
        {title:'skoda',hasChild:true,category_name:"skoda",test:'Show_result.js'},
        {title:'ferari',hasChild:true,category_name:"ferari",test:'Show_result.js'},
        {title:'merc',hasChild:true,category_name:"merc",test:'Show_result.js'}
    ];//End of Array


    //Creating an Array named rows
    var rows = [];

    //Traversing through the Array categories
    for(var v in categories)
    {
        //Creating a row Table View Row
          var row = Titanium.UI.createTableViewRow({
          hasChild:true,
          category_name:categories[v].category_name
          });    //End of 'createTableViewRow'

        //Creating Labels for Table View Row'
          var label = Ti.UI.createLabel({
           text:categories[v].title,
           font:{fontSize:20},
          });//End createlabel

          //Make the first Label All as Bold
          if(Number(v) === 0)
          {
            label.font = {fontWeight:'bold',fontSize:20};
        }//End of if

        //Add the labels Table View Row
          row.add(label);

          //Add the labels to named rows
          rows.push(row); 
    }// End of for loop

     //Create the Table View and put the Array rows as its data
    var table = Titanium.UI.createTableView({
    //data:rows   
    data:rows
    });//End of create Table

    //Add the table to the window
    win.add(table);


table.addEventListener('click', function(e)
{
        var i = e.index


        var win = Titanium.UI.createWindow({  
        barColor:'orange',
           backgroundColor:'white',
        fullscreen:false,modal:true,
        url:'Show_result.js',
        title:e.rowData.title,

        });


        win.open();    
    //}
});
