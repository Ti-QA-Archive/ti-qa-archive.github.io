nav group with tab group and transition

I have a two-part app. The first part is a nav group and it jumps around from window to window. The second part is a tab group. I want to do a flip transition when I switch from the navgroup to the tabgroup, but I am having trouble. I get poor transitions when I use tabgroup.open({transition}); but if I do navGroup.open(tabGroup); I just get the standard slide transition. This slide transition is very smooth and works well, but I would prefer if it were a flip transition.

//create homepage window
var homepagewindow = Titanium.UI.createWindow({ title: 'homepage', backgroundColor:'white', navBarHidden:true });

var window1 = Ti.UI.createWindow({backgroundColor:'yellow',navBarHidden:true});
var window2 = Ti.UI.createWindow({backgroundColor:'red',navBarHidden:true});
var window3 = Ti.UI.createWindow({backgroundColor:'blue',navBarHidden:true});

//Set up the Nav Group and assign the first window
    var navGroup = Ti.UI.iPhone.createNavigationGroup({
    window:homepagewindow
    });

//Set up the Tab Group
    var tabGroup = Titanium.UI.createTabGroup();

    //Set up the Tabs and assign their windows
    var tab = Titanium.UI.createTab({
    icon:'images/tabs/KS_nav_ui.png',
        title:'Tab1',
    window: window1
    });
    var tab2 = Titanium.UI.createTab({
    icon:'images/tabs/KS_nav_ui.png',
        title:'Tab2',
    window: window2
    });
    var tab3 = Titanium.UI.createTab({
    icon:'images/tabs/KS_nav_ui.png',
        title:'Tab3',
    window: window3
    });

    //add the tabs to the tab group
    tabGroup.addTab(tab);
    tabGroup.addTab(tab2);
    tabGroup.addTab(tab3);

//This is the main window of the application
var main = Ti.UI.createWindow({navBarHidden:true});
main.add(navGroup);
main.open();
