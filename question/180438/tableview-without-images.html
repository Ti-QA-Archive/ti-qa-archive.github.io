Tableview without images

HI, I need a tableview with 2 fields: a text and a image.

I use this code:

    var rows = db.execute('SELECT * FROM Termostatos ORDER BY Termostatos.Nombre');

    var RowsTabla=[];
    var i;
    var sectionTabla = Ti.UI.createTableViewSection({ headerTitle: txtTituloSeccion });
    RowsTabla.push(sectionTabla);
    for (i=0;i<NRegistros;i++){
        TablaNombreTermostatos[i]= rows.field(2);
        var nuevaRow=Ti.UI.createTableViewRow({ id: i, onClick: "TermoSel", title: rows.field(2), color: "white" });
        var imagen = Ti.UI.createImageView({
            image: 'BORRAR.png'
        });
        nuevaRow.add(imagen);
        RowsTabla.push(nuevaRow);

        rows.next();
    }
    var nuevaRow=(Ti.UI.createTableViewRow({ title: txtMasTermostato, color: "white" }));
    RowsTabla.push(nuevaRow);
    $.MenuLateral.setData(RowsTabla);

if i use the line

        nuevaRow.add(imagen);

the table is empty, but if i delete this line, the table is ok (but without image).

Where is the problem?? Thanks

  • Accepted Answer

    There are a couple of possible issues/solutions. First of all, if you are trying to add an image to your tableViewRow, the built in possible choices are 'leftImage' or 'rightImage'. This will put your image aligned left or right in your tableViewRow. If you change your row definition code to:

    var nuevaRow=Ti.UI.createTableViewRow({ id: i, onClick: "TermoSel", title: rows.field(2), color: "white", leftImage: ' BORRAR.png'});

    You should get your row with your image on the left side of the row.

    Since your row definition is the standard row (ie. you are defining titles, color, etc as the properties of your row) the row displays properly when you remove your image because adding your image is 'overwriting' your row, that is why you can't see them. If you want a custom row, then you should define the objects to go in your row and add them.

    For example, if you wanted to add an image and title and not use the built in row properties, then you would define a label and an image, then add them to your row.

    Ray

    • By default, the event will bubble up to the parent object (the row), so in your function TermoBorra, as the first line put 

      // prevent bubbling up to the row
    e.cancelBubble = true;

      Ray

    • The other option you have is to attach your eventListener to the table, then use the source property to determine which item in the row was clicked. If you attached your function to the table, then your code could look something like this:

      
var rows = db.execute('SELECT * FROM Termostatos ORDER BY Termostatos.Nombre');

    var RowsTabla=[];
    var i;
    var sectionTabla = Ti.UI.createTableViewSection({ headerTitle: txtTituloSeccion });
    RowsTabla.push(sectionTabla);
    for (i=0;i<NRegistros;i++){
        TablaNombreTermostatos[i]= rows.field(2);
        var nuevaRow=Ti.UI.createTableViewRow();
        var titulo=Ti.UI.createLabel({
            text: "       "+rows.field(2),
            color: "white" ,
            textAlign: Ti.UI.TEXT_ALIGNMENT_LEFT, 
            left:2
        });
        var imagen = Ti.UI.createImageView({
            image: '/papelera.png', 
            right: 10,
            width: '5%',
            height: '70%'
        });
        nuevaRow.add(titulo);
        nuevaRow.add(imagen);
        RowsTabla.push(nuevaRow);

        rows.next();
    }
    var nuevaRow=(Ti.UI.createTableViewRow({ title: txtMasTermostato, color: "white" }));
    RowsTabla.push(nuevaRow);

    $.MenuLateral.setData(RowsTabla);
    $.MenuLateral.addEventListener('click', function(e) {
    Ti.API.info( "I push /" + e.source + "/");   

    // Here if source is your Label, run function TermoSel
    // If source is your image, run TermoBorra
    }
