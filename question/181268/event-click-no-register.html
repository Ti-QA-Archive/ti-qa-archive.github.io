  • Try to register your click event in view (xml) file itself.

    
<Alloy>
    <Window class="container" id="login">
        <ImageView id="background"></ImageView>
        <ImageView id="mono_pesas"></ImageView>
        <ImageView id="lower_logo"></ImageView>
        <Label id="labelusuario" text="Usuario"></Label>
            <TextField id="usuario"></TextField>
        <Label id="labelpass"  text="Contraseña"></Label>
            <TextField id="pass"></TextField>
        <View id="forget_holder">
            <ImageView id="forget_image"></ImageView>
            <TextField id="email_forget"></TextField>
            <ImageView id="enviar_forget"></ImageView>
        </View>
        <ImageView id="login_button" onClick="doLogin"></ImageView>
    </Window>
</Alloy>

    
function doLogin(){
    console.log("click");
    Ti.API.info("click");
};

console.log($.login_button);
