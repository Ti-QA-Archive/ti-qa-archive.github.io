  • Nailed it! Thank you! Erase the title within the window and create a label within the TitleControl.

    <Tab class="nav" title="Home" icon="KS_nav_views.png">
            <Window id="win1">
                    <TitleControl>
                    <Label textAlign="Ti.UI.TEXT_ALIGNMENT_LEFT" width="100%">Home</Label>
                    </TitleControl>
                <RightNavButton>
                    <Button>Settings</Button>    
                </RightNavButton>
                <Label>I am Window 1</Label> 
                 <ScrollableView id="scrollableView" showPagingControl="true">
                    <View id="view1" backgroundColor="#123" />
                    <View id="view2" backgroundColor="#246" />
                    <View id="view3" backgroundColor="#48b" />
                </ScrollableView>
