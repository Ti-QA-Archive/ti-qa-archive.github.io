    Hi Juan,

    Following is the classic code. See if it helps. 

    
var win = Ti.UI.createWindow({
    navBarHidden: false
});

var actionBar;

win.addEventListener("open", function() {
    if (Ti.Platform.osname === "android") {
        if (! win.activity) {
            Ti.API.error("Can't access action bar on a lightweight window.");
        } else {
            actionBar = win.activity.actionBar;
            if (actionBar) {
                actionBar.icon = "/appicon.png";
                actionBar.title = "Action bar";
                actionBar.onHomeIconItemSelected = function() {
                    Ti.API.info("Home icon clicked!");
                };
            }

            win.activity.onCreateOptionsMenu = function(e) { 
    var menu = e.menu; 
    var menuItem1 = menu.add({ 
        title : "angry-bird", 
        icon : "/angry-bird.png", 
        showAsAction : Ti.Android.SHOW_AS_ACTION_IF_ROOM 
    }); 

    var menuItem2 = menu.add({ 
        title : "facebook", 
        icon : "/facebook.png", 
        showAsAction : Ti.Android.SHOW_AS_ACTION_IF_ROOM 
    }); 

    var menuItem3 = menu.add({ 
        title : "danger", 
        icon : "/danger.png", 
        showAsAction : Ti.Android.SHOW_AS_ACTION_IF_ROOM 
    }); 


    menuItem1.addEventListener("click", function(e) { 
        Ti.API.info("Angry Bird chirping"); 
    }); 

    menuItem2.addEventListener("click", function(e) { 
        Ti.API.info("Facebook Launched!"); 
    }); 

    menuItem3.addEventListener("click", function(e) { 
        Ti.API.info("Danger zone accessed!"); 
    }); 
};
        }
    }
});

win.open();

    angry-bird.png
    facebook.png
    danger.png

    Paste these images in app resource folder.

    Let know if you need any other than this.
