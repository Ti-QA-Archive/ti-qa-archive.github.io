  • //Back Button to exit the quiz
var Button_Back = Ti.UI.createButton ({
    style: Ti.UI.iPhone.SystemButtonStyle.BORDERED,
    title: 'Zurück'
});
Button_Back.addEventListener ('click', function (e) {
    Ti.UI.currentWindow.close ();
});

    App.Views.Defaults.FlexSpace = Ti.UI.createButton ({
    systemButton: Ti.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

    Ti.App.addEventListener ('Quiz::End', function () {
    Ti.UI.currentWindow.setToolbar ( [App.Views.Defaults.FlexSpace, Button_Back, App.Views.Defaults.FlexSpace]);
})
