    • Activity indicators must be used differently on Android and iOS in 2.x;

      • On Android, the activity indicator is a modal dialog that blocks the UI. Calling show displays the indicator, and calling hide removes it.
      • On iOS and Mobile Web, the activity indicator is a view like any other view, and must be added to a window or other top-level view before it can be shown. Note that, unlike most views, the activity indicator is hidden by default and must be shown explicitly.

      Change this line

      self.add(actInd);

      To become 

      //self.add(actInd);

      Then you can ONLY show the activity indicator when you need it, so above the return line is possibly right for this function, but you may need to adjust.

      actInd.show();
return self;

    • I have had a play around and here is a revised set of your own code with a couple of typo fixes and the activity indicator fixed. It looked like a CommonJS function so I built it as one.

      This gives me an activity indicator perfectly using Ti SDK 2.1.4.

      function create() {
    var actInd = Titanium.UI.createActivityIndicator({});

    actInd.message = 'Please wait...';
    //message will only shows in android.

    var self = Ti.UI.createWindow({
        barColor: "#50b849",
        backgroundColor: "#2d2d2d",
        navBarHidden: false,
        orientationModes: [
            Ti.UI.PORTRAIT
        ],
        title: 'About'
    });
    //self.add(actInd);
    actInd.show();

    var text = Ti.Filesystem.getFile(Ti.Filesystem.resourcesDirectory + "info.txt").read();

    var infoText = Ti.UI.createTextArea({
        width : "100%",
        height : "100%",
        color : "#fff",
        backgroundColor : "#2d2d2d",
        value : text,
        editable : false,
        scrollable : false,
        font : {
            fontWeight : 'bold',
            fontSize : 20,
            fontFamily : 'Helvetica Neue'
        },
        textAlign : Ti.UI.TEXT_ALIGNMENT_CENTER,
        autoLink : Ti.UI.AUTODETECT_LINK
    });

    infoText.addEventListener("click", function() {
        Ti.Platform.openURL("http://tellusanotherone.org/c2p");
    });

    var scrollView = Ti.UI.createScrollView({
        width : "100%",
        height : "100%",
        verticalBounce : true,
        scrollType : "vertical",
    });
    scrollView.add(infoText);
    self.add(infoText);

    self.addEventListener('close', function (e) {
        actInd.hide();
    });
    return self;
}
exports.create = create;
