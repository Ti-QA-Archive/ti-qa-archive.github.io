  • Hello Abdul,
    without seeing the source code difficult to identify the problem.
    I assume you want to show the picker on click of Save button and want to avoid to add the picker every time user clicks on Save button.
    there are 2 option to solve this issue.

    1.add the picker to window and set visible false property, just show/hide the picker on click of Save button.

var win = Ti.UI.createWindow({
    title: 'Title goes here',
    backgroundColor: '#123456'
});
var button = Ti.UI.createButton({
    title: 'Save',
    top: 50,
    height: 40,
    width: 100,
    backgroundColor:'red'
});
var isVisible =false;
var picker = Ti.UI.createPicker({
  bottom:0,
  visible:false
});

var data = [];
data[0]=Ti.UI.createPickerRow({title:'Bananas'});
data[1]=Ti.UI.createPickerRow({title:'Strawberries'});
data[2]=Ti.UI.createPickerRow({title:'Mangos'});
data[3]=Ti.UI.createPickerRow({title:'Grapes'});
picker.add(data);
picker.selectionIndicator = true;
win.add(picker);    
button.addEventListener('click', function() {
    if(isVisible == false){
        picker.visible = true;
        isVisible = true;
    }
    else{
        picker.visible = false;
        isVisible = false;
    }
});
win.add(button);
win.open();

2.add picker to window/view on click of Save button and check if it already added if yes then remove it.I assume you are adding the picker every time user clicks on Save button.
