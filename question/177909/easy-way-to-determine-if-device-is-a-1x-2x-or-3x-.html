  • This is some code I often use.

    switch (Ti.Platform.displayCaps.dpi) {
    case 320 :
        //iphone 4 or 5 retina, check height to determine if iphone 4 vs 5
        var isPhone5 = Ti.Platform.displayCaps.platformHeight > 480 ? true : false;
        break;
    case 160 :
        //iphone 3
        break;
    case 130 :
        //ipad 1 & 2 & iPad mini 1st gen
        break;
    case 260 :
        //ipad3+
        break;
    case 326 :
        //iphone 6
    case 401 :
        //iphone 6+
    default :
        //What the heck is this?
}
