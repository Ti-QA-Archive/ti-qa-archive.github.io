commonJs structure not working on simulator

i am refactoring an app and i would like to use the commonJS structure but is not working on the simulator which makes it imposible to implement.
i have:


// app.js
var win = Ti.UI.createWindow({
  backgroundColor: 'blue'
});

var ui = require('ui'); 
var button = Ti.UI.createButton({
  title: 'Test',
  top: 20,
  height: 50,
  width: 200
});
button.addEventListener('click', function(e){
  ui.alertIt('hi');
});
win.add(button);
win.open();

and then in the ui.js

// ui.js
exports.alertIt = function(e) {
  alert(e);
};

I am using Titanium Studio : 1.0.7
Titamiun SDK : 1.8.0.1
iphone SDK: 4.2 (also tried with 5.0)

Everytime i run it i get Script Error = Can't find module: ui
I double checked and ui.js it's in the resources folder together with the app.js

any ideas?
Regards.
