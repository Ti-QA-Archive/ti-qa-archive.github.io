ProgressBar when update database

Hi, I need help to fix my problem with progress bar into database updating process.
I use db.js for all database functions and ui.js for all ui related code.

Now into db I have created this function:

App.db.update = function() {
    var db = Ti.database.open('app');

    var test = 100;

    for ( var i=0; i < test; i++ ) {
        Ti.App.fireEvent( 'updateDbStart', function(e) {
            e.message = 'update row ' + i;
        });
    }

}

into ui.js use this example:

App.UI.example = function() {
    var win = Ti.UI.createWindow();

    App.db.update();

    Ti.App.addEventListener( 'updateDbStart', function(e) {
        alert(e.message);
    });
}

In this example the message var show only when the loop into db file is finish. If not separe db.js and ui.js and integrate into db loop working property.

Now need to earn the fire event system because not working in real time with my code.

Thanks for suggestions.
