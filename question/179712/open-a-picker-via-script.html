  • Here is the code, notice the image view called dropDownIndicator. When that is clicked I want the app to behave as if the user clicked on the picker called itemPicker. I want the options I added to data to show up on the screen and be selectable.

    Thanks.

    var selected = Ti.App.Properties.getString("itemDesc");
    var itemSelectLab = Ti.UI.createLabel({
        text:"Please select the item",
        top:"35%",
        left:"5%",
        font:{
            fontSize:"12dp"
        },
        color:"#406270"
    });
    buttonsView.add(itemSelectLab);

    var itemPicker = Ti.UI.createPicker({
        top:"32%",
        right:"10%",
        backgroundColor:"#AAA",
        height:"8%"
    });
    var dropDownIndicator = Ti.UI.createImageView({
        image:"/images/dropDown.png",
        top:"32%",
        height:"8%",
        right:"5%",
        width:"5%"
    });
    buttonsView.add(dropDownIndicator);
    dropDownIndicator.addEventListener("click",function(e){
        itemPicker.fireEvent("click",e);
    });

    var data = [];
    data[0]=Ti.UI.createPickerRow({title:"Hours Worked"});
    data[1]=Ti.UI.createPickerRow({title:"Calves"});
    data[2]=Ti.UI.createPickerRow({title:"Bales of Hay"});
    data[3]=Ti.UI.createPickerRow({title:"Skateboards"});
    data[4]=Ti.UI.createPickerRow({title:"Babysitting"});

    for(var i in data){
        if(data[i].title == selected){
            itemPicker.setSelectedRow(0,i,false);
        }
    }

    itemPicker.add(data);
    itemPicker.selectionIndicator = true;
    buttonsView.add(itemPicker);

    itemPicker.addEventListener("change",function(e){
        Ti.App.Properties.setString("itemDesc",itemPicker.getSelectedRow(0).title);
    });
