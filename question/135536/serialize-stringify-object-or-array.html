Serialize Stringify Object or Array

Hi experts,
in order to save an array of values in the Ti.App.Properties I first have to serialize the array or I can
use setList (which will use serialize as well).
But this doesn't work with the following structure.
What I'm doing wrong ?
All three outputs don't work.

    var aryData = [];
    for (var i=0; i < objFavs.length; i++) {
        aryData[i] = [];
        aryData[i]["id"] = objFavs[i].id;
        aryData[i]["bezeichnung"] = objFavs[i].bezeichnung;
        aryData[i]["beschreibung"] = objFavs[i].beschreibung;
        aryData[i]["valuecolumns"] = objFavs[i].valuecolumns;
    }
    Ti.App.Properties.setList("favoriten", aryData);
    Ti.API.info("schreibeary :" + JSON.stringify(aryData[0]));
    Ti.API.info("schreibefavs :" + JSON.stringify(objFavs));

TIA Stephan

  • Accepted Answer

    Try this:

    var aryData = [];

for(var i = 0; i < objFavs.length; i++) {
    // aryData[i] = [];
    aryData[i] = {};
    aryData[i]["id"] = objFavs[i].id;
    aryData[i]["bezeichnung"] = objFavs[i].bezeichnung;
    aryData[i]["beschreibung"] = objFavs[i].beschreibung;
    aryData[i]["valuecolumns"] = objFavs[i].valuecolumns;
}
