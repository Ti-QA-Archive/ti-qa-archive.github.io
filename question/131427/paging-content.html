  • Check this out 

    var win=Titanium.UI.createWindow({
    title:'Tab 1',
    backgroundColor:'#fff',
    orientationModes:[Ti.UI.LANDSCAPE_LEFT, Ti.UI.LANDSCAPE_RIGHT]
});

var view1 = Ti.UI.createView({
    backgroundColor : 'red',
    borderRadius : 10,
    width : 1030,
    height : 640,
    top : 0,

});

var view2 = Ti.UI.createView({
    backgroundColor : 'blue',
    borderRadius : 10,
    width : 1030,
    height : 640,
    top : 0,

});

var view3 = Ti.UI.createView({
    backgroundColor : 'green',
    borderRadius : 10,
    width : 1030,
    height : 640,
    top : 0,

});
view_array = [view1, view2, view3];
for(var view in view_array) {
    win.add(view_array[view]);
}

var forward_button = Titanium.UI.createButton({
    //image:'forward.png',
    title : 'F',
    top : 660,
    right : 100,
    width : 50,
    height : 50
});

var currentViewIndex = 0;

var showView = function(viewIndex) {
    for(var view in view_array) {
        view_array[view].hide();
    }
    view_array[viewIndex].show()
    setButtonStates(viewIndex);

}
var setButtonStates = function(viewIndex) {
    back_button.enabled = true;
    forward_button.enabled = true;
    if(viewIndex == view_array.length - 1) {
        forward_button.enabled = false;
    } else if(viewIndex == 0) {
        back_button.enabled = false;
    }
}

var funcRef_forward = function() {
    currentViewIndex++;
    showView(currentViewIndex);
}

forward_button.addEventListener('click', funcRef_forward);

var back_button = Titanium.UI.createButton({
    //image:'back.png',
    title : 'B',
    top : 660,
    right : 155,
    width : 50,
    height : 50
});

var funcRef_back = function() {
    currentViewIndex--;
    showView(currentViewIndex);
}

back_button.addEventListener('click', funcRef_back);
win.add(forward_button);

win.add(back_button);

showView(0);

win.open();
