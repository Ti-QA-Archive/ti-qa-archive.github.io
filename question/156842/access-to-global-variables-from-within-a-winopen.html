Access to global variables from within a win.open()

Hello, I am having trouble accessing variables defined in the app.js file from within windows that are opened after clicking on a table row. I was under the impression that variables defined in this way would be truly global? For example:

app.js

var foo = "bar";

var win = Ti.UI.createWindow();

var table     = Ti.UI.createTableView();
var tableView = Ti.UI.createTableViewSection();

tableView.add(Ti.UI.createTableViewRow({ title: "Foo" });

table.setData([ tableView ]);

table.addEventListener('click', function(e) {
    var newWin = UI.win({ title: "Foo", url: "bar.js" });
    win.open(newWin);
});

win.add(table);

bar.js

alert(foo);
