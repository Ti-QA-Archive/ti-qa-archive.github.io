Android Facebook login -- missing appid

This code fails on Android. The same code works on iOS. What am I doing wrong?

var fb = require('facebook);

fb.addEventListener('login', onFacebookLogin);
fb.addEventListener('logout', onFacebookLogout);

fb.applyProperties({
    appid : '[FACEBOOK APPID]',
    permissions : ['email'],
    forceDialogAuth : false
});

fb.authorize();

tiapp.xml has the following but nothing specific for android:

<property name="ti.facebook.appid">415070288583165</property>
<ios>
    <min-ios-ver>7.0</min-ios-ver>
        <plist>
            <dict>
                <key>FacebookAppID</key>
                <string>[FACEBOOK APPID]</string>

Getting the following error:

[ERROR] :  TiExceptionHandler: (main) [71,32442] ----- Titanium Javascript Runtime Error -----
[ERROR] :  TiExceptionHandler: (main) [1,32443] - In js/platform.js:289,14
[ERROR] :  TiExceptionHandler: (main) [0,32443] - Message: Uncaught Error: missing appid
[ERROR] :  TiExceptionHandler: (main) [0,32443] - Source:     facebook.authorize();
[ERROR] :  V8Exception: Exception occurred at js/platform.js:289: Uncaught Error: missing appid

Thank you for your help.

  • Hello Gopi,

    Thanks for your inquiry, First, Add the facebook module in your tiapp.xml file. Save it. Now use the below code 

    var win = Ti.UI.createWindow({backgroundColor: 'white'});
var fb = require('facebook');
fb.appid = "xxxxxxxxxxxxxxx";
fb.permissions =  ['publish_stream'];

fb.addEventListener('login', function(e) {
    if (e.success) {
        alert('Logged in');
    }
});
fb.addEventListener('logout', function(e) {
    alert('Logged out');
});
win.add(fb.createLoginButton({
    top : 50,
    style : fb.BUTTON_STYLE_WIDE
}));
win.open()

    replease the "fb.appid" value to your facebook application id value. Thanks.
