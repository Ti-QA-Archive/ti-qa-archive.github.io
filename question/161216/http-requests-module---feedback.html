  • I think every parameter makes sense. If you need a GET-only module without error handling, you can write a simple module:

    exports.ajax = function() {
    var options = arguments[0] || {};
    var self = Ti.Network.createHTTPClient({
        onload: options.onload(this.responseText)
    });
    self.open('GET',options.url,true);
    self.send();
}

    And now you can use:

    require('ajax')({
    url:'http://mickymouse.com/',
    onload: function(data){
        alert(data);    
});
