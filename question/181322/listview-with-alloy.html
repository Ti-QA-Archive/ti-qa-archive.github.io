    //CURRENT DAY
    var today = new Date();
    var dd = today.getDate();
    var mm = today.getMonth()+1; //January is 0!
    var yyyy = today.getFullYear();
    //

//DO the save process
    var task = Alloy.createModel('todo', {
        name : $.tareaInput.value,
        hour : $.horaInput.value,
        dateCreated : mm+'/'+dd+'/'+yyyy,
        dueDate: $.fechaInput.value,
        description: $.descInput.value,
        status: 1 //1 será el estado por defecto activo
    });
    myTasks.add(task);
    task.save();

 alert('Tarea guardada éxitosamente');

}
