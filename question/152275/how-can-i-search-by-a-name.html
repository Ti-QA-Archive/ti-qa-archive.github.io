    • Use like below.

      
var tableDataArray = []

for( var i = 0; i < arrayVideos.length; i++ ){
    var row = Ti.UI.createTableViewRow({
    title: arrayVideos[i].video_title
        description: arrayVideos[i].description
    })
    tableDataArray.push( row );
}

var search = Titanium.UI.createSearchBar({
    showCancel:true,
    top:0
});

var tableview = Ti.UI.createTableView({
    data: tableDataArray,
        search: search,
    filterAttribute: 'description'
});
