  • hello, Victor Your Question is not clear. What do you mean by "saved a photo from photo gallery"? is it your app select some photo in photo gallery and then save them in other place/folder, say in SD card? If so, Then here is a code to save any file or image from gallery to the specific folder of your SD card.

    var win = Titanium.UI.createWindow({
     title : 'Photo Upload',
     backgroundColor : '#fff',
     layout : 'vertical'
});

// Create a Button.
var uploadPhoto = Ti.UI.createButton({
     title : 'Upload Photo',
     height : Ti.UI.SIZE,
     width : Ti.UI.SIZE,
     top : 10,
     color : 'pink'
});

win.add(uploadPhoto);

//var path = [];

var imageDir = Ti.Filesystem.getFile(Ti.Filesystem.externalStorageDirectory, 'images');
if (! imageDir.exists()) {
     imageDir.createDirectory();
}
// Listen for click events.
uploadPhoto.addEventListener('click', function() {

     Titanium.Media.openPhotoGallery({
          success : function(event) {
               var filename = new Date().getTime() + "-ea.jpg";
               if (Ti.Filesystem.isExternalStoragePresent()) {
                     var imageFile = Ti.Filesystem.getFile(imageDir.resolve(), filename);
                     Ti.API.info("ImageFile path is: " + imageFile.resolve());
                     imageFile.write(event.media);
                     if (imageFile.write(event.media) === false) {
                          alert('error');
                     } else {
                          alert('see your sd Card project folder');
                     }

               } else {
                     alert("Your Memory Card is not Present");
               }

          }
     });
});

win.open();
