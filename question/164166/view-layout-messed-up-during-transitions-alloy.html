view layout messed up during transitions. Alloy

Hi, I have an alloy app that uses the iOS flip animation to transition between two views this work sfine however during the animation the second view (that the animation is flipping to) is shown with tss styling. Why is this and how can i fix it.

var state = true;
$.signUp.addEventListener('click', function(e) {
    var register=Alloy.createController('register').getView();

    switch (state) {
        case true:
            Ti.API.info('true');
            $.flipWrap.animate({
            view: register,
            transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
        });
        break;
        case false:
            Ti.API.info('false');
            $.flipWrap.animate({
            view: $.loginForm,
            transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_LEFT
        });
        break;
    }
    state = !state;
});
