Alloy databinding tableview row to model position in collection

Using Alloy databinding would it be possible to bind model position in the collection directly to the model in question when collection reshuffles?

What I'm trying to do is have a tableview bound to a collection (straight from Alloy tutorials) but when calling (moving rows): 

mytable.addEventListener('move', function(e){
   // update positon on each model in the collection with new table index
})

Sure, I could iterate over the tableview and trigger an event on every model to set new position, but isn't there a way to bind model property "position" directly to a tablerow and when its position changes inside the collection (which is my table) save the new value to the db?

Thanks in advance

    • Alright, no worries. I dug up Aaron Saunders code and modified it a little bit. I ended up doing something like this by extending collection with a new method:

      updatePositions : function() {
    var collection = this;
    var util = require("alloy/sync/util");
    var dbName = collection.config.adapter.db_name;
    var table = collection.config.adapter.collection_name;
    var columns = collection.config.columns;
    db = Ti.Database.open(dbName);
    db.execute("BEGIN;");
    collection.each(function(model) {
        if (!model.id) {
            model.id = util.guid();
            model.attributes[model.idAttribute ] = model.id;
        }
        var sql = "UPDATE " + table + " SET position = "+model.get('position')+" WHERE "+model.idAttribute+" = '"+model.id+"'";
        db.execute(sql);
    });
    db.execute("COMMIT;");
    db.close();
    collection.trigger('sync');
}

      So no need to .save() individual model. Calling this on collection suffice.
