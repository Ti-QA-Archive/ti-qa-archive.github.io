  • Well you can pass the object ( textfield / button object ) with the window you want to open and set the value from the new window.

    Maybe according to following code :

    MainWindow.js

    var win = Ti.UI.createWindow();
var button = Ti.UI.createButton( title : "Male" );
win.add(button);
button.addEventListener('click', function() 
{
    var win_set = Ti.UI.createWindow( url : "win_set.js" , customButtonObj : button );
    win_set.open();
});

win.open();

    win_set.js

    var win_set = Ti.UI.currentWindow;
var button_set = Ti.UI.createButton( title : "Change" );
win_set.add(button_set);
button_set.addEventListener('click', function() 
{
    win_set.customButtonObj.title = "Female";
    win_set.close();
});
