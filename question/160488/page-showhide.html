  • Try this simple code :

    ###index.xml 

    <Alloy>
    <Window id="win" title="Home">
        <Label  text="Main Window"/>
        <Button id="btn" title="open new window" onClick="openWindow"/>    
    </Window>
</Alloy>

    ###index.js

    $.win.open();

function openWindow(e)
{
    var winTest = Alloy.createController('test').getView();
        winTest.open();
}

    ###test.xml 

    <Alloy>
    <Window id="winTest" title="Test">
        <Label  text="New Window"/>
        <Button id="btn" title="closewindow" onClick="closeWindow"/>    
    </Window>
</Alloy>

    ###test.js

    
function closeWindow(e)
{
        $.winTest.close();
}
