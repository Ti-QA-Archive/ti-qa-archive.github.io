  • Hello Anton,

    See this…

    
var win = Titanium.UI.createWindow({
    title : 'Test',
    backgroundColor : '#fff',
    exitOnClose : true
});

var scrollView = Ti.UI.createScrollView({
    contentWidth : 'auto',
    contentHeight : 'auto',
    top : 0,
    layout : 'vertical',
    touchEnabled : true,
    backgroundColor : 'yellow',
});
win.add(scrollView);

var facebookview = Ti.UI.createView({
    width : 320,
    height : 50,
    zIndex : 20,
    backgroundColor : 'pink',
    opacity : 1.0,
    top : 5
});
scrollView.add(facebookview);

facebookview.addEventListener('click',function(e){
    alert('View Click');
});

var btn = Ti.UI.createButton({
    title : 'Click Me',
    top : 60,
    left : 30,
    right : 30,
    height : 40
});
scrollView.add(btn);

btn.addEventListener('click',function(e){
    alert('Button Click');
});

win.open();
