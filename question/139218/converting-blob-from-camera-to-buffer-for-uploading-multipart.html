Converting Blob from camera to buffer for uploading multipart

Hi,
internal camera gives me a blog. To send with multipart I have to convert TiBlob to Buffer.

I'am using this code:

exports.get = function(foo) {
    switch (typeof foo) {
        case 'string':
            return Ti.createBuffer({
                value : foo
            });
            break;
        case 'object':
            var instream = Ti.Stream.createStream({
                mode : Ti.Stream.MODE_READ,
                source : foo // e.media is a Blob
            });
            var buffer = Ti.createBuffer({
                length : foo.length
            });
            while ((instream.read(buffer)) > 0) {
            }
            instream.close();
            return buffer;
            break;
    }
}
