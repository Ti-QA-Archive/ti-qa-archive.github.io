Should I setup 30 different .js files (i.e window1.js, windows2.js) and dedicate an ID to each one, or is there a leaner approach I should consider.

  • How about this?

    var rootWindow = Titanium.UI.createWindow({backgroundColor : "blue", title : "Home"});
var button = Ti.UI.createButton({title: 'Start', height: 40, width: 80});
rootWindow.add(button);

var nav = Titanium.UI.iPhone.createNavigationGroup({
    window : rootWindow
});

var win = Titanium.UI.createWindow();
win.add(nav);
win.open();

button.addEventListener('click', function(){
    showData(1);
});

function showData(recordID) {
    // TODO - Fetch record based on recordID 

    var detailWin = Titanium.UI.createWindow({backgroundColor : "white", title : "Window #" + recordID});
    var label = Ti.UI.createLabel({text: 'Detail: ' + recordID, height: 100, width: '100%', color: 'red'});
    detailWin.add(label);

    var nextBtn = Ti.UI.createButton({title: 'Next'});
    detailWin.rightNavButton = nextBtn;
    nextBtn.addEventListener('click', function(){
        showData(recordID + 1);
    });

    nav.open(detailWin, {animated : true});
}

    It still use navigation, but wont display all records at all.
