    • Try below Code : 

      myapp.getRows = function() {
    var db = Ti.Database.open('weather');
    var data = [];
    var rs = db.execute('SELECT id, textField1 FROM cities');
    while (rs.isValidRow()) {
        var tblRow = Ti.UI.createTableViewRow({
            minRowHeight : 40,
            id : rowID
        });
        var rowID = rs.fieldByName('id');
        var buttonID = Ti.UI.createButton({
            title : rowID,
            width : 35,
            height : 35,
            left : 2,
            top : 2,
            id : rowID
        });
        tblRow.add(buttonID);
        buttonID.addEventListener('click', function(e) {
            var id = e.source.id
            var openDelete = Ti.Database.open('weather');
            var alertYesNo = Titanium.UI.createAlertDialog({
                message : 'Are you sure you want to delete',
                buttonNames : ['Yes', 'No']
            });
            alertYesNo.addEventListener('click', function(e) {
                if (e.index == 0) {
                    //clicked yes
                    function deleteID() {
                        openDelete.execute('DELETE FROM cities WHERE id=?', id);
                        openDelete.close();
                    };
                    deleteID();
                } else if (e.index == 1) {
                    //clicked no
                }

            });
            alertYesNo.show();
        });
        tblRow.add(Ti.UI.createLabel({
            text : rs.fieldByName('textField1'),
            font : {
                fontWeight : 'bold',
                fontSize : 20
            },
            color : 'black',
            left : 40,
            top : 2
        }));
        data.push(tblRow);
        rs.next();
    }
    db.close();
    return data;
};
