  • Hello Brad,

    Thanks for writing in community. Here I have attached my simple code for posting data from mobile apps to PHP based web server. Please check this code in your side. I hope it’s solving your problem.

    Titanium Code

    
var win1 = Titanium.UI.createWindow({

    title : 'Red Window'
});

var hour = Ti.UI.createTextField({
    height : 35,
    top : 10,
    left : 40,
    width : 240,
    hintText : 'hour',
    keyboardType : Ti.UI.KEYBOARD_DEFAULT,
    returnKeyType : Ti.UI.RETURNKEY_DEFAULT,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});

win1.add(hour);

var minute = Ti.UI.createTextField({
    height : 35,
    top : 10,
    left : 40,
    width : 240,
    hintText : 'minute',
    keyboardType : Ti.UI.KEYBOARD_DEFAULT,
    returnKeyType : Ti.UI.RETURNKEY_DEFAULT,
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED
});

win1.add(minute);
var url = 'your post url here';
var send = Ti.UI.createButton({
    title : 'send',

});

send.addEventListener('click', function() {

var data =  {
    hour : hour.value,
    minute : minute.value
};
var xhr = Ti.Network.createHTTPClient({
        onload : function() {
            Ti.API.info(this.responseText);
            Ti.API.info("TEXT:   " + this.responseText);
            json = JSON.parse(this.responseText);

        },
        onerror : function(e) {
            Ti.API.info("STATUS: " + this.status);
            Ti.API.info("TEXT:   " + this.responseText);
            Ti.API.info("ERROR:  " + e.error);
            alert(L('network_problem'));
        },
        timeout : 50000
    });

    xhr.open('POST', encodeURI(url));
    xhr.setRequestHeader('Content-Type', 'application/json');
    xhr.send(JSON.stringify(data));


});

win1.add(send);

win1.open();

<?php

if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $params = file_get_contents('php://input');
    $params = json_decode(str_replace('\"', '"', $params));

        echo json_encode((object) $params);

}

    ?>
