Webview-beforeload event

When I perform the before load event for webview. 

function function1()
{
alert("beforeload");
}

WebView.addEventListener('beforeload', function1);

But that function calls no. of times when html pages load..
I want to call function1 only one time before loading the webview.
Please specify the answer?

    var hasAlreadyLoaded = false;

function function1()
{
    if (hasAlreadyLoaded == true) {
        return;
    }

    hasAlreadyLoaded = true;


alert("beforeload");

}

WebView.addEventListener('beforeload', function1);
