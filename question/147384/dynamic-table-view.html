  • Hello Chand,

    See this…

    var win = Titanium.UI.currentWindow;

var tblArray = [];

var tableData = [ {title: 'Post 1'}, {title: 'Post 2'}, {title: 'Post 3'}, {title: 'Post 4'} ];

var table = Ti.UI.createTableView();

for (var i=0; i < tableData.length; i++) {
     var rows = Ti.UI.createTableViewRow({
         title : tableData[i].title
     });
     tblArray.push(rows);
};
table.setData(tblArray);
win.add(table);

  • Hi
    This is the code that uses sqlite database and fill the table at dynamic time this would suits you more
    app.js

    var db = require('db');
Ti.App.Quotes=new Array();
var myfontsize = (Titanium.Platform.displayCaps.platformHeight * 3) / 100;
var customFont = 'CutiveMono-Regular';
var HomeWin = Ti.UI.createWindow({
    backgroundColor : 'white',
    orientationModes : [Ti.UI.PORTRAIT]
})
Ti.App.myindex = 0;
var results = db.IQfun();
Ti.App.Quotes=results;
var tableView1 = Titanium.UI.createTableView();
HomeWin.add(tableView1);
var Row = [];

var length = 0;
if ( typeof results === "string") {
    length = 1;
} else {
    length = results.length;
}

for (var i = 0; i < length; i++) {
    var rss = results[i];

    var rssRow = Titanium.UI.createTableViewRow({
        height :'43dp',
        //top:'5%',
        className : 'rssrow'

    });
    var titleLabel = Titanium.UI.createLabel({
        text :i+1+')'+ rss,
        font : {
            fontSize : myfontsize,
        //    fontWeight : 'bold',
            fontFamily : customFont
        },
        width : 'auto',
        top : '10%',
        left : '5%',
        right : '5%',
        height : 'auto',
        color : 'black'
    });

    rssRow.add(titleLabel);

    Row.push(rssRow);
};

tableView1.setData(Row);

tableView1.addEventListener('click', function(e) {
    Ti.App.myindex = e.index;

    var DetailWin = Ti.UI.createWindow({
        url : 'DetailWin.js',
        backgroundColor : 'white',
        orientationModes : [Ti.UI.PORTRAIT]
    })
    DetailWin.open({
        modal : true
    });
});

HomeWin.open();

    db.js

    Ti.Database.install('Inspirational quotes.sqlite', 'scimed');
var db = Ti.Database.open('scimed');
function IQfun() {
    var dataArray = new Array();
    var counter = 0;
    var IQs = db.execute("select * from IQ");
    while (IQs.isValidRow()) {
        dataArray[counter] = IQs.fieldByName('IQ');
        IQs.next();
        counter++;
    };

    return dataArray;

}
function Authorfun() {
    var dataArray = new Array();
    var counter = 0;
    var IQs = db.execute("select * from IQ");
    while (IQs.isValidRow()) {
        dataArray[counter] = IQs.fieldByName('Author');
        IQs.next();
        counter++;
    };

    return dataArray;

}
exports.IQfun = IQfun;
exports.Authorfun = Authorfun;
