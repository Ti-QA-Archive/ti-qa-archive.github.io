function _db() {

        if ( Ti.App.Properties.getBool('initStartUp',false) ) { 
            db = Titanium.Database.open(filenameDb);
                    }else{
            Ti.App.Properties.setBool('initStartUp',true);
            db = Titanium.Database.open(filenameDb);

    return db;
}
