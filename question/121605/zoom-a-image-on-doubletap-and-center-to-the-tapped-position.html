Zoom a image on doubletap and center to the tapped position

I am trying to create a scroll view with images were when doubletapping the image should in by a factor 4, rotate a little and center to the tapped postion. I a trying to use the center property of the animation class, but a do not get the desired result.

var tapped = false;

image1.addEventListener('doubletap', function(e) {
    var a = Titanium.UI.createAnimation();

    if (tapped === false) {

    var t3 = Ti.UI.create2DMatrix();
        t3 = t3.rotate(10);
        t3 = t3.scale(4);
        //a.left -= e.x;
        //a.top -= e.y;
        a.center = {x:e.x, y:e.y};
        //alert(e.x + ' blaat ' + e.y);
        tapped = true;
    } else {
        var t3 = Ti.UI.create2DMatrix();
        t3 = t3.rotate(0);
        t3 = t3.scale(1);
        a.center = {x:'50%', y:'50%'};
        tapped = false;
    }

    a.transform = t3;
    a.duration = 400;
    image1.animate(a);
});

am I understanding the center property wrong? Is there another way to fix this?

    The easiest way to do this is add the image to a scrollView. It has all the properties you need to do this. You might be able to hack it with just the image view, but the scroll view would just be easier as it supports zoom without needing to use the 2D Matrix.
