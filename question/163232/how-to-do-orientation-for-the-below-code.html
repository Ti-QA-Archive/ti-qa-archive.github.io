function setAllLbl(tops,txt) {
    var label = Ti.UI.createLabel({
        top : tops,
        left :70,
        text : txt,
        color : '#000',
        font : {
            fontSize : 22,
            fontFamily : 'Arial'
        }
    });
    return label;

}

function textFieldProperty(values) {
    var property = {
        color : '#000',
        height : Ti.UI.SIZE,
        top : 10,
        left : 70,
        width : 200,
        value : values,
        font : {
            fontSize : 22,
            fontFamily : 'Arial'
        }
    };
    return property;
}

loginWin.open();
