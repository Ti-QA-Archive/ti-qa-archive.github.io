Drag and Drop animation not working with 3.1.3

Drag animation not working in 3.1.3 I'm using following code. 

var win = Ti.UI.createWindow({
    backgroundColor : "#fff"
});
var annoViewPosition;
var view = Ti.UI.createView({
    borderWidth : 1,
    borderColor : "#000000",
    left : 0,
    top : 0,
    height: 100,
    width: 100
});
win.add(view);

var lbltop;
var lblleft;
view.addEventListener('touchstart', function(e) {
        annoLblPosition = {
            top : e.source.top,
            left : e.source.left
        };
        lbltop = e.y;
        lblleft = e.x;
});

view.addEventListener('touchmove', function(e) {
    annoLblPosition.top += e.y - lbltop;
    annoLblPosition.left += e.x - lblleft;
    e.source.animate({
        top : annoLblPosition.top,
        left : annoLblPosition.left,
        duration : 5,
    });
});

win.open();

Even I tried kitchen sink animation code that given in basic animation to drag a circle. That even not working properly in 3.1.3 and 3.1.2.
