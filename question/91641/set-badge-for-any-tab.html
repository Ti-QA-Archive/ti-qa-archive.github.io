    Here is a complete project where a button in the second tab is updating the first tab… if this doesn't answer your question, the can I suggest that you update your question so someone can help you out

    // this sets the background color of the 
// master UIView (when there are no windows/tab groups on it)
Titanium.UI.setBackgroundColor('#000');

// create tab group
var tabGroup = Titanium.UI.createTabGroup();


//
// create base UI tab and root window
//
var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});
var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1,
    badgeCount:1
});

var label1 = Titanium.UI.createLabel({
    color:'#999',
    text:'I am Window 1',
    font:{fontSize:20,fontFamily:'Helvetica Neue'},
    textAlign:'center',
    width:'auto'
});

win1.add(label1);

//
// create controls tab and root window
//
var win2 = Titanium.UI.createWindow({  
    title:'Tab 2',
    backgroundColor:'#fff'
});
var tab2 = Titanium.UI.createTab({  
    icon:'KS_nav_ui.png',
    title:'Tab 2',
    window:win2,
    badgeCount:1
});

var updateBtn = Titanium.UI.createButton({
    color:'#999',
    title:'Update Badge One',
    height:35,
    width:150
});

var resetBtn = Titanium.UI.createButton({
    color:'#999',
    title:'Reset Badge One',
    top:35,
    height:35,
    width:150
});

updateBtn.addEventListener('click',function(evt){
    tab1.setBadge(tab1.badgeCount++);
});
resetBtn.addEventListener('click',function(evt){
    tab1.setBadge(null);
});
win2.add(updateBtn);
win2.add(resetBtn);



//
//  add tabs
//
tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);  


// open tab group
tabGroup.open();

    If you have the tab object you can set it's badge value.

    Create a new default two tab project and add this code at the end; the badges will be displayed.

    // set first tab to 1
tab1.setBadge(1);

// set second tab to 2
tab2.setBadge(2);
