    Very rough, since you original code doesn't actually work. Does not represent all best practices. Does not take into account all possible scenarios. It DOES remove your globals. It does actually run. Simply a demonstration:

    index.js

    <Alloy>
    <Window id="win">
        <TableView id="table"/>
    </Window>
</Alloy>

    index.tss

    "#win":{
    backgroundColor: "white"
}

    index.js

    $.win.addEventListener('open', function (e) {

    var url = 'http://www.json-generator.com/api/json/get/cpbFLzVLAO?indent=2';
    var xhr = Ti.Network.createHTTPClient({
        onload: function (e) { //alert(this.responseText); 
            var theJSON = {};
            try {
                theJSON = JSON.parse(this.responseText);
            } catch (error) {
                console.log(error);
            } finally {

                loadJson(theJSON);
            }


        },
        onerror: function (e) {
            Ti.API.info(this.responseText);
            Ti.API.info(this.status);
            Ti.API.info(e.error);
        },
        timeout: 5000
    });
    xhr.open('GET', url);
    xhr.send();

    function loadJson(theJSON) {
        console.log(theJSON.cars);
        if (theJSON.cars) {
            var tableData = [];
            for (var i = 0, length = theJSON.cars.length; i < length; i++) {
                var car = theJSON.cars[i];
                var row = Ti.UI.createTableViewRow({
                    height: '10%'
                });
                var name = Ti.UI.createLabel({
                    text: car.MakeofVehicle,
                    top: 0
                });
                var model = Ti.UI.createLabel({
                    text: car.ModelofVehicle,
                    bottom: 0
                });
                row.add(name);
                row.add(model);
                tableData.push(row);
            }
            $.table.data = tableData;
        }

    }

});

$.win.open();
