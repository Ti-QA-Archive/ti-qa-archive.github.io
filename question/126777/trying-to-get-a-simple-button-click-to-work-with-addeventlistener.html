Trying to get a simple button click to work with addEventListener

I can't seem to get an addEventListener to work.

app.js calls main_windows/home.js
home.js has an include to ui/tabs.js

I have the following button code:

var avatarShop = Ti.UI.createButton({
    title: 'AVATAR',
    width:80,
    height:60,
    backgroundColor:'orange',
    backgroundFocusedColor:'red',
    backgroundImage:'',
    backgroundFocusedImage:'',
    font: {
        fontSize:10
    },
    color:'#fff',
    top:400,
    left:0
});

var testAlert = Ti.UI.createAlertDialog({
    title:'It Worked!',
    message:"Nice'ly Done",
    buttonNames: ['OK', 'Awesome!']
});

avatarShop.addEventListener('click', function(e) {
    testAlert.show();
});

The alert works by it's self. The button loads, but the listener won't fire.

Any suggestions as to what I'm doing wrong?

iOS 4.1
Titanium Mobile 1.7.2

    After making some changes to get your code to run in a test case, the alert opens fine.

    Ti.UI.setBackgroundColor('#fff');

var win = Ti.UI.createWindow({
    title : 'test',
});


var avatarShop = Ti.UI.createButton({
    title: 'AVATAR',
    width:80,
    height:60,
    backgroundColor:'orange',
    backgroundFocusedColor:'red',
    backgroundImage:'',
    backgroundFocusedImage:'',
    font: {
        fontSize:10
    },
    color:'#fff',
    top:400,
    left:0
});

 win.add(avatarShop);

var testAlert = Ti.UI.createAlertDialog({
    title:'It Worked!',
    message:"Nice'ly Done",
    buttonNames: ['OK', 'Awesome!']
});

avatarShop.addEventListener('click', function(e) {
    testAlert.show();
});

win.open();
