Centering two labels in a view in Android

I was wondering if anyone might have an idea for how to center two side by side labels of variable width in a view. The following code will center the two labels on iOS but on Android they are left aligned. It seems the problem is that in Android the "internalContainer" view stretches to the width of "externalContainer", even though it's set to "auto". What I need it to do is stretch to the width of the content, as it does in iOS. Any ideas?

var win = Ti.UI.createWindow({  
    title:'Test',
      backgroundColor: '#fff'
});

var externalContaner = Ti.UI.createView({
    width: '200dp',
    height: '50dp',
    right: 0,
    backgroundColor: 'yellow'
});
win.add(externalContaner);

var internalContaner = Ti.UI.createView({
    width: 'auto',
    height: '50dp',
    right: 0,
    backgroundColor: 'yellow',
    layout: 'horizontal'
});
externalContaner.add(internalContaner);

internalContaner.add(Ti.UI.createLabel({
    text: 'Text 1',
    width: 'auto',
    color: '#000'
}));

internalContaner.add(Ti.UI.createLabel({
    text: 'Text 2',
    width: 'auto',
    color: '#000'
}));

win.open();

  • Jude

    If I understand your situation correctly, the code below is what you need (tested on iPhone and Android):

    var win = Ti.UI.createWindow({
  backgroundColor: '#fff',
  title:'Test',
  exitOnClose:true
});

var externalContainer = Ti.UI.createView({
  top: 20,
  height: '60dp',
  width: '100%',
  backgroundColor: 'red'
});
win.add(externalContainer);


var contanerLeft = Ti.UI.createView({
  backgroundColor: 'blue',
  left: 0,
  height: '50dp',
  width: '50%'
});
externalContainer.add(contanerLeft);

var contanerRight = Ti.UI.createView({
  backgroundColor: 'green',
  width: '50%',
  height: '50dp',
  right: 0
});
externalContainer.add(contanerRight);

contanerLeft.add(Ti.UI.createLabel({
  backgroundColor: 'yellow',
  width: 'auto',
  text: 'This label1'
}));

contanerRight.add(Ti.UI.createLabel({
  backgroundColor: 'yellow',
  width: 'auto',
  text: 'and this is label two'
}));

win.open();
