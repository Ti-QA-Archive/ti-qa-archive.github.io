  • Hi Ray khoury,

    First of all add ti.map module in your tiapp.xml and use Emulator Target as Google APIs(Google Inc.) when you create a new AVD from AVD Manager or Edit it.

    Now in app.js try this,

    var homeWin = Titanium.UI.createWindow({  
    backgroundColor:'#fff',
    navBarHidden:true,
    height:'100%',
    width:'100%',
});

var mapBtn = Titanium.UI.createButton(btn());
mapBtn.title='Map';
mapBtn.top='100dp';
homeWin.add(mapBtn);

// Listen for click events.
mapBtn.addEventListener('click', function() {
    var mapPage = Ti.UI.createWindow({
            url: '/map.js',
            backgroundColor:'#DDDEDE',
            navBarHidden:true,
            modal:true,
        });
        mapPage.open();
});

homeWin.open();

function btn()
{
    var proper = {
            width:'60dp',
            height:'40dp',
            borderRadius: 4,
            borderWidth: 2,
            borderColor: '#fff',
            backgroundGradient: {
                type: 'linear',
                colors: [
                    '#b5babd',
                    '#4a0000'
                ]   
            }
          };

    return proper;
}

    And in map.js,

    var mapPage = Titanium.UI.currentWindow;

//var map=require('ti.map');
var annotations = Ti.Map.createAnnotation({
        latitude: 23.7272512,
        longitude: 90.39666650000004,
        animate: true,
        pincolor: Ti.Map.ANNOTATION_RED,
    });

var mapView = Ti.Map.createView({
    mapType: Ti.Map.HYBRID_TYPE,
    region: {
        latitude:23.7272512,
        longitude:90.39666650000004,
        latitudeDelta:.01,
        longitudeDelta:.01
    },
    animate:true,
    regionFit:true,
    userLocation:true,
    annotations: [annotations]
});
mapPage.add(mapView);

    Hope it'll solve your problem. I've tested it and works fine.

    Thanks
