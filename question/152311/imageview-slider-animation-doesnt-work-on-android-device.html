ImageView Slider Animation doesnt work on Android Device

Hi,
I tried to animate 2 imageviews. I have an array of images which shall slide in to create some kind of stacking effect.
On the Emulator the animation works fine but on the device (S2) the images are just poping up and flickering.
Here is my code:

var images = ["/images/1.png", "/images/2.png"];
var counter = 1;
var actView = 0;
var anim = Ti.UI.createAnimation();
anim.left = 0;
anim.duration = 800;

$.imageSlogan0.image = images[0];

var slider = setInterval(function () {
    if(actView == 0){
        $.imageSlogan0.zIndex = 1;
        $.imageSlogan1.zIndex = 2;
        $.imageSlogan1.left = Ti.Platform.displayCaps.platformWidth;
        $.imageSlogan1.image = images[counter];
        $.imageSlogan1.animate(anim);
        actView = 1;
    }else{
        $.imageSlogan1.zIndex = 1;
        $.imageSlogan0.zIndex = 2;
        $.imageSlogan0.left = Ti.Platform.displayCaps.platformWidth;
        $.imageSlogan0.image = images[counter];
        $.imageSlogan0.animate(anim);
        actView = 0;        
    };
    counter++;
    if(counter == images.length){
        counter = 0;
    };
}, 10000);

I hope that anyone knows a solution.
Thanks
