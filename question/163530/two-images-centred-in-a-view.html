  • Hi,

    please try the below code.

    var win = Ti.UI.createWindow({
    backgroundColor: '#000', 
    fullscreen: true,
});
win.open();

var view=Ti.UI.createView({
    width:"100%",
    height:Ti.UI.SIZE
});
win.add(view);

var subView1=Ti.UI.createView({
    width:"50%",
    backgroundColor:'red',
    left:0,
    height:Ti.UI.SIZE
});
view.add(subView1);


var subView2=Ti.UI.createView({
    width:"50%",
    backgroundColor:'green',
    right:0,
    height:Ti.UI.SIZE
});
view.add(subView2);

var image1=Ti.UI.createImageView({
    image:'/KS_nav_views.png',
    right:0
});
subView1.add(image1);
var image2=Ti.UI.createImageView({
    image:'/KS_nav_views.png',
    left:0
});
subView2.add(image2);
