Titanium ignores zIndex = 0 on Android

This is a heads up for anyone relying on zIndex. Unlike CSS, zIndex in Titanium is not permissive with its values. I've had it ignore my values or sometimes crash. Here's an example of zIndex being ignored:

var win = Ti.UI.createWindow();

var smallBox = Ti.UI.createView(
{
    height          : 100,
    width           : 100,
    backgroundColor : 'red'
});

var bigBox = Ti.UI.createView(
{
    height          : 300,
    width           : 300,
    backgroundColor : 'blue'
});

win.add(smallBox);
win.add(bigBox);

win.open({fullscreen : true});

// The small box will be behind the big one
bigBox.zIndex   = 0;
smallBox.zIndex = 1;

// Uncomment to put the small box in front
//bigBox.zIndex   = 1;
//smallBox.zIndex = 2;
