      function getXmlData(url){

    xhr = Titanium.Network.createHTTPClient();
    xhr.open('GET',url);
    xhr.onload = function()
    {
        // Now parse the feed XML 
        var xml = this.responseXML;

        // Find the song title element 
        var songTitle = xhr.documentElement.getElementsByTagName("SONGTITLE");
    };
    xhr.send();
