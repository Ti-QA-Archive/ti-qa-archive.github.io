    The following has been tested and works on Titanium 1.7.X and Android 2.2. Note that this now properly uses the className property.

    var win = Titanium.UI.createWindow({
  title:'My App',
  backgroundColor: 'green'
});

var table_data = [];

for (var i = 0; i < 10; i++){
  var row = Titanium.UI.createTableViewRow({
    className: 'campaign',
    height: 150,
    width: 'auto'
  });

  var verticalView = Titanium.UI.createView({
    layout: 'vertical',
    right: 100,
    left: 0,
    height: 150,
    width: 200
  });

  var thisLabel1 = Ti.UI.createLabel({
    color:'black',
    text: 'CampaignName' + ' '+ i,
    textAlign: 'right',
    right: 0,
    font:{fontWeight:'bold',fontSize:16},
    left: 0,
    height: 35,
    width: 200,
    campaignLink:'http://www.google.com'
  });

  verticalView.add(thisLabel1);

  var thisLabel2 = Ti.UI.createLabel({
    color:'black',
    text: 'ProductName' + ' '+'CategoryName' ,
    font:{fontWeight:'bold'},
    textAlign: 'right',
    right: 0,
    left: 5,
    height: 35,
    width: 200
  });

  verticalView.add(thisLabel2);

  row.add(verticalView);

  table_data.push(row);

}

var table = Ti.UI.createTableView({
  data:table_data
});
win.add(table);
win.open();

win.addEventListener('click', function(e){
  if(e.source.campaignLink !== undefined){
    Ti.Platform.openURL(e.source.campaignLink);
  }
});

    Cheers
