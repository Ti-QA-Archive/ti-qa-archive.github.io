Views with buttons on Android become inactive after 1st instance

I have a win1 and it has a button to close it. Also I have a button to create a view1. view1 has a button to hide it.

When I click the hide button on the view1, the button on the win1 which supposed to create the view1 when clicked, becomes disabled. Why is it? It is only a problem in Android. Please help. Thank you.

var win1 = Titanium.UI.currentWindow;

var closeButton = Titanium.UI.createButton({
image:'images/icontest.png',
backgroundImage: 'none',
top:0,
right:0
    });

closeButton.addEventListener('click',function()
{
    win1.close({transition:Ti.UI.iPhone.AnimationStyle.CURL_DOWN});
});

 win1.add(closeButton);


 //Main view & button

    hi Jude,

    Replace this code and try it. I have changed some code instead of add just show that view. You are adding new view every time so just show don't add it.

    var win1 = Titanium.UI.currentWindow;

var closeButton = Titanium.UI.createButton({
image:'images/icontest.png',
backgroundImage: 'none',
top:0,
right:0
    });

closeButton.addEventListener('click',function()
{
    win1.close({transition:Ti.UI.iPhone.AnimationStyle.CURL_DOWN});
});

 win1.add(closeButton);

var view1=Ti.UI.createView({  
  backgroundColor: '#fff',
      borderColor: '#888',
      borderWidth: 4,
      height: 172,
      width: 275,
      top:50,
      opacity: 0.75,
      borderRadius: 8,
    visible : false
      });

  var closeButton2 = Titanium.UI.createButton({
  image:'images/icontest.png',
  backgroundImage: 'none',
  top:0,
  right:0
      });

  closeButton2.addEventListener('click',function()
{
    view1.hide();
});

  view1.add(closeButton2);

  var OpenButton = Titanium.UI.createButton({
  image:'images/icontest.png',
  backgroundImage: 'none',
  top:100,
  right:50
      });

win1.add(view1);

OpenButton.addEventListener('click',function()
  {
    view1.show();
  });

   win1.add(OpenButton);
