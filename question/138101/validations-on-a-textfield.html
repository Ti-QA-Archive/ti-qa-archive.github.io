  • Hello Lokesh,

    Try this….

    
var win3 = Ti.UI.currentWindow;
win3.backgroundColor = 'white';

var txt = Ti.UI.createTextField({
    top : 40,
    height : 30,
    width : 130,
    borderWidth : 3,
    borderColor : 'black'
});
win3.add(txt);

var btn = Ti.UI.createButton({
    title : 'click'
});
win3.add(btn);

function removeSpaces(string) {
 return string.split(' ').join('');
}

btn.addEventListener('click',function(e){
    removeSpaces(txt.value);
    if (removeSpaces(txt.value)) {
        alert(txt.value);
    }
    else{
        // alert(txt.value);
    }
});
