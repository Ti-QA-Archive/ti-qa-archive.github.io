    This is main structure

    <Window id="index">
    <NavigationWindow id="navWin" platform="ios">
        <Window class="container"></Window>
    </NavigationWindow>
</Window>

    This is page which open modal

    <Window class="container">
        <TitleControl>
            <View class="subtitle_header">
                <Label id="title_address" class="title"></Label>
                <Label id="sub_operation_name" class="subtitle"></Label>
            </View>
        </TitleControl>

        <ProgressBar id="pb" width="250" height="auto" min="0" max="1" value="0" color="#999" message="Loading ..." style="Titanium.UI.iPhone.ProgressBarStyle.PLAIN" zIndex="5" top="50%" />
        <Label id="lblNoResult" visible="false">No result found</Label>


        <Toolbar id="footer" bottom="0" borderTop="false" borderBottom="false" height="44">
            <!-- The Items tag sets the Toolbar.items property. -->
            <Items>
                <Button id="btnSort" height="28" width="28" backgroundImage="/images/icons/icon_sort.png" />
                <FlexSpace/>
                <Button id="btnFilter" height="28" width="27" backgroundImage="/images/icons/icon_filter.png" onClick="openFilter" />
                <FlexSpace/>
                <Button id="btnLocation" height="28" width="19" backgroundImage="/images/icons/icon_marker_slim_white.png" />
            </Items>
            <!-- Place additional views for the Toolbar here. -->
        </Toolbar>

        <TableView visible="false" bottom="45"></TableView>
    </Window>

    This is controller page which open modal

    var filter = Alloy.createController("filter");

function openFilter(e) {
    filter.openFilter();
}

    This is modal window

    var filter = Ti.UI.createWindow({
    title: 'Filter',
    backgroundColor: "#fff",
    barColor: "#2fb3f1",
    navTintColor: "#fff"
});

var btnResetFilter = Titanium.UI.createButton({
    title: "Reset"
});

var btnApplyFilter = Titanium.UI.createButton({
    title: "Apply"
});

var btnCloseFilter = Titanium.UI.createButton({
    title: "Cancel"
});
btnCloseFilter.addEventListener("click", function (e) {
    wrapperNavWin.close();
});

flexSpace = Titanium.UI.createButton({
    systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
});

var toolbar = Titanium.UI.iOS.createToolbar({
    items: [btnResetFilter, flexSpace, btnApplyFilter, flexSpace, btnCloseFilter],
    bottom: 0,
    borderTop: true,
    borderBottom: false,
    height: 44,
    zIndex: 3
});

filter.add(toolbar);

var wrapperNavWin = Ti.UI.iOS.createNavigationWindow({
    modal: true,
    window: filter
});

exports.openFilter = function () {
    wrapperNavWin.open();
};
