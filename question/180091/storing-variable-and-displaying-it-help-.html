  • Use the following code

    Titanium.UI.setBackgroundColor('#eeeeee');
    var win = Ti.UI.createWindow({});

    var signal=1;
    var popCount1 = 0, popCount2 =0;
    var imageView = Titanium.UI.createView({ height:'40%', width:'40%',top: 0, left: 0, backgroundImage: 'balloon.png' });
    imageView.addEventListener('click', function (e){
    if (signal) {

    popCount1 = popCount1+1;


imageView.backgroundImage = 'pop.png';
signal=0;

    } else {
    imageView.backgroundImage = 'balloon.png';
    signal=1;
    }
    Ti.API.info('imageView.image = ' + imageView.backgroundImage);

    Ti.API.info("popped balloons for imageView1 = "+popCount1);

    });

    var imageView2 = Titanium.UI.createView({ height:'40%', width:'40%',top: 200, left: 0, backgroundImage: 'balloon.png' });
    imageView2.addEventListener('click', function (e){
    if (signal) {

    popCount2 = popCount2+1;


imageView2.backgroundImage = 'pop.png';
signal=0;

    } else {
    imageView2.backgroundImage = 'balloon.png';
    signal=1;
    count++;
    }
    Ti.API.info('imageView2.image = ' + imageView2.backgroundImage);
    Ti.API.info("popped balloons for imageVie2 = "+popCount2);
    });

    win.add(imageView);

    it will give you popped ballons for both view when you will click on imageView. You can manage Ti.API.info("popped balloons for imageView1 = "+popCount1); as per your need.
