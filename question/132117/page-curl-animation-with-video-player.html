Page curl animation with Video Player

I am experiencing some major issues when it comes to page curl on a page that has a video player.

I basically what the page to curl up and display the next page when the user clicks on 'next'. The problem I am getting is when the user clicks on the 'next' button the page curl has problems animating the curl effect however eventually the next page is displayed.

I noticed that if you dont set the 'url' property from the video player or just set it a 'null' value then the page curl animation works perfectly!

Has anyone else experienced this problem? if so, could someone please provide me with a solution!

Here is my code:

Titanium.UI.setBackgroundColor('#000');

var tabGroup = Titanium.UI.createTabGroup();

var win1 = Titanium.UI.createWindow({  
    title:'Tab 1',
    backgroundColor:'#fff'
});

var view = Ti.UI.createView({ 
        backgroundColor: 'white',
        layout: 'vertical',
        top: 10,
        bottom: 10,
        left: 10,
        right: 10, 
        width: 340, 
        height: 'auto'
});

var activeMovie = Titanium.Media.createVideoPlayer({
    url: 'http://video-js.zencoder.com/oceans-clip.mp4',
    backgroundColor:'#111',
    mediaControlStyle:Titanium.Media.VIDEO_CONTROL_EMBEDDED,
    autoplay: true,
    width: 520,
    height: 360
});

var nextPage = Ti.UI.createButton({
    title:'NEXT',
    backgroundImage: 'none',
    width: 80,
    height: 30,
    borderColor: 'darkgray',
    color: 'black',
    borderRadius: 6,
});

view.add(activeMovie);
view.add(nextPage);

win1.add(view);

var tab1 = Titanium.UI.createTab({  
    icon:'KS_nav_views.png',
    title:'Tab 1',
    window:win1
});

tabGroup.addTab(tab1);  

tabGroup.open();

nextPage.addEventListener('click', function(e) {

    var tabGroup2 = Titanium.UI.createTabGroup();

    var backButton = Ti.UI.createButton({
        width: 100,
        height: 100,
        title: 'BACK',
        tabBarHidden: true,
    });    

    var articleDetails = Ti.UI.createWindow({
        backgroundColor:'black',
        barColor:'#000',
    });

    articleDetails.add(backButton);

    var tab1 = Titanium.UI.createTab({
        window:articleDetails
    });

    tabGroup2.addTab(tab1);  

    tabGroup2.open({ transition:Titanium.UI.iPhone.AnimationStyle.CURL_UP });

    backButton.addEventListener('click', function(e) {
        tabGroup2.close({ transition:Titanium.UI.iPhone.AnimationStyle.CURL_DOWN });
    });
});

To reconstruct the problem I am getting, copy and paste the code into the 'app.js' file and run the app. Click on the next button below the video and you will see the page curl animation problem I am experiencing. If you comment out the 'url' property in the video player object then run the app, the page curl animation works perfectly fine!

Many Thanks in advance!
