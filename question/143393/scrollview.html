    • Simple example:

      var win = Ti.UI.createWindow({
    backgroundColor:'#fff',
    layout:'vertical'
});

var scroll = Ti.UI.createScrollView({
    width:Ti.UI.FILL,
    height:Ti.UI.FILL,
    layout:'vertical'
});

var red = Ti.UI.createView({
    width:Ti.UI.FILL,
    height:300,
    backgroundColor:'red'
});
scroll.add(red);

var blue = Ti.UI.createView({
    width:Ti.UI.FILL,
    height:300,
    backgroundColor:'blue'
});
scroll.add(blue);

var green = Ti.UI.createView({
    width:Ti.UI.FILL,
    height:300,
    backgroundColor:'green'
});
scroll.add(green);


var wrapper = Ti.UI.createView({
    width:Ti.UI.FILL,
    height:Ti.UI.SIZE,
    layout:'horizontal'
});

var btng = Ti.UI.createButton({
    title:'Green'
});
btng.addEventListener('click', function() {
    scroll.contentOffset = {x:0,y:600};
});
wrapper.add(btng);

var btnb = Ti.UI.createButton({
    title:'Blue'
});
btnb.addEventListener('click', function() {
    scroll.contentOffset = {x:0,y:300};
});
wrapper.add(btnb);

var btnr = Ti.UI.createButton({
    title:'Red'
});
btnr.addEventListener('click', function() {
    scroll.contentOffset = {x:0,y:0};
});
wrapper.add(btnr);

win.add(wrapper);
win.add(scroll);
win.open();
