    you can create a navigation group that will handle your back button. here is a sample code for navigation group:

    var win= Titanium.UI.createWindow();
var win1 = Titanium.UI.createWindow({  
    title:'Enter Details',

});

var win2 = Titanium.UI.createWindow({
 url:'genel.js' //your path to genel.js

});

//create a navigation group
var nav = Titanium.UI.iPhone.createNavigationGroup({
   window: win1

});

win.add(nav);

// your button

var myButton = Titanium.UI.createButton({
     left: 196,
    top: 159,
    width: 104,
    height: 37,
    font: {fontFamily: 'Helvetica-Bold', fontSize: 15},
    title: 'Submit',
    color: '#324f85'
});

win1.add(myButton);

//on myButton event Handler just do this

myButton.addEventListener('click', function()
{
nav.open(win2,{animated:true});

}

win.open();
