No title or header bar with kitchen sink

Hi how come when i copy and paste the code in from the kitchen sink i gert no title/header bar/

copied from here
https://github.com/appcelerator/KitchenSink/blob/master/Resources/examples/table_view_edit_and_move.js

here is what i have in my app.js.



var tablesWin = Ti.UI.createWindow();


// create table view data object
var data = [
    {title:'Row 1', hasChild:true},
    {title:'Row 2', hasDetail:true},
    {title:'Delete only', moveable:false},
    {title:'Move only', editable:false},
    {title:'Neither', moveable:false, editable:false},
    {title:'Row 6'},
    {title:'Row 7'}
];

// create table view
var tableview = Titanium.UI.createTableView({
    data:data, editable:true, moveable:true
});

// add move event listener
tableview.addEventListener('move',function(e)
{
    Titanium.API.info("move - row="+e.row+", index="+e.index+", section="+e.section+", from = "+e.fromIndex);
});

// add table view to the window
tablesWin.add(tableview);

//
//  create edit/cancel buttons for nav bar
//
var edit = Titanium.UI.createButton({
    title:'Edit'
});

edit.addEventListener('click', function()
{
    tablesWin.setRightNavButton(cancel);
    tableview.editing = true;
});

var cancel = Titanium.UI.createButton({
    title:'Cancel',
    style:Titanium.UI.iPhone.SystemButtonStyle.DONE
});
cancel.addEventListener('click', function()
{
    tablesWin.setRightNavButton(edit);
    tableview.editing = false;
});

tablesWin.setRightNavButton(edit);

tablesWin.open();

    if your app consist of just a single win it will not have a title/header bar. To have a title bar your window must be inside a tabGroup, navGroup, open the window as a modal window, or create your own using Titanium.UI.iOS.Toolbar.
