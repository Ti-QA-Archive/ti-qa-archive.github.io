  • my final solution is -

    tableView.addEventListener('click', function(e) {

        if (e.row.hasChild && e.source.id !== "CHECKBOX") {
            e.row.applyProperties({ //can be used e.row.hasChild=false/true instead of this;
                hasChild : false
            });
            e.row.children[0].image = 'Tick.png';
        }

        Ti.API.info(e.row.hasChild);
        if (e.source.id == "CHECKBOX") {
            e.row.children[0].image = '';
            e.row.applyProperties({
                hasChild : true
            });
        }
    });
