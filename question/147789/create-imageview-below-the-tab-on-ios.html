    hi,

    just see this example

    var navWin1 = Titanium.UI.createWindow({
    backgroundColor : 'white'
});

var win = Ti.UI.createWindow({
    backgroundColor : 'red',
    navBarHidden : true
});

var imgView = Ti.UI.createImageView({
    backgroundColor : 'green',
    bottom : 0,
    height : 60,
    width : 320
});
win.add(imgView);

var navGroup1 = Ti.UI.iPhone.createNavigationGroup({
    window : win
});

navWin1.add(navGroup1);

navWin1.open();

// // Your Window
//
var tabGroup = Titanium.UI.createTabGroup({
    height : 400,
    top : 0
});

var win1 = Titanium.UI.createWindow({
    title : 'Tab 1',
    backgroundColor : '#fff',
    exitOnClose : true,
    backgroundColor : 'black'
});
var tab1 = Titanium.UI.createTab({
    icon : 'KS_nav_views.png',
    title : 'Tab 1',
    window : win1
});

tabGroup.addTab(tab1);

// open tab group
tabGroup.open();
