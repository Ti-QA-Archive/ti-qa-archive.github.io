  • Perfect work:

    var navi = Ti.UI.createWindow();
var tableView = Ti.UI.createTableView({});
navi.add(tableView);
navi.open();

var content = Ti.UI.createWindow({
    backgroundColor : 'green',
    left : 280
});
var b = Ti.UI.createLabel({
    left : 10,
    text:'?',
    top : 5,
    width : 30,
    height : 20
});
b.addEventListener('swipe', function(e) {
    if(e.direction == 'left') {
        content.animate(Ti.UI.createAnimation({
            duration : 200,
            left : 0
        }));
        b.text = '?';
    } else {
        content.animate(Ti.UI.createAnimation({
            duration : 200,
            left : 280
        }));
        b.text = '?';
    }
});
content.add(b);
content.open();

  • This code works partial: 

    var navi = Ti.UI.createWindow();
var tableView = Ti.UI.createTableView({});
navi.add(tableView);
navi.open();

var content = Ti.UI.createWindow({
    backgroundColor : 'red',
    left : 260
});
var b = Ti.UI.createLabel({
    font : {
        fontSize : 30
    },
    left : 10,
    borderColor:'black',
    borderRadius:15,
    borderWidth:2,
    top : 5,
    width : 30,
    height : 30
});
b.addEventListener('swipe',function(e){
    var offset = (e.direction=='left') ? -e.x/10 : e.x/10;
    content.left = content.left + offset;

});
content.add(b);
content.open();
