    tabGroup.addEventListener('focus', function(e)
{
    alert('tab changed to ' + e.index + ' old index ' + e.previousIndex);

    setTimeout(function()
    {
        Ti.API.info('tab ' + e.tab.title  + ' prevTab = ' + (e.previousTab ? e.previousTab.title : null));
Ti.API.info('active title ' + e.tab.title + ' old title ' + (e.previousTab ? e.previousTab.title : null);
    },1000);

});
