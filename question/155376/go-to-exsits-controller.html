    index.js

    function doClick(e) {  
    var c = Alloy.createController('home');
    var w = c.getView().open();
}

$.index.open();

    index.xml

    <Alloy>
    <Window class="container">
        <Label id="label" onClick="doClick">Click Me</Label>
    </Window>
</Alloy>

    home.xml

    <Alloy>
    <Window class="container">
        <Label id="label">This is home.xml</Label>
    </Window>
</Alloy>
