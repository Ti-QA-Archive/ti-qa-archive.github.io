    Hello Josselin S,

    Please check my following code. i hope it will help to you. 

    function winStart() {
    var view = Ti.UI.createView({
        width : '100%',
        height : 'auto'
    });
    var subTable1 = Ti.UI.createTableViewSection({
        headerTitle : 'Header1'
    });
    var subTable2 = Ti.UI.createTableViewSection({
        headerTitle : 'Header2'
    });

    var desc = [{
        title : 'title1',
        subtitle : "subtitle1",
        url : 'http://url1.com'
    }, {
        title : 'title2',
        subtitle : "subtitle2",
        url : 'http://url2.com'
    }];
    var table = Titanium.UI.createTableView({});
    var data = [];
    var customData = [];

    for (var i = desc.length - 1; i >= 0; i--) {
        var row = Ti.UI.createTableViewRow({
            height : 40,
            url : desc[i].url
        });
        var title = Titanium.UI.createLabel({
            text : desc[i].title,
            font : {
                fontSize : 16
            },
            width : 'auto',
            textAlign : 'left',
            top : 5,
            left : 10,
            height : 16
        });
        var subtitle = Titanium.UI.createLabel({
            text : desc[i].subtitle,
            font : {
                fontSize : 12
            },
            width : 'auto',
            textAlign : 'left',
            bottom : 5,
            left : 10,
            height : 12
        });
        row.add(title);
        row.add(subtitle);
        subTable1.add(row);
    }



    table.setData([subTable1, subTable2]);

    table.addEventListener('click', function(e) {
        var link = e.rowData.url;
        Ti.Platform.openURL(link);
    });

    view.add(table);

    return view;
}
