function addCover(opt) {
    // create a cover and add it to the list

    var cover = Alloy.createWidget('cover', 'widget', {
        url : opt.url, id : opt.id, size : coverSize, title : opt.title, image : opt.image, isnew : opt.isnew, date : opt.date
    });
    var c = cover.getView();

    $.coverlist.add(c);

    c.addEventListener("click", clickCover);
    c.addEventListener("longpress", longCover);
    c = null;
    cover = null;
}

download:

function doUpdate(id) {

    if (id < $.coverlist.children.length) {
        var cover = $.coverlist.children[id];

        // download rss file

        var file = downloader.createDownload({
            url : cover.url, id : cover.id, success : updateDownload, position : id, error : updateDownloadError
        });
        file.download();
        file = null;
        cover = null;
    } else {
        // all update done
    }
}

update:


function updateDownload(e) {
    // download done - parse file and update data
    var parser = require('parseRSS');
    var xml = "";
    var id = e.data.id;
    var date = "";
    var name = "";
    var image = "";
    var position = e.data.position;
    var cover = $.coverlist.children[position];

    xml = parser.createParser({
        file : e.file
    });
    if (xml != null && xml != undefined) {
        date = new Date(xml.getLatest()).getTime();
        name = xml.getName();
        image = xml.getCoverName();

        // get cover
        xml.downloadCover();

        // update name
        var pod = podcasts.where({alloy_id:id})[0];
        pod.set({
            "title" : name, "image" : image
        }).save();

        pod = null;

        var dateA = Number(cover.date);
        var dateB = Number(date);
        if (cover.date == undefined || cover.date == "" || cover.date == "undefined")
            dateA = 0;

        if (dateB > dateA) {
            // new
            var ep = xml.getEpisodes();

            for (var i = 0; i < ep.length; ++i) {
                var episode = Alloy.createModel("episode", {
                    title : ep[i].title, description : ep[i].description, timetotal : ep[i].timetotal, podcastID : id, url : ep[i].url
                });
                if (isUnique(episode, episodes, "url")) {
                    episodes.add(episode);
                    episode.save();
                } else {
                    Ti.API.info("already in");
                }
                episode = null;
            }
            ep = null;

        }
        dateA = null;
        dateB = null;
        xml.deleteFile({
            file : e.file
        });

    }

    parser = null;
    xml = null;
    id = null;
    date = null;
    name = null;
    image = null;
    position = null;
    cover = null;

    doUpdate(++currentUpdate);
}

  • Solution:
