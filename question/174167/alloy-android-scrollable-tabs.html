    Not a very specific question - an image with pointing would be helpful.

    Assuming that you are talking about the Android SDK 4.x version of ActionBar plus Tabs - then, that is the answer - ActionBar plus Tabs.

    This is a Titanium without Alloy example to give you an idea, create a classic test app and replace the contents of app.js with the code below;

    var win1 = Titanium.UI.createWindow({ title: 'Window 1' });
var tab1 = Ti.UI.createTab({ title: 'Tab 1', window: win1 });
var win2 = Titanium.UI.createWindow({ title: 'Window 2' });
var tab2 = Ti.UI.createTab({ title: 'Tab 2', window: win2 });
var lbl1 = Ti.UI.createLabel({ color: '#fff', text: 'Label 1' });
var lbl2 = Ti.UI.createLabel({ color: '#fff', text: 'Label 2' });
win1.add(lbl1);
win2.add(lbl2);
var tabGroup = Ti.UI.createTabGroup();
tabGroup.addEventListener('open', function(e) {
    var activity = tabGroup.getActivity();
    if (activity.actionBar) {
        activity.actionBar.title = "Test App"
    }
    activity.onCreateOptionsMenu = function(e) {
        var menu1 = e.menu.add({
            title: 'One',
            showAsAction: Ti.Android.SHOW_AS_ACTION_NEVER,
            icon: (Ti.Android.R.drawable.ic_menu_share ? Ti.Android.R.drawable.ic_menu_share : "my_search.png")
        });
        var menu2 = e.menu.add({
            title: 'Two',
            showAsAction: Ti.Android.SHOW_AS_ACTION_IF_ROOM,
            icon: (Ti.Android.R.drawable.ic_menu_share ? Ti.Android.R.drawable.ic_menu_share : "my_search.png")
        });
        var menu3 = e.menu.add({
            title: 'Three',
            showAsAction: Ti.Android.SHOW_AS_ACTION_IF_ROOM,
            icon: (Ti.Android.R.drawable.ic_menu_share ? Ti.Android.R.drawable.ic_menu_share : "my_search.png")
        });
        var menu4 = e.menu.add({
            title: 'Four',
            showAsAction: Ti.Android.SHOW_AS_ACTION_IF_ROOM,
            icon: (Ti.Android.R.drawable.ic_menu_share ? Ti.Android.R.drawable.ic_menu_share : "my_search.png")
        });
        var menu5 = e.menu.add({
            title: 'Five',
            showAsAction: Ti.Android.SHOW_AS_ACTION_IF_ROOM,
            icon: (Ti.Android.R.drawable.ic_menu_share ? Ti.Android.R.drawable.ic_menu_share : "my_search.png")
        });
        menu1.addEventListener('click', function (e) { alert('one'); });
        menu2.addEventListener('click', function (e) { alert('two'); });
        menu3.addEventListener('click', function (e) { alert('three'); });
        menu4.addEventListener('click', function (e) { alert('four'); });
        menu5.addEventListener('click', function (e) { alert('five'); });
    };
});
tabGroup.addTab(tab1);
tabGroup.addTab(tab2);
tabGroup.open();

    Also make sure your tiapp.xml does NOT include the following element;

    <supports-screens android:anyDensity="false"/>

    If it does - the overflow menu will appear cut off the screen.
