    var imageView = Ti.UI.createImageView({    
    left : 10,

tvRow.add(imageView);

Ti.Facebook.requestWithGraphPath(row['id'] + '?fields=picture', {}, "GET", function(ev) {
    if(ev.success) {
        x1 = JSON.parse(ev.result);
        x4 = x1['picture'];
        imageView.image = x4;        
    }
});
