  • A little modification in minh's reply as you need json which is inside GetJsonCatalogListResult .

    xhr1.onload = function() {
    if(this.status == 200) {
        var xml = Ti.XML.parseString(this.getResponseText());
        var catalogList = xml.documentElement.getElementsByTagName('GetJsonCatalogListResult');
    vat catalogListText = catalogList.item(0).text; // This will give you text as string which you can parse using json
    var jsonObj = JSON.parse(catalogListText ); // I think the test you get in catalogListText variable may contain </GetJsonCatalogListResult> tag. You can remove it than parse.

    }
};
