  • you must use property 'image' or 'backgroundImage' instead of 'url'
    'url' is deprecated .
    -example:-
    var Image = Titanium.UI.createImageView({
    borderRadius:10,
    backgroundImage:'images/image.jpg',
    bottom:'20%',
    right:'10%',
    top:'10%',
    left:'10%'
    });
    view.add(Image);
