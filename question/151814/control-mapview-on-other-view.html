    • Try this sample code : 

      var win = Ti.UI.createWindow();

var view = Ti.UI.createView({
    backgroundColor : "blue",
    height : "100dp",
    width : "100dp",
    top : "10dp",
    zIndex:"10"
});
win.add(view);

var map = Ti.Map.createView({
    animate : true,
    bottom : 0,
    height : Ti.UI.FILL,
    mapType : Ti.Map.STANDARD_TYPE,
    regionFit : true,
    top : 0,
    userLocation : true,
    width : Ti.UI.FILL
});
win.add(map);

win.open()
