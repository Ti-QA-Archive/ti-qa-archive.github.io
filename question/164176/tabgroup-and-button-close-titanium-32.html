    • Starting this late in the titanium game, you need to just start with Alloy. It is so much cleaner. You can look at your markup in the xml file and see just what the layout looks like, the relationship each objrct has to the others.

      index.xml

      <Alloy>
    <TabGroup>
        <Tab id="tab1">
            <Window id="win1">
            </Window>
        </Tab>
        <Tab id="tab2">
            <Window id="win2" onClick="openSecondWindow">
                <Button id="b1"/>
            </Window>
        </Tab>
        <Tab id="tab3">
            <Window id="win3">
                <Label id="label3"/>
            </Window>
        </Tab>
    </TabGroup>
</Alloy>

      index.tss

      "#tab1": {
    icon: '/images/Navigation_user.png'
}
"#tab2": {
    icon: '/images/Navigation_home.png',
}
"#tab3": {
    icon: '/images/Navigation_maps.png',
}
"#win1": {
    backgroundImage: '/images/Background_mycompte.png',
    fullscreen: false,
    navBarHidden: true,
}

"#win2": {
    backgroundColor: 'red',
    fullscreen: false,
    navBarHidden: true,
}

"#win3": {
    navBarHidden: true,
    backgroundColor: '#fff',
    fullscreen: false
}

"#b1": {
    title: 'PUSH'
}

"#label3": {
    color: '#999',
    font: {
        fontSize: 20,
        fontFamily: 'Helvetica Neue'
    },
    textAlign: 'center',
    width: 'auto',
    text: 'Label 3'
}

      index.js

      function openSecondWindow(){
    var secondWindow = Alloy.createController('secondWindow').getView();
    $.tab2.open(secondWindow);
}

$.index.open();

      secondWindow.xml

      <Alloy>
    <Window id="secondWin">
        <Button id="b2" onClick="closeSecondWindow"/>""
    </Window>
</Alloy>

      secondWindow.tss

      "#secondWin":{
    backgroundColor: 'blue'
}

"#b2":{
    title:"POP"
}

      secondWindow.js

      var args = arguments[0] || {};

function closeSecondWindow(){
    $.secondWin.close();
}
