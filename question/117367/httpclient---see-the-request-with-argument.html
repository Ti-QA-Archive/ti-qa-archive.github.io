    Example:

    var xhr = Ti.Network.createHTTPClient();
xhr.onload = function(){ alert(xhr.responseText); }
xhr.open('GET', 'http://your_suit.com');
var data_send = {
  data1:'data1',
  data2:'data2'
}
xhr.send(data);

    And code server (PHP)

    $data1 = $_GET['data1'];
$data2 = $_GET['data2'];
echo "Data1 -- {$data1}\nData2 -- {$data2}";
