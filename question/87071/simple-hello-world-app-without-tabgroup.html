  • Just answered myself :)
    win.open() was missing.
    Also, needed to call another .js
    Seems to be there's no currentWindow from app.js

    var win = Ti.UI.createWindow({
backgroundColor:'red',
title:'Simple Hellow World',
url:'main_windows/cleanwindow.js'
});

var l = Ti.UI.createLabel({
    text:'Hello World', 
    color:'#888',
    font:{fontSize:20},
    height:'auto',
    width:'auto',
});
win.open();
win.add(l);
