  • Try this code…

    var fb = require('facebook');

    fb.appid = 'YOUR APP ID';

    fb.permissions = "PERMISSIONS as required";

    var win = Ti.UI.createWindow({

    backgroundColor : '#44851354'

    });

    var btnGet = Ti.UI.createButton({

    title : 'Get Data',

height : 40,

width : 100,

top : 5,

borderColor : 'gray',

borderRadius : 10,

borderWidth : 3

    });

    btnGet.addEventListener('click', function(e) {

    var xhr = Ti.Network.createHTTPClient({

    onload : function(e) {

        alert(this.responseText);

        var data = JSON.parse(this.responseText);

    },

    onerror : function(e) {

        Ti.API.info("ERROR " + e.error);

        alert('ERROR :-->> ' + e.error);

    },

});

xhr.open('GET', 'https://graph.facebook.com/YOUR FRIENDS ID/photos?access_token=' +     fb.accessToken);

xhr.send();

    });

    fb.addEventListener('login', function(e) {

    if (e.success) {

    win.add(btnGet);

    win.remove(fbbtn);

} else if (e.error || e.cancelled) {

return;

}

    });

    var fbbtn = fb.createLoginButton();

    if (!fb.loggedIn) {

    win.add(fbbtn);

    } else {

    win.add(btnGet);

    }

    win.open();
