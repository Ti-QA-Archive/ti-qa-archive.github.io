iphone background showing when opening a view

i get a brief flash of my iphone desktop (without icons) when i open a couple of simple views from the tab bar ?

(titanium 1.2.2 - sdk 4.3 - iphone is ios 3.1.3)

app.js:

// home page
var win1 = Titanium.UI.createWindow({
    url:'about.js',
    titleid:'Home',
    barImage: 'nav-blue.png'
});
var tab1 = Titanium.UI.createTab({
    icon:'KS_nav_ui.png',
    titleid:'Home',
    window:win1
});
//home page

about.js :

var win = Titanium.UI.currentWindow;
win.backgroundColor = '#ffffff';

var currentSize = Ti.App.Properties.getInt("fontSize");



var label = Ti.UI.createLabel({
    text:'some blah blah',
    top:10,
    width:'auto',
    height:'auto',
    textAlign:'center',
    color:'#0000cc',
    font:{fontWeight:'bold',fontSize:18},
    shadowColor:'#aaa',
    shadowOffset:{x:1,y:1}    
});
win.add(label);

    text:'some blah blah',
