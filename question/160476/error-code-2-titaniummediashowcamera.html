Error Code 2 Titanium.Media.showCamera

Hi you guys,

Im using the function Titanium.Media.showCamera.
It does work on the android, but It doesn't work in the IOS.
Everytime the function to open the camera is triggered I received the error code 2.

Here is my code: 

Titanium.Media.showCamera({

        success : function(event) {
            alert("ok");
        },
        cancel : function() {

        },
        error : function(error) {
            alert("Error Code: " + error.code);
        },
        allowEditing : true,
        mediaTypes : [Ti.Media.MEDIA_TYPE_VIDEO, Ti.Media.MEDIA_TYPE_PHOTO],
    });

Does anyone knows what the code error 2 means?

  • Hi Gustavo,

    iPhone simulator can't emulate the camera, so you have to test on device.

    var win = Ti.UI.createWindow({
    backgroundColor : 'white',
    layout : 'vertical'
});

Titanium.Media.showCamera({

    success : function(event) {
        alert("ok");
    },
    cancel : function() {

    },
    error : function(error) {
        // called when there's an error
        var a = Titanium.UI.createAlertDialog({
            title : 'Camera'
        });
        if (error.code == Titanium.Media.NO_CAMERA) {
            a.setMessage('Please run this test on device');
        } else {
            a.setMessage('Unexpected error: ' + error.code);
        }
        a.show();
    },
    allowEditing : true,
    mediaTypes : [Ti.Media.MEDIA_TYPE_VIDEO, Ti.Media.MEDIA_TYPE_PHOTO],
});

win.open();
