  • Same code, beautified:

    function shuffle(o) { 
    for (var j, x, i = o.length; i; j = parseInt(Math.random() * i), x = o[--i], o[i] = o[j], o[j] = x); return o;
}; 
var level=3;
var sublevel=3; 
function discoverProblem(level, sublevel) {
    var obj = new Object();
    var prob = new Array();
    var grid = new Array();
    var min = 0;
    if (sublevel % 3 == 0) {
        min = 10;
    } else if (sublevel % 3 == 1) {
        min = 30;
    } else {
        min = -20;
    }
    var r = parseInt(Math.random() * min);
    for (var i = 0, j = r; i < (level*2); i++, j++){
        grid[i] = j;
        if (i == level) {j++;}
    }
    grid = shuffle(grid);
    // create JSON object for the solution
    var sol = '{ "results" : [';
    for (var i = 0; i < level; i++){
        prob[i] = grid[i]+grid[i+level];
        sol = sol + '{ "pair":"' + grid[i].toString() + ',' + grid[i+level].toString() + '" }';
        if ((i+1) < level) {
            sol = sol + ',';
        }
    }
    sol = sol + ']}';
    obj.prob = prob;
    obj.grid = grid;
    obj.sol = sol;
    //self.add(obj.prob); return obj;
}
