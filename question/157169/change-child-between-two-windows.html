    • You can use application level fire events and listeners. Look at this code how I used application level fire events to achieve your goal. 

      var win = Ti.UI.createWindow({
    backgroundColor:'white'
});

var btn = Ti.UI.createButton({
    title:'click me'
});

btn.addEventListener('click', function(){
    win.addEventListener('win2closed', function(e){
        Ti.API.info('in side loop');
        Ti.API.info(e);
    });
    var Win2Controller = require('win2');
    var win2Controller = new Win2Controller();
    var window2 = win2Controller.getView();
    // the window object
    window2.open();
});
win.add(btn);
Ti.App.addEventListener('win2closed', function(e){
    Ti.API.info('out side loop');
        Ti.API.info(e);
        btn.title = 'change';
    });
win.open();

      win2.js

      module.exports = function() {
    var self = {};
    // create the return value
    var win = Ti.UI.createWindow({
        backgroundColor : 'white'
    });
    var label = Ti.UI.createLabel({
        top : 20,
        width : Ti.UI.FILL,
        height : Ti.UI.SIZE,
        text : 'This is a text'
    });
    win.add(label);
    var closebutton = Ti.UI.createButton({
        top : 60,
        title : 'close'
    });
    win.add(closebutton);
    closebutton.addEventListener('click', function() {
        win.close();
        Ti.App.fireEvent('win2closed',{from:'window2'});

    });
    self.getView = function() {
        return win;
    };
    return self;
};
