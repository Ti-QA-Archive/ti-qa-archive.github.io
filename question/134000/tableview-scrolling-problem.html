Tableview scrolling problem

Hi,

Does anyone can tell me why I get this behavior with the tableview. I made this little video to show the problem:

See problem in action

Here's the code for my table view:

The declaration:


var questionTableView = Titanium.UI.createTableView({

    top: "70dp",
    left: "0dp",
    right: "0dp",
    minRowHeight: 47,
    height: "1000",
    backgroundColor: "#7f674f"
});

win.add(questionTableView);

The binding:


function bindTableView(_questions) {

    var data = [];

    //Loop on each questions    
    for(var i = 0; i < _questions.length; i++) {

        var row = Ti.UI.createTableViewRow();

        var question = Ti.UI.createLabel({

            text: _questions[i],
            height: "auto",
            left: "5dp",
            right: "5dp",
            top: "5dp",
            bottom: "5dp",
            textAlign: "left",
            font: { fontSize: 13, fontWeight: "bold" },
            color: "#fff"
        });

        //If we validate a selection    
        if(isValidation) {

            if(currentSelection == i) {

                if(i == questions[pp.app.currentQuestion].AnswerValue) {

                    row.backgroundColor = "#4f883c";

                } else {

                    row.backgroundColor = "#bf4334";
                }
            } else { //Bad selection

                //Good answer
                if(questions[pp.app.currentQuestion].AnswerValue == i) {

                    row.backgroundColor = "#4f883c";        
                } else {

                    //Wrong answer
                    row.backgroundColor = "#7f674f";        
                }
            }
        } else {
            //When we rewrite the table, we highlite the choice        
            if(currentSelection == i) {

                row.backgroundColor = "#f9a34d";
            } else {

                //Default color
                row.backgroundColor = "#7f674f";            
            }
        }

        row.add(question);
        row.hasChild = false;
        row.value = _questions[i].AnswerValue;
        row.height = "auto";

        data[i] = row;
    }

    questionTableView.data = data;
}

    Set the bottom: of your tableview to a point above that bottom ribbon.
    Remove your height. You have a height of 1000 on a device that in that orientation is what, 320?
