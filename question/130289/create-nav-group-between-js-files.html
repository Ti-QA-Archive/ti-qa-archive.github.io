index.js

var win = Ti.UI.createWindow({
    title:'Home',
    backgroundImage:'images/backgroundhome.png'
});
win.open();

var menu = Ti.UI.createView({
    top:250,
    left:10,
    width:305,
    height:198,
    backgroundImage:'images/menubox.png'
});
win.add(menu);

    var inv = Ti.UI.createButton({
        top:12,
        left:17,
        width:75,
        height:75,
        backgroundImage:'images/carIcon.png'
    });
    menu.add(inv);

        inv.addEventListener('click', function(){
            var invWin = Ti.UI.createWindow({url:'windows/inventory.js'});
            var nav = Ti.UI.iPhone.createNavigationGroup({window:win});        
            nav.open(invWin);
            win.add(nav);
        });

inventory.js

    • Ok, got it working using Ti 1.7.6 you can download the CI build if you want from here http://builds.appcelerator.com.s3.amazonaws.com/index.html

      Please note using the 1.7.x branch, there is a known issue with 1.8.x and 1.9.x that will cause you some issues here.

      I've also updated your code since it was easier for me to start with the blog sample.

      This was working for me using the iOS 5 Simulator. For some reason I had an issue in debug mode, but it was able to process without issue in Run mode.

      app.js

      Ti.include('test.js');
var first = Ti.UI.createWindow({
  backgroundColor:"#fff",
  title:"My App"
});
var label = Ti.UI.createLabel({ text: "poke me to open the next window" });
first.add(label);

//Here's the nav group that will hold them both...
var navGroup = Ti.UI.iPhone.createNavigationGroup({
    window:first
});
//Next, we’ll create a child window to push onto the stack:
//Here's a window we want to push onto the stack...
var second = fetchWindow();

second.add(Ti.UI.createLabel({text:"Here's the child"}));

//When the label on the first window receives a touch, open the second
label.addEventListener("click", function(e) {
  navGroup.open(second);
});

//This is the main window of the application
var main = Ti.UI.createWindow();
main.add(navGroup);
main.open();

      test.js

      function fetchWindow(){
var win = Ti.UI.createWindow({title: 'Inventory', backgroundColor:'purple'});
return win;
};
