    In an attempt to make a start troubleshooting this, I have developed the following usecase. I am discussing it with the core team, and will update you when I have some more info. Please also see TIMOB-3632, which is related.

    var win = Ti.UI.createWindow({
  backgroundColor : "white",
  layout:'vertical',
  exitOnClose:true
});

Ti.Media.audioSessionMode = Ti.Media.AUDIO_SESSION_MODE_AMBIENT;

var url = "http://st2.webradioworld.net:8036";  // direct stream - exception on 2.1, exception on 2.2

// Info page: http://www.radioparadise.com/content.php?name=Listen
// var url = "http://www.radioparadise.com/musiclinks/rp_64aac.m3u"; // exception on 2.1, exception on 2.2. Note that playlists are not expected to work
// var url = "http://173.244.214.174:8008";  // direct stream - exception on 2.1, exception on 2.2

// Info page: http://kzsulive.stanford.edu/
// var url = "http://kzsulive.stanford.edu/audio/kzsu-1-48.pls"; // exception on 2.1, exception on 2.2. Note that playlists are not expected to work
// var url = "http://171.66.118.51/kzsu-1-48.aac";  // direct stream - exception on 2.1, does not work on 2.2 but no exception

// var url = "http://radio.musicvictim.com:8012";  // direct stream - exception on 2.1, works on 2.2


var audio = Ti.Media.createAudioPlayer({
  url:url,
  preload:false,
  allowBackground:true
});

audio.format = Ti.Media.AUDIO_FORMAT_AAC;
audio.audioSessionMode = Ti.Media.AUDIO_SESSION_MODE_PLAYBACK;

var buttonControl = Ti.UI.createButton({
  title: 'Play',
  isSound:false,
  top:50
});

win.add(buttonControl);
win.open();

var nav_hidden = false;

buttonControl.addEventListener('click', function(){
  if(buttonControl.isSound){
    audio.stop();
    buttonControl.title = 'Play';
    buttonControl.isSound = !buttonControl.isSound;
    Ti.API.info("Stream stopped");
  } else {
    audio.start();
    buttonControl.title = 'Stop';
    buttonControl.isSound = !buttonControl.isSound;
    Ti.API.info("Stream started");
  }
});
