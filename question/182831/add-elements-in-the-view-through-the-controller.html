  • Try this Chen Hao,

    index.js

    
$.title.text="hello word";
var buttons=[];
for (var i=0; i < 4; i++){
    var btn = Titanium.UI.createButton({
        title: "item"+i,
        top: "6%",
        width:"100", 
        height:"50",
        font: {
            fontSize: 11,
            color:"#000",
            textAlign:"center"
        },
        backgroundColor: "#9DF"
    });
    $.view_body.add(btn);
    buttons.push(btn);
}

$.index.open();
