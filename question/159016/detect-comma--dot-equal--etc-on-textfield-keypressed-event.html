  • Hi Michael,

    Keypress event can only work with hardware key is pressed, you can see it in document as per your example you can see the log added to your consol when you press volume up and down, back etc. but It is not fired when you press keys in your keyboard displayed on the screen.

    Try this for demo.

    var win = Ti.UI.createWindow({
    backgroundColor : 'white'
});

win.open();

var txtAccountNo = Titanium.UI.createTextField({
    width:'200dp',
    height:'40dp',
    top:'160dp',
    enabled:true,
    _customProp:'',
    focusable:true,
    backgroundColor : "gray"
});

txtAccountNo.addEventListener('keypressed', function(e){
    Ti.API.info('::::: Keypressed Event');
});

win.add(txtAccountNo);
