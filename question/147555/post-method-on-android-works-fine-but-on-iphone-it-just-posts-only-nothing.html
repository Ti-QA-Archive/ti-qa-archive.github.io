    • Function to handle the request

      function getCurrectLocation(obj) {
    Ti.Geolocation.getCurrentPosition(function (e) {
        if (!e.success || e.error) {
            if (obj.error) {
                obj.error({
                    error: e.error,
                    code: e.code
                });
            }
        } else {
            if (obj.success) {
                obj.success({
                    coords: e.coords
                });
            }
        }
    });
}

      Usage;

      getCurrectLocation({
    success: function (e) {
        var lat1 = e.coords.latitude;
        var lon1 = e.coords.longitude;
        alert(JSON.stringify(e));
    },
    error: function (e) {
        alert(JSON.stringify(e));
    }
});
