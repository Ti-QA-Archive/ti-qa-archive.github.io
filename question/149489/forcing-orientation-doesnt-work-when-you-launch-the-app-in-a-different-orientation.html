  • When you create your application window, do you specify the orientation modes? I've noticed that adjusting the tiapp.xml doesn't always work. Have you tried…

    
function ApplicationWindow() {
  var self = Ti.UI.createWindow({
    fullscreen : true,
    orientationModes : [Ti.UI.PORTRAIT],
    layout : 'composite',
    height : aTi.Platform.displayCaps.platformHeight,
    width : Ti.Platform.displayCaps.platformWidth
  });
  return self;
}
module.exports = ApplicationWindow;

    This is a slimmed down snippet from one of my apps, but you can get the picture.
