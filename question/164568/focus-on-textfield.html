    • i did not get you exactly but i think you want to use a single function for more than one textField.

      try below..

      
function onFocus(e){
    e.source.borderColor = "red";
}

function onBlur(e){
    e.source.borderColor = "green";
}

for(var i=0; i < 10; i++){
    var txtFld = Ti.UI.createTextField({
        //propertis
    });
    txtFld.addEventListener('focus',onFocus);
    txtFld.addEventListener('blur',onBlur);
}
