Setting CustomTabBar image correctly

I've got a small issue using the customtabbar.js script from:

link to gitHub

Everything works great, apart from when the app first launches.

I select the 3rd tab as default (the middle one), which displays the Home screen in my app fine.

But the 'selected' graphic when the app launches selects the 1st tab, in this case 'Search' rather than the 'Home' one.

Can anyone suggest how to select the middle (Home) icon instead?

// Create the tab group
var tabGroup = Titanium.UI.createTabGroup();

// Assign windows & tabs
// IMPORTANT:
//          'tabBarHidden: true' should be set on all windows
//          height should be set to 480 - customTabBar's height (change 480 to app screen height)
var win1 = Titanium.UI.createWindow({ title:'Search', height: 368, tabBarHidden: true, barColor:'#005349', titleImage:'images/title_bounty.png', url:'search.js' });
var tab1 = Titanium.UI.createTab({ window:win1 });
var win2 = Titanium.UI.createWindow({ title:'Local', height: 368, tabBarHidden: true, barColor:'#005349', titleImage:'images/title_bounty.png', url:'local.js' });
var tab2 = Titanium.UI.createTab({ window:win2 });
var win3 = Titanium.UI.createWindow({ title:'Home', height: 368, tabBarHidden: true, barColor:'#005349', titleImage:'images/title_bounty.png', url:'home.js' });
var tab3 = Titanium.UI.createTab({ window:win3 });
var win4 = Titanium.UI.createWindow({ title:'Favourites', height: 368, tabBarHidden: true, barColor:'#005349', titleImage:'images/title_bounty.png', url:'favourites.js' });
var tab4 = Titanium.UI.createTab({ window:win4 });
var win5 = Titanium.UI.createWindow({ title:'News', height: 368, tabBarHidden: true, barColor:'#005349', titleImage:'images/title_bounty.png', url:'news.js' });
var tab5 = Titanium.UI.createTab({ window:win5 });
tabGroup.addTab(tab1);  
tabGroup.addTab(tab2);  
tabGroup.addTab(tab3);
tabGroup.addTab(tab4);
tabGroup.addTab(tab5);

// open tab group
tabGroup.open();

// Here is the magic
Ti.include("customTabBar.js");

var ctb = new CustomTabBar({
    tabBar: tabGroup,
    imagePath: 'images/tabs/',
    width: 64,
    height: 48,
    items: [
        { image: 'search.png', selected: 'search_over.png' },
        { image: 'local.png', selected: 'local_over.png' },
        { image: 'home.png', selected: 'home_over.png' },
        { image: 'favourites.png', selected: 'favourites_over.png' },
        { image: 'news.png', selected: 'news_over.png' }
    ]
});

// set middle tab to be open on app launch (0-4)
tabGroup.setActiveTab(tabGroup.tabs[2]);

Many thanks!
