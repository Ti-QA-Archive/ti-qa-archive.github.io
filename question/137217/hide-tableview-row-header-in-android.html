  • Hi,try this

    var table=Ti.UI.createTableView({
    //height:420,
    width:'100%',
    top:44,
    left:0,
    allowsSelection:true,
    separatorColor: '#999',
    search:searchBar,
    filterAttribute:'filter'
});
var row=Ti.UI.createTableViewRow({
            hasChild:true,
            filter: title
        });
