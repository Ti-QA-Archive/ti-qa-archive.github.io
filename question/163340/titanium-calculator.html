    Hi, Please try this code instead. Just copy paste this code

    var win = Ti.UI.createWindow({
    backgroundColor : 'white',
    height : Ti.UI.FILL,
    width : Ti.UI.FILL
});

var inputtxtField = Ti.UI.createTextField({
    height : 40,
    width : 120,
    backgroundColor : 'gray'
});
win.add(inputtxtField);

var lbl = Ti.UI.createLabel({
    top : 20,
    height : 40,
    width : 120,
    borderColor : 'gray',
    color:'black'
});

win.add(lbl);

inputtxtField.addEventListener('change', function(e) {
    try{
        var evaluatedValue = eval(this.value);
    }catch(e){
        Ti.API.info('error');
    }


    lbl.text = evaluatedValue;
});

win.open();
