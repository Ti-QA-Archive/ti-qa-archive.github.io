Hide BarTab on android

In the sdk 1.6 Use the following code to hide the tabbar in android:

Titanium.UI.setBackgroundColor('transparent');
var tabGroup = Titanium.UI.createTabGroup({
    barColor:'#3B7CE1',
    bottom:-50
});
var win1 = Titanium.UI.createWindow({  
    title:'Por Conocer',
    tabBarHidden: true,
    url:'appPorConocer/modMenus/menuPrincipal.js'
});
var tab = Titanium.UI.createTab({  
    window:win1
});
tabGroup.addTab(tab);  
tabGroup.open();
var f = function(){
    tabGroup.animate({top:-65,duration:20});
}
setTimeout(f,1000);

In the SDK 1.7 this code causes an error forcing me out of my program, how do I correct the error? or what is the code
to hide the tabbar is for sdk 1.7?
