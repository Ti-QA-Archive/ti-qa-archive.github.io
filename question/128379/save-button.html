Save button

How do I add a save button to a textfield? I want to save the text in the field evenif I leave the window, the text should be there when I get back.

  • Silvia

    Try the following:

    var win = Ti.UI.createWindow({
  backgroundColor:'#ddd',
  fullscreen:false,
  exitOnClose:true,
  layout:'vertical'
});

var textField = Ti.UI.createTextField({
  hintText: 'Enter some text',
  value:Ti.App.Properties.getString('textFieldContent', ''),
  height:60,
  top:60,
  left:60,
  width:350
});

win.add(textField);

var button = Ti.UI.createButton({
  title:'Save',
  height:60,
  top:5,
  left:60,
  width:350
});

win.add(button);

win.open();

textField.addEventListener('change', function(e){
  Ti.App.Properties.setString('textFieldContent', e.value);
});

button.addEventListener('click', function(e){
  Ti.App.Properties.setString('textFieldContent', textField.value);
});

    Hope this helps
