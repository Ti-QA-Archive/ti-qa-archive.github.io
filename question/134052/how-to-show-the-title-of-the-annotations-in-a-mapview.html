How to show the title of the annotations in a mapview?

I've managed to add a few annotations to a mapView but the user needs to click a annotation to show it's title. Is it possible to show the title of every annotations once they are on screen?

I tried adding this to the window with the mapView but it doesn't work as expected.

    mapView.addEventListener('complete', function() {
        for (var i in mapView.annotations) {
            mapView.annotations[i].fireEvent('click');
        }
    });

      And load them into a mapView here:

      function MapWindow(title) {
    var self = Ti.UI.createWindow({
        title:title,
        backgroundColor:'white'
    });

    var MapAnnotations = require('lib/MapAnnotations'),
        mapAnnotations = new MapAnnotations();

    var mapView = Ti.Map.createView({
        mapType: Titanium.Map.STANDARD_TYPE,
        annotations: mapAnnotations,
        region: {latitude: 59.859773, longitude:17.630181, 
            latitudeDelta:0.003, longitudeDelta:0.003},
        regionFit: true,
        userLocation: true,
        animate: true
    });

    //TODO: fix
    mapView.addEventListener('complete', function() {
        for (var i in mapView.annotations) {
            mapView.annotations[i].fireEvent('click', {
                source: mapView,
                type: 'click'
            });
        }
    })

    self.add(mapView);

    return self;
};

module.exports = MapWindow;
