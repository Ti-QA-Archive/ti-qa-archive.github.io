  • hello Allesandro.. i have this error too…
    what you need to do is declare the mapview outside and also ur annotation
    it goes like this..

    var myposition = Titanium.Map.createAnnotation({
        latitude:45.4155,
        longitude:10.9589,
        title:"Siete qui!",
        pincolor:Titanium.Map.ANNOTATION_RED,
        animate:true,
});
var mapview = Ti.Map.createView({
            mapType: Ti.Map.STANDARD_TYPE,
            region : { latitude: 45.4155, longitude: 10.9589, latitudeDelta:0.01, longitudeDelta:0.01},
            animate:true,
            regionFit:true,
            userLocation:true,
            annotations:[myposition]
});

Titanium.Geolocation.getCurrentPosition(function(e) { 
    if (!e.success || e.error) {
        Ti.UI.createAlertDialog({
            title: 'Attenzione',
            message: 'Problem',
            buttonNames:['OK']
        }).show();  
    }
    else
    {
        mapview.longitude = e.coords.longitude;
        mapview.latitude = e.coords.latitude;

            alert('long:' + longitude + ' lat: ' + latitude);   
    }

});
win.add(mapview);
