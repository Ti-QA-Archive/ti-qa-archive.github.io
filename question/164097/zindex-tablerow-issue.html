  • function createPhoneRow() {

     //Creates a table row with dropdown, textfield and remove btn

     var row = Ti.UI.createTableViewRow({
        width: Ti.UI.FILL, 
        height: 40

    });

    var inactiveDropDownImage = Ti.UI.createImageView({
        left: 0, top: 24,
        image: "/images/phone-dd-inactive.png"
    });
    row.add(inactiveDropDownImage);

    var selectedItemDropDown = Ti.UI.createLabel({
        left: 1, top: 10,
        text: "Mobile",
        color: "#000",
        font: {
               fontFamily: "Margot Regular",
            fontSize: 15
        }
    });
    row.add(selectedItemDropDown);

    var dropDownMenu = Ti.UI.createView({
        left: 1, top: 20,
        width: 100, height: 100,
        layout: 'vertical',
        visible: false,
        borderWidth: 1,
        borderColor: "#000",
        zIndex: 100
    });

    inactiveDropDownImage.addEventListener('click',function(e){
        dropDownMenu.show();
    });

    var arr = ["Mobile","Home","Other"];

    for (i=0; i < arr.length; i++){
        var item = Ti.UI.createLabel({
            height: 20, width: 100,
            zIndex: 100,
            text: arr[i],
            color: "#000"
        });
        dropDownMenu.add(item);

        item.addEventListener('click',function(e){
            selectedItemDropDown.text = this.text;
            dropDownMenu.hide();
        });
    }
    row.add(dropDownMenu);

    //Text field for phone number
    var phoneTextField = Ti.UI.createTextField({
        left: 110, top: 5,
        backgroundColor: 'none',
        value: "07945 654 894",
        color: "#6F1FB7",
        font: {
               fontFamily: "Margot Regular",
            fontSize: 18
        } 
    });
    row.add(phoneTextField);

    //Remove phone row button
    var removeRowBtn = Ti.UI.createButton({
        right:0, top:6,
        width: 25, height: 25,
        id: "removeRowBtn",
        backgroundImage: "/images/remove-btn.png",
        backgroundSelectedImage: "/images/remove-selected-btn.png"
    });
    row.add(removeRowBtn);

    return row;
}

$.contactDetailsTable.addEventListener('click', function(e){

        if(e.source.id == "btnAddPhone") {

        var phoneRow = createPhoneRow();

        $.contactDetailsTable.insertRowBefore(e.index, phoneRow);

    }

});

  • The zIndex value only applies to siblings of the same parent view - in this case the TableViewRow so you need to move your custom dropdown view to the window for it to appear above all rows.
