picker setSelectedRow

Hi,
I have custom data and already setup it to picker.

var tmptime = [{
    title : "5 " + L('minutes_point'),
    value : 5
}, {
    title : "10 " + L('minutes_point'),
    value : 10
}, {
    title : "15 " + L('minutes_point'),
    value : 15
}, {
    title : "20 " + L('minutes_point'),
    value : 20
}, {
    title : "30 " + L('minutes_point'),
    value : 30
}];

var pktime = [];

for (var i = 0; i < tmptime.length; i++) {
    var row = Titanium.UI.createPickerRow({
        title : tmptime[i].title,
        value : tmptime[i].value
    });

    pktime.push(row);
}

$.pkTracktimes.add(pktime);

and now i want to set selected row from value, but i don't know how to set it, please help me.
Thank you very much.

    Hi Vatthanachai Wongprasert!!!

    Please check the below code and I hope, it will help to you

    var pktime = [];
var value=30;
var selectedRow=0;

for (var i = 0; i < tmptime.length; i++) {
    var row = Titanium.UI.createPickerRow({
        title : tmptime[i].title,
        value : tmptime[i].value
    });

    pktime.push(row);
    if(value==tmptime[i].value)
    {
        selectedRow=i;
    }

}

$.pkTracktimes.add(pktime);

$.pkTracktimes.setSelectedRow(0, selectedRow, false);
