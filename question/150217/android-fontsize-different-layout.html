    Yes you could do it here is the solution for you

    var myfontsize = (Titanium.Platform.displayCaps.platformHeight * 5) / 100;
var labDate = Ti.UI.createLabel({
        text : 'Customized Font',
        color : 'white',
        font : {
            fontSize : myfontsize,
            fontFamily : customFont
        },
        height : Ti.UI.SIZE,
        top : '72%',
        width : Ti.UI.FILL,
        textAlign : 'center'
    });
    self.add(labDate);

    This would make the font size according to the height platform and would support multiple screens.It would also be applicable to IPhone

    I think you want to set custom size for resoluction…

    Ti.App.Properties.setInt('MaxWidth', Ti.Platform.displayCaps.platformWidth);
Ti.App.Properties.setInt('MaxHeight', Ti.Platform.displayCaps.platformHeight);

var screenWidth = Ti.App.Properties.getInt('MaxWidth');
var screenHeight = Ti.App.Properties.getInt('MaxHeight');


var labDate = Ti.UI.createLabel({
        text : 'Customized Font',
        color : 'white',
        font : {
            fontSize : screenWidth >= 320 ? 20 : 12,
            fontFamily : customFont
        },
        height : Ti.UI.SIZE,
        top : '72%',
        width : Ti.UI.FILL,
        textAlign : 'center'
    });
    self.add(labDate);
