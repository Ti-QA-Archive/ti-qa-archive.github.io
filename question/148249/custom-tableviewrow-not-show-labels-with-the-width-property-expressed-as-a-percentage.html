Custom tableviewrow not show labels with the width property expressed as a percentage

I created custom tableviewrow with the lables, but I have a problem in displaying lables: when I try to express width property value as a percentage, none of them appear. This is a bug or something wrong?

var win = Ti.UI.createWindow({
    layout : "vertical",
    width : "100%"
});
var data = [];
for(var i=0; i<5; i++){
    var row = Ti.UI.createTableViewRow({
        layout : "horizontal",
        width : "100%"
    });
    var lbl1 = Ti.UI.createLabel({
        height : 50,
        width : "40%",
        backgroundColor : "#aabbcc",
        text : "Lbl1, Row number " + i
    });
    row.add(lbl1);
    var lbl2 = Ti.UI.createLabel({
        height : 50,
        width : "60%",
        backgroundColor : "orange",
        text : "Lbl2, Row number " + i
    });
    row.add(lbl2);    
    data.push(row);
};
var tv = Ti.UI.createTableView({
    width : "100%",
    height : "50%",
    data : data,
    layout : "vertical"
})
win.add(tv);
win.open();

    This will work for you

    var win = Ti.UI.createWindow({
    layout : "vertical",
    width : "100%"
});
var data = [];
for(var i=0; i<5; i++){
    var row = Ti.UI.createTableViewRow({
        width : "100%"
    });
    var lbl1 = Ti.UI.createLabel({
        left:0,
        height : 50,
        width : "40%",
        backgroundColor : "#aabbcc",
        text : "Lbl1, Row number " + i
    });
    row.add(lbl1);
    var lbl2 = Ti.UI.createLabel({
        right:0,
        height : 50,
        width : "60%",
        backgroundColor : "orange",
        text : "Lbl2, Row number " + i
    });
    row.add(lbl2);  
    data.push(row);
};
var tv = Ti.UI.createTableView({
    width : "100%",
    height : "50%",
    data : data,
    layout : "vertical"
})
win.add(tv);
win.open();
