  • try this

    var row;
    for (var i=0, ilen = 10; i < ilen; i++){
      row = Ti.UI.createTableViewRow({
        color:'black',
        height: 200
      });

      var l = Ti.UI.createLabel({
        text:'black',
        height: 200
      });

      row.add(l);

      rows.push(row);
    }
    var table = Ti.UI.createTableView({
      data: rows,
      backgroundColor: 'white'
    });
