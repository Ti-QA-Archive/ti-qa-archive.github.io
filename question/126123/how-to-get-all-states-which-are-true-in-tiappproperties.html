  • // finds all properties that have "CHECKBOX" in the name and are true
    // adds them to the array "true_properties"

    var true_properties = [];
var props = Titanium.App.Properties.listProperties();
for (var c=0;c<props.length;c++)
{
    if ( props[c].indexOf("CHECKBOX") !== -1) {
       var value = Titanium.App.Properties.getBool(props[c]);
       Titanium.API.info(props[c]+" = "+value);
       if ( value ) {
          true_properties.push(props[c]);
       }
    }
}
