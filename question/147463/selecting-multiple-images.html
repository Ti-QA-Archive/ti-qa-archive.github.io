    • Now you have the imageurl property:

      exports.create = function() {
    var window = Ti.UI.createWindow({
        backgroundColor : 'black',
    });

    scrollgrid = Ti.UI.createScrollView({
        top : '100dp',
        contentHeight : Ti.UI.SIZE,
        contentWidth : Ti.UI.SIZE,
        height : Ti.UI.FILL,
        width : Ti.UI.FILL,
        showHorizontalScrollIndicator : false,
        showVerticalScrollIndicator : false
    });

    var imagesview = Ti.UI.createView({
        bottom : 8,
        height : Ti.UI.SIZE,
        layout : 'horizontal',
        left : 0,
        right : 0,
        top : 0,
        width : Ti.UI.FILL
    });

    var image = Titanium.UI.createImageView({
        height : Ti.Platform.displayCaps.platformWidth / 4.15,
        image : photos[p].picture,
        item_type1 : "Photo",
        done : false,
        width : Ti.Platform.displayCaps.platformWidth / 4.15
    });
    var view = Titanium.UI.createView({
        width : Ti.Platform.displayCaps.platformWidth / 4.15,
        height : Ti.Platform.displayCaps.platformWidth / 4.15,
        left : 4,
        top : 4,
    });
    var checkBox = Ti.UI.createView({
        width : 35,
        height : 35,
        done : false,
        top : 5,
        imageurl : photos[p].picture,  //  <======
        item_type : "CHECKBOX", // us this to know we clicked a checkbox
        right : 5,
        backgroundImage : '/icons/unchecked.png'
    });

    view.add(image);
    view.add(checkBox);
    imagesview.add(view);

    imagesview.addEventListener('click', function(evt) {
        if (evt.source.item_type === "CHECKBOX") {
            if (evt.source.done === true) {
                evt.source.done = false;
                evt.source.backgroundImage = '/icons/unchecked.png';
            } else {
                evt.source.done = true;
                evt.source.backgroundImage = '/icons/checked.png';
            }
        } else {
            alert("Click box to select the Picture.");
        }
    });
    scrollgrid.add(imagesview);
    window.add(scrollgrid);
    return window;
}
