Autocomplete for my classes and functions

Hi guys.
I am new in JS/Titanium programming an i would like to use autocomplete for my own classes and functions.
For example, if i wrote my own class like this:

function Test(params){
    this.txt    = params.txt;
    this.color    = params.color;
};
/**
 * Bla bla bla bla bla bla
 * @param {Boolean} red
 */
Test.prototype.helloWorld = function(red){
    if(red)
        Ti.API.warn(this.txt);
    else
        Ti.API.info(this.txt);
};

module.exports = Test;

var TestClass = require("Test");

var test = new TestClass({
    // ?????????? I get no suggestion about members "txt" and "color"
    txt : 'Hello World',
    color : 'grren'
});

test. // ?????????? I get no suggestion about function "helloWorld"

I would like to get suggestion about all methods of my Class if i write dot behind "test".
Also i would like to get suggestion about all posible members of my param for class Test, like "txt" and "color".

Thank you in advance.
