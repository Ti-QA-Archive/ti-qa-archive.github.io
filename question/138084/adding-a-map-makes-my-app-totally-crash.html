Adding a map makes my app totally crash

Hello everyone !

I'm totally new to developping applications and I'm a little stuck with this problem.

I created an adroid mobile app (Titanium SDK 2.0.2), on my Win 7 laptop. My app contains 2 tabs :

Titanium.UI.setBackgroundColor('#000');

var tabGroup = Titanium.UI.createTabGroup();

// Fenêtre
var fenetreAccueil = Titanium.UI.createWindow({  
    title:'Accueil',
    url: 'contenus/index.js',
    backgroundColor:'#fff'
});

// Onglet
var tabAccueil = Titanium.UI.createTab({
    icon: 'KS_nav_ui.png',
    title: "Accueil",
    window: fenetreAccueil
})


var fenetreMap = Titanium.UI.createWindow({  
    title:'Plan',
    url: 'contenus/map.js',
    backgroundColor:'#fff'
});

var tabMap = Titanium.UI.createTab({
    icon: 'KS_nav_views.png',
    title: "Plan",
    window: fenetreMap
})

tabGroup.addTab(tabAccueil);  
tabGroup.addTab(tabMap);  

// open tab group
tabGroup.open();

The first tab show the "index.js" content and works fine.


var win = Ti.UI.currentWindow;

var webView = Titanium.UI.createWebView({
    html: '<h1>Mobile API Project</h1><h2>Ma page HTML INDEX.JS</h2><p>Texte HTML<br/>une autre ligne !!</p>',
})

win.add(webView);

The second one show the app.js content… But when I click on this tab, the application crashes. I have a lot of error messages but one of them appears more "java.lang.NullPointerException"


var win = Titanium.UI.currentWindow;

var mapView = Titanium.Map.createView({
    mapType: Titanium.Map.STANDARD_TYPE,
    animate: true,
    region: true, //permet de définir une partie de la carte à afficher
    userlocation: false
})

win.add(mapView);
win.open();

Could someone help me, please ? :)

    I think your problem may be with the region property. I think you meant "regionFit", not "region". Region should be an object like this:

            var region = {
                latitude: 35.7795,
                longitude: -78.6389, 
                latitudeDelta: 0.01, 
                longitudeDelta: 0.01
           };

    I also strongly recommend that you not use the url property of windows. Instead, build your map window as a CommonJS module, like this:

    function MapWindow ()
{
    var _self,
        _mapView;

    _self = Ti.UI.createWindow ({
        title:'Plan',
        backgroundColor:'#fff'
    });

    _mapView = Titanium.Map.createView({
        mapType: Titanium.Map.STANDARD_TYPE,
        animate: true,
        regionFit: true, //permet de définir une partie de la carte à afficher
        userLocation: false
    });

    _self.add (_mapView);

    return _self;
}

module.exports = MapWindow;

    Then in your app.js:

    var MapWindow = require ('MapWindow');
var fenetreMap = MapWindow ();
