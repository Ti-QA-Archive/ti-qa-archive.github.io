  • You could do something like this

    tabGroup.addEventListener('focus', function(e) {
    if (e.tab === tab2){
        win2.init();
    }
});

    In your /ui/map.js

    function windowMap_Creation(){
    var self;
    self.init = function(){
        if (!!self){
            // setup window
        }
    };
    return self;
}

exports.windowMap_Creation = windowMap_Creation;
