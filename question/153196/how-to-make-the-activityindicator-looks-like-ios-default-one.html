How to make the ActivityIndicator looks like iOS' default one

Hi,

I use the Activity Indicator like this:

var activityIndicator = Titanium.UI.createActivityIndicator({
    top: (SCREEN_HEIGHT/2)-96,
    left: (SCREEN_WIDTH/2)-32,
    height:50,
    style:Titanium.UI.iPhone.ActivityIndicatorStyle.PLAIN,
    font: { fontFamily:'Helvetica Neue', fontSize:15, fontWeight:'bold' },
    color: 'white',
    message: 'Loading...',
    width: 210
});

activityIndicator.show();

this.loadingWindow.add(activityIndicator);

    Some very fast written code:

    var v = Ti.UI.createView({
background:"#9000" // yes 4 digits ;)
borderRadius:5,
width:100,
height:100
})

var actInd = Ti.UI.createActivityIndicator({ 
style:big_style_see_docs,
width:60,
height:60,
top:10
});

actInd.show()

var lbl = Ti.UI.createLabel({
text:'loading',
color:'#fff',
bottom:10
});


v.add(actInd);
v.add(lbl);

your_window.add(v);

    You will have to tweak it to make it how you like it :)
