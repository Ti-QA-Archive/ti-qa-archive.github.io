  • Want to try this one?

    var win = Ti.UI.createWindow({ backgroundColor : '#fff' });

// Create a TextField. 

var myTextField = Ti.UI.createTextField({ 
    color: "#000",
    height : 35, 
    top : 10, 
    left : 40, 
    width : 240, 
    hintText : 'This is hint text', 
    keyboardType : Ti.UI.KEYBOARD_NUMBER_PAD, 
    returnKeyType : Ti.UI.RETURNKEY_DEFAULT, 
    borderStyle : Ti.UI.INPUT_BORDERSTYLE_ROUNDED 
});

var inputText = Titanium.UI.createLabel({ 
    top : 150, 
    left : 10, 
    width : 300, 
    height : 'auto', 
    color : '#777', 
    font : { 
            fontSize : 13 
            }, 
    text : 'write something in the textbox' }); 



myTextField.addEventListener("change", function(event, type) {
   var value = myTextField.value;
   inputText.text=value.replace(/(\d)(?=(\d\d\d)+(?!\d))/g, "$1,");
});

win.add(inputText);
win.add(myTextField);

win.open();
