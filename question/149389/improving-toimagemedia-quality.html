hi,

  • Hi Usman, if you are targeting iOS you can save the Retina Resoution of your ImageView. 

    exports.saveRetinaImage = function(imageView, filename) {

    var image = imageView.toImage(null, true);
    var f = Ti.Filesystem.getFile(Ti.Filesystem.tempDirectory, filename);
    f.write(image);
    return;
};
