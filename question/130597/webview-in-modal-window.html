webview in modal window

I have a modal window that contains a webview and a button to close it. When I first open the window the webview is visible. When I close the window then re-open it the webview vanishes, i see onli the button.
This is my code:

var winGuida = Ti.UI.createWindow({
    modal : true,
    width : 280,
    height : 300,
    title : "Guida",
});

var winGuidaWebView = Ti.UI.createWebView({
    url: "guida.html",
    width: 260,
    height: 250,
    borderWidth: 2
});
winGuida.add(winGuidaWebView);

var closeWinGuidaBtn = Ti.UI.createButton({
    backgroundColor: "#aabbcc",
    borderRadius: 10,
    title: "chiudi guida",
    width: 100,
    height: 40,
    bottom: 5
});
winGuida.add(closeWinGuidaBtn);

closeWinGuidaBtn.addEventListener("click", function(){
    winGuida.close();
});

Ti.App.addEventListener("app:openWinGuida", function(){
    winGuida.open();        
});
