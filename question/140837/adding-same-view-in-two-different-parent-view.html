    In the same time? No

    You should be able add it to another parent if you remove it from the first one.

    But if you need it in the same time, I'd create a "class" and instantiate a new object for each subview you need.

    Something like this:

    var subview = function(title_text) {
    return Ti.UI.createLabel({text:title_text});
};

var sub_one = new subview('one');
var sub_two = new subview('two');

  • You can do it since everything is done by references.

    You can have 

    var oTab1 = Ti.UI.createTab()
var oTab2 = Ti.UI.createTab()

function letsRock()
{
    var oWindow = Ti.UI.createWindow()
    var tabs = [oTab1, oTab2]
    for ( var iTab = 0, ITab = tabs.length;  iTab < ITab;  iTab++ )
    {
        var oTab = tabs[iTab]
              oTab.window = oWindow
    }
}

    You could even achieve "at the same time" by using Ti.App.addEventListener and Ti.App.fireEvent

    function updateTab( oTab, oWindow )
{
    oTab.window = oWindow
}

var eventName = 'New_window'

var oTab1 = Ti.UI.createTab()
var oTab2 = Ti.UI.createTab()
Ti.App.addEventListener( eventName, function(e){ updateTab( oTab1, e.data.oWindow ) } )
Ti.App.addEventListener( eventName, function(e){ updateTab( oTab2, e.data.oWindow ) } )

var oWindow = Ti.UI.createWindow()
Ti.App.fireEvent( eventName, {oWindow: oWindow} )

    • Yes, you simply add to both parents. That's why when oWindow.close() states

      > It will deallocate the window and all of its children if they are not referenced by another object

      So you can have

      var oLabel = Ti.UI.createLabel({text:'foo'})

var oView1 = Ti.UI.createView()
var oView2 = Ti.UI.createView()

oView1.add(oLabel)
oView2.add(oLabel)

oLabel.text = ' the hee hee'

      Bare in mind that since you've created a reference to the sub View object, if you modify it, it will be reflected in both parents. Maybe this is what you are striving for. if not you can check this article on how to Deep copy in JS
