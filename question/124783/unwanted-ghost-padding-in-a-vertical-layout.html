Unwanted Ghost-Padding in a Vertical layout!

I try to create an article with some paragraphs (in my example uiLabelRed, uiLabelBlue and uiLabelGreen). Each paragraph should have a bottom margin of 10dp. I dont know exactly why, but its not possible to add bottom: '10dp' on each paragraph, so i added top: '10pd' to the following paragraph.

On the screenshot you see the strange result.

My problem is, that each row i´ve added also add a Ghost-Bottom-Padding of 10dp to the "uiViewBackground" element. I´ve commented these 30pd (21 pixel) on the screenshot.

My questions are:

  • Did I made a mistake here in my source causing this problem with the Ghost-Padding?
  • Is there a trick, how to add a bottom-margin to a paragraph? (bottom: '10dp' dont work)

I hope somebody can test this.. at the moment i´ve no idea how to solve this.

var uiWindow = Ti.UI.createWindow({
    navBarHidden: true
});

var uiViewBackground = Ti.UI.createView({
    layout: 'vertical',
    backgroundColor: '#FFFFFF'
});

var uiViewArticle = Ti.UI.createView({
    layout: 'vertical',
    backgroundColor: '#aabbcc'
});

// add intro red
var uiLabelRed = Ti.UI.createLabel({
    text: 'I´m the RED bar (10dp top)!',
    backgroundColor: '#AA0000',
    width: '100%',
    height: 'auto',
    top: '10dp'
});

// add intro red
var uiLabelGreen = Ti.UI.createLabel({
    text: 'I´m the GREEN bar (10dp top)!',
    backgroundColor: '#00AA00',
    width: '100%',
    height: 'auto',
    top: '10dp'
});

// add intro red
var uiLabelBlue = Ti.UI.createLabel({
    text: 'I´m the BLUE bar (10dp top)!',
    backgroundColor: '#0000AA',
    width: '100%',
    top: '10dp'
});

// add all 3 labels with 10dp (7 pixel) top
uiViewArticle.add(uiLabelRed);
uiViewArticle.add(uiLabelGreen);
uiViewArticle.add(uiLabelBlue);

// after adding the tree labels, the background
// view got 30dp (21 pixel) extra at the bottom! 
uiViewBackground.add(uiViewArticle);

uiWindow.add(uiViewBackground);

uiWindow.open();

  • It looks like you are working with the Android Emulator. I don't have the Android environment set up, but I replicated your issue on the iPhone Simulator and was able to get it working as you described. Basically, you need the height:&#39;auto&#39; on the parent object, which is uiViewArticle. Then you can remove layout:&#39;vertical&#39; from uiViewBackground because you only have one child object in that view. I also set top:0 on uiViewArticle. I also set height:&#39;auto&#39; on uiLabelBlue. I think this will give you the results you desire. Let me know.

    var uiWindow = Ti.UI.createWindow({
    navBarHidden: true
});

var uiViewBackground = Ti.UI.createView({
    backgroundColor: '#FFFFFF'
});

var uiViewArticle = Ti.UI.createView({
    layout: 'vertical',
    backgroundColor: '#aabbcc',
    height:'auto',
    top:0
});

// add intro red
var uiLabelRed = Ti.UI.createLabel({
    text: 'I´m the RED bar (10dp top)!',
    backgroundColor: '#AA0000',
    width: '100%',
    height: 'auto',
    top: '10dp'
});

// add intro red
var uiLabelGreen = Ti.UI.createLabel({
    text: 'I´m the GREEN bar (10dp top)!',
    backgroundColor: '#00AA00',
    width: '100%',
    height: 'auto',
    top: '10dp'
});

// add intro red
var uiLabelBlue = Ti.UI.createLabel({
    text: 'I´m the BLUE bar (10dp top)!',
    backgroundColor: '#0000AA',
    width: '100%',
    top: '10dp',
    height:'auto'
});

// add all 3 labels with 10dp (7 pixel) top
uiViewArticle.add(uiLabelRed);
uiViewArticle.add(uiLabelGreen);
uiViewArticle.add(uiLabelBlue);

// after adding the tree labels, the background
// view got 30dp (21 pixel) extra at the bottom! 
uiViewBackground.add(uiViewArticle);

uiWindow.add(uiViewBackground);

uiWindow.open();

/**
 * encapsulate the given uiElement by an view with layout vertical
 * Because of this, the uiElement kan use top/bottom to set a margin!
 */
function uiElementWithMargin(uiElement) {

    var container = Ti.UI.createView({
        layout: 'vertical'
    });
    container.add(uiElement);

    return container;
}
