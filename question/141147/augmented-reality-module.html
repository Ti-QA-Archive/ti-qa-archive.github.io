    • Im working on that as we speak. This is my approach:

      Titanium.AR = Ti.AR = require('ti.ar');

/* Create the canvas */
var arView = Ti.AR.createView();

/* Custom view */
var customView = Ti.UI.createView({
    backgroundImage: "poi_background.png"
    width: "100dp",
    height: "50dp"
});

/* Labels */
var title = Ti.UI.createLabel({
    text: "Apple's nest",
    left: "10dp",
    top: "10dp",
    height: "20dp",
    type: "title" /* Must be defined */
    .....
});

var subtitle = Ti.UI.createLabel({
    text: "Where great things happen",
    left: "10dp",
    top: "30dp",
    height: "20dp"
    type: "subtitle" /* Must be defined */
    .....
});

customView.add(title, subtitle);

/* The POIView holding our custom view */
var poiView = Ti.AR.createPOIView({
    latitude: 59.263796,
    longitude: 11.143848,
    view: customView
});

/* Add POI to canvas */
arView.addPOIs([poiView]);

/* Open the camera and place our POI's */
arView.start();

/* Now you can add eventlisteners etc.. */
arView.addEventListener("location", function(e) {});
arView.addEventListener("heading", function(e) {});
arView.addEventListener("poi:clicked", function(e) {});

      Let me know if there are any other/better ways of doing it.
