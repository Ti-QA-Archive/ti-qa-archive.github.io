var date = Titanium.App.Properties.getString("looking_date");

    exports.call='titanium';

    var win1 = Titanium.UI.createWindow({  
        barColor:'orange',
           backgroundColor:'white',

        variableReceived: callparams
        });


        win1.open();    


    })

    In some.js

    do this

    var winTemp = Titanium.UI.currentWindow;

var callparamsTemp = winTemp.variableReceived;

    then all your passed vales will come into callparamsTemp.
