Zoom In & out as per scale property of pinch evenlistener

I want the view to zoom in & out on basis of my pinch gesture. I have written the following block of code. I need some suggestion as I have not yet succeeded.

var win = Titanium.UI.createWindow({  

    orientationModes:[Ti.UI.LANDSCAPE_LEFT, Ti.UI.LANDSCAPE_RIGHT],
    backgroundColor:'white'

});
win.open();

var currentScale = 1.0;
var currentTranslation = {x:0.0, y:0.0};
var lastTranslation = {x:0.0, y:0.0};

var view5 = Ti.UI.createImageView({
    url:'view5_background1.png',
    borderRadius : 10,
    width : 1030,
    height : 600,
    top : 0,

});

function updateTransform(image)
{
    var transform = Ti.UI.create2DMatrix()
        .scale(lastScale*currentScale);

    transform.tx = lastTranslation.x+currentTranslation.x;
    transform.ty = lastTranslation.y+currentTranslation.y;

    view5.transform = transform;
};



view5.addEventListener('pinch', function(e){
    alert('You pinched to ' + e.scale*100 + '%');
    currentScale = e.scale;
    updateTransform(view5);

});



win.add(view5);

  • view5.addEventListener('pinch',function(e){
    var t = Titanium.UI.create2DMatrix().scale(e.scale);
    e.source.transform = t;
});
