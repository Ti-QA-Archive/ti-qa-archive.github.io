    font:{fontSize:'14dp',fontWeight:'normal'},
    color:'#000'

    • here are full code:

      Titanium.UI.setBackgroundColor('#000');

var tabGroup = Ti.UI.createTabGroup({
    backgroundImage: '/images/tabbar.png',
});

var notNetworkAlert = Titanium.UI.createAlertDialog({
    message: 'Not connection with server',
    buttonNames: ['Close']
});


var settings = Ti.UI.createButton({
    title: 'Settings'
});

var news = Ti.UI.createWindow({  
    backgroundColor:'#fff',
    barImage: '/images/navbar.png'
});

var tab1 = Ti.UI.createTab({ 
    icon:'news.png',
    title:'News',
    window:news
});

var data = [];

var xhr = Titanium.Network.createHTTPClient();

xhr.onload = function(){
    var returnedJson = JSON.parse(this.responseText);
     var i = 0;
    for(var v in returnedJson.news){
        title = returnedJson.news[v].title;
        text = returnedJson.news[v].text1;
        picture = returnedJson.news[v].picture;
        comments = returnedJson.news[v].comments;
        if(comments > 0) {
            comments = ' <font color="#ff0000">(' + comments + ')</font>';
        }
        else {
            comments = '';
        }
        data[i] = Ti.UI.createTableViewRow({
            height: 'auto',
        });        
        var titleLabel = Ti.UI.createLabel({
            html: title,
            top: 10,
            left: 10,
            right: 140,
            height: 'auto',
            maxHeight: 50,
            font: {fontFamily: 'Arial', fontSize: 12, fontWeight: 'bold'}
        });
        data[i].add(titleLabel);    
        var textLabel = Ti.UI.createLabel({
            text: text,
            top: 60,
            left: 10,
            right: 140,
            bottom: 10,
            height: 50,
            font: {fontFamily: 'Arial', fontSize: 11}
        });
        data[i].add(textLabel);
        var image = Ti.UI.createImageView({
            top: 10,
            right: 10,
            width: 120,
            height: 80,
            url: picture
        });
        data[i].add(image);
        i++;
    }
    var newsTable = Ti.UI.createTableView({
        data:data
    });    
    news.add(newsTable);
};
xhr.onerror = function(e) {
    notNetworkAlert.show();
}
xhr.open('GET','http://domain.com/json.php');
xhr.send();

news.setRightNavButton(settings);
tabGroup.addTab(tab1);
tabGroup.open();
